41  class GarmanKlassAbstract :
42  public LocalVolatilityEstimator<IntervalPrice> {
43  protected:
44  Real yearFraction_;
45  virtual Real calculatePoint(const IntervalPrice &p) = 0;
46  public:
47  explicit GarmanKlassAbstract(Real y) :
48  yearFraction_(y) {}
49  TimeSeries<Volatility>
50  calculate(const TimeSeries<IntervalPrice> &quoteSeries) {
51  TimeSeries<Volatility> retval;
52  TimeSeries<IntervalPrice>::const_iterator prev, next, cur, start;
53  start = quoteSeries.begin();
54  for (cur = start; cur != quoteSeries.end(); ++cur) {
55  retval[cur->first] =
56  std::sqrt(std::fabs(calculatePoint(cur->second))/
57  yearFraction_);
58  }
59  return retval;
60  }

75  /* This template factors out common functionality found in
76  classes which rely on the difference between the previous day's
77  close price and today's open price. */
78  template <class T>
79  class GarmanKlassOpenClose : public T {
80  protected:
81  Real f_;
82  Real a_;
83  public:
84  GarmanKlassOpenClose(Real y, Real marketOpenFraction,
85  Real a) :
86  T(y), f_(marketOpenFraction), a_(a) {};
87  TimeSeries<Volatility>
88  calculate(const TimeSeries<IntervalPrice> &quoteSeries) {
89  TimeSeries<Volatility> retval;
90  TimeSeries<IntervalPrice>::const_iterator prev, next, cur, start;
91  start = quoteSeries.begin();
92  ++start;
93  for (cur = start; cur != quoteSeries.end(); ++cur) {
94  prev = cur; --prev;
95  Real c0 = std::log(prev->second.close());
96  Real o1 = std::log(cur->second.open());
97  Real sigma2 =
98  a_ * (o1 - c0) * (o1 - c0) / f_ +
99  (1-a_) * T::calculatePoint(cur->second) /
100  (1-f_);
101 
102  retval[cur->first] = std::sqrt(sigma2/T::yearFraction_);
103  }
104  return retval;
105  }
106  };

144  class GarmanKlassSigma4 :
145  public GarmanKlassAbstract {
146  public:

148  GarmanKlassAbstract(y) {};
149  protected:
150  Real calculatePoint(const IntervalPrice &p) {
151  Real u = std::log(p.high()/p.open());
152  Real d = std::log(p.low()/p.open());
153  Real c = std::log(p.close()/p.open());
154  return 0.511 * (u-d)*(u-d) -
155  0.019 * (c*(u+d)-2*u*d) -
156  0.383 * c * c;
