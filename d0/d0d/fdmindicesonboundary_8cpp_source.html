27  FdmIndicesOnBoundary::FdmIndicesOnBoundary(
28  const ext::shared_ptr<FdmLinearOpLayout>& layout,
29  Size direction, FdmDirichletBoundary::Side side) {
30 
31  std::vector<Size> newDim(layout->dim());
32  newDim[direction] = 1;
33  const Size hyperSize
34  = std::accumulate(newDim.begin(), newDim.end(), Size(1),
35  std::multiplies<Size>());
36  indices_.resize(hyperSize);
37 
38  Size i=0;
39  const FdmLinearOpIterator endIter = layout->end();
40  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
41  ++iter) {
42  if ( ( side == FdmDirichletBoundary::Lower
43  && iter.coordinates()[direction] == 0)
44  || ( side == FdmDirichletBoundary::Upper
45  && iter.coordinates()[direction]
46  == layout->dim()[direction]-1)) {
47  QL_REQUIRE(hyperSize > i, "index missmatch");
48  indices_[i++] = iter.index();
49  }
50  }

53  }
