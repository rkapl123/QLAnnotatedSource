261  inline void operator()(Array &newPoint, const Array &currentPoint, const Array &temp) const {
262  QL_REQUIRE(newPoint.size() == currentPoint.size(), "Incompatible input");
263  QL_REQUIRE(newPoint.size() == lower_.size(), "Incompatible input");
264  QL_REQUIRE(newPoint.size() == temp.size(), "Incompatible input");
265  for (Size i = 0; i < currentPoint.size(); i++) {
266  newPoint[i] = lower_[i] - 1.0;
267  while (newPoint[i] < lower_[i] || newPoint[i] > upper_[i]) {
268  Real draw = uniform_();
269  Real sign = static_cast<int>(0.5 < draw) - static_cast<int>(draw < 0.5);
270  Real y = sign*temp[i] * (std::pow(1.0 + 1.0 / temp[i],
271  std::abs(2 * draw - 1.0)) - 1.0);
272  newPoint[i] = currentPoint[i] + y*(upper_[i] - lower_[i]);
273  }
274  }
275  };
276  private:
277  Array lower_, upper_;
278  base_generator_type generator_;
279  uniform distribution_;
280  mutable uniform_variate uniform_;
281  };

443  class ReannealingFiniteDifferences {
444  public:
445  ReannealingFiniteDifferences(Real initialTemp,
446  Size dimension,
447  const Array & lower = Array(),
448  const Array & upper = Array(),
449  Real stepSize = 1e-7,
450  Real minSize = 1e-10,
451  Real functionTol = 1e-10)
452  : stepSize_(stepSize), minSize_(minSize),
453  functionTol_(functionTol), N_(dimension), bound_(false),
454  lower_(lower), upper_(upper), initialTemp_(dimension, initialTemp),
455  bounded_(dimension, 1.0) {
456  if (!lower.empty() && !upper.empty()) {
457  QL_REQUIRE(lower.size() == N_, "Incompatible input");
458  QL_REQUIRE(upper.size() == N_, "Incompatible input");
459  bound_ = true;
460  for (Size i = 0; i < N_; i++) {
461  bounded_[i] = upper[i] - lower[i];
462  }
463  }
464  }
465  inline void setProblem(Problem &P) { problem_ = &P; };
466  inline void operator()(Array & steps, const Array &currentPoint,
467  Real currentValue, const Array & currTemp) const {
468  QL_REQUIRE(currTemp.size() == N_, "Incompatible input");
469  QL_REQUIRE(steps.size() == N_, "Incompatible input");
470 
471  Array finiteDiffs(N_, 0.0);
472  double finiteDiffMax = 0.0;
473  Array ofssetPoint(currentPoint);
474  for (Size i = 0; i < N_; i++) {
475  ofssetPoint[i] += stepSize_;
476  finiteDiffs[i] = bounded_[i] * std::abs((problem_->value(ofssetPoint) - currentValue) / stepSize_);
477  ofssetPoint[i] -= stepSize_;
478  if (finiteDiffs[i] < minSize_)
479  finiteDiffs[i] = minSize_;
480  if (finiteDiffs[i] > finiteDiffMax)
481  finiteDiffMax = finiteDiffs[i];
482  }
483  for (Size i = 0; i < N_; i++) {
484  Real tRatio = initialTemp_[i] / currTemp[i];
485  Real sRatio = finiteDiffMax / finiteDiffs[i];
486  if (sRatio*tRatio < functionTol_)
487  steps[i] = std::pow(std::fabs(std::log(functionTol_)),
488  Integer(N_));
489  else
490  steps[i] = std::pow(std::fabs(std::log(sRatio*tRatio)),
491  Integer(N_));
492  }
493  }
494  private:
495  Problem *problem_;
496  Real stepSize_, minSize_, functionTol_;
497  Size N_;
498  bool bound_;
499  Array lower_, upper_, initialTemp_, bounded_;
500  };
