243  inline void operator()(Array &newPoint, const Array &currentPoint, const Array &temp) const {
244  QL_REQUIRE(newPoint.size() == currentPoint.size(), "Incompatible input");
245  QL_REQUIRE(newPoint.size() == lower_.size(), "Incompatible input");
246  QL_REQUIRE(newPoint.size() == temp.size(), "Incompatible input");
247  for (Size i = 0; i < currentPoint.size(); i++) {
248  newPoint[i] = lower_[i] - 1.0;
249  while (newPoint[i] < lower_[i] || newPoint[i] > upper_[i]) {
250  Real draw = uniform_();
251  Real sign = (0.5 < draw) - (draw < 0.5);
252  Real y = sign*temp[i] * (std::pow(1.0 + 1.0 / temp[i],
253  std::abs(2 * draw - 1.0)) - 1.0);
254  newPoint[i] = currentPoint[i] + y*(upper_[i] - lower_[i]);
255  }
256  }
257  };
258  private:
259  Array lower_, upper_;
260  base_generator_type generator_;
261  uniform distribution_;
262  mutable uniform_variate uniform_;
263  };

424  class ReannealingFiniteDifferences {
425  public:
426  ReannealingFiniteDifferences(Real initialTemp,
427  Size dimension,
428  const Array & lower = Array(),
429  const Array & upper = Array(),
430  Real stepSize = 1e-7,
431  Real minSize = 1e-10,
432  Real functionTol = 1e-10)
433  : stepSize_(stepSize), minSize_(minSize),
434  functionTol_(functionTol), N_(dimension), bound_(false),
435  lower_(lower), upper_(upper), initialTemp_(dimension, initialTemp),
436  bounded_(dimension, 1.0) {
437  if (lower.size() > 0 && upper.size() > 0) {
438  QL_REQUIRE(lower.size() == N_, "Incompatible input");
439  QL_REQUIRE(upper.size() == N_, "Incompatible input");
440  bound_ = true;
441  for (Size i = 0; i < N_; i++) {
442  bounded_[i] = upper[i] - lower[i];
443  }
444  }
445  }
446  inline void setProblem(Problem &P) { problem_ = &P; };
447  inline void operator()(Array & steps, const Array &currentPoint,
448  Real currentValue, const Array & currTemp) const {
449  QL_REQUIRE(currTemp.size() == N_, "Incompatible input");
450  QL_REQUIRE(steps.size() == N_, "Incompatible input");
451 
452  Array finiteDiffs(N_, 0.0);
453  double finiteDiffMax = 0.0;
454  Array ofssetPoint(currentPoint);
455  for (Size i = 0; i < N_; i++) {
456  ofssetPoint[i] += stepSize_;
457  finiteDiffs[i] = bounded_[i] * std::abs((problem_->value(ofssetPoint) - currentValue) / stepSize_);
458  ofssetPoint[i] -= stepSize_;
459  if (finiteDiffs[i] < minSize_)
460  finiteDiffs[i] = minSize_;
461  if (finiteDiffs[i] > finiteDiffMax)
462  finiteDiffMax = finiteDiffs[i];
463  }
464  for (Size i = 0; i < N_; i++) {
465  Real tRatio = initialTemp_[i] / currTemp[i];
466  Real sRatio = finiteDiffMax / finiteDiffs[i];
467  if (sRatio*tRatio < functionTol_)
468  steps[i] = std::pow(std::fabs(std::log(functionTol_)),
469  Integer(N_));
470  else
471  steps[i] = std::pow(std::fabs(std::log(sRatio*tRatio)),
472  Integer(N_));
473  }
474  }
475  private:
476  Problem *problem_;
477  Real stepSize_, minSize_, functionTol_;
478  Size N_;
479  bool bound_;
480  Array lower_, upper_, initialTemp_, bounded_;
481  };
