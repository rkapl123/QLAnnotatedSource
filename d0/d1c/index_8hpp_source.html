89  template <class DateIterator, class ValueIterator>
90  void addFixings(DateIterator dBegin, DateIterator dEnd,
91  ValueIterator vBegin,
92  bool forceOverwrite = false) {
93  checkNativeFixingsAllowed();
94  std::string tag = name();
95  TimeSeries<Real> h = IndexManager::instance().getHistory(tag);
96  bool noInvalidFixing = true, noDuplicatedFixing = true;
97  Date invalidDate, duplicatedDate;
98  Real nullValue = Null<Real>();
99  Real invalidValue = Null<Real>();
100  Real duplicatedValue = Null<Real>();
101  while (dBegin != dEnd) {
102  bool validFixing = isValidFixingDate(*dBegin);
103  Real currentValue = h[*dBegin];
104  bool missingFixing = forceOverwrite || currentValue == nullValue;
105  if (validFixing) {
106  if (missingFixing)
107  h[*(dBegin++)] = *(vBegin++);
108  else if (close(currentValue,*(vBegin))) {
109  ++dBegin;
110  ++vBegin;
111  } else {
112  noDuplicatedFixing = false;
113  duplicatedDate = *(dBegin++);
114  duplicatedValue = *(vBegin++);
115  }
116  } else {
117  noInvalidFixing = false;
118  invalidDate = *(dBegin++);
119  invalidValue = *(vBegin++);
120  }
121  }
122  IndexManager::instance().setHistory(tag, h);
123  QL_REQUIRE(noInvalidFixing,
124  "At least one invalid fixing provided: " <<
125  invalidDate.weekday() << " " << invalidDate <<
126  ", " << invalidValue);
127  QL_REQUIRE(noDuplicatedFixing,
128  "At least one duplicated fixing provided: " <<
129  duplicatedDate << ", " << duplicatedValue <<
130  " while " << h[duplicatedDate] <<
131  " value is already present");
132  }
134  void clearFixings();
135  private:
137  void checkNativeFixingsAllowed();
138  };
