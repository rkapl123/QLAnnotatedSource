42  void TreeVanillaSwapEngine::calculate() const {
43 
44  QL_REQUIRE(!model_.empty(), "no model specified");
45 
46  Date referenceDate;
47  DayCounter dayCounter;
48 
49  ext::shared_ptr<TermStructureConsistentModel> tsmodel =
50  ext::dynamic_pointer_cast<TermStructureConsistentModel>(*model_);
51  if (tsmodel != 0) {
52  referenceDate = tsmodel->termStructure()->referenceDate();
53  dayCounter = tsmodel->termStructure()->dayCounter();
54  } else {
55  referenceDate = termStructure_->referenceDate();
56  dayCounter = termStructure_->dayCounter();
57  }
58 
59  DiscretizedSwap swap(arguments_, referenceDate, dayCounter);
60  std::vector<Time> times = swap.mandatoryTimes();
61 
62  ext::shared_ptr<Lattice> lattice;
63  if (lattice_ != 0) {
64  lattice = lattice_;
65  } else {
66  TimeGrid timeGrid(times.begin(), times.end(), timeSteps_);
67  lattice = model_->tree(timeGrid);
68  }
69 
70  Time maxTime = *std::max_element(times.begin(), times.end());
71  swap.initialize(lattice, maxTime);
72  swap.rollback(0.0);
73 
74  results_.value = swap.presentValue();
75  }
