75  template <Size N> inline
76  FdmNdimSolver<N>::FdmNdimSolver(
77  const FdmSolverDesc& solverDesc,
78  const FdmSchemeDesc& schemeDesc,
79  const boost::shared_ptr<FdmLinearOpComposite>& op)
80  : solverDesc_(solverDesc),
81  schemeDesc_(schemeDesc),
82  op_(op),
83  thetaCondition_(new FdmSnapshotCondition(
84  0.99*std::min(1.0/365.0,
85  solverDesc.condition->stoppingTimes().empty()
86  ? solverDesc.maturity :
87  solverDesc.condition->stoppingTimes().front()))),
88  conditions_(FdmStepConditionComposite::joinConditions(thetaCondition_,
89  solverDesc.condition)),
90  x_ (solverDesc.mesher->layout()->dim().size()),
91  initialValues_(solverDesc.mesher->layout()->size()),
92  extrapolation_(std::vector<bool>(N, false)) {
93 
94  const boost::shared_ptr<FdmMesher> mesher = solverDesc.mesher;
95  const boost::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
96 
97  QL_REQUIRE(layout->dim().size() == N, "solver dim " << N
98  << "does not fit to layout dim " << layout->size());
99 
100  for (Size i=0; i < N; ++i) {
101  x_[i].reserve(layout->dim()[i]);
102  }
103 
104  const FdmLinearOpIterator endIter = layout->end();
105  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
106  ++iter) {
107 
108  initialValues_[iter.index()] = solverDesc_.calculator
109  ->avgInnerValue(iter, solverDesc.maturity);
110 
111  const std::vector<Size>& c = iter.coordinates();
112  for (Size i=0; i < N; ++i) {
113  if (!(std::accumulate(c.begin(), c.end(), 0)-c[i])) {
114  x_[i].push_back(mesher->location(iter, i));
115  }
116  }
117  }
118 
119  f_ = boost::shared_ptr<data_table>(new data_table(x_));
120  }

123  template <Size N> inline
124  void FdmNdimSolver<N>::performCalculations() const {
125  Array rhs(initialValues_.size());
126  std::copy(initialValues_.begin(), initialValues_.end(), rhs.begin());
127 
128  FdmBackwardSolver(op_, solverDesc_.bcSet, conditions_, schemeDesc_)
129  .rollback(rhs, solverDesc_.maturity, 0.0,
130  solverDesc_.timeSteps, solverDesc_.dampingSteps);
131 
132  const boost::shared_ptr<FdmLinearOpLayout> layout
133  = solverDesc_.mesher->layout();
134 
135  const FdmLinearOpIterator endIter = layout->end();
136  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
137  ++iter) {
138  setValue(*f_, iter.coordinates(), rhs[iter.index()]);
139  }
140 
141  interp_ = boost::shared_ptr<MultiCubicSpline<N> >(
142  new MultiCubicSpline<N>(x_, *f_, extrapolation_));
143  }

146  template <Size N> inline
147  Real FdmNdimSolver<N>::thetaAt(const std::vector<Real>& x) const {
148  QL_REQUIRE(conditions_->stoppingTimes().front() > 0.0,
149  "stopping time at zero-> can't calculate theta");
150  calculate();
151  const Array& rhs = thetaCondition_->getValues();
152  const boost::shared_ptr<FdmLinearOpLayout> layout
153  = solverDesc_.mesher->layout();
154 
155  data_table f(x_);
156 
157  const FdmLinearOpIterator endIter = layout->end();
158  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
159  ++iter) {
160  setValue(f, iter.coordinates(), rhs[iter.index()]);
161  }
162 
163  return (MultiCubicSpline<N>(x_, f)(x)
164  - interpolateAt(x)) / thetaCondition_->getTime();
165  }
