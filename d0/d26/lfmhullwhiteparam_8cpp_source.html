26  LfmHullWhiteParameterization::LfmHullWhiteParameterization(
27  const ext::shared_ptr<LiborForwardModelProcess> & process,
28  const ext::shared_ptr<OptionletVolatilityStructure> & capletVol,
29  const Matrix& correlation, Size factors)
30  : LfmCovarianceParameterization(process->size(), factors),
31  diffusion_ (size_-1, factors_),
32  fixingTimes_(process->fixingTimes()) {
33 
34  Matrix sqrtCorr(size_-1, factors_, 1.0);
35  if (correlation.empty()) {
36  QL_REQUIRE(factors_ == 1,
37  "correlation matrix must be given for "
38  "multi factor models");
39  } else {
40  QL_REQUIRE( correlation.rows() == size_-1
41  && correlation.rows() == correlation.columns(),
42  "wrong dimesion of the correlation matrix");
43 
44  QL_REQUIRE(factors_ <= size_-1,
45  "too many factors for given LFM process");
46 
47  Matrix tmpSqrtCorr = pseudoSqrt(correlation,
48  SalvagingAlgorithm::Spectral);
49 
50  // reduce to n factor model
51  // "Reconstructing a valid correlation matrix from invalid data"
52  // (<http://www.quarchome.org/correlationmatrix.pdf>)
53  for (Size i=0; i < size_-1; ++i) {
54  std::transform(
55  tmpSqrtCorr[i], tmpSqrtCorr[i]+factors_, sqrtCorr[i],
56  divide_by<Real>(std::sqrt(std::inner_product(
57  tmpSqrtCorr[i],tmpSqrtCorr[i]+factors_,
58  tmpSqrtCorr[i], 0.0))));
59  }
60  }
61 
62  std::vector<Volatility> lambda;
63  const std::vector<Time> fixingTimes = process->fixingTimes();
64  const std::vector<Date> fixingDates = process->fixingDates();
65 
66  for (Size i = 1; i < size_; ++i) {
67  Real cumVar = 0.0;
68  for (Size j = 1; j < i; ++j) {
69  cumVar += lambda[i-j-1] * lambda[i-j-1]
70  * (fixingTimes[j+1] - fixingTimes[j]);
71  }
72 
73  const Volatility vol = capletVol->volatility(fixingDates[i], 0.0);
74  const Volatility var = vol * vol
75  * capletVol->dayCounter().yearFraction(fixingDates[0],
76  fixingDates[i]);
77 
78  lambda.push_back(std::sqrt( (var - cumVar)
79  / (fixingTimes[1] - fixingTimes[0])) );
80 
81  for (Size q=0; q<factors_; ++q) {
82  diffusion_[i-1][q] = sqrtCorr[i-1][q] * lambda.back();
83  }
84  }
85 
86  covariance_ = diffusion_ * transpose(diffusion_);
87  }

96  Disposable<Matrix> LfmHullWhiteParameterization::diffusion(
97  Time t, const Array&) const {
98  Matrix tmp(size_, factors_, 0.0);
99  const Size m = nextIndexReset(t);
100 
101  for (Size k=m; k<size_; ++k) {
102  for (Size q=0; q<factors_; ++q) {
103  tmp[k][q] = diffusion_[k-m][q];
104  }
105  }

123  Disposable<Matrix> LfmHullWhiteParameterization::integratedCovariance(
124  Time t, const Array&) const {
125 
126  Matrix tmp(size_, size_, 0.0);
127 
128  Size last = std::lower_bound(fixingTimes_.begin(),
129  fixingTimes_.end(), t)
130  - fixingTimes_.begin();
131 
132  for (Size i=0; i<last; ++i) {
133  const Time dt = ((i+1<last)? fixingTimes_[i+1] : t )
134  - fixingTimes_[i];
135 
136  for (Size k=i; k<size_-1; ++k) {
137  for (Size l=i; l<size_-1; ++l) {
138  tmp[k+1][l+1]+= covariance_[k-i][l-i]*dt;
139  }
140  }
141  }
142 
143  return tmp;
144  }
