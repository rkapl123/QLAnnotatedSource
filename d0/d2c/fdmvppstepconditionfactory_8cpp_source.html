30 FdmVPPStepConditionFactory::FdmVPPStepConditionFactory(VanillaVPPOption::arguments args)
31 : args_(std::move(args)) {
32 QL_REQUIRE(!( args_.nStarts != Null<Size>()
33 && args_.nRunningHours != Null<Size>()),
34 "start and running hour limt together is not supported");
35
36 if ( args_.nRunningHours == Null<Size>()
37 && args_.nStarts == Null<Size>()) {
38 type_ = Vanilla;
39 }
40 else if (args_.nRunningHours == Null<Size>()) {
41 type_ = StartLimit;
42 }
43 else {
44 type_ = RunningHourLimit;
45 }
46 }

48 ext::shared_ptr<Fdm1dMesher>
49 FdmVPPStepConditionFactory::stateMesher() const {
50 Size nStates;
51 switch (type_) {
52 case Vanilla:
53 nStates = 2*args_.tMinUp + args_.tMinDown;
54 break;
55 case StartLimit:
56 nStates = FdmVPPStartLimitStepCondition::nStates(
57 args_.tMinUp,args_.tMinDown, args_.nStarts);
58 break;
59 default:
60 QL_FAIL("vpp type is not supported");
61 }
62
63 return ext::shared_ptr<Fdm1dMesher>(
64 new Uniform1dMesher(0.0, 1.0, nStates));
65 }
