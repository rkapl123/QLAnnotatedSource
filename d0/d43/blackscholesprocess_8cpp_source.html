75  Real GeneralizedBlackScholesProcess::drift(Time t, Real x) const {
76  Real sigma = diffusion(t,x);
77  // we could be more anticipatory if we know the right dt
78  // for which the drift will be used
79  Time t1 = t + 0.0001;
80  return riskFreeRate_->forwardRate(t,t1,Continuous,NoFrequency,true)
81  - dividendYield_->forwardRate(t,t1,Continuous,NoFrequency,true)
82  - 0.5 * sigma * sigma;
83  }

93  Real GeneralizedBlackScholesProcess::expectation(Time t0,
94  Real x0,
95  Time dt) const {
96  localVolatility(); // trigger update
97  if(isStrikeIndependent_ && !forceDiscretization_) {
98  // exact value for curves
99  return x0 *
100  std::exp(dt * (riskFreeRate_->forwardRate(t0, t0 + dt, Continuous,
101  NoFrequency, true) -
102  dividendYield_->forwardRate(
103  t0, t0 + dt, Continuous, NoFrequency, true)));
104  } else {
105  QL_FAIL("not implemented");
106  }
107  }

109  Real GeneralizedBlackScholesProcess::stdDeviation(Time t0, Real x0, Time dt) const {
110  localVolatility(); // trigger update
111  if(isStrikeIndependent_ && !forceDiscretization_) {
112  // exact value for curves
113  return std::sqrt(variance(t0,x0,dt));
114  }

120  Real GeneralizedBlackScholesProcess::variance(Time t0, Real x0, Time dt) const {
121  localVolatility(); // trigger update
122  if(isStrikeIndependent_ && !forceDiscretization_) {
123  // exact value for curves
124  return blackVolatility_->blackVariance(t0 + dt, 0.01) -
125  blackVolatility_->blackVariance(t0, 0.01);
126  }
127  else{
128  return discretization_->variance(*this,t0,x0,dt);
129  }
130  }

132  Real GeneralizedBlackScholesProcess::evolve(Time t0, Real x0,
133  Time dt, Real dw) const {
134  localVolatility(); // trigger update
135  if (isStrikeIndependent_ && !forceDiscretization_) {
136  // exact value for curves
137  Real var = variance(t0, x0, dt);
138  Real drift = (riskFreeRate_->forwardRate(t0, t0 + dt, Continuous,
139  NoFrequency, true) -
140  dividendYield_->forwardRate(t0, t0 + dt, Continuous,
141  NoFrequency, true)) *
142  dt -
143  0.5 * var;
144  return apply(x0, std::sqrt(var) * dw + drift);
145  } else
146  return apply(x0, discretization_->drift(*this, t0, x0, dt) +
147  stdDeviation(t0, x0, dt) * dw);
148  }

180  const Handle<LocalVolTermStructure>&
181  GeneralizedBlackScholesProcess::localVolatility() const {
182  if (hasExternalLocalVol_)
183  return externalLocalVolTS_;
184 
185  if (!updated_) {
186  isStrikeIndependent_=true;
187 
188  // constant Black vol?
189  ext::shared_ptr<BlackConstantVol> constVol =
190  ext::dynamic_pointer_cast<BlackConstantVol>(
191  *blackVolatility());
192  if (constVol != 0) {
193  // ok, the local vol is constant too.
194  localVolatility_.linkTo(ext::make_shared<LocalConstantVol>(
195  constVol->referenceDate(),
196  constVol->blackVol(0.0, x0_->value()),
197  constVol->dayCounter()));
198  updated_ = true;
199  return localVolatility_;
200  }
201 
202  // ok, so it's not constant. Maybe it's strike-independent?
203  ext::shared_ptr<BlackVarianceCurve> volCurve =
204  ext::dynamic_pointer_cast<BlackVarianceCurve>(
205  *blackVolatility());
206  if (volCurve != 0) {
207  // ok, we can use the optimized algorithm
208  localVolatility_.linkTo(ext::make_shared<LocalVolCurve>(
209  Handle<BlackVarianceCurve>(volCurve)));
210  updated_ = true;
211  return localVolatility_;
212  }
213 
214  // ok, so it's strike-dependent. Never mind.
215  localVolatility_.linkTo(
216  ext::make_shared<LocalVolSurface>(blackVolatility_, riskFreeRate_,
217  dividendYield_, x0_->value()));
218  updated_ = true;
219  isStrikeIndependent_ = false;
220  return localVolatility_;
221 
222  } else {
223  return localVolatility_;
224  }
225  }
