63 inline Real
64 SegmentIntegral::integrate(const ext::function<Real (Real)>& f,
65 Real a,
66 Real b) const {
67 if(close_enough(a,b))
68 return 0.0;
69 Real dx = (b-a)/intervals_;
70 Real sum = 0.5*(f(a)+f(b));
71 Real end = b - 0.5*dx;
72 for (Real x = a+dx; x < end; x += dx)
73 sum += f(x);
74 return sum*dx;
75 }
