42  Size maxEvaluations,
43  Volatility minVol,

46  QL_REQUIRE(!isExpired(), "option expired");
47 
48  boost::shared_ptr<SimpleQuote> volQuote(new SimpleQuote);
49 
50  boost::shared_ptr<GeneralizedBlackScholesProcess> newProcess =
51  detail::ImpliedVolatilityHelper::clone(process, volQuote);
52 
53  // engines are built-in for the time being
54  boost::scoped_ptr<PricingEngine> engine;
55  switch (exercise_->type()) {
56  case Exercise::European:
57  engine.reset(new AnalyticEuropeanEngine(newProcess));
58  break;
59  case Exercise::American:
60  engine.reset(new FDAmericanEngine<CrankNicolson>(newProcess));
61  break;
62  case Exercise::Bermudan:
63  engine.reset(new FDBermudanEngine<CrankNicolson>(newProcess));
64  break;
65  default:
66  QL_FAIL("unknown exercise type");
67  }
68 
69  return detail::ImpliedVolatilityHelper::calculate(*this,
70  *engine,
71  *volQuote,
72  targetValue,
73  accuracy,
74  maxEvaluations,
75  minVol, maxVol);
76  }
