45  template <class Curve>
46  Real dontThrowFallback(const BootstrapError<Curve>& error,
47  Real xMin, Real xMax, Size steps) {
48 
49  QL_REQUIRE(xMin < xMax, "Expected xMin to be less than xMax");
50 
51  // Set the initial value of the result to xMin and store the absolute bootstrap error at xMin
52  Real result = xMin;
53  Real absError = std::abs(error(xMin));
54  Real minError = absError;
55 
56  // Step out to xMax
57  Real stepSize = (xMax - xMin) / steps;
58  for (Size i = 0; i < steps; i++) {
59 
60  // Get absolute bootstrap error at updated x value
61  xMin += stepSize;
62  absError = std::abs(error(xMin));
63 
64  // If this absolute bootstrap error is less than the minimum, update result and minError
65  if (absError < minError) {
66  result = xMin;
67  minError = absError;
68  }
69  }
70 
71  return result;
72  }

74 }

155  template <class Curve>
156  void IterativeBootstrap<Curve>::initialize() const {
157  // ensure helpers are sorted
158  std::sort(ts_->instruments_.begin(), ts_->instruments_.end(),
159  detail::BootstrapHelperSorter());
160  // skip expired helpers
161  Date firstDate = Traits::initialDate(ts_);
162  QL_REQUIRE(ts_->instruments_[n_-1]->pillarDate()>firstDate,
163  "all instruments expired");
164  firstAliveHelper_ = 0;
165  while (ts_->instruments_[firstAliveHelper_]->pillarDate() <= firstDate)
166  ++firstAliveHelper_;
167  alive_ = n_-firstAliveHelper_;
168  Size nodes = alive_+1;
169  QL_REQUIRE(nodes >= Interpolator::requiredPoints,
170  "not enough alive instruments: " << alive_ <<
171  " provided, " << Interpolator::requiredPoints-1 <<
172  " required");
173 
174  // calculate dates and times, create errors_
175  std::vector<Date>& dates = ts_->dates_;
176  std::vector<Time>& times = ts_->times_;
177  dates.resize(alive_+1);
178  times.resize(alive_+1);
179  errors_.resize(alive_+1);
180  dates[0] = firstDate;
181  times[0] = ts_->timeFromReference(dates[0]);
182 
183  Date latestRelevantDate, maxDate = firstDate;
184  // pillar counter: i
185  // helper counter: j
186  for (Size i=1, j=firstAliveHelper_; j<n_; ++i, ++j) {
187  const ext::shared_ptr<typename Traits::helper>& helper =
188  ts_->instruments_[j];
189  dates[i] = helper->pillarDate();
190  times[i] = ts_->timeFromReference(dates[i]);
191  // check for duplicated pillars
192  QL_REQUIRE(dates[i-1]!=dates[i],
193  "more than one instrument with pillar " << dates[i]);
194 
195  latestRelevantDate = helper->latestRelevantDate();
196  // check that the helper is really extending the curve, i.e. that
197  // pillar-sorted helpers are also sorted by latestRelevantDate
198  QL_REQUIRE(latestRelevantDate > maxDate,
199  io::ordinal(j+1) << " instrument (pillar: " <<
200  dates[i] << ") has latestRelevantDate (" <<
201  latestRelevantDate << ") before or equal to "
202  "previous instrument's latestRelevantDate (" <<
203  maxDate << ")");
204  maxDate = latestRelevantDate;
205 
206  // when a pillar date is different from the last relevant date the
207  // convergence loop is required even if the Interpolator is local
208  if (dates[i] != latestRelevantDate)
209  loopRequired_ = true;
210 
211  errors_[i] = ext::shared_ptr<BootstrapError<Curve> >(new
212  BootstrapError<Curve>(ts_, helper, i));
213  }
214  ts_->maxDate_ = maxDate;
215 
216  // set initial guess only if the current curve cannot be used as guess
217  if (!validCurve_ || ts_->data_.size()!=alive_+1) {
218  // ts_->data_[0] is the only relevant item,
219  // but reasonable numbers might be needed for the whole data vector
220  // because, e.g., of interpolation's early checks
221  ts_->data_ = std::vector<Real>(alive_+1, Traits::initialValue(ts_));
222  previousData_.resize(alive_+1);
223  validCurve_ = false;
224  }
225  initialized_ = true;
226  }

228  template <class Curve>
229  void IterativeBootstrap<Curve>::calculate() const {
230 
231  // we might have to call initialize even if the curve is initialized
232  // and not moving, just because helpers might be date relative and change
233  // with evaluation date change.
234  // anyway it makes little sense to use date relative helpers with a
235  // non-moving curve if the evaluation date changes
236  if (!initialized_ || ts_->moving_)
237  initialize();
238 
239  // setup helpers
240  for (Size j=firstAliveHelper_; j<n_; ++j) {
241  const ext::shared_ptr<typename Traits::helper>& helper =
242  ts_->instruments_[j];
243  // check for valid quote
244  QL_REQUIRE(helper->quote()->isValid(),
245  io::ordinal(j + 1) << " instrument (maturity: " <<
246  helper->maturityDate() << ", pillar: " <<
247  helper->pillarDate() << ") has an invalid quote");
248  // don't try this at home!
249  // This call creates helpers, and removes "const".
250  // There is a significant interaction with observability.
251  helper->setTermStructure(const_cast<Curve*>(ts_));
252  }
253 
254  const std::vector<Time>& times = ts_->times_;
255  const std::vector<Real>& data = ts_->data_;
256  Real accuracy = accuracy_ != Null<Real>() ? accuracy_ : ts_->accuracy_;
257 
258  Size maxIterations = Traits::maxIterations()-1;
259 
260  // there might be a valid curve state to use as guess
261  bool validData = validCurve_;
262 
263  for (Size iteration=0; ; ++iteration) {
264  previousData_ = ts_->data_;
265 
266  // Store min value and max value at each pillar so that we can expand search if necessary.
267  std::vector<Real> minValues(alive_+1, Null<Real>());
268  std::vector<Real> maxValues(alive_+1, Null<Real>());
269  std::vector<Size> attempts(alive_+1, 1);
270 
271  for (Size i=1; i<=alive_; ++i) { // pillar loop
272 
273  // shorter aliases for readability and to avoid duplication
274  Real& min = minValues[i];
275  Real& max = maxValues[i];
276 
277  // bracket root and calculate guess
278  if (min == Null<Real>()) {
279  // First attempt; we take min and max either from
280  // explicit constructor parameter or from traits
281  min = (minValue_ != Null<Real>() ? minValue_ :
282  Traits::minValueAfter(i, ts_, validData, firstAliveHelper_));
283  max = (maxValue_ != Null<Real>() ? maxValue_ :
284  Traits::maxValueAfter(i, ts_, validData, firstAliveHelper_));
285  } else {
286  // Extending a previous attempt. A negative min
287  // is enlarged; a positive one is shrunk towards 0.
288  min = (min < 0.0 ? min * minFactor_ : min / minFactor_);
289  // The opposite holds for the max.
290  max = (max > 0.0 ? max * maxFactor_ : max / maxFactor_);
291  }
292  Real guess = Traits::guess(i, ts_, validData, firstAliveHelper_);
293 
294  // adjust guess if needed
295  if (guess >= max)
296  guess = max - (max - min) / 5.0;
297  else if (guess <= min)
298  guess = min + (max - min) / 5.0;
299 
300  // extend interpolation if needed
301  if (!validData) {
302  try { // extend interpolation a point at a time
303  // including the pillar to be boostrapped
304  ts_->interpolation_ = ts_->interpolator_.interpolate(
305  times.begin(), times.begin()+i+1, data.begin());
306  } catch (...) {
307  if (!Interpolator::global)
308  throw; // no chance to fix it in a later iteration
309 
310  // otherwise use Linear while the target
311  // interpolation is not usable yet
312  ts_->interpolation_ = Linear().interpolate(
313  times.begin(), times.begin()+i+1, data.begin());
314  }
315  ts_->interpolation_.update();
316  }
317 
318  try {
319  if (validData)
320  solver_.solve(*errors_[i], accuracy, guess, min, max);
321  else
322  firstSolver_.solve(*errors_[i], accuracy, guess, min, max);
323  } catch (std::exception &e) {
324  if (validCurve_) {
325  // the previous curve state might have been a
326  // bad guess, so we retry without using it.
327  // This would be tricky to do here (we're
328  // inside multiple nested for loops, we need
329  // to re-initialize...), so we invalidate the
330  // curve, make a recursive call and then exit.
331  validCurve_ = initialized_ = false;
332  calculate();
333  return;
334  }
335 
336  // If we have more attempts left on this iteration, try again. Note that the max and min
337  // bounds will be widened on the retry.
338  if (attempts[i] < maxAttempts_) {
339  attempts[i]++;
340  i--;
341  continue;
342  }
343 
344  if (dontThrow_) {
345  // Use the fallback value
346  ts_->data_[i] = detail::dontThrowFallback(*errors_[i], min, max, dontThrowSteps_);
347 
348  // Remember to update the interpolation. If we don't and we are on the last "i", we will still
349  // have the last attempted value in the solver being used in ts_->interpolation_.
350  ts_->interpolation_.update();
351  } else {
352  QL_FAIL(io::ordinal(iteration + 1) << " iteration: failed "
353  "at " << io::ordinal(i) << " alive instrument, "
354  "pillar " << errors_[i]->helper()->pillarDate() <<
355  ", maturity " << errors_[i]->helper()->maturityDate() <<
356  ", reference date " << ts_->dates_[0] <<
357  ": " << e.what());
358  }
359  }
360  }
361 
362  if (!loopRequired_)
363  break;
364 
365  // exit condition
366  Real change = std::fabs(data[1]-previousData_[1]);
367  for (Size i=2; i<=alive_; ++i)
368  change = std::max(change, std::fabs(data[i]-previousData_[i]));
369  if (change<=accuracy) // convergence reached
370  break;
371 
372  // If we hit the max number of iterations and dontThrow is true, just use what we have
373  if (iteration == maxIterations) {
374  if (dontThrow_) {
375  break;
376  } else {
377  QL_FAIL("convergence not reached after " << iteration <<
378  " iterations; last improvement " << change <<
379  ", required accuracy " << accuracy);
380  }
381  }
382 
383  validData = true;
384  }
385  validCurve_ = true;
386  }
387 
388 }
