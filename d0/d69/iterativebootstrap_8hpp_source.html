69  void IterativeBootstrap<Curve>::setup(Curve* ts) {
70 
71  ts_ = ts;
72  n_ = ts_->instruments_.size();
73  QL_REQUIRE(n_ > 0, "no bootstrap helpers given")
74  for (Size j=0; j<n_; ++j)
75  ts_->registerWith(ts_->instruments_[j]);
76 
77  // do not initialize yet: instruments could be invalid here
78  // but valid later when bootstrapping is actually required
79  }

81  template <class Curve>
82  void IterativeBootstrap<Curve>::initialize() const {
83  // ensure helpers are sorted
84  std::sort(ts_->instruments_.begin(), ts_->instruments_.end(),
85  detail::BootstrapHelperSorter());
86  // skip expired helpers
87  Date firstDate = Traits::initialDate(ts_);
88  QL_REQUIRE(ts_->instruments_[n_-1]->pillarDate()>firstDate,
89  "all instruments expired");
90  firstAliveHelper_ = 0;
91  while (ts_->instruments_[firstAliveHelper_]->pillarDate() <= firstDate)
92  ++firstAliveHelper_;
93  alive_ = n_-firstAliveHelper_;
94  QL_REQUIRE(alive_>=Interpolator::requiredPoints-1,
95  "not enough alive instruments: " << alive_ <<
96  " provided, " << Interpolator::requiredPoints-1 <<
97  " required");
98 
99  // calculate dates and times, create errors_
100  std::vector<Date>& dates = ts_->dates_;
101  std::vector<Time>& times = ts_->times_;
102  dates.resize(alive_+1);
103  times.resize(alive_+1);
104  errors_.resize(alive_+1);
105  dates[0] = firstDate;
106  times[0] = ts_->timeFromReference(dates[0]);
107 
108  Date latestRelevantDate, maxDate = firstDate;
109  // pillar counter: i
110  // helper counter: j
111  for (Size i=1, j=firstAliveHelper_; j<n_; ++i, ++j) {
112  const boost::shared_ptr<typename Traits::helper>& helper =
113  ts_->instruments_[j];
114  dates[i] = helper->pillarDate();
115  times[i] = ts_->timeFromReference(dates[i]);
116  // check for duplicated pillars
117  QL_REQUIRE(dates[i-1]!=dates[i],
118  "more than one instrument with pillar " << dates[i]);
119 
120  latestRelevantDate = helper->latestRelevantDate();
121  // check that the helper is really extending the curve, i.e. that
122  // pillar-sorted helpers are also sorted by latestRelevantDate
123  QL_REQUIRE(latestRelevantDate > maxDate,
124  io::ordinal(j+1) << " instrument (pillar: " <<
125  dates[i] << ") has latestRelevantDate (" <<
126  latestRelevantDate << ") before or equal to "
127  "previous instrument's latestRelevantDate (" <<
128  maxDate << ")");
129  maxDate = latestRelevantDate;
130 
131  // when a pillar date is different from the last relevant date the
132  // convergence loop is required even if the Interpolator is local
133  if (dates[i] != latestRelevantDate)
134  loopRequired_ = true;
135 
136  errors_[i] = boost::shared_ptr<BootstrapError<Curve> >(new
137  BootstrapError<Curve>(ts_, helper, i));
138  }
139  ts_->maxDate_ = maxDate;
140 
141  // set initial guess only if the current curve cannot be used as guess
142  if (!validCurve_ || ts_->data_.size()!=alive_+1) {
143  // ts_->data_[0] is the only relevant item,
144  // but reasonable numbers might be needed for the whole data vector
145  // because, e.g., of interpolation's early checks
146  ts_->data_ = std::vector<Real>(alive_+1, Traits::initialValue(ts_));
147  previousData_.resize(alive_+1);
148  }
149  initialized_ = true;
150  }

152  template <class Curve>
153  void IterativeBootstrap<Curve>::calculate() const {
154 
155  // we might have to call initialize even if the curve is initialized
156  // and not moving, just because helpers might be date relative and change
157  // with evaluation date change.
158  // anyway it makes little sense to use date relative helpers with a
159  // non-moving curve if the evaluation date changes
160  if (!initialized_ || ts_->moving_)
161  initialize();
162 
163  // setup helpers
164  for (Size j=firstAliveHelper_; j<n_; ++j) {
165  const boost::shared_ptr<typename Traits::helper>& helper =
166  ts_->instruments_[j];
167  // check for valid quote
168  QL_REQUIRE(helper->quote()->isValid(),
169  io::ordinal(j + 1) << " instrument (maturity: " <<
170  helper->maturityDate() << ", pillar: " <<
171  helper->pillarDate() << ") has an invalid quote");
172  // don't try this at home!
173  // This call creates helpers, and removes "const".
174  // There is a significant interaction with observability.
175  helper->setTermStructure(const_cast<Curve*>(ts_));
176  }
177 
178  const std::vector<Time>& times = ts_->times_;
179  const std::vector<Real>& data = ts_->data_;
180  Real accuracy = ts_->accuracy_;
181 
182  Size maxIterations = Traits::maxIterations()-1;
183 
184  // there might be a valid curve state to use as guess
185  bool validData = validCurve_;
186 
187  for (Size iteration=0; ; ++iteration) {
188  previousData_ = ts_->data_;
189 
190  for (Size i=1; i<=alive_; ++i) { // pillar loop
191 
192  // bracket root and calculate guess
193  Real min = Traits::minValueAfter(i, ts_, validData,
194  firstAliveHelper_);
195  Real max = Traits::maxValueAfter(i, ts_, validData,
196  firstAliveHelper_);
197  Real guess = Traits::guess(i, ts_, validData,
198  firstAliveHelper_);
199  // adjust guess if needed
200  if (guess>=max)
201  guess = max - (max-min)/5.0;
202  else if (guess<=min)
203  guess = min + (max-min)/5.0;
204 
205  // extend interpolation if needed
206  if (!validData) {
207  try { // extend interpolation a point at a time
208  // including the pillar to be boostrapped
209  ts_->interpolation_ = ts_->interpolator_.interpolate(
210  times.begin(), times.begin()+i+1, data.begin());
211  } catch (...) {
212  if (!Interpolator::global)
213  throw; // no chance to fix it in a later iteration
214 
215  // otherwise use Linear while the target
216  // interpolation is not usable yet
217  ts_->interpolation_ = Linear().interpolate(
218  times.begin(), times.begin()+i+1, data.begin());
219  }
220  ts_->interpolation_.update();
221  }
222 
223  try {
224  if (validData)
225  solver_.solve(*errors_[i], accuracy, guess, min, max);
226  else
227  firstSolver_.solve(*errors_[i], accuracy,guess,min,max);
228  } catch (std::exception &e) {
229  if (validCurve_) {
230  // the previous curve state might have been a
231  // bad guess, so we retry without using it.
232  // This would be tricky to do here (we're
233  // inside multiple nested for loops, we need
234  // to re-initialize...), so we invalidate the
235  // curve, make a recursive call and then exit.
236  validCurve_ = initialized_ = false;
237  calculate();
238  return;
239  }
240  QL_FAIL(io::ordinal(iteration+1) << " iteration: failed "
241  "at " << io::ordinal(i) << " alive instrument, "
242  "pillar " << errors_[i]->helper()->pillarDate() <<
243  ", maturity " << errors_[i]->helper()->maturityDate() <<
244  ", reference date " << ts_->dates_[0] <<
245  ": " << e.what());
246  }
247  }
248 
249  if (!loopRequired_)
250  break;
251 
252  // exit condition
253  Real change = std::fabs(data[1]-previousData_[1]);
254  for (Size i=2; i<=alive_; ++i)
255  change = std::max(change, std::fabs(data[i]-previousData_[i]));
256  if (change<=accuracy) // convergence reached
257  break;
258 
259  QL_REQUIRE(iteration<maxIterations,
260  "convergence not reached after " << iteration <<
261  " iterations; last improvement " << change <<
262  ", required accuracy " << accuracy);
263 
264  validData = true;
265  }
266  validCurve_ = true;
267  }
