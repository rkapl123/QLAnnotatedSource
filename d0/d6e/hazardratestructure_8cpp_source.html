41  template <class F>
42  struct remapper {
43  F f;
44  Time T;
45  remapper(const F& f, Time T) : f(f), T(T) {}
46  // This remaps [-1,1] to [0,T]. No differential included.
47  Real operator()(Real x) const {
48  const Real arg = (x+1.0)*T/2.0;
49  return f(arg);
50  }
51  };

53  template <class F>
54  remapper<F> remap(const F& f, Time T) {
55  return remapper<F>(f,T);

82  Probability HazardRateStructure::survivalProbabilityImpl(Time t) const {
83  static GaussChebyshevIntegration integral(48);
84  // this stores the address of the method to integrate (so that
85  // we don't have to insert its full expression inside the
86  // integral below--it's long enough already)
87  Real (HazardRateStructure::*f)(Time) const =
88  &HazardRateStructure::hazardRateImpl;
89  // the Gauss-Chebyshev quadratures integrate over [-1,1],
90  // hence the remapping (and the Jacobian term t/2)
91  return std::exp(-integral(remap(bind(f,this,_1), t)) * t/2.0);
92  }
