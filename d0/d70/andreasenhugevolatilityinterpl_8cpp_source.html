59  class AndreasenHugeCostFunction : public CostFunction {
60  public:
61  AndreasenHugeCostFunction(
62  const Array& marketNPVs,
63  const Array& marketVegas,
64  const Array& lnMarketStrikes,
65  const Array& previousNPVs,
66  const ext::shared_ptr<FdmMesherComposite>& mesher,
67  Time dT,
68  AndreasenHugeVolatilityInterpl::InterpolationType interpolationType)
69  : marketNPVs_(marketNPVs), marketVegas_(marketVegas), lnMarketStrikes_(lnMarketStrikes),
70  previousNPVs_(previousNPVs), mesher_(mesher), nGridPoints_(mesher->layout()->size()),
71  dT_(dT), interpolationType_((lnMarketStrikes_.size() > 1) ?
72  interpolationType :
73  AndreasenHugeVolatilityInterpl::PiecewiseConstant),
74  dxMap_(FirstDerivativeOp(0, mesher_)), dxxMap_(SecondDerivativeOp(0, mesher_)),
75  d2CdK2_(dxMap_.mult(Array(mesher->layout()->size(), -1.0)).add(dxxMap_)),
76  mapT_(0, mesher_) {}

78  Disposable<Array> d2CdK2(const Array& c) const {
79  return d2CdK2_.apply(c);
80  }

82  Disposable<Array> solveFor(
83  Time dT, const Array& sig, const Array& b) const {
84 
85  Array x(lnMarketStrikes_.size());
86  Interpolation sigInterpl;
87 
88  switch (interpolationType_) {
89  case AndreasenHugeVolatilityInterpl::CubicSpline:
90  sigInterpl = CubicNaturalSpline(
91  lnMarketStrikes_.begin(), lnMarketStrikes_.end(),
92  sig.begin());
93  break;
94  case AndreasenHugeVolatilityInterpl::Linear:
95  sigInterpl = LinearInterpolation(
96  lnMarketStrikes_.begin(), lnMarketStrikes_.end(),
97  sig.begin());
98  break;
99  case AndreasenHugeVolatilityInterpl::PiecewiseConstant:
100  for (Size i=0; i < x.size()-1; ++i)
101  x[i] = 0.5*(lnMarketStrikes_[i] + lnMarketStrikes_[i+1]);
102  x.back() = lnMarketStrikes_.back();
103 
104  sigInterpl = BackwardFlatInterpolation(
105  x.begin(), x.end(), sig.begin());
106  break;
107  default:
108  QL_FAIL("unknown interpolation type");
109  }
110 
111  const ext::shared_ptr<FdmLinearOpLayout> layout =
112  mesher_->layout();
113  const FdmLinearOpIterator endIter = layout->end();
114 
115  Array z(layout->size());
116  for (FdmLinearOpIterator iter = layout->begin();
117  iter!=endIter; ++iter) {
118  const Size i = iter.index();
119  const Real lnStrike = mesher_->location(iter, 0);
120 
121  const Real vol = sigInterpl(
122  std::min(std::max(lnStrike, lnMarketStrikes_.front()),
123  lnMarketStrikes_.back()), true);
124 
125  z[i] = 0.5*vol*vol;
126  }
127 
128  mapT_.axpyb(z, dxMap_, dxxMap_.mult(-z), Array());
129  return mapT_.mult(Array(z.size(), dT)).solve_splitting(b, 1.0);
130 
131  }
132 
133  Disposable<Array> apply(const Array& c) const {
134  return -mapT_.apply(c);
135  }
136 
137  Disposable<Array> values(const Array& sig) const {
138  Array newNPVs = solveFor(dT_, sig, previousNPVs_);
139 
140  const std::vector<Real>& gridPoints =
141  mesher_->getFdm1dMeshers().front()->locations();
142 
143  const MonotonicCubicNaturalSpline interpl(
144  gridPoints.begin(), gridPoints.end(), newNPVs.begin());
145 
146  Array retVal(lnMarketStrikes_.size());
147  for (Size i=0; i < retVal.size(); ++i) {
148  const Real strike = lnMarketStrikes_[i];
149  retVal[i] = interpl(strike) - marketNPVs_[i];
150  }
151  return retVal;
152  }

154  Disposable<Array> vegaCalibrationError(const Array& sig) const {

156  }
157 
158  Disposable<Array> initialValues() const {
159  Array retVal(lnMarketStrikes_.size(), 0.25);
160  return retVal;
161  }

164  private:
165  const Array marketNPVs_, marketVegas_;
166  const Array lnMarketStrikes_, previousNPVs_;
167  const ext::shared_ptr<FdmMesherComposite> mesher_;
168  const Size nGridPoints_;

179  class CombinedCostFunction : public CostFunction {
180  public:
181  CombinedCostFunction(
182  const ext::shared_ptr<AndreasenHugeCostFunction>& putCostFct,
183  const ext::shared_ptr<AndreasenHugeCostFunction>& callCostFct)
184  : putCostFct_(putCostFct),
185  callCostFct_(callCostFct) { }
186 
187  Disposable<Array> values(const Array& sig) const {
188  if ((putCostFct_ != 0) && (callCostFct_ != 0)) {
189  const Array pv = putCostFct_->values(sig);
190  const Array cv = callCostFct_->values(sig);
191 
192  Array retVal(pv.size() + cv.size());
193  std::copy(pv.begin(), pv.end(), retVal.begin());
194  std::copy(cv.begin(), cv.end(), retVal.begin() + cv.size());
195 
196  return retVal;
197  } else if (putCostFct_ != 0)
198  return putCostFct_->values(sig);
199  else if (callCostFct_ != 0)
200  return callCostFct_->values(sig);
201  else
202  QL_FAIL("internal error: cost function not set");
203  }

206  if ((putCostFct_ != 0) && (callCostFct_ != 0))
207  return 0.5*( putCostFct_->initialValues()
208  + callCostFct_->initialValues());
209  else if (putCostFct_ != 0)
210  return putCostFct_->initialValues();
211  else if (callCostFct_ != 0)
212  return callCostFct_->initialValues();
213  else
214  QL_FAIL("internal error: cost function not set");
215  }

217  private:
218  const ext::shared_ptr<AndreasenHugeCostFunction> putCostFct_;
219  const ext::shared_ptr<AndreasenHugeCostFunction> callCostFct_;

224  AndreasenHugeVolatilityInterpl::AndreasenHugeVolatilityInterpl(
225  const CalibrationSet& calibrationSet,
226  const Handle<Quote>& spot,
227  const Handle<YieldTermStructure>& rTS,
228  const Handle<YieldTermStructure>& qTS,
229  InterpolationType interplationType,
230  CalibrationType calibrationType,
231  Size nGridPoints,
232  Real _minStrike,
233  Real _maxStrike,
234  const ext::shared_ptr<OptimizationMethod>& optimizationMethod,
235  const EndCriteria& endCriteria)
236  : spot_(spot),
237  rTS_(rTS),
238  qTS_(qTS),
239  interpolationType_(interplationType),
240  calibrationType_(calibrationType),
241  nGridPoints_(nGridPoints),
242  minStrike_(_minStrike),
243  maxStrike_(_maxStrike),
244  optimizationMethod_(optimizationMethod),
245  endCriteria_(endCriteria) {
246  QL_REQUIRE(nGridPoints > 2 && !calibrationSet.empty(), "undefined grid or calibration set");
247 
248  std::set<Real> strikes;
249  std::set<Date> expiries;
250 
251  calibrationSet_.reserve(calibrationSet.size());
252  for (Size i=0; i < calibrationSet.size(); ++i) {
253 
254  const ext::shared_ptr<Exercise> exercise =
255  calibrationSet[i].first->exercise();
256 
257  QL_REQUIRE(exercise->type() == Exercise::European,
258  "European option required");
259 
260  const Date expiry = exercise->lastDate();
261  expiries.insert(expiry);
262 
263  const ext::shared_ptr<PlainVanillaPayoff> payoff =
264  ext::dynamic_pointer_cast<PlainVanillaPayoff>(
265  calibrationSet[i].first->payoff());
266 
267  QL_REQUIRE(payoff, "plain vanilla payoff required");
268 
269  const Real strike = payoff->strike();
270  strikes.insert(strike);
271 
272  calibrationSet_.push_back(
273  std::make_pair(
274  ext::make_shared<VanillaOption>(payoff, exercise),
275  calibrationSet[i].second)
276  );
277 
278  registerWith(calibrationSet[i].second);
279  }
280 
281  strikes_.assign(strikes.begin(), strikes.end());
282  expiries_.assign(expiries.begin(), expiries.end());
283 
284  dT_.resize(expiries_.size());
285  expiryTimes_.resize(expiries_.size());
286 
287  calibrationMatrix_ = std::vector< std::vector<Size> >(
288  expiries.size(), std::vector<Size>(strikes.size(), Null<Size>()));
289 
290  for (Size i=0; i < calibrationSet.size(); ++i) {
291  const Date expiry =
292  calibrationSet[i].first->exercise()->lastDate();
293 
294  const Size l = std::distance(expiries.begin(), expiries.lower_bound(expiry));
295 
296  const Real strike =
297  ext::dynamic_pointer_cast<PlainVanillaPayoff>(
298  calibrationSet[i].first->payoff())->strike();
299 
300  const Size k = std::distance(strikes_.begin(),
301  std::find_if(strikes_.begin(), strikes_.end(),
302  close_enough_to(strike)));
303 
304  calibrationMatrix_[l][k] = i;
305  }
306 
307  registerWith(spot_);
308  registerWith(rTS_);
309  registerWith(qTS_);
310  }

312  ext::shared_ptr<AndreasenHugeCostFunction>
313  AndreasenHugeVolatilityInterpl::buildCostFunction(
314  Size iExpiry, Option::Type optionType,
315  const Array& previousNPVs) const {
316 
317  if (calibrationType_ != CallPut
318  && ( (calibrationType_ == Call && optionType ==Option::Put)
319  || (calibrationType_ == Put && optionType ==Option::Call)))
320  return ext::shared_ptr<AndreasenHugeCostFunction>();
321 
322  const Time expiryTime = expiryTimes_[iExpiry];
323 
324  const DiscountFactor discount = rTS_->discount(expiryTime);
325  const Real fwd = spot_->value()*qTS_->discount(expiryTime)/discount;
326 
327  const Size nOptions = std::count_if(
328  calibrationMatrix_[iExpiry].begin(),
329  calibrationMatrix_[iExpiry].end(),
330  not_null<Size>());
331 
332  Array lnMarketStrikes(nOptions),
333  marketNPVs(nOptions), marketVegas(nOptions);
334 
335  // calculate undiscounted market prices
336  for (Size j=0, k=0; j < strikes_.size(); ++j) {
337  const Size idx = calibrationMatrix_[iExpiry][j];
338 
339  if (idx != Null<Size>()) {
340 
341  const Volatility vol = calibrationSet_[idx].second->value();
342  const Real stdDev = vol*std::sqrt(expiryTime);
343 
344  const BlackCalculator calculator(
345  optionType, strikes_[j], fwd, stdDev, discount);
346 
347  const Real npv = calculator.value();
348  const Real vega = calculator.vega(expiryTime);
349 
350  marketNPVs[k] = npv/(discount*fwd);
351  marketVegas[k] = vega/(discount*fwd);
352  lnMarketStrikes[k++] = std::log(strikes_[j]/fwd);
353  }
354  }
355 
356  return ext::make_shared<AndreasenHugeCostFunction>(
357  marketNPVs,
358  marketVegas,
359  lnMarketStrikes,
360  previousNPVs,
361  mesher_,
362  dT_[iExpiry],
363  interpolationType_);
364  }

367  void AndreasenHugeVolatilityInterpl::performCalculations() const {
368  QL_REQUIRE(maxStrike() > minStrike(),
369  "max strike must be greater than min strike");
370 
371  const DayCounter dc = rTS_->dayCounter();
372  for (Size i=0; i < expiryTimes_.size(); ++i) {
373  expiryTimes_[i] =
374  dc.yearFraction(rTS_->referenceDate(), expiries_[i]);
375  dT_[i] = expiryTimes_[i] - ( (i==0)? 0.0 : expiryTimes_[i-1]);
376  }
377 
378  mesher_ =
379  ext::make_shared<FdmMesherComposite>(
380  ext::make_shared<Concentrating1dMesher>(
381  std::log(minStrike()/spot_->value()),
382  std::log(maxStrike()/spot_->value()),
383  nGridPoints_,
384  std::pair<Real, Real>(0.0, 0.025)));
385 
386  gridPoints_ = mesher_->locations(0);
387  gridInFwd_ = Exp(gridPoints_)*spot_->value();
388 
389  localVolCache_.clear();
390  calibrationResults_.clear();
391 
392  avgError_ = 0.0;
393  minError_ = std::numeric_limits<Real>::max();
394  maxError_ = 0.0;
395 
396  calibrationResults_.reserve(expiries_.size());
397 
398  Array npvPuts(nGridPoints_);
399  Array npvCalls(nGridPoints_);
400 
401  for (Size i=0; i < nGridPoints_; ++i) {
402  const Real strike = std::exp(gridPoints_[i]);
403  npvPuts[i] = PlainVanillaPayoff(Option::Put, strike)(1.0);
404  npvCalls[i]= PlainVanillaPayoff(Option::Call, strike)(1.0);
405  }
406 
407  for (Size i=0; i < expiries_.size(); ++i) {
408  const ext::shared_ptr<AndreasenHugeCostFunction> putCostFct =
409  buildCostFunction(i, Option::Put, npvPuts);
410  const ext::shared_ptr<AndreasenHugeCostFunction> callCostFct =
411  buildCostFunction(i, Option::Call, npvCalls);
412 
413  CombinedCostFunction costFunction(putCostFct, callCostFct);
414 
415  PositiveConstraint positiveConstraint;
416  Problem problem(costFunction,
417  positiveConstraint, costFunction.initialValues());
418 
419  optimizationMethod_->minimize(problem, endCriteria_);
420 
421  const Array& sig = problem.currentValue();
422 
423  const SingleStepCalibrationResult calibrationResult = {
424  npvPuts, npvCalls, sig,
425  (calibrationType_ == Call)? callCostFct : putCostFct
426  };
427 
428  calibrationResults_.push_back(calibrationResult);
429 
430  Array vegaDiffs(sig.size());
431  switch (calibrationType_) {
432  case CallPut: {
433  const Array vegaPutDiffs =
434  putCostFct->vegaCalibrationError(sig);
435  const Array vegaCallDiffs =
436  callCostFct->vegaCalibrationError(sig);
437 
438  const Real fwd = spot_->value()*
439  qTS_->discount(expiryTimes_[i])/rTS_->discount(expiryTimes_[i]);
440 
441  for (Size j=0; j < vegaDiffs.size(); ++j)
442  vegaDiffs[j] = std::fabs(
443  (fwd > gridInFwd_[j])? vegaPutDiffs[j] : vegaCallDiffs[j]);
444  }
445  break;
446  case Put:
447  vegaDiffs = Abs(putCostFct->vegaCalibrationError(sig));
448  break;
449  case Call:
450  vegaDiffs = Abs(callCostFct->vegaCalibrationError(sig));
451  break;
452  default:
453  QL_FAIL("unknown calibration type");
454  }
455 
456  avgError_ +=
457  std::accumulate(vegaDiffs.begin(), vegaDiffs.end(), 0.0);
458  minError_ = std::min(minError_,
459  *std::min_element(vegaDiffs.begin(), vegaDiffs.end()));
460  maxError_ = std::max(maxError_,
461  *std::max_element(vegaDiffs.begin(), vegaDiffs.end()));
462 
463  if (putCostFct != 0)
464  npvPuts = putCostFct->solveFor(dT_[i], sig, npvPuts);
465  if (callCostFct != 0)
466  npvCalls= callCostFct->solveFor(dT_[i], sig, npvCalls);
467  }
468 
469  avgError_ /= calibrationSet_.size();
470  }

533  Real AndreasenHugeVolatilityInterpl::optionPrice(
534  Time t, Real strike, Option::Type optionType) const {
535 
536  TimeValueCacheType::const_iterator f = priceCache_.find(t);
537 
538  const DiscountFactor df = rTS_->discount(t);
539 
540  if (f != priceCache_.end()) {
541  const Real fwd = ext::get<0>(f->second);
542 
543  Real price = getCacheValue(strike, f);
544 
545  if (optionType == Option::Put
546  && (calibrationType_ == Call || calibrationType_ == CallPut))
547  price = price + strike/fwd - 1.0;
548  else if (optionType == Option::Call && calibrationType_ == Put)
549  price = 1.0 - strike/fwd + price;
550 
551  return price*df*fwd;
552  }
553 
554  calculate();
555 
556 
557  ext::shared_ptr<Array> prices(
558  ext::make_shared<Array>(gridPoints_));
559 
560  switch (calibrationType_) {
561  case Put:
562  (*prices) = getPriceSlice(t, Option::Put);
563  break;
564  case Call:
565  case CallPut:
566  (*prices) = getPriceSlice(t, Option::Call);
567  break;
568  default:
569  QL_FAIL("unknown calibration type");
570  }
571 
572  Real fwd = spot_->value()*qTS_->discount(t)/df;
573 
574  priceCache_[t] = ext::make_tuple(
575  fwd, prices,
576  ext::make_shared<CubicNaturalSpline>(
577  gridPoints_.begin()+1, gridPoints_.end()-1,
578  prices->begin()+1));
579 
580  return this->optionPrice(t, strike, optionType);
581  }

583  Disposable<Array> AndreasenHugeVolatilityInterpl::getLocalVolSlice(
584  Time t, Option::Type optionType) const {
585 
586  const Size iu = getExerciseTimeIdx(t);
587 
588  const Array& previousNPVs =
589  (optionType == Option::Call)? calibrationResults_[iu].callNPVs
590  : calibrationResults_[iu].putNPVs;
591 
592  const ext::shared_ptr<AndreasenHugeCostFunction> costFunction
593  = calibrationResults_[iu].costFunction;
594 
595  const Time dt = (iu == 0) ? t : t-expiryTimes_[iu-1];
596  const Array& sig = calibrationResults_[iu].sigmas;
597 
598  const Array cAtJ = costFunction->solveFor(dt, sig, previousNPVs);
599 
600  const Array dCdT =
601  costFunction->solveFor(dt, sig,
602  costFunction->apply(
603  costFunction->solveFor(dt, sig, previousNPVs)));
604 
605  const Array d2CdK2 = costFunction->d2CdK2(cAtJ);
606 
607  Array localVol = Sqrt(2*dCdT/d2CdK2);
608 
609  for (Size i=1; i < localVol.size()-1; ++i)
610  if (!boost::math::isfinite(localVol[i]) || localVol[i] < 0.0)
611  localVol[i] = 0.25;
612 
613  return localVol;
614  }

616  Volatility AndreasenHugeVolatilityInterpl::localVol(Time t, Real strike)
617  const {
618  TimeValueCacheType::const_iterator f = localVolCache_.find(t);
619 
620  if (f != localVolCache_.end())
621  return getCacheValue(strike, f);
622 
623  calculate();
624 
625  ext::shared_ptr<Array> localVol(
626  ext::make_shared<Array>(gridPoints_.size()));
627 
628  switch (calibrationType_) {
629  case CallPut: {
630  const Array putLocalVol = getLocalVolSlice(t, Option::Put);
631  const Array callLocalVol = getLocalVolSlice(t, Option::Call);
632 
633  for (Size i=0, n=localVol->size(); i < n; ++i)
634  (*localVol)[i] =
635  (gridPoints_[i] > 0.0)? callLocalVol[i] : putLocalVol[i];
636  }
637  break;
638  case Put:
639  (*localVol) = getLocalVolSlice(t, Option::Put);
640  break;
641  case Call:
642  (*localVol) = getLocalVolSlice(t, Option::Call);
643  break;
644  default:
645  QL_FAIL("unknown calibration type");
646  }
647 
648  Real fwd = spot_->value()*qTS_->discount(t)/rTS_->discount(t);
649 
650  localVolCache_[t] = ext::make_tuple(
