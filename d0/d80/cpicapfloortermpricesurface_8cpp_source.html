27  CPICapFloorTermPriceSurface::CPICapFloorTermPriceSurface(
28  Real nominal,
29  Real baseRate, // avoids an uncontrolled crash if index has no TS
30  const Period& observationLag,
31  const Calendar& cal, // calendar in index may not be useful
32  const BusinessDayConvention& bdc,
33  const DayCounter& dc,
34  const Handle<ZeroInflationIndex>& zii,
35  Handle<YieldTermStructure> yts,
36  const std::vector<Rate>& cStrikes,
37  const std::vector<Rate>& fStrikes,
38  const std::vector<Period>& cfMaturities,
39  const Matrix& cPrice,
40  const Matrix& fPrice)
41  : InflationTermStructure(
42  0, cal, baseRate, observationLag, zii->frequency(), zii->interpolated(), dc),
43  zii_(zii), nominalTS_(std::move(yts)), cStrikes_(cStrikes), fStrikes_(fStrikes),
44  cfMaturities_(cfMaturities), cPrice_(cPrice), fPrice_(fPrice), nominal_(nominal), bdc_(bdc) {
45 
46  // does the index have a TS?
47  QL_REQUIRE(!zii_->zeroInflationTermStructure().empty(), "ZITS missing from index");
48  QL_REQUIRE(!nominalTS_.empty(), "nominal TS missing");
49 
50  // data consistency checking, enough data?
51  QL_REQUIRE(fStrikes_.size() > 1, "not enough floor strikes");
52  QL_REQUIRE(cStrikes_.size() > 1, "not enough cap strikes");
53  QL_REQUIRE(cfMaturities_.size() > 1, "not enough maturities");
54  QL_REQUIRE(fStrikes_.size() == fPrice.rows(),
55  "floor strikes vs floor price rows not equal");
56  QL_REQUIRE(cStrikes_.size() == cPrice.rows(),
57  "cap strikes vs cap price rows not equal");
58  QL_REQUIRE(cfMaturities_.size() == fPrice.columns(),
59  "maturities vs floor price columns not equal");
60  QL_REQUIRE(cfMaturities_.size() == cPrice.columns(),
61  "maturities vs cap price columns not equal");
62 
63  // data has correct properties (positive, monotonic)?
64  for(Size j = 0; j <cfMaturities_.size(); j++) {
65  QL_REQUIRE( cfMaturities[j] > Period(0,Days), "non-positive maturities");
66  if(j>0) {
67  QL_REQUIRE( cfMaturities[j] > cfMaturities[j-1],
68  "non-increasing maturities");
69  }
70  for(Size i = 0; i <fPrice_.rows(); i++) {
71  QL_REQUIRE( fPrice_[i][j] > 0.0,
72  "non-positive floor price: " << fPrice_[i][j] );
73  if(i>0) {
74  QL_REQUIRE( fPrice_[i][j] >= fPrice_[i-1][j],
75  "non-increasing floor prices");
76  }
77  }
78  for(Size i = 0; i <cPrice_.rows(); i++) {
79  QL_REQUIRE( cPrice_[i][j] > 0.0,
80  "non-positive cap price: " << cPrice_[i][j] );
81  if(i>0) {
82  QL_REQUIRE( cPrice_[i][j] <= cPrice_[i-1][j],
83  "non-decreasing cap prices: "
84  << cPrice_[i][j] << " then " << cPrice_[i-1][j]);
85  }
86  }
87  }
88 
89 
90  // Get the set of strikes, noting that repeats, overlaps are
91  // expected between caps and floors but that no overlap in the
92  // output is allowed so no repeats or overlaps are used
93  cfStrikes_ = std::vector<Rate>();
94  for(Size i = 0; i <fStrikes_.size(); i++)
95  cfStrikes_.push_back( fStrikes[i] );
96  Real eps = 0.0000001;
97  Rate maxFstrike = fStrikes_.back();
98  for(Size i = 0; i < cStrikes_.size(); i++) {
99  Rate k = cStrikes[i];
100  if (k > maxFstrike + eps) cfStrikes_.push_back(k);
101  }
102 
103  // final consistency checking
104  QL_REQUIRE(cfStrikes_.size() > 2, "overall not enough strikes");
105  for (Size i = 1; i < cfStrikes_.size(); i++)
106  QL_REQUIRE( cfStrikes_[i] > cfStrikes_[i-1],
107  "cfStrikes not increasing");
108  }
