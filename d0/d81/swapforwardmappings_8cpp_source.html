44  // compute derivative of swap-rate to underlying forward rate
45  Real SwapForwardMappings::swapDerivative(const CurveState& cs,
46  Size startIndex,
47  Size endIndex,
48  Size forwardIndex)
49  {
50  if (forwardIndex < startIndex)
51  return 0.0;
52  if (forwardIndex >= endIndex)
53  return 0.0;
54 
55  Real numerator = cs.discountRatio(startIndex, endIndex)-1;
56  Real swapAnnuity = annuity(cs, startIndex, endIndex,endIndex);
57 
58  Real ratio = cs.rateTaus()[forwardIndex] /
59  (1 + cs.rateTaus()[forwardIndex] * cs.forwardRate(forwardIndex));
60 
61  Real part1 = ratio*(numerator+1)/swapAnnuity;
62  Real part2 = numerator/(swapAnnuity*swapAnnuity);
63 
64  if (forwardIndex >=1)
65  part2 *= ratio* annuity(cs, startIndex, forwardIndex, endIndex);
66  else
67  part2 = 0.0;
68 
69  return part1-part2;
70  }

72  Disposable<Matrix>
73  SwapForwardMappings::coterminalSwapForwardJacobian(const CurveState& cs)
74  {
75  Size n = cs.numberOfRates();
76  const std::vector<Rate>& f = cs.forwardRates();
77  const std::vector<Time>& tau = cs.rateTaus();
78 
79  // coterminal floating leg values
80  std::vector<Real> a(n);
81  for (Size k=0; k<n; ++k)
82  a[k] = cs.discountRatio(k,n)-1.0;
83  //p[k]-p[n];
84 
85  Matrix jacobian = Matrix(n, n, 0.0);
86  for (Size i=0; i<n; ++i) { // i = swap rate index
87  for (Size j=i; j<n; ++j) { // j = forward rate index
88  Real bi = cs.coterminalSwapAnnuity(n,i);
89  Real bj = cs.coterminalSwapAnnuity(n,j);
90  jacobian[i][j] =
91  // p[j+1]*tau[j]/b[i] +
92  tau[j]/cs.coterminalSwapAnnuity(j+1,i) +
93  // tau[j]/(1.0+f[j]*tau[j]) *
94  tau[j]/(1.0+f[j]*tau[j]) *
95  // (-a[j]*b[i]+a[i]*b[j])/(b[i]*b[i]);
96  (-a[j]*bi+a[i]*bj)/(bi*bi);
97 
98  }
99  }
100  return jacobian;
101  }

104  SwapForwardMappings::coterminalSwapZedMatrix(const CurveState& cs,
105  const Spread displacement) {
106  Size n = cs.numberOfRates();
107  Matrix zMatrix = coterminalSwapForwardJacobian(cs);
108  const std::vector<Rate>& f = cs.forwardRates();
109  const std::vector<Rate>& sr = cs.coterminalSwapRates();

182  SwapForwardMappings::swaptionImpliedVolatility(const MarketModel& volStructure,
183  Size startIndex,
184  Size endIndex)
185  {
186  QL_REQUIRE(startIndex < endIndex, "start index must be before end index in swaptionImpliedVolatility");
187 
188  LMMCurveState cs(volStructure.evolution().rateTimes());
189  cs.setOnForwardRates(volStructure.initialRates());
190  Real displacement = volStructure.displacements()[0];
191 
192  Matrix cmsZed(cmSwapZedMatrix(cs, endIndex-startIndex,displacement));
193 
194  Real variance=0.0;
195 
196  Size index=0;
197 
198  const EvolutionDescription& evolution(volStructure.evolution());
199  Size factors = volStructure.numberOfFactors();
200 
201  while (index < evolution.numberOfSteps() && startIndex >= evolution.firstAliveRate()[index] )
202  {
203  const Matrix& thisPseudo = volStructure.pseudoRoot(index);
204  Real thisVariance =0.0;
205 
206  for (Size f=0; f < factors; ++f)
207  {
208  Real sum=0.0;
209 
210  for (Size j=startIndex; j < endIndex;++j)
211  {
212  sum += cmsZed[startIndex][j]*thisPseudo[j][f];
213 
214  }
215  thisVariance += sum*sum;
216 
217  }
218  variance += thisVariance;
219  ++index;
220 
221  }
222 
223  Real expiry = evolution.rateTimes()[startIndex];
224  return std::sqrt(variance/expiry);
225 
226  }
