60  Disposable<Matrix> StochasticProcessArray::diffusion(
61  Time t, const Array& x) const {
62  Matrix tmp = sqrtCorrelation_;
63  for (Size i=0; i<size(); ++i) {
64  Real sigma = processes_[i]->diffusion(t, x[i]);
65  std::transform(tmp.row_begin(i), tmp.row_end(i),
66  tmp.row_begin(i),
67  multiply_by<Real>(sigma));
68  }
