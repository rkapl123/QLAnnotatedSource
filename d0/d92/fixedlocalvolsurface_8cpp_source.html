30 Date time2Date(const Date referenceDate, const DayCounter& dc, Time t) {
31 t-=1e4*QL_EPSILON; // add a small buffer for rounding errors
32 Date d(referenceDate);
33 while(dc.yearFraction(referenceDate, d+=Period(1, Years)) < t);
34 d-=Period(1, Years);
35 while(dc.yearFraction(referenceDate, d+=Period(1, Months)) < t);
36 d-=Period(1, Months);
37 while(dc.yearFraction(referenceDate, d++) < t);
38 return d;
39 }
40 }

108 void FixedLocalVolSurface::checkSurface() {
109 QL_REQUIRE(times_.size()==localVolMatrix_->columns(),
110 "mismatch between date vector and vol matrix colums");
111 for (const auto& strike : strikes_) {
112 QL_REQUIRE(strike->size() == localVolMatrix_->rows(),
113 "mismatch between money-strike vector and "
114 "vol matrix rows");
115 }
116
117 for (Size j=1; j<times_.size(); j++) {
118 QL_REQUIRE(times_[j]>times_[j-1],
119 "dates must be sorted unique!");
120 }
121
122 for (const auto& strike : strikes_)
123 for (Size j = 1; j < strike->size(); j++) {
124 QL_REQUIRE((*strike)[j] >= (*strike)[j - 1], "strikes must be sorted");
125 }
126 }

141 Volatility FixedLocalVolSurface::localVolImpl(Time t, Real strike) const {
142 t = std::min(times_.back(), std::max(t, times_.front()));
143
144 const Size idx = std::distance(times_.begin(),
145 std::lower_bound(times_.begin(), times_.end(), t));
146
147 if (close_enough(t, times_[idx])) {
148 if (strikes_[idx]->front() < strikes_[idx]->back())
149 return localVolInterpol_[idx](strike, true);
150 else
151 return (*localVolMatrix_)[localVolMatrix_->rows()/2][idx];
152 }
153 else {
154 Real earlierStrike = strike, laterStrike = strike;
155 if (lowerExtrapolation_ == ConstantExtrapolation) {
156 if (strike < strikes_[idx-1]->front())
157 earlierStrike = strikes_[idx-1]->front();
158 if (strike < strikes_[idx]->front())
159 laterStrike = strikes_[idx]->front();
160 }
161
162 if (upperExtrapolation_ == ConstantExtrapolation) {
163 if (strike > strikes_[idx-1]->back())
164 earlierStrike = strikes_[idx-1]->back();
165 if (strike > strikes_[idx]->back())
166 laterStrike = strikes_[idx]->back();
167 }
168
169 const Real earlyVol =
170 (strikes_[idx-1]->front() < strikes_[idx-1]->back())
171 ? localVolInterpol_[idx-1](earlierStrike, true)
172 : (*localVolMatrix_)[localVolMatrix_->rows()/2][idx-1];
173 const Real laterVol = localVolInterpol_[idx](laterStrike, true);
174
175 return earlyVol
176 + (laterVol-earlyVol)/(times_[idx]-times_[idx-1])
177 *(t-times_[idx-1]);
178 }
179 }
