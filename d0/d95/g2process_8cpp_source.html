48  Disposable<Matrix> G2Process::diffusion(Time, const Array&) const {
49  /* the correlation matrix is
50  | 1 rho |
51  | rho 1 |
52  whose square root (which is used here) is
53  | 1 0 |
54  | rho sqrt(1-rho^2) |
55  */
56  Matrix tmp(2,2);
57  Real sigma1 = sigma_;
58  Real sigma2 = eta_;
59  tmp[0][0] = sigma1; tmp[0][1] = 0.0;
60  tmp[1][0] = rho_*sigma1; tmp[1][1] = std::sqrt(1.0-rho_*rho_)*sigma2;
61  return tmp;
62  }

72  Disposable<Matrix> G2Process::stdDeviation(Time t0, const Array& x0,
73  Time dt) const {
74  /* the correlation matrix is
75  | 1 rho |
76  | rho 1 |
77  whose square root (which is used here) is
78  | 1 0 |
79  | rho sqrt(1-rho^2) |
80  */
81  Matrix tmp(2,2);
82  Real sigma1 = xProcess_->stdDeviation(t0, x0[0], dt);
83  Real sigma2 = yProcess_->stdDeviation(t0, x0[1], dt);
84  Real expa = std::exp(-a_*dt), expb = std::exp(-b_*dt);
85  Real H = (rho_*sigma_*eta_)/(a_+b_)*(1-expa*expb);
86  Real den =
87  (0.5*sigma_*eta_)*std::sqrt((1-expa*expa)*(1-expb*expb)/(a_*b_));
88  Real newRho = H/den;
89  tmp[0][0] = sigma1;
90  tmp[0][1] = 0.0;
91  tmp[1][0] = newRho*sigma2;
92  tmp[1][1] = std::sqrt(1.0-newRho*newRho)*sigma2;
93  return tmp;
94  }

197  Real G2ForwardProcess::xForwardDrift(Time t, Time T) const {
198  Real expatT = std::exp(-a_*(T-t));
199  Real expbtT = std::exp(-b_*(T-t));
200 
201  return -(sigma_*sigma_/a_) * (1-expatT)
202  - (rho_*sigma_*eta_/b_) * (1-expbtT);
203  }
204 
205  Real G2ForwardProcess::yForwardDrift(Time t, Time T) const {
206  Real expatT = std::exp(-a_*(T-t));
207  Real expbtT = std::exp(-b_*(T-t));
208 
209  return -(eta_*eta_/b_) * (1-expbtT)
210  - (rho_*sigma_*eta_/a_) * (1-expatT);
211  }

213  Real G2ForwardProcess::Mx_T(Real s, Real t, Real T) const {
214  Real M;
215  M = ( (sigma_*sigma_)/(a_*a_) + (rho_*sigma_*eta_)/(a_*b_) )
216  * (1-std::exp(-a_*(t-s)));
217  M += -(sigma_*sigma_)/(2*a_*a_) *
218  (std::exp(-a_*(T-t))-std::exp(-a_*(T+t-2*s)));
219  M += -(rho_*sigma_*eta_)/(b_*(a_+b_))
220  * (std::exp(-b_*(T-t)) -std::exp(-b_*T-a_*t+(a_+b_)*s));
221  return M;
222  }
223 
224  Real G2ForwardProcess::My_T(Real s, Real t, Real T) const {
225  Real M;
226  M = ( (eta_*eta_)/(b_*b_) + (rho_*sigma_*eta_)/(a_*b_) )
227  * (1-std::exp(-b_*(t-s)));
228  M += -(eta_*eta_)/(2*b_*b_) *
229  (std::exp(-b_*(T-t))-std::exp(-b_*(T+t-2*s)));
230  M += -(rho_*sigma_*eta_)/(a_*(a_+b_))
231  * (std::exp(-a_*(T-t))-std::exp(-a_*T-b_*t+(a_+b_)*s));
232  return M;
233  }
