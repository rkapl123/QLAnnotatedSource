34  IrregularSwap::IrregularSwap(
35  Type type,
36  const Leg& fixLeg,
37  const Leg& floatLeg)
38  : Swap(2), type_(type) {
39 
40 
41  switch (type_) {
42  case Payer:
43  payer_[0] = -1.0;
44  payer_[1] = +1.0;
45  break;
46  case Receiver:
47  payer_[0] = +1.0;
48  payer_[1] = -1.0;
49  break;
50  default:
51  QL_FAIL("Unknown Irregular-swap type");
52  }
53 
54  //Fixed leg
55  legs_[0] = fixLeg;
56 
57  for (Leg::const_iterator i = legs_[0].begin(); i < legs_[0].end(); ++i)
58  registerWith(*i);
59 
60  //Floating Leg
61  legs_[1] = floatLeg;
62 
63  for (Leg::const_iterator i = legs_[1].begin(); i < legs_[1].end(); ++i)
64  registerWith(*i);
65 
66  }

69  void IrregularSwap::setupArguments(PricingEngine::arguments* args) const {
70 
71  //Debug: to be done
72 
73  Swap::setupArguments(args);
74 
75  IrregularSwap::arguments* arguments =
76  dynamic_cast<IrregularSwap::arguments*>(args);
77 
78  if (!arguments) // it's a swap engine...
79  return;
80 
81  arguments->type = type_;
82  //arguments->nominal = nominal_;
83 
84 
85  const Leg& fixedCoupons = fixedLeg();
86 
87  arguments->fixedResetDates = arguments->fixedPayDates = std::vector<Date>(fixedCoupons.size());
88  arguments->fixedNominals = arguments->fixedCoupons = std::vector<Real>(fixedCoupons.size());
89 
90 
91  for (Size i=0; i<fixedCoupons.size(); ++i) {
92  boost::shared_ptr<FixedRateCoupon> coupon =
93  boost::dynamic_pointer_cast<FixedRateCoupon>(fixedCoupons[i]);
94 
95  arguments->fixedPayDates[i] = coupon->date();
96  arguments->fixedResetDates[i] = coupon->accrualStartDate();
97  arguments->fixedCoupons[i] = coupon->amount();
98  arguments->fixedNominals[i] = coupon->nominal();
99  }
100 
101 
102  const Leg& floatingCoupons = floatingLeg();
103 
104  arguments->floatingResetDates = arguments->floatingPayDates
105  = arguments->floatingFixingDates = std::vector<Date>(floatingCoupons.size());
106  arguments->floatingAccrualTimes = std::vector<Time>(floatingCoupons.size());
107  arguments->floatingSpreads = std::vector<Spread>(floatingCoupons.size());
108  arguments->floatingNominals = arguments->floatingCoupons = std::vector<Real>(floatingCoupons.size());
109 
110  for (Size i=0; i<floatingCoupons.size(); ++i) {
111  boost::shared_ptr<IborCoupon> coupon =
112  boost::dynamic_pointer_cast<IborCoupon>(floatingCoupons[i]);
113 
114  arguments->floatingResetDates[i] = coupon->accrualStartDate();
115  arguments->floatingPayDates[i] = coupon->date();
116 
117  arguments->floatingFixingDates[i] = coupon->fixingDate();
118  arguments->floatingAccrualTimes[i] = coupon->accrualPeriod();
119  arguments->floatingSpreads[i] = coupon->spread();
120  arguments->floatingNominals[i] = coupon->nominal();
121 
122  try {
123  arguments->floatingCoupons[i] = coupon->amount();
124  } catch (Error&) {
125  arguments->floatingCoupons[i] = Null<Real>();
126  }
127  }
128 
129  }

175  void IrregularSwap::fetchResults(const PricingEngine::results* r) const {
176  Swap::fetchResults(r);
177 
178  const IrregularSwap::results* results =
179  dynamic_cast<const IrregularSwap::results*>(r);
180  if (results) { // might be a swap engine, so no error is thrown
181  fairRate_ = results->fairRate;
182  fairSpread_ = results->fairSpread;
183  } else {
184  fairRate_ = Null<Rate>();
185  fairSpread_ = Null<Spread>();
186  }
187 
188  if (fairRate_ == Null<Rate>()) {
189  // calculate it from other results
190  if (legBPS_[0] != Null<Real>())
191  fairRate_ = 0.0; // Debug: legs_[0]->fixedRate_ - NPV_/(legBPS_[0]/basisPoint);
192  }
193  if (fairSpread_ == Null<Spread>()) {
194  // ditto
195  if (legBPS_[1] != Null<Real>())
196  fairSpread_ = 0.0; //DEBUG: spread_ - NPV_/(legBPS_[1]/basisPoint);
197  }
198  }

200  void IrregularSwap::arguments::validate() const {
201 
202  Swap::arguments::validate();
203 
204  QL_REQUIRE(fixedResetDates.size() == fixedPayDates.size(),
205  "number of fixed start dates different from "

208  "number of fixed payment dates different from "

211  "number of floating start dates different from "

214  "number of floating fixing dates different from "
215  "number of floating payment dates");
216  QL_REQUIRE(floatingAccrualTimes.size() == floatingPayDates.size(),
217  "number of floating accrual Times different from "
218  "number of floating payment dates");
219  QL_REQUIRE(floatingSpreads.size() == floatingPayDates.size(),
220  "number of floating spreads different from "
221  "number of floating payment dates");
222  QL_REQUIRE(floatingPayDates.size() == floatingCoupons.size(),
223  "number of floating payment dates different from "
