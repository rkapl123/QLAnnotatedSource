26  void Gaussian1dSwaptionEngine::calculate() const {
27 
28  QL_REQUIRE(arguments_.settlementType == Settlement::Physical,
29  "cash-settled swaptions not yet implemented ...");
30 
31  Date settlement = model_->termStructure()->referenceDate();
32 
33  if (arguments_.exercise->dates().back() <=
34  settlement) { // swaption is expired, possibly generated swap is not
35  // valued
36  results_.value = 0.0;
37  return;
38  }
39 
40  int idx = static_cast<int>(arguments_.exercise->dates().size()) - 1;
41  int minIdxAlive = static_cast<int>(
42  std::upper_bound(arguments_.exercise->dates().begin(),
43  arguments_.exercise->dates().end(), settlement) -
44  arguments_.exercise->dates().begin());
45 
46  VanillaSwap swap = *arguments_.swap;
47  Option::Type type =
48  arguments_.type == VanillaSwap::Payer ? Option::Call : Option::Put;
49  Schedule fixedSchedule = swap.fixedSchedule();
50  Schedule floatSchedule = swap.floatingSchedule();
51 
52  Array npv0(2 * integrationPoints_ + 1, 0.0),
53  npv1(2 * integrationPoints_ + 1, 0.0);
54  Array z = model_->yGrid(stddevs_, integrationPoints_);
55  Array p(z.size(), 0.0);
56 
57  // for probability computation
58  std::vector<Array> npvp0, npvp1;
59  if (probabilities_ != None) {
60  for (Size i = 0; i < static_cast<Size>(idx - minIdxAlive + 2); ++i) {
61  Array npvTmp0(2 * integrationPoints_ + 1, 0.0);
62  Array npvTmp1(2 * integrationPoints_ + 1, 0.0);
63  npvp0.push_back(npvTmp0);
64  npvp1.push_back(npvTmp1);
65  }
66  }
67  // end probabkility computation
68 
69  Date expiry1 = Null<Date>(), expiry0;
70  Time expiry1Time = Null<Real>(), expiry0Time;
71 
72  do {
73 
74  if (idx == minIdxAlive - 1)
75  expiry0 = settlement;
76  else
77  expiry0 = arguments_.exercise->dates()[idx];
78 
79  expiry0Time = std::max(
80  model_->termStructure()->timeFromReference(expiry0), 0.0);
81 
82  Size j1 =
83  std::upper_bound(fixedSchedule.dates().begin(),
84  fixedSchedule.dates().end(), expiry0 - 1) -
85  fixedSchedule.dates().begin();
86  Size k1 =
87  std::upper_bound(floatSchedule.dates().begin(),
88  floatSchedule.dates().end(), expiry0 - 1) -
89  floatSchedule.dates().begin();
90 
91  // a lazy object is not thread safe, neither is the caching
92  // in gsrprocess. therefore we trigger computations here such
93  // that neither lazy object recalculation nor write access
94  // during caching occurs in the parallized loop below.
95  // this is known to work for the gsr and markov functional
96  // model implementations of Gaussian1dModel
97 #ifdef _OPENMP
98  if (expiry1Time != Null<Real>())
99  model_->yGrid(stddevs_, integrationPoints_, expiry1Time,
100  expiry0Time, 0.0);
101  if (expiry0 > settlement) {
102  for (Size l = k1; l < arguments_.floatingCoupons.size(); l++) {
103  model_->forwardRate(arguments_.floatingFixingDates[l],
104  expiry0, 0.0,
105  arguments_.swap->iborIndex());
106  model_->zerobond(arguments_.floatingPayDates[l], expiry0,
107  0.0, discountCurve_);
108  }
109  for (Size l = j1; l < arguments_.fixedCoupons.size(); l++) {
110  model_->zerobond(arguments_.fixedPayDates[l], expiry0, 0.0,
111  discountCurve_);
112  }
113  model_->numeraire(expiry0Time, 0.0, discountCurve_);
114  }
115 #endif
116 
117 #pragma omp parallel for default(shared) firstprivate(p) if(expiry0>settlement)
118  for (long k = 0; k < (expiry0 > settlement ? (long)npv0.size() : 1);
119  k++) {
120 
121  Real price = 0.0;
122  if (expiry1Time != Null<Real>()) {
123  Array yg = model_->yGrid(stddevs_, integrationPoints_,
124  expiry1Time, expiry0Time,
125  expiry0 > settlement ? z[k] : 0.0);
126  CubicInterpolation payoff0(
127  z.begin(), z.end(), npv1.begin(),
128  CubicInterpolation::Spline, true,
129  CubicInterpolation::Lagrange, 0.0,
130  CubicInterpolation::Lagrange, 0.0);
131  for (Size i = 0; i < yg.size(); i++) {
132  p[i] = payoff0(yg[i], true);
133  }
134  CubicInterpolation payoff1(
135  z.begin(), z.end(), p.begin(),
136  CubicInterpolation::Spline, true,
137  CubicInterpolation::Lagrange, 0.0,
138  CubicInterpolation::Lagrange, 0.0);
139  for (Size i = 0; i < z.size() - 1; i++) {
140  price += model_->gaussianShiftedPolynomialIntegral(
141  0.0, payoff1.cCoefficients()[i],
142  payoff1.bCoefficients()[i],
143  payoff1.aCoefficients()[i], p[i], z[i], z[i],
144  z[i + 1]);
145  }
146  if (extrapolatePayoff_) {
147  if (flatPayoffExtrapolation_) {
148  price += model_->gaussianShiftedPolynomialIntegral(
149  0.0, 0.0, 0.0, 0.0, p[z.size() - 2],
150  z[z.size() - 2], z[z.size() - 1], 100.0);
151  price += model_->gaussianShiftedPolynomialIntegral(
152  0.0, 0.0, 0.0, 0.0, p[0], z[0], -100.0, z[0]);
153  } else {
154  if (type == Option::Call)
155  price +=
156  model_->gaussianShiftedPolynomialIntegral(
157  0.0,
158  payoff1.cCoefficients()[z.size() - 2],
159  payoff1.bCoefficients()[z.size() - 2],
160  payoff1.aCoefficients()[z.size() - 2],
161  p[z.size() - 2], z[z.size() - 2],
162  z[z.size() - 1], 100.0);
163  if (type == Option::Put)
164  price +=
165  model_->gaussianShiftedPolynomialIntegral(
166  0.0, payoff1.cCoefficients()[0],
167  payoff1.bCoefficients()[0],
168  payoff1.aCoefficients()[0], p[0], z[0],
169  -100.0, z[0]);
170  }
171  }
172  }
173 
174  npv0[k] = price;
175 
176  // for probability computation
177  if (probabilities_ != None) {
178  for (Size m = 0; m < npvp0.size(); m++) {
179  Real price = 0.0;
180  if (expiry1Time != Null<Real>()) {
181  Array yg = model_->yGrid(
182  stddevs_, integrationPoints_, expiry1Time,
183  expiry0Time, expiry0 > settlement ? z[k] : 0.0);
184  CubicInterpolation payoff0(
185  z.begin(), z.end(), npvp1[m].begin(),
186  CubicInterpolation::Spline, true,
187  CubicInterpolation::Lagrange, 0.0,
188  CubicInterpolation::Lagrange, 0.0);
189  for (Size i = 0; i < yg.size(); i++) {
190  p[i] = payoff0(yg[i], true);
191  }
192  CubicInterpolation payoff1(
193  z.begin(), z.end(), p.begin(),
194  CubicInterpolation::Spline, true,
195  CubicInterpolation::Lagrange, 0.0,
196  CubicInterpolation::Lagrange, 0.0);
197  for (Size i = 0; i < z.size() - 1; i++) {
198  price +=
199  model_->gaussianShiftedPolynomialIntegral(
200  0.0, payoff1.cCoefficients()[i],
201  payoff1.bCoefficients()[i],
202  payoff1.aCoefficients()[i], p[i], z[i],
203  z[i], z[i + 1]);
204  }
205  if (extrapolatePayoff_) {
206  if (flatPayoffExtrapolation_) {
207  price +=
208  model_
209  ->gaussianShiftedPolynomialIntegral(
210  0.0, 0.0, 0.0, 0.0,
211  p[z.size() - 2],
212  z[z.size() - 2],
213  z[z.size() - 1], 100.0);
214  price +=
215  model_
216  ->gaussianShiftedPolynomialIntegral(
217  0.0, 0.0, 0.0, 0.0, p[0],
218  z[0], -100.0, z[0]);
219  } else {
220  if (type == Option::Call)
221  price +=
222  model_
223  ->gaussianShiftedPolynomialIntegral(
224  0.0,
225  payoff1.cCoefficients()
226  [z.size() - 2],
227  payoff1.bCoefficients()
228  [z.size() - 2],
229  payoff1.aCoefficients()
230  [z.size() - 2],
231  p[z.size() - 2],
232  z[z.size() - 2],
233  z[z.size() - 1], 100.0);
234  if (type == Option::Put)
235  price +=
236  model_
237  ->gaussianShiftedPolynomialIntegral(
238  0.0,
239  payoff1
240  .cCoefficients()[0],
241  payoff1
242  .bCoefficients()[0],
243  payoff1
244  .aCoefficients()[0],
245  p[0], z[0], -100.0, z[0]);
246  }
247  }
248  }
249 
250  npvp0[m][k] = price;
251  }
252  }
253  // end probability computation
254 
255  if (expiry0 > settlement) {
256  Real floatingLegNpv = 0.0;
257  for (Size l = k1; l < arguments_.floatingCoupons.size();
258  l++) {
259  floatingLegNpv +=
260  arguments_.nominal *
261  arguments_.floatingAccrualTimes[l] *
262  (arguments_.floatingSpreads[l] +
263  model_->forwardRate(
264  arguments_.floatingFixingDates[l], expiry0,
265  z[k], arguments_.swap->iborIndex())) *
266  model_->zerobond(arguments_.floatingPayDates[l],
267  expiry0, z[k], discountCurve_);
268  }
269  Real fixedLegNpv = 0.0;
270  for (Size l = j1; l < arguments_.fixedCoupons.size(); l++) {
271  fixedLegNpv +=
272  arguments_.fixedCoupons[l] *
273  model_->zerobond(arguments_.fixedPayDates[l],
274  expiry0, z[k], discountCurve_);
275  }
276  Real exerciseValue =
277  (type == Option::Call ? 1.0 : -1.0) *
278  (floatingLegNpv - fixedLegNpv) /
279  model_->numeraire(expiry0Time, z[k], discountCurve_);
280 
281  // for probability computation
282  if (probabilities_ != None) {
283  if (idx == static_cast<int>(
284  arguments_.exercise->dates().size()) -
285  1) // if true we are at the latest date,
286  // so we init
287  // the no call probability
288  npvp0.back()[k] =
289  probabilities_ == Naive
290  ? 1.0
291  : 1.0 / (model_->zerobond(expiry0Time, 0.0,
292  0.0,
293  discountCurve_) *
294  model_->numeraire(expiry0, z[k],
295  discountCurve_));
296  if (exerciseValue >= npv0[k]) {
297  npvp0[idx - minIdxAlive][k] =
298  probabilities_ == Naive
299  ? 1.0
300  : 1.0 /
301  (model_->zerobond(expiry0Time, 0.0,
302  0.0,
303  discountCurve_) *
304  model_->numeraire(expiry0Time, z[k],
305  discountCurve_));
306  for (Size ii = idx - minIdxAlive + 1;
307  ii < npvp0.size(); ii++)
308  npvp0[ii][k] = 0.0;
309  }
310  }
311  // end probability computation
312 
313  npv0[k] = std::max(npv0[k], exerciseValue);
314  }
315  }
316 
317  npv1.swap(npv0);
318 
319  // for probability computation
320  if (probabilities_ != None) {
321  for (Size i = 0; i < npvp0.size(); i++)
322  npvp1[i].swap(npvp0[i]);
323  }
324  // end probability computation
325 
326  expiry1 = expiry0;
327  expiry1Time = expiry0Time;
328 
329  } while (--idx >= minIdxAlive - 1);
330 
331  results_.value = npv1[0] * model_->numeraire(0.0, 0.0, discountCurve_);
332 
333  // for probability computation
334  if (probabilities_ != None) {
335  std::vector<Real> prob(npvp0.size());
336  for (Size i = 0; i < npvp0.size(); i++) {
337  prob[i] = npvp1[i][0] *
338  (probabilities_ == Naive
339  ? 1.0
340  : model_->numeraire(0.0, 0.0, discountCurve_));
341  }
342  results_.additionalResults["probabilities"] = prob;
343  }
344  // end probability computation
345  }
346 }
