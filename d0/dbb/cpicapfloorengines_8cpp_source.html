44  void InterpolatingCPICapFloorEngine::calculate() const
45  {
46  Real npv = 0.0;
47 
48  // what is the difference between the observationLag of the surface
49  // and the observationLag of the cap/floor?
50  // \TODO next line will fail if units are different
51  Period lagDiff = arguments_.observationLag - priceSurf_->observationLag();
52  // next line will fail if units are different if Period() is not well written
53  QL_REQUIRE(lagDiff >= Period(0, Months), "InterpolatingCPICapFloorEngine: "
54  "lag difference must be non-negative: " << lagDiff);
55 
56  // we now need an effective maturity to use in the price surface because this uses
57  // maturity of calibration instruments as its time axis, N.B. this must also
58  // use the roll because the surface does
59  Date effectiveMaturity = arguments_.payDate - lagDiff;
60 
61 
62  // what interpolation do we use? Index / flat / linear
63  if (arguments_.observationInterpolation == CPI::AsIndex) {
64  // same as index means we can just use the price surface
65  // since this uses the index
66  if (arguments_.type == Option::Call) {
67  npv = priceSurf_->capPrice(effectiveMaturity, arguments_.strike);
68  } else {
69  npv = priceSurf_->floorPrice(effectiveMaturity, arguments_.strike);
70  }
71 
72 
73  } else {
74  std::pair<Date,Date> dd = inflationPeriod(effectiveMaturity, arguments_.infIndex->frequency());
75  Real priceStart = 0.0;
76 
77  if (arguments_.type == Option::Call) {
78  priceStart = priceSurf_->capPrice(dd.first, arguments_.strike);
79  } else {
80  priceStart = priceSurf_->floorPrice(dd.first, arguments_.strike);
81  }
82 
83  // if we use a flat index vs the interpolated one ...
84  if (arguments_.observationInterpolation == CPI::Flat) {
85  // then use the price for the first day in the period because the value cannot change after then
86  npv = priceStart;
87 
88  } else {
89  // linear interpolation will be very close
90  Real priceEnd = 0.0;
91  if (arguments_.type == Option::Call) {
92  priceEnd = priceSurf_->capPrice((dd.second+Period(1,Days)), arguments_.strike);
93  } else {
94  priceEnd = priceSurf_->floorPrice((dd.second+Period(1,Days)), arguments_.strike);
95  }
96 
97  npv = priceStart + (priceEnd - priceStart) * (effectiveMaturity - dd.first)
98  / ( (dd.second+Period(1,Days)) - dd.first); // can't get to next period'
99  }
100 
101  }
102  results_.value = npv;
103  }
