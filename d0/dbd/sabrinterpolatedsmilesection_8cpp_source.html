103  void SabrInterpolatedSmileSection::performCalculations() const {
104  forwardValue_ = forward_->value();
105  vols_.clear();
106  actualStrikes_.clear();
107  // we populate the volatilities, skipping the invalid ones
108  for (Size i=0; i<volHandles_.size(); ++i) {
109  if (volHandles_[i]->isValid()) {
110  if (hasFloatingStrikes_) {
111  actualStrikes_.push_back(forwardValue_ + strikes_[i]);
112  vols_.push_back(atmVolatility_->value() + volHandles_[i]->value());
113  } else {
114  actualStrikes_.push_back(strikes_[i]);
115  vols_.push_back(volHandles_[i]->value());
116  }
117  }
118  }
119  // we are recreating the sabrinterpolation object unconditionnaly to
120  // avoid iterator invalidation
121  createInterpolation();
122  sabrInterpolation_->update();
123  }
