78  void calculate() const {
79  Real spot = process_->x0();
80  QL_REQUIRE(spot >= 0.0, "negative or null underlying given");
81  QL_REQUIRE(!triggered(spot), "barrier touched");
82  McSimulation<SingleVariate,RNG,S>::calculate(requiredTolerance_,
83  requiredSamples_,
84  maxSamples_);
85  results_.value = this->mcModel_->sampleAccumulator().mean();
86  if (RNG::allowsErrorEstimate)
87  results_.errorEstimate =
88  this->mcModel_->sampleAccumulator().errorEstimate();
89  }
90  protected:
91  // McSimulation implementation
92  TimeGrid timeGrid() const;
93  boost::shared_ptr<path_generator_type> pathGenerator() const {
94  TimeGrid grid = timeGrid();
95  typename RNG::rsg_type gen =
96  RNG::make_sequence_generator(grid.size()-1,seed_);
97  return boost::shared_ptr<path_generator_type>(
98  new path_generator_type(process_,
99  grid, gen, brownianBridge_));
100  }
101  boost::shared_ptr<path_pricer_type> pathPricer() const;

104  Size timeSteps_, timeStepsPerYear_;
105  Size requiredSamples_, maxSamples_;
106  Real requiredTolerance_;
107  bool isBiased_;

218  template <class RNG, class S>
219  inline TimeGrid MCBarrierEngine<RNG,S>::timeGrid() const {
220 
221  Time residualTime = process_->time(arguments_.exercise->lastDate());
222  if (timeSteps_ != Null<Size>()) {
223  return TimeGrid(residualTime, timeSteps_);
224  } else if (timeStepsPerYear_ != Null<Size>()) {
225  Size steps = static_cast<Size>(timeStepsPerYear_*residualTime);
226  return TimeGrid(residualTime, std::max<Size>(steps, 1));
227  } else {
228  QL_FAIL("time steps not specified");
229  }
230  }

233  template <class RNG, class S>
234  inline
235  boost::shared_ptr<typename MCBarrierEngine<RNG,S>::path_pricer_type>
236  MCBarrierEngine<RNG,S>::pathPricer() const {
237  boost::shared_ptr<PlainVanillaPayoff> payoff =
238  boost::dynamic_pointer_cast<PlainVanillaPayoff>(arguments_.payoff);
239  QL_REQUIRE(payoff, "non-plain payoff given");
240 
241  TimeGrid grid = timeGrid();
242  std::vector<DiscountFactor> discounts(grid.size());
243  for (Size i=0; i<grid.size(); i++)
244  discounts[i] = process_->riskFreeRate()->discount(grid[i]);
245 
246  // do this with template parameters?
247  if (isBiased_) {
248  return boost::shared_ptr<
249  typename MCBarrierEngine<RNG,S>::path_pricer_type>(
250  new BiasedBarrierPathPricer(
251  arguments_.barrierType,
252  arguments_.barrier,
253  arguments_.rebate,
254  payoff->optionType(),
255  payoff->strike(),
256  discounts));
257  } else {
258  PseudoRandom::ursg_type sequenceGen(grid.size()-1,
259  PseudoRandom::urng_type(5));
260  return boost::shared_ptr<
261  typename MCBarrierEngine<RNG,S>::path_pricer_type>(
262  new BarrierPathPricer(
263  arguments_.barrierType,
264  arguments_.barrier,
265  arguments_.rebate,
266  payoff->optionType(),
267  payoff->strike(),
268  discounts,
269  process_,
270  sequenceGen));
271  }
272  }
