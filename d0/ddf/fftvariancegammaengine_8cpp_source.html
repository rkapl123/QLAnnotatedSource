59  std::complex<Real> FFTVarianceGammaEngine::complexFourierTransform(std::complex<Real> u) const
60  {
61  Real s = process_->x0();
62 
63  std::complex<Real> i1(0, 1);
64 
65  Real omega = std::log(1.0 - theta_ * nu_ - sigma_*sigma_ * nu_ / 2.0) / nu_;
66  std::complex<Real> phi = std::exp(i1 * u * (std::log(s) + omega * t_))
67  * std::pow(dividendDiscount_/ riskFreeDiscount_, i1 * u);
68  phi = phi * (std::pow((1.0 - i1 * theta_ * nu_ * u + sigma_*sigma_ * nu_ * u*u / 2.0), (-t_ / nu_)));
69 
70  return phi;
71  }
