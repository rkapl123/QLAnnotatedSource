39 void FFTVarianceGammaEngine::precalculateExpiry(Date d)
40 {
41 ext::shared_ptr<VarianceGammaProcess> process =
42 ext::dynamic_pointer_cast<VarianceGammaProcess>(process_);
43
44 dividendDiscount_ =
45 process->dividendYield()->discount(d);
46 riskFreeDiscount_ =
47 process->riskFreeRate()->discount(d);
48
49 DayCounter rfdc = process->riskFreeRate()->dayCounter();
50 t_ = rfdc.yearFraction(process->riskFreeRate()->referenceDate(), d);
51
52 sigma_ = process->sigma();
53 nu_ = process->nu();
54 theta_ = process->theta();
55 }

57 std::complex<Real> FFTVarianceGammaEngine::complexFourierTransform(std::complex<Real> u) const
58 {
59 Real s = process_->x0();
60
61 std::complex<Real> i1(0, 1);
62
63 Real omega = std::log(1.0 - theta_ * nu_ - sigma_*sigma_ * nu_ / 2.0) / nu_;
64 std::complex<Real> phi = std::exp(i1 * u * (std::log(s) + omega * t_))
65 * std::pow(dividendDiscount_/ riskFreeDiscount_, i1 * u);
66 phi = phi * (std::pow((1.0 - i1 * theta_ * nu_ * u + sigma_*sigma_ * nu_ * u*u / 2.0), (-t_ / nu_)));
67
68 return phi;
69 }
