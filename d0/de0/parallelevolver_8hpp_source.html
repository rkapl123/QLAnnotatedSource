83 template <class Evolver>
84 class ParallelEvolver {
85 public:
86 // typedefs
87 typedef ParallelEvolverTraits<typename Evolver::traits> traits;
88 typedef typename traits::operator_type operator_type;
89 typedef typename traits::array_type array_type;
90 typedef typename traits::bc_set bc_set;
91 // constructors
92 ParallelEvolver(const operator_type& L,
93 const bc_set& bcs) {
94 evolvers_.reserve(L.size());
95 for (Size i=0; i < L.size(); i++) {
96 evolvers_.push_back(ext::shared_ptr<Evolver>(new
97 Evolver(L[i], bcs[i])));
98 }
99 }
100 void step(array_type& a,
101 Time t) {
102 //#pragma omp parallel for
103 for (Size i=0; i < evolvers_.size(); i++) {
104 evolvers_[i]->step(a[i], t);
105 }
106 }
107 void setStep(Time dt) {
108 for (Size i=0; i < evolvers_.size(); i++) {
109 evolvers_[i]->setStep(dt);
110 }
111 }
112 private:
113 std::vector<ext::shared_ptr<Evolver> > evolvers_;
114 };
