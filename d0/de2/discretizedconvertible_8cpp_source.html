27  DiscretizedConvertible::DiscretizedConvertible(
28  const ConvertibleBond::option::arguments& args,
29  const ext::shared_ptr<GeneralizedBlackScholesProcess>& process,
30  const TimeGrid& grid)
31  : arguments_(args), process_(process) {
32 
33  dividendValues_ = Array(arguments_.dividends.size(), 0.0);
34 
35  Date settlementDate = process_->riskFreeRate()->referenceDate();
36  for (Size i=0; i<arguments_.dividends.size(); i++) {
37  if (arguments_.dividends[i]->date() >= settlementDate) {
38  dividendValues_[i] =
39  arguments_.dividends[i]->amount() *
40  process_->riskFreeRate()->discount(
41  arguments_.dividends[i]->date());
42  }
43  }
44 
45  DayCounter dayCounter = process_->riskFreeRate()->dayCounter();
46  Date bondSettlement = arguments_.settlementDate;
47 
48  stoppingTimes_.resize(arguments_.exercise->dates().size());
49  for (Size i=0; i<stoppingTimes_.size(); ++i)
50  stoppingTimes_[i] =
51  dayCounter.yearFraction(bondSettlement,
52  arguments_.exercise->date(i));
53 
54  callabilityTimes_.resize(arguments_.callabilityDates.size());
55  for (Size i=0; i<callabilityTimes_.size(); ++i)
56  callabilityTimes_[i] =
57  dayCounter.yearFraction(bondSettlement,
58  arguments_.callabilityDates[i]);
59 
60  couponTimes_.resize(arguments_.couponDates.size());
61  for (Size i=0; i<couponTimes_.size(); ++i)
62  couponTimes_[i] =
63  dayCounter.yearFraction(bondSettlement,
64  arguments_.couponDates[i]);
65 
66  dividendTimes_.resize(arguments_.dividendDates.size());
67  for (Size i=0; i<dividendTimes_.size(); ++i)
68  dividendTimes_[i] =
69  dayCounter.yearFraction(bondSettlement,
70  arguments_.dividendDates[i]);
71 
72  if (!grid.empty()) {
73  // adjust times to grid
74  for (Size i=0; i<stoppingTimes_.size(); i++)
75  stoppingTimes_[i] = grid.closestTime(stoppingTimes_[i]);
76  for (Size i=0; i<couponTimes_.size(); i++)
77  couponTimes_[i] = grid.closestTime(couponTimes_[i]);
78  for (Size i=0; i<callabilityTimes_.size(); i++)
79  callabilityTimes_[i] = grid.closestTime(callabilityTimes_[i]);
80  for (Size i=0; i<dividendTimes_.size(); i++)
81  dividendTimes_[i] = grid.closestTime(dividendTimes_[i]);
82  }
83  }

85  void DiscretizedConvertible::reset(Size size) {
86 
87  // Set to bond redemption values
88  values_ = Array(size, arguments_.redemption);
89 
90  // coupon amounts should be added when adjusting
91  // values_ = Array(size, arguments_.cashFlows.back()->amount());
92 
93  conversionProbability_ = Array(size, 0.0);
94  spreadAdjustedRate_ = Array(size, 0.0);
95 
96  DayCounter rfdc = process_->riskFreeRate()->dayCounter();
97 
98  // this takes care of convertibility and conversion probabilities
99  adjustValues();
100 
101  Real creditSpread = arguments_.creditSpread->value();
102 
103  Date exercise = arguments_.exercise->lastDate();
104 
105  Rate riskFreeRate =
106  process_->riskFreeRate()->zeroRate(exercise, rfdc,
107  Continuous, NoFrequency);
108 
109  // Calculate blended discount rate to be used on roll back.
110  for (Size j=0; j<values_.size(); j++) {
111  spreadAdjustedRate_[j] =
112  conversionProbability_[j] * riskFreeRate +
113  (1-conversionProbability_[j])*(riskFreeRate + creditSpread);
114  }
115  }

117  void DiscretizedConvertible::postAdjustValuesImpl() {
118 
119  bool convertible = false;
120  switch (arguments_.exercise->type()) {
121  case Exercise::American:
122  if (time() <= stoppingTimes_[1] && time() >= stoppingTimes_[0])
123  convertible = true;
124  break;
125  case Exercise::European:
126  if (isOnTime(stoppingTimes_[0]))
127  convertible = true;
128  break;
129  case Exercise::Bermudan:
130  for (Size i=0; i<stoppingTimes_.size(); ++i) {
131  if (isOnTime(stoppingTimes_[i]))
132  convertible = true;
133  }
134  break;
135  default:
136  QL_FAIL("invalid option type");
137  }
138 
139  for (Size i=0; i<callabilityTimes_.size(); i++) {
140  if (isOnTime(callabilityTimes_[i]))
141  applyCallability(i,convertible);
142  }
143 
144  for (Size i=0; i<couponTimes_.size(); i++) {
145  if (isOnTime(couponTimes_[i]))
146  addCoupon(i);
147  }
148 
149  if (convertible)
150  applyConvertibility();
151  }

153  void DiscretizedConvertible::applyConvertibility() {
154  Array grid = adjustedGrid();
155  for (Size j=0; j<values_.size(); j++) {
156  Real payoff = arguments_.conversionRatio*grid[j];
157  if (values_[j] <= payoff) {
158  values_[j] = payoff;
159  conversionProbability_[j] = 1.0;
160  }
161  }

164  void DiscretizedConvertible::applyCallability(Size i, bool convertible) {
165  Size j;
166  Array grid = adjustedGrid();
167  switch (arguments_.callabilityTypes[i]) {
168  case Callability::Call:
169  if (arguments_.callabilityTriggers[i] != Null<Real>()) {
170  Real conversionValue =
171  arguments_.redemption/arguments_.conversionRatio;
172  Real trigger =
173  conversionValue*arguments_.callabilityTriggers[i];
174  for (j=0; j<values_.size(); j++) {
175  // the callability is conditioned by the trigger...
176  if (grid[j] >= trigger) {
177  // ...and might trigger conversion
178  values_[j] =
179  std::min(std::max(
180  arguments_.callabilityPrices[i],
181  arguments_.conversionRatio*grid[j]),
182  values_[j]);
183  }
184  }
185  } else if (convertible) {
186  for (j=0; j<values_.size(); j++) {
187  // exercising the callability might trigger conversion
188  values_[j] =
189  std::min(std::max(arguments_.callabilityPrices[i],
190  arguments_.conversionRatio*grid[j]),
191  values_[j]);
192  }
193  } else {
194  for (j=0; j<values_.size(); j++) {
195  values_[j] = std::min(arguments_.callabilityPrices[i],
196  values_[j]);
197  }
198  }
199  break;
200  case Callability::Put:
201  for (j=0; j<values_.size(); j++) {
202  values_[j] = std::max(values_[j],
203  arguments_.callabilityPrices[i]);
204  }
205  break;
206  default:
207  QL_FAIL("unknown callability type");
208  }
209  }

215  Disposable<Array> DiscretizedConvertible::adjustedGrid() const {
216  Time t = time();
217  Array grid = method()->grid(t);
218  // add back all dividend amounts in the future
219  for (Size i=0; i<arguments_.dividends.size(); i++) {
220  Time dividendTime = dividendTimes_[i];
221  if (dividendTime >= t || close(dividendTime,t)) {
222  const ext::shared_ptr<Dividend>& d = arguments_.dividends[i];
223  DiscountFactor dividendDiscount =
224  process_->riskFreeRate()->discount(dividendTime) /
225  process_->riskFreeRate()->discount(t);
226  for (Size j=0; j<grid.size(); j++)
227  grid[j] += d->amount(grid[j])*dividendDiscount;
228  }
229  }
230  return grid;
231  }
