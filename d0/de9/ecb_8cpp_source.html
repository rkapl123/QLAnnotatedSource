77 Date ECB::date(const string& ecbCode,
78 const Date& refDate) {
79
80 QL_REQUIRE(isECBcode(ecbCode),
81 ecbCode << " is not a valid ECB code");
82
83 string code = to_upper_copy(ecbCode);
84 string monthString = code.substr(0, 3);
85 Month m;
86 if (monthString=="JAN") m = January;
87 else if (monthString=="FEB") m = February;
88 else if (monthString=="MAR") m = March;
89 else if (monthString=="APR") m = April;
90 else if (monthString=="MAY") m = May;
91 else if (monthString=="JUN") m = June;
92 else if (monthString=="JUL") m = July;
93 else if (monthString=="AUG") m = August;
94 else if (monthString=="SEP") m = September;
95 else if (monthString=="OCT") m = October;
96 else if (monthString=="NOV") m = November;
97 else if (monthString=="DEC") m = December;
98 else QL_FAIL("not an ECB month (and it should have been)");
99
100 Year y = std::stoi(code.substr(3, 2));
101 Date referenceDate = (refDate != Date() ?
102 refDate :
103 Date(Settings::instance().evaluationDate()));
104 Year referenceYear = (referenceDate.year() % 100);
105 y += referenceDate.year() - referenceYear;
106 if (y<Date::minDate().year())
107 return ECB::nextDate(Date::minDate());
108
109 return ECB::nextDate(Date(1, m, y) - 1);
110 }

173 Date ECB::nextDate(const Date& date) {
174 Date d = (date == Date() ?
175 Settings::instance().evaluationDate() :
176 date);
177
178 auto i = std::upper_bound(knownDates().begin(), knownDates().end(), d);

180 QL_REQUIRE(i!=knownDates().end(),
181 "ECB dates after " << *(--knownDates().end()) << " are unknown");

198 bool ECB::isECBcode(const std::string& ecbCode) {
199
200 if (ecbCode.length() != 5)
201 return false;
202
203 string code = to_upper_copy(ecbCode);
204
205 string str1("0123456789");
206 string::size_type loc = str1.find(code.substr(3, 1), 0);
207 if (loc == string::npos)
208 return false;
209 loc = str1.find(code.substr(4, 1), 0);
210 if (loc == string::npos)
211 return false;
212
213 string monthString = code.substr(0, 3);
214 if (monthString=="JAN") return true;
215 else if (monthString=="FEB") return true;
216 else if (monthString=="MAR") return true;
217 else if (monthString=="APR") return true;
218 else if (monthString=="MAY") return true;
219 else if (monthString=="JUN") return true;
220 else if (monthString=="JUL") return true;
221 else if (monthString=="AUG") return true;
222 else if (monthString=="SEP") return true;
223 else if (monthString=="OCT") return true;
224 else if (monthString=="NOV") return true;
225 else if (monthString=="DEC") return true;
226 else return false;
227 }

229 string ECB::nextCode(const std::string& ecbCode) {
230 QL_REQUIRE(isECBcode(ecbCode),
231 ecbCode << " is not a valid ECB code");
232
233 string code = to_upper_copy(ecbCode);
234 std::ostringstream result;
235
236 string monthString = code.substr(0, 3);
237 if (monthString=="JAN") result << "FEB" << code.substr(3, 2);
238 else if (monthString=="FEB") result << "MAR" << code.substr(3, 2);
239 else if (monthString=="MAR") result << "APR" << code.substr(3, 2);
240 else if (monthString=="APR") result << "MAY" << code.substr(3, 2);
241 else if (monthString=="MAY") result << "JUN" << code.substr(3, 2);
242 else if (monthString=="JUN") result << "JUL" << code.substr(3, 2);
243 else if (monthString=="JUL") result << "AUG" << code.substr(3, 2);
244 else if (monthString=="AUG") result << "SEP" << code.substr(3, 2);
245 else if (monthString=="SEP") result << "OCT" << code.substr(3, 2);
246 else if (monthString=="OCT") result << "NOV" << code.substr(3, 2);
247 else if (monthString=="NOV") result << "DEC" << code.substr(3, 2);
248 else if (monthString=="DEC") {
249 unsigned int y = (std::stoi(code.substr(3, 2)) + 1) % 100;
250 string padding;
251 if (y < 10)
252 padding = "0";
253
254 result << "JAN" << padding << y;
255 } else QL_FAIL("not an ECB month (and it should have been)");
256
257
258 #if defined(QL_EXTRA_SAFETY_CHECKS)
259 QL_ENSURE(isECBcode(result.str()),
260 "the result " << result.str() <<
261 " is an invalid ECB code");
262 #endif
263 return result.str();
264 }
