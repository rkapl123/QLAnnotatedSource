83  Date ECB::date(const string& ecbCode,
84  const Date& refDate) {
85 
86  QL_REQUIRE(isECBcode(ecbCode),
87  ecbCode << " is not a valid ECB code");
88 
89  string code = to_upper_copy(ecbCode);
90  string monthString = code.substr(0, 3);
91  Month m;
92  if (monthString=="JAN") m = January;
93  else if (monthString=="FEB") m = February;
94  else if (monthString=="MAR") m = March;
95  else if (monthString=="APR") m = April;
96  else if (monthString=="MAY") m = May;
97  else if (monthString=="JUN") m = June;
98  else if (monthString=="JUL") m = July;
99  else if (monthString=="AUG") m = August;
100  else if (monthString=="SEP") m = September;
101  else if (monthString=="OCT") m = October;
102  else if (monthString=="NOV") m = November;
103  else if (monthString=="DEC") m = December;
104  else QL_FAIL("not an ECB month (and it should have been)");
105 
106  // lexical_cast causes compilation errors with x64
107  //Year y = boost::lexical_cast<Year>(code.substr(3, 2));
108 
109  Year y = io::to_integer(code.substr(3, 2));
110  Date referenceDate = (refDate != Date() ?
111  refDate :
112  Date(Settings::instance().evaluationDate()));
113  Year referenceYear = (referenceDate.year() % 100);
114  y += referenceDate.year() - referenceYear;
115  if (y<Date::minDate().year())
116  return ECB::nextDate(Date::minDate());
117 
118  return ECB::nextDate(Date(1, m, y) - 1);
119  }

209  bool ECB::isECBcode(const std::string& ecbCode) {
210 
211  if (ecbCode.length() != 5)
212  return false;
213 
214  string code = to_upper_copy(ecbCode);
215 
216  string str1("0123456789");
217  string::size_type loc = str1.find(code.substr(3, 1), 0);
218  if (loc == string::npos)
219  return false;
220  loc = str1.find(code.substr(4, 1), 0);
221  if (loc == string::npos)
222  return false;
223 
224  string monthString = code.substr(0, 3);
225  if (monthString=="JAN") return true;
226  else if (monthString=="FEB") return true;
227  else if (monthString=="MAR") return true;
228  else if (monthString=="APR") return true;
