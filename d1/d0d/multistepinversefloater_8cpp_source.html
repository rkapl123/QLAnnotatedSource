50  bool MultiStepInverseFloater::nextTimeStep(
51  const CurveState& currentState,
52  std::vector<Size>& numberCashFlowsThisStep,
53  std::vector<std::vector<MarketModelMultiProduct::CashFlow> >&
54  genCashFlows)
55  {
56  Rate liborRate = currentState.forwardRate(currentIndex_);
57  Real inverseFloatingCoupon = std::max((fixedStrikes_[currentIndex_] - fixedMultipliers_[currentIndex_]*liborRate),0.0)*fixedAccruals_[currentIndex_] ;
58  Real floatingCoupon = (liborRate+floatingSpreads_[currentIndex_])*floatingAccruals_[currentIndex_];
59 
60  genCashFlows[0][0].timeIndex = currentIndex_;
61  genCashFlows[0][0].amount =multiplier_*(inverseFloatingCoupon - floatingCoupon);
62 
63  numberCashFlowsThisStep[0] = 1;
64  ++currentIndex_;
65 
66  return (currentIndex_ == lastIndex_);
67  }
