84  ext::shared_ptr<FdmStepConditionComposite>
85  FdmStepConditionComposite::vanillaComposite(
86  const DividendSchedule& cashFlow,
87  const ext::shared_ptr<Exercise>& exercise,
88  const ext::shared_ptr<FdmMesher>& mesher,
89  const ext::shared_ptr<FdmInnerValueCalculator>& calculator,
90  const Date& refDate,
91  const DayCounter& dayCounter) {
92 
93  std::list<std::vector<Time> > stoppingTimes;
94  std::list<ext::shared_ptr<StepCondition<Array> > > stepConditions;
95 
96  if(!cashFlow.empty()) {
97  ext::shared_ptr<FdmDividendHandler> dividendCondition(
98  new FdmDividendHandler(cashFlow, mesher,
99  refDate, dayCounter, 0));
100  stepConditions.push_back(dividendCondition);
101  stoppingTimes.push_back(dividendCondition->dividendTimes());
102  }
103 
104  QL_REQUIRE( exercise->type() == Exercise::American
105  || exercise->type() == Exercise::European
106  || exercise->type() == Exercise::Bermudan,
107  "exercise type is not supported");
108  if (exercise->type() == Exercise::American) {
109  stepConditions.push_back(ext::shared_ptr<StepCondition<Array> >(
110  new FdmAmericanStepCondition(mesher,calculator)));
111  }
112  else if (exercise->type() == Exercise::Bermudan) {
113  ext::shared_ptr<FdmBermudanStepCondition> bermudanCondition(
114  new FdmBermudanStepCondition(exercise->dates(),
115  refDate, dayCounter,
116  mesher, calculator));
117  stepConditions.push_back(bermudanCondition);
118  stoppingTimes.push_back(bermudanCondition->exerciseTimes());
119  }
120 
121  return ext::make_shared<FdmStepConditionComposite>(
122  stoppingTimes, stepConditions);
123 
124  }
