29  ZeroCouponInflationSwapHelper::ZeroCouponInflationSwapHelper(
30  const Handle<Quote>& quote,
31  const Period& swapObsLag, // <= index availability lag
32  const Date& maturity,
33  const Calendar& calendar, // index may have null calendar as valid on every day
34  BusinessDayConvention paymentConvention,
35  const DayCounter& dayCounter,
36  const boost::shared_ptr<ZeroInflationIndex>& zii)
37  : BootstrapHelper<ZeroInflationTermStructure>(quote),
38  swapObsLag_(swapObsLag), maturity_(maturity), calendar_(calendar),
39  paymentConvention_(paymentConvention), dayCounter_(dayCounter),
40  zii_(zii) {
41 
42  if (zii_->interpolated()) {
43  // if interpolated then simple
44  earliestDate_ = maturity_ - swapObsLag_;
45  latestDate_ = maturity_ - swapObsLag_;
46  } else {
47  // but if NOT interpolated then the value is valid
48  // for every day in an inflation period so you actually
49  // get an extended validity, however for curve building
50  // just put the first date because using that convention
51  // for the base date throughout
52  std::pair<Date,Date> limStart = inflationPeriod(maturity_ - swapObsLag_,
53  zii_->frequency());
54  earliestDate_ = limStart.first;
55  latestDate_ = limStart.first;
56  }
57 
58  // check that the observation lag of the swap
59  // is compatible with the availability lag of the index AND
60  // it's interpolation (assuming the start day is spot)
61  if (zii_->interpolated()) {
62  Period pShift(zii_->frequency());
63  QL_REQUIRE(swapObsLag_ - pShift > zii_->availabilityLag(),
64  "inconsistency between swap observation of index "
65  << swapObsLag_ <<
66  " index availability " << zii_->availabilityLag() <<
67  " index period " << pShift <<
68  " and index availability " << zii_->availabilityLag() <<
69  " need (obsLag-index period) > availLag");
70  }
71 
72  registerWith(Settings::instance().evaluationDate());
73  }

83  void ZeroCouponInflationSwapHelper::setTermStructure(
84  ZeroInflationTermStructure* z) {
85 
86  BootstrapHelper<ZeroInflationTermStructure>::setTermStructure(z);
87 
88  // set up a new ZCIIS
89  // but this one does NOT own its inflation term structure
90  const bool own = false;
91  Rate K = quote()->value();
92 
93  // The effect of the new inflation term structure is
94  // felt via the effect on the inflation index
95  Handle<ZeroInflationTermStructure> zits(
96  boost::shared_ptr<ZeroInflationTermStructure>(z, null_deleter()), own);
97 
98  boost::shared_ptr<ZeroInflationIndex> new_zii = zii_->clone(zits);
99 
100  Real nominal = 1000000.0; // has to be something but doesn't matter what
101  Date start = z->nominalTermStructure()->referenceDate();
102  zciis_.reset(new ZeroCouponInflationSwap(
103  ZeroCouponInflationSwap::Payer,
104  nominal, start, maturity_,
105  calendar_, paymentConvention_, dayCounter_, K, // fixed side & fixed rate
106  new_zii, swapObsLag_));
107  // Because very simple instrument only takes
108  // standard discounting swap engine.
109  zciis_->setPricingEngine(boost::shared_ptr<PricingEngine>(
110  new DiscountingSwapEngine(z->nominalTermStructure())));
111  }

169  void YearOnYearInflationSwapHelper::setTermStructure(
170  YoYInflationTermStructure* y) {
171 
172  BootstrapHelper<YoYInflationTermStructure>::setTermStructure(y);
173 
174  // set up a new YYIIS
175  // but this one does NOT own its inflation term structure
176  const bool own = false;
177 
178  // The effect of the new inflation term structure is
179  // felt via the effect on the inflation index
180  Handle<YoYInflationTermStructure> yyts(
181  boost::shared_ptr<YoYInflationTermStructure>(y, null_deleter()), own);
182 
183  boost::shared_ptr<YoYInflationIndex> new_yii = yii_->clone(yyts);
184 
185  // always works because tenor is always 1 year so
186  // no problem with different days-in-month
187  Date from = Settings::instance().evaluationDate();
188  Date to = maturity_;
189  Schedule fixedSchedule = MakeSchedule().from(from).to(to)
190  .withTenor(1*Years)
191  .withConvention(Unadjusted)
192  .withCalendar(calendar_)// fixed leg gets cal from sched
193  .backwards();
194  Schedule yoySchedule = fixedSchedule;
195  Spread spread = 0.0;
196  Rate fixedRate = quote()->value();
197 
198  Real nominal = 1000000.0; // has to be something but doesn't matter what
199  yyiis_.reset(new YearOnYearInflationSwap(YearOnYearInflationSwap::Payer,
200  nominal,
201  fixedSchedule,
202  fixedRate,
203  dayCounter_,
204  yoySchedule,
205  new_yii,
206  swapObsLag_,
207  spread,
208  dayCounter_,
209  calendar_, // inflation index does not have a calendar
210  paymentConvention_));
211 
212 
213  // Because very simple instrument only takes
214  // standard discounting swap engine.
215  yyiis_->setPricingEngine(boost::shared_ptr<PricingEngine>(
216  new DiscountingSwapEngine(y->nominalTermStructure())));
