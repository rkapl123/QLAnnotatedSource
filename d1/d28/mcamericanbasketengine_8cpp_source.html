28  AmericanBasketPathPricer::AmericanBasketPathPricer(
29  Size assetNumber,
30  const ext::shared_ptr<Payoff>& payoff,
31  Size polynomOrder,
32  LsmBasisSystem::PolynomType polynomType)
33  : assetNumber_ (assetNumber),
34  payoff_ (payoff),
35  scalingValue_(1.0),
36  v_ (LsmBasisSystem::multiPathBasisSystem(assetNumber_,
37  polynomOrder,
38  polynomType)) {
39  QL_REQUIRE( polynomType == LsmBasisSystem::Monomial
40  || polynomType == LsmBasisSystem::Laguerre
41  || polynomType == LsmBasisSystem::Hermite
42  || polynomType == LsmBasisSystem::Hyperbolic
43  || polynomType == LsmBasisSystem::Chebyshev2nd,
44  "insufficient polynom type");
45  using namespace ext::placeholders;
46  const ext::shared_ptr<BasketPayoff> basketPayoff
47  = ext::dynamic_pointer_cast<BasketPayoff>(payoff_);
48  QL_REQUIRE(basketPayoff, "payoff not a basket payoff");
49 
50  const ext::shared_ptr<StrikedTypePayoff> strikePayoff
51  = ext::dynamic_pointer_cast<StrikedTypePayoff>(basketPayoff->basePayoff());
52 
53  if (strikePayoff != 0) {
54  scalingValue_/=strikePayoff->strike();
55  }
56 
57  v_.push_back(ext::bind(&AmericanBasketPathPricer::payoff,
58  this, _1));
59  }

61  Array AmericanBasketPathPricer::state(const MultiPath& path,
62  Size t) const {
63  QL_REQUIRE(path.assetNumber() == assetNumber_, "invalid multipath");
64 
65  Array tmp(assetNumber_);
66  for (Size i=0; i<assetNumber_; ++i) {
67  tmp[i] = path[i][t]*scalingValue_;
68  }
69 
70  return tmp;
71  }

73  Real AmericanBasketPathPricer::payoff(const Array& state) const {
74  const ext::shared_ptr<BasketPayoff> basketPayoff
75  = ext::dynamic_pointer_cast<BasketPayoff>(payoff_);
76  QL_REQUIRE(basketPayoff, "payoff not a basket payoff");
77 
78  Real value = basketPayoff->accumulate(state);
79  return (*payoff_)(value/scalingValue_);
