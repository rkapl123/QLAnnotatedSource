45  Date::serial_type Thirty360::US_Impl::dayCount(const Date& d1,
46  const Date& d2) const {
47  Day dd1 = d1.dayOfMonth(), dd2 = d2.dayOfMonth();
48  Integer mm1 = d1.month(), mm2 = d2.month();
49  Year yy1 = d1.year(), yy2 = d2.year();
50 
51  if (dd2 == 31 && dd1 < 30) { dd2 = 1; mm2++; }
52 
53  return 360*(yy2-yy1) + 30*(mm2-mm1-1) +
54  std::max(Integer(0),30-dd1) + std::min(Integer(30),dd2);
55  }
