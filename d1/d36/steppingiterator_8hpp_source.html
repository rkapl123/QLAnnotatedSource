38  class step_iterator :
39  public boost::iterator_adaptor<step_iterator<Iterator>, Iterator> {
40  private:
41  typedef boost::iterator_adaptor<step_iterator<Iterator>, Iterator>
42  super_t;
43  // a Size would mess up integer division in distance_to
44  BigInteger step_;
45  public:
46  step_iterator() {}
47  explicit step_iterator(const Iterator& base, Size step)
48  : super_t(base), step_(static_cast<BigInteger>(step)) {}
49  template <class OtherIterator>
50  step_iterator(const step_iterator<OtherIterator>& i,
51  typename boost::enable_if_convertible
52  <OtherIterator,Iterator>::type* = 0)
53  : super_t(i.base()), step_(static_cast<BigInteger>(i.step())) {}
54  // inspector
55  Size step() const { return static_cast<Size>(this->step_); }
56  // iterator adapter interface
57  void increment() {
58  std::advance(this->base_reference(), step_);
59  }
60  void decrement() {
61  std::advance(this->base_reference(), -step_);
62  }
63  void advance(typename super_t::difference_type n) {
64  this->base_reference() += n*(this->step_);
65  }
66  typename super_t::difference_type
67  distance_to(const step_iterator& i) const {
68  return (i.base()-this->base())/(this->step_);
69  }
70  };
