40  class MonomialFct {
41  public:
42  explicit MonomialFct(Size order): order_(order) {}
43  inline Real operator()(const Real x) const {
44  Real ret = 1.0;

53  /* multiplies [Real -> Real] functors
54  to create [Array -> Real] functor */
55  class MultiDimFct {
56  public:
57  explicit MultiDimFct(VF_R b) : b_(std::move(b)) {
58  QL_REQUIRE(!b_.empty(), "zero size basis");
59  }
60  inline Real operator()(const Array& a) const {
61  #if defined(QL_EXTRA_SAFETY_CHECKS)
62  QL_REQUIRE(b_.size()==a.size(), "wrong argument size");
63  #endif
64  Real ret = b_[0](a[0]);
65  for(Size i=1; i<b_.size(); ++i)
66  ret *= b_[i](a[i]);
67  return ret;
68  }
69  private:
70  const VF_R b_;
71  };

73  // check size and order of tuples
74  void check_tuples(const VV& v, Size dim, Size order) {
75  for (const auto& i : v) {
76  QL_REQUIRE(dim == i.size(), "wrong tuple size");
77  QL_REQUIRE(order == std::accumulate(i.begin(), i.end(), 0UL), "wrong tuple order");
78  }

81  // build order N+1 tuples from order N tuples
82  VV next_order_tuples(const VV& v) {
83  const Size order = std::accumulate(v[0].begin(), v[0].end(), 0UL);
84  const Size dim = v[0].size();
85 
86  check_tuples(v, dim, order);
87 
88  // the set of unique tuples
89  std::set<std::vector<Size> > tuples;
90  std::vector<Size> x;
91  for(Size i=0; i<dim; ++i) {
92  // increase i-th value in every tuple by 1
93  for (const auto& j : v) {
94  x = j;
95  x[i] += 1;
96  tuples.insert(x);
97  }
98  }
99 
100  VV ret(tuples.begin(), tuples.end());
101  return ret;
102  }
103  }

107  VF_R LsmBasisSystem::pathBasisSystem(Size order, PolynomType polyType) {
108  VF_R ret(order+1);
109  for (Size i=0; i<=order; ++i) {
110  switch (polyType) {
111  case Monomial:
112  ret[i] = MonomialFct(i);
113  break;
114  case Laguerre:
115  {
116  GaussLaguerrePolynomial p;
117  ret[i] = [=](Real x){ return p.weightedValue(i, x); };
118  }
119  break;
120  case Hermite:
121  {
122  GaussHermitePolynomial p;
123  ret[i] = [=](Real x){ return p.weightedValue(i, x); };
124  }
125  break;
126  case Hyperbolic:
127  {
128  GaussHyperbolicPolynomial p;
129  ret[i] = [=](Real x){ return p.weightedValue(i, x); };
130  }
131  break;
132  case Legendre:
133  {
134  GaussLegendrePolynomial p;
135  ret[i] = [=](Real x){ return p.weightedValue(i, x); };
136  }
137  break;
138  case Chebyshev:
139  {
140  GaussChebyshevPolynomial p;
141  ret[i] = [=](Real x){ return p.weightedValue(i, x); };
142  }
143  break;
144  case Chebyshev2nd:
145  {
146  GaussChebyshev2ndPolynomial p;
147  ret[i] = [=](Real x){ return p.weightedValue(i, x); };
148  }
149  break;
150  default:
151  QL_FAIL("unknown regression type");
152  }
153  }
154  return ret;
155  }

157  VF_A LsmBasisSystem::multiPathBasisSystem(Size dim, Size order,
158  PolynomType polyType) {
159  QL_REQUIRE(dim>0, "zero dimension");
160  // get single factor basis
161  VF_R pathBasis = pathBasisSystem(order, polyType);
162  VF_A ret;
163  // 0-th order term
164  VF_R term(dim, pathBasis[0]);
165  ret.push_back(MultiDimFct(term));
166  // start with all 0 tuple
167  VV tuples(1, std::vector<Size>(dim));
168  // add multi-factor terms
169  for(Size i=1; i<=order; ++i) {
170  tuples = next_order_tuples(tuples);
171  // now we have all tuples of order i
172  // for each tuple add the corresponding term
173  for (auto& tuple : tuples) {
174  for(Size k=0; k<dim; ++k)
175  term[k] = pathBasis[tuple[k]];
176  ret.push_back(MultiDimFct(term));
177  }
178  }
179  return ret;
180  }
