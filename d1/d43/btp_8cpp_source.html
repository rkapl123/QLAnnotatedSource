91  RendistatoBasket::RendistatoBasket(
92  const std::vector<ext::shared_ptr<BTP> >& btps,
93  const std::vector<Real>& outstandings,
94  const std::vector<Handle<Quote> >& cleanPriceQuotes)
95  : btps_(btps), outstandings_(outstandings), quotes_(cleanPriceQuotes) {
96 
97  QL_REQUIRE(!btps_.empty(), "empty RendistatoCalculator Basket");
98  Size k = btps_.size();
99 
100  QL_REQUIRE(outstandings_.size()==k,
101  "mismatch between number of BTPs (" << k <<
102  ") and number of outstandings (" <<
103  outstandings_.size() << ")");
104  QL_REQUIRE(quotes_.size()==k,
105  "mismatch between number of BTPs (" << k <<
106  ") and number of clean prices quotes (" <<
107  quotes_.size() << ")");
108 
109  // require non-negative outstanding
110  for (Size i=0; i<k; ++i) {
111  QL_REQUIRE(outstandings[i]>=0,
112  "negative outstanding for " << io::ordinal(i) <<
113  " bond, maturity " << btps[i]->maturityDate());
114  // add check for prices ??
115  }
116 
117  // TODO: filter out expired bonds, zero outstanding bond, etc
118 
119  QL_REQUIRE(!btps_.empty(), "invalid bonds only in RendistatoCalculator Basket");
120  n_ = btps_.size();
121 
122  outstanding_ = 0.0;
123  for (Size i=0; i<n_; ++i)
124  outstanding_ += outstandings[i];
125 
126  weights_.resize(n_);
127  for (Size i=0; i<n_; ++i) {
128  weights_[i] = outstandings[i]/outstanding_;
129  registerWith(quotes_[i]);
130  }
131 
132  }

135  RendistatoCalculator::RendistatoCalculator(
136  const ext::shared_ptr<RendistatoBasket>& basket,
137  const ext::shared_ptr<Euribor>& euriborIndex,
138  const Handle<YieldTermStructure>& discountCurve)
139  : basket_(basket),
140  euriborIndex_(euriborIndex), discountCurve_(discountCurve),
141  yields_(basket_->size(), 0.05), durations_(basket_->size()),
142  nSwaps_(15), // TODO: generalize number of swaps and their lengths
143  swaps_(nSwaps_),
144  swapLengths_(nSwaps_), swapBondDurations_(nSwaps_, Null<Time>()),
145  swapBondYields_(nSwaps_, 0.05), swapRates_(nSwaps_, Null<Rate>())
146  {
147  registerWith(basket_);
148  registerWith(euriborIndex_);
149  registerWith(discountCurve_);
150 
151  Rate dummyRate = 0.05;
152  for (Size i=0; i<nSwaps_; ++i) {
153  swapLengths_[i] = static_cast<Real>(i+1);
154  swaps_[i] = MakeVanillaSwap(
155  swapLengths_[i]*Years, euriborIndex_, dummyRate, 1*Days)
156  .withDiscountingTermStructure(discountCurve_);
157  }
158  }

160  void RendistatoCalculator::performCalculations() const {
161 
162  const std::vector<ext::shared_ptr<BTP> >& btps = basket_->btps();
163  const std::vector<Handle<Quote> >& quotes = basket_->cleanPriceQuotes();
164  Date bondSettlementDate = btps[0]->settlementDate();
165  for (Size i=0; i<basket_->size(); ++i) {
166  yields_[i] = BondFunctions::yield(
167  *btps[i], quotes[i]->value(),
168  ActualActual(ActualActual::ISMA), Compounded, Annual,
169  bondSettlementDate,
170  // accuracy, maxIterations, guess
171  1.0e-10, 100, yields_[i]);
172  durations_[i] = BondFunctions::duration(
173  *btps[i], yields_[i],
174  ActualActual(ActualActual::ISMA), Compounded, Annual,
175  Duration::Modified, bondSettlementDate);
176  }
177  duration_ = std::inner_product(basket_->weights().begin(),
178  basket_->weights().end(),
179  durations_.begin(), 0.0);
180 
181  Natural settlDays = 2;
182  DayCounter fixedDayCount = swaps_[0]->fixedDayCount();
183  equivalentSwapIndex_ = nSwaps_-1;
184  swapRates_[0]= swaps_[0]->fairRate();
185  FixedRateBond swapBond(settlDays,
186  100.0, // faceAmount
187  swaps_[0]->fixedSchedule(),
188  std::vector<Rate>(1, swapRates_[0]),
189  fixedDayCount,
190  Following, // paymentConvention
191  100.0); // redemption
192  swapBondYields_[0] = BondFunctions::yield(swapBond,
193  100.0, // floating leg NPV including end payment
194  ActualActual(ActualActual::ISMA), Compounded, Annual,
195  bondSettlementDate,
196  // accuracy, maxIterations, guess
197  1.0e-10, 100, swapBondYields_[0]);
198  swapBondDurations_[0] = BondFunctions::duration(
199  swapBond, swapBondYields_[0],
200  ActualActual(ActualActual::ISMA), Compounded, Annual,
201  Duration::Modified, bondSettlementDate);
202  for (Size i=1; i<nSwaps_; ++i) {
203  swapRates_[i]= swaps_[i]->fairRate();
204  FixedRateBond swapBond(settlDays,
205  100.0, // faceAmount
206  swaps_[i]->fixedSchedule(),
207  std::vector<Rate>(1, swapRates_[i]),
208  fixedDayCount,
209  Following, // paymentConvention
210  100.0); // redemption
211  swapBondYields_[i] = BondFunctions::yield(swapBond,
212  100.0, // floating leg NPV including end payment
213  ActualActual(ActualActual::ISMA), Compounded, Annual,
214  bondSettlementDate,
215  // accuracy, maxIterations, guess
216  1.0e-10, 100, swapBondYields_[i]);
217  swapBondDurations_[i] = BondFunctions::duration(
218  swapBond, swapBondYields_[i],
219  ActualActual(ActualActual::ISMA), Compounded, Annual,
220  Duration::Modified, bondSettlementDate);
221  if (swapBondDurations_[i] > duration_) {
222  equivalentSwapIndex_ = i-1;
223  break; // exit the loop
224  }
225  }
226  }
