334  template <class Model>
335  typename SwaptionVolCube1x<Model>::Cube
336  SwaptionVolCube1x<Model>::sabrCalibration(const Cube &marketVolCube) const {
337 
338  const std::vector<Time>& optionTimes = marketVolCube.optionTimes();
339  const std::vector<Time>& swapLengths = marketVolCube.swapLengths();
340  const std::vector<Date>& optionDates = marketVolCube.optionDates();
341  const std::vector<Period>& swapTenors = marketVolCube.swapTenors();
342  Matrix alphas(optionTimes.size(), swapLengths.size(),0.);
343  Matrix betas(alphas);
344  Matrix nus(alphas);
345  Matrix rhos(alphas);
346  Matrix forwards(alphas);
347  Matrix errors(alphas);
348  Matrix maxErrors(alphas);
349  Matrix endCriteria(alphas);
350 
351  const std::vector<Matrix>& tmpMarketVolCube = marketVolCube.points();
352 
353  std::vector<Real> strikes(strikeSpreads_.size());
354  std::vector<Real> volatilities(strikeSpreads_.size());
355 
356  for (Size j=0; j<optionTimes.size(); j++) {
357  for (Size k=0; k<swapLengths.size(); k++) {
358  Rate atmForward = atmStrike(optionDates[j], swapTenors[k]);
359  Real shiftTmp = atmVol_->shift(optionTimes[j], swapLengths[k]);
360  strikes.clear();
361  volatilities.clear();
362  for (Size i=0; i<nStrikes_; i++){
363  Real strike = atmForward+strikeSpreads_[i];
364  if(strike + shiftTmp >=cutoffStrike_) {
365  strikes.push_back(strike);
366  volatilities.push_back(tmpMarketVolCube[i][j][k]);
367  }
368  }
369 
370  const std::vector<Real>& guess =
371  parametersGuess_(optionTimes[j], swapLengths[k]);
372 
373  const ext::shared_ptr<typename Model::Interpolation> sabrInterpolation =
374  ext::shared_ptr<typename Model::Interpolation>(new
375  (typename Model::Interpolation)(strikes.begin(), strikes.end(),
376  volatilities.begin(),
377  optionTimes[j], atmForward,
378  guess[0], guess[1],
379  guess[2], guess[3],
380  isParameterFixed_[0],
381  isParameterFixed_[1],
382  isParameterFixed_[2],
383  isParameterFixed_[3],
384  vegaWeightedSmileFit_,
385  endCriteria_,
386  optMethod_,
387  errorAccept_,
388  useMaxError_,
389  maxGuesses_,
390  shiftTmp));
391  sabrInterpolation->update();
392 
393  Real rmsError = sabrInterpolation->rmsError();
394  Real maxError = sabrInterpolation->maxError();
395  alphas [j][k] = sabrInterpolation->alpha();
396  betas [j][k] = sabrInterpolation->beta();
397  nus [j][k] = sabrInterpolation->nu();
398  rhos [j][k] = sabrInterpolation->rho();
399  forwards [j][k] = atmForward;
400  errors [j][k] = rmsError;
401  maxErrors [j][k] = maxError;
402  endCriteria[j][k] = sabrInterpolation->endCriteria();
403 
404  QL_ENSURE(endCriteria[j][k]!=EndCriteria::MaxIterations,
405  "global swaptions calibration failed: "
406  "MaxIterations reached: " << "\n" <<
407  "option maturity = " << optionDates[j] << ", \n" <<
408  "swap tenor = " << swapTenors[k] << ", \n" <<
409  "rms error = " << io::rate(errors[j][k]) << ", \n" <<
410  "max error = " << io::rate(maxErrors[j][k]) << ", \n" <<
411  " alpha = " << alphas[j][k] << "n" <<
412  " beta = " << betas[j][k] << "\n" <<
413  " nu = " << nus[j][k] << "\n" <<
414  " rho = " << rhos[j][k] << "\n"
415  );
416 
417  QL_ENSURE((useMaxError_ ? maxError : rmsError) < maxErrorTolerance_,
418  "global swaptions calibration failed: "
419  "error tolerance exceeded: "
420  << "\n"
421  << "using " << (useMaxError_ ? "maxError" : "rmsError")
422  << " tolerance " << maxErrorTolerance_ << ", \n"
423  << "option maturity = " << optionDates[j] << ", \n"
424  << "swap tenor = " << swapTenors[k] << ", \n"
425  << "rms error = " << io::rate(errors[j][k]) << ", \n"
426  << "max error = " << io::rate(maxErrors[j][k]) << ", \n"
427  << " alpha = " << alphas[j][k] << "n"
428  << " beta = " << betas[j][k] << "\n"
429  << " nu = " << nus[j][k] << "\n"
430  << " rho = " << rhos[j][k] << "\n");
431  }
432  }
433  Cube sabrParametersCube(optionDates, swapTenors,
434  optionTimes, swapLengths, 8,
435  true, backwardFlat_);
436  sabrParametersCube.setLayer(0, alphas);
437  sabrParametersCube.setLayer(1, betas);
438  sabrParametersCube.setLayer(2, nus);
439  sabrParametersCube.setLayer(3, rhos);
440  sabrParametersCube.setLayer(4, forwards);
441  sabrParametersCube.setLayer(5, errors);
442  sabrParametersCube.setLayer(6, maxErrors);
443  sabrParametersCube.setLayer(7, endCriteria);
444 
445  return sabrParametersCube;
446 
447  }

477  strikes.push_back(strike);

506  calibrationResult[0]=sabrInterpolation->alpha();
507  calibrationResult[1]=sabrInterpolation->beta();

509  calibrationResult[3]=sabrInterpolation->rho();

523  ", nu " << calibrationResult[2] <<
524  ", rho " << calibrationResult[3] <<
525  ", max error " << calibrationResult[6] <<
526  ", error " << calibrationResult[5]
527  );
528

530  "section calibration failed: "
531  "option tenor " << optionDates[j] <<
532  ", swap tenor " << swapTenors[k] <<
533  (useMaxError_ ? ": max error " : ": error ") <<
534  (useMaxError_ ? calibrationResult[6] : calibrationResult[5]) <<
535  ", alpha " << calibrationResult[0] <<
536  ", beta " << calibrationResult[1] <<
537  ", nu " << calibrationResult[2] <<
538  ", rho " << calibrationResult[3] <<
539  (useMaxError_ ? ": error" : ": max error ") <<
540  (useMaxError_ ? calibrationResult[5] : calibrationResult[6])
541  );
542 
543  parametersCube.setPoint(optionDates[j], swapTenors[k],
544  optionTimes[j], swapLengths[k],
545  calibrationResult);
546  parametersCube.updateInterpolators();
547  }

646  template<class Model> std::vector<Real> SwaptionVolCube1x<Model>::spreadVolInterpolation(
647  const Date& atmOptionDate, const Period& atmSwapTenor) const {
648 
649  Time atmOptionTime = timeFromReference(atmOptionDate);
650  Time atmTimeLength = swapLength(atmSwapTenor);
651 
652  std::vector<Real> result;
653  const std::vector<Time>& optionTimes(sparseParameters_.optionTimes());
654  const std::vector<Time>& swapLengths(sparseParameters_.swapLengths());
655  const std::vector<Date>& optionDates =
656  sparseParameters_.optionDates();
657  const std::vector<Period>& swapTenors = sparseParameters_.swapTenors();
658 
659  std::vector<Real>::const_iterator optionTimesPreviousNode,
660  swapLengthsPreviousNode;
661 
662  optionTimesPreviousNode = std::lower_bound(optionTimes.begin(),
663  optionTimes.end(),
664  atmOptionTime);
665  Size optionTimesPreviousIndex =
666  optionTimesPreviousNode - optionTimes.begin();
667  if (optionTimesPreviousIndex >0)
668  optionTimesPreviousIndex --;
669 
670  swapLengthsPreviousNode = std::lower_bound(swapLengths.begin(),
671  swapLengths.end(),
672  atmTimeLength);
673  Size swapLengthsPreviousIndex = swapLengthsPreviousNode - swapLengths.begin();
674  if (swapLengthsPreviousIndex >0)
675  swapLengthsPreviousIndex --;
676 
677  std::vector< std::vector<ext::shared_ptr<SmileSection> > > smiles;
678  std::vector<ext::shared_ptr<SmileSection> > smilesOnPreviousExpiry;
679  std::vector<ext::shared_ptr<SmileSection> > smilesOnNextExpiry;
680 
681  QL_REQUIRE(optionTimesPreviousIndex+1 < sparseSmiles_.size(),
682  "optionTimesPreviousIndex+1 >= sparseSmiles_.size()");
683  QL_REQUIRE(swapLengthsPreviousIndex+1 < sparseSmiles_[0].size(),
684  "swapLengthsPreviousIndex+1 >= sparseSmiles_[0].size()");
685  smilesOnPreviousExpiry.push_back(
686  sparseSmiles_[optionTimesPreviousIndex][swapLengthsPreviousIndex]);
687  smilesOnPreviousExpiry.push_back(
688  sparseSmiles_[optionTimesPreviousIndex][swapLengthsPreviousIndex+1]);
689  smilesOnNextExpiry.push_back(
690  sparseSmiles_[optionTimesPreviousIndex+1][swapLengthsPreviousIndex]);
691  smilesOnNextExpiry.push_back(
692  sparseSmiles_[optionTimesPreviousIndex+1][swapLengthsPreviousIndex+1]);
693 
694  smiles.push_back(smilesOnPreviousExpiry);
695  smiles.push_back(smilesOnNextExpiry);
696 
697  std::vector<Real> optionsNodes(2);
698  optionsNodes[0] = optionTimes[optionTimesPreviousIndex];
699  optionsNodes[1] = optionTimes[optionTimesPreviousIndex+1];
700 
701  std::vector<Date> optionsDateNodes(2);
702  optionsDateNodes[0] = optionDates[optionTimesPreviousIndex];
703  optionsDateNodes[1] = optionDates[optionTimesPreviousIndex+1];
704 
705  std::vector<Real> swapLengthsNodes(2);
706  swapLengthsNodes[0] = swapLengths[swapLengthsPreviousIndex];
707  swapLengthsNodes[1] = swapLengths[swapLengthsPreviousIndex+1];
708 
709  std::vector<Period> swapTenorNodes(2);
710  swapTenorNodes[0] = swapTenors[swapLengthsPreviousIndex];
711  swapTenorNodes[1] = swapTenors[swapLengthsPreviousIndex+1];
712 
713  Rate atmForward = atmStrike(atmOptionDate, atmSwapTenor);
714  Real shift = atmVol_->shift(atmOptionTime, atmTimeLength);
715 
716  Matrix atmForwards(2, 2, 0.0);
717  Matrix atmShifts(2,2,0.0);
718  Matrix atmVols(2, 2, 0.0);
719  for (Size i=0; i<2; i++) {
720  for (Size j=0; j<2; j++) {
721  atmForwards[i][j] = atmStrike(optionsDateNodes[i],
722  swapTenorNodes[j]);
723  atmShifts[i][j] = atmVol_->shift(optionsNodes[i], swapLengthsNodes[j]);
724  // atmVols[i][j] = smiles[i][j]->volatility(atmForwards[i][j]);
725  atmVols[i][j] = atmVol_->volatility(
726  optionsDateNodes[i], swapTenorNodes[j], atmForwards[i][j]);
727  /* With the old implementation the interpolated spreads on ATM
728  volatilities were null even if the spreads on ATM volatilities to be
729  interpolated were non-zero. The new implementation removes
730  this behaviour, but introduces a small ERROR in the cube:
731  even if no spreads are applied on any cube ATM volatility corresponding
732  to quoted smile sections (that is ATM volatilities in sparse cube), the
733  cube ATM volatilities corresponding to not quoted smile sections (that
734  is ATM volatilities in dense cube) are no more exactly the quoted values,
735  but that ones PLUS the linear interpolation of the fit errors on the ATM
736  volatilities in sparse cube whose spreads are used in the calculation.
737  A similar imprecision is introduced to the volatilities in dense cube
738  whith moneyness near to 1.
739  (See below how spreadVols are calculated).
740  The extent of this error depends on the quality of the fit: in case of
741  good fits it is negligibile.
742  */
743  }
744  }
745 
746  for (Size k=0; k<nStrikes_; k++){
747  const Real strike = std::max(atmForward + strikeSpreads_[k],cutoffStrike_-shift);
748  const Real moneyness = (atmForward+shift)/(strike+shift);
749 
750  Matrix strikes(2,2,0.);
751  Matrix spreadVols(2,2,0.);
752  for (Size i=0; i<2; i++){
753  for (Size j=0; j<2; j++){
754  strikes[i][j] = (atmForwards[i][j]+atmShifts[i][j])/moneyness - atmShifts[i][j];
755  spreadVols[i][j] =
756  smiles[i][j]->volatility(strikes[i][j]) - atmVols[i][j];
757  }
758  }
759  Cube localInterpolator(optionsDateNodes, swapTenorNodes,
760  optionsNodes, swapLengthsNodes, 1);
761  localInterpolator.setLayer(0, spreadVols);
762  localInterpolator.updateInterpolators();
763 
764  result.push_back(localInterpolator(atmOptionTime, atmTimeLength)[0]);
765  }
766  return result;
767  }
