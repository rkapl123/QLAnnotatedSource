108  class TwoFactorModel::ShortRateTree
109  : public TreeLattice2D<TwoFactorModel::ShortRateTree,TrinomialTree> {
110  public:
112  ShortRateTree(const ext::shared_ptr<TrinomialTree>& tree1,
113  const ext::shared_ptr<TrinomialTree>& tree2,
114  const ext::shared_ptr<ShortRateDynamics>& dynamics);
115 
116  DiscountFactor discount(Size i, Size index) const {
117  Size modulo = tree1_->size(i);
118  Size index1 = index % modulo;
119  Size index2 = index / modulo;
120 
121  Real x = tree1_->underlying(i, index1);
122  Real y = tree2_->underlying(i, index2);
123 
124  Rate r = dynamics_->shortRate(timeGrid()[i], x, y);
125  return std::exp(-r*timeGrid().dt(i));
126  }
127  private:
128  ext::shared_ptr<ShortRateDynamics> dynamics_;
129  };
