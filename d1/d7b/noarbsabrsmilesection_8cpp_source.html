73 Real NoArbSabrSmileSection::volatilityImpl(Rate strike) const {
74 
75  Real impliedVol = 0.0;
76  try {
77  Option::Type type;
78  if (strike >= forward_)
79  type = Option::Call;
80  else
81  type = Option::Put;
82  impliedVol =
83  blackFormulaImpliedStdDev(type, strike, forward_,
84  optionPrice(strike, type, 1.0), 1.0) /
85  std::sqrt(exerciseTime());
86  } catch (...) {
87  }
88  if (impliedVol == 0.0)
89  // fall back on Hagan 2002 expansion
90  impliedVol =
91  unsafeSabrVolatility(strike, forward_, exerciseTime(), params_[0],
92  params_[1], params_[2], params_[3]);
93 
94  return impliedVol;
95 }
