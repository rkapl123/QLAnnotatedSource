76Real NoArbSabrSmileSection::volatilityImpl(Rate strike) const {
77
78 Real impliedVol = 0.0;
79 try {
80 Option::Type type;
81 if (strike >= forward_)
82 type = Option::Call;
83 else
84 type = Option::Put;
85 impliedVol =
86 blackFormulaImpliedStdDev(type, strike, forward_,
87 optionPrice(strike, type, 1.0), 1.0) /
88 std::sqrt(exerciseTime());
89 } catch (...) {
90 }
91 if (impliedVol == 0.0)
92 // fall back on Hagan 2002 expansion
93 impliedVol =
94 unsafeSabrVolatility(strike, forward_, exerciseTime(), params_[0],
95 params_[1], params_[2], params_[3], volatilityType());
96
97 return impliedVol;
98}
