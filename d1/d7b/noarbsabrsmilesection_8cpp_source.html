74 Real NoArbSabrSmileSection::volatilityImpl(Rate strike) const {
75 
76  Real impliedVol = 0.0;
77  try {
78  Option::Type type;
79  if (strike >= forward_)
80  type = Option::Call;
81  else
82  type = Option::Put;
83  impliedVol =
84  blackFormulaImpliedStdDev(type, strike, forward_,
85  optionPrice(strike, type, 1.0), 1.0) /
86  std::sqrt(exerciseTime());
87  } catch (...) {
88  }
89  if (impliedVol == 0.0)
90  // fall back on Hagan 2002 expansion
91  impliedVol =
92  unsafeSabrVolatility(strike, forward_, exerciseTime(), params_[0],
93  params_[1], params_[2], params_[3]);
94 
95  return impliedVol;
96 }
