75 Real NoArbSabrSmileSection::volatilityImpl(Rate strike) const {
76 
77  Real impliedVol = 0.0;
78  try {
79  Option::Type type;
80  if (strike >= forward_)
81  type = Option::Call;
82  else
83  type = Option::Put;
84  impliedVol =
85  blackFormulaImpliedStdDev(type, strike, forward_,
86  optionPrice(strike, type, 1.0), 1.0) /
87  std::sqrt(exerciseTime());
88  } catch (...) {
89  }
90  if (impliedVol == 0.0)
91  // fall back on Hagan 2002 expansion
92  impliedVol =
93  unsafeSabrVolatility(strike, forward_, exerciseTime(), params_[0],
94  params_[1], params_[2], params_[3]);
95 
96  return impliedVol;
97 }
