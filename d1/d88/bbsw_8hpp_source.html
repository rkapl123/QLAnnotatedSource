35 /*! Bbsw rate fixed by AFMA.

37 See <http://www.afma.com.au/data/BBSW>.

39 class Bbsw : public IborIndex {
40 public:
41 Bbsw(const Period& tenor,
42 const Handle<YieldTermStructure>& h = {})
43 : IborIndex("Bbsw", tenor,
44 0, // settlement days
45 AUDCurrency(), Australia(),
46 HalfMonthModifiedFollowing, true,
