59  void applyTo(Array& a,

61  DiscountFactor B = std::exp(-rate_ * (t - resTime_));
62  //#pragma omp parallel for
63  for (Size i = 0; i < a.size(); i++) {
64  a[i] = std::max(a[i], B * impl_->getValue(a, i));
65  }
66  }
