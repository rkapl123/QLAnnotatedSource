39 class QL_DEPRECATED ShoutCondition : public StandardStepCondition {
40 public:
41 ShoutCondition(const Array& intrinsicValues,
42 Time resTime,
43 Rate rate)
44 : resTime_(resTime), rate_(rate),
45 impl_(new ArrayImpl(intrinsicValues)) {}
46
47 void applyTo(Array& a, Time t) const override {
48 DiscountFactor B = std::exp(-rate_ * (t - resTime_));
49 //#pragma omp parallel for
50 for (Size i = 0; i < a.size(); i++) {
51 a[i] = std::max(a[i], B * impl_->getValue(a, i));
52 }
53 }
54
