114  template <class Func, class Predicate>
115  std::pair<Real,Size> expectationValue(const Func& f,
116  const Predicate& inRange) const {
117  Real num = 0.0, den = 0.0;
118  Size N = 0;
119  std::vector<std::pair<Real,Real> >::const_iterator i;
120  for (i=samples_.begin(); i!=samples_.end(); ++i) {
121  Real x = i->first, w = i->second;
122  if (inRange(x)) {
123  num += f(x)*w;
124  den += w;
125  N += 1;
126  }
127  }
128  if (N == 0)
129  return std::make_pair<Real,Size>(Null<Real>(),0);
130  else
131  return std::make_pair(num/den,N);
132  }
