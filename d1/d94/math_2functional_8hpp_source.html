284 template <class F, class R>
285 class QL_DEPRECATED clipped_function {
286 public:
287 typedef typename F::argument_type argument_type;
288 typedef typename F::result_type result_type;
289 clipped_function(const F& f, const R& r) : f_(f), r_(r) {}
290 result_type operator()(const argument_type& x) const {
291 return r_(x) ? f_(x) : result_type();
292 }
293 private:
294 F f_;
295 R r_;
296 };
