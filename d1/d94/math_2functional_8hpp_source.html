308 template <class F, class R>
309 class QL_DEPRECATED clipped_function {
310 public:
311 typedef typename F::argument_type argument_type;
312 typedef typename F::result_type result_type;
313 QL_DEPRECATED_DISABLE_WARNING
314 clipped_function(const F& f, const R& r) : f_(f), r_(r) {}
315 QL_DEPRECATED_ENABLE_WARNING
316 result_type operator()(const argument_type& x) const {
317 return r_(x) ? f_(x) : result_type();
318 }
319 private:
320 F f_;
321 R r_;
322 };
