149  Real continuousToConv(Real oas,
150  const Bond &b,
151  const Handle<YieldTermStructure>& yts,
152  const DayCounter& dayCounter,
153  Compounding compounding,
154  Frequency frequency)
155  {
156  double zz=yts->zeroRate(b.maturityDate(),
157  dayCounter,

161  dayCounter,
162  Continuous,
163  NoFrequency);
164  InterestRate spreadedRate(oas+zz,
165  dayCounter,
166  Continuous,
167  NoFrequency);

169  compounding,
170  frequency,
171  yts->referenceDate(),
172  b.maturityDate()).rate();
173  double sr=spreadedRate.equivalentRate(dayCounter,

177  b.maturityDate()).rate();
178  // Return the spread

240  Spread CallableBond::OAS(Real cleanPrice,
241  const Handle<YieldTermStructure>& engineTS,
242  const DayCounter& dayCounter,
243  Compounding compounding,
244  Frequency frequency,
245  Date settlement,
246  Real accuracy,
247  Size maxIterations,
248  Spread guess)
249  {
250  if (settlement == Date())
251  settlement = settlementDate();
252 
253  Real dirtyPrice = cleanPrice + accruedAmount(settlement);
254 
255  boost::function<Real(Real)> f = NPVSpreadHelper(*this);
256  OASHelper obj(f, dirtyPrice);
257 
258  Brent solver;
259  solver.setMaxEvaluations(maxIterations);
260 
261  Real step = 0.001;
262  Spread oas=solver.solve(obj, accuracy, guess, step);
263 
264  return continuousToConv(oas,
265  *this,
266  engineTS,
267  dayCounter,
268  compounding,
269  frequency);
270  }

298  Real CallableBond::effectiveDuration(Real oas,
299  const Handle<YieldTermStructure>& engineTS,
300  const DayCounter& dayCounter,
301  Compounding compounding,
302  Frequency frequency,
303  Real bump)
304  {
305  Real P = cleanPriceOAS(oas,
306  engineTS,
307  dayCounter,
308  compounding,
309  frequency);
310 
311  Real Ppp = cleanPriceOAS(oas+bump,
312  engineTS,
313  dayCounter,
314  compounding,
315  frequency);
316  Real Pmm = cleanPriceOAS(oas-bump,
317  engineTS,
318  dayCounter,
319  compounding,
320  frequency);
321 
322  if ( P == 0.0 )
323  return 0;
324  else
325  {
326  return (Pmm-Ppp)/(2*P*bump);
327  }
328  }

404  Real CallableFixedRateBond::accrued(Date settlement) const {
405 
406  if (settlement == Date()) settlement = settlementDate();
407 
408  const bool IncludeToday = false;
409  for (Size i = 0; i<cashflows_.size(); ++i) {
410  // the first coupon paying after d is the one we're after
411  if (!cashflows_[i]->hasOccurred(settlement,IncludeToday)) {
412  boost::shared_ptr<Coupon> coupon =
413  boost::dynamic_pointer_cast<Coupon>(cashflows_[i]);
414  if (coupon)
415  // !!!
416  return coupon->accruedAmount(settlement) /
417  notional(settlement) * 100.0;
418  else
419  return 0.0;
420  }
421  }
422  return 0.0;
423  }

426  void CallableFixedRateBond::setupArguments(
427  PricingEngine::arguments* args) const {
428 
429  Bond::setupArguments(args);
430  CallableBond::arguments* arguments =
431  dynamic_cast<CallableBond::arguments*>(args);
432 
433  QL_REQUIRE(arguments != 0, "no arguments given");
434 
435  Date settlement = arguments->settlementDate;
436 
437  arguments->redemption = redemption()->amount();
438  arguments->redemptionDate = redemption()->date();
439 
440  const Leg& cfs = cashflows();
441 
442  arguments->couponDates.clear();
443  arguments->couponDates.reserve(cfs.size()-1);
444  arguments->couponAmounts.clear();
445  arguments->couponAmounts.reserve(cfs.size()-1);
446 
447  for (Size i=0; i<cfs.size()-1; i++) {
448  if (!cfs[i]->hasOccurred(settlement, false)) {
449  arguments->couponDates.push_back(cfs[i]->date());
450  arguments->couponAmounts.push_back(cfs[i]->amount());
451  }
452  }
453 
454  arguments->callabilityPrices.clear();
455  arguments->callabilityDates.clear();
456  arguments->callabilityPrices.reserve(putCallSchedule_.size());
457  arguments->callabilityDates.reserve(putCallSchedule_.size());
458 
459  arguments->paymentDayCounter = paymentDayCounter_;
460  arguments->frequency = frequency_;
461 
462  arguments->putCallSchedule = putCallSchedule_;
463  for (Size i=0; i<putCallSchedule_.size(); i++) {
464  if (!putCallSchedule_[i]->hasOccurred(settlement, false)) {
465  arguments->callabilityDates.push_back(
466  putCallSchedule_[i]->date());
467  arguments->callabilityPrices.push_back(
468  putCallSchedule_[i]->price().amount());
469 
470  if (putCallSchedule_[i]->price().type()==
471  Callability::Price::Clean) {
472  /* calling accrued() forces accrued interest to be zero
473  if future option date is also coupon date, so that dirty
474  price = clean price. Use here because callability is
475  always applied before coupon in the tree engine.
476  */
477  arguments->callabilityPrices.back() +=
478  this->accrued(putCallSchedule_[i]->date());
479  }
480  }
481  }
482 
483  arguments->spread = 0.0;
484  }
