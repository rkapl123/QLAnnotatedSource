112  Real EnergyCommodity::calculateFxConversionFactor(

127  Real EnergyCommodity::calculateUnitCost(const CommodityType& commodityType,
128  const CommodityUnitCost& unitCost,
129  const Date& evaluationDate) const {
130  if (unitCost.amount().value() != 0) {
131  const Currency& baseCurrency =
132  CommoditySettings::instance().currency();
133  const UnitOfMeasure baseUnitOfMeasure =
134  CommoditySettings::instance().unitOfMeasure();
135  Real unitCostUomConversionFactor =
136  calculateUomConversionFactor(commodityType,
137  unitCost.unitOfMeasure(),
138  baseUnitOfMeasure);
139  Real unitCostFxConversionFactor =
140  calculateFxConversionFactor(unitCost.amount().currency(),
141  baseCurrency, evaluationDate);
142  return unitCost.amount().value() * unitCostUomConversionFactor
143  * unitCostFxConversionFactor;
144  }
145  return 0;
146  }

148  void EnergyCommodity::calculateSecondaryCostAmounts(
149  const CommodityType& commodityType,
150  Real totalQuantityValue,
151  const Date& evaluationDate) const {
152  secondaryCostAmounts_.clear();
153  if (secondaryCosts_ != 0) {
154  const Currency& baseCurrency =
155  CommoditySettings::instance().currency();
156  try {
157  for (SecondaryCosts::const_iterator i = secondaryCosts_->begin();
158  i != secondaryCosts_->end(); ++i) {
159  if (boost::any_cast<CommodityUnitCost>(&i->second) != 0) {
160  Real value =
161  calculateUnitCost(
162  commodityType,
163  boost::any_cast<CommodityUnitCost>(i->second),
164  evaluationDate) * totalQuantityValue;
165  secondaryCostAmounts_[i->first] =
166  Money(baseCurrency, value);
167  } else if (boost::any_cast<Money>(&i->second) != 0) {
168  const Money& amount = boost::any_cast<Money>(i->second);
169  Real fxConversionFactor =
170  calculateFxConversionFactor(amount.currency(),
171  baseCurrency,
172  evaluationDate);
173  secondaryCostAmounts_[i->first] =
174  Money(baseCurrency,
175  amount.value() * fxConversionFactor);
176  }
177  }
178  } catch (const std::exception& e) {
179  QL_FAIL("error calculating secondary costs: " << e.what());
180  }
181  }
182  }
