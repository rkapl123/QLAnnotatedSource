108 inline void MultiCurveSensitivities::performCalculations() const {
109  std::vector< Rate > sensiVector;
110  origZeros_ = allZeros();
111  for (const auto& allQuote : allQuotes_) {
112  Rate bps = +1e-4;
113  Rate origQuote = allQuote->value();
114  ext::shared_ptr<SimpleQuote> q =
115  ext::dynamic_pointer_cast<SimpleQuote>(allQuote.currentLink());
116  q->setValue(origQuote + bps);
117  try {
118  std::vector<Rate> tmp(allZeros());
119  for (Size i = 0; i < tmp.size(); ++i)
120  sensiVector.push_back((tmp[i] - origZeros_[i]) / bps);
121  q->setValue(origQuote);
122  } catch (...) {
123  q->setValue(origQuote);
124  QL_FAIL("Application of shift to quote led to exception.");
125  }
126  }
127  Matrix result(origZeros_.size(), origZeros_.size(), sensiVector.begin(), sensiVector.end());
128  sensi_ = result;
129  invSensi_ = inverse(sensi_);
130 }

142 inline std::vector< std::pair< Date, Real > > MultiCurveSensitivities::allNodes() const {
143  std::vector< std::pair< Date, Real > > result;
144  for (const auto& it : curves_) {
145  ext::shared_ptr<PiecewiseYieldCurve<ZeroYield, Linear> > curve =
146  ext::dynamic_pointer_cast<PiecewiseYieldCurve<ZeroYield, Linear> >(
147  it.second.currentLink());
148  result.reserve(result.size() + curve->nodes().size() - 1);
149  for (std::vector<std::pair<Date, Real> >::const_iterator p = curve->nodes().begin() + 1;
150  p != curve->nodes().end(); ++p)
151  result.push_back(*p);
152  }
153  return result;
154 }
