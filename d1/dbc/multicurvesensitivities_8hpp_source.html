110inline void MultiCurveSensitivities::performCalculations() const {
111 std::vector< Rate > sensiVector;
112 origZeros_ = allZeros();
113 for (const auto& allQuote : allQuotes_) {
114 Rate bps = +1e-4;
115 Rate origQuote = allQuote->value();
116 ext::shared_ptr<SimpleQuote> q =
117 ext::dynamic_pointer_cast<SimpleQuote>(allQuote.currentLink());
118 q->setValue(origQuote + bps);
119 try {
120 std::vector<Rate> tmp(allZeros());
121 for (Size i = 0; i < tmp.size(); ++i)
122 sensiVector.push_back((tmp[i] - origZeros_[i]) / bps);
123 q->setValue(origQuote);
124 } catch (...) {
125 q->setValue(origQuote);
126 QL_FAIL("Application of shift to quote led to exception.");
127 }
128 }
129 Matrix result(origZeros_.size(), origZeros_.size(), sensiVector.begin(), sensiVector.end());
130 sensi_ = result;
131 invSensi_ = inverse(sensi_);
132}

144inline std::vector< std::pair< Date, Real > > MultiCurveSensitivities::allNodes() const {
145 std::vector< std::pair< Date, Real > > result;
146 for (const auto& it : curves_) {
147 ext::shared_ptr<PiecewiseYieldCurve<ZeroYield, Linear> > curve =
148 ext::dynamic_pointer_cast<PiecewiseYieldCurve<ZeroYield, Linear> >(
149 it.second.currentLink());
150 result.reserve(result.size() + curve->nodes().size() - 1);
151 for (std::vector<std::pair<Date, Real> >::const_iterator p = curve->nodes().begin() + 1;
152 p != curve->nodes().end(); ++p)
153 result.push_back(*p);
154 }
155 return result;
156}
