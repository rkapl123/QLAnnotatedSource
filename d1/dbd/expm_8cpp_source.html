54  Disposable<Matrix> Expm(const Matrix& M, Real t, Real tol) {
55  const Size n = M.rows();
56  QL_REQUIRE(n == M.columns(), "Expm expects a square matrix");
57 
58  AdaptiveRungeKutta<> rk(tol);
59  AdaptiveRungeKutta<>::OdeFct odeFct = MatrixVectorProductFct(M);
60 
61  Matrix result(n, n);
62  for (Size i=0; i < n; ++i) {
63  std::vector<Real> x0(n, 0.0);
64  x0[i] = 1.0;
65 
66  const std::vector<Real> r = rk(odeFct, x0, 0.0, t);
67  std::copy(r.begin(), r.end(), result.column_begin(i));
68  }
69  return result;
70  }
