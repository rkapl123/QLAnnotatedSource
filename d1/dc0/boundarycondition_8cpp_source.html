27 void NeumannBC::applyBeforeApplying(TridiagonalOperator& L) const {
28 switch (side_) {
29 case Lower:
30 L.setFirstRow(-1.0,1.0);
31 break;
32 case Upper:
33 L.setLastRow(-1.0,1.0);
34 break;
35 default:
36 QL_FAIL("unknown side for Neumann boundary condition");
37 }
38 }

40 void NeumannBC::applyAfterApplying(Array& u) const {
41 switch (side_) {
42 case Lower:
43 u[0] = u[1] - value_;
44 break;
45 case Upper:
46 u[u.size()-1] = u[u.size()-2] + value_;
47 break;
48 default:
49 QL_FAIL("unknown side for Neumann boundary condition");
50 }
