36  HaltonRsg::HaltonRsg(Size dimensionality, unsigned long seed,
37  bool randomStart, bool randomShift)
38  : dimensionality_(dimensionality), sequenceCounter_(0),
39  sequence_(std::vector<Real> (dimensionality), 1.0),
40  randomStart_(dimensionality, 0UL),
41  randomShift_(dimensionality, 0.0) {
42 
43  QL_REQUIRE(dimensionality>0,
44  "dimensionality must be greater than 0");
45 
46  if (randomStart || randomShift) {
47  RandomSequenceGenerator<MersenneTwisterUniformRng>
48  uniformRsg(dimensionality_, seed);
49  if (randomStart)
50  randomStart_ = uniformRsg.nextInt32Sequence();

53  }

57  const HaltonRsg::sample_type& HaltonRsg::nextSequence() const {
58  ++sequenceCounter_;
59  for (Size i=0; i<dimensionality_; ++i) {
60  double h = 0.0;
61  unsigned long b = PrimeNumbers::get(i);
62  double f = 1.0;
63  unsigned long k = sequenceCounter_+randomStart_[i];
64  while (k != 0U) {
65  f /= b;
66  h += (k%b)*f;
67  k /= b;
68  }
69  sequence_.value[i] = h+randomShift_[i];
70  sequence_.value[i] -= long(sequence_.value[i]);
71  }
72  return sequence_;
73  }
