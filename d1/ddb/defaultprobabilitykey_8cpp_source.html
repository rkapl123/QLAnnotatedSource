39  bool operator==(const DefaultProbKey& lhs, const DefaultProbKey& rhs) {
40  if(lhs.seniority() != rhs.seniority()) return false;
41  if(lhs.currency() != rhs.currency()) return false;

43  Size mySize = rhs.eventTypes().size();
44  if(mySize != lhs.eventTypes().size()) return false;
45  // the all types must be equal in the weak sense.

48  *boost::lambda::_1 == *rhs.eventTypes()[i]) == lhs.eventTypes().end())

51  return true;
52  }
