52  Real FdmLogInnerValue::avgInnerValue(
53  const FdmLinearOpIterator& iter, Time t) {
54  if (avgInnerValues_.empty()) {
55  // calculate caching values
56  avgInnerValues_.resize(mesher_->layout()->dim()[direction_]);
57  std::deque<bool> initialized(avgInnerValues_.size(), false);
58 
59  const boost::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
60  const FdmLinearOpIterator endIter = layout->end();
61  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
62  ++iter) {
63  const Size xn = iter.coordinates()[direction_];
64  if (!initialized[xn]) {
65  initialized[xn] = true;
66  avgInnerValues_[xn] = avgInnerValueCalc(iter, t);
67  }
68  }
69  }
70 
71  return avgInnerValues_[iter.coordinates()[direction_]];
72  }

74  Real FdmLogInnerValue::avgInnerValueCalc(
75  const FdmLinearOpIterator& iter, Time t) {
76  const Size dim = mesher_->layout()->dim()[direction_];
77  const Size coord = iter.coordinates()[direction_];
78  const Real loc = mesher_->location(iter,direction_);
79  Real a = loc;
80  Real b = loc;
81  if (coord > 0) {
82  a -= mesher_->dminus(iter, direction_)/2.0;
83  }
84  if (coord < dim-1) {
85  b += mesher_->dplus(iter, direction_)/2.0;
86  }
87  boost::function1<Real, Real> f = compose(
88  std::bind1st(std::mem_fun(&Payoff::operator()), payoff_.get()),
89  std::ptr_fun<Real,Real>(std::exp));
90 
91  Real retVal;
92  try {
93  const Real acc
94  = ((f(a) != 0.0 || f(b) != 0.0) ? (f(a)+f(b))*5e-5 : 1e-4);
95  retVal = SimpsonIntegral(acc, 8)(f, a, b)/(b-a);
96  }
97  catch (Error&) {
98  // use default value
99  retVal = innerValue(iter, t);
100  }
101 
102  return retVal;
103  }
