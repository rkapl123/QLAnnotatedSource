64  Real FdmCellAveragingInnerValue::avgInnerValue(const FdmLinearOpIterator& iter, Time t) {
65  if (avgInnerValues_.empty()) {
66  // calculate caching values
67  avgInnerValues_.resize(mesher_->layout()->dim()[direction_]);
68  std::deque<bool> initialized(avgInnerValues_.size(), false);
69 
70  const ext::shared_ptr<FdmLinearOpLayout> layout =
71  mesher_->layout();
72  const FdmLinearOpIterator endIter = layout->end();
73  for (FdmLinearOpIterator i = layout->begin(); i != endIter; ++i) {
74  const Size xn = i.coordinates()[direction_];
75  if (!initialized[xn]) {
76  initialized[xn] = true;
77  avgInnerValues_[xn] = avgInnerValueCalc(i, t);
78  }
79  }
80  }
81 
82  return avgInnerValues_[iter.coordinates()[direction_]];
83  }

85  Real FdmCellAveragingInnerValue::avgInnerValueCalc(const FdmLinearOpIterator& iter, Time t) {
86  const Size dim = mesher_->layout()->dim()[direction_];
87  const Size coord = iter.coordinates()[direction_];
88 
89  if (coord == 0 || coord == dim-1)
90  return innerValue(iter, t);
91 
92  const Real loc = mesher_->location(iter,direction_);
93  const Real a = loc - mesher_->dminus(iter, direction_)/2.0;
94  const Real b = loc + mesher_->dplus(iter, direction_)/2.0;
95 
96  mapped_payoff f(*payoff_, gridMapping_);
97 
98  Real retVal;
99  try {
100  const Real acc
101  = ((f(a) != 0.0 || f(b) != 0.0) ? (f(a)+f(b))*5e-5 : 1e-4);
102  retVal = SimpsonIntegral(acc, 8)(f, a, b)/(b-a);
103  }
104  catch (Error&) {
105  // use default value
106  retVal = innerValue(iter, t);
107  }
108 
109  return retVal;
110  }
