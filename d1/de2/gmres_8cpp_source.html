69  GMRESResult GMRES::solveImpl(const Array& b, const Array& x0) const {
70  const Real bn = Norm2(b);
71  if (bn == 0.0) {
72  GMRESResult result = { std::list<Real>(1, 0.0), b };
73  return result;
74  }
75 
76  Array x = ((!x0.empty()) ? x0 : Array(b.size(), 0.0));
77  Array r = b - A_(x);
78 
79  const Real g = Norm2(r);
80  if (g/bn < relTol_) {
81  GMRESResult result = { std::list<Real>(1, g/bn), x };
82  return result;
83  }
84 
85  std::vector<Array> v(1, r/g);
86  std::vector<Array> h(1, Array(maxIter_, 0.0));
87  std::vector<Real> c(maxIter_+1), s(maxIter_+1), z(maxIter_+1);
88 
89  z[0] = g;
90 
91  std::list<Real> errors(1, g/bn);
92 
93  for (Size j=0; j < maxIter_ && errors.back() >= relTol_; ++j) {
94  h.push_back(Array(maxIter_, 0.0));
95  Array w = A_((M_)? M_(v[j]) : v[j]);
96 
97  for (Size i=0; i <= j; ++i) {
98  h[i][j] = DotProduct(w, v[i]);
99  w -= h[i][j] * v[i];
100  }
101 
102  h[j+1][j] = Norm2(w);
103 
104  if (h[j+1][j] < QL_EPSILON*QL_EPSILON)
105  break;
106 
107  v.push_back(w / h[j+1][j]);
108 
109  for (Size i=0; i < j; ++i) {
110  const Real h0 = c[i]*h[i][j] + s[i]*h[i+1][j];
111  const Real h1 =-s[i]*h[i][j] + c[i]*h[i+1][j];
112 
113  h[i][j] = h0;
114  h[i+1][j] = h1;
115  }
116 
117  const Real nu = std::sqrt( square<Real>()(h[j][j])
118  + square<Real>()(h[j+1][j]));
119 
120  c[j] = h[j][j]/nu;
121  s[j] = h[j+1][j]/nu;
122 
123  h[j][j] = nu;
124  h[j+1][j] = 0.0;
125 
126  z[j+1] = -s[j]*z[j];
127  z[j] = c[j] * z[j];
128 
129  errors.push_back(std::fabs(z[j+1]/bn));
130  }
131 
132  const Size k = v.size()-1;
133 
134  Array y(k, 0.0);
135  y[k-1]=z[k-1]/h[k-1][k-1];
136 
137  for (Integer i=k-2; i >= 0; --i) {
138  y[i] = (z[i] - std::inner_product(
139  h[i].begin()+i+1, h[i].begin()+k, y.begin()+i+1, 0.0))/h[i][i];
140  }
141 
142  Array xm = std::inner_product(
143  v.begin(), v.begin()+k, y.begin(), Array(x.size(), 0.0));
144 
145  xm = x + ((M_)? M_(xm) : xm);
146 
147  GMRESResult result = { errors, xm };
148  return result;
149  }
