36  EndCriteria::Type DifferentialEvolution::minimize(Problem& p, const EndCriteria& endCriteria) {
37  EndCriteria::Type ecType;
38 
39  upperBound_ = p.constraint().upperBound(p.currentValue());
40  lowerBound_ = p.constraint().lowerBound(p.currentValue());
41  currGenSizeWeights_ = Array(configuration().populationMembers,
42  configuration().stepsizeWeight);
43  currGenCrossover_ = Array(configuration().populationMembers,
44  configuration().crossoverProbability);
45 
46  std::vector<Candidate> population(configuration().populationMembers,
47  Candidate(p.currentValue().size()));
48  fillInitialPopulation(population, p);
49 
50  std::partial_sort(population.begin(), population.begin() + 1, population.end(),
51  sort_by_cost());
52  bestMemberEver_ = population.front();
53  Real fxOld = population.front().cost;
54  Size iteration = 0, stationaryPointIteration = 0;
55 
56  // main loop - calculate consecutive emerging populations
57  while (!endCriteria.checkMaxIterations(iteration++, ecType)) {
58  calculateNextGeneration(population, p.costFunction());
59  std::partial_sort(population.begin(), population.begin() + 1, population.end(),
60  sort_by_cost());
61  if (population.front().cost < bestMemberEver_.cost)
62  bestMemberEver_ = population.front();
63  Real fxNew = population.front().cost;
64  if (endCriteria.checkStationaryFunctionValue(fxOld, fxNew, stationaryPointIteration,
65  ecType))
66  break;
67  fxOld = fxNew;
68  };
69  p.setCurrentValue(bestMemberEver_.values);
70  p.setFunctionValue(bestMemberEver_.cost);
71  return ecType;
72  }

74  void DifferentialEvolution::calculateNextGeneration(
75  std::vector<Candidate>& population,
76  const CostFunction& costFunction) const {
77 
78  std::vector<Candidate> mirrorPopulation;
79  std::vector<Candidate> oldPopulation = population;
80 
81  switch (configuration().strategy) {
82 
83  case Rand1Standard: {
84  std::random_shuffle(population.begin(), population.end());
85  std::vector<Candidate> shuffledPop1 = population;
86  std::random_shuffle(population.begin(), population.end());
87  std::vector<Candidate> shuffledPop2 = population;
88  std::random_shuffle(population.begin(), population.end());
89  mirrorPopulation = shuffledPop1;
90 
91  for (Size popIter = 0; popIter < population.size(); popIter++) {
92  population[popIter].values = population[popIter].values
93  + configuration().stepsizeWeight
94  * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
95  }
96  }
97  break;
98 
99  case BestMemberWithJitter: {
100  std::random_shuffle(population.begin(), population.end());
101  std::vector<Candidate> shuffledPop1 = population;
102  std::random_shuffle(population.begin(), population.end());
103  Array jitter(population[0].values.size(), 0.0);
104 
105  for (Size popIter = 0; popIter < population.size(); popIter++) {
106  for (Size jitterIter = 0; jitterIter < jitter.size(); jitterIter++) {
107  jitter[jitterIter] = rng_.nextReal();
108  }
109  population[popIter].values = bestMemberEver_.values
110  + (shuffledPop1[popIter].values - population[popIter].values)
111  * (0.0001 * jitter + configuration().stepsizeWeight);
112  }
113  mirrorPopulation = std::vector<Candidate>(population.size(),
114  bestMemberEver_);
115  }
116  break;
117 
118  case CurrentToBest2Diffs: {
119  std::random_shuffle(population.begin(), population.end());
120  std::vector<Candidate> shuffledPop1 = population;
121  std::random_shuffle(population.begin(), population.end());
122 
123  for (Size popIter = 0; popIter < population.size(); popIter++) {
124  population[popIter].values = oldPopulation[popIter].values
125  + configuration().stepsizeWeight
126  * (bestMemberEver_.values - oldPopulation[popIter].values)
127  + configuration().stepsizeWeight
128  * (population[popIter].values - shuffledPop1[popIter].values);
129  }
130  mirrorPopulation = shuffledPop1;
131  }
132  break;
133 
134  case Rand1DiffWithPerVectorDither: {
135  std::random_shuffle(population.begin(), population.end());
136  std::vector<Candidate> shuffledPop1 = population;
137  std::random_shuffle(population.begin(), population.end());
138  std::vector<Candidate> shuffledPop2 = population;
139  std::random_shuffle(population.begin(), population.end());
140  mirrorPopulation = shuffledPop1;
141  Array FWeight = Array(population.front().values.size(), 0.0);
142  for (Size fwIter = 0; fwIter < FWeight.size(); fwIter++)
143  FWeight[fwIter] = (1.0 - configuration().stepsizeWeight)
144  * rng_.nextReal() + configuration().stepsizeWeight;
145  for (Size popIter = 0; popIter < population.size(); popIter++) {
146  population[popIter].values = population[popIter].values
147  + FWeight * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
148  }
149  }
150  break;
151 
152  case Rand1DiffWithDither: {
153  std::random_shuffle(population.begin(), population.end());
154  std::vector<Candidate> shuffledPop1 = population;
155  std::random_shuffle(population.begin(), population.end());
156  std::vector<Candidate> shuffledPop2 = population;
157  std::random_shuffle(population.begin(), population.end());
158  mirrorPopulation = shuffledPop1;
159  Real FWeight = (1.0 - configuration().stepsizeWeight) * rng_.nextReal()
160  + configuration().stepsizeWeight;
161  for (Size popIter = 0; popIter < population.size(); popIter++) {
162  population[popIter].values = population[popIter].values
163  + FWeight * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
164  }
165  }
166  break;
167 
168  case EitherOrWithOptimalRecombination: {
169  std::random_shuffle(population.begin(), population.end());
170  std::vector<Candidate> shuffledPop1 = population;
171  std::random_shuffle(population.begin(), population.end());
172  std::vector<Candidate> shuffledPop2 = population;
173  std::random_shuffle(population.begin(), population.end());
174  mirrorPopulation = shuffledPop1;
175  Real probFWeight = 0.5;
176  if (rng_.nextReal() < probFWeight) {
177  for (Size popIter = 0; popIter < population.size(); popIter++) {
178  population[popIter].values = oldPopulation[popIter].values
179  + configuration().stepsizeWeight
180  * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
181  }
182  } else {
183  Real K = 0.5 * (configuration().stepsizeWeight + 1); // invariant with respect to probFWeight used
184  for (Size popIter = 0; popIter < population.size(); popIter++) {
185  population[popIter].values = oldPopulation[popIter].values
186  + K
187  * (shuffledPop1[popIter].values - shuffledPop2[popIter].values
188  - 2.0 * population[popIter].values);
189  }
190  }
191  }
192  break;
193 
194  case Rand1SelfadaptiveWithRotation: {
195  std::random_shuffle(population.begin(), population.end());
196  std::vector<Candidate> shuffledPop1 = population;
197  std::random_shuffle(population.begin(), population.end());
198  std::vector<Candidate> shuffledPop2 = population;
199  std::random_shuffle(population.begin(), population.end());
200  mirrorPopulation = shuffledPop1;
201 
202  adaptSizeWeights();
203 
204  for (Size popIter = 0; popIter < population.size(); popIter++) {
205  if (rng_.nextReal() < 0.1){
206  population[popIter].values = rotateArray(bestMemberEver_.values);
207  }else {
208  population[popIter].values = bestMemberEver_.values
209  + currGenSizeWeights_[popIter]
210  * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
211  }
212  }
213  }
214  break;
215 
216  default:
217  QL_FAIL("Unknown strategy ("
218  << Integer(configuration().strategy) << ")");
219  }
220  // in order to avoid unnecessary copying we use the same population object for mutants
221  crossover(oldPopulation, population, population, mirrorPopulation,
222  costFunction);
223  }

225  void DifferentialEvolution::crossover(
226  const std::vector<Candidate>& oldPopulation,
227  std::vector<Candidate>& population,
228  const std::vector<Candidate>& mutantPopulation,
229  const std::vector<Candidate>& mirrorPopulation,
230  const CostFunction& costFunction) const {
231 
232  if (configuration().crossoverIsAdaptive) {
233  adaptCrossover();
234  }
235 
236  Array mutationProbabilities = getMutationProbabilities(population);
237 
238  std::vector<Array> crossoverMask(population.size(),
239  Array(population.front().values.size(), 1.0));
240  std::vector<Array> invCrossoverMask = crossoverMask;
241  getCrossoverMask(crossoverMask, invCrossoverMask, mutationProbabilities);
242 
243  // crossover of the old and mutant population
244  for (Size popIter = 0; popIter < population.size(); popIter++) {
245  population[popIter].values = oldPopulation[popIter].values * invCrossoverMask[popIter]
246  + mutantPopulation[popIter].values * crossoverMask[popIter];
247  // immediately apply bounds if specified
248  if (configuration().applyBounds) {
249  for (Size memIter = 0; memIter < population[popIter].values.size(); memIter++) {
250  if (population[popIter].values[memIter] > upperBound_[memIter])
251  population[popIter].values[memIter] = upperBound_[memIter]
252  + rng_.nextReal()
253  * (mirrorPopulation[popIter].values[memIter]
254  - upperBound_[memIter]);
255  if (population[popIter].values[memIter] < lowerBound_[memIter])
256  population[popIter].values[memIter] = lowerBound_[memIter]
257  + rng_.nextReal()
258  * (mirrorPopulation[popIter].values[memIter]
259  - lowerBound_[memIter]);
260  }
261  }
262  // evaluate objective function as soon as possible to avoid unnecessary loops
263  try {
264  population[popIter].cost = costFunction.value(population[popIter].values);
265  } catch (Error&) {
266  population[popIter].cost = QL_MAX_REAL;
267  }
268  }
269  }

271  void DifferentialEvolution::getCrossoverMask(
272  std::vector<Array> & crossoverMask,
273  std::vector<Array> & invCrossoverMask,
274  const Array & mutationProbabilities) const {
275  for (Size cmIter = 0; cmIter < crossoverMask.size(); cmIter++) {
276  for (Size memIter = 0; memIter < crossoverMask[cmIter].size(); memIter++) {
277  if (rng_.nextReal() < mutationProbabilities[cmIter]) {
278  invCrossoverMask[cmIter][memIter] = 0.0;
279  } else {
280  crossoverMask[cmIter][memIter] = 0.0;
281  }
282  }
283  }

286  Array DifferentialEvolution::getMutationProbabilities(
287  const std::vector<Candidate> & population) const {
288  Array mutationProbabilities = currGenCrossover_;
289  switch (configuration().crossoverType) {
290  case Normal:
291  break;
292  case Binomial:
293  mutationProbabilities = currGenCrossover_
294  * (1.0 - 1.0 / population.front().values.size())
295  + 1.0 / population.front().values.size();
296  break;
297  case Exponential:
298  for (Size coIter = 0;coIter< currGenCrossover_.size(); coIter++){
299  mutationProbabilities[coIter] =
300  (1.0 - std::pow(currGenCrossover_[coIter],
301  (int) population.front().values.size()))
302  / (population.front().values.size()
303  * (1.0 - currGenCrossover_[coIter]));
304  }
305  break;
306  default:
307  QL_FAIL("Unknown crossover type ("
308  << Integer(configuration().crossoverType) << ")");
309  break;
310  }
311  return mutationProbabilities;
312  }

319  void DifferentialEvolution::adaptSizeWeights() const {
320  // [=Fl & =Fu] respectively see Brest, J. et al., 2006,
321  // "Self-Adapting Control Parameters in Differential
322  // Evolution"
323  Real sizeWeightLowerBound = 0.1, sizeWeightUpperBound = 0.9;
324  // [=tau1] A Comparative Study on Numerical Benchmark
325  // Problems." page 649 for reference
326  Real sizeWeightChangeProb = 0.1;
327  for (Size coIter = 0;coIter < currGenSizeWeights_.size(); coIter++){
328  if (rng_.nextReal() < sizeWeightChangeProb)
329  currGenSizeWeights_[coIter] = sizeWeightLowerBound + rng_.nextReal() * sizeWeightUpperBound;
330  }
331  }

333  void DifferentialEvolution::adaptCrossover() const {
334  Real crossoverChangeProb = 0.1; // [=tau2]
335  for (Size coIter = 0;coIter < currGenCrossover_.size(); coIter++){

341  void DifferentialEvolution::fillInitialPopulation(
342  std::vector<Candidate> & population,
343  const Problem& p) const {
344 
345  // use initial values provided by the user
346  population.front().values = p.currentValue();
347  population.front().cost = p.costFunction().value(population.front().values);
348  // rest of the initial population is random
349  for (Size j = 1; j < population.size(); ++j) {
350  for (Size i = 0; i < p.currentValue().size(); ++i) {
351  Real l = lowerBound_[i], u = upperBound_[i];
352  population[j].values[i] = l + (u-l)*rng_.nextReal();
353  }
354  population[j].cost = p.costFunction().value(population[j].values);
355  }
356  }
