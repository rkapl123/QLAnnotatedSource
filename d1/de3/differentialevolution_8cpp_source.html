49  EndCriteria::Type DifferentialEvolution::minimize(Problem& p, const EndCriteria& endCriteria) {
50  EndCriteria::Type ecType;
51  p.reset();
52 
53  if (configuration().upperBound.empty()) {
54  upperBound_ = p.constraint().upperBound(p.currentValue());
55  } else {
56  QL_REQUIRE(configuration().upperBound.size() == p.currentValue().size(),
57  "wrong upper bound size in differential evolution configuration");
58  upperBound_ = configuration().upperBound;
59  }
60  if (configuration().lowerBound.empty()) {
61  lowerBound_ = p.constraint().lowerBound(p.currentValue());
62  } else {
63  QL_REQUIRE(configuration().lowerBound.size() == p.currentValue().size(),
64  "wrong lower bound size in differential evolution configuration");
65  lowerBound_ = configuration().lowerBound;
66  }
67  currGenSizeWeights_ =
68  Array(configuration().populationMembers, configuration().stepsizeWeight);
69  currGenCrossover_ = Array(configuration().populationMembers,
70  configuration().crossoverProbability);
71 
72  std::vector<Candidate> population;
73  if (!configuration().initialPopulation.empty()) {
74  population.resize(configuration().initialPopulation.size());
75  for (Size i = 0; i < population.size(); ++i) {
76  population[i].values = configuration().initialPopulation[i];
77  QL_REQUIRE(population[i].values.size() == p.currentValue().size(),
78  "wrong values size in initial population");
79  population[i].cost = p.costFunction().value(population[i].values);
80  }
81  } else {
82  population = std::vector<Candidate>(configuration().populationMembers,
83  Candidate(p.currentValue().size()));
84  fillInitialPopulation(population, p);
85  }
86 
87  std::partial_sort(population.begin(), population.begin() + 1, population.end(),
88  sort_by_cost());
89  bestMemberEver_ = population.front();
90  Real fxOld = population.front().cost;
91  Size iteration = 0, stationaryPointIteration = 0;
92 
93  // main loop - calculate consecutive emerging populations
94  while (!endCriteria.checkMaxIterations(iteration++, ecType)) {
95  calculateNextGeneration(population, p);
96  std::partial_sort(population.begin(), population.begin() + 1, population.end(),
97  sort_by_cost());
98  if (population.front().cost < bestMemberEver_.cost)
99  bestMemberEver_ = population.front();
100  Real fxNew = population.front().cost;
101  if (endCriteria.checkStationaryFunctionValue(fxOld, fxNew, stationaryPointIteration,
102  ecType))
103  break;
104  fxOld = fxNew;
105  };
106  p.setCurrentValue(bestMemberEver_.values);
107  p.setFunctionValue(bestMemberEver_.cost);
108  return ecType;
109  }

111  void DifferentialEvolution::calculateNextGeneration(
112  std::vector<Candidate>& population,
113  Problem& p) const {
114 
115  std::vector<Candidate> mirrorPopulation;
116  std::vector<Candidate> oldPopulation = population;
117 
118  switch (configuration().strategy) {
119 
120  case Rand1Standard: {
121  randomize(population.begin(), population.end(), rng_);
122  std::vector<Candidate> shuffledPop1 = population;
123  randomize(population.begin(), population.end(), rng_);
124  std::vector<Candidate> shuffledPop2 = population;
125  randomize(population.begin(), population.end(), rng_);
126  mirrorPopulation = shuffledPop1;
127 
128  for (Size popIter = 0; popIter < population.size(); popIter++) {
129  population[popIter].values = population[popIter].values
130  + configuration().stepsizeWeight
131  * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
132  }
133  }
134  break;
135 
136  case BestMemberWithJitter: {
137  randomize(population.begin(), population.end(), rng_);
138  std::vector<Candidate> shuffledPop1 = population;
139  randomize(population.begin(), population.end(), rng_);
140  Array jitter(population[0].values.size(), 0.0);
141 
142  for (Size popIter = 0; popIter < population.size(); popIter++) {
143  for (Size jitterIter = 0; jitterIter < jitter.size(); jitterIter++) {
144  jitter[jitterIter] = rng_.nextReal();
145  }
146  population[popIter].values = bestMemberEver_.values
147  + (shuffledPop1[popIter].values - population[popIter].values)
148  * (0.0001 * jitter + configuration().stepsizeWeight);
149  }
150  mirrorPopulation = std::vector<Candidate>(population.size(),
151  bestMemberEver_);
152  }
153  break;
154 
155  case CurrentToBest2Diffs: {
156  randomize(population.begin(), population.end(), rng_);
157  std::vector<Candidate> shuffledPop1 = population;
158  randomize(population.begin(), population.end(), rng_);
159 
160  for (Size popIter = 0; popIter < population.size(); popIter++) {
161  population[popIter].values = oldPopulation[popIter].values
162  + configuration().stepsizeWeight
163  * (bestMemberEver_.values - oldPopulation[popIter].values)
164  + configuration().stepsizeWeight
165  * (population[popIter].values - shuffledPop1[popIter].values);
166  }
167  mirrorPopulation = shuffledPop1;
168  }
169  break;
170 
171  case Rand1DiffWithPerVectorDither: {
172  randomize(population.begin(), population.end(), rng_);
173  std::vector<Candidate> shuffledPop1 = population;
174  randomize(population.begin(), population.end(), rng_);
175  std::vector<Candidate> shuffledPop2 = population;
176  randomize(population.begin(), population.end(), rng_);
177  mirrorPopulation = shuffledPop1;
178  Array FWeight = Array(population.front().values.size(), 0.0);
179  for (Size fwIter = 0; fwIter < FWeight.size(); fwIter++)
180  FWeight[fwIter] = (1.0 - configuration().stepsizeWeight)
181  * rng_.nextReal() + configuration().stepsizeWeight;
182  for (Size popIter = 0; popIter < population.size(); popIter++) {
183  population[popIter].values = population[popIter].values
184  + FWeight * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
185  }
186  }
187  break;
188 
189  case Rand1DiffWithDither: {
190  randomize(population.begin(), population.end(), rng_);
191  std::vector<Candidate> shuffledPop1 = population;
192  randomize(population.begin(), population.end(), rng_);
193  std::vector<Candidate> shuffledPop2 = population;
194  randomize(population.begin(), population.end(), rng_);
195  mirrorPopulation = shuffledPop1;
196  Real FWeight = (1.0 - configuration().stepsizeWeight) * rng_.nextReal()
197  + configuration().stepsizeWeight;
198  for (Size popIter = 0; popIter < population.size(); popIter++) {
199  population[popIter].values = population[popIter].values
200  + FWeight * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
201  }
202  }
203  break;
204 
205  case EitherOrWithOptimalRecombination: {
206  randomize(population.begin(), population.end(), rng_);
207  std::vector<Candidate> shuffledPop1 = population;
208  randomize(population.begin(), population.end(), rng_);
209  std::vector<Candidate> shuffledPop2 = population;
210  randomize(population.begin(), population.end(), rng_);
211  mirrorPopulation = shuffledPop1;
212  Real probFWeight = 0.5;
213  if (rng_.nextReal() < probFWeight) {
214  for (Size popIter = 0; popIter < population.size(); popIter++) {
215  population[popIter].values = oldPopulation[popIter].values
216  + configuration().stepsizeWeight
217  * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
218  }
219  } else {
220  Real K = 0.5 * (configuration().stepsizeWeight + 1); // invariant with respect to probFWeight used
221  for (Size popIter = 0; popIter < population.size(); popIter++) {
222  population[popIter].values = oldPopulation[popIter].values
223  + K
224  * (shuffledPop1[popIter].values - shuffledPop2[popIter].values
225  - 2.0 * population[popIter].values);
226  }
227  }
228  }
229  break;
230 
231  case Rand1SelfadaptiveWithRotation: {
232  randomize(population.begin(), population.end(), rng_);
233  std::vector<Candidate> shuffledPop1 = population;
234  randomize(population.begin(), population.end(), rng_);
235  std::vector<Candidate> shuffledPop2 = population;
236  randomize(population.begin(), population.end(), rng_);
237  mirrorPopulation = shuffledPop1;
238 
239  adaptSizeWeights();
240 
241  for (Size popIter = 0; popIter < population.size(); popIter++) {
242  if (rng_.nextReal() < 0.1){
243  population[popIter].values = rotateArray(bestMemberEver_.values);
244  }else {
245  population[popIter].values = bestMemberEver_.values
246  + currGenSizeWeights_[popIter]
247  * (shuffledPop1[popIter].values - shuffledPop2[popIter].values);
248  }
249  }
250  }
251  break;
252 
253  default:
254  QL_FAIL("Unknown strategy ("
255  << Integer(configuration().strategy) << ")");
256  }
257  // in order to avoid unnecessary copying we use the same population object for mutants
258  crossover(oldPopulation, population, population, mirrorPopulation, p);
259  }

261  void DifferentialEvolution::crossover(
262  const std::vector<Candidate>& oldPopulation,
263  std::vector<Candidate>& population,
264  const std::vector<Candidate>& mutantPopulation,
265  const std::vector<Candidate>& mirrorPopulation,
266  Problem& p) const {
267 
268  if (configuration().crossoverIsAdaptive) {
269  adaptCrossover();
270  }
271 
272  Array mutationProbabilities = getMutationProbabilities(population);
273 
274  std::vector<Array> crossoverMask(population.size(),
275  Array(population.front().values.size(), 1.0));
276  std::vector<Array> invCrossoverMask = crossoverMask;
277  getCrossoverMask(crossoverMask, invCrossoverMask, mutationProbabilities);
278 
279  // crossover of the old and mutant population
280  for (Size popIter = 0; popIter < population.size(); popIter++) {
281  population[popIter].values = oldPopulation[popIter].values * invCrossoverMask[popIter]
282  + mutantPopulation[popIter].values * crossoverMask[popIter];
283  // immediately apply bounds if specified
284  if (configuration().applyBounds) {
285  for (Size memIter = 0; memIter < population[popIter].values.size(); memIter++) {
286  if (population[popIter].values[memIter] > upperBound_[memIter])
287  population[popIter].values[memIter] = upperBound_[memIter]
288  + rng_.nextReal()
289  * (mirrorPopulation[popIter].values[memIter]
290  - upperBound_[memIter]);
291  if (population[popIter].values[memIter] < lowerBound_[memIter])
292  population[popIter].values[memIter] = lowerBound_[memIter]
293  + rng_.nextReal()
294  * (mirrorPopulation[popIter].values[memIter]
295  - lowerBound_[memIter]);
296  }
297  }
298  // evaluate objective function as soon as possible to avoid unnecessary loops
299  try {
300  population[popIter].cost = p.value(population[popIter].values);
301  } catch (Error&) {
302  population[popIter].cost = QL_MAX_REAL;
303  }
304  if(!boost::math::isfinite(population[popIter].cost))
305  population[popIter].cost = QL_MAX_REAL;
306 
307  }
308  }

310  void DifferentialEvolution::getCrossoverMask(
311  std::vector<Array> & crossoverMask,
312  std::vector<Array> & invCrossoverMask,
313  const Array & mutationProbabilities) const {
314  for (Size cmIter = 0; cmIter < crossoverMask.size(); cmIter++) {
315  for (Size memIter = 0; memIter < crossoverMask[cmIter].size(); memIter++) {
316  if (rng_.nextReal() < mutationProbabilities[cmIter]) {
317  invCrossoverMask[cmIter][memIter] = 0.0;
318  } else {
319  crossoverMask[cmIter][memIter] = 0.0;
320  }
321  }
322  }

325  Array DifferentialEvolution::getMutationProbabilities(
326  const std::vector<Candidate> & population) const {
327  Array mutationProbabilities = currGenCrossover_;
328  switch (configuration().crossoverType) {
329  case Normal:
330  break;
331  case Binomial:
332  mutationProbabilities = currGenCrossover_
333  * (1.0 - 1.0 / population.front().values.size())
334  + 1.0 / population.front().values.size();
335  break;
336  case Exponential:
337  for (Size coIter = 0;coIter< currGenCrossover_.size(); coIter++){
338  mutationProbabilities[coIter] =
339  (1.0 - std::pow(currGenCrossover_[coIter],
340  (int) population.front().values.size()))
341  / (population.front().values.size()
342  * (1.0 - currGenCrossover_[coIter]));
343  }
344  break;
345  default:
346  QL_FAIL("Unknown crossover type ("
347  << Integer(configuration().crossoverType) << ")");
348  break;
349  }
350  return mutationProbabilities;
351  }

358  void DifferentialEvolution::adaptSizeWeights() const {
359  // [=Fl & =Fu] respectively see Brest, J. et al., 2006,
360  // "Self-Adapting Control Parameters in Differential
361  // Evolution"
362  Real sizeWeightLowerBound = 0.1, sizeWeightUpperBound = 0.9;
363  // [=tau1] A Comparative Study on Numerical Benchmark
364  // Problems." page 649 for reference
365  Real sizeWeightChangeProb = 0.1;
366  for (Size coIter = 0;coIter < currGenSizeWeights_.size(); coIter++){
367  if (rng_.nextReal() < sizeWeightChangeProb)
368  currGenSizeWeights_[coIter] = sizeWeightLowerBound + rng_.nextReal() * sizeWeightUpperBound;
369  }

380  void DifferentialEvolution::fillInitialPopulation(
381  std::vector<Candidate> & population,
382  const Problem& p) const {
383 
384  // use initial values provided by the user
385  population.front().values = p.currentValue();
386  population.front().cost = p.costFunction().value(population.front().values);
387  // rest of the initial population is random
388  for (Size j = 1; j < population.size(); ++j) {
389  for (Size i = 0; i < p.currentValue().size(); ++i) {
390  Real l = lowerBound_[i], u = upperBound_[i];
391  population[j].values[i] = l + (u-l)*rng_.nextReal();
392  }
393  population[j].cost = p.costFunction().value(population[j].values);
394  if(!boost::math::isfinite(population[j].cost))
395  population[j].cost = QL_MAX_REAL;
396  }
397  }
