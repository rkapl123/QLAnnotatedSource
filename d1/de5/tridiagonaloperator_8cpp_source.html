62  Disposable<Array> TridiagonalOperator::applyTo(const Array& v) const {
63  QL_REQUIRE(n_!=0,
64  "uninitialized TridiagonalOperator");
65  QL_REQUIRE(v.size()==n_,
66  "vector of the wrong size " << v.size() <<
67  " instead of " << n_);
68  Array result(n_);
69  std::transform(diagonal_.begin(), diagonal_.end(),
70  v.begin(),
71  result.begin(),
72  std::multiplies<Real>());
73 
74  // matricial product
75  result[0] += upperDiagonal_[0]*v[1];
76  for (Size j=1; j<=n_-2; j++)
77  result[j] += lowerDiagonal_[j-1]*v[j-1]+
78  upperDiagonal_[j]*v[j+1];
79  result[n_-1] += lowerDiagonal_[n_-2]*v[n_-2];
80 
81  return result;
82  }

91  void TridiagonalOperator::solveFor(const Array& rhs,
92  Array& result) const {
93 
94  QL_REQUIRE(n_!=0,
95  "uninitialized TridiagonalOperator");
96  QL_REQUIRE(rhs.size()==n_,
97  "rhs vector of size " << rhs.size() <<
98  " instead of " << n_);
99 
100  Real bet = diagonal_[0];
101  QL_REQUIRE(!close(bet, 0.0),
102  "diagonal's first element (" << bet <<
103  ") cannot be close to zero");
104  result[0] = rhs[0]/bet;
105  for (Size j=1; j<=n_-1; ++j) {
106  temp_[j] = upperDiagonal_[j-1]/bet;
107  bet = diagonal_[j]-lowerDiagonal_[j-1]*temp_[j];
108  QL_ENSURE(!close(bet, 0.0), "division by zero");
109  result[j] = (rhs[j] - lowerDiagonal_[j-1]*result[j-1])/bet;
110  }
111  // cannot be j>=0 with Size j
112  for (Size j=n_-2; j>0; --j)
113  result[j] -= temp_[j+1]*result[j+1];
114  result[0] -= temp_[1]*result[1];
115  }

117  Disposable<Array> TridiagonalOperator::SOR(const Array& rhs,
118  Real tol) const {
119  QL_REQUIRE(n_!=0,
120  "uninitialized TridiagonalOperator");
121  QL_REQUIRE(rhs.size()==n_,
122  "rhs vector of size " << rhs.size() <<
123  " instead of " << n_);
124 
125  // initial guess
126  Array result = rhs;
127 
128  // solve tridiagonal system with SOR technique
129  Real omega = 1.5;
130  Real err = 2.0*tol;
131  Real temp;
132  for (Size sorIteration=0; err>tol ; ++sorIteration) {
133  QL_REQUIRE(sorIteration<100000,
134  "tolerance (" << tol << ") not reached in " <<
135  sorIteration << " iterations. " <<
136  "The error still is " << err);
137 
138  temp = omega * (rhs[0] -
139  upperDiagonal_[0] * result[1]-
140  diagonal_[0] * result[0])/diagonal_[0];
141  err = temp*temp;
142  result[0] += temp;
143  Size i;
144  for (i=1; i<n_-1 ; ++i) {
145  temp = omega *(rhs[i] -
146  upperDiagonal_[i] * result[i+1]-
147  diagonal_[i] * result[i] -
148  lowerDiagonal_[i-1] * result[i-1])/diagonal_[i];
149  err += temp * temp;
150  result[i] += temp;
151  }
152 
153  temp = omega * (rhs[i] -
154  diagonal_[i] * result[i] -
155  lowerDiagonal_[i-1] * result[i-1])/diagonal_[i];
156  err += temp*temp;
157  result[i] += temp;
158  }
159  return result;
160  }
