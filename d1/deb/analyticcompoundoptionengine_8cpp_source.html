28  // Helper Class needed to solve an implicit problem of finding a
29  // spot to a corresponding option price.
30  class ImpliedSpotHelper {
31  public:
32  ImpliedSpotHelper(DiscountFactor dividendDiscount,
33  DiscountFactor riskFreeDiscount,
34  Real standardDeviation,
35  const ext::shared_ptr<PlainVanillaPayoff>& payoff,
36  Real strike)
37  : dividendDiscount_(dividendDiscount), riskFreeDiscount_(riskFreeDiscount),
38  standardDeviation_(standardDeviation), strike_(strike), payoff_(payoff) {}
39  Real operator()(Real spot) const {
40  Real forwardPrice = spot*dividendDiscount_/riskFreeDiscount_;
41  Real value = blackFormula(payoff_, forwardPrice,
42  standardDeviation_,riskFreeDiscount_);
43  return value - strike_;
44  }
45  private:
46  DiscountFactor dividendDiscount_;
47  DiscountFactor riskFreeDiscount_;
48  Real standardDeviation_;
49  Real strike_;
50  ext::shared_ptr<PlainVanillaPayoff> payoff_;
51  };

61  void AnalyticCompoundOptionEngine::calculate() const {
62 
63  QL_REQUIRE(strikeDaughter()>0.0,
64  "Daughter strike must be positive");
65 
66  QL_REQUIRE(strikeMother()>0.0,
67  "Mother strike must be positive");
68 
69  QL_REQUIRE(spot() >= 0.0, "negative or null underlying given");
70 
71  /* Solver Setup ***************************************************/
72  Date helpDate(process_->riskFreeRate()->referenceDate());
73  Date helpMaturity=helpDate+(maturityDaughter()-maturityMother())*Days;
74  Real vol =process_->blackVolatility()->blackVol(helpMaturity,
75  strikeDaughter());
76 
77  Time helpTimeToMat=process_->time(helpMaturity);
78  vol=vol*std::sqrt(helpTimeToMat);
79 
80  DiscountFactor dividendDiscount =
81  process_->dividendYield()->discount(helpMaturity);
82 
83  DiscountFactor riskFreeDiscount =
84  process_->riskFreeRate()->discount(helpMaturity);
85 
86 
87  ext::shared_ptr<ImpliedSpotHelper> f(
88  new ImpliedSpotHelper(dividendDiscount, riskFreeDiscount,
89  vol, payoffDaughter(), strikeMother()));
90 
91  Brent solver;
92  solver.setMaxEvaluations(1000);
93  Real accuracy = 1.0e-6;
94 
95  Real sSolved=solver.solve(*f, accuracy, strikeDaughter(), 1.0e-6, strikeDaughter()*1000.0);
96  Real X=transformX(sSolved); // transform stock to return as in Wystup's book
97  /* Solver Setup Finished*****************************************/
98 
99  Real phi=typeDaughter(); // -1 or 1
100  Real w=typeMother(); // -1 or 1
101 
102  Real rho=std::sqrt(residualTimeMother()/residualTimeDaughter());
103  BivariateCumulativeNormalDistributionDr78 N2(w*rho) ;
104 
105  DiscountFactor ddD=dividendDiscountDaughter();
106  DiscountFactor rdD=riskFreeDiscountDaughter();
107  //DiscountFactor ddM=dividendDiscountMother();
108  DiscountFactor rdM=riskFreeDiscountMother();
109 
110  Real XmSM=X-stdDeviationMother();
111  Real S=spot();
112  Real dP=dPlus();
113  Real dPT12=dPlusTau12(sSolved);
114  Real vD=volatilityDaughter();
115 
116  Real dM=dMinus();
117  Real strD=strikeDaughter();
118  Real strM=strikeMother();
119  Real rTM=residualTimeMother();
120  Real rTD=residualTimeDaughter();
121 
122  Real rD=riskFreeRateDaughter();
123  Real dD=dividendRateDaughter();
124 
125  Real N2XmSM=N2(-phi*w*XmSM,phi*dP);
126  Real N2X=N2(-phi*w*X,phi*dM);
127  Real NeX=N_(-phi*w*e(X));
128  Real NX=N_(-phi*w*X);
129  Real NT12=N_(phi*dPT12);
130  Real ndP=n_(dP);
131  Real nXm=n_(XmSM);
132  Real invMTime=1/std::sqrt(rTM);
133  Real invDTime=1/std::sqrt(rTD);
134 
135  Real tempRes=phi*w*S*ddD*N2XmSM-phi*w*strD*rdD*N2X-w*strM*rdM*NX;
136  Real tempDelta=phi*w*ddD*N2XmSM;
137  Real tempGamma=(ddD/(vD*S))*(invMTime*nXm*NT12+w*invDTime*ndP*NeX);
138  Real tempVega=ddD*S*((1/invMTime)*nXm*NT12+w*(1/invDTime)*ndP*NeX);
139  Real tempTheta=phi*w*dD*S*ddD*N2XmSM-phi*w*rD*strD*rdD*N2X-w*rD*strM*rdM*NX;
140  tempTheta-=0.5*vD*S*ddD*(invMTime*nXm*NT12+w*invDTime*ndP*NeX);
141 
142  results_.value=tempRes;
143  results_.delta=tempDelta;
144  results_.gamma=tempGamma;
145  results_.vega=tempVega;
146  results_.theta=tempTheta;
147  }

248  Real AnalyticCompoundOptionEngine::dPlus() const {
249  Real forward = spot() * dividendDiscountDaughter() / riskFreeDiscountDaughter();

280  Real AnalyticCompoundOptionEngine::transformX(Real X) const {
281 
282  Real sd=stdDeviationMother();
283  Real resX=riskFreeDiscountMother()*X/(spot()*dividendDiscountMother());
284  resX=resX*std::exp(0.5*sd*sd);
285  resX=std::log(resX);
286 
287  return resX/sd;
288  }

290  Real AnalyticCompoundOptionEngine::e(Real X) const {
