28  // Helper Class needed to solve an implicit problem of finding a
29  // spot to a corresponding option price.
30  class ImpliedSpotHelper {
31  public:
32  ImpliedSpotHelper(DiscountFactor dividendDiscount,
33  DiscountFactor riskFreeDiscount,
34  Real standardDeviation ,
35  boost::shared_ptr<PlainVanillaPayoff> payoff,
36  Real strike)
37  : dividendDiscount_(dividendDiscount),
38  riskFreeDiscount_(riskFreeDiscount),
39  standardDeviation_(standardDeviation),
40  strike_(strike),payoff_(payoff) {}
41  Real operator()(Real spot) const {
42  Real forwardPrice = spot*dividendDiscount_/riskFreeDiscount_;
43  Real value = blackFormula(payoff_, forwardPrice,
44  standardDeviation_,riskFreeDiscount_);
45  return value - strike_;
46  }
47  private:
48  DiscountFactor dividendDiscount_;
49  DiscountFactor riskFreeDiscount_;
50  Real standardDeviation_;
51  Real strike_;
52  boost::shared_ptr<PlainVanillaPayoff> payoff_;
53  };

63  void AnalyticCompoundOptionEngine::calculate() const {
64 
65  QL_REQUIRE(strikeDaughter()>0.0,
66  "Daughter strike must be positive");
67 
68  QL_REQUIRE(strikeMother()>0.0,
69  "Mother strike must be positive");
70 
71  QL_REQUIRE(spot() >= 0.0, "negative or null underlying given");
72 
73  /* Solver Setup ***************************************************/
74  Date helpDate(process_->riskFreeRate()->referenceDate());
75  Date helpMaturity=helpDate+(maturityDaughter()-maturityMother())*Days;
76  Real vol =process_->blackVolatility()->blackVol(helpMaturity,
77  strikeDaughter());
78 
79  Time helpTimeToMat=process_->time(helpMaturity);
80  vol=vol*std::sqrt(helpTimeToMat);
81 
82  DiscountFactor dividendDiscount =
83  process_->dividendYield()->discount(helpMaturity);
84 
85  DiscountFactor riskFreeDiscount =
86  process_->riskFreeRate()->discount(helpMaturity);
87 
88 
89  boost::shared_ptr<ImpliedSpotHelper> f(
90  new ImpliedSpotHelper(dividendDiscount, riskFreeDiscount,
91  vol, payoffDaughter(), strikeMother()));
92 
93  Brent solver;
94  solver.setMaxEvaluations(1000);
95  Real accuracy = 1.0e-6;
96 
97  Real sSolved=solver.solve(*f, accuracy, strikeDaughter(), 1.0e-6, strikeDaughter()*1000.0);
98  Real X=transformX(sSolved); // transform stock to return as in Wystup's book
99  /* Solver Setup Finished*****************************************/
100 
101  Real phi=typeDaughter(); // -1 or 1
102  Real w=typeMother(); // -1 or 1
103 
104  Real rho=std::sqrt(residualTimeMother()/residualTimeDaughter());
105  BivariateCumulativeNormalDistributionDr78 N2(w*rho) ;
106 
107  DiscountFactor ddD=dividendDiscountDaughter();
108  DiscountFactor rdD=riskFreeDiscountDaughter();
109  //DiscountFactor ddM=dividendDiscountMother();
110  DiscountFactor rdM=riskFreeDiscountMother();
111 
112  Real XmSM=X-stdDeviationMother();
113  Real S=spot();
114  Real dP=dPlus();
115  Real dPT12=dPlusTau12(sSolved);
116  Real vD=volatilityDaughter();
117 
118  Real dM=dMinus();
119  Real strD=strikeDaughter();
120  Real strM=strikeMother();
121  Real rTM=residualTimeMother();
122  Real rTD=residualTimeDaughter();
123 
124  Real rD=riskFreeRateDaughter();
125  Real dD=dividendRateDaughter();
126 
127  Real N2XmSM=N2(-phi*w*XmSM,phi*dP);
128  Real N2X=N2(-phi*w*X,phi*dM);
129  Real NeX=N_(-phi*w*e(X));
130  Real NX=N_(-phi*w*X);
131  Real NT12=N_(phi*dPT12);
132  Real ndP=n_(dP);
133  Real nXm=n_(XmSM);
134  Real invMTime=1/std::sqrt(rTM);
135  Real invDTime=1/std::sqrt(rTD);
136 
137  Real tempRes=phi*w*S*ddD*N2XmSM-phi*w*strD*rdD*N2X-w*strM*rdM*NX;
138  Real tempDelta=phi*w*ddD*N2XmSM;
139  Real tempGamma=(ddD/(vD*S))*(invMTime*nXm*NT12+w*invDTime*ndP*NeX);
140  Real tempVega=ddD*S*((1/invMTime)*nXm*NT12+w*(1/invDTime)*ndP*NeX);
141  Real tempTheta=phi*w*dD*S*ddD*N2XmSM-phi*w*rD*strD*rdD*N2X-w*rD*strM*rdM*NX;
142  tempTheta-=0.5*vD*S*ddD*(invMTime*nXm*NT12+w*invDTime*ndP*NeX);
143 
144  results_.value=tempRes;
145  results_.delta=tempDelta;
146  results_.gamma=tempGamma;
147  results_.vega=tempVega;
148  results_.theta=tempTheta;
149  }

282  Real AnalyticCompoundOptionEngine::transformX(Real X) const {
283 
284  Real sd=stdDeviationMother();
285  Real resX=riskFreeDiscountMother()*X/(spot()*dividendDiscountMother());
286  resX=resX*std::exp(0.5*sd*sd);
287  resX=std::log(resX);
288 
289  return resX/sd;
290  }

292  Real AnalyticCompoundOptionEngine::e(Real X) const {
