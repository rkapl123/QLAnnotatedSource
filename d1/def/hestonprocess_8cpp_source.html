73  Disposable<Array> HestonProcess::drift(Time t, const Array& x) const {
74  Array tmp(2);
75  const Real vol = (x[1] > 0.0) ? std::sqrt(x[1])
76  : (discretization_ == Reflection) ? - std::sqrt(-x[1])
77  : 0.0;
78 
79  tmp[0] = riskFreeRate_->forwardRate(t, t, Continuous)
80  - dividendYield_->forwardRate(t, t, Continuous)
81  - 0.5 * vol * vol;
82 
83  tmp[1] = kappa_*
84  (theta_-((discretization_==PartialTruncation) ? x[1] : vol*vol));
85  return tmp;
86  }

88  Disposable<Matrix> HestonProcess::diffusion(Time, const Array& x) const {
89  /* the correlation matrix is
90  | 1 rho |
91  | rho 1 |
92  whose square root (which is used here) is
93  | 1 0 |
94  | rho sqrt(1-rho^2) |
95  */
96  Matrix tmp(2,2);
97  const Real vol = (x[1] > 0.0) ? std::sqrt(x[1])
98  : (discretization_ == Reflection) ? -std::sqrt(-x[1])
99  : 1e-8; // set vol to (almost) zero but still
100  // expose some correlation information
101  const Real sigma2 = sigma_ * vol;
102  const Real sqrhov = std::sqrt(1.0 - rho_*rho_);
103 
104  tmp[0][0] = vol; tmp[0][1] = 0.0;
105  tmp[1][0] = rho_*sigma2; tmp[1][1] = sqrhov*sigma2;
106  return tmp;
107  }

118  // This is the continuous version of a characteristic function
119  // for the exact sampling of the Heston process, s. page 8, formula 13,
120  // M. Broadie, O. Kaya, Exact Simulation of Stochastic Volatility and
121  // other Affine Jump Diffusion Processes
122  // http://finmath.stanford.edu/seminars/documents/Broadie.pdf
123  //
124  // This version does not need a branch correction procedure.
125  // For details please see:
126  // Roger Lord, "Efficient Pricing Algorithms for exotic Derivatives",
127  // http://repub.eur.nl/pub/13917/LordR-Thesis.pdf
128  std::complex<Real> Phi(const HestonProcess& process,
129  const std::complex<Real>& a,
130  Real nu_0, Real nu_t, Time dt) {
131  const Real theta = process.theta();
132  const Real kappa = process.kappa();
133  const Real sigma = process.sigma();
134 
135  const Volatility sigma2 = sigma*sigma;
136  const std::complex<Real> ga = std::sqrt(
137  kappa*kappa - 2*sigma2*a*std::complex<Real>(0.0, 1.0));
138  const Real d = 4*theta*kappa/sigma2;
139 
140  const Real nu = 0.5*d-1;
141  const std::complex<Real> z
142  = ga*std::exp(-0.5*ga*dt)/(1.0-std::exp(-ga*dt));
143  const std::complex<Real> log_z
144  = -0.5*ga*dt + std::log(ga/(1.0-std::exp(-ga*dt)));
145 
146  const std::complex<Real> alpha
147  = 4.0*ga*std::exp(-0.5*ga*dt)/(sigma2*(1.0-std::exp(-ga*dt)));
148  const std::complex<Real> beta = 4.0*kappa*std::exp(-0.5*kappa*dt)
149  /(sigma2*(1.0-std::exp(-kappa*dt)));
150 
151  return ga*std::exp(-0.5*(ga-kappa)*dt)*(1-std::exp(-kappa*dt))
152  / (kappa*(1.0-std::exp(-ga*dt)))
153  *std::exp((nu_0+nu_t)/sigma2 * (
154  kappa*(1.0+std::exp(-kappa*dt))/(1.0-std::exp(-kappa*dt))
155  - ga*(1.0+std::exp(-ga*dt))/(1.0-std::exp(-ga*dt))))
156  *std::exp(nu*log_z)/std::pow(z, nu)
157  *((nu_t > 1e-8)
158  ? modifiedBesselFunction_i(
159  nu, std::sqrt(nu_0*nu_t)*alpha)
160  / modifiedBesselFunction_i(
161  nu, std::sqrt(nu_0*nu_t)*beta)
162  : std::pow(alpha/beta, nu)
163  );
164  }

177  Real int_ph(const HestonProcess& process,
178  Real a, Real x, Real y, Real nu_0, Real nu_t, Time t) {
179  static const GaussLaguerreIntegration gaussLaguerreIntegration(128);
180 
181  const Real rho = process.rho();
182  const Real kappa = process.kappa();
183  const Real sigma = process.sigma();
184  const Real x0 = std::log(process.s0()->value());
185 
186  return gaussLaguerreIntegration(
187  [&](Real u){ return ph(process, y, u, nu_0, nu_t, t); })
188  / std::sqrt(2*M_PI*(1-rho*rho)*y)
189  * std::exp(-0.5*square<Real>()( x - x0 - a
190  + y*(0.5-rho*kappa/sigma))
191  /(y*(1-rho*rho)));
192  }

195  Real pade(Real x, const Real* nominator, const Real* denominator, Size m) {
196  Real n=0.0, d=0.0;
197  for (Integer i=m-1; i >= 0; --i) {
198  n = (n+nominator[i])*x;
199  d = (d+denominator[i])*x;
200  }
201  return (1+n)/(1+d);
202  }

205  // http://wikipedia.org/wiki/Sine_integral#Sine_integral
206  Real Si(Real x) {
207  if (x <=4.0) {
208  const Real n[] =
209  { -4.54393409816329991e-2,1.15457225751016682e-3,
210  -1.41018536821330254e-5,9.43280809438713025e-8,
211  -3.53201978997168357e-10,7.08240282274875911e-13,
212  -6.05338212010422477e-16 };

214  { 1.01162145739225565e-2,4.99175116169755106e-5,

223  const Real fn[] =
224  { 7.44437068161936700618e2,1.96396372895146869801e5,

255  Real cornishFisherEps(const HestonProcess& process,
256  Real nu_0, Real nu_t, Time dt, Real eps) {
257  // use moment generating function to get the
258  // first,second, third and fourth moment of the distribution
259  const Real d = 1e-2;
260  const Real p2 = Phi(process, std::complex<Real>(0, -2*d),
261  nu_0, nu_t, dt).real();
262  const Real p1 = Phi(process, std::complex<Real>(0, -d),
263  nu_0, nu_t, dt).real();
264  const Real p0 = Phi(process, std::complex<Real>(0, 0),
265  nu_0, nu_t, dt).real();
266  const Real pm1= Phi(process, std::complex<Real>(0, d),
267  nu_0, nu_t, dt).real();
268  const Real pm2= Phi(process, std::complex<Real>(0, 2*d),
269  nu_0, nu_t, dt).real();
270 
271  const Real avg = (pm2-8*pm1+8*p1-p2)/(12*d);
272  const Real m2 = (-pm2+16*pm1-30*p0+16*p1-p2)/(12*d*d);
273  const Real var = m2 - avg*avg;
274  const Real stdDev = std::sqrt(var);
275 
276  const Real m3 = (-0.5*pm2 + pm1 - p1 + 0.5*p2)/(d*d*d);
277  const Real skew
278  = (m3 - 3*var*avg - avg*avg*avg) / (var*stdDev);
279 
280  const Real m4 = (pm2 - 4*pm1 + 6*p0 - 4*p1 + p2)/(d*d*d*d);
281  const Real kurt
282  = (m4 - 4*m3*avg + 6*m2*avg*avg - 3*avg*avg*avg*avg)
283  /(var*var);
284 
285  // Cornish-Fisher relation to come up with an improved
286  // estimate of 1-F(u_\eps) < \eps
287  const Real q = InverseCumulativeNormal()(1-eps);
288  const Real w = q + (q*q-1)/6*skew + (q*q*q-3*q)/24*(kurt-3)
289  - (2*q*q*q-5*q)/36*skew*skew;
290 
291  return avg + w*stdDev;
292  }

294  Real cdf_nu_ds(const HestonProcess& process,
295  Real x, Real nu_0, Real nu_t, Time dt,
296  HestonProcess::Discretization discretization) {
297  const Real eps = 1e-4;
298  const Real u_eps = std::min(100.0,
299  std::max(0.1, cornishFisherEps(process, nu_0, nu_t, dt, eps)));
300 
301  switch (discretization) {
302  case HestonProcess::BroadieKayaExactSchemeLaguerre:
303  {
304  static const GaussLaguerreIntegration
305  gaussLaguerreIntegration(128);
306 
307  // get the upper bound for the integration
308  Real upper = u_eps/2.0;
309  while (std::abs(Phi(process,upper,nu_0,nu_t,dt)/upper)
310  > eps) upper*=2.0;
311 
312  return (x < upper)
313  ? std::max(0.0, std::min(1.0,
314  gaussLaguerreIntegration(
315  [&](Real u){ return ch(process, x, u, nu_0, nu_t, dt); })))
316  : 1.0;
317  }
318  case HestonProcess::BroadieKayaExactSchemeLobatto:
319  {
320  // get the upper bound for the integration
321  Real upper = u_eps/2.0;
322  while (std::abs(Phi(process, upper,nu_0,nu_t,dt)/upper)
323  > eps) upper*=2.0;
324 
325  return (x < upper)
326  ? std::max(0.0, std::min(1.0,
327  GaussLobattoIntegral(Null<Size>(), eps)(
328  [&](Real xi){ return ch(process, x, xi, nu_0, nu_t, dt); },
329  QL_EPSILON, upper)))
330  : 1.0;
331  }
332  case HestonProcess::BroadieKayaExactSchemeTrapezoidal:
333  {
334  const Real h = 0.05;
335 
336  Real si = Si(0.5*h*x);
337  Real s = M_2_PI*si;
338  std::complex<Real> f;
339  Size j = 0;
340  do {
341  ++j;
342  const Real u = h*j;
343  const Real si_n = Si(x*(u+0.5*h));
344 
345  f = Phi(process, u, nu_0, nu_t, dt);
346  s+= M_2_PI*f.real()*(si_n-si);
347  si = si_n;
348  }
349  while (M_2_PI*std::abs(f)/j > eps);
350 
351  return s;
352  }
353  default:
354  QL_FAIL("unknown integration method");
355  }
356  }
357  }

366  Real HestonProcess::pdf(Real x, Real v, Time t, Real eps) const {
367  const Real k = sigma_*sigma_*(1-std::exp(-kappa_*t))/(4*kappa_);
368  const Real a = std::log( dividendYield_->discount(t)
369  / riskFreeRate_->discount(t))
370  + rho_/sigma_*(v - v0_ - kappa_*theta_*t);
371 
372  const Real x0 = std::log(s0()->value());
373  Real upper = std::max(0.1, -(x-x0-a)/(0.5-rho_*kappa_/sigma_)), f=0, df=1;
374 
375  while (df > 0.0 || f > 0.1*eps) {
376  const Real f1 = x-x0-a+upper*(0.5-rho_*kappa_/sigma_);
377  const Real f2 = -0.5*f1*f1/(upper*(1-rho_*rho_));
378 
379  df = 1/std::sqrt(2*M_PI*(1-rho_*rho_))
380  * ( -0.5/(upper*std::sqrt(upper))*std::exp(f2)
381  + 1/std::sqrt(upper)*std::exp(f2)*(-0.5/(1-rho_*rho_))
382  *(-1/(upper*upper)*f1*f1
383  + 2/upper*f1*(0.5-rho_*kappa_/sigma_)));
384 
385  f = std::exp(f2)/ std::sqrt(2*M_PI*(1-rho_*rho_)*upper);
386  upper*=1.5;
387  }
388 
389  upper = 2.0*cornishFisherEps(*this, v0_, v, t,1e-3);
390 
391  return SegmentIntegral(100)(
392  [&](Real xi){ return int_ph(*this, a, x, xi, v0_, v, t); },
393  QL_EPSILON, upper)
394  * boost::math::pdf(
395  boost::math::non_central_chi_squared_distribution<Real>(
396  4*theta_*kappa_/(sigma_*sigma_),
397  4*kappa_*std::exp(-kappa_*t)
398  /((sigma_*sigma_)*(1-std::exp(-kappa_*t)))*v0_),
399  v/k) / k;
400  }

402  Disposable<Array> HestonProcess::evolve(Time t0, const Array& x0,
403  Time dt, const Array& dw) const {
404  Array retVal(2);
405  Real vol, vol2, mu, nu, dy;
406 
407  const Real sdt = std::sqrt(dt);
408  const Real sqrhov = std::sqrt(1.0 - rho_*rho_);
409 
410  switch (discretization_) {
411  // For the definition of PartialTruncation, FullTruncation
412  // and Reflection see Lord, R., R. Koekkoek and D. van Dijk (2006),
413  // "A Comparison of biased simulation schemes for
414  // stochastic volatility models",
415  // Working Paper, Tinbergen Institute
416  case PartialTruncation:
417  vol = (x0[1] > 0.0) ? std::sqrt(x0[1]) : 0.0;
418  vol2 = sigma_ * vol;
419  mu = riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
420  - dividendYield_->forwardRate(t0, t0+dt, Continuous)
421  - 0.5 * vol * vol;
422  nu = kappa_*(theta_ - x0[1]);
423 
424  retVal[0] = x0[0] * std::exp(mu*dt+vol*dw[0]*sdt);
425  retVal[1] = x0[1] + nu*dt + vol2*sdt*(rho_*dw[0] + sqrhov*dw[1]);
426  break;
427  case FullTruncation:
428  vol = (x0[1] > 0.0) ? std::sqrt(x0[1]) : 0.0;
429  vol2 = sigma_ * vol;
430  mu = riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
431  - dividendYield_->forwardRate(t0, t0+dt, Continuous)
432  - 0.5 * vol * vol;
433  nu = kappa_*(theta_ - vol*vol);
434 
435  retVal[0] = x0[0] * std::exp(mu*dt+vol*dw[0]*sdt);
436  retVal[1] = x0[1] + nu*dt + vol2*sdt*(rho_*dw[0] + sqrhov*dw[1]);
437  break;
438  case Reflection:
439  vol = std::sqrt(std::fabs(x0[1]));
440  vol2 = sigma_ * vol;
441  mu = riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
442  - dividendYield_->forwardRate(t0, t0+dt, Continuous)
443  - 0.5 * vol*vol;
444  nu = kappa_*(theta_ - vol*vol);
445 
446  retVal[0] = x0[0]*std::exp(mu*dt+vol*dw[0]*sdt);
447  retVal[1] = vol*vol
448  +nu*dt + vol2*sdt*(rho_*dw[0] + sqrhov*dw[1]);
449  break;
450  case NonCentralChiSquareVariance:
451  // use Alan Lewis trick to decorrelate the equity and the variance
452  // process by using y(t)=x(t)-\frac{rho}{sigma}\nu(t)
453  // and Ito's Lemma. Then use exact sampling for the variance
454  // process. For further details please read the Wilmott thread
455  // "QuantLib code is very high quality"
456  vol = (x0[1] > 0.0) ? std::sqrt(x0[1]) : 0.0;
457  mu = riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
458  - dividendYield_->forwardRate(t0, t0+dt, Continuous)
459  - 0.5 * vol*vol;
460 
461  retVal[1] = varianceDistribution(x0[1], dw[1], dt);
462  dy = (mu - rho_/sigma_*kappa_
463  *(theta_-vol*vol)) * dt + vol*sqrhov*dw[0]*sdt;
464 
465  retVal[0] = x0[0]*std::exp(dy + rho_/sigma_*(retVal[1]-x0[1]));
466  break;
467  case QuadraticExponential:
468  case QuadraticExponentialMartingale:
469  {
470  // for details of the quadratic exponential discretization scheme
471  // see Leif Andersen,
472  // Efficient Simulation of the Heston Stochastic Volatility Model
473  const Real ex = std::exp(-kappa_*dt);
474 
475  const Real m = theta_+(x0[1]-theta_)*ex;
476  const Real s2 = x0[1]*sigma_*sigma_*ex/kappa_*(1-ex)
477  + theta_*sigma_*sigma_/(2*kappa_)*(1-ex)*(1-ex);
478  const Real psi = s2/(m*m);
479 
480  const Real g1 = 0.5;
481  const Real g2 = 0.5;
482  Real k0 = -rho_*kappa_*theta_*dt/sigma_;
483  const Real k1 = g1*dt*(kappa_*rho_/sigma_-0.5)-rho_/sigma_;
484  const Real k2 = g2*dt*(kappa_*rho_/sigma_-0.5)+rho_/sigma_;
485  const Real k3 = g1*dt*(1-rho_*rho_);
486  const Real k4 = g2*dt*(1-rho_*rho_);
487  const Real A = k2+0.5*k4;
488 
489  if (psi < 1.5) {
490  const Real b2 = 2/psi-1+std::sqrt(2/psi*(2/psi-1));
491  const Real b = std::sqrt(b2);
492  const Real a = m/(1+b2);
493 
494  if (discretization_ == QuadraticExponentialMartingale) {
495  // martingale correction
496  QL_REQUIRE(A < 1/(2*a), "illegal value");
497  k0 = -A*b2*a/(1-2*A*a)+0.5*std::log(1-2*A*a)
498  -(k1+0.5*k3)*x0[1];
499  }
500  retVal[1] = a*(b+dw[1])*(b+dw[1]);
501  }
502  else {
503  const Real p = (psi-1)/(psi+1);
504  const Real beta = (1-p)/m;
505 
506  const Real u = CumulativeNormalDistribution()(dw[1]);
507 
508  if (discretization_ == QuadraticExponentialMartingale) {
509  // martingale correction
510  QL_REQUIRE(A < beta, "illegal value");
511  k0 = -std::log(p+beta*(1-p)/(beta-A))-(k1+0.5*k3)*x0[1];
512  }
513  retVal[1] = ((u <= p) ? 0.0 : std::log((1-p)/(1-u))/beta);
514  }
515 
516  mu = riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
517  - dividendYield_->forwardRate(t0, t0+dt, Continuous);
518 
519  retVal[0] = x0[0]*std::exp(mu*dt + k0 + k1*x0[1] + k2*retVal[1]
520  +std::sqrt(k3*x0[1]+k4*retVal[1])*dw[0]);
521  }
522  break;
523  case BroadieKayaExactSchemeLobatto:
524  case BroadieKayaExactSchemeLaguerre:
525  case BroadieKayaExactSchemeTrapezoidal:
526  {
527  const Real nu_0 = x0[1];
528  const Real nu_t = varianceDistribution(nu_0, dw[1], dt);
529 
530  const Real x = std::min(1.0-QL_EPSILON,
531  std::max(0.0, CumulativeNormalDistribution()(dw[2])));
532 
533  const Real vds = Brent().solve(
534  [&](Real xi){ return cdf_nu_ds_minus_x(*this, xi, nu_0, nu_t, dt, discretization_, x); },
535  1e-5, theta_*dt, 0.1*theta_*dt);
536 
537  const Real vdw
538  = (nu_t - nu_0 - kappa_*theta_*dt + kappa_*vds)/sigma_;
539 
540  mu = ( riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
541  -dividendYield_->forwardRate(t0, t0+dt, Continuous))*dt
542  - 0.5*vds + rho_*vdw;
543 
544  const Volatility sig = std::sqrt((1-rho_*rho_)*vds);
545  const Real s = x0[0]*std::exp(mu + sig*dw[0]);
546 
547  retVal[0] = s;
548  retVal[1] = nu_t;
549  }
550  break;
551  default:
552  QL_FAIL("unknown discretization schema");
553  }
554 
555  return retVal;
556  }
