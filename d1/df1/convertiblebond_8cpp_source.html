197  void ConvertibleBond::option::setupArguments(
198  PricingEngine::arguments* args) const {
199 
200  OneAssetOption::setupArguments(args);
201 
202  ConvertibleBond::option::arguments* moreArgs =
203  dynamic_cast<ConvertibleBond::option::arguments*>(args);
204  QL_REQUIRE(moreArgs != 0, "wrong argument type");
205 
206  moreArgs->conversionRatio = conversionRatio_;
207 
208  Date settlement = bond_->settlementDate();
209 
210  Size n = callability_.size();
211  moreArgs->callabilityDates.clear();
212  moreArgs->callabilityTypes.clear();
213  moreArgs->callabilityPrices.clear();
214  moreArgs->callabilityTriggers.clear();
215  moreArgs->callabilityDates.reserve(n);
216  moreArgs->callabilityTypes.reserve(n);
217  moreArgs->callabilityPrices.reserve(n);
218  moreArgs->callabilityTriggers.reserve(n);
219  for (Size i=0; i<n; i++) {
220  if (!callability_[i]->hasOccurred(settlement, false)) {
221  moreArgs->callabilityTypes.push_back(callability_[i]->type());
222  moreArgs->callabilityDates.push_back(callability_[i]->date());
223  moreArgs->callabilityPrices.push_back(
224  callability_[i]->price().amount());
225  if (callability_[i]->price().type() ==
226  Callability::Price::Clean)
227  moreArgs->callabilityPrices.back() +=
228  bond_->accruedAmount(callability_[i]->date());
229  boost::shared_ptr<SoftCallability> softCall =
230  boost::dynamic_pointer_cast<SoftCallability>(
231  callability_[i]);
232  if (softCall)
233  moreArgs->callabilityTriggers.push_back(
234  softCall->trigger());
235  else
236  moreArgs->callabilityTriggers.push_back(Null<Real>());
237  }
238  }
239 
240  const Leg& cashflows = bond_->cashflows();
241 
242  moreArgs->couponDates.clear();
243  moreArgs->couponAmounts.clear();
244  for (Size i=0; i<cashflows.size()-1; i++) {
245  if (!cashflows[i]->hasOccurred(settlement, false)) {
246  moreArgs->couponDates.push_back(cashflows[i]->date());
247  moreArgs->couponAmounts.push_back(cashflows[i]->amount());
248  }
249  }
250 
251  moreArgs->dividends.clear();
252  moreArgs->dividendDates.clear();
253  for (Size i=0; i<dividends_.size(); i++) {
254  if (!dividends_[i]->hasOccurred(settlement, false)) {
255  moreArgs->dividends.push_back(dividends_[i]);
256  moreArgs->dividendDates.push_back(dividends_[i]->date());
257  }
258  }
259 
260  moreArgs->creditSpread = creditSpread_;
261  moreArgs->issueDate = issueDate_;
262  moreArgs->settlementDate = settlement;
263  moreArgs->settlementDays = settlementDays_;
264  moreArgs->redemption = redemption_;
265  }

268  void ConvertibleBond::option::arguments::validate() const {
269 
270  OneAssetOption::arguments::validate();
271 
272  QL_REQUIRE(conversionRatio != Null<Real>(), "null conversion ratio");
273  QL_REQUIRE(conversionRatio > 0.0,
274  "positive conversion ratio required: "
275  << conversionRatio << " not allowed");
276 
277  QL_REQUIRE(redemption != Null<Real>(), "null redemption");
278  QL_REQUIRE(redemption >= 0.0,
279  "positive redemption required: "
280  << redemption << " not allowed");
281 
282  QL_REQUIRE(settlementDate != Date(), "null settlement date");
283 
284  QL_REQUIRE(settlementDays != Null<Natural>(), "null settlement days");
285 
286  QL_REQUIRE(callabilityDates.size() == callabilityTypes.size(),
287  "different number of callability dates and types");
288  QL_REQUIRE(callabilityDates.size() == callabilityPrices.size(),
289  "different number of callability dates and prices");
290  QL_REQUIRE(callabilityDates.size() == callabilityTriggers.size(),
291  "different number of callability dates and triggers");
292 
293  QL_REQUIRE(couponDates.size() == couponAmounts.size(),
294  "different number of coupon dates and amounts");
295  }
