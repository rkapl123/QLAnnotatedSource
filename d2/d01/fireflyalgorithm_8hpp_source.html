120  class FireflyAlgorithm::Intensity {
121  friend class FireflyAlgorithm;
122  public:
123  virtual ~Intensity() {}
125  void findBrightest();
126  protected:
127  Size Mfa_, N_;
128  const std::vector<Array> *x_;
129  const std::vector<std::pair<Real, Size> > *values_;
130  std::vector<Array> *xI_;
131 
132  virtual Real intensityImpl(Real valueX, Real valueY, Real distance) = 0;
133  inline Real distance(const Array& x, const Array& y) const {
134  Real d = 0.0;
135  for (Size i = 0; i < N_; i++) {
136  Real diff = x[i] - y[i];
137  d += diff*diff;
138  }
139  return d;
140  }
141 
142  private:
143  void init(FireflyAlgorithm *fa) {
144  x_ = &fa->x_;
145  xI_ = &fa->xI_;
146  values_ = &fa->values_;
147  Mfa_ = fa->Mfa_;
148  N_ = fa->N_;
149  }
150  };

158  : beta0_(beta0), betaMin_(betaMin), gamma_(gamma) {}
159  protected:
160  Real intensityImpl(Real valueX, Real valueY, Real d){
161  return (beta0_ - betaMin_)*std::exp(-gamma_*d) + betaMin_;

213  /* Random walk given by distribution template parameter. The
214  distribution must be compatible with boost's Random
215  variate_generator

217  template <class Distribution>
218  class DistributionRandomWalk : public FireflyAlgorithm::RandomWalk {
219  public:
220  typedef IsotropicRandomWalk<Distribution, base_generator_type> WalkRandom;
221  explicit DistributionRandomWalk(Distribution dist,
222  Real delta = 0.9,
223  unsigned long seed = SeedGenerator::instance().get())
224  : walkRandom_(base_generator_type(seed), dist, 1, Array(1, 1.0), seed),
225  delta_(delta) {}
226  protected:
227  void walkImpl(Array & xRW) {
228  walkRandom_.nextReal(&xRW[0]);
229  xRW *= delta_;
230  }

232  FireflyAlgorithm::RandomWalk::init(fa);
233  walkRandom_.setDimension(N_, *lX_, *uX_);
234  }

265  /* Gaussian random walk, but size of step decreases with each iteration step
266  */
267  class DecreasingGaussianWalk : public GaussianWalk {
268  public:
269  explicit DecreasingGaussianWalk(
270  Real sigma,
271  Real delta = 0.9,
272  unsigned long seed = SeedGenerator::instance().get())
273  : GaussianWalk(sigma, delta, seed), delta0_(delta) {}
274  protected:
275  void walkImpl(Array & xRW) {
276  iteration_++;
277  if (iteration_ > Mfa_) {
278  //Every time all the fireflies have been processed
279  //multiply delta by itself
280  iteration_ = 0;
281  delta_ *= delta_;
282  }
283  GaussianWalk::walkImpl(xRW);
284  }
285  void init(FireflyAlgorithm *fa) {
286  GaussianWalk::init(fa);
287  iteration_ = 0;
288  delta_ = delta0_;
289  }
290  private:
291  Real delta0_;
292  Size iteration_;
293  };
