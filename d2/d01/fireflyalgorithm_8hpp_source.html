126  virtual Real intensityImpl(Real valueX, Real valueY, Real distance) = 0;
127  inline Real distance(const Array& x, const Array& y) {
128  Real d = 0.0;
129  for (Size i = 0; i < N_; i++) {
130  Real diff = x[i] - y[i];
131  d += diff*diff;
132  }
133  return d;
134  }
135  private:

137  x_ = &fa->x_;
138  xI_ = &fa->xI_;
139  values_ = &fa->values_;
140  Mfa_ = fa->Mfa_;

142  }

211  class DistributionRandomWalk : public FireflyAlgorithm::RandomWalk {
212  public:
213  typedef IsotropicRandomWalk<Distribution, base_generator_type> WalkRandom;
214  DistributionRandomWalk(Distribution dist,
215  Real delta = 0.9,
216  unsigned long seed = SeedGenerator::instance().get()) :
217  walkRandom_(base_generator_type(seed), dist, 1, Array(1, 1.0), seed),
218  delta_(delta) {}

221  walkRandom_.nextReal(&xRW[0]);
222  xRW *= delta_;
223  }

257  /* Gaussian random walk, but size of step decreases with each iteration step
258  */
259  class DecreasingGaussianWalk : public GaussianWalk {
260  public:
261  DecreasingGaussianWalk(Real sigma, Real delta = 0.9, unsigned long seed = SeedGenerator::instance().get()):
262  GaussianWalk(sigma, delta, seed), delta0_(delta){}
263  protected:
264  void walkImpl(Array & xRW) {
265  iteration_++;
266  if (iteration_ > Mfa_) {
267  //Every time all the fireflies have been processed
268  //multiply delta by itself
269  iteration_ = 0;
270  delta_ *= delta_;
271  }
272  GaussianWalk::walkImpl(xRW);
273  }

275  GaussianWalk::init(fa);
276  iteration_ = 0;
277  delta_ = delta0_;
