149  Rate ZeroInflationTermStructure::zeroRate(const Date &d, const Period& instObsLag,
150  bool forceLinearInterpolation,
151  bool extrapolate) const {
152 
153  Period useLag = instObsLag;
154  if (instObsLag == Period(-1,Days)) {
155  useLag = observationLag();
156  }
157 
158  Rate zeroRate;
159  if (forceLinearInterpolation) {
160  std::pair<Date,Date> dd = inflationPeriod(d-useLag, frequency());
161  dd.second = dd.second + Period(1,Days);
162  Real dp = dd.second - dd.first;
163  Real dt = d - dd.first;
164  // if we are interpolating we only check the exact point
165  // this prevents falling off the end at curve maturity
166  InflationTermStructure::checkRange(d, extrapolate);
167  Time t1 = timeFromReference(dd.first);
168  Time t2 = timeFromReference(dd.second);
169  Rate z1 = zeroRateImpl(t1);
170  Rate z2 = zeroRateImpl(t2);
171  zeroRate = z1 + (z2-z1) * (dt/dp);
172  } else {
173  if (indexIsInterpolated()) {
174  InflationTermStructure::checkRange(d-useLag, extrapolate);
175  Time t = timeFromReference(d-useLag);
176  zeroRate = zeroRateImpl(t);
177  } else {
178  std::pair<Date,Date> dd = inflationPeriod(d-useLag, frequency());
179  InflationTermStructure::checkRange(dd.first, extrapolate);
180  Time t = timeFromReference(dd.first);
181  zeroRate = zeroRateImpl(t);
182  }
183  }
184 
185  if (hasSeasonality()) {
186  zeroRate = seasonality()->correctZeroRate(d-useLag, zeroRate, *this);
187  }
188  return zeroRate;
189  }

236  Rate YoYInflationTermStructure::yoyRate(const Date &d, const Period& instObsLag,
237  bool forceLinearInterpolation,
238  bool extrapolate) const {
239 
240  Period useLag = instObsLag;
241  if (instObsLag == Period(-1,Days)) {
242  useLag = observationLag();
243  }
244 
245  Rate yoyRate;
246  if (forceLinearInterpolation) {
247  std::pair<Date,Date> dd = inflationPeriod(d-useLag, frequency());
248  dd.second = dd.second + Period(1,Days);
249  Real dp = dd.second - dd.first;
250  Real dt = (d-useLag) - dd.first;
251  // if we are interpolating we only check the exact point
252  // this prevents falling off the end at curve maturity
253  InflationTermStructure::checkRange(d, extrapolate);
254  Time t1 = timeFromReference(dd.first);
255  Time t2 = timeFromReference(dd.second);
256  Rate y1 = yoyRateImpl(t1);
257  Rate y2 = yoyRateImpl(t2);
258  yoyRate = y1 + (y2-y1) * (dt/dp);
259  } else {
260  if (indexIsInterpolated()) {
261  InflationTermStructure::checkRange(d-useLag, extrapolate);
262  Time t = timeFromReference(d-useLag);
263  yoyRate = yoyRateImpl(t);
264  } else {
265  std::pair<Date,Date> dd = inflationPeriod(d-useLag, frequency());
266  InflationTermStructure::checkRange(dd.first, extrapolate);
267  Time t = timeFromReference(dd.first);
268  yoyRate = yoyRateImpl(t);
269  }
270  }
271 
272  if (hasSeasonality()) {
273  yoyRate = seasonality()->correctYoYRate(d-useLag, yoyRate, *this);
274  }
275  return yoyRate;
276  }

287  std::pair<Date,Date> inflationPeriod(const Date& d,
288  Frequency frequency) {
289 
290  Month month = d.month();
291  Year year = d.year();
292 
293  Month startMonth, endMonth;
294  switch (frequency) {
295  case Annual:
296  startMonth = January;
297  endMonth = December;
298  break;
299  case Semiannual:
300  startMonth = Month(6*((month-1)/6) + 1);
301  endMonth = Month(startMonth + 5);
302  break;
303  case Quarterly:
304  startMonth = Month(3*((month-1)/3) + 1);
305  endMonth = Month(startMonth + 2);
306  break;
307  case Monthly:
308  startMonth = endMonth = month;
309  break;
310  default:
311  QL_FAIL("Frequency not handled: " << frequency);
312  break;
313  };
314 
315  Date startDate = Date(1, startMonth, year);
316  Date endDate = Date::endOfMonth(Date(1, endMonth, year));
317 
318  return std::make_pair(startDate,endDate);
319  }

322  Time inflationYearFraction(Frequency f, bool indexIsInterpolated,
323  const DayCounter &dayCounter,
324  const Date &d1, const Date &d2) {
325 
326  Time t=0;
327  if (indexIsInterpolated) {
328  // N.B. we do not use linear interpolation between flat
329  // fixing forecasts for forecasts. This avoids awkwardnesses
330  // when bootstrapping the inflation curve.
331  t = dayCounter.yearFraction(d1, d2);
332  } else {
333  // I.e. fixing is constant for the whole inflation period.
334  // Use the value for half way along the period.
335  // But the inflation time is the time between period starts
336  std::pair<Date,Date> limD1 = inflationPeriod(d1, f);
337  std::pair<Date,Date> limD2 = inflationPeriod(d2, f);
338  t = dayCounter.yearFraction(limD1.first, limD2.first);
339  }
340 
341  return t;
342  }
