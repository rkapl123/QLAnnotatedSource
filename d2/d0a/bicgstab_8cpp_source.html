37  BiCGStabResult BiCGstab::solve(const Array& b, const Array& x0) const {
38  Real bnorm2 = Norm2(b);
39  if (bnorm2 == 0.0) {
40  BiCGStabResult result = { 0, 0.0, b};
41  return result;
42  }
43 
44  Array x = ((!x0.empty()) ? x0 : Array(b.size(), 0.0));
45  Array r = b - A_(x);
46 
47  Array rTld = r;
48  Array p, pTld, v, s, sTld, t;
49  Real omega = 1.0;
50  Real rho, rhoTld=1.0;
51  Real alpha = 0.0, beta;
52  Real error = Norm2(r)/bnorm2;
53 
54  Size i;
55  for (i=0; i < maxIter_ && error >= relTol_; ++i) {
56  rho = DotProduct(rTld, r);
57  if (rho == 0.0 || omega == 0.0)
58  break;
59 
60  if (i != 0U) {
61  beta = (rho / rhoTld) * (alpha / omega);
62  p = r + beta * (p - omega * v);
63  } else {
64  p = r;
65  }
66 
67  pTld = ((M_) != 0 ? M_(p) : p);
68  v = A_(pTld);
69 
70  alpha = rho/DotProduct(rTld, v);
71  s = r-alpha*v;
72  if (Norm2(s) < relTol_*bnorm2) {
73  x += alpha*pTld;
74  error = Norm2(s)/bnorm2;
75  break;
76  }
77 
78  sTld = ((M_) != 0 ? M_(s) : s);
79  t = A_(sTld);
80  omega = DotProduct(t,s)/DotProduct(t,t);
81  x += alpha*pTld + omega*sTld;
82  r = s - omega*t;
83  error = Norm2(r)/bnorm2;
84  rhoTld = rho;
85  }
86 
87  QL_REQUIRE(i < maxIter_, "max number of iterations exceeded");
88  QL_REQUIRE(error < relTol_, "could not converge");
89 
90  BiCGStabResult result = { i, error, x};
91  return result;
92  }
