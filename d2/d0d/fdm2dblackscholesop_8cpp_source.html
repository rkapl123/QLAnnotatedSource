70  void Fdm2dBlackScholesOp::setTime(Time t1, Time t2) {
71  opX_.setTime(t1, t2);
72  opY_.setTime(t1, t2);
73 
74  if (localVol1_ != nullptr) {
75  const ext::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
76  const FdmLinearOpIterator endIter = layout->end();
77 
78  Array vol1(layout->size()), vol2(layout->size());
79  for (FdmLinearOpIterator iter = layout->begin();
80  iter!=endIter; ++iter) {
81  const Size i = iter.index();
82 
83  if (illegalLocalVolOverwrite_ < 0.0) {
84  vol1[i] = localVol1_->localVol(0.5*(t1+t2), x_[i], true);
85  vol2[i] = localVol2_->localVol(0.5*(t1+t2), y_[i], true);
86  }
87  else {
88  try {
89  vol1[i] = localVol1_->localVol(0.5*(t1+t2), x_[i],true);
90  } catch (Error&) {
91  vol1[i] = illegalLocalVolOverwrite_;
92  }
93  try {
94  vol2[i] = localVol2_->localVol(0.5*(t1+t2), y_[i],true);
95  } catch (Error&) {
96  vol2[i] = illegalLocalVolOverwrite_;
97  }
98 
99  }
100  }
101  corrMapT_ = corrMapTemplate_.mult(vol1*vol2);
102  } else {
103  const Real vol1 = p1_
104  ->blackVolatility()->blackForwardVol(t1, t2, p1_->x0());
105 
106  const Real vol2 = p2_
107  ->blackVolatility()->blackForwardVol(t1, t2, p2_->x0());
108 
109  corrMapT_ = corrMapTemplate_
110  .mult(Array(mesher_->layout()->size(), vol1*vol2));
111  }
112 
113  currentForwardRate_ = p1_->riskFreeRate()
114  ->forwardRate(t1, t2, Continuous).rate();
115  }
