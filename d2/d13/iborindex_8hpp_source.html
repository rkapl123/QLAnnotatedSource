113  inline Rate IborIndex::forecastFixing(const Date& d1,
114  const Date& d2,
115  Time t) const {
116  QL_REQUIRE(!termStructure_.empty(),
117  "null term structure set to this instance of " << name());
118  DiscountFactor disc1 = termStructure_->discount(d1);
119  DiscountFactor disc2 = termStructure_->discount(d2);
120  return (disc1/disc2 - 1.0) / t;
121  }
