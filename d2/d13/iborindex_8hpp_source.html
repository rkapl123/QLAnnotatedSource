115  inline Rate IborIndex::forecastFixing(const Date& d1,
116  const Date& d2,
117  Time t) const {
118  QL_REQUIRE(!termStructure_.empty(),
119  "null term structure set to this instance of " << name());
120  DiscountFactor disc1 = termStructure_->discount(d1);
121  DiscountFactor disc2 = termStructure_->discount(d2);
122  return (disc1/disc2 - 1.0) / t;
