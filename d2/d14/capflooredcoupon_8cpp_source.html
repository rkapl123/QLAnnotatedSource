81  Rate CappedFlooredCoupon::rate() const {
82  QL_REQUIRE(underlying_->pricer(), "pricer not set");
83  Rate swapletRate = underlying_->rate();
84  Rate floorletRate = 0.;
85  if(isFloored_)
86  floorletRate = underlying_->pricer()->floorletRate(effectiveFloor());
87  Rate capletRate = 0.;
88  if(isCapped_)
89  capletRate = underlying_->pricer()->capletRate(effectiveCap());
90  return swapletRate + floorletRate - capletRate;
91  }
