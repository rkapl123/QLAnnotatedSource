62  Quantity UnitOfMeasureConversion::convert(const Quantity& quantity) const {
63  switch (data_->type) {
64  case Direct:
65  if (quantity.unitOfMeasure() == data_->source)
66  return Quantity(quantity.commodityType(),
67  data_->target,
68  quantity.amount()*data_->conversionFactor);
69  else if (quantity.unitOfMeasure() == data_->target)
70  return Quantity(quantity.commodityType(),
71  data_->source,
72  quantity.amount()/data_->conversionFactor);
73  else
74  QL_FAIL("direct conversion not applicable");
75  case Derived:
76  if (quantity.unitOfMeasure()
77  == data_->conversionFactorChain.first->source() ||
78  quantity.unitOfMeasure()
79  == data_->conversionFactorChain.first->target())
80  return data_->conversionFactorChain.second->convert(
81  data_->conversionFactorChain.first->convert(quantity));
82  else if (quantity.unitOfMeasure()
83  == data_->conversionFactorChain.second->source() ||
84  quantity.unitOfMeasure()
85  == data_->conversionFactorChain.second->target())
86  return data_->conversionFactorChain.first->convert(
87  data_->conversionFactorChain.second->convert(quantity));
88  else
89  QL_FAIL("derived conversion factor not applicable");
90  default:
91  QL_FAIL("unknown conversion-factor type");
92  }
93  }

95  UnitOfMeasureConversion UnitOfMeasureConversion::chain(
96  const UnitOfMeasureConversion& r1,
97  const UnitOfMeasureConversion& r2) {
98  UnitOfMeasureConversion result(r1, r2);
99  result.data_->type = Derived;
100  if (r1.data_->source == r2.data_->source) {
101  result.data_->source = r1.data_->target;
102  result.data_->target = r2.data_->target;
103  result.data_->conversionFactor =
104  r2.data_->conversionFactor/r1.data_->conversionFactor;
105  } else if (r1.data_->source == r2.data_->target) {
106  result.data_->source = r1.data_->target;
107  result.data_->target = r2.data_->source;
108  result.data_->conversionFactor =
109  1.0/(r1.data_->conversionFactor*r2.data_->conversionFactor);
110  } else if (r1.data_->target == r2.data_->source) {
111  result.data_->source = r1.data_->source;
112  result.data_->target = r2.data_->target;
113  result.data_->conversionFactor =
114  r1.data_->conversionFactor*r2.data_->conversionFactor;
115  } else if (r1.data_->target == r2.data_->target) {
116  result.data_->source = r1.data_->source;
117  result.data_->target = r2.data_->source;
118  result.data_->conversionFactor =
119  r1.data_->conversionFactor/r2.data_->conversionFactor;
120  } else {
121  QL_FAIL("conversion factors not chainable");
122  }
123  return result;
124  }
