31  FwdPeriodAdapter::FwdPeriodAdapter(
32  const boost::shared_ptr<MarketModel>& largeModel,
33  Size period,
34  Size offset,
35  const std::vector<Spread>& newDisplacements)
36  :
37  numberOfFactors_(largeModel->numberOfFactors()),
38  numberOfRates_((largeModel->numberOfRates()-offset) / (period > 0 ? period : 1) ),
39  numberOfSteps_(largeModel->numberOfSteps()),
40  pseudoRoots_(numberOfSteps_, Matrix(numberOfRates_,
41  numberOfFactors_)),
42  displacements_(newDisplacements)
43  {
44  QL_REQUIRE( period >0, "period must be greater than zero in fwdperiodadapter");
45  QL_REQUIRE(period > offset, "period must be greater than offset in fwdperiodadapter");
46 
47  const std::vector<Spread>& largeDisplacements_ =
48  largeModel->displacements();
49 
50  if (displacements_.size() == 1)
51  {
52  Real dis = displacements_[0];
53  displacements_.resize(numberOfRates_);
54  std::fill(displacements_.begin(), displacements_.end(), dis);
55  }
56 
57  if (displacements_.size() ==0) // if not specified use average across rate
58  {
59  displacements_.reserve(numberOfRates_);
60  Size m=0;
61  Real sum=0.0;
62  for (Size k=0; k < numberOfRates_; ++k)
63  {
64  for (Size l=0; l < period; ++l, ++m)
65  sum+= largeDisplacements_[m];
66 
67  displacements_.push_back(sum/period);
68  }
69  }
70  QL_REQUIRE( displacements_.size() == numberOfRates_,"newDisplacements should be empty,1, or number of new rates in fwdperiodadapter");
71 
72  LMMCurveState largeCS(largeModel->evolution().rateTimes());
73  largeCS.setOnForwardRates(largeModel->initialRates());
74 
75  LMMCurveState smallCS(
76  ForwardForwardMappings::RestrictCurveState(largeCS,
77  period, offset
78  ));
79 
80  initialRates_ =smallCS.forwardRates();
81 
82  Real finalReset = smallCS.rateTimes()[smallCS.numberOfRates()-1];
83  std::vector<Time> oldEvolutionTimes(largeModel->evolution().evolutionTimes());
84  std::vector<Time> newEvolutionTimes;
85  for (Size i =0; i < oldEvolutionTimes.size() && oldEvolutionTimes[i]<= finalReset; ++i)
86  newEvolutionTimes.push_back(oldEvolutionTimes[i]);
87 
88  evolution_=EvolutionDescription(smallCS.rateTimes(),
89  newEvolutionTimes);
90 
91  numberOfSteps_ = newEvolutionTimes.size();
92 
93 
94  const std::vector<Time>& rateTimes =
95  smallCS.rateTimes();
96  // we must ensure we step through all rateTimes
97  const std::vector<Time>& evolutionTimes =
98  evolution_.evolutionTimes();
99 
100  std::set<Time> setTimes(evolutionTimes.begin(),evolutionTimes.end());
101 
102  for (Size i=0; i < rateTimes.size()-1; ++i)
103  QL_REQUIRE(setTimes.find(rateTimes[i]) != setTimes.end(),
104  "every new rate time except last must be an evolution time in fwdperiod adapter");
105 
106 
107  Matrix YMatrix =
108  ForwardForwardMappings::YMatrix( largeCS,
109  largeDisplacements_,
110  displacements_,
111  period,
112  offset
113  );
114 
115  const std::vector<Size>& alive =
116  evolution_.firstAliveRate();
117 
118  for (Size k = 0; k<numberOfSteps_; ++k) {
119  pseudoRoots_[k]=YMatrix*largeModel->pseudoRoot(k);
120  for (Size i=0; i<alive[k]; ++i)
121  std::fill(pseudoRoots_[k].row_begin(i),
122  pseudoRoots_[k].row_end(i),
123  0.0);
124  }
125  }
