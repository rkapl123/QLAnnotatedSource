29  LogNormalCotSwapRatePc::LogNormalCotSwapRatePc(
30  const ext::shared_ptr<MarketModel>& marketModel,
31  const BrownianGeneratorFactory& factory,
32  const std::vector<Size>& numeraires,
33  Size initialStep)
34  : marketModel_(marketModel),
35  numeraires_(numeraires),
36  initialStep_(initialStep),
37  numberOfRates_(marketModel->numberOfRates()),
38  numberOfFactors_(marketModel_->numberOfFactors()),
39  curveState_(marketModel->evolution().rateTimes()),
40  swapRates_(marketModel->initialRates()),
41  displacements_(marketModel->displacements()),
42  logSwapRates_(numberOfRates_), initialLogSwapRates_(numberOfRates_),
43  drifts1_(numberOfRates_), drifts2_(numberOfRates_),
44  initialDrifts_(numberOfRates_),
45  brownians_(numberOfFactors_), correlatedBrownians_(numberOfRates_),
46  alive_(marketModel->evolution().firstAliveRate())
47  {
48  checkCompatibility(marketModel->evolution(), numeraires);
49 
50  Size steps = marketModel->evolution().numberOfSteps();
51 
52  generator_ = factory.create(numberOfFactors_, steps-initialStep_);
53 
54  currentStep_ = initialStep_;
55 
56  calculators_.reserve(steps);
57  fixedDrifts_.reserve(steps);
58  for (Size j=0; j<steps; ++j) {
59  const Matrix& A = marketModel_->pseudoRoot(j);
60  calculators_.push_back(SMMDriftCalculator(A,
61  displacements_,
62  marketModel->evolution().rateTaus(),
63  numeraires[j],
64  alive_[j]));
65  std::vector<Real> fixed(numberOfRates_);
66  for (Size k=0; k<numberOfRates_; ++k) {
67  Real variance =
68  std::inner_product(A.row_begin(k), A.row_end(k),
69  A.row_begin(k), 0.0);
70  fixed[k] = -0.5*variance;
71  }
72  fixedDrifts_.push_back(fixed);
73  }
74 
75  setCoterminalSwapRates(marketModel_->initialRates());
76  }

82  void LogNormalCotSwapRatePc::setCoterminalSwapRates(const std::vector<Real>& swapRates)
83  {
84  QL_REQUIRE(swapRates.size()==numberOfRates_,
85  "mismatch between swapRates and rateTimes");
86  for (Size i=0; i<numberOfRates_; ++i)
87  initialLogSwapRates_[i] = std::log(swapRates[i] +
88  displacements_[i]);
89  curveState_.setOnCoterminalSwapRates(swapRates);
90  calculators_[initialStep_].compute(curveState_, initialDrifts_);
91  }

107  Real LogNormalCotSwapRatePc::advanceStep()
108  {
109  //we're going from T1 to T2
110 
111  //a) compute drifts D1 at T1;
112  if (currentStep_ > initialStep_)
113  calculators_[currentStep_].compute(curveState_, drifts1_);
114  else
115  std::copy(initialDrifts_.begin(), initialDrifts_.end(),
116  drifts1_.begin());
117 
118  //b) evolve forwards up to T2 using D1;
119  Real weight = generator_->nextStep(brownians_);
120  const Matrix& A = marketModel_->pseudoRoot(currentStep_);
121  const std::vector<Real>& fixedDrift = fixedDrifts_[currentStep_];
122 
123  Size i, alive = alive_[currentStep_];
124  for (i=alive; i<numberOfRates_; ++i) {
125  logSwapRates_[i] += drifts1_[i] + fixedDrift[i];
126  logSwapRates_[i] +=
127  std::inner_product(A.row_begin(i), A.row_end(i),
128  brownians_.begin(), 0.0);
129  swapRates_[i] = std::exp(logSwapRates_[i]) - displacements_[i];
130  }
131 
132  // intermediate curve state update
133  curveState_.setOnCoterminalSwapRates(swapRates_);
134 
135  //c) recompute drifts D2 using the predicted forwards;
136  calculators_[currentStep_].compute(curveState_, drifts2_);
137 
138  //d) correct forwards using both drifts
139  for (i=alive; i<numberOfRates_; ++i) {
140  logSwapRates_[i] += (drifts2_[i]-drifts1_[i])/2.0;
141  swapRates_[i] = std::exp(logSwapRates_[i]) - displacements_[i];
142  }
143 
144  //e) update curve state
145  curveState_.setOnCoterminalSwapRates(swapRates_);
146 
147  ++currentStep_;
148 
149  return weight;
150  }
