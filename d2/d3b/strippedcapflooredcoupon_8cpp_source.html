40  Rate StrippedCappedFlooredCoupon::rate() const {
41 
42  QL_REQUIRE(underlying_->underlying()->pricer() != nullptr, "pricer not set");
43  underlying_->underlying()->pricer()->initialize(*underlying_->underlying());
44  Rate floorletRate = 0.0;
45  if (underlying_->isFloored())
46  floorletRate = underlying_->underlying()->pricer()->floorletRate(
47  underlying_->effectiveFloor());
48  Rate capletRate = 0.0;
49  if (underlying_->isCapped())
50  capletRate =
51  underlying_->underlying()->pricer()->capletRate(underlying_->effectiveCap());
52 
53  // if the underlying is collared we return the value of the embedded
54  // collar, otherwise the value of a long floor or a long cap respectively
55 
56  return (underlying_->isFloored() && underlying_->isCapped())
57  ? floorletRate - capletRate
58  : floorletRate + capletRate;
59  }
