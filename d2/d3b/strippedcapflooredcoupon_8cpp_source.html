38  Rate StrippedCappedFlooredCoupon::rate() const {
39 
40  QL_REQUIRE(underlying_->underlying()->pricer() != NULL, "pricer not set");
41  underlying_->underlying()->pricer()->initialize(*underlying_->underlying());
42  Rate floorletRate = 0.0;
43  if (underlying_->isFloored())
44  floorletRate = underlying_->underlying()->pricer()->floorletRate(
45  underlying_->effectiveFloor());
46  Rate capletRate = 0.0;
47  if (underlying_->isCapped())
48  capletRate =
49  underlying_->underlying()->pricer()->capletRate(underlying_->effectiveCap());
50 
51  // if the underlying is collared we return the value of the embedded
52  // collar, otherwise the value of a long floor or a long cap respectively
53 
54  return (underlying_->isFloored() && underlying_->isCapped())
55  ? floorletRate - capletRate
56  : floorletRate + capletRate;
57  }

115  for (Leg::const_iterator i = underlyingLeg_.begin();
116  i != underlyingLeg_.end(); ++i) {
117  if ((c = boost::dynamic_pointer_cast<CappedFlooredCoupon>(*i)) !=
118  NULL) {
119  resultLeg.push_back(
120  boost::make_shared<StrippedCappedFlooredCoupon>(c));
121  } else {
122  resultLeg.push_back(*i);
123  }
124  }
