38  std::pair<Real, Volatility>
39  BSMRNDCalculator::distributionParams(Real x, Time t) const {
40  const Volatility stdDev =
41  process_->blackVolatility()->blackVol(t, std::exp(x))*std::sqrt(t);
42  const Real mean = std::log(process_->x0()) - 0.5*stdDev*stdDev
43  + std::log( process_->dividendYield()->discount(t)
44  / process_->riskFreeRate()->discount(t));
45 
46  return std::make_pair(mean, stdDev);
47  }
