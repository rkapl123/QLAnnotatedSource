27  UniformGridMesher::UniformGridMesher(
28  const boost::shared_ptr<FdmLinearOpLayout> & layout,
29  const std::vector<std::pair<Real, Real> > & boundaries)
30  : FdmMesher (layout),
31  dx_ (new Real[layout->dim().size()]),
32  locations_(layout->dim().size()) {
33 
34  QL_REQUIRE(boundaries.size() == layout->dim().size(),
35  "inconsistent boundaries given");
36 
37  for (Size i=0; i < layout->dim().size(); ++i) {
38  dx_[i] = (boundaries[i].second - boundaries[i].first)
39  / (layout->dim()[i]-1);
40 
41  locations_[i] = std::vector<Real>(layout->dim()[i]);
42  for (Size j=0; j < layout->dim()[i]; ++j) {
43  locations_[i][j] = boundaries[i].first + j*dx_[i];
44  }
45  }
46  }

51  const FdmLinearOpIterator endIter = layout_->end();
52  for (FdmLinearOpIterator iter = layout_->begin();
53  iter != endIter; ++iter) {
54  retVal[iter.index()] = locations_[d][iter.coordinates()[d]];
55  }
56 
57  return retVal;
