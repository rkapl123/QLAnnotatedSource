77  Real YoYInflationCouponPricer::optionletPrice(Option::Type optionType,
78  Real effStrike) const {
79  Date fixingDate = coupon_->fixingDate();
80  if (fixingDate <= Settings::instance().evaluationDate()) {
81  // the amount is determined
82  Real a, b;
83  if (optionType==Option::Call) {
84  a = coupon_->indexFixing();
85  b = effStrike;
86  } else {
87  a = effStrike;
88  b = coupon_->indexFixing();
89  }
90  return std::max(a - b, 0.0)* coupon_->accrualPeriod()*discount_;
91  } else {
92  // not yet determined, use Black/DD1/Bachelier/whatever from Impl
93  QL_REQUIRE(!capletVolatility().empty(),
94  "missing optionlet volatility");
95  Real stdDev =
96  std::sqrt(capletVolatility()->totalVariance(fixingDate,
97  effStrike));
98  Rate fixing = optionletPriceImp(optionType,
99  effStrike,
100  adjustedFixing(),
101  stdDev);
102  return fixing * coupon_->accrualPeriod() * discount_;
103  }
104  }

117  void YoYInflationCouponPricer::initialize(const InflationCoupon& coupon) {
118  coupon_ = dynamic_cast<const YoYInflationCoupon*>(&coupon);
119  QL_REQUIRE(coupon_, "year-on-year inflation coupon needed");
120  gearing_ = coupon_->gearing();
121  spread_ = coupon_->spread();
122  paymentDate_ = coupon_->date();
123  rateCurve_ = boost::dynamic_pointer_cast<YoYInflationIndex>(coupon.index())
124  ->yoyInflationTermStructure()
125  ->nominalTermStructure();
126 
127  // past or future fixing is managed in YoYInflationIndex::fixing()
128  // use yield curve from index (which sets discount)
129 
130  discount_ = 1.0;
131  if (paymentDate_ > rateCurve_->referenceDate())
132  discount_ = rateCurve_->discount(paymentDate_);
133 
134  spreadLegValue_ = spread_ * coupon_->accrualPeriod()* discount_;
135 
136  }
