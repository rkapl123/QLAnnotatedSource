98  Real YoYInflationCouponPricer::optionletRate(Option::Type optionType,
99  Real effStrike) const {
100  Date fixingDate = coupon_->fixingDate();
101  if (fixingDate <= Settings::instance().evaluationDate()) {
102  // the amount is determined
103  Real a, b;
104  if (optionType==Option::Call) {
105  a = coupon_->indexFixing();
106  b = effStrike;
107  } else {
108  a = effStrike;
109  b = coupon_->indexFixing();
110  }
111  return std::max(a - b, 0.0);
112  } else {
113  // not yet determined, use Black/DD1/Bachelier/whatever from Impl
114  QL_REQUIRE(!capletVolatility().empty(),
115  "missing optionlet volatility");
116  Real stdDev =
117  std::sqrt(capletVolatility()->totalVariance(fixingDate,
118  effStrike));
119  return optionletPriceImp(optionType,
120  effStrike,
121  adjustedFixing(),
122  stdDev);
123  }
124  }

137  void YoYInflationCouponPricer::initialize(const InflationCoupon& coupon) {
138  coupon_ = dynamic_cast<const YoYInflationCoupon*>(&coupon);
139  QL_REQUIRE(coupon_, "year-on-year inflation coupon needed");
140  gearing_ = coupon_->gearing();
141  spread_ = coupon_->spread();
142  paymentDate_ = coupon_->date();
143  rateCurve_ =
144  !nominalTermStructure_.empty() ?
145  nominalTermStructure_ :
146  ext::dynamic_pointer_cast<YoYInflationIndex>(coupon.index())
147  ->yoyInflationTermStructure()
148  ->nominalTermStructure();
149 
150  // past or future fixing is managed in YoYInflationIndex::fixing()
151  // use yield curve from index (which sets discount)
152 
153  discount_ = 1.0;
154  if (paymentDate_ > rateCurve_->referenceDate()) {
155  if (rateCurve_.empty()) {
156  // allow to extract rates, but mark the discount as invalid for prices
157  discount_ = Null<Real>();
158  } else {
159  discount_ = rateCurve_->discount(paymentDate_);
160  }
161  }
162  }

171  Rate YoYInflationCouponPricer::swapletRate() const {
172  // This way we do not require the index to have
173  // a yield curve, i.e. we do not get the problem
174  // that a discounting-instrument-pricer is used
175  // with a different yield curve
176  return gearing_ * adjustedFixing() + spread_;
