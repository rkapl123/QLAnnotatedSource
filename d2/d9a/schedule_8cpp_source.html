31  Date nextTwentieth(const Date& d, DateGeneration::Rule rule) {
32  Date result = Date(20, d.month(), d.year());
33  if (result < d)
34  result += 1*Months;
35  if (rule == DateGeneration::TwentiethIMM ||
36  rule == DateGeneration::OldCDS ||
37  rule == DateGeneration::CDS ||
38  rule == DateGeneration::CDS2015) {
39  Month m = result.month();
40  if (m % 3 != 0) { // not a main IMM nmonth
41  Integer skip = 3 - m%3;
42  result += skip*Months;
43  }
44  }
45  return result;
46  }

79  Schedule::Schedule(Date effectiveDate,
80  const Date& terminationDate,
81  const Period& tenor,
82  Calendar cal,
83  BusinessDayConvention convention,
84  BusinessDayConvention terminationDateConvention,
85  DateGeneration::Rule rule,
86  bool endOfMonth,
87  const Date& first,
88  const Date& nextToLast)
89  : tenor_(tenor), calendar_(std::move(cal)), convention_(convention),
90  terminationDateConvention_(terminationDateConvention), rule_(rule),
91  endOfMonth_(allowsEndOfMonth(tenor) ? endOfMonth : false),
92  firstDate_(first == effectiveDate ? Date() : first),
93  nextToLastDate_(nextToLast == terminationDate ? Date() : nextToLast) {
94  // sanity checks
95  QL_REQUIRE(terminationDate != Date(), "null termination date");
96 
97  // in many cases (e.g. non-expired bonds) the effective date is not
98  // really necessary. In these cases a decent placeholder is enough
99  if (effectiveDate==Date() && first==Date()
100  && rule==DateGeneration::Backward) {
101  Date evalDate = Settings::instance().evaluationDate();
102  QL_REQUIRE(evalDate < terminationDate, "null effective date");
103  Natural y;
104  if (nextToLast != Date()) {
105  y = (nextToLast - evalDate)/366 + 1;
106  effectiveDate = nextToLast - y*Years;
107  } else {
108  y = (terminationDate - evalDate)/366 + 1;
109  effectiveDate = terminationDate - y*Years;
110  }
111  } else
112  QL_REQUIRE(effectiveDate != Date(), "null effective date");
113 
114  QL_REQUIRE(effectiveDate < terminationDate,
115  "effective date (" << effectiveDate
116  << ") later than or equal to termination date ("
117  << terminationDate << ")");
118 
119  if (tenor.length()==0)
120  rule_ = DateGeneration::Zero;
121  else
122  QL_REQUIRE(tenor.length()>0,
123  "non positive tenor (" << tenor << ") not allowed");
124 
125  if (firstDate_ != Date()) {
126  switch (*rule_) {
127  case DateGeneration::Backward:
128  case DateGeneration::Forward:
129  QL_REQUIRE(firstDate_ > effectiveDate &&
130  firstDate_ <= terminationDate,
131  "first date (" << firstDate_ <<
132  ") out of effective-termination date range (" <<
133  effectiveDate << ", " << terminationDate << "]");
134  // we should ensure that the above condition is still
135  // verified after adjustment
136  break;
137  case DateGeneration::ThirdWednesday:
138  QL_REQUIRE(IMM::isIMMdate(firstDate_, false),
139  "first date (" << firstDate_ <<
140  ") is not an IMM date");
141  break;
142  case DateGeneration::Zero:
143  case DateGeneration::Twentieth:
144  case DateGeneration::TwentiethIMM:
145  case DateGeneration::OldCDS:
146  case DateGeneration::CDS:
147  case DateGeneration::CDS2015:
148  QL_FAIL("first date incompatible with " << *rule_ <<
149  " date generation rule");
150  default:
151  QL_FAIL("unknown rule (" << Integer(*rule_) << ")");
152  }
153  }
154  if (nextToLastDate_ != Date()) {
155  switch (*rule_) {
156  case DateGeneration::Backward:
157  case DateGeneration::Forward:
158  QL_REQUIRE(nextToLastDate_ >= effectiveDate &&
159  nextToLastDate_ < terminationDate,
160  "next to last date (" << nextToLastDate_ <<
161  ") out of effective-termination date range [" <<
162  effectiveDate << ", " << terminationDate << ")");
163  // we should ensure that the above condition is still
164  // verified after adjustment
165  break;
166  case DateGeneration::ThirdWednesday:
167  QL_REQUIRE(IMM::isIMMdate(nextToLastDate_, false),
168  "next-to-last date (" << nextToLastDate_ <<
169  ") is not an IMM date");
170  break;
171  case DateGeneration::Zero:
172  case DateGeneration::Twentieth:
173  case DateGeneration::TwentiethIMM:
174  case DateGeneration::OldCDS:
175  case DateGeneration::CDS:
176  case DateGeneration::CDS2015:
177  QL_FAIL("next to last date incompatible with " << *rule_ <<
178  " date generation rule");
179  default:
180  QL_FAIL("unknown rule (" << Integer(*rule_) << ")");
181  }
182  }
183 
184 
185  // calendar needed for endOfMonth adjustment
186  Calendar nullCalendar = NullCalendar();
187  Integer periods = 1;
188  Date seed, exitDate;
189  switch (*rule_) {
190 
191  case DateGeneration::Zero:
192  tenor_ = 0*Years;
193  dates_.push_back(effectiveDate);
194  dates_.push_back(terminationDate);
195  isRegular_.push_back(true);
196  break;
197 
198  case DateGeneration::Backward:
199 
200  dates_.push_back(terminationDate);
201 
202  seed = terminationDate;
203  if (nextToLastDate_ != Date()) {
204  dates_.insert(dates_.begin(), nextToLastDate_);
205  Date temp = nullCalendar.advance(seed,
206  -periods*(*tenor_), convention, *endOfMonth_);
207  if (temp!=nextToLastDate_)
208  isRegular_.insert(isRegular_.begin(), false);
209  else
210  isRegular_.insert(isRegular_.begin(), true);
211  seed = nextToLastDate_;
212  }
213 
214  exitDate = effectiveDate;
215  if (firstDate_ != Date())
216  exitDate = firstDate_;
217 
218  for (;;) {
219  Date temp = nullCalendar.advance(seed,
220  -periods*(*tenor_), convention, *endOfMonth_);
221  if (temp < exitDate) {
222  if (firstDate_ != Date() &&
223  (calendar_.adjust(dates_.front(),convention)!=
224  calendar_.adjust(firstDate_,convention))) {
225  dates_.insert(dates_.begin(), firstDate_);
226  isRegular_.insert(isRegular_.begin(), false);
227  }
228  break;
229  } else {
230  // skip dates that would result in duplicates
231  // after adjustment
232  if (calendar_.adjust(dates_.front(),convention)!=
233  calendar_.adjust(temp,convention)) {
234  dates_.insert(dates_.begin(), temp);
235  isRegular_.insert(isRegular_.begin(), true);
236  }
237  ++periods;
238  }
239  }
240 
241  if (calendar_.adjust(dates_.front(),convention)!=
242  calendar_.adjust(effectiveDate,convention)) {
243  dates_.insert(dates_.begin(), effectiveDate);
244  isRegular_.insert(isRegular_.begin(), false);
245  }
246  break;
247 
248  case DateGeneration::Twentieth:
249  case DateGeneration::TwentiethIMM:
250  case DateGeneration::ThirdWednesday:
251  case DateGeneration::OldCDS:
252  case DateGeneration::CDS:
253  case DateGeneration::CDS2015:
254  QL_REQUIRE(!*endOfMonth_,
255  "endOfMonth convention incompatible with " << *rule_ <<
256  " date generation rule");
257  // fall through
258  case DateGeneration::Forward:
259 
260  if (*rule_ == DateGeneration::CDS || *rule_ == DateGeneration::CDS2015) {
261  Date prev20th = previousTwentieth(effectiveDate, *rule_);
262  if (calendar_.adjust(prev20th, convention) > effectiveDate) {
263  dates_.push_back(prev20th - 3 * Months);
264  isRegular_.push_back(true);
265  }
266  dates_.push_back(prev20th);
267  } else {
268  dates_.push_back(effectiveDate);
269  }
270 
271  seed = dates_.back();
272 
273  if (firstDate_!=Date()) {
274  dates_.push_back(firstDate_);
275  Date temp = nullCalendar.advance(seed, periods*(*tenor_),
276  convention, *endOfMonth_);
277  if (temp!=firstDate_)
278  isRegular_.push_back(false);
279  else
280  isRegular_.push_back(true);
281  seed = firstDate_;
282  } else if (*rule_ == DateGeneration::Twentieth ||
283  *rule_ == DateGeneration::TwentiethIMM ||
284  *rule_ == DateGeneration::OldCDS ||
285  *rule_ == DateGeneration::CDS ||
286  *rule_ == DateGeneration::CDS2015) {
287  Date next20th = nextTwentieth(effectiveDate, *rule_);
288  if (*rule_ == DateGeneration::OldCDS) {
289  // distance rule inforced in natural days
290  static const Date::serial_type stubDays = 30;
291  if (next20th - effectiveDate < stubDays) {
292  // +1 will skip this one and get the next
293  next20th = nextTwentieth(next20th + 1, *rule_);
294  }
295  }
296  if (next20th != effectiveDate) {
297  dates_.push_back(next20th);
298  isRegular_.push_back(*rule_ == DateGeneration::CDS || *rule_ == DateGeneration::CDS2015);
299  seed = next20th;
300  }
301  }
302 
303  exitDate = terminationDate;
304  if (nextToLastDate_ != Date())
305  exitDate = nextToLastDate_;
306  for (;;) {
307  Date temp = nullCalendar.advance(seed, periods*(*tenor_),
308  convention, *endOfMonth_);
309  if (temp > exitDate) {
310  if (nextToLastDate_ != Date() &&
311  (calendar_.adjust(dates_.back(),convention)!=
312  calendar_.adjust(nextToLastDate_,convention))) {
313  dates_.push_back(nextToLastDate_);
314  isRegular_.push_back(false);
315  }
316  break;
317  } else {
318  // skip dates that would result in duplicates
319  // after adjustment
320  if (calendar_.adjust(dates_.back(),convention)!=
321  calendar_.adjust(temp,convention)) {
322  dates_.push_back(temp);
323  isRegular_.push_back(true);
324  }
325  ++periods;
326  }
327  }
328 
329  if (calendar_.adjust(dates_.back(),terminationDateConvention)!=
330  calendar_.adjust(terminationDate,terminationDateConvention)) {
331  if (*rule_ == DateGeneration::Twentieth ||
332  *rule_ == DateGeneration::TwentiethIMM ||
333  *rule_ == DateGeneration::OldCDS ||
334  *rule_ == DateGeneration::CDS ||
335  *rule_ == DateGeneration::CDS2015) {
336  dates_.push_back(nextTwentieth(terminationDate, *rule_));
337  isRegular_.push_back(true);
338  } else {
339  dates_.push_back(terminationDate);
340  isRegular_.push_back(false);
341  }
342  }
343 
344  break;
345 
346  default:
347  QL_FAIL("unknown rule (" << Integer(*rule_) << ")");
348  }
349 
350  // adjustments
351  if (*rule_==DateGeneration::ThirdWednesday)
352  for (Size i=1; i<dates_.size()-1; ++i)
353  dates_[i] = Date::nthWeekday(3, Wednesday,
354  dates_[i].month(),
355  dates_[i].year());
356 
357  if (*endOfMonth_ && calendar_.isEndOfMonth(seed)) {
358  // adjust to end of month
359  if (convention == Unadjusted) {
360  for (Size i=1; i<dates_.size()-1; ++i)
361  dates_[i] = Date::endOfMonth(dates_[i]);
362  } else {
363  for (Size i=1; i<dates_.size()-1; ++i)
364  dates_[i] = calendar_.endOfMonth(dates_[i]);
365  }
366  Date d1 = dates_.front(), d2 = dates_.back();
367  if (terminationDateConvention != Unadjusted) {
368  d1 = calendar_.endOfMonth(dates_.front());
369  d2 = calendar_.endOfMonth(dates_.back());
370  } else {
371  // the termination date is the first if going backwards,
372  // the last otherwise.
373  if (*rule_ == DateGeneration::Backward)
374  d2 = Date::endOfMonth(dates_.back());
375  else
376  d1 = Date::endOfMonth(dates_.front());
377  }
378  // if the eom adjustment leads to a single date schedule
379  // we do not apply it
380  if(d1 != d2) {
381  dates_.front() = d1;
382  dates_.back() = d2;
383  }
384  } else {
385  // first date not adjusted for old CDS schedules
386  if (*rule_ != DateGeneration::OldCDS)
387  dates_[0] = calendar_.adjust(dates_[0], convention);
388  for (Size i=1; i<dates_.size()-1; ++i)
389  dates_[i] = calendar_.adjust(dates_[i], convention);
390 
391  // termination date is NOT adjusted as per ISDA
392  // specifications, unless otherwise specified in the
393  // confirmation of the deal or unless we're creating a CDS
394  // schedule
395  if (terminationDateConvention != Unadjusted
396  && *rule_ != DateGeneration::CDS
397  && *rule_ != DateGeneration::CDS2015) {
398  dates_.back() = calendar_.adjust(dates_.back(),
399  terminationDateConvention);
400  }
401  }
402 
403  // Final safety checks to remove extra next-to-last date, if
404  // necessary. It can happen to be equal or later than the end
405  // date due to EOM adjustments (see the Schedule test suite
406  // for an example).
407  if (dates_.size() >= 2 && dates_[dates_.size()-2] >= dates_.back()) {
408  // there might be two dates only, then isRegular_ has size one
409  if (isRegular_.size() >= 2) {
410  isRegular_[isRegular_.size() - 2] =
411  (dates_[dates_.size() - 2] == dates_.back());
412  }
413  dates_[dates_.size() - 2] = dates_.back();
414  dates_.pop_back();
415  isRegular_.pop_back();
416  }
417  if (dates_.size() >= 2 && dates_[1] <= dates_.front()) {
418  isRegular_[1] =
419  (dates_[1] == dates_.front());
420  dates_[1] = dates_.front();
421  dates_.erase(dates_.begin());
422  isRegular_.erase(isRegular_.begin());
423  }
424 
425  QL_ENSURE(dates_.size()>1,
426  "degenerate single date (" << dates_[0] << ") schedule" <<
427  "\n seed date: " << seed <<
428  "\n exit date: " << exitDate <<
429  "\n effective date: " << effectiveDate <<
430  "\n first date: " << first <<
431  "\n next to last date: " << nextToLast <<
432  "\n termination date: " << terminationDate <<
433  "\n generation rule: " << *rule_ <<
434  "\n end of month: " << *endOfMonth_);
435  }

437  Schedule Schedule::after(const Date& truncationDate) const {
438  Schedule result = *this;
439 
440  QL_REQUIRE(truncationDate < result.dates_.back(),
441  "truncation date " << truncationDate <<
442  " must be before the last schedule date " <<
443  result.dates_.back());
444  if (truncationDate > result.dates_[0]) {
445  // remove earlier dates
446  while (result.dates_[0] < truncationDate) {
447  result.dates_.erase(result.dates_.begin());
448  if (!result.isRegular_.empty())
449  result.isRegular_.erase(result.isRegular_.begin());
450  }
451 
452  // add truncationDate if missing
453  if (truncationDate != result.dates_.front()) {
454  result.dates_.insert(result.dates_.begin(), truncationDate);
455  result.isRegular_.insert(result.isRegular_.begin(), false);
456  result.terminationDateConvention_ = Unadjusted;
457  }
458  else {
459  result.terminationDateConvention_ = convention_;
460  }
461 
462  if (result.nextToLastDate_ <= truncationDate)
463  result.nextToLastDate_ = Date();
464  if (result.firstDate_ <= truncationDate)
465  result.firstDate_ = Date();
466  }
467 
468  return result;
469  }

471  Schedule Schedule::until(const Date& truncationDate) const {
472  Schedule result = *this;
473 
474  QL_REQUIRE(truncationDate>result.dates_[0],
475  "truncation date " << truncationDate <<
476  " must be later than schedule first date " <<
477  result.dates_[0]);
478  if (truncationDate<result.dates_.back()) {
479  // remove later dates
480  while (result.dates_.back()>truncationDate) {
481  result.dates_.pop_back();
482  if(!result.isRegular_.empty())
483  result.isRegular_.pop_back();
484  }
485 
486  // add truncationDate if missing
487  if (truncationDate!=result.dates_.back()) {
488  result.dates_.push_back(truncationDate);
489  result.isRegular_.push_back(false);
490  result.terminationDateConvention_ = Unadjusted;
491  } else {
492  result.terminationDateConvention_ = convention_;
493  }

495  if (result.nextToLastDate_>=truncationDate)
496  result.nextToLastDate_ = Date();
497  if (result.firstDate_>=truncationDate)
498  result.firstDate_ = Date();
499  }

610  MakeSchedule::operator Schedule() const {
611  // check for mandatory arguments
612  QL_REQUIRE(effectiveDate_ != Date(), "effective date not provided");
613  QL_REQUIRE(terminationDate_ != Date(), "termination date not provided");
614  QL_REQUIRE(tenor_, "tenor/frequency not provided");
615 
616  // set dynamic defaults:
617  BusinessDayConvention convention;
618  // if a convention was set, we use it.
619  if (convention_) { // NOLINT(readability-implicit-bool-conversion)
620  convention = *convention_;
621  } else {
622  if (!calendar_.empty()) {
623  // ...if we set a calendar, we probably want it to be used;
624  convention = Following;
625  } else {
626  // if not, we don't care.
627  convention = Unadjusted;
628  }
629  }
630 
631  BusinessDayConvention terminationDateConvention;
632  // if set explicitly, we use it;
633  if (terminationDateConvention_) { // NOLINT(readability-implicit-bool-conversion)
634  terminationDateConvention = *terminationDateConvention_;
635  } else {
636  // Unadjusted as per ISDA specification
637  terminationDateConvention = convention;
638  }
639 
640  Calendar calendar = calendar_;
641  // if no calendar was set...
642  if (calendar.empty()) {
643  // ...we use a null one.
644  calendar = NullCalendar();
645  }
646 
647  return Schedule(effectiveDate_, terminationDate_, *tenor_, calendar,
648  convention, terminationDateConvention,
649  rule_, endOfMonth_, firstDate_, nextToLastDate_);
650  }

652  Date previousTwentieth(const Date& d, DateGeneration::Rule rule) {
653  Date result = Date(20, d.month(), d.year());
654  if (result > d)
655  result -= 1 * Months;
656  if (rule == DateGeneration::TwentiethIMM ||
657  rule == DateGeneration::OldCDS ||
658  rule == DateGeneration::CDS ||
659  rule == DateGeneration::CDS2015) {
660  Month m = result.month();
661  if (m % 3 != 0) { // not a main IMM nmonth
662  Integer skip = m % 3;
663  result -= skip * Months;
664  }
665  }
666  return result;
667  }
