41  void JumpDiffusionEngine::calculate() const {
42 
43  Real jumpSquareVol = process_->logJumpVolatility()->value()
44  * process_->logJumpVolatility()->value();
45  Real muPlusHalfSquareVol = process_->logMeanJump()->value()
46  + 0.5*jumpSquareVol;
47  // mean jump size
48  Real k = std::exp(muPlusHalfSquareVol) - 1.0;
49  Real lambda = (k+1.0) * process_->jumpIntensity()->value();
50 
51  boost::shared_ptr<StrikedTypePayoff> payoff =
52  boost::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
53  QL_REQUIRE(payoff, "non-striked payoff given");
54 
55  Real variance =
56  process_->blackVolatility()->blackVariance(
57  arguments_.exercise->lastDate(),
58  payoff->strike());
59 
60  DayCounter voldc = process_->blackVolatility()->dayCounter();
61  Calendar volcal = process_->blackVolatility()->calendar();
62  Date volRefDate = process_->blackVolatility()->referenceDate();
63  Time t = voldc.yearFraction(volRefDate,
64  arguments_.exercise->lastDate());
65  Rate riskFreeRate = -std::log(process_->riskFreeRate()->discount(
66  arguments_.exercise->lastDate()))/t;
67  Date rateRefDate = process_->riskFreeRate()->referenceDate();
68 
69  PoissonDistribution p(lambda*t);
70 
71  Handle<Quote> stateVariable = process_->stateVariable();
72  Handle<YieldTermStructure> dividendTS = process_->dividendYield();
73  RelinkableHandle<YieldTermStructure> riskFreeTS(
74  *process_->riskFreeRate());
75  RelinkableHandle<BlackVolTermStructure> volTS(
76  *process_->blackVolatility());
77 
78  boost::shared_ptr<GeneralizedBlackScholesProcess> bsProcess(
79  new GeneralizedBlackScholesProcess(stateVariable, dividendTS,
80  riskFreeTS, volTS));
81 
82  AnalyticEuropeanEngine baseEngine(bsProcess);
83 
84  VanillaOption::arguments* baseArguments =
85  dynamic_cast<VanillaOption::arguments*>(baseEngine.getArguments());
86 
87  baseArguments->payoff = arguments_.payoff;
88  baseArguments->exercise = arguments_.exercise;
89 
90  baseArguments->validate();
91 
92  const VanillaOption::results* baseResults =
93  dynamic_cast<const VanillaOption::results*>(
94  baseEngine.getResults());
95 
96  results_.value = 0.0;
97  results_.delta = 0.0;
98  results_.gamma = 0.0;
99  results_.theta = 0.0;
100  results_.vega = 0.0;
101  results_.rho = 0.0;
102  results_.dividendRho = 0.0;
103 
104  Real r, v, weight, lastContribution = 1.0;
105  Size i;
106  Real theta_correction;
107  // Haug arbitrary criterium is:
108  //for (i=0; i<11; i++) {
109  for (i=0; (lastContribution>relativeAccuracy_ && i<maxIterations_)
110  || i < Size(lambda*t); i++) {
111 
112  // constant vol/rate assumption. It should be relaxed
113  v = std::sqrt((variance + i*jumpSquareVol)/t);
114  r = riskFreeRate - process_->jumpIntensity()->value()*k
115  + i*muPlusHalfSquareVol/t;
116  riskFreeTS.linkTo(boost::shared_ptr<YieldTermStructure>(new
117  FlatForward(rateRefDate, r, voldc)));
118  volTS.linkTo(boost::shared_ptr<BlackVolTermStructure>(new
119  BlackConstantVol(rateRefDate, volcal, v, voldc)));
120 
121  baseArguments->validate();
122  baseEngine.calculate();
123 
124  weight = p(Size(i));
125  results_.value += weight * baseResults->value;
126  results_.delta += weight * baseResults->delta;
127  results_.gamma += weight * baseResults->gamma;
128  results_.vega += weight * (std::sqrt(variance/t)/v)*
129  baseResults->vega;
130  // theta modified
131  theta_correction = baseResults->vega*((i*jumpSquareVol)/
132  (2.0*v*t*t)) +
133  baseResults->rho*i*muPlusHalfSquareVol/(t*t);
134  results_.theta += weight *(baseResults->theta + theta_correction +
135  lambda*baseResults->value);
136  if(i != 0){
137  results_.theta -= (p(Size(i-1))*lambda* baseResults->value);
138  }
139  //end theta calculation
140  results_.rho += weight * baseResults->rho;
141  results_.dividendRho += weight * baseResults->dividendRho;
142 
143  lastContribution = std::fabs(baseResults->value /
144  (std::fabs(results_.value)>QL_EPSILON ? results_.value : 1.0));
145 
146  lastContribution = std::max<Real>(lastContribution,
147  std::fabs(baseResults->delta /
148  (std::fabs(results_.delta)>QL_EPSILON ? results_.delta : 1.0)));
149 
150  lastContribution = std::max<Real>(lastContribution,
151  std::fabs(baseResults->gamma /
152  (std::fabs(results_.gamma)>QL_EPSILON ? results_.gamma : 1.0)));
153 
154  lastContribution = std::max<Real>(lastContribution,
155  std::fabs(baseResults->theta /
156  (std::fabs(results_.theta)>QL_EPSILON ? results_.theta : 1.0)));
157 
158  lastContribution = std::max<Real>(lastContribution,
159  std::fabs(baseResults->vega /
160  (std::fabs(results_.vega)>QL_EPSILON ? results_.vega : 1.0)));
161 
162  lastContribution = std::max<Real>(lastContribution,
163  std::fabs(baseResults->rho /
164  (std::fabs(results_.rho)>QL_EPSILON ? results_.rho : 1.0)));
165 
166  lastContribution = std::max<Real>(lastContribution,
167  std::fabs(baseResults->dividendRho /
168  (std::fabs(results_.dividendRho)>QL_EPSILON ?
169  results_.dividendRho : 1.0)));
170 
171  lastContribution *= weight;
172  }
173  QL_ENSURE(i<maxIterations_,
174  i << " iterations have been not enough to reach "
175  << "the required " << relativeAccuracy_
176  << " accuracy. The " << io::ordinal(i)
177  << " addendum was " << lastContribution
178  << " while the running sum was " << results_.value);
179  }
