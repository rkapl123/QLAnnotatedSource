31  LogNormalFwdRateEulerConstrained::LogNormalFwdRateEulerConstrained(
32  const ext::shared_ptr<MarketModel>& marketModel,
33  const BrownianGeneratorFactory& factory,
34  const std::vector<Size>& numeraires,
35  Size initialStep)
36  : marketModel_(marketModel),
37  numeraires_(numeraires),
38  initialStep_(initialStep),
39  numberOfRates_(marketModel->numberOfRates()),
40  numberOfFactors_(marketModel_->numberOfFactors()),
41  curveState_(marketModel->evolution().rateTimes()),
42  forwards_(marketModel->initialRates()),
43  displacements_(marketModel->displacements()),
44  logForwards_(numberOfRates_), initialLogForwards_(numberOfRates_),
45  drifts1_(numberOfRates_), initialDrifts_(numberOfRates_),
46  brownians_(numberOfFactors_), correlatedBrownians_(numberOfRates_),
47  alive_(marketModel->evolution().firstAliveRate())
48  {
49  checkCompatibility(marketModel->evolution(), numeraires);
50 
51  Size steps = marketModel->evolution().numberOfSteps();
52 
53  generator_ = factory.create(numberOfFactors_, steps-initialStep_);
54  currentStep_ = initialStep_;
55 
56  calculators_.reserve(steps);
57  variances_.reserve(steps);
58  fixedDrifts_.reserve(steps);
59  for (Size j=0; j<steps; ++j) {
60  const Matrix& A = marketModel_->pseudoRoot(j);
61  calculators_.push_back(LMMDriftCalculator(A,
62  displacements_,
63  marketModel->evolution().rateTaus(),
64  numeraires[j],
65  alive_[j]));
66  std::vector<Real> fixed(numberOfRates_);
67  std::vector<Real> variances(numberOfRates_);
68  for (Size k=0; k<numberOfRates_; ++k) {
69  Real variance =
70  std::inner_product(A.row_begin(k), A.row_end(k),
71  A.row_begin(k), 0.0);
72  variances[k] = variance;
73  fixed[k] = -0.5*variance;
74  }
75  variances_.push_back(variances);
76  fixedDrifts_.push_back(fixed);
77  }
78 
79  setForwards(marketModel_->initialRates());
80  }

100  void LogNormalFwdRateEulerConstrained::setConstraintType(
101  const std::vector<Size>& startIndexOfSwapRate,
102  const std::vector<Size>& endIndexOfSwapRate)
103  {
104  QL_REQUIRE(startIndexOfSwapRate.size() == numeraires_.size(),
105  "Size mismatch in constraint specification.");
106  QL_REQUIRE(endIndexOfSwapRate.size() == numeraires_.size(), "Size mismatch in constraint specification.");
107 
108  startIndexOfSwapRate_=startIndexOfSwapRate;
109  endIndexOfSwapRate_ = endIndexOfSwapRate;
110 
111  covariances_.clear();
112  covariances_.reserve(startIndexOfSwapRate_.size());
113 
114  std::vector<Real> covariances(numberOfRates_);
115 
116  for (Size i=0; i < startIndexOfSwapRate_.size(); ++i) {
117  QL_REQUIRE(startIndexOfSwapRate_[i]+1 == endIndexOfSwapRate_[i],
118  "constrained euler currently only implemented for forward rates");
119 
120  const Matrix& A = marketModel_->pseudoRoot(currentStep_);
121 
122  for (Size j=0; j < numberOfRates_; ++j) {
123  Real cov=0.0;
124  for (Size k=0; k < numberOfFactors_; ++k)
125  cov += A[startIndexOfSwapRate_[i]][k]*A[j][k];
126  covariances[j] = cov;
127 
128  }
129  covariances_.push_back(covariances);
130  }

160  Real LogNormalFwdRateEulerConstrained::advanceStep()
161  {
162  // we're going from T1 to T2
163 
164  // a) compute drifts D1 at T1;
165  if (currentStep_ > initialStep_) {
166  calculators_[currentStep_].compute(forwards_, drifts1_);
167  } else {
168  std::copy(initialDrifts_.begin(), initialDrifts_.end(),
169  drifts1_.begin());
170  }
171 
172  // b) evolve forwards up to T2 using D1;
173  Real weight = generator_->nextStep(brownians_);
174  const Matrix& A = marketModel_->pseudoRoot(currentStep_);
175  const std::vector<Real>& fixedDrift = fixedDrifts_[currentStep_];
176 
177  Size alive = alive_[currentStep_];
178  for (Size i=alive; i<numberOfRates_; i++) {
179  logForwards_[i] += drifts1_[i] + fixedDrift[i];
180  logForwards_[i] +=
181  std::inner_product(A.row_begin(i), A.row_end(i),
182  brownians_.begin(), 0.0);
183  }
184 
185  // check constraint active
186  if (isConstraintActive_[currentStep_]) {
187  Size index = startIndexOfSwapRate_[currentStep_];
188 
189  // compute error
190  Real requiredShift = rateConstraints_[currentStep_] - logForwards_[index];
191 
192  Real multiplier = requiredShift/variances_[currentStep_][index];
193 
194  // now shift each rate by multiplier * weighting of index rate
195  // across the step
196  for (Size i=alive; i<numberOfRates_; i++) {
197  // we only need a small part of cov matrix
198  logForwards_[i] += multiplier*covariances_[currentStep_][i];
199  }
200 
201  // now we need to calculate the LR shift, we have moved Gaussian Z_k by multiplier * a_{index,k}
202  // divide original density by density of shifted normal
203  Real weightsEffect = 1.0;
204 
205  CumulativeNormalDistribution phi;
206  for (Size k=0; k < numberOfFactors_; k++) {
207  Real shift = multiplier * A[index][k];
208  Real originalDensity = phi.derivative(brownians_[k]+shift);
209  // the density of the draw after changes in original measure
210  Real newDensity = phi.derivative(brownians_[k]);
211  // the density of the draw after changes in new measure, shifts cancel
212  weightsEffect*= originalDensity/newDensity;
213  }
214 
215  weight *= weightsEffect;
216  }
217 
218  for (Size i=alive; i<numberOfRates_; i++)
219  forwards_[i] = std::exp(logForwards_[i]) - displacements_[i];
220 
221  // c) update curve state
222  curveState_.setOnForwardRates(forwards_);
223 
224  ++currentStep_;
225 
226  return weight;
227  }
