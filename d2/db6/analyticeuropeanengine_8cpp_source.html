41  void AnalyticEuropeanEngine::calculate() const {
42 
43  // if the discount curve is not specified, we default to the
44  // risk free rate curve embedded within the GBM process
45  ext::shared_ptr<YieldTermStructure> discountPtr =
46  discountCurve_.empty() ?
47  process_->riskFreeRate().currentLink() :
48  discountCurve_.currentLink();
49 
50  QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
51  "not an European option");
52 
53  ext::shared_ptr<StrikedTypePayoff> payoff =
54  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
55  QL_REQUIRE(payoff, "non-striked payoff given");
56 
57  Real variance =
58  process_->blackVolatility()->blackVariance(
59  arguments_.exercise->lastDate(),
60  payoff->strike());
61  DiscountFactor dividendDiscount =
62  process_->dividendYield()->discount(
63  arguments_.exercise->lastDate());
64  DiscountFactor df = discountPtr->discount(arguments_.exercise->lastDate());
65  DiscountFactor riskFreeDiscountForFwdEstimation =
66  process_->riskFreeRate()->discount(arguments_.exercise->lastDate());
67  Real spot = process_->stateVariable()->value();
68  QL_REQUIRE(spot > 0.0, "negative or null underlying given");
69  Real forwardPrice = spot * dividendDiscount / riskFreeDiscountForFwdEstimation;
70 
71  BlackCalculator black(payoff, forwardPrice, std::sqrt(variance),df);
72 
73 
74  results_.value = black.value();
75  results_.delta = black.delta(spot);
76  results_.deltaForward = black.deltaForward();
77  results_.elasticity = black.elasticity(spot);
78  results_.gamma = black.gamma(spot);
79 
80  DayCounter rfdc = discountPtr->dayCounter();
81  DayCounter divdc = process_->dividendYield()->dayCounter();
82  DayCounter voldc = process_->blackVolatility()->dayCounter();
83  Time t = rfdc.yearFraction(process_->riskFreeRate()->referenceDate(),
84  arguments_.exercise->lastDate());
85  results_.rho = black.rho(t);
86 
87  t = divdc.yearFraction(process_->dividendYield()->referenceDate(),
88  arguments_.exercise->lastDate());
89  results_.dividendRho = black.dividendRho(t);
90 
91  t = voldc.yearFraction(process_->blackVolatility()->referenceDate(),
92  arguments_.exercise->lastDate());
93  results_.vega = black.vega(t);
94  try {
95  results_.theta = black.theta(spot, t);
96  results_.thetaPerDay =
97  black.thetaPerDay(spot, t);
98  } catch (Error&) {
99  results_.theta = Null<Real>();
100  results_.thetaPerDay = Null<Real>();
101  }
102 
103  results_.strikeSensitivity = black.strikeSensitivity();
104  results_.itmCashProbability = black.itmCashProbability();
105 
106  Real tte = process_->blackVolatility()->timeFromReference(arguments_.exercise->lastDate());
107  results_.additionalResults["spot"] = spot;
108  results_.additionalResults["dividendDiscount"] = dividendDiscount;
109  results_.additionalResults["riskFreeDiscount"] = riskFreeDiscountForFwdEstimation;
110  results_.additionalResults["forward"] = forwardPrice;
111  results_.additionalResults["strike"] = payoff->strike();
112  results_.additionalResults["volatility"] = std::sqrt(variance / tte);
113  results_.additionalResults["timeToExpiry"] = tte;
114  }
115 
116 }
