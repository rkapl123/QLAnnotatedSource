46  Real BarrierPathPricer::operator()(const Path& path) const {
47  static Size null = Null<Size>();
48  Size n = path.length();
49  QL_REQUIRE(n>1, "the path cannot be empty");
50 
51  bool isOptionActive = false;
52  Size knockNode = null;
53  Real asset_price = path.front();
54  Real new_asset_price;
55  Real x, y;
56  Volatility vol;
57  TimeGrid timeGrid = path.timeGrid();
58  Time dt;
59  std::vector<Real> u = sequenceGen_.nextSequence().value;
60  Size i;
61 
62  switch (barrierType_) {
63  case Barrier::DownIn:
64  isOptionActive = false;
65  for (i = 0; i < n-1; i++) {
66  new_asset_price = path[i+1];
67  // terminal or initial vol?
68  vol = diffProcess_->diffusion(timeGrid[i],asset_price);
69  dt = timeGrid.dt(i);
70 
71  x = std::log(new_asset_price / asset_price);
72  y = 0.5*(x - std::sqrt (x*x - 2*vol*vol*dt*std::log(u[i])));
73  y = asset_price * std::exp(y);
74  if (y <= barrier_) {
75  isOptionActive = true;
76  if (knockNode == null)
77  knockNode = i+1;
78  }
79  asset_price = new_asset_price;
80  }
81  break;
82  case Barrier::UpIn:
83  isOptionActive = false;
84  for (i = 0; i < n-1; i++) {
85  new_asset_price = path[i+1];
86  // terminal or initial vol?
87  vol = diffProcess_->diffusion(timeGrid[i],asset_price);
88  dt = timeGrid.dt(i);
89 
90  x = std::log(new_asset_price / asset_price);
91  y = 0.5*(x + std::sqrt(x*x - 2*vol*vol*dt*std::log((1-u[i]))));
92  y = asset_price * std::exp(y);
93  if (y >= barrier_) {
94  isOptionActive = true;
95  if (knockNode == null)
96  knockNode = i+1;
97  }
98  asset_price = new_asset_price;
99  }
100  break;
101  case Barrier::DownOut:
102  isOptionActive = true;
103  for (i = 0; i < n-1; i++) {
104  new_asset_price = path[i+1];
105  // terminal or initial vol?
106  vol = diffProcess_->diffusion(timeGrid[i],asset_price);
107  dt = timeGrid.dt(i);
108 
109  x = std::log(new_asset_price / asset_price);
110  y = 0.5*(x - std::sqrt(x*x - 2*vol*vol*dt*std::log(u[i])));
111  y = asset_price * std::exp(y);
112  if (y <= barrier_) {
113  isOptionActive = false;
114  if (knockNode == null)
115  knockNode = i+1;
116  }
117  asset_price = new_asset_price;
118  }
119  break;
120  case Barrier::UpOut:
121  isOptionActive = true;
122  for (i = 0; i < n-1; i++) {
123  new_asset_price = path[i+1];
124  // terminal or initial vol?
125  vol = diffProcess_->diffusion(timeGrid[i],asset_price);
126  dt = timeGrid.dt(i);
127 
128  x = std::log(new_asset_price / asset_price);
129  y = 0.5*(x + std::sqrt(x*x - 2*vol*vol*dt*std::log((1-u[i]))));
130  y = asset_price * std::exp(y);
131  if (y >= barrier_) {
132  isOptionActive = false;
133  if (knockNode == null)
134  knockNode = i+1;
135  }
136  asset_price = new_asset_price;
137  }
138  break;
139  default:
140  QL_FAIL("unknown barrier type");
141  }
142 
143  if (isOptionActive) {
144  return payoff_(asset_price) * discounts_.back();
145  } else {
146  switch (barrierType_) {
147  case Barrier::UpIn:
148  case Barrier::DownIn:
149  return rebate_*discounts_.back();
150  case Barrier::UpOut:
151  case Barrier::DownOut:

155  }
