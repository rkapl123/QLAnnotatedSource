229  class base_cubic_spline {
230  public:
231  typedef Real argument_type;
232  typedef Real result_type;
233  typedef base_data data;
234  typedef base_data_table data_table;
235  typedef base_output_data output_data;
236  base_cubic_spline(const data &d, const data &d2,
237  const data_table& y, data_table &y2,
238  output_data &v) {
239  Size dim = d.first.size();
240  Size j = 1, k = 2, l = 3;
241  result_type &w = ((y2[dim] = y[1]) -= y[0]) /= d[0],
242  &u = ((y2[0] = y[2]) -= y[1]) /= d[1], &t = v[dim];
243  y2[1] = -d[1] / d2[0], v[1] = 6.0 * (u - w) / d2[0];
244  for(; k < dim; u = w, j = k, k = l, ++l) {
245  w = (y[l]-y[k])/d[k];
246  u = (u-w)*6.0;
247  (y2[k] = d[k]) /= ((t = -y2[j]) *= d[j]) -= d2[j];
248  (v[k] = (u += d[j] * v[j])) /= t;
249  }
250  y2[0] = y2[dim] = 0.0;
251  while (k != 0U) {
252  (y2[k-1] *= y2[l-1]) += v[k-1];
253  --k; --l;
254  }
255  }
256  };

506  // the data is checked and, in case of insufficient number of points,
507  // exception is thrown BEFORE the main body of interpolation begins
508  template <Size i>
509  void MultiCubicSpline<i>::set_shared_increments() const {
510  SplineGrid x(i), y(i);
511  Size k = 0, dim = 0;
512  for(Size j = 0; j < i; k = 0, ++j) {
513  const std::vector<Real> &v = grid_[j];
514  if((dim = v.size() - 1) > 2) {
515  std::vector<Real> tmp1(dim);
516  x[j].swap(tmp1);
517  std::vector<Real> tmp2(dim - 1);
518  y[j].swap(tmp2);
519  for(; k < dim; ++k) {
520  if((x[j][k] = v[k + 1] - v[k]) <= 0.0) break;
521  if (k != 0U)
522  y[j][k - 1] = 2.0 * (v[k + 1] - v[k - 1]);
523  }
524  }
525  QL_REQUIRE(dim >= 3,
526  "Dimension " << j
527  << " : not enough points for interpolation");
528  QL_REQUIRE(k >= dim,
529  "Dimension " << j << " : invalid data");
530  }
531 
532  typename c_splint::data tmp1(x), tmp2(y);
533  d_.swap(tmp1);
534  d2_.swap(tmp2);
535  }

537  #ifndef __DOXYGEN__
538  // the argument value is checked and, in out of boundaries case,
539  // exception is thrown BEFORE the main body of interpolation begins
540  template <Size i>
541  void MultiCubicSpline<i>::set_shared_coefficients(
542  const typename MultiCubicSpline<i>::argument_type &x) const {
543  for(Size j = 0; j < i; ++j) {
544  Size &k = i_[j], sz = grid_[j].size() - 1;
545  const std::vector<Real> &v = grid_[j];
546  if(x[j] < v[0] || x[j] >= v[sz]) {
547  QL_REQUIRE(ae_[j],
548  "Dimension " << j
549  << ": extrapolation is not allowed.");
550  a_[j] = 1.0, a2_[j] = b_[j] = b2_[j] = 0.0;
551  k = x[j] < v[0] ? 0 : sz;
552  }
553  else {
554  k = v[k] <= x[j] && x[j] < v[k + 1] ? k :
555  std::upper_bound(v.begin(),v.end(),x[j])-v.begin()-1;
556  Real h = v[k + 1] - v[k];
557  a_[j] = (v[k + 1] - x[j]) / h, b_[j] = (x[j] - v[k]) / h;
558  a2_[j] = (a_[j] * a_[j] * a_[j] - a_[j]) * h * h / 6.0,
559  b2_[j] = (b_[j] * b_[j] * b_[j] - b_[j]) * h * h / 6.0;
560  }
561  }
562  }
563  #endif
