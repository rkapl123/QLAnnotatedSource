233  class base_cubic_spline : public std::unary_function<Real,Real> {
234  public:
235  typedef base_data data;
236  typedef base_data_table data_table;
237  typedef base_output_data output_data;
238  base_cubic_spline(const data &d, const data &d2,
239  const data_table& y, data_table &y2,
240  output_data &v) {
241  Size dim = d.first.size();
242  Size j = 1, k = 2, l = 3;
243  result_type &w = ((y2[dim] = y[1]) -= y[0]) /= d[0],
244  &u = ((y2[0] = y[2]) -= y[1]) /= d[1], &t = v[dim];
245  y2[1] = -d[1] / d2[0], v[1] = 6.0 * (u - w) / d2[0];
246  for(; k < dim; u = w, j = k, k = l, ++l) {
247  w = (y[l]-y[k])/d[k];
248  u = (u-w)*6.0;
249  (y2[k] = d[k]) /= ((t = -y2[j]) *= d[j]) -= d2[j];
250  (v[k] = (u += d[j] * v[j])) /= t;
251  }
252  y2[0] = y2[dim] = 0.0;
253  while (k) {
254  (y2[k-1] *= y2[l-1]) += v[k-1];
255  --k; --l;
256  }
257  }
258  };

505  // the data is checked and, in case of insufficient number of points,
506  // exception is thrown BEFORE the main body of interpolation begins
507  template <Size i>
508  void MultiCubicSpline<i>::set_shared_increments() const {
509  SplineGrid x(i), y(i);
510  Size k = 0, dim = 0;
511  for(Size j = 0; j < i; k = 0, ++j) {
512  const std::vector<Real> &v = grid_[j];
513  if((dim = v.size() - 1) > 2) {
514  std::vector<Real> tmp1(dim);
515  x[j].swap(tmp1);
516  std::vector<Real> tmp2(dim - 1);
517  y[j].swap(tmp2);
518  for(; k < dim; ++k) {
519  if((x[j][k] = v[k + 1] - v[k]) <= 0.0) break;
520  if(k) y[j][k - 1] = 2.0 * (v[k + 1] - v[k - 1]);
521  }
522  }
523  QL_REQUIRE(dim >= 3,
524  "Dimension " << j
525  << " : not enough points for interpolation");
526  QL_REQUIRE(k >= dim,
527  "Dimension " << j << " : invalid data");
528  }
529 
530  typename c_splint::data tmp1(x), tmp2(y);
531  d_.swap(tmp1);
532  d2_.swap(tmp2);
533  }

536  // the argument value is checked and, in out of boundaries case,
537  // exception is thrown BEFORE the main body of interpolation begins
538  template <Size i>
539  void MultiCubicSpline<i>::set_shared_coefficients(
540  const typename MultiCubicSpline<i>::argument_type &x) const {
541  for(Size j = 0; j < i; ++j) {
542  Size &k = i_[j], sz = grid_[j].size() - 1;
543  const std::vector<Real> &v = grid_[j];
544  if(x[j] < v[0] || x[j] >= v[sz]) {
545  QL_REQUIRE(ae_[j],
546  "Dimension " << j
547  << ": extrapolation is not allowed.");
548  a_[j] = 1.0, a2_[j] = b_[j] = b2_[j] = 0.0;
549  k = x[j] < v[0] ? 0 : sz;
550  }
551  else {
552  k = v[k] <= x[j] && x[j] < v[k + 1] ? k :
553  std::upper_bound(v.begin(),v.end(),x[j])-v.begin()-1;
554  Real h = v[k + 1] - v[k];
555  a_[j] = (v[k + 1] - x[j]) / h, b_[j] = (x[j] - v[k]) / h;
556  a2_[j] = (a_[j] * a_[j] * a_[j] - a_[j]) * h * h / 6.0,
557  b2_[j] = (b_[j] * b_[j] * b_[j] - b_[j]) * h * h / 6.0;
558  }
559  }
560  }
561  #endif
