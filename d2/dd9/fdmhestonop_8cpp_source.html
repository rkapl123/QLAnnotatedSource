31  FdmHestonEquityPart::FdmHestonEquityPart(
32  const ext::shared_ptr<FdmMesher>& mesher,
33  const ext::shared_ptr<YieldTermStructure>& rTS,
34  const ext::shared_ptr<YieldTermStructure>& qTS,
35  const ext::shared_ptr<FdmQuantoHelper>& quantoHelper,
36  const ext::shared_ptr<LocalVolTermStructure>& leverageFct)
37  : varianceValues_(0.5*mesher->locations(1)),
38  dxMap_ (FirstDerivativeOp(0, mesher)),
39  dxxMap_(SecondDerivativeOp(0, mesher).mult(0.5*mesher->locations(1))),
40  mapT_ (0, mesher),
41  mesher_(mesher),
42  rTS_(rTS),
43  qTS_(qTS),
44  quantoHelper_(quantoHelper),
45  leverageFct_(leverageFct) {
46 
47  // on the boundary s_min and s_max the second derivative
48  // d^2V/dS^2 is zero and due to Ito's Lemma the variance term
49  // in the drift should vanish.
50  ext::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
51  FdmLinearOpIterator endIter = layout->end();
52  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
53  ++iter) {
54  if ( iter.coordinates()[0] == 0
55  || iter.coordinates()[0] == layout->dim()[0]-1) {
56  varianceValues_[iter.index()] = 0.0;
57  }
58  }
59  volatilityValues_ = Sqrt(2*varianceValues_);
60  }

62  void FdmHestonEquityPart::setTime(Time t1, Time t2) {
63  const Rate r = rTS_->forwardRate(t1, t2, Continuous).rate();
64  const Rate q = qTS_->forwardRate(t1, t2, Continuous).rate();
65 
66  L_ = getLeverageFctSlice(t1, t2);
67  const Array Lsquare = L_*L_;
68 
69  if (quantoHelper_ != 0) {
70  mapT_.axpyb(r - q - varianceValues_*Lsquare
71  - quantoHelper_->quantoAdjustment(
72  volatilityValues_*L_, t1, t2),
73  dxMap_, dxxMap_.mult(Lsquare), Array(1, -0.5*r));
74  }
75  else {
76  mapT_.axpyb(r - q - varianceValues_*Lsquare, dxMap_,
77  dxxMap_.mult(Lsquare), Array(1, -0.5*r));
78  }
79  }

81  Disposable<Array> FdmHestonEquityPart::getLeverageFctSlice(Time t1, Time t2)
82  const {
83  const ext::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
84  Array v(layout->size(), 1.0);
85 
86  if (!leverageFct_) {
87  return v;
88  }
89  const Real t = 0.5*(t1+t2);
90  const Time time = std::min(leverageFct_->maxTime(), t);
91 
92  const FdmLinearOpIterator endIter = layout->end();
93  for (FdmLinearOpIterator iter = layout->begin();
94  iter!=endIter; ++iter) {
95  const Size nx = iter.coordinates()[0];
96 
97  if (iter.coordinates()[1] == 0) {
98  const Real x = std::exp(mesher_->location(iter, 0));
99  const Real spot = std::min(leverageFct_->maxStrike(),
100  std::max(leverageFct_->minStrike(), x));
101  v[nx] = std::max(0.01, leverageFct_->localVol(time, spot, true));
102  }
103  else {
104  v[iter.index()] = v[nx];
105  }
106  }
107  return v;
108  }
