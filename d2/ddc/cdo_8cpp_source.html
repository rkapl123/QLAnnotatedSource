69  Real back = nominals_.back();
70  for (Size i = 0; i < n; i++)
71  nominals_.push_back(back);
72  }
73

75  "nominal size " << nominals_.size()
76  << " != basket size " << basket_.size());
77 
78  nominal_ = 0;
79  for (Size i = 0; i < nominals_.size(); i++) {
80  lgds_.push_back (nominals_[i] * (1.0 - recoveryRate_));
81  nominal_ += nominals_[i];

100  Real CDO::expectedTrancheLoss (Date d) const {
101  if (d <= basket_.front()->referenceDate())
102  return 0;
103 
104  vector<Real> defProb (basket_.size());
105  for (Size j = 0; j < basket_.size(); j++)
106  defProb[j] = basket_[j]->defaultProbability (d);
107 
108  LossDistBucketing op (nBuckets_, xMax_);
109  Distribution dist = copula_->integral (op, lgds_, defProb);
110 
111  return dist.trancheExpectedValue (xMin_, xMax_);
112

123  void CDO::performCalculations() const {
124 
125  QL_REQUIRE(!yieldTS_.empty(), "no yield term structure set");
126 
127  errorEstimate_ = Null<Real>();
128 
129  NPV_ = 0.0;
130  premiumValue_ = 0;
131  protectionValue_ = 0;
132  error_ = 0;
133 
134  /* Expectations e1 and e2 are portfolio loss given default,
135  i.e. with recovery already "bult in". Multiplication by
136  (1-r) is therefore not necessary, neither in premium nor
137  protection value calculation.
138  */
139 
140  Real e1 = 0;
141  Date today = yieldTS_->referenceDate();
142  if (premiumSchedule_[0] > today)
143  e1 = expectedTrancheLoss (premiumSchedule_[0]);
144 
145  for (Size i = 1; i < premiumSchedule_.size(); i++) {
146  Date d2 = premiumSchedule_[i];
147  if (d2 < today)
148  continue;
149 
150  Date d1 = premiumSchedule_[i-1];
151 
152  Date d, d0 = d1;
153  do {
154  d = NullCalendar().advance (d0 > today ? d0 : today,
155  integrationStep_);
156  if (d > d2)
157  d = d2;
158 
159  Real e2 = expectedTrancheLoss (d);
160 
161  premiumValue_ += (xMax_ - xMin_ - e2)
162  * premiumRate_ * dayCounter_.yearFraction (d0, d)
163  * yieldTS_->discount (d);
164 
165  if (e2 < e1) {
166  error_ ++;
167  }
168 
169  protectionValue_ -= (e2 - e1) * yieldTS_->discount (d);
170 
171  d0 = d;
172  e1 = e2;
173  }
174  while (d < d2);
175  }
176 
177  if (premiumSchedule_[0] >= today)
178  upfrontPremiumValue_ = (xMax_ - xMin_) * upfrontPremiumRate_ *
179  yieldTS_->discount(premiumSchedule_[0]);
180  else
181  upfrontPremiumValue_ = 0.0;
182 
183  if (!protectionSeller_) {
184  premiumValue_ *= -1;
185  upfrontPremiumValue_ *= -1;
186  protectionValue_ *= -1;
187  }
188 
189  NPV_ = premiumValue_ + protectionValue_ + upfrontPremiumValue_;
190  }
