53  void MarketModelComposite::add(
54  const Clone<MarketModelMultiProduct>& product,
55  Real multiplier) {
56  QL_REQUIRE(!finalized_, "product already finalized");
57  EvolutionDescription d = product->evolution();
58  if (!components_.empty()) {
59  // enforce preconditions
60  EvolutionDescription d1 =
61  components_.front().product->evolution();
62  const std::vector<Time>& rateTimes1 = d1.rateTimes();
63  const std::vector<Time>& rateTimes2 = d.rateTimes();
64  QL_REQUIRE(rateTimes1.size() == rateTimes2.size() &&
65  std::equal(rateTimes1.begin(), rateTimes1.end(),
66  rateTimes2.begin()),
67  "incompatible rate times");
68  }
69  components_.push_back(SubProduct());
70  components_.back().product = product;
71  components_.back().multiplier = multiplier;
72  components_.back().done = false;
73  allEvolutionTimes_.push_back(d.evolutionTimes());
74  }

82  void MarketModelComposite::finalize() {
83  QL_REQUIRE(!finalized_, "product already finalized");
84  QL_REQUIRE(!components_.empty(), "no sub-product provided");
85 
86  // fetch the rate times from the first subproduct (we checked
87  // they're all the same)
88  EvolutionDescription description =
89  components_.front().product->evolution();
90  rateTimes_ = description.rateTimes();
91 
92  mergeTimes(allEvolutionTimes_, evolutionTimes_, isInSubset_);
93 
94  std::vector<Time> allCashflowTimes;
95 
96  // now, for each subproduct...
97  iterator i;
98  for (i=components_.begin(); i!=components_.end(); ++i) {
99  EvolutionDescription d = i->product->evolution();
100  // ...collect all possible cash-flow times...
101  const std::vector<Time>& cashflowTimes =
102  i->product->possibleCashFlowTimes();
103  allCashflowTimes.insert(allCashflowTimes.end(),
104  cashflowTimes.begin(),
105  cashflowTimes.end());
106  // ...allocate working vectors...
107  i->numberOfCashflows =
108  std::vector<Size>(i->product->numberOfProducts());
109  i->cashflows =
110  std::vector<std::vector<CashFlow> >(
111  i->product->numberOfProducts(),
112  std::vector<CashFlow>(i->product
113  ->maxNumberOfCashFlowsPerProductPerStep()));
114  }
115 
116  // all information having been collected, we can sort and
117  // compact the vector of all cash-flow times...
118  std::sort(allCashflowTimes.begin(), allCashflowTimes.end());
119  std::vector<Time>::iterator end = std::unique(allCashflowTimes.begin(),
120  allCashflowTimes.end());
121  //std::copy(allCashflowTimes.begin(), end,
122  // std::back_inserter(cashflowTimes_));
123  cashflowTimes_.insert(cashflowTimes_.end(),
124  allCashflowTimes.begin(), end);
125  // ...and map each product's cash-flow time into the total vector.
126  for (i=components_.begin(); i!=components_.end(); ++i) {
127  const std::vector<Time>& productTimes =
128  i->product->possibleCashFlowTimes();
129  i->timeIndices = std::vector<Size>(productTimes.size());
130  for (Size j=0; j<productTimes.size(); ++j) {
131  i->timeIndices[j] =
132  std::find(cashflowTimes_.begin(), cashflowTimes_.end(),
133  productTimes[j]) - cashflowTimes_.begin();
134  }
135  }
136 
137  evolution_ = EvolutionDescription(rateTimes_, evolutionTimes_);
138 
139  // all done.
140  finalized_ = true;
141  }
