36 void DiscountingBondEngine::calculate() const {
37 QL_REQUIRE(!discountCurve_.empty(),
38 "discounting term structure handle is empty");
39
40 results_.valuationDate = (*discountCurve_)->referenceDate();
41
42 bool includeRefDateFlows = includeSettlementDateFlows_ ? // NOLINT(readability-implicit-bool-conversion)
43 *includeSettlementDateFlows_ :
44 Settings::instance().includeReferenceDateEvents();
45
46 results_.value = CashFlows::npv(arguments_.cashflows,
47 **discountCurve_,
48 includeRefDateFlows,
49 results_.valuationDate,
50 results_.valuationDate);
51
52 // a bond's cashflow on settlement date is never taken into
53 // account, so we might have to play it safe and recalculate
54 if (!includeRefDateFlows
55 && results_.valuationDate == arguments_.settlementDate) {
56 // same parameters as above, we can avoid another call
57 results_.settlementValue = results_.value;
58 } else {
59 // no such luck
60 results_.settlementValue =
61 CashFlows::npv(arguments_.cashflows,
62 **discountCurve_,
63 false,
64 arguments_.settlementDate,
65 arguments_.settlementDate);
66 }
67 }
