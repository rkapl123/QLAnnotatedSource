33  void DiscountingBondEngine::calculate() const {
34  QL_REQUIRE(!discountCurve_.empty(),
35  "discounting term structure handle is empty");
36 
37  results_.valuationDate = (*discountCurve_)->referenceDate();
38 
39  bool includeRefDateFlows = includeSettlementDateFlows_ ? // NOLINT(readability-implicit-bool-conversion)
40  *includeSettlementDateFlows_ :
41  Settings::instance().includeReferenceDateEvents();
42 
43  results_.value = CashFlows::npv(arguments_.cashflows,
44  **discountCurve_,
45  includeRefDateFlows,
46  results_.valuationDate,
47  results_.valuationDate);
48 
49  // a bond's cashflow on settlement date is never taken into
50  // account, so we might have to play it safe and recalculate
51  if (!includeRefDateFlows
52  && results_.valuationDate == arguments_.settlementDate) {
53  // same parameters as above, we can avoid another call
54  results_.settlementValue = results_.value;
55  } else {
56  // no such luck
57  results_.settlementValue =
58  CashFlows::npv(arguments_.cashflows,
59  **discountCurve_,
60  false,
61  arguments_.settlementDate,
62  arguments_.settlementDate);
63  }
64  }
