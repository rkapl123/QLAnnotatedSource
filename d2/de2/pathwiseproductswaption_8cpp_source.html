54  bool MarketModelPathwiseCoterminalSwaptionsDeflated::nextTimeStep(
55  const CurveState& currentState,
56  std::vector<Size>& numberCashFlowsThisStep,
57  std::vector<std::vector<MarketModelPathwiseMultiProduct::CashFlow> >& cashFlowsGenerated)
58  {
59  Rate swapRate = currentState.coterminalSwapRate(currentIndex_);
60  cashFlowsGenerated[currentIndex_][0].timeIndex = currentIndex_;
61 
62  Real annuity = currentState.coterminalSwapAnnuity(currentIndex_,currentIndex_);
63  cashFlowsGenerated[currentIndex_][0].amount[0] =
64  (swapRate-strikes_[currentIndex_])*annuity;
65 
66  std::fill(numberCashFlowsThisStep.begin(),
67  numberCashFlowsThisStep.end(),0);
68 
69  if ( cashFlowsGenerated[currentIndex_][0].amount[0] >0)
70  {
71  numberCashFlowsThisStep[currentIndex_] = 1;
72  for (Size i=1; i <= numberRates_; ++i)
73  cashFlowsGenerated[currentIndex_][0].amount[i] =0;
74 
75  for (Size k=currentIndex_; k < numberRates_; ++k)
76  {
77  cashFlowsGenerated[currentIndex_][0].amount[k+1] = (rateTimes_[k+1]-rateTimes_[k])*currentState.discountRatio(k+1,currentIndex_);
78 
79  Real multiplier = - (rateTimes_[k+1]-rateTimes_[k])*currentState.discountRatio(k+1,k);
80 
81  for (Size l=k; l < numberRates_; ++l)
82  cashFlowsGenerated[currentIndex_][0].amount[k+1] +=(currentState.forwardRate(l)-strikes_[currentIndex_])*(rateTimes_[l+1]-rateTimes_[l])
83  * multiplier*currentState.discountRatio(l+1,currentIndex_);
84  }
85  }
86  ++currentIndex_;
87  return (currentIndex_ == strikes_.size());
88  }

163  bool MarketModelPathwiseCoterminalSwaptionsNumericalDeflated::nextTimeStep(
164  const CurveState& currentState,
165  std::vector<Size>& numberCashFlowsThisStep,
166  std::vector<std::vector<MarketModelPathwiseMultiProduct::CashFlow> >& cashFlowsGenerated)
167  {
168  Rate swapRate = currentState.coterminalSwapRate(currentIndex_);
169  cashFlowsGenerated[currentIndex_][0].timeIndex = currentIndex_;
170 
171  Real annuity = currentState.coterminalSwapAnnuity(currentIndex_,currentIndex_);
172  cashFlowsGenerated[currentIndex_][0].amount[0] =
173  (swapRate-strikes_[currentIndex_])*annuity;
174 
175  std::fill(numberCashFlowsThisStep.begin(),
176  numberCashFlowsThisStep.end(),0);
177 
178  if ( cashFlowsGenerated[currentIndex_][0].amount[0] >0)
179  {
180  numberCashFlowsThisStep[currentIndex_] = 1;
181  for (Size i=1; i <= numberRates_; ++i)
182  cashFlowsGenerated[currentIndex_][0].amount[i] =0;
183 
184  for (Size k=currentIndex_; k < numberRates_; ++k)
185  {
186  forwards_ = currentState.forwardRates();
187  forwards_[k] += bumpSize_;
188  up_.setOnForwardRates(forwards_);
189 
190  forwards_[k] -= bumpSize_;
191  forwards_[k] -= bumpSize_;
192  down_.setOnForwardRates(forwards_);
193 
194  Real upSR = up_.coterminalSwapRate(currentIndex_);
195  Real upAnnuity = up_.coterminalSwapAnnuity(currentIndex_,currentIndex_);
196  Real upValue = (upSR - strikes_[currentIndex_])* upAnnuity;
197 
198  Real downSR = down_.coterminalSwapRate(currentIndex_);
199  Real downAnnuity = down_.coterminalSwapAnnuity(currentIndex_,currentIndex_);
200  Real downValue = (downSR - strikes_[currentIndex_])* downAnnuity;
201 
202  Real deriv = (upValue - downValue)/(2.0*bumpSize_);
203 
204  cashFlowsGenerated[currentIndex_][0].amount[k+1] = deriv;
205 
206  }
207  }
208  ++currentIndex_;
209  return (currentIndex_ == strikes_.size());
210  }
