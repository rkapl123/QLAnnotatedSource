34  Distribution::Distribution (int nBuckets, Real xmin, Real xmax)
35  //-------------------------------------------------------------------------
36  : size_(nBuckets),
37  xmin_(xmin), xmax_(xmax), count_(nBuckets),
38  x_(nBuckets,0), dx_(nBuckets,0),
39  density_(nBuckets,0),
40  cumulativeDensity_(nBuckets,0),
41  excessProbability_(nBuckets,0),
42  cumulativeExcessProbability_(nBuckets,0),
43  average_(nBuckets,0),
44  overFlow_(0), underFlow_(0),
45  isNormalized_(false) {
46  for (int i = 0; i < nBuckets; i++) {
47  dx_[i] = (xmax - xmin) / nBuckets;
48  x_[i] = (i == 0 ? xmin : x_[i-1] + dx_[i-1]);
49  }
50  // ensure we match exactly the domain, otherwise we might fail the
51  // locate test because of precission mismatches
52  dx_.back() = xmax - x_.back();
53  }

56  int Distribution::locate (Real x) {
57  //-------------------------------------------------------------------------
58  QL_REQUIRE ((x >= x_.front() || close(x, x_.front())) &&
59  (x <= x_.back() + dx_.back()
60  || close(x, x_.back() + dx_.back())),
61  "coordinate " << x
62  << " out of range [" << x_.front() << "; "
63  << x_.back() + dx_.back() << "]");
64  for (Size i = 0; i < x_.size(); i++) {
65  if (x_[i] > x)
66  return i - 1;
67  }
68  return x_.size() - 1;
69  }

79  void Distribution::add (Real value) {
80  //-------------------------------------------------------------------------
81  isNormalized_ = false;
82  if (value < x_.front()) underFlow_++;
83  else {
84  for (Size i = 0; i < count_.size(); i++) {
85  if (x_[i] + dx_[i] > value) {
86  count_[i]++;
87  average_[i] += value;
88  return;
89  }
90  }
91  overFlow_++;
92  }
93  }

112  void Distribution::normalize () {
113  //-------------------------------------------------------------------------
114  if (isNormalized_)
115  return;
116 
117  int count = underFlow_ + overFlow_;
118  for (int i = 0; i < size_; i++)
119  count += count_[i];
120 
121  excessProbability_[0] = 1.0;
122  cumulativeExcessProbability_[0] = 0.0;
123  for (int i = 0; i < size_; i++) {
124  if (count > 0) {
125  density_[i] = 1.0 / dx_[i] * count_[i] / count;
126  if (count_[i] > 0)
127  average_[i] /= count_[i];
128  }
129  if (density_[i] == 0.0)
130  average_[i] = x_[i] + dx_[i]/2;
131 
132  cumulativeDensity_[i] = density_[i] * dx_[i];
133  if (i > 0) {
134  cumulativeDensity_[i] += cumulativeDensity_[i-1];
135  excessProbability_[i] = 1.0 - cumulativeDensity_[i-1];
136 // excessProbability_[i] = excessProbability_[i-1]
137 // - density_[i-1] * dx_[i-1];
138 // cumulativeExcessProbability_[i]
139 // = (excessProbability_[i-1] +
140 // excessProbability_[i]) / 2 * dx_[i-1]
141 // + cumulativeExcessProbability_[i-1];
142  cumulativeExcessProbability_[i]
143  = excessProbability_[i-1] * dx_[i-1]
144  + cumulativeExcessProbability_[i-1];
145  }
146  }
147 
148  isNormalized_ = true;
149  }

221  Real Distribution::cumulativeDensity (Real x) {
222  //-------------------------------------------------------------------------
223  Real tiny = dx_.back() * 1e-3;
224  QL_REQUIRE (x > 0, "x must be positive");
225  normalize();
226  for (int i = 0; i < size_; i++) {
227  if (x_[i] + dx_[i] + tiny >= x)
228  return ((x - x_[i]) * cumulativeDensity_[i]
229  + (x_[i] + dx_[i] - x) * cumulativeDensity_[i-1]) / dx_[i];
230  }
231  QL_FAIL ("x = " << x << " beyond distribution cutoff "
232  << x_.back() + dx_.back());
233  }

236  // Dangerous to perform calls to members after this; transform and clone?
237  void Distribution::tranche (Real attachmentPoint, Real detachmentPoint) {
238  //-------------------------------------------------------------------------
239  QL_REQUIRE (attachmentPoint < detachmentPoint,
240  "attachment >= detachment point");
241  QL_REQUIRE (x_.back() > attachmentPoint &&
242  x_.back()+dx_.back() >= detachmentPoint,
243  "attachment or detachment too large");
244 
245  normalize();
246 
247  // shift
248  while (x_[0] < attachmentPoint) {
249  x_.erase(x_.begin());
250  dx_.erase(dx_.begin());
251  count_.erase(count_.begin());
252  density_.erase(density_.begin());
253  cumulativeDensity_.erase(cumulativeDensity_.begin());
254  excessProbability_.erase(excessProbability_.begin());
255  }
256 
257  // remove losses over detachment point:
258  auto detachPosit = std::find_if(x_.begin(), x_.end(), greater_than<Real>(detachmentPoint));
259  if(detachPosit != x_.end())
260  x_.erase(detachPosit + 1, x_.end());
261 
262  size_ = x_.size();
263  cumulativeDensity_.erase(cumulativeDensity_.begin() + size_,
264  cumulativeDensity_.end());
265  cumulativeDensity_.back() = 1.;
266  count_.erase(count_.begin() + size_, count_.end());
267  dx_.erase(dx_.begin() + size_, dx_.end());
268 
269  // truncate
270  for (double& i : x_) {
271  i = std::min(std::max(i - attachmentPoint, 0.), detachmentPoint - attachmentPoint);
272  }
273 
274  density_.clear();
275  excessProbability_.clear();
276  cumulativeExcessProbability_.clear(); //? reuse?
277  density_.push_back((cumulativeDensity_[0]-0.)/dx_[0]);
278  excessProbability_.push_back(1.);
279  for(Integer i=1; i<size_-1; i++) {
280  excessProbability_.push_back(1.-cumulativeDensity_[i-1]);
281  density_.push_back((cumulativeDensity_[i]-
282  cumulativeDensity_[i-1])/dx_[i]);
283  }
284  excessProbability_.push_back(1.-cumulativeDensity_.back());
285  density_.push_back((1.-cumulativeDensity_.back())/dx_.back());
286  }

289  Distribution ManipulateDistribution::convolve (const Distribution& d1,
290  const Distribution& d2) {
291  //-------------------------------------------------------------------------
292  // force equal constant bucket sizes
293  QL_REQUIRE (d1.dx_[0] == d2.dx_[0], "bucket sizes differ in d1 and d2");
294  for (Size i = 1; i < d1.size(); i++)
295  QL_REQUIRE (d1.dx_[i] == d1.dx_[i-1], "bucket size varies in d1");
296  for (Size i = 1; i < d2.size(); i++)
297  QL_REQUIRE (d2.dx_[i] == d2.dx_[i-1], "bucket size varies in d2");
298 
299  // force offset 0
300  QL_REQUIRE (d1.xmin_ == 0.0 && d2.xmin_ == 0.0,
301  "distributions offset larger than 0");
302 
303  Distribution dist(d1.size() + d2.size() - 1,
304  0.0, // assuming both distributions have xmin = 0
305  d1.xmax_ + d2.xmax_);
306 
307  for (Size i1 = 0; i1 < d1.size(); i1++) {
308  Real dx = d1.dx_[i1];
309  for (Size i2 = 0; i2 < d2.size(); i2++)
310  dist.density_[i1+i2] = d1.density_[i1] * d2.density_[i2] * dx;
311  }
312 
313  // update cumulated and excess
314  dist.excessProbability_[0] = 1.0;
315  for (Size i = 0; i < dist.size(); i++) {
316  dist.cumulativeDensity_[i] = dist.density_[i] * dist.dx_[i];
317  if (i > 0) {
318  dist.cumulativeDensity_[i] += dist.cumulativeDensity_[i-1];
319  dist.excessProbability_[i] = dist.excessProbability_[i-1]
320  - dist.density_[i-1] * dist.dx_[i-1];
321  }
322  }
323 
324  return dist;
325  }

329  Real Distribution::expectedShortfall (Real percValue) {
330  //-------------------------------------------------------------------------
331  QL_REQUIRE(percValue >= 0. && percValue <= 1.,
332  "Incorrect percentile");
333  normalize();
334  Real expected = 0;
335  Integer iVal = locate(confidenceLevel(percValue));
336 
337  if(iVal == size_-1) return x_.back();
338 
339  for (int i = iVal; i < size_; i++)
340  expected += x_[i] *
341  (cumulativeDensity_[i] - cumulativeDensity_[i-1]);
342  return expected/(1.-cumulativeDensity_.at(iVal));
343  }
