26  CallSpecifiedMultiProduct::CallSpecifiedMultiProduct(
27  const Clone<MarketModelMultiProduct>& underlying,
28  const Clone<ExerciseStrategy<CurveState> >& strategy,
29  const Clone<MarketModelMultiProduct>& rebate)
30  : underlying_(underlying),
31  strategy_(strategy),
32  rebate_(rebate),
33  callable_(true)
34  {
35 
36  Size products = underlying_->numberOfProducts();
37  EvolutionDescription d1 = underlying->evolution();
38  const std::vector<Time>& rateTimes1 = d1.rateTimes();
39  const std::vector<Time>& evolutionTimes1 = d1.evolutionTimes();
40  const std::vector<Time>& exerciseTimes = strategy->exerciseTimes();
41 
42  if (!rebate_.empty())
43  {
44  EvolutionDescription d2 = rebate_->evolution();
45  const std::vector<Time>& rateTimes2 = d2.rateTimes();
46  QL_REQUIRE(rateTimes1.size() == rateTimes2.size() &&
47  std::equal(rateTimes1.begin(), rateTimes1.end(),
48  rateTimes2.begin()),
49  "incompatible rate times");
50  }
51  else
52  {
53  EvolutionDescription description(rateTimes1, exerciseTimes);
54  Matrix amounts(products, exerciseTimes.size(), 0.0);
55 
56  rebate_ = MarketModelCashRebate(description, exerciseTimes,
57  amounts, products);
58  }
59 
60  std::vector<Time> mergedEvolutionTimes;
61  std::vector<std::vector<Time> > allEvolutionTimes(4);
62  allEvolutionTimes[0] = evolutionTimes1;
63  allEvolutionTimes[1] = exerciseTimes;
64  allEvolutionTimes[2] = rebate_->evolution().evolutionTimes();
65  allEvolutionTimes[3] = strategy->relevantTimes();
66 
67  mergeTimes(allEvolutionTimes,
68  mergedEvolutionTimes,
69  isPresent_);
70 
71  // TODO: add relevant rates
72  evolution_ = EvolutionDescription(rateTimes1, mergedEvolutionTimes);
73 
74  cashFlowTimes_ = underlying_->possibleCashFlowTimes();
75  rebateOffset_ = cashFlowTimes_.size();
76  const std::vector<Time> rebateTimes = rebate_->possibleCashFlowTimes();
77  cashFlowTimes_.insert(cashFlowTimes_.end(),
78  rebateTimes.begin(), rebateTimes.end());
79 
80  dummyCashFlowsThisStep_ = std::vector<Size>(products, 0);
81  Size n = rebate_->maxNumberOfCashFlowsPerProductPerStep();
82  dummyCashFlowsGenerated_ =
83  std::vector<std::vector<CashFlow> >(products,
84  std::vector<CashFlow>(n));
85  }

126  bool CallSpecifiedMultiProduct::nextTimeStep(
127  const CurveState& currentState,
128  std::vector<Size>& numberCashFlowsThisStep,
129  std::vector<std::vector<CashFlow> >& cashFlowsGenerated)
130  {
131 
132  bool isUnderlyingTime = isPresent_[0][currentIndex_];
133  bool isExerciseTime = isPresent_[1][currentIndex_];
134  bool isRebateTime = isPresent_[2][currentIndex_];
135  bool isStrategyRelevantTime = isPresent_[3][currentIndex_];
136 
137  bool done = false;
138 
139  if (!wasCalled_ && isStrategyRelevantTime)
140  strategy_->nextStep(currentState);
141 
142 
143  if (!wasCalled_ && isExerciseTime && callable_)
144  wasCalled_ = strategy_->exercise(currentState);
145 
146  if (wasCalled_)
147  {
148  if (isRebateTime)
149  {
150  done = rebate_->nextTimeStep(currentState,
151  numberCashFlowsThisStep,
152  cashFlowsGenerated);
153  for (Size i=0; i<numberCashFlowsThisStep.size(); ++i)
154  for (Size j=0; j<numberCashFlowsThisStep[i]; ++j)
155  cashFlowsGenerated[i][j].timeIndex += rebateOffset_;
156  }
157  }
158  else
159  {
160  if (isRebateTime)
161  rebate_->nextTimeStep(currentState,
162  dummyCashFlowsThisStep_,
163  dummyCashFlowsGenerated_);
164  if (isUnderlyingTime)
165  done = underlying_->nextTimeStep(currentState,
166  numberCashFlowsThisStep,
167  cashFlowsGenerated);
168  }
169 
170  ++currentIndex_;
171  return done || currentIndex_ == evolution_.evolutionTimes().size();
172  }
