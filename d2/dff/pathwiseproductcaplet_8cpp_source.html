69  bool MarketModelPathwiseMultiCaplet::nextTimeStep(
70  const CurveState& currentState,
71  std::vector<Size>& numberCashFlowsThisStep,
72  std::vector<std::vector<MarketModelPathwiseMultiProduct::CashFlow> >& cashFlowsGenerated)
73  {
74  Rate liborRate = currentState.forwardRate(currentIndex_);
75  cashFlowsGenerated[currentIndex_][0].timeIndex = currentIndex_;
76  cashFlowsGenerated[currentIndex_][0].amount[0] =
77  (liborRate-strikes_[currentIndex_])*accruals_[currentIndex_];
78 
79  std::fill(numberCashFlowsThisStep.begin(),
80  numberCashFlowsThisStep.end(),0);
81 
82  if ( cashFlowsGenerated[currentIndex_][0].amount[0] >0)
83  {
84  numberCashFlowsThisStep[currentIndex_] = 1;
85  for (Size i=1; i <= numberRates_; ++i)
86  cashFlowsGenerated[currentIndex_][0].amount[i] =0;
87 
88  cashFlowsGenerated[currentIndex_][0].amount[currentIndex_+1] = accruals_[currentIndex_];
89  }
90  ++currentIndex_;
91  return (currentIndex_ == strikes_.size());
92  }

210  bool MarketModelPathwiseMultiDeflatedCaplet::nextTimeStep(
211  const CurveState& currentState,
212  std::vector<Size>& numberCashFlowsThisStep,
213  std::vector<std::vector<MarketModelPathwiseMultiProduct::CashFlow> >& cashFlowsGenerated)
214  {
215  Rate liborRate = currentState.forwardRate(currentIndex_);
216  cashFlowsGenerated[currentIndex_][0].timeIndex = currentIndex_;
217  cashFlowsGenerated[currentIndex_][0].amount[0] =
218  (liborRate-strikes_[currentIndex_])*accruals_[currentIndex_]*currentState.discountRatio(currentIndex_+1,0);
219 
220  std::fill(numberCashFlowsThisStep.begin(),
221  numberCashFlowsThisStep.end(),0);
222 
223  if ( cashFlowsGenerated[currentIndex_][0].amount[0] >0)
224  {
225  numberCashFlowsThisStep[currentIndex_] = 1;
226  for (Size i=1; i <= numberRates_; ++i)
227  cashFlowsGenerated[currentIndex_][0].amount[i] =0;
228 
229  cashFlowsGenerated[currentIndex_][0].amount[currentIndex_+1] = accruals_[currentIndex_]*currentState.discountRatio(currentIndex_+1,0);
230 
231  for (Size i=0; i <= currentIndex_; ++i)
232  {
233  Real stepDF = currentState.discountRatio(i+1,i);
234  cashFlowsGenerated[currentIndex_][0].amount[i+1] -= accruals_[i]*stepDF
235  *cashFlowsGenerated[currentIndex_][0].amount[0];
236  }
237  }
238  ++currentIndex_;
239  return (currentIndex_ == strikes_.size());
240  }

305  for (Size j=0; j < underlyingCaplets_.maxNumberOfCashFlowsPerProductPerStep(); ++j)

353  bool MarketModelPathwiseMultiDeflatedCap::nextTimeStep(
354  const CurveState& currentState,
355  std::vector<Size>& numberCashFlowsThisStep,
356  std::vector<std::vector<MarketModelPathwiseMultiProduct::CashFlow> >& cashFlowsGenerated)
357  {
358 
359  bool done = underlyingCaplets_.nextTimeStep(currentState, innerCashFlowSizes_, innerCashFlowsGenerated_);
360 
361  for (Size k=0; k < startsAndEnds_.size(); ++k)
362  numberCashFlowsThisStep[k]=0;
363 
364  for (Size j=0; j < numberRates_; ++j)
365  {
366  if (innerCashFlowSizes_[j]>0)
367  {
368  for (Size k=0; k < startsAndEnds_.size(); ++k)
369  {
370  if (startsAndEnds_[k].first <= j && j < startsAndEnds_[k].second)
371  {
372  for (Size l=0; l < innerCashFlowSizes_[j]; ++l)
373  cashFlowsGenerated[k][numberCashFlowsThisStep[k]++] = innerCashFlowsGenerated_[j][l];
374  }
375  }
376  }
377  }
378 
379  return done;
380  }
