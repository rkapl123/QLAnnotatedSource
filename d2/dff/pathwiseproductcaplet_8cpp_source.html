70  bool MarketModelPathwiseMultiCaplet::nextTimeStep(
71  const CurveState& currentState,
72  std::vector<Size>& numberCashFlowsThisStep,
73  std::vector<std::vector<MarketModelPathwiseMultiProduct::CashFlow> >& cashFlowsGenerated)
74  {
75  Rate liborRate = currentState.forwardRate(currentIndex_);
76  cashFlowsGenerated[currentIndex_][0].timeIndex = currentIndex_;
77  cashFlowsGenerated[currentIndex_][0].amount[0] =
78  (liborRate-strikes_[currentIndex_])*accruals_[currentIndex_];
79 
80  std::fill(numberCashFlowsThisStep.begin(),
81  numberCashFlowsThisStep.end(),0);
82 
83  if ( cashFlowsGenerated[currentIndex_][0].amount[0] >0)
84  {
85  numberCashFlowsThisStep[currentIndex_] = 1;
86  for (Size i=1; i <= numberRates_; ++i)
87  cashFlowsGenerated[currentIndex_][0].amount[i] =0;
88 
89  cashFlowsGenerated[currentIndex_][0].amount[currentIndex_+1] = accruals_[currentIndex_];
90  }
91  ++currentIndex_;
92  return (currentIndex_ == strikes_.size());
93  }

211  bool MarketModelPathwiseMultiDeflatedCaplet::nextTimeStep(
212  const CurveState& currentState,
213  std::vector<Size>& numberCashFlowsThisStep,
214  std::vector<std::vector<MarketModelPathwiseMultiProduct::CashFlow> >& cashFlowsGenerated)
215  {
216  Rate liborRate = currentState.forwardRate(currentIndex_);
217  cashFlowsGenerated[currentIndex_][0].timeIndex = currentIndex_;
218  cashFlowsGenerated[currentIndex_][0].amount[0] =
219  (liborRate-strikes_[currentIndex_])*accruals_[currentIndex_]*currentState.discountRatio(currentIndex_+1,0);
220 
221  std::fill(numberCashFlowsThisStep.begin(),
222  numberCashFlowsThisStep.end(),0);
223 
224  if ( cashFlowsGenerated[currentIndex_][0].amount[0] >0)
225  {
226  numberCashFlowsThisStep[currentIndex_] = 1;
227  for (Size i=1; i <= numberRates_; ++i)
228  cashFlowsGenerated[currentIndex_][0].amount[i] =0;
229 
230  cashFlowsGenerated[currentIndex_][0].amount[currentIndex_+1] = accruals_[currentIndex_]*currentState.discountRatio(currentIndex_+1,0);
231 
232  for (Size i=0; i <= currentIndex_; ++i)
233  {
234  Real stepDF = currentState.discountRatio(i+1,i);
235  cashFlowsGenerated[currentIndex_][0].amount[i+1] -= accruals_[i]*stepDF
236  *cashFlowsGenerated[currentIndex_][0].amount[0];
237  }
238  }
239  ++currentIndex_;
240  return (currentIndex_ == strikes_.size());
241  }

286  const std::vector<Time>& rateTimes,
287  const std::vector<Real>& accruals,
288  const std::vector<Time>& paymentTimes,
289  Rate strike,
290  std::vector<std::pair<Size, Size> > startsAndEnds)
291  : underlyingCaplets_(rateTimes, accruals, paymentTimes, strike), numberRates_(accruals.size()),
292  startsAndEnds_(std::move(startsAndEnds)) {
293  for (Size j=0; j < startsAndEnds_.size(); ++j)
294  {
295  QL_REQUIRE(startsAndEnds_[j].first < startsAndEnds_[j].second,"a cap must start before it ends: " << j << startsAndEnds_[j].first << startsAndEnds_[j].second );
296  QL_REQUIRE(startsAndEnds_[j].second <= accruals.size() ,"a cap must end when the underlying caplets: " << j << startsAndEnds_[j].first << startsAndEnds_[j].second );
297 
298  }
299 
300  innerCashFlowSizes_.resize(accruals.size());
301  innerCashFlowsGenerated_.resize(accruals.size());
302  for (auto& i : innerCashFlowsGenerated_) {
303  i.resize(underlyingCaplets_.maxNumberOfCashFlowsPerProductPerStep());
304  for (Size j=0; j < underlyingCaplets_.maxNumberOfCashFlowsPerProductPerStep(); ++j)
305  i[j].amount.resize(accruals.size() + 1);
306  }
307  }

349  bool MarketModelPathwiseMultiDeflatedCap::nextTimeStep(
350  const CurveState& currentState,
351  std::vector<Size>& numberCashFlowsThisStep,
352  std::vector<std::vector<MarketModelPathwiseMultiProduct::CashFlow> >& cashFlowsGenerated)
353  {
354 
355  bool done = underlyingCaplets_.nextTimeStep(currentState, innerCashFlowSizes_, innerCashFlowsGenerated_);
356 
357  for (Size k=0; k < startsAndEnds_.size(); ++k)
358  numberCashFlowsThisStep[k]=0;
359 
360  for (Size j=0; j < numberRates_; ++j)
361  {
362  if (innerCashFlowSizes_[j]>0)
363  {
364  for (Size k=0; k < startsAndEnds_.size(); ++k)
365  {
366  if (startsAndEnds_[k].first <= j && j < startsAndEnds_[k].second)
367  {
368  for (Size l=0; l < innerCashFlowSizes_[j]; ++l)
369  cashFlowsGenerated[k][numberCashFlowsThisStep[k]++] = innerCashFlowsGenerated_[j][l];
370  }
371  }
372  }
373  }
374 
375  return done;
376  }
