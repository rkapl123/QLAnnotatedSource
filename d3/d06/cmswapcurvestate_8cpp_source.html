41  void CMSwapCurveState::setOnCMSwapRates(const std::vector<Rate>& rates,
42  Size firstValidIndex) {
43  QL_REQUIRE(rates.size()==numberOfRates_,
44  "rates mismatch: " <<
45  numberOfRates_ << " required, " <<
46  rates.size() << " provided");
47  QL_REQUIRE(firstValidIndex<numberOfRates_,
48  "first valid index must be less than " <<
49  numberOfRates_ << ": " <<
50  firstValidIndex << " not allowed");
51 
52  // first copy input...
53  first_ = firstValidIndex;
54  std::copy(rates.begin()+first_, rates.end(),
55  cmSwapRates_.begin()+first_);
56  // ...then calculate discount ratios and annuities
57 
58  // taken care at constructor time
59  //discRatios_[numberOfRates_] = 1.0;
60  //cmSwapAnnuities_[numberOfRates_-1] = rateTaus_[numberOfRates_-1];
61 
62  // assume i+1 and do i
63  Integer oldAnnuityEndIndex = numberOfRates_;
64  for (Size i=numberOfRates_-1; i>first_; --i)
65  {
66  // formula 6.1 Joshi Liesch
67  Integer endIndex = std::min(i + spanningFwds_,numberOfRates_);
68  Integer annuityEndIndex = std::min(i + spanningFwds_-1,numberOfRates_);
69 
70  discRatios_[i] = discRatios_[endIndex] +
71  cmSwapRates_[i]*cmSwapAnnuities_[i];
72  cmSwapAnnuities_[i-1]= cmSwapAnnuities_[i]
73  +discRatios_[i] * rateTaus_[i-1];
74 
75  if (annuityEndIndex < oldAnnuityEndIndex)
76  cmSwapAnnuities_[i-1]-=discRatios_[oldAnnuityEndIndex] * rateTaus_[oldAnnuityEndIndex-1 ];
77 
78  oldAnnuityEndIndex = annuityEndIndex;
79  }
80  Integer endIndex = std::min(first_ + spanningFwds_,numberOfRates_);
81 
82  discRatios_[first_]= discRatios_[endIndex] +
83  cmSwapRates_[first_]*cmSwapAnnuities_[first_];
84 
85  // lazy evaluation of forward and coterminal swap rates&annuities
86  }
