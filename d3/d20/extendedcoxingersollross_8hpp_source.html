113  class ExtendedCoxIngersollRoss::FittingParameter
114  : public TermStructureFittingParameter {
115  private:
116  class Impl : public Parameter::Impl {
117  public:
118  Impl(const Handle<YieldTermStructure>& termStructure,
119  Real theta, Real k, Real sigma, Real x0)
120  : termStructure_(termStructure),
121  theta_(theta), k_(k), sigma_(sigma), x0_(x0) {}
122 
123  Real value(const Array&, Time t) const {
124  Rate forwardRate =
125  termStructure_->forwardRate(t, t, Continuous, NoFrequency);
126  Real h = std::sqrt(k_*k_ + 2.0*sigma_*sigma_);
127  Real expth = std::exp(t*h);
128  Real temp = 2.0*h + (k_+h)*(expth-1.0);
129  Real phi = forwardRate -
130  2.0*k_*theta_*(expth - 1.0)/temp -
131  x0_*4.0*h*h*expth/(temp*temp);
132  return phi;
133  }
134  private:
135  Handle<YieldTermStructure> termStructure_;
136  Real theta_, k_, sigma_, x0_;
137  };
138  public:
139  FittingParameter(const Handle<YieldTermStructure>& termStructure,
140  Real theta, Real k, Real sigma, Real x0)
141  : TermStructureFittingParameter(ext::shared_ptr<Parameter::Impl>(
142  new FittingParameter::Impl(
143  termStructure, theta, k, sigma, x0))) {}
144  };
