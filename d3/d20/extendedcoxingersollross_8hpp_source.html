113 class ExtendedCoxIngersollRoss::FittingParameter
114 : public TermStructureFittingParameter {
115 private:
116 class Impl : public Parameter::Impl {
117 public:
118 Impl(Handle<YieldTermStructure> termStructure, Real theta, Real k, Real sigma, Real x0)
119 : termStructure_(std::move(termStructure)), theta_(theta), k_(k), sigma_(sigma),
120 x0_(x0) {}
121
122 Real value(const Array&, Time t) const override {
123 Rate forwardRate =
124 termStructure_->forwardRate(t, t, Continuous, NoFrequency);
125 Real h = std::sqrt(k_*k_ + 2.0*sigma_*sigma_);
126 Real expth = std::exp(t*h);
127 Real temp = 2.0*h + (k_+h)*(expth-1.0);
128 Real phi = forwardRate -
129 2.0*k_*theta_*(expth - 1.0)/temp -
130 x0_*4.0*h*h*expth/(temp*temp);
131 return phi;
132 }
133
134 private:
135 Handle<YieldTermStructure> termStructure_;
136 Real theta_, k_, sigma_, x0_;
137 };
138 public:
139 FittingParameter(const Handle<YieldTermStructure>& termStructure,
140 Real theta, Real k, Real sigma, Real x0)
141 : TermStructureFittingParameter(ext::shared_ptr<Parameter::Impl>(
142 new FittingParameter::Impl(
143 termStructure, theta, k, sigma, x0))) {}
144 };
