48  void AnalyticPDFHestonEngine::calculate() const {
49  // this is an European option pricer
50  QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
51  "not an European option");
52 
53  const boost::shared_ptr<HestonProcess>& process = model_->process();
54 
55  const Time t = process->time(arguments_.exercise->lastDate());
56 
57  const Real xMax = 8.0 * std::sqrt(process->theta()*t
58  + (process->v0() - process->theta())
59  *(1-std::exp(-process->kappa()*t))/process->kappa());
60 
61  const Real x0 = std::log(process->s0()->value());
62  const Real rD = process->riskFreeRate()->discount(t);
63  const Real qD = process->dividendYield()->discount(t);
64 
65  const Real drift = x0 + std::log(rD/qD);
66 
67  results_.value = GaussLobattoIntegral(
68  maxIntegrationIterations_, integrationEps_)(
69  boost::bind(&AnalyticPDFHestonEngine::weightedPayoff, this,_1, t),
70  -xMax+drift, xMax+drift);
71  }

86  Real AnalyticPDFHestonEngine::weightedPayoff(Real x_t, Time t) const {
87  const DiscountFactor rD
88  = model_->process()->riskFreeRate()->discount(t);
89 
90  const Real s_t = std::exp(x_t);
91  const Real payoff = (*arguments_.payoff)(s_t);
92 
93  return (payoff != 0.0) ? payoff*Pv(x_t, t)*rD : 0.0;
94  }
