82  template <class I1, class I2>
83  class VannaVolgaInterpolationImpl
84  : public Interpolation::templateImpl<I1,I2> {
85  public:
86  VannaVolgaInterpolationImpl(const I1& xBegin, const I1& xEnd,
87  const I2& yBegin,
88  Real spot,
89  DiscountFactor dDiscount,
90  DiscountFactor fDiscount,
91  Time T)
92  : Interpolation::templateImpl<I1,I2>(xBegin, xEnd, yBegin,
93  VannaVolga::requiredPoints),
94  spot_(spot), dDiscount_(dDiscount), fDiscount_(fDiscount), T_(T) {
95  QL_REQUIRE(this->xEnd_-this->xBegin_ == 3,
96  "Vanna Volga Interpolator only interpolates 3 volatilities in strike space");
97  }
98  void update() {
99  //atmVol should be the second vol
100  atmVol_ = this->yBegin_[1];
101  fwd_ = spot_*fDiscount_/dDiscount_;
102  for(Size i = 0; i < 3; i++){
103  premiaBS.push_back(blackFormula(Option::Call, this->xBegin_[i], fwd_, atmVol_ * std::sqrt(T_), dDiscount_));
104  premiaMKT.push_back(blackFormula(Option::Call, this->xBegin_[i], fwd_, this->yBegin_[i] * std::sqrt(T_), dDiscount_));
105  vegas.push_back(vega(this->xBegin_[i]));
106  }
107  }
108  Real value(Real k) const {
109  Real x1 = vega(k)/vegas[0]
110  * (std::log(this->xBegin_[1]/k) * std::log(this->xBegin_[2]/k))
111  / (std::log(this->xBegin_[1]/this->xBegin_[0]) * std::log(this->xBegin_[2]/this->xBegin_[0]));
112  Real x2 = vega(k)/vegas[1]
113  * (std::log(k/this->xBegin_[0]) * std::log(this->xBegin_[2]/k))
114  / (std::log(this->xBegin_[1]/this->xBegin_[0]) * std::log(this->xBegin_[2]/this->xBegin_[1]));
115  Real x3 = vega(k)/vegas[2]
116  * (std::log(k/this->xBegin_[0]) * std::log(k/this->xBegin_[1]))
117  / (std::log(this->xBegin_[2]/this->xBegin_[0]) * std::log(this->xBegin_[2]/this->xBegin_[1]));
118 
119  Real cBS = blackFormula(Option::Call, k, fwd_, atmVol_ * std::sqrt(T_), dDiscount_);
120  Real c = cBS + x1*(premiaMKT[0] - premiaBS[0]) + x2*(premiaMKT[1] - premiaBS[1]) + x3*(premiaMKT[2] - premiaBS[2]);
121  Real std = blackFormulaImpliedStdDev(Option::Call, k, fwd_, c, dDiscount_);
122  return std / sqrt(T_);
123  }

125  QL_FAIL("Vanna Volga primitive not implemented");
126  }

128  QL_FAIL("Vanna Volga derivative not implemented");

145  Real vega(Real k) const {
146  Real d1 = (std::log(fwd_/k) + 0.5 * std::pow(atmVol_, 2.0) * T_)/(atmVol_ * std::sqrt(T_));
147  NormalDistribution norm;
148  return spot_ * dDiscount_ * std::sqrt(T_) * norm(d1);
149  }
150  };
