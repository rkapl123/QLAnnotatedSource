217  template<class Spec>
218  void BlackStyleSwaptionEngine<Spec>::calculate() const {
219  static const Spread basisPoint = 1.0e-4;
220 
221  Date exerciseDate = arguments_.exercise->date(0);
222 
223  // the part of the swap preceding exerciseDate should be truncated
224  // to avoid taking into account unwanted cashflows
225  // for the moment we add a check avoiding this situation
226  VanillaSwap swap = *arguments_.swap;
227  const Leg& fixedLeg = swap.fixedLeg();
228  ext::shared_ptr<FixedRateCoupon> firstCoupon =
229  ext::dynamic_pointer_cast<FixedRateCoupon>(fixedLeg[0]);
230  QL_REQUIRE(firstCoupon->accrualStartDate() >= exerciseDate,
231  "swap start (" << firstCoupon->accrualStartDate() << ") before exercise date ("
232  << exerciseDate << ") not supported in Black swaption engine");
233 
234  Rate strike = swap.fixedRate();
235 
236  // using the discounting curve
237  // swap.iborIndex() might be using a different forwarding curve
238  swap.setPricingEngine(ext::shared_ptr<PricingEngine>(new
239  DiscountingSwapEngine(discountCurve_, false)));
240  Rate atmForward = swap.fairRate();
241 
242  // Volatilities are quoted for zero-spreaded swaps.
243  // Therefore, any spread on the floating leg must be removed
244  // with a corresponding correction on the fixed leg.
245  if (swap.spread()!=0.0) {
246  Spread correction = swap.spread() *
247  std::fabs(swap.floatingLegBPS()/swap.fixedLegBPS());
248  strike -= correction;
249  atmForward -= correction;
250  results_.additionalResults["spreadCorrection"] = correction;
251  } else {
252  results_.additionalResults["spreadCorrection"] = 0.0;
253  }
254  results_.additionalResults["strike"] = strike;
255  results_.additionalResults["atmForward"] = atmForward;
256 
257  // using the discounting curve
258  swap.setPricingEngine(ext::shared_ptr<PricingEngine>(
259  new DiscountingSwapEngine(discountCurve_, false)));
260  Real annuity;
261  if (arguments_.settlementType == Settlement::Physical ||
262  (arguments_.settlementType == Settlement::Cash &&
263  arguments_.settlementMethod ==
264  Settlement::CollateralizedCashPrice)) {
265  annuity = std::fabs(swap.fixedLegBPS()) / basisPoint;
266  } else if (arguments_.settlementType == Settlement::Cash &&
267  arguments_.settlementMethod == Settlement::ParYieldCurve) {
268  DayCounter dayCount = firstCoupon->dayCounter();
269  // we assume that the cash settlement date is equal
270  // to the swap start date
271  Date discountDate = model_ == DiscountCurve
272  ? firstCoupon->accrualStartDate()
273  : discountCurve_->referenceDate();
274  Real fixedLegCashBPS = CashFlows::bps(
275  fixedLeg,
276  InterestRate(atmForward, dayCount, Compounded, Annual), false,
277  discountDate);
278  annuity = std::fabs(fixedLegCashBPS / basisPoint) *
279  discountCurve_->discount(discountDate);
280  } else {
281  QL_FAIL("invalid (settlementType, settlementMethod) pair");
282  }
283  results_.additionalResults["annuity"] = annuity;
284 
285  Time swapLength = vol_->swapLength(swap.floatingSchedule().dates().front(),
286  swap.floatingSchedule().dates().back());
287 
288  // swapLength is rounded to whole months. To ensure we can read a variance
289  // and a shift from vol_ we floor swapLength at 1/12 here therefore.
290  swapLength = std::max(swapLength, 1.0 / 12.0);
291  results_.additionalResults["swapLength"] = swapLength;
292 
293  Real variance = vol_->blackVariance(exerciseDate, swapLength, strike);
294 
295  Real displacement =
296  vol_->volatilityType() == ShiftedLognormal ?
297  vol_->shift(exerciseDate, swapLength) : 0.0;
298 
299  Real stdDev = std::sqrt(variance);
300  results_.additionalResults["stdDev"] = stdDev;
301  Option::Type w = (arguments_.type==VanillaSwap::Payer) ?
302  Option::Call : Option::Put;
303  results_.value = Spec().value(w, strike, atmForward, stdDev, annuity, displacement);
304 
305  Time exerciseTime = vol_->timeFromReference(exerciseDate);
306  results_.additionalResults["vega"] = Spec().vega(
307  strike, atmForward, stdDev, exerciseTime, annuity, displacement);
308  results_.additionalResults["delta"] = Spec().delta(
309  w, strike, atmForward, stdDev, annuity, displacement);
310  results_.additionalResults["timeToExpiry"] = exerciseTime;
311  results_.additionalResults["impliedVolatility"] = stdDev / std::sqrt(exerciseTime);
312  }
