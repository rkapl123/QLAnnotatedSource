42  FuturesRateHelper::FuturesRateHelper(const Handle<Quote>& price,
43  const Date& iborStartDate,
44  Natural lengthInMonths,
45  const Calendar& calendar,
46  BusinessDayConvention convention,
47  bool endOfMonth,
48  const DayCounter& dayCounter,
49  const Handle<Quote>& convAdj,
50  Futures::Type type)
51  : RateHelper(price), convAdj_(convAdj) {
52  switch (type) {
53  case Futures::IMM:
54  QL_REQUIRE(IMM::isIMMdate(iborStartDate, false),
55  iborStartDate << " is not a valid IMM date");
56  break;
57  case Futures::ASX:
58  QL_REQUIRE(ASX::isASXdate(iborStartDate, false),
59  iborStartDate << " is not a valid ASX date");
60  break;
61  default:
62  QL_FAIL("unknown futures type (" << Integer(type) << ")");
63  }
64  earliestDate_ = iborStartDate;
65  maturityDate_ = calendar.advance(iborStartDate, lengthInMonths*Months,
66  convention, endOfMonth);
67  yearFraction_ = dayCounter.yearFraction(earliestDate_, maturityDate_);
68  pillarDate_ = latestDate_ = latestRelevantDate_ = maturityDate_;
69 
70  registerWith(convAdj_);
71  }

266  Real FuturesRateHelper::impliedQuote() const {
267  QL_REQUIRE(termStructure_ != 0, "term structure not set");
268  Rate forwardRate = (termStructure_->discount(earliestDate_) /
269  termStructure_->discount(maturityDate_) - 1.0) / yearFraction_;
270  Rate convAdj = convAdj_.empty() ? 0.0 : convAdj_->value();
271  // Convexity, as FRA/futures adjustment, has been used in the
272  // past to take into account futures margining vs FRA.
273  // Therefore, there's no requirement for it to be non-negative.
274  Rate futureRate = forwardRate + convAdj;
275  return 100.0 * (1.0 - futureRate);
276  }

562  void FraRateHelper::initializeDates() {
563  // if the evaluation date is not a business day
564  // then move to the next business day
565  Date referenceDate =
566  iborIndex_->fixingCalendar().adjust(evaluationDate_);
567  Date spotDate = iborIndex_->fixingCalendar().advance(
568  referenceDate, iborIndex_->fixingDays()*Days);
569  earliestDate_ = iborIndex_->fixingCalendar().advance(
570  spotDate,
571  periodToStart_,
572  iborIndex_->businessDayConvention(),
573  iborIndex_->endOfMonth());
574  // maturity date is calculated from spot date
575  maturityDate_ = iborIndex_->fixingCalendar().advance(
576  spotDate,
577  periodToStart_ + iborIndex_->tenor(),
578  iborIndex_->businessDayConvention(),
579  iborIndex_->endOfMonth());
580  // latest relevant date is calculated from earliestDate_ instead
581  latestRelevantDate_ = iborIndex_->maturityDate(earliestDate_);
582 
583  switch (pillarChoice_) {
584  case Pillar::MaturityDate:
585  pillarDate_ = maturityDate_;
586  break;
587  case Pillar::LastRelevantDate:
588  pillarDate_ = latestRelevantDate_;
589  break;
590  case Pillar::CustomDate:
591  // pillarDate_ already assigned at construction time
592  QL_REQUIRE(pillarDate_ >= earliestDate_,
593  "pillar date (" << pillarDate_ << ") must be later "
594  "than or equal to the instrument's earliest date (" <<
595  earliestDate_ << ")");
596  QL_REQUIRE(pillarDate_ <= latestRelevantDate_,
597  "pillar date (" << pillarDate_ << ") must be before "
598  "or equal to the instrument's latest relevant date (" <<
599  latestRelevantDate_ << ")");
600  break;
601  default:
602  QL_FAIL("unknown Pillar::Choice(" << Integer(pillarChoice_) << ")");
603  }
604 
605  latestDate_ = pillarDate_; // backward compatibility
606 
607  fixingDate_ = iborIndex_->fixingDate(earliestDate_);
608  }

764  void SwapRateHelper::initializeDates() {
765 
766  // 1. do not pass the spread here, as it might be a Quote
767  // i.e. it can dinamically change
768  // 2. input discount curve Handle might be empty now but it could
769  // be assigned a curve later; use a RelinkableHandle here
770  swap_ = MakeVanillaSwap(tenor_, iborIndex_, 0.0, fwdStart_)
771  .withSettlementDays(settlementDays_)
772  .withDiscountingTermStructure(discountRelinkableHandle_)
773  .withFixedLegDayCount(fixedDayCount_)
774  .withFixedLegTenor(Period(fixedFrequency_))
775  .withFixedLegConvention(fixedConvention_)
776  .withFixedLegTerminationDateConvention(fixedConvention_)
777  .withFixedLegCalendar(calendar_)
778  .withFloatingLegCalendar(calendar_);
779 
780  earliestDate_ = swap_->startDate();
781  maturityDate_ = swap_->maturityDate();
782 
783  shared_ptr<IborCoupon> lastCoupon =
784  boost::dynamic_pointer_cast<IborCoupon>(swap_->floatingLeg().back());
785  latestRelevantDate_ = std::max(maturityDate_, lastCoupon->fixingEndDate());
786 
787  switch (pillarChoice_) {
788  case Pillar::MaturityDate:
789  pillarDate_ = maturityDate_;
790  break;
791  case Pillar::LastRelevantDate:
792  pillarDate_ = latestRelevantDate_;
793  break;
794  case Pillar::CustomDate:
795  // pillarDate_ already assigned at construction time
796  QL_REQUIRE(pillarDate_ >= earliestDate_,
797  "pillar date (" << pillarDate_ << ") must be later "
798  "than or equal to the instrument's earliest date (" <<
799  earliestDate_ << ")");
800  QL_REQUIRE(pillarDate_ <= latestRelevantDate_,
801  "pillar date (" << pillarDate_ << ") must be before "
802  "or equal to the instrument's latest relevant date (" <<
803  latestRelevantDate_ << ")");
804  break;
805  default:
806  QL_FAIL("unknown Pillar::Choice(" << Integer(pillarChoice_) << ")");
807  }
808 
809  latestDate_ = pillarDate_; // backward compatibility
810 
811  }

880  void BMASwapRateHelper::initializeDates() {

882  // then move to the next business day
883  JointCalendar jc(calendar_,
884  iborIndex_->fixingCalendar());

886  earliestDate_ =

890

892  shared_ptr<BMAIndex> clonedIndex(new BMAIndex(termStructureHandle_));

906  .endOfMonth(iborIndex_->endOfMonth())
907  .backwards();

911  0.75, // arbitrary
912  0.0,
913  iborIndex_,
914  iborIndex_->dayCounter(),
915  bmaSchedule,
916  clonedIndex,
917  bmaDayCount_));
918  swap_->setPricingEngine(shared_ptr<PricingEngine>(new
919  DiscountingSwapEngine(iborIndex_->forwardingTermStructure())));
