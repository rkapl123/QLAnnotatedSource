59  std::pair<Real, Real> SquareRootCLVModel::nonCentralChiSquaredParams(
60  const Date& d) const {
61 
62  const Time t = bsProcess_->time(d);
63 
64  const Real kappa = sqrtProcess_->a();
65  const Real theta = sqrtProcess_->b();
66  const Real sigma = sqrtProcess_->sigma();
67 
68  const Real df = 4*theta*kappa/(sigma*sigma);
69  const Real ncp = 4*kappa*std::exp(-kappa*t)
70  / (sigma*sigma*(1-std::exp(-kappa*t)))*sqrtProcess_->x0();
71 
72  return std::make_pair(df, ncp);
73  }

76  Disposable<Array> SquareRootCLVModel::collocationPointsX(const Date& d)
77  const {
78  const std::pair<Real, Real> p = nonCentralChiSquaredParams(d);
79 
80  Array x = GaussianQuadrature(lagrangeOrder_,
81  GaussNonCentralChiSquaredPolynomial(p.first, p.second))
82  .x();
83 
84  std::sort(x.begin(), x.end());
85 
86  const boost::math::non_central_chi_squared_distribution<Real>
87  dist(p.first, p.second);
88 
89  const Real xMin = std::max(x.front(),
90  (pMin_ == Null<Real>())
91  ? 0.0 : boost::math::quantile(dist, pMin_));
92 
93  const Real xMax = std::min(x.back(),
94  (pMax_ == Null<Real>())
95  ? QL_MAX_REAL : boost::math::quantile(dist, pMax_));
96 
97  const Real b = xMin - x.front();
98  const Real a = (xMax - xMin)/(x.back() - x.front());
99 
100  for (Size i=0, n=x.size(); i < n; ++i) {
101  x[i] = a*x[i] + b;
102  }
103 
104  return x;
105  }

107  Disposable<Array> SquareRootCLVModel::collocationPointsY(const Date& d)
108  const {
109 
110  const Array x = collocationPointsX(d);
111  const std::pair<Real, Real> params = nonCentralChiSquaredParams(d);
112  const boost::math::non_central_chi_squared_distribution<Real>
113  dist(params.first, params.second);
114 
115  Array s(x.size());
116  for (Size i=0, n=s.size(); i < n; ++i) {
117  const Real q = boost::math::cdf(dist, x[i]);
118 
119  s[i] = invCDF(d, q);
120  }
121 
122  return s;
123  }

135  const SquareRootCLVModel& model)
136  : s_(ext::make_shared<Matrix>(

138  x_(ext::make_shared<Matrix>(

145  bsProcess = model.bsProcess_;

154  std::copy(y.begin(), y.end(), s_->row_begin(i));

157 
158  interpl.insert(
159  std::make_pair(maturity,
160  ext::make_shared<LagrangeInterpolation>(
161  x_->row_begin(i), x_->row_end(i),
162  s_->row_begin(i))));
163  }

166  Real SquareRootCLVModel::MappingFunction::operator()(Time t,Real x) const {
167  const interpl_type::const_iterator ge = interpl.lower_bound(t);
168 
169  if (close_enough(ge->first, t)) {
170  return (*ge->second)(x, true);
171  }
172 
173  QL_REQUIRE(ge != interpl.end() && ge != interpl.begin(),
174  "extrapolation to large or small t is not allowed");
175 
176  const Time t1 = ge->first;
177  const Real y1 = (*ge->second)(x, true);
178 
179  interpl_type::const_iterator lt = ge;
180  std::advance(lt, -1);
181 
182  const Time t0 = lt->first;
183  const Real y0 = (*lt->second)(x, true);
184 
185  return y0 + (y1 - y0)/(t1 - t0)*(t - t0);
186  }
