34  DiscretizedCallableFixedRateBond::DiscretizedCallableFixedRateBond(
35  const CallableBond::arguments& args,
36  const Date& referenceDate,
37  const DayCounter& dayCounter)
38  : arguments_(args) {
39 
40  redemptionTime_ = dayCounter.yearFraction(referenceDate,
41  args.redemptionDate);
42 
43  couponTimes_.resize(args.couponDates.size());
44  for (Size i=0; i<couponTimes_.size(); ++i)
45  couponTimes_[i] =
46  dayCounter.yearFraction(referenceDate,
47  args.couponDates[i]);
48 
49  callabilityTimes_.resize(args.callabilityDates.size());
50  for (Size i=0; i<callabilityTimes_.size(); ++i)
51  callabilityTimes_[i] =
52  dayCounter.yearFraction(referenceDate,
53  args.callabilityDates[i]);
54 
55  // To avoid mispricing, we snap exercise dates to the closest coupon date.
56  for (Size i=0; i<callabilityTimes_.size(); i++) {
57  Time& exerciseTime = callabilityTimes_[i];
58  for (Size j=0; j<couponTimes_.size(); j++) {
59  if (withinNextWeek(exerciseTime, couponTimes_[j])) {
60  exerciseTime = couponTimes_[j];
61  break;
62  }
63  }
64  }
65  }

121  void DiscretizedCallableFixedRateBond::applyCallability(Size i) {
122  Size j;
123  switch (arguments_.putCallSchedule[i]->type() ) {
124  case Callability::Call:
125  for (j=0; j<values_.size(); j++) {
126  values_[j] =
127  std::min(arguments_.callabilityPrices[i],
128  values_[j]);
129  }
130  break;
131  case Callability::Put:
132  for (j=0; j<values_.size(); j++) {
133  values_[j] = std::max(values_[j],
134  arguments_.callabilityPrices[i]);
135  }
136  break;
137  default:
138  QL_FAIL("unknown callability type");
139  }
