183  class SimpleRandomInertia : public ParticleSwarmOptimization::Inertia {
184  public:
185  SimpleRandomInertia(Real threshold = 0.5, unsigned long seed = SeedGenerator::instance().get())
186  : threshold_(threshold), rng_(seed) {
187  QL_REQUIRE(threshold_ >= 0.0 && threshold_ < 1.0, "Threshold must be a Real in [0, 1)");
188  }
189  inline void setSize(Size M, Size N, Real c0, const EndCriteria &endCriteria) {
190  M_ = M;
191  c0_ = c0;
192  }
193  inline void setValues() {
194  for (Size i = 0; i < M_; i++) {
195  Real val = c0_*(threshold_ + (1.0 - threshold_)*rng_.nextReal());
196  (*V_)[i] *= val;
197  }
198  }
199  private:
200  Real c0_, threshold_;
201  Size M_;
202  MersenneTwisterUniformRng rng_;

206  /* Inertia value gets decreased every iteration until it reaches

209  class DecreasingInertia : public ParticleSwarmOptimization::Inertia {
210  public:
211  DecreasingInertia(Real threshold = 0.5)
212  : threshold_(threshold) {
213  QL_REQUIRE(threshold_ >= 0.0 && threshold_ < 1.0, "Threshold must be a Real in [0, 1)");
214  }
215  inline void setSize(Size M, Size N, Real c0, const EndCriteria &endCriteria) {
216  N_ = N;
217  c0_ = c0;
218  iteration_ = 0;
219  maxIterations_ = endCriteria.maxIterations();
220  }
221  inline void setValues() {
222  Real c0 = c0_*(threshold_ + (1.0 - threshold_)*(maxIterations_ - iteration_) / maxIterations_);
223  for (Size i = 0; i < M_; i++) {
224  (*V_)[i] *= c0;
225  }
226  }
227  private:
228  Real c0_, threshold_;
229  Size M_, N_, maxIterations_, iteration_;

259  /* As long as the particle keeps getting frequent updates to its
260  personal best value, the inertia behaves like a SimpleRandomInertia,
261  but after a number of iterations without improvement, the behaviour
262  changes to that of a Levy flight ~ u^{-1/\alpha}
263  */
264  class LevyFlightInertia : public ParticleSwarmOptimization::Inertia {
265  public:
266  typedef IsotropicRandomWalk<LevyFlightDistribution, base_generator_type> IsotropicLevyFlight;
267  LevyFlightInertia(Real alpha, Size threshold,
268  unsigned long seed = SeedGenerator::instance().get())
269  :rng_(seed), flight_(base_generator_type(seed), LevyFlightDistribution(1.0, alpha),
270  1, Array(1, 1.0), seed),
271  threshold_(threshold) {};
272  inline void setSize(Size M, Size N, Real c0, const EndCriteria &endCriteria) {
273  M_ = M;
274  N_ = N;
275  c0_ = c0;
276  adaptiveCounter_ = std::vector<Size>(M_, 0);
277  }
278  inline void setValues() {
279  for (Size i = 0; i < M_; i++) {
280  if ((*pBF_)[i] < personalBestF_[i]) {
281  personalBestF_[i] = (*pBF_)[i];
282  adaptiveCounter_[i] = 0;
283  }
284  else {
285  adaptiveCounter_[i]++;
286  }
287  if (adaptiveCounter_[i] <= threshold_) {
288  //Simple Random Inertia
289  (*V_)[i] *= c0_*(0.5 + 0.5*rng_.nextReal());
290  }
291  else {
292  //If particle has not found a new personal best after threshold_ iterations
293  //then trigger a Levy flight pattern for the speed
294  flight_.nextReal<Real *>(&(*V_)[i][0]);
295  }
296  }
297  }
298  protected:
299  void init(ParticleSwarmOptimization *pso) {
300  ParticleSwarmOptimization::Inertia::init(pso);
301  personalBestF_ = *pBF_;
302  flight_.setDimension(N_, *lX_, *uX_);
303  }
304  private:
305  MersenneTwisterUniformRng rng_;
306  IsotropicLevyFlight flight_;
307  Array personalBestF_;
308  std::vector<Size> adaptiveCounter_;
309  Real c0_;
310  Size M_, N_;
311  Size threshold_;
312  };

343  /* The global best as seen by each particle is the best from amongst
344  all particles
345  */
346  class GlobalTopology : public ParticleSwarmOptimization::Topology {
347  public:
348  inline void setSize(Size M) { M_ = M; }
349  inline void findSocialBest() {
350  Real bestF = (*pBF_)[0];
351  Size bestP = 0;
352  for (Size i = 1; i < M_; i++) {
353  if (bestF < (*pBF_)[i]) {
354  bestF = (*pBF_)[i];
355  bestP = i;
356  }
357  }
358  Array& x = (*pBX_)[bestP];
359  for (Size i = 0; i < M_; i++) {
360  if (i != bestP) {
361  (*gBX_)[i] = x;
362  (*gBF_)[i] = bestF;
363  }
364  }
365  }
366 
367  private:
368  Size M_;
369  };

392  /* H.M. Emara, Adaptive Clubs-based Particle Swarm Optimization
393  Each particle is originally assigned to a default number of clubs
394  from among the total set. The best as seen by each particle is the
395  best from amongst the clubs to which the particle belongs.
396  Underperforming particles join more clubs randomly (up to a maximum
