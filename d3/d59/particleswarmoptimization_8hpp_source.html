184  class SimpleRandomInertia : public ParticleSwarmOptimization::Inertia {
185  public:
186  SimpleRandomInertia(Real threshold = 0.5, unsigned long seed = SeedGenerator::instance().get())
187  : threshold_(threshold), rng_(seed) {
188  QL_REQUIRE(threshold_ >= 0.0 && threshold_ < 1.0, "Threshold must be a Real in [0, 1)");
189  }
190  inline void setSize(Size M, Size N, Real c0, const EndCriteria& endCriteria) override {
191  M_ = M;
192  c0_ = c0;
193  }
194  inline void setValues() override {
195  for (Size i = 0; i < M_; i++) {
196  Real val = c0_*(threshold_ + (1.0 - threshold_)*rng_.nextReal());
197  (*V_)[i] *= val;
198  }
199  }
200 
201  private:
202  Real c0_, threshold_;
203  Size M_;
204  MersenneTwisterUniformRng rng_;

227  }

263  /* As long as the particle keeps getting frequent updates to its
264  personal best value, the inertia behaves like a SimpleRandomInertia,
265  but after a number of iterations without improvement, the behaviour
266  changes to that of a Levy flight ~ u^{-1/\alpha}
267  */
268  class LevyFlightInertia : public ParticleSwarmOptimization::Inertia {
269  public:
270  typedef IsotropicRandomWalk<LevyFlightDistribution, base_generator_type> IsotropicLevyFlight;
271  LevyFlightInertia(Real alpha, Size threshold,
272  unsigned long seed = SeedGenerator::instance().get())
273  :rng_(seed), flight_(base_generator_type(seed), LevyFlightDistribution(1.0, alpha),
274  1, Array(1, 1.0), seed),
275  threshold_(threshold) {};
276  inline void setSize(Size M, Size N, Real c0, const EndCriteria& endCriteria) override {
277  M_ = M;
278  N_ = N;
279  c0_ = c0;
280  adaptiveCounter_ = std::vector<Size>(M_, 0);
281  }
282  inline void setValues() override {
283  for (Size i = 0; i < M_; i++) {
284  if ((*pBF_)[i] < personalBestF_[i]) {
285  personalBestF_[i] = (*pBF_)[i];
286  adaptiveCounter_[i] = 0;
287  }
288  else {
289  adaptiveCounter_[i]++;
290  }
291  if (adaptiveCounter_[i] <= threshold_) {
292  //Simple Random Inertia
293  (*V_)[i] *= c0_*(0.5 + 0.5*rng_.nextReal());
294  }
295  else {
296  //If particle has not found a new personal best after threshold_ iterations
297  //then trigger a Levy flight pattern for the speed
298  flight_.nextReal<Real *>(&(*V_)[i][0]);
299  }
300  }
301  }
302 
303  protected:
304  void init(ParticleSwarmOptimization* pso) override {
305  ParticleSwarmOptimization::Inertia::init(pso);
306  personalBestF_ = *pBF_;
307  flight_.setDimension(N_, *lX_, *uX_);
308  }
309 
310  private:
311  MersenneTwisterUniformRng rng_;
312  IsotropicLevyFlight flight_;
313  Array personalBestF_;
314  std::vector<Size> adaptiveCounter_;
315  Real c0_;
316  Size M_, N_;
317  Size threshold_;
318  };

349  /* The global best as seen by each particle is the best from amongst
350  all particles
351  */
352  class GlobalTopology : public ParticleSwarmOptimization::Topology {
353  public:
354  inline void setSize(Size M) override { M_ = M; }
355  inline void findSocialBest() override {
356  Real bestF = (*pBF_)[0];
357  Size bestP = 0;
358  for (Size i = 1; i < M_; i++) {
359  if (bestF < (*pBF_)[i]) {
360  bestF = (*pBF_)[i];
361  bestP = i;
362  }
363  }
364  Array& x = (*pBX_)[bestP];
365  for (Size i = 0; i < M_; i++) {
366  if (i != bestP) {
367  (*gBX_)[i] = x;
368  (*gBF_)[i] = bestF;
369  }
370  }
371  }
372 
373  private:
374  Size M_;
375  };

398  /* H.M. Emara, Adaptive Clubs-based Particle Swarm Optimization
399  Each particle is originally assigned to a default number of clubs
400  from among the total set. The best as seen by each particle is the
401  best from amongst the clubs to which the particle belongs.
402  Underperforming particles join more clubs randomly (up to a maximum
403  number) to widen the particles that influence them, while
404  overperforming particles leave clubs randomly (down to a minimum
405  number) to avoid early convergence to local minima.
406  */
