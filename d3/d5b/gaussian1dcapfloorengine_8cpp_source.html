26  void Gaussian1dCapFloorEngine::calculate() const {
27 
28  for (Size i = 0; i < arguments_.spreads.size(); i++)
29  QL_REQUIRE(arguments_.spreads[i] == 0.0,
30  "Non zero spreads (" << arguments_.spreads[i]
31  << ") are not allowed.");
32 
33  Size optionlets = arguments_.startDates.size();
34  std::vector<Real> values(optionlets, 0.0);
35  std::vector<Real> forwards(optionlets, 0.0);
36  Real value = 0.0;
37 
38  Date settlement = model_->termStructure()->referenceDate();
39 
40  CapFloor::Type type = arguments_.type;
41 
42  Array z = model_->yGrid(stddevs_, integrationPoints_);
43  Array p(z.size());
44 
45  for (Size i = 0; i < optionlets; ++i) {
46 
47  Date valueDate = arguments_.startDates[i];
48  Date paymentDate = arguments_.endDates[i];
49  ext::shared_ptr<IborIndex> iborIndex =
50  ext::dynamic_pointer_cast<IborIndex>(arguments_.indexes[i]);
51  // if we do not find an ibor index with associated forwarding curve
52  // we fall back on the model curve
53 
54  if (paymentDate > settlement) {
55 
56  Real f = arguments_.nominals[i] * arguments_.gearings[i];
57  Date fixingDate = arguments_.fixingDates[i];
58  Time fixingTime =
59  model_->termStructure()->timeFromReference(fixingDate);
60 
61  Real strike;
62 
63  if (type == CapFloor::Cap || type == CapFloor::Collar) {
64  strike = arguments_.capRates[i];
65  if (fixingDate <= settlement) {
66  values[i] =
67  std::max(arguments_.forwards[i] - strike, 0.0) * f *
68  arguments_.accrualTimes[i];
69  } else {
70 
71  // todo add openmp support later on (as in gaussian1dswaptionengine)
72 
73  for (Size j = 0; j < z.size(); j++) {
74  Real floatingLegNpv;
75  if (iborIndex != NULL)
76  floatingLegNpv =
77  arguments_.accrualTimes[i] *
78  model_->forwardRate(fixingDate, fixingDate,
79  z[j], iborIndex) *
80  model_->zerobond(paymentDate, fixingDate,
81  z[j], discountCurve_);
82  else
83  floatingLegNpv =
84  (model_->zerobond(valueDate, fixingDate,
85  z[j]) -
86  model_->zerobond(paymentDate, fixingDate,
87  z[j]));
88  Real fixedLegNpv =
89  arguments_.capRates[i] *
90  arguments_.accrualTimes[i] *
91  model_->zerobond(paymentDate, fixingDate, z[j]);
92  p[j] =
93  std::max((floatingLegNpv - fixedLegNpv), 0.0) /
94  model_->numeraire(fixingTime, z[j],
95  discountCurve_);
96  }
97  CubicInterpolation payoff(
98  z.begin(), z.end(), p.begin(),
99  CubicInterpolation::Spline, true,
100  CubicInterpolation::Lagrange, 0.0,
101  CubicInterpolation::Lagrange, 0.0);
102  Real price = 0.0;
103  for (Size j = 0; j < z.size() - 1; j++) {
104  price += model_->gaussianShiftedPolynomialIntegral(
105  0.0, payoff.cCoefficients()[j],
106  payoff.bCoefficients()[j],
107  payoff.aCoefficients()[j], p[j], z[j], z[j],
108  z[j + 1]);
109  }
110  if (extrapolatePayoff_) {
111  if (flatPayoffExtrapolation_) {
112  price +=
113  model_->gaussianShiftedPolynomialIntegral(
114  0.0, 0.0, 0.0, 0.0, p[z.size() - 2],
115  z[z.size() - 2], z[z.size() - 1],
116  100.0);
117  price +=
118  model_->gaussianShiftedPolynomialIntegral(
119  0.0, 0.0, 0.0, 0.0, p[0], z[0], -100.0,
120  z[0]);
121  } else {
122  price +=
123  model_->gaussianShiftedPolynomialIntegral(
124  0.0,
125  payoff.cCoefficients()[z.size() - 2],
126  payoff.bCoefficients()[z.size() - 2],
127  payoff.aCoefficients()[z.size() - 2],
128  p[z.size() - 2], z[z.size() - 2],
129  z[z.size() - 1], 100.0);
130  }
131  }
132  values[i] =
133  price *
134  model_->numeraire(0.0, 0.0, discountCurve_) * f;
135  }
136  }
137  if (type == CapFloor::Floor || type == CapFloor::Collar) {
138  strike = arguments_.floorRates[i];
139  Real floorlet;
140  if (fixingDate <= settlement) {
141  floorlet =
142  std::max(-(arguments_.forwards[i] - strike), 0.0) *
143  f * arguments_.accrualTimes[i];
144  } else {
145  for (Size j = 0; j < z.size(); j++) {
146  Real floatingLegNpv;
147  if (iborIndex != NULL)
148  floatingLegNpv =
149  arguments_.accrualTimes[i] *
150  model_->forwardRate(fixingDate, fixingDate,
151  z[j], iborIndex) *
152  model_->zerobond(paymentDate, fixingDate,
153  z[j], discountCurve_);
154  else
155  floatingLegNpv =
156  (model_->zerobond(valueDate, fixingDate,
157  z[j]) -
158  model_->zerobond(paymentDate, fixingDate,
159  z[j]));
160  Real fixedLegNpv =
161  arguments_.floorRates[i] *
162  arguments_.accrualTimes[i] *
163  model_->zerobond(paymentDate, fixingDate, z[j]);
164  p[j] =
165  std::max(-(floatingLegNpv - fixedLegNpv), 0.0) /
166  model_->numeraire(fixingTime, z[j],
167  discountCurve_);
168  }
169  CubicInterpolation payoff(
170  z.begin(), z.end(), p.begin(),
171  CubicInterpolation::Spline, true,
172  CubicInterpolation::Lagrange, 0.0,
173  CubicInterpolation::Lagrange, 0.0);
174  Real price = 0.0;
175  for (Size j = 0; j < z.size() - 1; j++) {
176  price += model_->gaussianShiftedPolynomialIntegral(
177  0.0, payoff.cCoefficients()[j],
178  payoff.bCoefficients()[j],
179  payoff.aCoefficients()[j], p[j], z[j], z[j],
180  z[j + 1]);
181  }
182  if (extrapolatePayoff_) {
183  if (flatPayoffExtrapolation_) {
184  price +=
185  model_->gaussianShiftedPolynomialIntegral(
186  0.0, 0.0, 0.0, 0.0, p[z.size() - 2],
187  z[z.size() - 2], z[z.size() - 1],
188  100.0);
189  price +=
190  model_->gaussianShiftedPolynomialIntegral(
191  0.0, 0.0, 0.0, 0.0, p[0], z[0], -100.0,
192  z[0]);
193  } else {
194  price +=
195  model_->gaussianShiftedPolynomialIntegral(
196  0.0, payoff.cCoefficients()[0],
197  payoff.bCoefficients()[0],
198  payoff.aCoefficients()[0], p[0], z[0],
199  -100.0, z[0]);
200  }
201  }
202  floorlet = price *
203  model_->numeraire(0.0, 0.0, discountCurve_) *
204  f;
205  }
206  if (type == CapFloor::Floor) {
207  values[i] = floorlet;
208  } else {
209  // a collar is long a cap and short a floor
210  values[i] -= floorlet;
211  }
212  }
213 
214  value += values[i];
215  }
216  }
217 
218  results_.value = value;
219 
220  results_.additionalResults["optionletsPrice"] = values;
221  results_.additionalResults["optionletsAtmForward"] = forwards;
222  }
223
