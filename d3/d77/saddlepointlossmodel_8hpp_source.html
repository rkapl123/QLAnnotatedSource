625  template<class CP>
626  Real SaddlePointLossModel<CP>::CumulantGeneratingCond(
627  const std::vector<Real>& invUncondProbs,
628  Real lossFraction,
629  const std::vector<Real>& mktFactor) const
630  {
631  const Size nNames = remainingNotionals_.size();
632  Real sum = 0.;
633 
634  for(Size iName=0; iName < nNames; iName++) {
635  Probability pBuffer =
636  copula_->conditionalDefaultProbabilityInvP(
637  invUncondProbs[iName], iName, mktFactor);
638  sum += std::log(1. - pBuffer +
639  pBuffer * std::exp(remainingNotionals_[iName] *
640  (1.-copula_->conditionalRecoveryInvP(invUncondProbs[iName],
641  iName, mktFactor)) * lossFraction / remainingNotional_));
642  }
643  return sum;
644  }

646  template<class CP>
647  Real SaddlePointLossModel<CP>::CumGen1stDerivativeCond(
648  const std::vector<Real>& invUncondProbs,
649  Real saddle,
650  const std::vector<Real>& mktFactor) const
651  {
652  const Size nNames = remainingNotionals_.size();
653  Real sum = 0.;
654 
655  for(Size iName=0; iName < nNames; iName++) {
656  Probability pBuffer =
657  copula_->conditionalDefaultProbabilityInvP(
658  invUncondProbs[iName], iName, mktFactor);
659  // loss in fractional units
660  Real lossInDef = remainingNotionals_[iName] *
661  (1.-copula_->conditionalRecoveryInvP(invUncondProbs[iName],
662  iName, mktFactor)) / remainingNotional_;
663  Real midFactor = pBuffer * std::exp(lossInDef * saddle);
664  sum += lossInDef * midFactor / (1.-pBuffer + midFactor);
665  }
666  return sum;

840  const std::vector<Real>& invUncondPs,
841  Real lossLevel, // in total portfolio loss fractional unit
842  const std::vector<Real>& mktFactor,
843  Real accuracy,
844  Natural maxEvaluations
845  ) const
846  {

848  // REQUIRE that loss level is below the max loss attainable in
849  // the portfolio, otherwise theres no solution...
850  SaddleObjectiveFunction f(*this, lossLevel, invUncondPs, mktFactor);
851

854  for(Size iName=0; iName<nNames; iName++)
855  lgds.push_back(remainingNotionals_[iName] *
856  (1.-copula_->conditionalRecoveryInvP(invUncondPs[iName], iName,
857  mktFactor)) );
858

860  // position of the name with the largest relative exposure loss (i.e.:
861  // largest: N_i LGD_i / N_{total})
862  Size iNamMax = std::distance(lgds.begin(),

872  // smaller than the true value:

875  (pMaxName*lgds[iNamMax]/remainingNotional_-pMaxName*deltaMin));

882  // by the minimum/maximum attainable point. Typically the functionals
883  // to integrate will have a low dependency on this point.
884  if(lossLevel < minLoss) return saddleMin;
885

887  std::log((lgds[iNamMax]/remainingNotional_
888  -deltaMin)*(1.-pMaxName)/(pMaxName*deltaMin));

890  CumGen1stDerivativeCond(invUncondPs, saddleMax, mktFactor);
891  if(lossLevel > maxLoss) return saddleMax;

894  Real guess = (saddleMin+saddleMax)/2.;

899  if(guess > saddleMax) guess = (saddleMin+saddleMax)/2.;
900  */

902  return solverBrent.solve(f, accuracy, guess, saddleMin, saddleMax);
