41  void AnalyticGJRGARCHEngine::calculate() const {
42  // this is a european option pricer
43  QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
44  "not an European option");
45 
46  // plain vanilla
47  boost::shared_ptr<StrikedTypePayoff> payoff =
48  boost::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
49  QL_REQUIRE(payoff, "non-striked payoff given");
50 
51  const boost::shared_ptr<GJRGARCHProcess>& process = model_->process();
52 
53  const Rate riskFreeDiscount = process->riskFreeRate()->discount(
54  arguments_.exercise->lastDate());
55  const Rate dividendDiscount = process->dividendYield()->discount(
56  arguments_.exercise->lastDate());
57  const Real spotPrice = process->s0()->value();
58  QL_REQUIRE(spotPrice > 0.0, "negative or null underlying given");
59  const Real strikePrice = payoff->strike();
60  const Real term = process->time(arguments_.exercise->lastDate());
61  Size T = Size(process->daysPerYear()*term+0.5);
62  Real r = -std::log(riskFreeDiscount/dividendDiscount)/(process->daysPerYear()*term);
63  Real h1 = process->v0();
64  Real b0 = process->omega();
65  Real b2 = process->alpha();
66  Real b1 = process->beta();
67  Real b3 = process->gamma();
68  Real la = process->lambda();
69  Real N = CumulativeNormalDistribution()(la);
70  Real n = std::exp(-la*la/2)/(M_SQRTPI*M_SQRT2);
71  const Real s = spotPrice;
72  const Real x = strikePrice;
73  Real m1, m2, m3, v1, v2, v3, z1, z2, x1;
74  Real ex, ex2, ex3, ex4;
75  Real sEh = 0.0, sEh2 = 0.0, sEhh = 0.0, sEh1_2eh = 0.0;
76  Real sEhhh = 0.0, sEh2h = 0.0, sEhh2 = 0.0, sEh3 = 0.0;
77  Real sEh1_2eh2 = 0.0, sEh3_2eh = 0.0, sEh1_2ehh = 0.0, sEhh1_2eh = 0.0;
78  Real sEhe2h = 0.0, sEh1_2eh1_2eh = 0.0;
79  Real sEh3_2e3h = 0.0;
80  Real SD1, SD2, SD3;
81  Real ST1, ST2, ST3, ST4;
82  Real SQ2, SQ4, SQ5;
83  Size i, j, k;
84  Real stdev, sigma, k3, k4;
85  Real d, del, d_, C, A3, A4, Capp;
86  bool constants_match = false;
87 
88  if (!init_ || b1 != b1_ || b2 != b2_ || b3 != b3_ || la != la_) {
89  // compute the useful coefficients
90  m1 = b1 + (b2+b3*N)*(1+la*la) + b3*la*n; // ok
91  m2 = b1*b1 + b2*b2*(pow(la,4)+6*la*la+3)
92  + (b3*b3+2*b2*b3)*( pow(la,4)*N
93  +pow(la,3)*n+6*la*la*N+5*la*n+3*N)
94  + 2*b1*b2*(1+la*la) + 2*b3*b1*(la*la*N+la*n+N); // ok
95  m3 = pow(b1,3)
96  + (3*b3*b3*b1+6*b1*b2*b3)*(pow(la,3)*n+5*la*n+3*N
97  +pow(la,4)*N+6*la*la*N)
98  + pow(b2,3)*(15+pow(la,6)+15*pow(la,4)+45*la*la)
99  + (pow(b3,3)+3*b2*b2*b3+3*b3*b3*b2)
100  *(pow(la,5)*n+14*pow(la,3)*n+33*la*n+15*N
101  +15*pow(la,4)*N+45*la*la*N+pow(la,6)*N)
102  + 3*b1*b1*b2*(1+la*la) + 3*b1*b1*b3*(la*n+N+la*la*N)
103  + 3*b1*b2*b2*(3+pow(la,4)+6*la*la); // ok
104  v1 = -2*b2*la - 2*b3*(n+la*N); // ok
105  v2 = -4*b2*b2*(3*la+pow(la,3))
106  - (4*b3*b3+8*b2*b3)*(la*la*n+2*n+pow(la,3)*N+3*la*N)
107  - 4*b1*b2*la - 4*b3*b1*(n+la*N); // ok
108  v3 = -12*b3*b1*(b3+2*b2)*(la*la*n+2*n+pow(la,3)*N+3*la*N)
109  - 6*pow(b2,3)*la*(15+pow(la,4)+10*la*la)
110  - 6*b3*(b3*b3+3*b2*b2+3*b3*b2)
111  *(9*la*la*n+8*n+15*la*N+pow(la,4)*n+pow(la,5)*N
112  +10*pow(la,3)*N)
113  - 6*b1*b1*b2*la - 6*b3*b1*b1*(n+la*N)
114  - 12*b2*b2*b1*(3*la+std::pow(la,3)); // ok
115  z1 = b1 + b2*(3+la*la) + b3*(la*n+3*N+la*la*N); // ok
116  z2 = b1*b1 + b2*b2*(15+pow(la,4)+18*la*la)
117  + (b3*b3+2*b2*b3)*(pow(la,3)*n+17*la*n+15*N
118  +pow(la,4)*N+18*la*la*N)
119  + 2*b1*b2*(3+la*la) + 2*b3*b1*(la*n+3*N+la*la*N); // ok
120  x1 = -6*b2*la - 2*b3*(4*n+3*la*N); // ok
121  b1_ = b1; b2_ = b2; b3_ = b3; la_ = la;
122  m1_ = m1; m2_ = m2; m3_ = m3;
123  v1_ = v1; v2_ = v2; v3_ = v3; z1_ = z1; z2_ = z2; x1_ = x1;
124  } else {
125  // these assignments are never used ?
126  // b1 = b1_; b2 = b2_; b3 = b3_; la = la_;
127  // m1 = m1_; m2 = m2_; m3 = m3_;
128  // v1 = v1_; v2 = v2_; v3 = v3_; z1 = z1_; z2 = z2_; x1 = x1_;
129  constants_match = true;
130  }
131 
132  // compute the first four moments
133  if (!init_ || !constants_match || b0 != b0_ || h1 != h1_ || T != T_) {
134  // these assignments are never used ?
135  //b1 = b1_; b2 = b2_; b3 = b3_; la = la_;
136  m1 = m1_; m2 = m2_; m3 = m3_;
137  v1 = v1_; v2 = v2_; /*v3 = v3_;*/ z1 = z1_; /*z2 = z2_;*/ x1 = x1_;
138 
139  boost::scoped_array<Real> m1ai(new Real[T]);
140  boost::scoped_array<Real> m2ai(new Real[T]);
141  boost::scoped_array<Real> m3ai(new Real[T]);
142  m1ai[0] = m2ai[0] = m3ai[0] = 1.0;
143  for (i=1; i < T; ++i) {
144  m1ai[i] = m1ai[i-1]*m1;
145  m2ai[i] = m2ai[i-1]*m2;
146  m3ai[i] = m3ai[i-1]*m3;
147  }
148 
149  for (i = 0; i < T; ++i) {
150  Real m1i = m1ai[i];
151  Real m2i = m2ai[i];
152  Real m3i = m3ai[i];
153 
154  Real m1im2i = m1i-m2i, m1im3i = m1i-m3i, m2im3i = m2i-m3i;
155  Real Eh = b0*(1-m1i)/(1-m1) + m1i*h1; // ko
156  Real Eh2 = b0*b0*((1+m1)*(1-m2i)/(1-m2)
157  - 2*m1*m1im2i/(m1-m2))/(1-m1)
158  + 2*b0*m1*m1im2i*h1/(m1-m2)
159  + m2i*h1*h1; // ko
160  Real Eh3 = pow(b0,3)*(
161  (1-m3i)/(1-m3)
162  + 3*m2*((1-m3i)/(1-m3)-m2im3i/(m2-m3))/(1-m2)
163  + 3*m1*((1-m3i)/(1-m3)-m1im3i/(m1-m3))/(1-m1)
164  + 6*m1*m2*(
165  ((1-m3i)/(1-m3)-m2im3i/(m2-m3))/(1-m2)
166  + (m2im3i/(m2-m3)-m1im3i/(m1-m3))/(m1-m2)
167  )/(1-m1))
168  + 3*b0*b0*m1*h1*(m1im3i/(m1-m3)
169  +2*m2*(m1im3i/(m1-m3)-m2im3i/(m2-m3))/(m1-m2))
170  + 3*b0*m2*h1*h1*m2im3i/(m2-m3)
171  + m3i*h1*h1*h1; // ko
172  Real Eh3_2 = .375*std::pow(Eh,-0.5)*Eh2+.625*std::pow(Eh,1.5);
173  Real Eh5_2 = 1.875*std::pow(Eh,0.5)*Eh2-.875*std::pow(Eh,2.5);
174  sEh += Eh;
175  sEh2 += Eh2;
176  sEh3 += Eh3;
177  for (j = 0; j < T-i-1; ++j) {
178  Real Ehh = b0*Eh*(1-m1ai[j+1])/(1-m1)+ Eh2*m1ai[j+1]; // ko
179  Real Ehh2 = b0*b0*Eh*((1+m1)*(1-m2ai[j+1])/(1-m2)
180  - 2*m1*(m1ai[j+1]
181  -m2ai[j+1])/(m1-m2))/(1-m1)
182  + 2*b0*m1*Eh2*(m1ai[j+1]-m2ai[j+1])/(m1-m2)
183  + m2ai[j+1]*Eh3; // ko
184  Real Eh2h = b0*Eh2*(1-m1ai[j+1])/(1-m1)
185  + m1ai[j+1]*Eh3; // ok
186  Real Eh1_2eh = v1*m1ai[j]*Eh3_2; // ko
187  Real Eh1_2eh2 = 2*b0*v1*(m1ai[j+1]
188  -m2ai[j+1])*Eh3_2/(m1-m2)
189  + v2*m2ai[j]*Eh5_2; // ko
190  Real Ehij = b0*(1-m1ai[i+j+1])/(1-m1)
191  + m1ai[i+j+1]*h1; // ko
192  Real Ehh3_2 = 0.375*Ehh2/std::sqrt(Ehij)
193  + 0.75*std::sqrt(Ehij)*Ehh
194  - 0.125*std::pow(Ehij,1.5)*Eh; // ko
195  Real Eh3_2eh = v1*m1ai[j]*Eh5_2; // ko
196  Real Eh3_2e3h = x1*m1ai[j]*Eh5_2; // ok
197  Real Eh1_2eh3_2 = 0.375*Eh1_2eh2/std::sqrt(Ehij)
198  + 0.75*std::sqrt(Ehij)*Eh1_2eh; // ko
199  sEhh += Ehh;
200  sEh1_2eh += Eh1_2eh;
201  sEhh2 += Ehh2;
202  sEh2h += Eh2h;
203  sEh1_2eh2 += Eh1_2eh2;
204  sEh3_2eh += Eh3_2eh;
205  sEhe2h += b0*Eh*(1-m1ai[j+1])/(1-m1)
206  + z1*m1ai[j]*Eh2; // ko
207  sEh3_2e3h += Eh3_2e3h; // ok
208  for (k = 0; k < T-i-j-2; ++k) {
209  Real Ehhh = b0*Ehh*(1-m1ai[k+1])/(1-m1)
210  + m1ai[k+1]*Ehh2; //ko
211  Real Eh1_2ehh = b0*Eh1_2eh*(1-m1ai[k+1])/(1-m1)
212  + m1ai[k+1]*Eh1_2eh2; // ko
213  sEhhh += Ehhh;
214  sEh1_2ehh += Eh1_2ehh;
215  sEhh1_2eh += v1*m1ai[k]*Ehh3_2; // ko
216  sEh1_2eh1_2eh += v1*m1ai[k]*Eh1_2eh3_2; // ko
217  }
218  }
219  }
220 
221  ex = T*r - 0.5*sEh;
222  SD1 = 2*sEhh + sEh2;
223  SD2 = sEh;
224  SD3 = sEh1_2eh;
225  ex2 = T*T*r*r - T*r*sEh + 0.25*SD1 + SD2 - SD3;
226  ST1 = 6*sEhhh + (3*sEhh2 + (3*sEh2h + sEh3));
227  ST2 = 3*sEh1_2eh;
228  ST3 = 2*sEhh1_2eh + (2*sEh1_2ehh + (2*sEh3_2eh + sEh1_2eh2));
229  ST4 = sEhe2h + (sEhh + (sEh2 + 2*sEh1_2eh1_2eh));
230  ex3 = pow(T*r,3) - 1.5*T*T*r*r*sEh
231  + 3*T*r*(SD1/4+SD2-SD3) + (ST2-ST1/8+3*ST3/4-3*ST4/2);
232  SQ2 = 6*sEhe2h + (12*sEh1_2eh1_2eh + 3*sEh2);
233  SQ4 = 2*sEhhh + 2*sEhh2;
234  SQ5 = 3*sEhh1_2eh + 3*sEh1_2ehh + 3*sEh3_2eh
235  + 3*sEh1_2eh2 + sEh3_2e3h;
236  ex4 = pow(T*r,4) - 2*pow(T*r,3)*sEh
237  + 6*T*T*r*r*(SD1/4+SD2-SD3) + T*r*(4*ST2-ST1/2+3*ST3-6*ST4)
238  + (SQ2+3*SQ4/2-2*SQ5);
239 
240  // compute variance, skewness, kurtosis
241  sigma = ex2 - ex*ex;
242  // 3rd central moment mu3
243  k3 = ex3 - 3*sigma*ex - ex*ex*ex;
244  // 4th central moment mu4
245  k4 = ex4 + 6*ex*ex*ex2 - 3*ex*ex*ex*ex - 4*ex*ex3;
246  k3 /= std::pow(sigma,1.5); // 3rd standardized moment, ie skewness
247  k4 /= pow(sigma,2); // 4th standardized moment, ie kurtosis
248  ex_ = ex; sigma_ = sigma;
249  k3_ = k3; k4_ = k4; r_ = r; T_ = T; b0_ = b0; h1_ = h1;
250  } else {
251  ex = ex_; sigma = sigma_;
252  k3 = k3_; k4 = k4_; r = r_; T = T_; /*b0 = b0_; h1 = h1_;*/ // never used ?
253  }
254 
255  // compute call option price
256  stdev = std::sqrt(sigma);
257  del = (ex - r*T + sigma/2)/stdev;
258  d = (std::log(s/x) + (r*T+sigma/2))/stdev;
259  d_ = d+del;
260  C = s*std::exp(del*stdev)*CumulativeNormalDistribution()(d_)
261  - x*std::exp(-r*T)*CumulativeNormalDistribution()(d_-stdev);
262  A3 = s*std::exp(del*stdev)*stdev*((2*stdev-d_)
263  *std::exp(-d_*d_/2)/std::sqrt(2*M_PI)
264  +sigma*CumulativeNormalDistribution()(d_))/6;
265  A4 = s*std::exp(del*stdev)*stdev*(
266  (d_*d_-1-3*stdev*(d_-stdev))*exp(-d_*d_/2)/std::sqrt(2*M_PI)
267  -sigma*stdev*CumulativeNormalDistribution()(d_))/24;
268  Capp = C + k3*A3 + (k4-3)*A4;
269  init_ = true;
270 
271  switch (payoff->optionType()) {
272  case Option::Call:
273  results_.value = Capp;
274  break;
275  case Option::Put:
276  results_.value = Capp+strikePrice*riskFreeDiscount/dividendDiscount
277  -spotPrice;
278  break;
279  default:
280  QL_FAIL("unknown option type");
281  }
282  }
