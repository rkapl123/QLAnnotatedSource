115 // Reverse iterators
116 // The following class makes compilation fail for the code
117 // that calls rbegin or rend with a container that does not
118 // support reverse iterators. All the rest TimeSeries class
119 // features should compile and work for this type of
120 // containers.
121 template <class container, class iterator_category>
122 struct reverse {
123 typedef boost::reverse_iterator<typename container::const_iterator>
124 const_reverse_iterator;
125 reverse(const container& c) : c_(c) {}
126 const_reverse_iterator rbegin() const {
127 return const_reverse_iterator(c_.end());
128 }
129 const_reverse_iterator rend() const {
130 return const_reverse_iterator(c_.begin());
131 }
132 const container& c_;
133 };
134
135 // This class defines reverse iterator features via
136 // container's native calls.
137 template <class container>
138 struct reverse<container, std::bidirectional_iterator_tag> {
139 typedef typename container::const_reverse_iterator
140 const_reverse_iterator;
141 reverse(const container& c) : c_(c) {}
142 const_reverse_iterator rbegin() const { return c_.rbegin(); }
143 const_reverse_iterator rend() const { return c_.rend(); }
144 const container& c_;
145 };

147 // The following typedef enables reverse iterators for
148 // bidirectional_iterator_tag category.
149 typedef typename boost::mpl::if_ <
150 boost::mpl::or_ <
151 std::is_same<iterator_category, std::bidirectional_iterator_tag>,
152 std::is_base_of<std::bidirectional_iterator_tag, iterator_category> >,
153 std::bidirectional_iterator_tag,
154 std::input_iterator_tag>::type enable_reverse;
