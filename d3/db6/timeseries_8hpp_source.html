114 // Reverse iterators
115 // The following class makes compilation fail for the code
116 // that calls rbegin or rend with a container that does not
117 // support reverse iterators. All the rest TimeSeries class
118 // features should compile and work for this type of
119 // containers.
120 template <class container, class iterator_category>
121 struct reverse {
122 typedef std::reverse_iterator<typename container::const_iterator>
123 const_reverse_iterator;
124 reverse(const container& c) : c_(c) {}
125 const_reverse_iterator rbegin() const {
126 return const_reverse_iterator(c_.end());
127 }
128 const_reverse_iterator rend() const {
129 return const_reverse_iterator(c_.begin());
130 }
131 const container& c_;
132 };
133
134 // This class defines reverse iterator features via
135 // container's native calls.
136 template <class container>
137 struct reverse<container, std::bidirectional_iterator_tag> {
138 typedef typename container::const_reverse_iterator
139 const_reverse_iterator;
140 reverse(const container& c) : c_(c) {}
141 const_reverse_iterator rbegin() const { return c_.rbegin(); }
142 const_reverse_iterator rend() const { return c_.rend(); }
143 const container& c_;
144 };

146 // The following typedef enables reverse iterators for
147 // bidirectional_iterator_tag category.
148 typedef std::conditional_t<
149 std::is_same<iterator_category, std::bidirectional_iterator_tag>::value ||
150 std::is_base_of<std::bidirectional_iterator_tag, iterator_category>::value,
151 std::bidirectional_iterator_tag, std::input_iterator_tag> enable_reverse;
