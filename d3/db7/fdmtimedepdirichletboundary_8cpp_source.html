65  void FdmTimeDepDirichletBoundary::applyAfterApplying(array_type& a) const {
66  QL_REQUIRE(indices_.size() == values_.size(),
67  "values on boundary size (" << values_.size()
68  << ") does not match hypersurface size ("
69  << indices_.size() << ")");
70  for (std::vector<Size>::const_iterator iter = indices_.begin();
71  iter != indices_.end(); ++iter) {
72  a[*iter] = values_[iter - indices_.begin()];
73  }
74  }
