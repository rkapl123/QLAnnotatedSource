33  Real flatVolCovariance(Time t1,Time t2,
34  Time T, Time S,
35  Volatility v1, Volatility v2) {
36  QL_REQUIRE(t1<=t2,
37  "integrations bounds (" << t1 <<
38  "," << t2 << ") are in reverse order");
39  Time cutOff = std::min(S,T);
40  if (t1>=cutOff) {
41  return 0.0;
42  } else {
43  cutOff = std::min(t2, cutOff);
44  return (cutOff - t1)*v1*v2;
45  }
46  }

48  FlatVol::FlatVol(
49  const vector<Volatility>& vols,
50  const ext::shared_ptr<PiecewiseConstantCorrelation>& corr,
51  const EvolutionDescription& evolution,
52  Size numberOfFactors,
53  const vector<Rate>& initialRates,
54  const vector<Spread>& displacements)
55  : numberOfFactors_(numberOfFactors),
56  numberOfRates_(initialRates.size()),
57  numberOfSteps_(evolution.evolutionTimes().size()),
58  initialRates_(initialRates),
59  displacements_(displacements),
60  evolution_(evolution),
61  pseudoRoots_(numberOfSteps_, Matrix(numberOfRates_, numberOfFactors_))
62  {
63  const vector<Time>& rateTimes = evolution.rateTimes();
64  QL_REQUIRE(numberOfRates_==rateTimes.size()-1,
65  "mismatch between number of rates (" << numberOfRates_ <<
66  ") and rate times");
67  QL_REQUIRE(numberOfRates_==displacements.size(),
68  "mismatch between number of rates (" << numberOfRates_ <<
69  ") and displacements (" << displacements.size() << ")");
70  QL_REQUIRE(numberOfRates_==vols.size(),
71  "mismatch between number of rates (" << numberOfRates_ <<
72  ") and vols (" << vols.size() << ")");
73  QL_REQUIRE(numberOfRates_<=numberOfFactors_*numberOfSteps_,
74  "number of rates (" << numberOfRates_ <<
75  ") greater than number of factors (" << numberOfFactors_
76  << ") times number of steps (" << numberOfSteps_ << ")");
77  QL_REQUIRE(numberOfFactors<=numberOfRates_,
78  "number of factors (" << numberOfFactors <<
79  ") cannot be greater than numberOfRates (" <<
80  numberOfRates_ << ")");
81  QL_REQUIRE(numberOfFactors>0,
82  "number of factors (" << numberOfFactors <<
83  ") must be greater than zero");
84 
85  Time effStopTime = 0.0;
86  const vector<Time>& corrTimes = corr->times();
87  const vector<Time>& evolTimes = evolution.evolutionTimes();
88  Matrix covariance(numberOfRates_, numberOfRates_);
89  for (Size k=0, kk=0; k<numberOfSteps_; ++k) {
90  // one covariance per evolution step
91  std::fill(covariance.begin(), covariance.end(), 0.0);
92 
93  // there might be more than one correlation matrix
94  // in a single evolution step
95  for (; corrTimes[kk]<evolTimes[k]; ++kk) {
96  Time effStartTime = effStopTime;
97  effStopTime = corrTimes[kk];
98  const Matrix& corrMatrix = corr->correlation(kk);
99  for (Size i=0; i<numberOfRates_; ++i) {
100  for (Size j=i; j<numberOfRates_; ++j) {
101  Real cov = flatVolCovariance(effStartTime, effStopTime,
102  rateTimes[i], rateTimes[j],
103  vols[i], vols[j]);
104  covariance[i][j] += cov * corrMatrix[i][j];
105  }
106  }
107  }
108  // last part in the evolution step
109  Time effStartTime = effStopTime;
110  effStopTime = evolTimes[k];
111  const Matrix& corrMatrix = corr->correlation(kk);
112  for (Size i=0; i<numberOfRates_; ++i) {
113  for (Size j=i; j<numberOfRates_; ++j) {
114  Real cov = flatVolCovariance(effStartTime, effStopTime,
115  rateTimes[i], rateTimes[j],
116  vols[i], vols[j]);
117  covariance[i][j] += cov * corrMatrix[i][j];
118  }
119  }
120  // no more use for the kk-th correlation matrix
121  while (kk<corrTimes.size() && corrTimes[kk]<=evolTimes[k])
122  ++kk;
123 
124  // make it symmetric
125  for (Size i=0; i<numberOfRates_; ++i) {
126  for (Size j=i+1; j<numberOfRates_; ++j) {
127  covariance[j][i] = covariance[i][j];
128  }
129  }
130 
131  pseudoRoots_[k] = rankReducedSqrt(covariance,
132  numberOfFactors, 1.0,
133  SalvagingAlgorithm::None);
134 
135  QL_ENSURE(pseudoRoots_[k].rows()==numberOfRates_,
136  "step " << k
137  << " flat vol wrong number of rows: "
138  << pseudoRoots_[k].rows()
139  << " instead of " << numberOfRates_);
140  QL_ENSURE(pseudoRoots_[k].columns()==numberOfFactors,
141  "step " << k
142  << " flat vol wrong number of columns: "
143  << pseudoRoots_[k].columns()
144  << " instead of " << numberOfFactors_);
145  }
146  }

166  FlatVolFactory::create(const EvolutionDescription& evolution,
167  Size numberOfFactors) const {
168  const vector<Time>& rateTimes = evolution.rateTimes();
169  Size numberOfRates = rateTimes.size()-1;
170 
171  vector<Rate> initialRates(numberOfRates);
172  for (Size i=0; i<numberOfRates; ++i)
173  initialRates[i] = yieldCurve_->forwardRate(rateTimes[i],
174  rateTimes[i+1],
175  Simple);
176 
177  vector<Volatility> displacedVolatilities(numberOfRates);
178  for (Size i=0; i<numberOfRates; ++i) {
179  Volatility vol = // to be changes
180  volatility_(rateTimes[i]);
181  displacedVolatilities[i] =
182  initialRates[i]*vol/(initialRates[i]+displacement_);
183  }
184 
185  vector<Spread> displacements(numberOfRates, displacement_);
186 
187  Matrix correlations = exponentialCorrelations(evolution.rateTimes(),
188  longTermCorrelation_,
189  beta_);
190  ext::shared_ptr<PiecewiseConstantCorrelation> corr(new
191  TimeHomogeneousForwardCorrelation(correlations,
192  rateTimes));
193  return ext::shared_ptr<MarketModel>(new
194  FlatVol(displacedVolatilities,
195  corr,
196  evolution,
197  numberOfFactors,
198  initialRates,
199  displacements));
