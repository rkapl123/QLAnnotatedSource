55  void ImplicitEulerScheme::step(array_type& a, Time t) {
56  QL_REQUIRE(t-dt_ > -1e-8, "a step towards negative time given");
57  map_->setTime(std::max(0.0, t-dt_), t);
58  bcSet_.setTime(std::max(0.0, t-dt_));
59 
60  bcSet_.applyBeforeSolving(*map_, a);
61 
62  if (map_->size() == 1) {
63  a = map_->solve_splitting(0, a, -dt_);
64  }
65  else if (solverType_ == BiCGstab) {
66  const BiCGStabResult result =
67  QuantLib::BiCGstab(
68  boost::function<Disposable<Array>(const Array&)>(
69  boost::bind(&ImplicitEulerScheme::apply, this, _1)),
70  std::max(Size(10), a.size()), relTol_,
71  boost::function<Disposable<Array>(const Array&)>(
72  boost::bind(&FdmLinearOpComposite::preconditioner,
73  map_, _1, -dt_))
74  ).solve(a, a);
75 
76  (*iterations_) += result.iterations;
77  a = result.x;
78  }
79  else if (solverType_ == GMRES) {
80  const GMRESResult result =
81  QuantLib::GMRES(
82  boost::function<Disposable<Array>(const Array&)>(
83  boost::bind(&ImplicitEulerScheme::apply, this, _1)),
84  std::max(Size(10), a.size()/10u), relTol_,
85  boost::function<Disposable<Array>(const Array&)>(
86  boost::bind(&FdmLinearOpComposite::preconditioner,
87  map_, _1, -dt_))
88  ).solve(a, a);
89 
90  (*iterations_) += result.errors.size();
91  a = result.x;
92  }
93  else
94  QL_FAIL("unknown/illegal solver type");
95 
96  bcSet_.applyAfterSolving(a);
97  }
