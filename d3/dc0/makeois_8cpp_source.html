54  MakeOIS::operator ext::shared_ptr<OvernightIndexedSwap>() const {
55 
56  Date startDate;
57  if (effectiveDate_ != Date())
58  startDate = effectiveDate_;
59  else {
60  Date refDate = Settings::instance().evaluationDate();
61  // if the evaluation date is not a business day
62  // then move to the next business day
63  refDate = calendar_.adjust(refDate);
64  Date spotDate = calendar_.advance(refDate,
65  settlementDays_*Days);
66  startDate = spotDate+forwardStart_;
67  if (forwardStart_.length()<0)
68  startDate = calendar_.adjust(startDate, Preceding);
69  else
70  startDate = calendar_.adjust(startDate, Following);
71  }
72 
73  // OIS end of month default
74  bool usedEndOfMonth =
75  isDefaultEOM_ ? calendar_.isEndOfMonth(startDate) : endOfMonth_;
76 
77  Date endDate = terminationDate_;
78  if (endDate == Date()) {
79  if (usedEndOfMonth)
80  endDate = calendar_.advance(startDate,
81  swapTenor_,
82  ModifiedFollowing,
83  usedEndOfMonth);
84  else
85  endDate = startDate + swapTenor_;
86  }
87 
88  Schedule schedule(startDate, endDate,
89  Period(paymentFrequency_),
90  calendar_,
91  ModifiedFollowing,
92  ModifiedFollowing,
93  rule_,
94  usedEndOfMonth);
95 
96  Rate usedFixedRate = fixedRate_;
97  if (fixedRate_ == Null<Rate>()) {
98  OvernightIndexedSwap temp(type_, nominal_,
99  schedule,
100  0.0, // fixed rate
101  fixedDayCount_,
102  overnightIndex_, overnightSpread_,
103  paymentLag_, paymentAdjustment_,
104  paymentCalendar_, telescopicValueDates_);
105  if (engine_ == 0) {
106  Handle<YieldTermStructure> disc =
107  overnightIndex_->forwardingTermStructure();
108  QL_REQUIRE(!disc.empty(),
109  "null term structure set to this instance of " <<
110  overnightIndex_->name());
111  bool includeSettlementDateFlows = false;
112  ext::shared_ptr<PricingEngine> engine(new
113  DiscountingSwapEngine(disc, includeSettlementDateFlows));
114  temp.setPricingEngine(engine);
115  } else
116  temp.setPricingEngine(engine_);
117 
118  usedFixedRate = temp.fairRate();
119  }
120 
121  ext::shared_ptr<OvernightIndexedSwap> ois(new
122  OvernightIndexedSwap(type_, nominal_,
123  schedule,
124  usedFixedRate, fixedDayCount_,
125  overnightIndex_, overnightSpread_,
126  paymentLag_, paymentAdjustment_,
127  paymentCalendar_, telescopicValueDates_));
128 
129  if (engine_ == 0) {
130  Handle<YieldTermStructure> disc =
131  overnightIndex_->forwardingTermStructure();
132  bool includeSettlementDateFlows = false;
133  ext::shared_ptr<PricingEngine> engine(new
134  DiscountingSwapEngine(disc, includeSettlementDateFlows));
135  ois->setPricingEngine(engine);
136  } else
137  ois->setPricingEngine(engine_);
138 
139  return ois;
140  }
