89  void HestonSLVMCModel::performCalculations() const {
90  const boost::shared_ptr<HestonProcess> hestonProcess
91  = hestonModel_->process();
92  const boost::shared_ptr<Quote> spot
93  = hestonProcess->s0().currentLink();
94  const boost::shared_ptr<YieldTermStructure> rTS
95  = hestonProcess->riskFreeRate().currentLink();
96  const boost::shared_ptr<YieldTermStructure> qTS
97  = hestonProcess->dividendYield().currentLink();
98 
99  const Real v0 = hestonProcess->v0();
100  const DayCounter dc = hestonProcess->riskFreeRate()->dayCounter();
101  const Date referenceDate = hestonProcess->riskFreeRate()->referenceDate();
102 
103  const Volatility lv0
104  = localVol_->localVol(0.0, spot->value())/std::sqrt(v0);
105 
106  const boost::shared_ptr<Matrix> L(new Matrix(nBins_, timeGrid_->size()));
107 
108  std::vector<boost::shared_ptr<std::vector<Real> > >
109  vStrikes(timeGrid_->size());
110  for (Size i=0; i < timeGrid_->size(); ++i) {
111  const Integer u = nBins_/2;
112  const Real dx = spot->value()*std::sqrt(QL_EPSILON);
113 
114  vStrikes[i] = boost::make_shared<std::vector<Real> >(nBins_);
115 
116  for (Integer j=0; j < Integer(nBins_); ++j)
117  vStrikes[i]->at(j) = spot->value() + (j - u)*dx;
118  }
119 
120  std::fill(L->column_begin(0),L->column_end(0), lv0);
121 
122  leverageFunction_ = boost::make_shared<FixedLocalVolSurface>(
123  referenceDate,
124  std::vector<Time>(timeGrid_->begin(), timeGrid_->end()),
125  vStrikes, L, dc);
126 
127  const boost::shared_ptr<HestonSLVProcess> slvProcess
128  = boost::make_shared<HestonSLVProcess>(hestonProcess, leverageFunction_);
129 
130  std::vector<std::pair<Real, Real> > pairs(
131  calibrationPaths_, std::make_pair(spot->value(), v0));
132 
133  const Size k = calibrationPaths_ / nBins_;
134  const Size m = calibrationPaths_ % nBins_;
135 
136  const Size timeSteps = timeGrid_->size()-1;
137 
138  typedef boost::multi_array<Real, 3> path_type;
139  path_type paths(boost::extents[calibrationPaths_][timeSteps][2]);
140 
141  const boost::shared_ptr<BrownianGenerator> brownianGenerator =
142  brownianGeneratorFactory_->create(2, timeSteps);
143 
144  for (Size i=0; i < calibrationPaths_; ++i) {
145  brownianGenerator->nextPath();
146  std::vector<Real> tmp(2);
147  for (Size j=0; j < timeSteps; ++j) {
148  brownianGenerator->nextStep(tmp);
149  paths[i][j][0] = tmp[0];
150  paths[i][j][1] = tmp[1];
151  }
152  }
153 
154  for (Size n=1; n < timeGrid_->size(); ++n) {
155  const Time t = timeGrid_->at(n-1);
156  const Time dt = timeGrid_->dt(n-1);
157 
158  Array x0(2), dw(2);
159 
160  for (Size i=0; i < calibrationPaths_; ++i) {
161  x0[0] = pairs[i].first;
162  x0[1] = pairs[i].second;
163 
164  dw[0] = paths[i][n-1][0];
165  dw[1] = paths[i][n-1][1];
166 
167  x0 = slvProcess->evolve(t, x0, dt, dw);
168 
169  pairs[i].first = x0[0];
170  pairs[i].second = x0[1];
171  }
172 
173  std::sort(pairs.begin(), pairs.end());
174 
175  Size s = 0u, e = 0u;
176  for (Size i=0; i < nBins_; ++i) {
177  const Size inc = k + (i < m);
178  e = s + inc;
179 
180  Real sum=0.0;
181  for (Size j=s; j < e; ++j) {
182  sum+=pairs[j].second;
183  }
184  sum/=inc;
185 
186  vStrikes[n]->at(i) = 0.5*(pairs[e-1].first + pairs[s].first);
187  (*L)[i][n] = std::sqrt(square<Real>()(
188  localVol_->localVol(t, vStrikes[n]->at(i), true))/sum);
189 
190  s = e;
191  }
192 
193  leverageFunction_->setInterpolation<Linear>();
194  }
195  }
