36 explicit FdmLinearOpLayout(std::vector<Size> dim)
37 : dim_(std::move(dim)), spacing_(dim_.size()) {
38 spacing_[0] = 1;
39 std::partial_sum(dim_.begin(), dim_.end()-1,
40 spacing_.begin()+1, std::multiplies<>());
41
42 size_ = spacing_.back()*dim_.back();
43 }
