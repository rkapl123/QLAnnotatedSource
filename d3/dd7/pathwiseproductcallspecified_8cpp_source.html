29  CallSpecifiedPathwiseMultiProduct::CallSpecifiedPathwiseMultiProduct(
30  const Clone<MarketModelPathwiseMultiProduct>& underlying,
31  const Clone<ExerciseStrategy<CurveState> >& strategy,
32  const Clone<MarketModelPathwiseMultiProduct>& rebate)
33  : underlying_(underlying),
34  strategy_(strategy),
35  rebate_(rebate),
36  callable_(true)
37  {
38 
39  Size products = underlying_->numberOfProducts();
40  EvolutionDescription d1 = underlying->evolution();
41  const std::vector<Time>& rateTimes1 = d1.rateTimes();
42  const std::vector<Time>& evolutionTimes1 = d1.evolutionTimes();
43  const std::vector<Time>& exerciseTimes = strategy->exerciseTimes();
44 
45  if (!rebate_.empty())
46  {
47  EvolutionDescription d2 = rebate_->evolution();
48  const std::vector<Time>& rateTimes2 = d2.rateTimes();
49  QL_REQUIRE(rateTimes1.size() == rateTimes2.size() &&
50  std::equal(rateTimes1.begin(), rateTimes1.end(),
51  rateTimes2.begin()),
52  "incompatible rate times");
53 
54  QL_REQUIRE(underlying_->alreadyDeflated() == rebate_->alreadyDeflated(), "incompatible deflations");
55  }
56  else
57  {
58  EvolutionDescription description(rateTimes1, exerciseTimes);
59  Matrix amounts(products, exerciseTimes.size(), 0.0);
60 
61  rebate_ = MarketModelPathwiseCashRebate(description,
62  exerciseTimes,
63  amounts,
64  products);
65  }
66 
67  std::vector<Time> mergedEvolutionTimes;
68  std::vector<std::vector<Time> > allEvolutionTimes(4);
69  allEvolutionTimes[0] = evolutionTimes1;
70  allEvolutionTimes[1] = exerciseTimes;
71  allEvolutionTimes[2] = rebate_->evolution().evolutionTimes();
72  allEvolutionTimes[3] = strategy->relevantTimes();
73 
74  mergeTimes(allEvolutionTimes,
75  mergedEvolutionTimes,
76  isPresent_);
77 
78  // TODO: add relevant rates
79  evolution_ = EvolutionDescription(rateTimes1, mergedEvolutionTimes);
80 
81  cashFlowTimes_ = underlying_->possibleCashFlowTimes();
82  rebateOffset_ = cashFlowTimes_.size();
83  const std::vector<Time> rebateTimes = rebate_->possibleCashFlowTimes();
84  cashFlowTimes_.insert(cashFlowTimes_.end(),
85  rebateTimes.begin(), rebateTimes.end());
86 
87  dummyCashFlowsThisStep_ = std::vector<Size>(products, 0);
88  Size n = rebate_->maxNumberOfCashFlowsPerProductPerStep();
89 
90  CashFlow modelCashFlow;
91  modelCashFlow.amount.resize(d1.numberOfRates()+1);
92 
93  dummyCashFlowsGenerated_ =
94  std::vector<std::vector<CashFlow> >(products,
95  std::vector<CashFlow>(n, modelCashFlow));
96  }

142  bool CallSpecifiedPathwiseMultiProduct::nextTimeStep(
143  const CurveState& currentState,
144  std::vector<Size>& numberCashFlowsThisStep,
145  std::vector<std::vector<CashFlow> >& cashFlowsGenerated)
146  {
147 
148  bool isUnderlyingTime = isPresent_[0][currentIndex_];
149  bool isExerciseTime = isPresent_[1][currentIndex_];
150  bool isRebateTime = isPresent_[2][currentIndex_];
151  bool isStrategyRelevantTime = isPresent_[3][currentIndex_];
152 
153  bool done = false;
154 
155  if (!wasCalled_ && isStrategyRelevantTime)
156  strategy_->nextStep(currentState);
157 
158 
159  if (!wasCalled_ && isExerciseTime && callable_)
160  wasCalled_ = strategy_->exercise(currentState);
161 
162  if (wasCalled_)
163  {
164  if (isRebateTime)
165  {
166  done = rebate_->nextTimeStep(currentState,
167  numberCashFlowsThisStep,
168  cashFlowsGenerated);
169  for (Size i=0; i<numberCashFlowsThisStep.size(); ++i)
170  for (Size j=0; j<numberCashFlowsThisStep[i]; ++j)
171  cashFlowsGenerated[i][j].timeIndex += rebateOffset_;
172  }
173  }
174  else
175  {
176  if (isRebateTime)
177  rebate_->nextTimeStep(currentState,
178  dummyCashFlowsThisStep_,
179  dummyCashFlowsGenerated_);
180  if (isUnderlyingTime)
181  done = underlying_->nextTimeStep(currentState,
182  numberCashFlowsThisStep,
183  cashFlowsGenerated);
184  }
185 
186  ++currentIndex_;
187  return done || currentIndex_ == evolution_.evolutionTimes().size();
188  }
