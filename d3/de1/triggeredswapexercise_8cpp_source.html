27  TriggeredSwapExercise::TriggeredSwapExercise(
28  const std::vector<Time>& rateTimes,
29  const std::vector<Time>& exerciseTimes,
30  const std::vector<Rate>& strikes)
31  : rateTimes_(rateTimes), exerciseTimes_(exerciseTimes),
32  strikes_(strikes), currentStep_(0), rateIndex_(exerciseTimes.size()),
33  evolution_(rateTimes, exerciseTimes) {
34  Size j = 0;
35  for (Size i=0; i<exerciseTimes.size(); ++i) {
36  while (j < rateTimes.size() && rateTimes[j] < exerciseTimes[i])
37  ++j;
38  rateIndex_[i] = j;
39  }
40  }
