28  TriggeredSwapExercise::TriggeredSwapExercise(const std::vector<Time>& rateTimes,
29  const std::vector<Time>& exerciseTimes,
30  std::vector<Rate> strikes)
31  : rateTimes_(rateTimes), exerciseTimes_(exerciseTimes), strikes_(std::move(strikes)),
32  currentStep_(0), rateIndex_(exerciseTimes.size()), evolution_(rateTimes, exerciseTimes) {
33  Size j = 0;
34  for (Size i=0; i<exerciseTimes.size(); ++i) {
35  while (j < rateTimes.size() && rateTimes[j] < exerciseTimes[i])
36  ++j;
37  rateIndex_[i] = j;
38  }
39  }
