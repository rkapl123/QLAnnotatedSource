53  void FdmBlackScholesOp::setTime(Time t1, Time t2) {
54  const Rate r = rTS_->forwardRate(t1, t2, Continuous).rate();
55  const Rate q = qTS_->forwardRate(t1, t2, Continuous).rate();
56 
57  if (localVol_) {
58  const boost::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
59  const FdmLinearOpIterator endIter = layout->end();
60 
61  Array v(layout->size());
62  for (FdmLinearOpIterator iter = layout->begin();
63  iter!=endIter; ++iter) {
64  const Size i = iter.index();
65 
66  if (illegalLocalVolOverwrite_ < 0.0) {
67  v[i] = square<Real>()(
68  localVol_->localVol(0.5*(t1+t2), x_[i], true));
69  }
70  else {
71  try {
72  v[i] = square<Real>()(
73  localVol_->localVol(0.5*(t1+t2), x_[i], true));
74  } catch (Error&) {
75  v[i] = square<Real>()(illegalLocalVolOverwrite_);
76  }
77 
78  }
79  }
80  mapT_.axpyb(r - q - 0.5*v, dxMap_,
81  dxxMap_.mult(0.5*v), Array(1, -r));
82  }
83  else {
84  const Real v
85  = volTS_->blackForwardVariance(t1, t2, strike_)/(t2-t1);
86  mapT_.axpyb(Array(1, r - q - 0.5*v), dxMap_,
87  dxxMap_.mult(0.5*Array(mesher_->layout()->size(), v)),
88  Array(1, -r));
89  }
90  }
