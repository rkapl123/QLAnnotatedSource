167  Real integral(Real p) const {
168  QL_REQUIRE(p >= 0 && p <= 1, "probability p=" << p
169  << " out of range [0,1]");
170  calculate();
171 
172  Real avg = 0;
173  for (Size k = 0; k < steps(); k++) {
174  Real pp = conditionalProbability(p, m(k));
175  avg += pp * densitydm(k);
176  }
177  return avg;
178  }

195  Real avg = 0.0;
196  for (Size i = 0; i < steps_; i++) {
197  std::vector<Real> conditional

215  template <class F>
216  Distribution integral(const F& f,
217  const std::vector<Real>& nominals,
218  const std::vector<Real>& probabilities) const {
219  calculate();
220 
221  Distribution dist(f.buckets(), 0.0, f.maximum());
222  for (Size i = 0; i < steps(); i++) {
223  std::vector<Real> conditional
224  = conditionalProbability(probabilities, m(i));
225  Distribution d = f(nominals, conditional);
226  for (Size j = 0; j < dist.size(); j++)
227  dist.addDensity(j, d.density(j) * densitydm(i));
228  }
229  return dist;
230  }
