26  SecondOrderMixedDerivativeOp::SecondOrderMixedDerivativeOp(
27  Size d0, Size d1,
28  const ext::shared_ptr<FdmMesher>& mesher)
29  : NinePointLinearOp(d0, d1, mesher) {
30 
31  const ext::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
32  const FdmLinearOpIterator endIter = layout->end();
33 
34  for (FdmLinearOpIterator iter = layout->begin(); iter!=endIter; ++iter) {
35  const Size i = iter.index();
36  const Real hm_d0 = mesher->dminus(iter, d0_);
37  const Real hp_d0 = mesher->dplus(iter, d0_);
38  const Real hm_d1 = mesher->dminus(iter, d1_);
39  const Real hp_d1 = mesher->dplus(iter, d1_);
40 
41  const Real zetam1 = hm_d0*(hm_d0+hp_d0);
42  const Real zeta0 = hm_d0*hp_d0;
43  const Real zetap1 = hp_d0*(hm_d0+hp_d0);
44  const Real phim1 = hm_d1*(hm_d1+hp_d1);
45  const Real phi0 = hm_d1*hp_d1;
46  const Real phip1 = hp_d1*(hm_d1+hp_d1);
47 
48  const Size c0 = iter.coordinates()[d0_];
49  const Size c1 = iter.coordinates()[d1_];
50  if (c0 == 0 && c1 == 0) {
51  // lower left corner
52  a00_[i] = a01_[i] = a02_[i] = a10_[i] = a20_[i] = 0.0;
53  a21_[i] = a12_[i] = -(a11_[i] = a22_[i] = 1.0/(hp_d0*hp_d1));
54  }
55  else if (c0 == layout->dim()[d0_]-1 && c1 == 0) {
56  // upper left corner
57  a22_[i] = a21_[i] = a20_[i] = a10_[i] = a00_[i] = 0.0;
58  a11_[i] = a02_[i] = -(a01_[i] = a12_[i] = 1.0/(hm_d0*hp_d1));
59  }
60  else if (c0 == 0 && c1 == layout->dim()[d1_]-1) {
61  // lower right corner
62  a00_[i] = a01_[i] = a02_[i] = a12_[i] = a22_[i] = 0.0;
63  a20_[i] = a11_[i] = -(a10_[i] = a21_[i] = 1.0/(hp_d0*hm_d1));
64  }
65  else if (c0 == layout->dim()[d0_]-1 && c1 == layout->dim()[d1_]-1) {
66  // upper right corner
67  a20_[i] = a21_[i] = a22_[i] = a12_[i] = a02_[i] = 0.0;
68  a10_[i] = a01_[i] = -(a00_[i] = a11_[i] = 1.0/(hm_d0*hm_d1));
69  }
70  else if (c0 == 0) {
71  // lower side
72  a00_[i] = a01_[i] = a02_[i] = 0.0;
73 
74  a20_[i] = -(a10_[i] = hp_d1/(hp_d0*phim1));
75  a11_[i] = -(a21_[i] = (hp_d1-hm_d1)/(hp_d0*phi0));
76  a12_[i] = -(a22_[i] = hm_d1/(hp_d0*phip1));
77  }
78  else if (c0 == layout->dim()[d0_]-1) {
79  // upper side
80  a20_[i] = a21_[i] = a22_[i] = 0.0;
81 
82  a10_[i] = -(a00_[i] = hp_d1/(hm_d0*phim1));
83  a01_[i] = -(a11_[i] = (hp_d1-hm_d1)/(hm_d0*phi0));
84  a02_[i] = -(a12_[i] = hm_d1/(hm_d0*phip1));
85  }
86  else if (c1 == 0) {
87  // left side
88  a00_[i] = a10_[i] = a20_[i] = 0.0;
89 
90  a02_[i] = -(a01_[i] = hp_d0/(zetam1*hp_d1));
91  a11_[i] = -(a12_[i] = (hp_d0-hm_d0)/(zeta0*hp_d1));
92  a21_[i] = -(a22_[i] = hm_d0/(zetap1*hp_d1));
93  }
94  else if (c1 == layout->dim()[d1_]-1) {
95  // right side
96  a22_[i] = a12_[i] = a02_[i] = 0.0;
97 
98  a01_[i] = -(a00_[i] = hp_d0/(zetam1*hm_d1));
99  a10_[i] = -(a11_[i] = (hp_d0-hm_d0)/(zeta0*hm_d1));
100  a20_[i] = -(a21_[i] = hm_d0/(zetap1*hm_d1));
101  }
102  else {
103  a00_[i] = hp_d0*hp_d1/(zetam1*phim1);
104  a10_[i] = -(hp_d0-hm_d0)*hp_d1/(zeta0*phim1);
105  a20_[i] = -hm_d0*hp_d1/(zetap1*phim1);
106  a01_[i] = -hp_d0*(hp_d1-hm_d1)/(zetam1*phi0);
107  a11_[i] = (hp_d0-hm_d0)*(hp_d1-hm_d1)/(zeta0*phi0);
108  a21_[i] = hm_d0*(hp_d1-hm_d1)/(zetap1*phi0);
109  a02_[i] = -hp_d0*hm_d1/(zetam1*phip1);
110  a12_[i] = hm_d1*(hp_d0-hm_d0)/(zeta0*phip1);
111  a22_[i] = hm_d0*hm_d1/(zetap1*phip1);
112  }
113  }
114  }
