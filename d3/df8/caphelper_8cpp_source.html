79  void CapHelper::performCalculations() const {
80 
81  Period indexTenor = index_->tenor();
82  Rate fixedRate = 0.04; // dummy value
83  Date startDate, maturity;
84  if (includeFirstSwaplet_) {
85  startDate = termStructure_->referenceDate();
86  maturity = termStructure_->referenceDate() + length_;
87  } else {
88  startDate = termStructure_->referenceDate() + indexTenor;
89  maturity = termStructure_->referenceDate() + length_;
90  }
91  boost::shared_ptr<IborIndex> dummyIndex(new
92  IborIndex("dummy",
93  indexTenor,
94  index_->fixingDays(),
95  index_->currency(),
96  index_->fixingCalendar(),
97  index_->businessDayConvention(),
98  index_->endOfMonth(),
99  termStructure_->dayCounter(),
100  termStructure_));
101 
102  std::vector<Real> nominals(1,1.0);
103 
104  Schedule floatSchedule(startDate, maturity,
105  index_->tenor(), index_->fixingCalendar(),
106  index_->businessDayConvention(),
107  index_->businessDayConvention(),
108  DateGeneration::Forward, false);
109  Leg floatingLeg = IborLeg(floatSchedule, index_)
110  .withNotionals(nominals)
111  .withPaymentAdjustment(index_->businessDayConvention())
112  .withFixingDays(0);
113 
114  Schedule fixedSchedule(startDate, maturity, Period(fixedLegFrequency_),
115  index_->fixingCalendar(),
116  Unadjusted, Unadjusted,
117  DateGeneration::Forward, false);
118  Leg fixedLeg = FixedRateLeg(fixedSchedule)
119  .withNotionals(nominals)
120  .withCouponRates(fixedRate, fixedLegDayCounter_)
121  .withPaymentAdjustment(index_->businessDayConvention());
122 
123  Swap swap(floatingLeg, fixedLeg);
124  swap.setPricingEngine(boost::shared_ptr<PricingEngine>(
125  new DiscountingSwapEngine(termStructure_, false)));
126  Rate fairRate = fixedRate - swap.NPV()/(swap.legBPS(1)/1.0e-4);
127  cap_ = boost::shared_ptr<Cap>(new Cap(floatingLeg,
128  std::vector<Rate>(1, fairRate)));
129 
130  CalibrationHelper::performCalculations();
131 
132  }
