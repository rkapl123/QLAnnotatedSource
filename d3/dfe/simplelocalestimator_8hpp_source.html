42  TimeSeries<Volatility>
43  calculate(const TimeSeries<Real> &quoteSeries) {
44  TimeSeries<Volatility> retval;
45  TimeSeries<Real>::const_iterator prev, next, cur, start;
46  start = quoteSeries.begin();
47  ++start;
48  for (cur = start; cur != quoteSeries.end(); ++cur) {
49  prev = cur; --prev;
50  retval[cur->first] =
51  std::fabs(std::log(cur->second/prev->second))/
52  std::sqrt(yearFraction_);
53  }
54  return retval;
55  }
