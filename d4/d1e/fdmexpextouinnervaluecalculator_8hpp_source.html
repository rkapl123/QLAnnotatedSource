38  FdmExpExtOUInnerValueCalculator(
39  const ext::shared_ptr<Payoff>& payoff,
40  const ext::shared_ptr<FdmMesher>& mesher,
41  const ext::shared_ptr<Shape>& shape = ext::shared_ptr<Shape>(),
42  Size direction = 0)
43  : direction_(direction),
44  payoff_(payoff),

48  Real innerValue(const FdmLinearOpIterator& iter, Time t) {
49  const Real u = mesher_->location(iter, direction_);
50 
51  Real f = 0;
52  if (shape_ != 0) {
53  f = std::lower_bound(shape_->begin(), shape_->end(),
54  std::pair<Time, Real>(t-std::sqrt(QL_EPSILON), 0.0))->second;
55  }
56 
57  return (*payoff_)(std::exp(f + u));
58  }
59  Real avgInnerValue(const FdmLinearOpIterator& iter, Time t) {
60  return innerValue(iter, t);
61  }
62  private:
63  const Size direction_;
64  const ext::shared_ptr<Payoff> payoff_;
65  const ext::shared_ptr<FdmMesher> mesher_;
66  const ext::shared_ptr<Shape> shape_;
67  };
