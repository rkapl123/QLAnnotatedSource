28  Real Gaussian1dModel::forwardRate(const Date& fixing,
29  const Date& referenceDate,
30  const Real y,
31  const ext::shared_ptr<IborIndex>& iborIdx) const {
32 
33  QL_REQUIRE(iborIdx != NULL, "no ibor index given");
34 
35  calculate();
36 
37  if (fixing <= (evaluationDate_ + (enforcesTodaysHistoricFixings_ ? 0 : -1)))
38  return iborIdx->fixing(fixing);
39 
40  Handle<YieldTermStructure> yts = iborIdx->forwardingTermStructure(); // might be empty, then
41  // use model curve
42 
43  Date valueDate = iborIdx->valueDate(fixing);
44  Date endDate = iborIdx->fixingCalendar().advance(
45  valueDate, iborIdx->tenor(), iborIdx->businessDayConvention(), iborIdx->endOfMonth());
46  // FIXME Here we should use the calculation date calendar ?
47  Real dcf = iborIdx->dayCounter().yearFraction(valueDate, endDate);
48 
49  return (zerobond(valueDate, referenceDate, y, yts) -
50  zerobond(endDate, referenceDate, y, yts)) /
51  (dcf * zerobond(endDate, referenceDate, y, yts));
52 }

54 Real Gaussian1dModel::swapRate(const Date& fixing,
55  const Period& tenor,
56  const Date& referenceDate,
57  const Real y,
58  const ext::shared_ptr<SwapIndex>& swapIdx) const {
59 
60  QL_REQUIRE(swapIdx != NULL, "no swap index given");
61 
62  calculate();
63 
64  if (fixing <= (evaluationDate_ + (enforcesTodaysHistoricFixings_ ? 0 : -1)))
65  return swapIdx->fixing(fixing);
66 
67  Handle<YieldTermStructure> ytsf =
68  swapIdx->iborIndex()->forwardingTermStructure();
69  Handle<YieldTermStructure> ytsd =
70  swapIdx->discountingTermStructure(); // either might be empty, then
71  // use model curve
72 
73  Schedule sched, floatSched;
74 
75  ext::shared_ptr<VanillaSwap> underlying =
76  underlyingSwap(swapIdx, fixing, tenor);
77 
78  sched = underlying->fixedSchedule();
79 
80  ext::shared_ptr<OvernightIndexedSwapIndex> oisIdx =
81  ext::dynamic_pointer_cast<OvernightIndexedSwapIndex>(swapIdx);
82  if (oisIdx != NULL) {
83  floatSched = sched;
84  } else {
85  floatSched = underlying->floatingSchedule();
86  }
87 
88  Real annuity = swapAnnuity(fixing, tenor, referenceDate, y,
89  swapIdx); // should be fine for
90  // overnightindexed swap indices as
91  // well
92  Rate floatleg = 0.0;
93  if (ytsf.empty() && ytsd.empty()) { // simple 100-formula can be used
94  // only in one curve setup
95  floatleg =
96  (zerobond(sched.dates().front(), referenceDate, y) -
97  zerobond(sched.calendar().adjust(sched.dates().back(),
98  underlying->paymentConvention()),
99  referenceDate, y));
100  } else {
101  for (Size i = 1; i < floatSched.size(); i++) {
102  floatleg +=
103  (zerobond(floatSched[i - 1], referenceDate, y, ytsf) /
104  zerobond(floatSched[i], referenceDate, y, ytsf) -
105  1.0) *
106  zerobond(floatSched.calendar().adjust(
107  floatSched[i], underlying->paymentConvention()),
108  referenceDate, y, ytsd);
109  }
110  }
111  return floatleg / annuity;
112 }

114 Real Gaussian1dModel::swapAnnuity(const Date& fixing,
115  const Period& tenor,
116  const Date& referenceDate,
117  const Real y,
118  const ext::shared_ptr<SwapIndex>& swapIdx) const {
119 
120  QL_REQUIRE(swapIdx != NULL, "no swap index given");
121 
122  calculate();
123 
124  Handle<YieldTermStructure> ytsd =
125  swapIdx->discountingTermStructure(); // might be empty, then use
126  // model curve
127 
128  ext::shared_ptr<VanillaSwap> underlying =
129  underlyingSwap(swapIdx, fixing, tenor);
130 
131  Schedule sched = underlying->fixedSchedule();
132 
133  Real annuity = 0.0;
134  for (unsigned int j = 1; j < sched.size(); j++) {
135  annuity += zerobond(sched.calendar().adjust(
136  sched.date(j), underlying->paymentConvention()),
137  referenceDate, y, ytsd) *
138  swapIdx->dayCounter().yearFraction(sched.date(j - 1),
139  sched.date(j));
140  }
141  return annuity;
142 }

144 Real Gaussian1dModel::zerobondOption(
145  const Option::Type &type, const Date &expiry, const Date &valueDate,
146  const Date &maturity, const Rate strike, const Date &referenceDate,
147  const Real y, const Handle<YieldTermStructure> &yts, const Real yStdDevs,
148  const Size yGridPoints, const bool extrapolatePayoff,
149  const bool flatPayoffExtrapolation) const {
150 
151  calculate();
152 
153  Time fixingTime = termStructure()->timeFromReference(expiry);
154  Time referenceTime =
155  referenceDate == Null<Date>()
156  ? 0.0
157  : termStructure()->timeFromReference(referenceDate);
158 
159  Array yg = yGrid(yStdDevs, yGridPoints, fixingTime, referenceTime, y);
160  Array z = yGrid(yStdDevs, yGridPoints);
161 
162  Array p(yg.size());
163 
164  for (Size i = 0; i < yg.size(); i++) {
165  Real expValDsc = zerobond(valueDate, expiry, yg[i], yts);
166  Real discount =
167  zerobond(maturity, expiry, yg[i], yts) / expValDsc;
168  p[i] =
169  std::max((type == Option::Call ? 1.0 : -1.0) * (discount - strike),
170  0.0) /
171  numeraire(fixingTime, yg[i], yts) * expValDsc;
172  }
173 
174  CubicInterpolation payoff(
175  z.begin(), z.end(), p.begin(), CubicInterpolation::Spline, true,
176  CubicInterpolation::Lagrange, 0.0, CubicInterpolation::Lagrange, 0.0);
177 
178  Real price = 0.0;
179  for (Size i = 0; i < z.size() - 1; i++) {
180  price += gaussianShiftedPolynomialIntegral(
181  0.0, payoff.cCoefficients()[i], payoff.bCoefficients()[i],
182  payoff.aCoefficients()[i], p[i], z[i], z[i], z[i + 1]);
183  }
184  if (extrapolatePayoff) {
185  if (flatPayoffExtrapolation) {
186  price += gaussianShiftedPolynomialIntegral(
187  0.0, 0.0, 0.0, 0.0, p[z.size() - 2], z[z.size() - 2],
188  z[z.size() - 1], 100.0);
189  price += gaussianShiftedPolynomialIntegral(0.0, 0.0, 0.0, 0.0, p[0],
190  z[0], -100.0, z[0]);
191  } else {
192  if (type == Option::Call)
193  price += gaussianShiftedPolynomialIntegral(
194  0.0, payoff.cCoefficients()[z.size() - 2],
195  payoff.bCoefficients()[z.size() - 2],
196  payoff.aCoefficients()[z.size() - 2], p[z.size() - 2],
197  z[z.size() - 2], z[z.size() - 1], 100.0);
198  if (type == Option::Put)
199  price += gaussianShiftedPolynomialIntegral(
200  0.0, payoff.cCoefficients()[0], payoff.bCoefficients()[0],
201  payoff.aCoefficients()[0], p[0], z[0], -100.0, z[0]);
202  }
203  }
204 
205  return numeraire(referenceTime, y, yts) * price;
206 }

208 Real Gaussian1dModel::gaussianPolynomialIntegral(
209  const Real a, const Real b, const Real c, const Real d, const Real e,
210  const Real y0, const Real y1) {
211 
212 #ifdef GAUSS1D_ENABLE_NTL
213  const boost::math::ntl::RR aa = 4.0 * a, ba = 2.0 * M_SQRT2 * b,
214  ca = 2.0 * c, da = M_SQRT2 * d;
215  const boost::math::ntl::RR x0 = y0 * M_SQRT1_2, x1 = y1 * M_SQRT1_2;
216  const boost::math::ntl::RR res =
217  (0.125 * (3.0 * aa + 2.0 * ca + 4.0 * e) * boost::math::erf(x1) -
218  1.0 / (4.0 * M_SQRTPI) * exp(-x1 * x1) *
219  (2.0 * aa * x1 * x1 * x1 + 3.0 * aa * x1 +
220  2.0 * ba * (x1 * x1 + 1.0) + 2.0 * ca * x1 + 2.0 * da)) -
221  (0.125 * (3.0 * aa + 2.0 * ca + 4.0 * e) * boost::math::erf(x0) -
222  1.0 / (4.0 * M_SQRTPI) * exp(-x0 * x0) *
223  (2.0 * aa * x0 * x0 * x0 + 3.0 * aa * x0 +
224  2.0 * ba * (x0 * x0 + 1.0) + 2.0 * ca * x0 + 2.0 * da));
225  return NTL::to_double(res.value());
226 #else
227  const Real aa = 4.0 * a, ba = 2.0 * M_SQRT2 * b, ca = 2.0 * c,
228  da = M_SQRT2 * d;
229  const Real x0 = y0 * M_SQRT1_2, x1 = y1 * M_SQRT1_2;
230  return (0.125 * (3.0 * aa + 2.0 * ca + 4.0 * e) * boost::math::erf(x1) -
231  1.0 / (4.0 * M_SQRTPI) * exp(-x1 * x1) *
232  (2.0 * aa * x1 * x1 * x1 + 3.0 * aa * x1 +
233  2.0 * ba * (x1 * x1 + 1.0) + 2.0 * ca * x1 + 2.0 * da)) -
234  (0.125 * (3.0 * aa + 2.0 * ca + 4.0 * e) * boost::math::erf(x0) -
235  1.0 / (4.0 * M_SQRTPI) * exp(-x0 * x0) *
236  (2.0 * aa * x0 * x0 * x0 + 3.0 * aa * x0 +
237  2.0 * ba * (x0 * x0 + 1.0) + 2.0 * ca * x0 + 2.0 * da));
238 #endif
239 }

241 Real Gaussian1dModel::gaussianShiftedPolynomialIntegral(
242  const Real a, const Real b, const Real c, const Real d, const Real e,
243  const Real h, const Real x0, const Real x1) {
244  return gaussianPolynomialIntegral(
245  a, -4.0 * a * h + b, 6.0 * a * h * h - 3.0 * b * h + c,
246  -4 * a * h * h * h + 3.0 * b * h * h - 2.0 * c * h + d,
247  a * h * h * h * h - b * h * h * h + c * h * h - d * h + e, x0, x1);
248 }

250 Disposable<Array> Gaussian1dModel::yGrid(
251  const Real stdDevs, const int gridPoints, const Real T, const Real t, const Real y) const {
252 
253  // we use that the standard deviation is independent of $x$ here !
254 
255  QL_REQUIRE(stateProcess_ != NULL, "state process not set");
256 
257  Array result(2 * gridPoints + 1, 0.0);
258 
259  Real x_t, e_0_t, e_t_T, stdDev_0_t, stdDev_t_T;
260  Real stdDev_0_T = stateProcess_->stdDeviation(0.0, 0.0, T);
261  Real e_0_T = stateProcess_->expectation(0.0, 0.0, T);
262 
263  if (t < QL_EPSILON) {
264  // stdDev_0_t = 0.0;
265  stdDev_t_T = stdDev_0_T;
266  // e_0_t = 0.0;
267  // x_t = 0.0;
268  e_t_T = e_0_T;
269  } else {
270  stdDev_0_t = stateProcess_->stdDeviation(0.0, 0.0, t);
271  stdDev_t_T = stateProcess_->stdDeviation(t, 0.0, T - t);
272  e_0_t = stateProcess_->expectation(0.0, 0.0, t);
273  x_t = y * stdDev_0_t + e_0_t;
274  e_t_T = stateProcess_->expectation(t, x_t, T - t);
275  }
276 
277  Real h = stdDevs / ((Real)gridPoints);
278 
279  for (int j = -gridPoints; j <= gridPoints; j++) {
280  result[j + gridPoints] =
281  (e_t_T + stdDev_t_T * ((Real)j) * h - e_0_T) / stdDev_0_T;
282  }
283 
284  return result;
285 }
