31  Real MTBrownianGenerator::nextStep(std::vector<Real>& output) {
32  #if defined(QL_EXTRA_SAFETY_CHECKS)
33  QL_REQUIRE(output.size() == factors_, "size mismatch");
34  QL_REQUIRE(lastStep_<steps_, "uniform sequence exhausted");
35  #endif
36  // no copying, just fetching a reference
37  const std::vector<Real>& currentSequence = generator_.lastSequence().value;
38  Size start = lastStep_*factors_, end = (lastStep_+1)*factors_;
39  std::transform(currentSequence.begin()+start,
40  currentSequence.begin()+end,
41  output.begin(),
42  inverseCumulative_);
43  ++lastStep_;
44  return 1.0;
45  }
