63  struct cFunction {
64  // this is just a helper class where we do not want virtual
65  // functions
66  cFunction(Real f, Real s, Real a, Real b)
67  : f_(f), s_(s), a_(a), b_(b), exponential_(false) {}
68  cFunction(Real a, Real b) : a_(a), b_(b), exponential_(true) {}
69  Real operator()(Real k) const {
70  if (exponential_)
71  return std::exp(-a_ * k + b_);
72  if (s_ < QL_EPSILON)
73  return std::max(f_ - k, 0.0) + a_ * k + b_;
74  boost::math::normal normal;
75  Real d1 = std::log(f_ / k) / s_ + s_ / 2.0;
76  Real d2 = d1 - s_;
77  return f_ * boost::math::cdf(normal, d1) -
78  k * boost::math::cdf(normal, d2) + a_ * k + b_;
79  }
80  Real f_, s_, a_, b_;
81  const bool exponential_;
82  };

84  struct aHelper {
85  aHelper(Real k0, Real k1, Real c0, Real c1, Real c0p, Real c1p)
86  : k0_(k0), k1_(k1), c0_(c0), c1_(c1), c0p_(c0p), c1p_(c1p) {}
87  Real operator()(Real a) const {
88  boost::math::normal normal;
89  Real d20 = boost::math::quantile(normal, -c0p_ + a);
90  Real d21 = boost::math::quantile(normal, -c1p_ + a);
91  Real alpha = (d20 - d21) / (std::log(k0_) - std::log(k1_));
92  Real beta = d20 - alpha * std::log(k0_);
93  s_ = -1.0 / alpha;
94  f_ = std::exp(s_ * (beta + s_ / 2.0));
95  QL_REQUIRE(f_ < QL_KAHALE_FMAX, "dummy"); // this is caught
96  cFunction cTmp(f_, s_, a, 0.0);
97  b_ = c0_ - cTmp(k0_);
98  cFunction c(f_, s_, a, b_);
99  return c(k1_) - c1_;
100  }
101  Real k0_, k1_, c0_, c1_, c0p_, c1p_;
102  mutable Real s_, f_, b_;
103  };

105  struct sHelper {
106  sHelper(Real k0, Real c0, Real c0p) : k0_(k0), c0_(c0), c0p_(c0p) {}
107  Real operator()(Real s) const {
108  s = std::max(s, 0.0);
109  boost::math::normal normal;
110  Real d20 = boost::math::quantile(normal, -c0p_);
111  f_ = k0_ * std::exp(s * d20 + s * s / 2.0);
112  QL_REQUIRE(f_ < QL_KAHALE_FMAX, "dummy"); // this is caught
113  cFunction c(f_, s, 0.0, 0.0);
114  return c(k0_) - c0_;
115  }

117  mutable Real f_;
