28  NinePointLinearOp::NinePointLinearOp(
29  Size d0, Size d1,
30  const boost::shared_ptr<FdmMesher>& mesher)
31  : d0_(d0), d1_(d1),
32  i00_(new Size[mesher->layout()->size()]),
33  i10_(new Size[mesher->layout()->size()]),
34  i20_(new Size[mesher->layout()->size()]),
35  i01_(new Size[mesher->layout()->size()]),
36  i21_(new Size[mesher->layout()->size()]),
37  i02_(new Size[mesher->layout()->size()]),
38  i12_(new Size[mesher->layout()->size()]),
39  i22_(new Size[mesher->layout()->size()]),
40  a00_(new Real[mesher->layout()->size()]),
41  a10_(new Real[mesher->layout()->size()]),
42  a20_(new Real[mesher->layout()->size()]),
43  a01_(new Real[mesher->layout()->size()]),
44  a11_(new Real[mesher->layout()->size()]),
45  a21_(new Real[mesher->layout()->size()]),
46  a02_(new Real[mesher->layout()->size()]),
47  a12_(new Real[mesher->layout()->size()]),
48  a22_(new Real[mesher->layout()->size()]),
49  mesher_(mesher) {
50 
51  QL_REQUIRE( d0_ != d1_
52  && d0_ < mesher->layout()->dim().size()
53  && d1_ < mesher->layout()->dim().size(),
54  "inconsistent derivative directions");
55 
56  const boost::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
57  const FdmLinearOpIterator endIter = layout->end();
58 
59  for (FdmLinearOpIterator iter = layout->begin(); iter!=endIter; ++iter) {
60  const Size i = iter.index();
61 
62  i10_[i] = layout->neighbourhood(iter, d1_, -1);
63  i01_[i] = layout->neighbourhood(iter, d0_, -1);
64  i21_[i] = layout->neighbourhood(iter, d0_, 1);
65  i12_[i] = layout->neighbourhood(iter, d1_, 1);
66  i00_[i] = layout->neighbourhood(iter, d0_, -1, d1_, -1);
67  i20_[i] = layout->neighbourhood(iter, d0_, 1, d1_, -1);
68  i02_[i] = layout->neighbourhood(iter, d0_, -1, d1_, 1);
69  i22_[i] = layout->neighbourhood(iter, d0_, 1, d1_, 1);
70  }
71  }

131  Disposable<Array> NinePointLinearOp::apply(const Array& u)
132  const {
133 
134  const boost::shared_ptr<FdmLinearOpLayout> index=mesher_->layout();
135  QL_REQUIRE(u.size() == index->size(),"inconsistent length of r "
136  << u.size() << " vs " << index->size());
137 
138  Array retVal(u.size());
139  // direct access to make the following code faster.
140  const Real *a00(a00_.get()), *a01(a01_.get()), *a02(a02_.get());
141  const Real *a10(a10_.get()), *a11(a11_.get()), *a12(a12_.get());
142  const Real *a20(a20_.get()), *a21(a21_.get()), *a22(a22_.get());
143  const Size *i00(i00_.get()), *i01(i01_.get()), *i02(i02_.get());
144  const Size *i10(i10_.get()), *i12(i12_.get());
145  const Size *i20(i20_.get()), *i21(i21_.get()), *i22(i22_.get());
146 
147  //#pragma omp parallel for
148  for (Size i=0; i < retVal.size(); ++i) {
149  retVal[i] = a00[i]*u[i00[i]]
150  + a01[i]*u[i01[i]]
151  + a02[i]*u[i02[i]]
152  + a10[i]*u[i10[i]]
153  + a11[i]*u[i]
154  + a12[i]*u[i12[i]]
155  + a20[i]*u[i20[i]]
156  + a21[i]*u[i21[i]]
157  + a22[i]*u[i22[i]];
158  }
159  return retVal;
160  }
