47  void FdmLocalVolFwdOp::setTime(Time t1, Time t2) {
48  const Rate r = rTS_->forwardRate(t1, t2, Continuous).rate();
49  const Rate q = qTS_->forwardRate(t1, t2, Continuous).rate();
50 
51  const boost::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
52  const FdmLinearOpIterator endIter = layout->end();
53 
54  Array v(layout->size());
55  for (FdmLinearOpIterator iter = layout->begin();
56  iter != endIter; ++iter) {
57  const Size i = iter.index();
58 
59  v[i] = square<Real>()(
60  localVol_->localVol(0.5*(t1+t2), x_[i], true));
61  }
62  mapT_.axpyb(Array(1, 1.0), dxMap_.multR(- r + q + 0.5*v),
63  dxxMap_.multR(0.5*v), Array(1, 0.0));
64  }
