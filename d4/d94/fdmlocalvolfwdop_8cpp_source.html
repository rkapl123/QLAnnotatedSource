41  void FdmLocalVolFwdOp::setTime(Time t1, Time t2) {
42  const Rate r = rTS_->forwardRate(t1, t2, Continuous).rate();
43  const Rate q = qTS_->forwardRate(t1, t2, Continuous).rate();
44 
45  const ext::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
46  const FdmLinearOpIterator endIter = layout->end();
47 
48  Array v(layout->size());
49  for (FdmLinearOpIterator iter = layout->begin();
50  iter != endIter; ++iter) {
51  const Size i = iter.index();
52 
53  v[i] = square<Real>()(
54  localVol_->localVol(0.5*(t1+t2), x_[i], true));
55  }
56  mapT_.axpyb(Array(1, 1.0), dxMap_.multR(- r + q + 0.5*v),
57  dxxMap_.multR(0.5*v), Array(1, 0.0));
58  }
