84  inline BinomialDistribution::BinomialDistribution(Real p,

114  inline Real BinomialDistribution::operator()(BigNatural k) const {
115 
116  if (k > n_) return 0.0;
117 
118  // p==1.0
119  if (logP_==0.0)
120  return (k==n_ ? 1.0 : 0.0);
121  // p==0.0
122  else if (logOneMinusP_==0.0)
123  return (k==0 ? 1.0 : 0.0);
124  else
125  return std::exp(binomialCoefficientLn(n_, k) +
126  k * logP_ + (n_-k) * logOneMinusP_);
127  }

137  inline Real PeizerPrattMethod2Inversion(Real z, BigNatural n) {
138 
139  QL_REQUIRE(n%2==1,
140  "n must be an odd number: " << n << " not allowed");
141 
142  Real result = (z/(n+1.0/3.0+0.1/(n+1.0)));
143  result *= result;
144  result = std::exp(-result*(n+1.0/6.0));
145  result = 0.5 + (z>0 ? 1 : -1) * std::sqrt((0.25 * (1.0-result)));
146  return result;
147  }
