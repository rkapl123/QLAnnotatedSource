46 boost::shared_ptr<SmileSection>
47 Gaussian1dSwaptionVolatility::smileSectionImpl(Time optionTime,
48  Time swapLength) const {
49  DateHelper hlp(*this, optionTime);
50  NewtonSafe newton;
51  Date d(static_cast<Date::serial_type>(newton.solve(
52  hlp, 0.1,
53  365.25 * optionTime + static_cast<Real>(referenceDate().serialNumber()),
54  1.0)));
55  Period tenor(
56  static_cast<Integer>(Rounding(0).operator()(swapLength * 12.0)),
57  Months);
58  d = indexBase_->fixingCalendar().adjust(d);
59  return smileSectionImpl(d, tenor);
60 }
