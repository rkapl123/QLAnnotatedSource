44 Gaussian1dSwaptionVolatility::smileSectionImpl(Time optionTime,
45  Time swapLength) const {
46  DateHelper hlp(*this, optionTime);
47  NewtonSafe newton;
48  Date d(static_cast<Date::serial_type>(newton.solve(
49  hlp, 0.1,
50  365.25 * optionTime + static_cast<Real>(referenceDate().serialNumber()),
51  1.0)));
52  Period tenor(
53  static_cast<Integer>(Rounding(0)(swapLength * 12.0)),
54  Months);
55  d = indexBase_->fixingCalendar().adjust(d);
56  return smileSectionImpl(d, tenor);
