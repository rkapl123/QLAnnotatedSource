29  Date previousWednesday(const Date& date) {
30  Weekday w = date.weekday();
31  if (w >= 4) // roll back w-4 days
32  return date - (w - 4) * Days;
33  else // roll forward 4-w days and back one week
34  return date + (4 - w - 7) * Days;
35  }

66  bool BMAIndex::isValidFixingDate(const Date& date) const {
67  Calendar cal = fixingCalendar();
68  // either the fixing date is last Wednesday, or all days
69  // between last Wednesday included and the fixing date are
70  // holidays
71  for (Date d = previousWednesday(date); d<date; ++d) {
72  if (cal.isBusinessDay(d))
73  return false;
74  }
75  // also, the fixing date itself must be a business day
76  return cal.isBusinessDay(date);
77  }

83  Date BMAIndex::maturityDate(const Date& valueDate) const {
84  Calendar cal = fixingCalendar();
85  Date fixingDate = cal.advance(valueDate, -1, Days);
86  Date nextWednesday = previousWednesday(fixingDate+7);
87  return cal.advance(nextWednesday, 1, Days);
