49  FdmSolverDesc FdCIRVanillaEngine::getSolverDesc(Real) const {
50  DividendSchedule dividendSchedule = DividendSchedule();
51 
52  const ext::shared_ptr<StrikedTypePayoff> payoff =
53  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
54  const Time maturity = bsProcess_->time(arguments_.exercise->lastDate());
55 
56  // The short rate mesher
57  const ext::shared_ptr<Fdm1dMesher> shortRateMesher(
58  new FdmSimpleProcess1dMesher(rGrid_, cirProcess_, maturity, tGrid_));
59 
60  // The equity mesher
61  const ext::shared_ptr<Fdm1dMesher> equityMesher(
62  new FdmBlackScholesMesher(
63  xGrid_, bsProcess_, maturity, payoff->strike(),
64  Null<Real>(), Null<Real>(), 0.0001, 1.5,
65  std::pair<Real, Real>(payoff->strike(), 0.1),
66  dividendSchedule, quantoHelper_,
67  0.0));
68 
69  const ext::shared_ptr<FdmMesher> mesher(
70  new FdmMesherComposite(equityMesher, shortRateMesher));
71 
72  // Calculator
73  const ext::shared_ptr<FdmInnerValueCalculator> calculator(
74  new FdmLogInnerValue(arguments_.payoff, mesher, 0));
75 
76  // Step conditions
77  const ext::shared_ptr<FdmStepConditionComposite> conditions =
78  FdmStepConditionComposite::vanillaComposite(
79  arguments_.cashFlow, arguments_.exercise,
80  mesher, calculator,
81  bsProcess_->riskFreeRate()->referenceDate(),
82  bsProcess_->riskFreeRate()->dayCounter());
83 
84  // Boundary conditions
85  const FdmBoundaryConditionSet boundaries;
86 
87  // Solver
88  FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
89  calculator, maturity,
90  tGrid_, dampingSteps_ };
91 
92  return solverDesc;
93  }

95  void FdCIRVanillaEngine::calculate() const {
96  const ext::shared_ptr<StrikedTypePayoff> payoff =
97  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
98 
99  ext::shared_ptr<FdmCIRSolver> solver(new FdmCIRSolver(
100  Handle<CoxIngersollRossProcess>(cirProcess_),
101  Handle<GeneralizedBlackScholesProcess>(bsProcess_),
102  getSolverDesc(1.5), schemeDesc_,
103  rho_, payoff->strike()));
104 
105  const Real r0 = cirProcess_->x0();
106  const Real spot = bsProcess_->x0();
107 
108  results_.value = solver->valueAt(spot, r0);
109  results_.delta = solver->deltaAt(spot, r0);
110  results_.gamma = solver->gammaAt(spot, r0);
111  results_.theta = solver->thetaAt(spot, r0);
112  }
