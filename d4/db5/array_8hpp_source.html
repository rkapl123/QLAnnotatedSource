243  template <class I>
244  inline void _fill_array_(Array& a,
245  boost::scoped_array<Real>& data_,
246  Size& n_,
247  I begin, I end,
248  const boost::true_type&) {
249  // we got redirected here from a call like Array(3, 4)
250  // because it matched the constructor below exactly with
251  // ForwardIterator = int. What we wanted was fill an
252  // Array with a given value, which we do here.
253  Size n = begin;
254  Real value = end;
255  data_.reset(n ? new Real[n] : (Real*)(0));
256  n_ = n;
257  std::fill(a.begin(),a.end(),value);
258  }
