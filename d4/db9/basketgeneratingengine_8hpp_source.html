41  /* \warning the generated calibrating swaptions have a strike floored at
42  0.1bp (minus lognormal shift, if applicable), this is not true for atm
43  swaptions where the strike is generated in the swaption helper.
44 
45  \warning the standardSwapBase index should have associated forward and
46  discount curves. These curves are used for setup of the swaption helper.
47  This means that the market price of the calibration instrument is calculated
48  using these curves. Therefore the model price must be calculated using the
49  same curves, otherwise the calibration gets incosistent, i.e. the pricing
50  engine used for model calibration has to be capable of using the same curves
51  as associated to the index. Also the volatility structure passed to construct
52  the calibration helper should use curves that are consistent with the model
53  calibration curve setup. Finally the discountCurve given in the constructor
54  should be the same curve as the discounting curve of the swapIndex used to
55  determine the calibration basket. */

119  Real NPV(const ext::shared_ptr<VanillaSwap>& swap,
120  Real fixedRate,
121  Real nominal,
122  Real y,
123  int type) const {
124  Real npv = 0.0;
125  for (const auto& i : swap->fixedLeg()) {
126  ext::shared_ptr<FixedRateCoupon> c =
127  ext::dynamic_pointer_cast<FixedRateCoupon>(i);
128  npv -=
129  fixedRate * c->accrualPeriod() * nominal *
130  mdl_->zerobond(c->date(), expiry_, y,
131  indexBase_->discountingTermStructure());
132  }
133  for (const auto& i : swap->floatingLeg()) {
134  ext::shared_ptr<IborCoupon> c = ext::dynamic_pointer_cast<IborCoupon>(i);
135  npv +=
136  mdl_->forwardRate(c->fixingDate(), expiry_, y,
137  c->iborIndex()) *
138  c->accrualPeriod() * nominal *
139  mdl_->zerobond(c->date(), expiry_, y,
140  indexBase_->discountingTermStructure());
141  }
142  return (Real)type * npv;
143  }

154  Disposable<Array> values(const Array& v) const override {
155  // transformations
156  int type = type_; // start with same type as non standard
157  // underlying (1 means payer, -1 receiver)
158  Real nominal = std::fabs(v[0]);
159  if (v[0] < 0.0)
160  type *= -1;
161  Real maturity = std::min(std::fabs(v[1]), maxMaturity_);
162 
163  Real fixedRate = v[2]; // allow for negative rates explicitly
164  // (though it might not be reasonable for calibration depending
165  // on the model to calibrate and the market instrument quotation)
166  Size years = (Size)std::floor(maturity);
167  maturity -= (Real)years;
168  maturity *= 12.0;
169  Size months = (Size)std::floor(maturity);
170  Real alpha = 1.0 - (maturity - (Real)months);
171  if (years == 0 && months == 0) {
172  months = 1; // ensure a maturity of at least one month ...
173  alpha = 1.0; // ... but in this case only look at the lower
174  // maturity swap
175  }
176  // maturity -= (Real)months; maturity *= 365.25;
177  // Size days = (Size)std::floor(maturity);
178  // Real alpha = 1.0-(maturity-(Real)days);
179  // generate swap
180  Period lowerPeriod =
181  years * Years + months * Months; //+days*Days;
182  Period upperPeriod = lowerPeriod + 1 * Months; // 1*Days;
183  ext::shared_ptr<SwapIndex> tmpIndexLower, tmpIndexUpper;
184  tmpIndexLower = indexBase_->clone(lowerPeriod);
185  tmpIndexUpper = indexBase_->clone(upperPeriod);
186  ext::shared_ptr<VanillaSwap> swapLower =
187  tmpIndexLower->underlyingSwap(expiry_);
188  ext::shared_ptr<VanillaSwap> swapUpper =
189  tmpIndexUpper->underlyingSwap(expiry_);
190  // compute npv, delta, gamma
191  Real npvm =
192  alpha * NPV(swapLower, fixedRate, nominal, -h_, type) +
193  (1.0 - alpha) *
194  NPV(swapUpper, fixedRate, nominal, -h_, type);
195  Real npv =
196  alpha * NPV(swapLower, fixedRate, nominal, 0.0, type) +
197  (1.0 - alpha) *
198  NPV(swapUpper, fixedRate, nominal, 0.0, type);
199  Real npvu =
200  alpha * NPV(swapLower, fixedRate, nominal, h_, type) +
201  (1.0 - alpha) *
202  NPV(swapUpper, fixedRate, nominal, h_, type);
203  Real delta = (npvu - npvm) / (2.0 * h_);
204  Real gamma = (npvu - 2.0 * npv + npvm) / (h_ * h_);
205 
206  // debug output global standard underlying npv
207  // Real xtmp = -5.0;
208  // std::cout << "globalStandardNpv;";

210  // std::cout << alpha *NPV(swapLower, fixedRate, nominal, xtmp,

214  // << ";";
215  // xtmp += 0.1;
216  // }
217  // std::cout << std::endl;
218  // end debug output
219 
220  // return target function values
221  Array res(3);
222  res[0] = (npv - npv_) / delta_;
223  res[1] = (delta - delta_) / delta_;
224  res[2] = (gamma - gamma_) / gamma_;
225  return res;
226  }
