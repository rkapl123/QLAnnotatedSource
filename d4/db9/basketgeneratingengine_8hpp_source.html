40  /* \warning the generated calibrating swaptions have a strike floored at
41  0.1bp (minus lognormal shift, if applicable), this is not true for atm
42  swaptions where the strike is generated in the swaption helper.
43 
44  \warning the standardSwapBase index should have associated forward and
45  discount curves. These curves are used for setup of the swaption helper.
46  This means that the market price of the calibration instrument is calculated
47  using these curves. Therefore the model price must be calculated using the
48  same curves, otherwise the calibration gets incosistent, i.e. the pricing
49  engine used for model calibration has to be capable of using the same curves
50  as associated to the index. Also the volatility structure passed to construct
51  the calibration helper should use curves that are consistent with the model
52  calibration curve setup. Finally the discountCurve given in the constructor
53  should be the same curve as the discounting curve of the swapIndex used to
54  determine the calibration basket. */

114  Real NPV(boost::shared_ptr<VanillaSwap> swap, Real fixedRate,
115  Real nominal, Real y, int type) const {
116  Real npv = 0.0;
117  for (Size i = 0; i < swap->fixedLeg().size(); i++) {
118  boost::shared_ptr<FixedRateCoupon> c =
119  boost::dynamic_pointer_cast<FixedRateCoupon>(
120  swap->fixedLeg()[i]);
121  npv -=
122  fixedRate * c->accrualPeriod() * nominal *
123  mdl_->zerobond(c->date(), expiry_, y,
124  indexBase_->discountingTermStructure());
125  }
126  for (Size i = 0; i < swap->floatingLeg().size(); i++) {
127  boost::shared_ptr<IborCoupon> c =
128  boost::dynamic_pointer_cast<IborCoupon>(
129  swap->floatingLeg()[i]);
130  npv +=
131  mdl_->forwardRate(c->fixingDate(), expiry_, y,
132  c->iborIndex()) *
133  c->accrualPeriod() * nominal *
134  mdl_->zerobond(c->date(), expiry_, y,
135  indexBase_->discountingTermStructure());
136  }
137  return (Real)type * npv;
138  }

140  Real value(const Array &v) const {
141  Array vals = values(v);
142  Real res = 0.0;
143  for (Size i = 0; i < vals.size(); i++) {
144  res += vals[i] * vals[i];
145  }
146  return std::sqrt(res / vals.size());
147  }

149  Disposable<Array> values(const Array &v) const {
150  // transformations
151  int type = type_; // start with same type as non standard
152  // underlying (1 means payer, -1 receiver)
153  Real nominal = std::fabs(v[0]);
154  if (v[0] < 0.0)
155  type *= -1;
156  Real maturity = std::min(std::fabs(v[1]), maxMaturity_);
157 
158  Real fixedRate = v[2]; // allow for negative rates explicitly
159  // (though it might not be reasonable for calibration depending
160  // on the model to calibrate and the market instrument quotation)
161  Size years = (Size)std::floor(maturity);
162  maturity -= (Real)years;
163  maturity *= 12.0;
164  Size months = (Size)std::floor(maturity);
165  Real alpha = 1.0 - (maturity - (Real)months);
166  if (years == 0 && months == 0) {
167  months = 1; // ensure a maturity of at least one month ...
168  alpha = 1.0; // ... but in this case only look at the lower
169  // maturity swap
170  }
171  // maturity -= (Real)months; maturity *= 365.25;
172  // Size days = (Size)std::floor(maturity);
173  // Real alpha = 1.0-(maturity-(Real)days);
174  // generate swap
175  Period lowerPeriod =
176  years * Years + months * Months; //+days*Days;
177  Period upperPeriod = lowerPeriod + 1 * Months; // 1*Days;
178  boost::shared_ptr<SwapIndex> tmpIndexLower, tmpIndexUpper;
179  tmpIndexLower = indexBase_->clone(lowerPeriod);
180  tmpIndexUpper = indexBase_->clone(upperPeriod);
181  boost::shared_ptr<VanillaSwap> swapLower =
182  tmpIndexLower->underlyingSwap(expiry_);
183  boost::shared_ptr<VanillaSwap> swapUpper =
184  tmpIndexUpper->underlyingSwap(expiry_);
185  // compute npv, delta, gamma
186  Real npvm =
187  alpha * NPV(swapLower, fixedRate, nominal, -h_, type) +
188  (1.0 - alpha) *
189  NPV(swapUpper, fixedRate, nominal, -h_, type);
190  Real npv =
191  alpha * NPV(swapLower, fixedRate, nominal, 0.0, type) +
192  (1.0 - alpha) *
193  NPV(swapUpper, fixedRate, nominal, 0.0, type);
194  Real npvu =
195  alpha * NPV(swapLower, fixedRate, nominal, h_, type) +
196  (1.0 - alpha) *
197  NPV(swapUpper, fixedRate, nominal, h_, type);
198  Real delta = (npvu - npvm) / (2.0 * h_);
199  Real gamma = (npvu - 2.0 * npv + npvm) / (h_ * h_);
200 
201  // debug output global standard underlying npv
202  // Real xtmp = -5.0;

216  Array res(3);
217  res[0] = (npv - npv_) / delta_;
218  res[1] = (delta - delta_) / delta_;
219  res[2] = (gamma - gamma_) / gamma_;
220  return res;
221  }
