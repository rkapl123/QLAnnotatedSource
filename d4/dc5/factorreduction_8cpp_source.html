26  Disposable<std::vector<Real> >
27  factorReduction(Matrix mtrx,
28  Size maxIters) {
29  static Real tolerance = 1.e-6;
30 
31  QL_REQUIRE(mtrx.rows() == mtrx.columns(),
32  "Input matrix is not square");
33 
34  const Size n = mtrx.columns();
35 
36  #if defined(QL_EXTRA_SAFETY_CHECKS)
37  // check symmetry
38  for (Size iRow=0; iRow<mtrx.rows(); iRow++)
39  for (Size iCol=0; iCol<iRow; iCol++)
40  QL_REQUIRE(mtrx[iRow][iCol] == mtrx[iCol][iRow],
41  "input matrix is not symmetric");
42  QL_REQUIRE(*std::max_element(mtrx.begin(), mtrx.end()) <=1 &&
43  *std::min_element(mtrx.begin(), mtrx.end()) >=-1,
44  "input matrix data is not correlation data");
45  #endif
46 
47  // Initial guess value
48  std::vector<Real> previousCorrels(n, 0.);
49  for(Size iCol=0; iCol<n; iCol++) {
50  for(Size iRow=0; iRow<n; iRow++)
51  previousCorrels[iCol] +=
52  mtrx[iRow][iCol] * mtrx[iRow][iCol];
53  previousCorrels[iCol] =
54  std::sqrt((previousCorrels[iCol]-1.)/(n-1.));
55  }
56 
57  // iterative solution
58  Size iteration = 0;
59  Real distance;
60  do {
61  // patch Matrix diagonal
62  for(Size iCol=0; iCol<n; iCol++)
63  mtrx[iCol][iCol] =
64  previousCorrels[iCol];
65  // compute eigenvector decomposition
66  SymmetricSchurDecomposition ssDec(mtrx);
67  //const Matrix& eigenVect = ssDec.eigenvectors();
68  const Array& eigenVals = ssDec.eigenvalues();
69  Array::const_iterator itMaxEval =
70  std::max_element(eigenVals.begin(), eigenVals.end());
71  // We do not need the max value, only the position of the
72  // corresponding eigenvector
73  Size iMax = std::distance(eigenVals.begin(), itMaxEval);
74  std::vector<Real> newCorrels, distances;
75  for(Size iCol=0; iCol<n; iCol++) {
76  Real thisCorrel = mtrx[iMax][iCol];
77  newCorrels.push_back(thisCorrel);
78  // strictly is:
79  // abs(\sqrt{\rho}- \sqrt{\rho_{old}})/\sqrt{\rho_{old}}
80  distances.push_back(
81  std::abs(thisCorrel - previousCorrels[iCol])/
82  previousCorrels[iCol]);
83  }
84  previousCorrels = newCorrels;
85  distance = *std::max_element(distances.begin(), distances.end());
86  }while(distance > tolerance && ++iteration <= maxIters );
87 
88  // test it did not go up to the max iteration and the matrix can
89  // be reduced to one factor.
90  QL_REQUIRE(iteration < maxIters,
91  "convergence not reached after " <<
92  iteration << " iterations");
93 
94  #if defined(QL_EXTRA_SAFETY_CHECKS)
95  QL_REQUIRE(*std::max_element(previousCorrels.begin(),
96  previousCorrels.end()) <=1 &&
97  *std::min_element(previousCorrels.begin(),
98  previousCorrels.end()) >=-1,
99  "matrix can not be decomposed to a single factor dependence");
100  #endif
101 
102  return previousCorrels;
103  }
