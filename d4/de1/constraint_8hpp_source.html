34  class Constraint {
35  protected:
37  class Impl {
38  public:
39  virtual ~Impl() = default;
41  virtual bool test(const Array& params) const = 0;
43  virtual Array upperBound(const Array& params) const {
44  return Array(params.size(),
45  std::numeric_limits < Array::value_type > ::max());
46  }
48  virtual Array lowerBound(const Array& params) const {
49  return Array(params.size(),
50  -std::numeric_limits < Array::value_type > ::max());
51  }
52  };
53  ext::shared_ptr<Impl> impl_;
54  public:
55  bool empty() const { return !impl_; }
56  bool test(const Array& p) const { return impl_->test(p); }
57  Array upperBound(const Array& params) const {
58  Array result = impl_->upperBound(params);
59  QL_REQUIRE(params.size() == result.size(),
60  "upper bound size (" << result.size()
61  << ") not equal to params size ("
62  << params.size() << ")");
63  return result;
64  }
65  Array lowerBound(const Array& params) const {
66  Array result = impl_->lowerBound(params);
67  QL_REQUIRE(params.size() == result.size(),
68  "lower bound size (" << result.size()
69  << ") not equal to params size ("
70  << params.size() << ")");
71  return result;
72  }
73  Real update(Array& p, const Array& direction, Real beta) const;
74  Constraint(ext::shared_ptr<Impl> impl = ext::shared_ptr<Impl>());
75  };

147  class CompositeConstraint : public Constraint {
148  private:
149  class Impl : public Constraint::Impl {
150  public:
151  Impl(Constraint c1, Constraint c2) : c1_(std::move(c1)), c2_(std::move(c2)) {}
152  bool test(const Array& params) const override {
153  return c1_.test(params) && c2_.test(params);
154  }
155  Array upperBound(const Array& params) const override {
156  Array c1ub = c1_.upperBound(params);
157  Array c2ub = c2_.upperBound(params);
158  Array rtrnArray(c1ub.size(), 0.0);
159  for (Size iter = 0; iter < c1ub.size(); iter++) {
160  rtrnArray.at(iter) = std::min(c1ub.at(iter), c2ub.at(iter));
161  }
162  return rtrnArray;
163  }
164  Array lowerBound(const Array& params) const override {
165  Array c1lb = c1_.lowerBound(params);
166  Array c2lb = c2_.lowerBound(params);
167  Array rtrnArray(c1lb.size(), 0.0);
168  for (Size iter = 0; iter < c1lb.size(); iter++) {
169  rtrnArray.at(iter) = std::max(c1lb.at(iter), c2lb.at(iter));
170  }
171  return rtrnArray;
172  }
173 
174  private:
175  Constraint c1_, c2_;
176  };
177  public:
178  CompositeConstraint(const Constraint& c1, const Constraint& c2)
179  : Constraint(ext::shared_ptr<Constraint::Impl>(
180  new CompositeConstraint::Impl(c1,c2))) {}
181  };
