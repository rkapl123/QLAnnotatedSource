88 
89  if (isWeekend(w)

92  // Sao Paulo City Day
93  || (d == 25 && m == January)
94  // Tiradentes Day
95  || (d == 21 && m == April)
96  // Labor Day
97  || (d == 1 && m == May)
98  // Revolution Day
99  || (d == 9 && m == July)
100  // Independence Day
101  || (d == 7 && m == September)
102  // Nossa Sra. Aparecida Day
103  || (d == 12 && m == October)
104  // All Souls Day
105  || (d == 2 && m == November)
106  // Republic Day
107  || (d == 15 && m == November)
108  // Black Consciousness Day
109  || (d == 20 && m == November && y >= 2007)
110  // Christmas Eve
111  || (d == 24 && m == December)
112  // Christmas
113  || (d == 25 && m == December)
114  // Passion of Christ
115  || (dd == em-3)
116  // Carnival
117  || (dd == em-49 || dd == em-48)
118  // Corpus Christi
119  || (dd == em+59)
120  // last business day of the year
121  || (m == December && (d == 31 || (d >= 29 && w == Friday)))
122  )
123  return false; // NOLINT(readability-simplify-boolean-expr)
124  return true;
