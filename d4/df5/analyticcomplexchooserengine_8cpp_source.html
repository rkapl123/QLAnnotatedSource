41  Real AnalyticComplexChooserEngine::ComplexChooser() const{
42  Real S = process_->x0();
43  Real b;
44  Real v;
45  Real Xc = arguments_.strikeCall;
46  Real Xp = arguments_.strikePut;
47  Time T = choosingTime();
48  Time Tc = callMaturity()-choosingTime();
49  Time Tp = putMaturity()-choosingTime();
50 
51  Real i = CriticalValueChooser();
52 
53  b = riskFreeRate(choosingTime()) - dividendYield(choosingTime());
54  v = volatility(T);
55  Real d1 = (log(S / i) + (b + pow(v, 2) / 2)*T) / (v*sqrt(T));
56  Real d2 = d1 - v*sqrt(T);
57 
58  b = riskFreeRate(callMaturity()) - dividendYield(callMaturity());
59  v = volatility(Tc);
60  Real y1 = (log(S / Xc) + (b + pow(v, 2) / 2)*Tc) / (v*sqrt(Tc));
61 
62  b = riskFreeRate(putMaturity()) - dividendYield(putMaturity());
63  v = volatility(Tp);
64  Real y2 = (log(S / Xp) + (b + pow(v, 2) / 2)*Tp) / (v*sqrt(Tp));
65 
66  Real rho1 = sqrt(T / Tc);
67  Real rho2 = sqrt(T / Tp);
68  b = riskFreeRate(callMaturity()) - dividendYield(callMaturity());
69  Real r = riskFreeRate(callMaturity());
70  Real ComplexChooser = S * exp((b - r)*Tc) * BivariateCumulativeNormalDistributionDr78(rho1)(d1, y1)
71  - Xc * exp(-r*Tc)*BivariateCumulativeNormalDistributionDr78(rho1)(d2, y1 - v * sqrt(Tc)) ;
72  b = riskFreeRate(putMaturity()) - dividendYield(putMaturity());
73  r = riskFreeRate(putMaturity());
74  ComplexChooser-= S * exp((b - r)*Tp) * BivariateCumulativeNormalDistributionDr78(rho2)(-d1, -y2);
75  ComplexChooser+= Xp * exp(-r*Tp) * BivariateCumulativeNormalDistributionDr78(rho2)(-d2, -y2 + v * sqrt(Tp));
76  return ComplexChooser;
77  }

79  BlackScholesCalculator AnalyticComplexChooserEngine::bsCalculator(
80  Real spot, Option::Type optionType) const {
81  Real vol;
82  DiscountFactor growth;
83  DiscountFactor discount;
84 
85  //payoff
86  ext::shared_ptr<PlainVanillaPayoff > vanillaPayoff;
87  if (optionType == Option::Call){
88  //TC-T
89  Time t=callMaturity()-choosingTime()-choosingTime();
90  vanillaPayoff = ext::make_shared<PlainVanillaPayoff>(
91  Option::Call, strike(Option::Call));
92  //QuantLib requires sigma * sqrt(T) rather than just sigma/volatility
93  vol = volatility(t) * std::sqrt(t);
94  growth = dividendDiscount(t);
95  discount = riskFreeDiscount(t);
96  } else{
97  Time t=putMaturity()-choosingTime()-choosingTime();
98  vanillaPayoff = ext::make_shared<PlainVanillaPayoff>(
99  Option::Put, strike(Option::Put));
100  vol = volatility(t) * std::sqrt(t);
101  growth = dividendDiscount(t);
102  discount = riskFreeDiscount(t);
103  }
104 
105  BlackScholesCalculator bs(vanillaPayoff, spot, growth, vol, discount);
106  return bs;
107  }

109  Real AnalyticComplexChooserEngine::CriticalValueChooser() const{
110  Real Sv = process_->x0();
111 
112  BlackScholesCalculator bs=bsCalculator(Sv,Option::Call);
113  Real ci = bs.value();
114  Real dc = bs.delta();
115 
116  bs=bsCalculator(Sv,Option::Put);
117  Real Pi = bs.value();
118  Real dp = bs.delta();
119 
120  Real yi = ci - Pi;
121  Real di = dc - dp;
122  Real epsilon = 0.001;
123 
124  //Newton-Raphson process
125  while (std::fabs(yi) > epsilon){
126  Sv = Sv - yi / di;
127 
128  bs=bsCalculator(Sv,Option::Call);
129  ci = bs.value();
130  dc = bs.delta();
131 
132  bs=bsCalculator(Sv,Option::Put);
133  Pi = bs.value();
134  dp = bs.delta();
135 
136  yi = ci - Pi;
137  di = dc - dp;
138  }
139  return Sv;
140  }
