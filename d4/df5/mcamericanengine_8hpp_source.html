179 template <class RNG, class S, class RNG_Calibration>
180 inline void MCAmericanEngine<RNG, S, RNG_Calibration>::calculate() const {
181 MCLongstaffSchwartzEngine<VanillaOption::engine, SingleVariate, RNG, S,
182 RNG_Calibration>::calculate();
183 if (this->controlVariate_) {
184 // control variate might lead to small negative
185 // option values for deep OTM options
186 this->results_.value = std::max(0.0, this->results_.value);
187 }

190 template <class RNG, class S, class RNG_Calibration>
191 inline ext::shared_ptr<LongstaffSchwartzPathPricer<Path> >
192 MCAmericanEngine<RNG, S, RNG_Calibration>::lsmPathPricer() const {
193 ext::shared_ptr<GeneralizedBlackScholesProcess> process =
194 ext::dynamic_pointer_cast<GeneralizedBlackScholesProcess>(
195 this->process_);
196 QL_REQUIRE(process, "generalized Black-Scholes process required");
197
198 ext::shared_ptr<EarlyExercise> exercise =
199 ext::dynamic_pointer_cast<EarlyExercise>(
200 this->arguments_.exercise);
201 QL_REQUIRE(exercise, "wrong exercise given");
202 QL_REQUIRE(!exercise->payoffAtExpiry(),
203 "payoff at expiry not handled");
204
205 ext::shared_ptr<AmericanPathPricer> earlyExercisePathPricer(
206 new AmericanPathPricer(this->arguments_.payoff,
207 polynomialOrder_, polynomialType_));
208
209 return ext::make_shared<LongstaffSchwartzPathPricer<Path> > (
210
211 this->timeGrid(),
212 earlyExercisePathPricer,
213 *(process->riskFreeRate()));
214 }

216 template <class RNG, class S, class RNG_Calibration>
217 inline ext::shared_ptr<PathPricer<Path> >
218 MCAmericanEngine<RNG, S, RNG_Calibration>::controlPathPricer() const {
219 ext::shared_ptr<StrikedTypePayoff> payoff =
220 ext::dynamic_pointer_cast<StrikedTypePayoff>(
221 this->arguments_.payoff);
222 QL_REQUIRE(payoff, "StrikedTypePayoff needed for control variate");
223
224 ext::shared_ptr<GeneralizedBlackScholesProcess> process =
225 ext::dynamic_pointer_cast<GeneralizedBlackScholesProcess>(
226 this->process_);
227 QL_REQUIRE(process, "generalized Black-Scholes process required");
228
229 return ext::shared_ptr<PathPricer<Path> >(
230 new EuropeanPathPricer(
231 payoff->optionType(),
232 payoff->strike(),
233 process->riskFreeRate()->discount(this->timeGrid().back()))
234 );
235 }

249 template <class RNG, class S, class RNG_Calibration>
250 inline Real
251 MCAmericanEngine<RNG, S, RNG_Calibration>::controlVariateValue() const {
252 ext::shared_ptr<PricingEngine> controlPE =
253 this->controlPricingEngine();
254
255 QL_REQUIRE(controlPE,
256 "engine does not provide "
257 "control variation pricing engine");
258
259 auto* controlArguments = dynamic_cast<VanillaOption::arguments*>(controlPE->getArguments());
260 *controlArguments = this->arguments_;
261 controlArguments->exercise = ext::shared_ptr<Exercise>(
262 new EuropeanExercise(this->arguments_.exercise->lastDate()));
263
264 controlPE->calculate();
265
266 const auto* controlResults =
267 dynamic_cast<const VanillaOption::results*>(controlPE->getResults());
268
269 return controlResults->value;
270 }
