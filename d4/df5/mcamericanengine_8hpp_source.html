183  template <class RNG, class S, class RNG_Calibration>
184  inline ext::shared_ptr<LongstaffSchwartzPathPricer<Path> >
185  MCAmericanEngine<RNG, S, RNG_Calibration>::lsmPathPricer() const {
186  ext::shared_ptr<GeneralizedBlackScholesProcess> process =
187  ext::dynamic_pointer_cast<GeneralizedBlackScholesProcess>(
188  this->process_);
189  QL_REQUIRE(process, "generalized Black-Scholes process required");
190 
191  ext::shared_ptr<EarlyExercise> exercise =
192  ext::dynamic_pointer_cast<EarlyExercise>(
193  this->arguments_.exercise);
194  QL_REQUIRE(exercise, "wrong exercise given");
195  QL_REQUIRE(!exercise->payoffAtExpiry(),
196  "payoff at expiry not handled");
197 
198  ext::shared_ptr<AmericanPathPricer> earlyExercisePathPricer(
199  new AmericanPathPricer(this->arguments_.payoff,
200  polynomOrder_, polynomType_));
201 
202  return ext::make_shared<LongstaffSchwartzPathPricer<Path> > (
203 
204  this->timeGrid(),
205  earlyExercisePathPricer,
206  *(process->riskFreeRate()));
207  }

242  template <class RNG, class S, class RNG_Calibration>
243  inline Real
244  MCAmericanEngine<RNG, S, RNG_Calibration>::controlVariateValue() const {
245  ext::shared_ptr<PricingEngine> controlPE =
246  this->controlPricingEngine();
247 
248  QL_REQUIRE(controlPE,
249  "engine does not provide "
250  "control variation pricing engine");
251 
252  auto* controlArguments = dynamic_cast<VanillaOption::arguments*>(controlPE->getArguments());
253  *controlArguments = this->arguments_;
254  controlArguments->exercise = ext::shared_ptr<Exercise>(
255  new EuropeanExercise(this->arguments_.exercise->lastDate()));
256 
257  controlPE->calculate();
258 
259  const auto* controlResults =
260  dynamic_cast<const VanillaOption::results*>(controlPE->getResults());
261 
262  return controlResults->value;
263  }
