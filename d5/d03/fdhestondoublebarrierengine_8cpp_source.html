48  void FdHestonDoubleBarrierEngine::calculate() const {
49 
50  QL_REQUIRE(arguments_.barrierType == DoubleBarrier::KnockOut,
51  "only Knock-Out double barrier options are supported");
52 
53  // 1. Mesher
54  const boost::shared_ptr<HestonProcess>& process = model_->process();
55  const Time maturity = process->time(arguments_.exercise->lastDate());
56 
57  // 1.1 The variance mesher
58  const Size tGridMin = 5;
59  const boost::shared_ptr<FdmHestonVarianceMesher> varianceMesher(
60  new FdmHestonVarianceMesher(vGrid_, process, maturity,
61  std::max(tGridMin, tGrid_/50)));
62 
63  // 1.2 The equity mesher
64  const boost::shared_ptr<StrikedTypePayoff> payoff =
65  boost::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
66 
67  Real xMin = std::log(arguments_.barrier_lo);
68  Real xMax = std::log(arguments_.barrier_hi);
69 
70  const boost::shared_ptr<Fdm1dMesher> equityMesher(
71  new FdmBlackScholesMesher(
72  xGrid_,
73  FdmBlackScholesMesher::processHelper(
74  process->s0(), process->dividendYield(),
75  process->riskFreeRate(), varianceMesher->volaEstimate()),
76  maturity, payoff->strike(), xMin, xMax));
77 
78  const boost::shared_ptr<FdmMesher> mesher (
79  new FdmMesherComposite(equityMesher, varianceMesher));
80 
81  // 2. Calculator
82  const boost::shared_ptr<FdmInnerValueCalculator> calculator(
83  new FdmLogInnerValue(payoff, mesher, 0));
84 
85  // 3. Step conditions
86  std::list<boost::shared_ptr<StepCondition<Array> > > stepConditions;
87  std::list<std::vector<Time> > stoppingTimes;
88 
89  QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
90  "only european style option are supported");
91 
92  boost::shared_ptr<FdmStepConditionComposite> conditions(
93  new FdmStepConditionComposite(stoppingTimes, stepConditions));
94 
95  // 4. Boundary conditions
96  FdmBoundaryConditionSet boundaries;
97  boundaries.push_back(FdmBoundaryConditionSet::value_type(
98  new FdmDirichletBoundary(mesher, arguments_.rebate, 0,
99  FdmDirichletBoundary::Lower)));
100 
101  boundaries.push_back(FdmBoundaryConditionSet::value_type(
102  new FdmDirichletBoundary(mesher, arguments_.rebate, 0,
103  FdmDirichletBoundary::Upper)));
104 
105  // 5. Solver
106  FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
107  calculator, maturity,
108  tGrid_, dampingSteps_ };
109 
110  boost::shared_ptr<FdmHestonSolver> solver(new FdmHestonSolver(
111  Handle<HestonProcess>(process), solverDesc, schemeDesc_,
112  Handle<FdmQuantoHelper>(), leverageFct_));
113 
114  const Real spot = process->s0()->value();
115  results_.value = solver->valueAt(spot, process->v0());
116  results_.delta = solver->deltaAt(spot, process->v0());
117  results_.gamma = solver->gammaAt(spot, process->v0());
118  results_.theta = solver->thetaAt(spot, process->v0());
119  }
120 }
