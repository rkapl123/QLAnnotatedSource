47 MakeCreditDefaultSwap::operator ext::shared_ptr<CreditDefaultSwap>() const {
48
49 Date tradeDate = (tradeDate_ != Date()) ? tradeDate_ : Settings::instance().evaluationDate();
50 Date upfrontDate = WeekendsOnly().advance(tradeDate, cashSettlementDays_, Days);
51
52 Date protectionStart;
53 if (rule_ == DateGeneration::CDS2015 || rule_ == DateGeneration::CDS) {
54 protectionStart = tradeDate;
55 } else {
56 protectionStart = tradeDate + 1;
57 }
58
59 Date end;
60 if (tenor_) { // NOLINT(readability-implicit-bool-conversion)
61 if (rule_ == DateGeneration::CDS2015 || rule_ == DateGeneration::CDS || rule_ == DateGeneration::OldCDS) {
62 end = cdsMaturity(tradeDate, *tenor_, rule_);
63 } else {
64 end = tradeDate + *tenor_;
65 }
66 } else {
67 // we have two exclusive constructors; if we don't have a tenor, we have a term date
68 end = *termDate_; // NOLINT(bugprone-unchecked-optional-access)
69 }
70
71 Schedule schedule(protectionStart, end, couponTenor_, WeekendsOnly(), Following,
72 Unadjusted, rule_, false);
73
74 ext::shared_ptr<CreditDefaultSwap> cds =
75 ext::make_shared<CreditDefaultSwap>(
76 side_, nominal_, upfrontRate_, couponRate_, schedule, Following,
77 dayCounter_, true, true, protectionStart, upfrontDate,
78 ext::shared_ptr<Claim>(), lastPeriodDayCounter_, true, tradeDate, cashSettlementDays_);
79
80 cds->setPricingEngine(engine_);
81 return cds;
82
83 }
