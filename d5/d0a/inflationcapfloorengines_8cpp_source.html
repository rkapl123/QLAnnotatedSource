50  void YoYInflationCapFloorEngine::calculate() const {
51 
52  // copy black version then adapt to others
53 
54  Real value = 0.0;
55  Size optionlets = arguments_.startDates.size();
56  std::vector<Real> values(optionlets, 0.0);
57  std::vector<Real> stdDevs(optionlets, 0.0);
58  std::vector<Real> forwards(optionlets, 0.0);
59  YoYInflationCapFloor::Type type = arguments_.type;
60 
61  Handle<YoYInflationTermStructure> yoyTS
62  = index()->yoyInflationTermStructure();
63  Handle<YieldTermStructure> nominalTS =
64  !nominalTermStructure_.empty() ?
65  nominalTermStructure_ :
66  yoyTS->nominalTermStructure();
67  Date settlement = nominalTS->referenceDate();
68 
69  for (Size i=0; i<optionlets; ++i) {
70  Date paymentDate = arguments_.payDates[i];
71  if (paymentDate > settlement) { // discard expired caplets
72  DiscountFactor d = arguments_.nominals[i] *
73  arguments_.gearings[i] *
74  nominalTS->discount(paymentDate) *
75  arguments_.accrualTimes[i];
76 
77  // We explicitly have the index and assume that
78  // the fixing is natural, i.e. no convexity adjustment.
79  // If that was required then we would also need
80  // nominal vols in the pricing engine, i.e. a different engine.
81  // This also means that we do not need the coupon to have
82  // a pricing engine to return the swaplet rate and then
83  // the adjusted fixing in the instrument.
84  forwards[i] = yoyTS->yoyRate(arguments_.fixingDates[i],Period(0,Days));
85  Rate forward = forwards[i];
86 
87  Date fixingDate = arguments_.fixingDates[i];
88  Time sqrtTime = 0.0;
89  if (fixingDate > volatility_->baseDate()){
90  sqrtTime = std::sqrt(
91  volatility_->timeFromBase(fixingDate));
92  }
93 
94  if (type == YoYInflationCapFloor::Cap || type == YoYInflationCapFloor::Collar) {
95  Rate strike = arguments_.capRates[i];
96  if (sqrtTime>0.0) {
97  stdDevs[i] = std::sqrt(
98  volatility_->totalVariance(fixingDate, strike, Period(0,Days)));
99 
100  }
101 
102  // sttDev=0 for already-fixed dates so everything on forward
103  values[i] = optionletImpl(Option::Call, strike,
104  forward, stdDevs[i], d);
105  }
106  if (type == YoYInflationCapFloor::Floor || type == YoYInflationCapFloor::Collar) {
107  Rate strike = arguments_.floorRates[i];
108  if (sqrtTime>0.0) {
109  stdDevs[i] = std::sqrt(
110  volatility_->totalVariance(fixingDate, strike, Period(0,Days)));
111  }
112  Real floorlet = optionletImpl(Option::Put, strike,
113  forward, stdDevs[i], d);
114  if (type == YoYInflationCapFloor::Floor) {
115  values[i] = floorlet;
116  } else {
117  // a collar is long a cap and short a floor
118  values[i] -= floorlet;
119  }
120 
121  }
122  value += values[i];
123  }
124  }
125  results_.value = value;
126 
127  results_.additionalResults["optionletsPrice"] = values;
128  results_.additionalResults["optionletsAtmForward"] = forwards;
129  if (type != YoYInflationCapFloor::Collar)
130  results_.additionalResults["optionletsStdDev"] = stdDevs;
131  }
