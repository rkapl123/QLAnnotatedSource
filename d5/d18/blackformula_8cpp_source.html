65  Real blackFormula(Option::Type optionType,
66  Real strike,
67  Real forward,
68  Real stdDev,
69  Real discount,
70  Real displacement)
71  {
72  checkParameters(strike, forward, displacement);
73  QL_REQUIRE(stdDev>=0.0,
74  "stdDev (" << stdDev << ") must be non-negative");
75  QL_REQUIRE(discount>0.0,
76  "discount (" << discount << ") must be positive");
77 
78  if (stdDev==0.0)
79  return std::max((forward-strike)*optionType, Real(0.0))*discount;
80 
81  forward = forward + displacement;
82  strike = strike + displacement;
83 
84  // since displacement is non-negative strike==0 iff displacement==0
85  // so returning forward*discount is OK
86  if (strike==0.0)
87  return (optionType==Option::Call ? forward*discount : 0.0);
88 
89  Real d1 = std::log(forward/strike)/stdDev + 0.5*stdDev;
90  Real d2 = d1 - stdDev;
91  CumulativeNormalDistribution phi;
92  Real nd1 = phi(optionType*d1);
93  Real nd2 = phi(optionType*d2);
94  Real result = discount * optionType * (forward*nd1 - strike*nd2);
95  QL_ENSURE(result>=0.0,
96  "negative value (" << result << ") for " <<
97  stdDev << " stdDev, " <<
98  optionType << " option, " <<
99  strike << " strike , " <<
100  forward << " forward");
101  return result;
102  }

113  Real blackFormulaForwardDerivative(Option::Type optionType,
114  Real strike,
115  Real forward,
116  Real stdDev,
117  Real discount,
118  Real displacement)
119  {
120  checkParameters(strike, forward, displacement);
121  QL_REQUIRE(stdDev>=0.0,
122  "stdDev (" << stdDev << ") must be non-negative");
123  QL_REQUIRE(discount>0.0,
124  "discount (" << discount << ") must be positive");
125 
126  if (stdDev==0.0)
127  return optionType * std::max(1.0 * boost::math::sign((forward - strike) * optionType), 0.0) * discount;
128 
129  forward = forward + displacement;
130  strike = strike + displacement;
131 
132  if (strike==0.0)
133  return (optionType==Option::Call ? discount : 0.0);
134 
135  Real d1 = std::log(forward/strike)/stdDev + 0.5*stdDev;
136  CumulativeNormalDistribution phi;
137  return optionType * phi(optionType * d1) * discount;
138  }

150  Real blackFormulaImpliedStdDevApproximation(Option::Type optionType,
151  Real strike,
152  Real forward,
153  Real blackPrice,
154  Real discount,
155  Real displacement)
156  {
157  checkParameters(strike, forward, displacement);
158  QL_REQUIRE(blackPrice>=0.0,
159  "blackPrice (" << blackPrice << ") must be non-negative");
160  QL_REQUIRE(discount>0.0,
161  "discount (" << discount << ") must be positive");
162 
163  Real stdDev;
164  forward = forward + displacement;
165  strike = strike + displacement;
166  if (strike==forward)
167  // Brenner-Subrahmanyan (1988) and Feinstein (1988) ATM approx.
168  stdDev = blackPrice/discount*std::sqrt(2.0 * M_PI)/forward;
169  else {
170  // Corrado and Miller extended moneyness approximation
171  Real moneynessDelta = optionType*(forward-strike);
172  Real moneynessDelta_2 = moneynessDelta/2.0;
173  Real temp = blackPrice/discount - moneynessDelta_2;
174  Real moneynessDelta_PI = moneynessDelta*moneynessDelta/M_PI;
175  Real temp2 = temp*temp-moneynessDelta_PI;
176  if (temp2<0.0) // approximation breaks down, 2 alternatives:
177  // 1. zero it
178  temp2=0.0;
179  // 2. Manaster-Koehler (1982) efficient Newton-Raphson seed
180  //return std::fabs(std::log(forward/strike))*std::sqrt(2.0);
181  temp2 = std::sqrt(temp2);
182  temp += temp2;
183  temp *= std::sqrt(2.0 * M_PI);
184  stdDev = temp/(forward+strike);
185  }
186  QL_ENSURE(stdDev>=0.0,
187  "stdDev (" << stdDev << ") must be non-negative");
188  return stdDev;
189  }

201  Real blackFormulaImpliedStdDevChambers(Option::Type optionType,
202  Real strike,
203  Real forward,
204  Real blackPrice,
205  Real blackAtmPrice,
206  Real discount,
207  Real displacement) {
208  checkParameters(strike, forward, displacement);
209  QL_REQUIRE(blackPrice >= 0.0,
210  "blackPrice (" << blackPrice << ") must be non-negative");
211  QL_REQUIRE(blackAtmPrice >= 0.0, "blackAtmPrice ("
212  << blackAtmPrice
213  << ") must be non-negative");
214  QL_REQUIRE(discount > 0.0, "discount (" << discount
215  << ") must be positive");
216 
217  Real stdDev;
218 
219  forward = forward + displacement;
220  strike = strike + displacement;
221  blackPrice /= discount;
222  blackAtmPrice /= discount;
223 
224  Real s0 = M_SQRT2 * M_SQRTPI * blackAtmPrice /
225  forward; // Brenner-Subrahmanyam formula
226  Real priceAtmVol =
227  blackFormula(optionType, strike, forward, s0, 1.0, 0.0);
228  Real dc = blackPrice - priceAtmVol;
229 
230  if (close(dc, 0.0)) {
231  stdDev = s0;
232  } else {
233  Real d1 =
234  blackFormulaStdDevDerivative(strike, forward, s0, 1.0, 0.0);
235  Real d2 = blackFormulaStdDevSecondDerivative(strike, forward, s0,
236  1.0, 0.0);
237  Real ds = 0.0;
238  Real tmp = d1 * d1 + 2.0 * d2 * dc;
239  if (std::fabs(d2) > 1E-10 && tmp >= 0.0)
240  ds = (-d1 + std::sqrt(tmp)) / d2; // second order approximation
241  else
242  if(std::fabs(d1) > 1E-10)
243  ds = dc / d1; // first order approximation
244  stdDev = s0 + ds;
245  }
246 
247  QL_ENSURE(stdDev >= 0.0, "stdDev (" << stdDev
248  << ") must be non-negative");
249  return stdDev;
250  }

271  Real blackFormulaImpliedStdDevApproximationRS(
272  Option::Type type, Real K, Real F,
273  Real marketValue, Real df, Real displacement) {
274 
275  checkParameters(K, F, displacement);
276  QL_REQUIRE(marketValue >= 0.0,
277  "blackPrice (" << marketValue << ") must be non-negative");
278  QL_REQUIRE(df > 0.0, "discount (" << df << ") must be positive");
279 
280  F = F + displacement;
281  K = K + displacement;
282 
283  const Real ey = F/K;
284  const Real ey2 = ey*ey;
285  const Real y = std::log(ey);
286  const Real alpha = marketValue/(K*df);
287  const Real R = 2*alpha + ((type == Option::Call) ? -ey+1.0 : ey-1.0);
288  const Real R2 = R*R;
289 
290  const Real a = std::exp((1.0-M_2_PI)*y);
291  const Real A = square<Real>()(a - 1.0/a);
292  const Real b = std::exp(M_2_PI*y);
293  const Real B = 4.0*(b + 1/b)
294  - 2*K/F*(a + 1.0/a)*(ey2 + 1 - R2);
295  const Real C = (R2-square<Real>()(ey-1))*(square<Real>()(ey+1)-R2)/ey2;
296 
297  const Real beta = 2*C/(B+std::sqrt(B*B+4*A*C));
298  const Real gamma = -M_PI_2*std::log(beta);
299 
300  if (y >= 0.0) {
301  const Real M0 = K*df*(
302  (type == Option::Call) ? ey*Af(std::sqrt(2*y)) - 0.5
303  : 0.5-ey*Af(-std::sqrt(2*y)));
304 
305  if (marketValue <= M0)
306  return std::sqrt(gamma+y)-std::sqrt(gamma-y);
307  else
308  return std::sqrt(gamma+y)+std::sqrt(gamma-y);
309  }
310  else {
311  const Real M0 = K*df*(
312  (type == Option::Call) ? 0.5*ey - Af(-std::sqrt(-2*y))
313  : Af(std::sqrt(-2*y)) - 0.5*ey);
314 
315  if (marketValue <= M0)
316  return std::sqrt(gamma-y)-std::sqrt(gamma+y);
317  else
318  return std::sqrt(gamma+y)+std::sqrt(gamma-y);
319  }
320  }

332  class BlackImpliedStdDevHelper {
333  public:
334  BlackImpliedStdDevHelper(Option::Type optionType,
335  Real strike,
336  Real forward,
337  Real undiscountedBlackPrice,
338  Real displacement = 0.0)
339  : halfOptionType_(0.5*optionType), signedStrike_(optionType*(strike+displacement)),
340  signedForward_(optionType*(forward+displacement)),
341  undiscountedBlackPrice_(undiscountedBlackPrice)
342  {
343  checkParameters(strike, forward, displacement);
344  QL_REQUIRE(undiscountedBlackPrice>=0.0,
345  "undiscounted Black price (" <<
346  undiscountedBlackPrice << ") must be non-negative");
347  signedMoneyness_ = optionType*std::log((forward+displacement)/(strike+displacement));
348  }
349  Real operator()(Real stdDev) const {
350  #if defined(QL_EXTRA_SAFETY_CHECKS)
351  QL_REQUIRE(stdDev>=0.0,
352  "stdDev (" << stdDev << ") must be non-negative");
353  #endif
354  if (stdDev==0.0)
355  return std::max(signedForward_-signedStrike_, Real(0.0))
356  - undiscountedBlackPrice_;
357  Real temp = halfOptionType_*stdDev;
358  Real d = signedMoneyness_/stdDev;
359  Real signedD1 = d + temp;
360  Real signedD2 = d - temp;
361  Real result = signedForward_ * N_(signedD1)
362  - signedStrike_ * N_(signedD2);
363  // numerical inaccuracies can yield a negative answer
364  return std::max(Real(0.0), result) - undiscountedBlackPrice_;
365  }
366  Real derivative(Real stdDev) const {
367  #if defined(QL_EXTRA_SAFETY_CHECKS)
368  QL_REQUIRE(stdDev>=0.0,
369  "stdDev (" << stdDev << ") must be non-negative");
370  #endif
371  Real signedD1 = signedMoneyness_/stdDev + halfOptionType_*stdDev;
372  return signedForward_*N_.derivative(signedD1);
373  }
374  private:
375  Real halfOptionType_;
376  Real signedStrike_, signedForward_;
377  Real undiscountedBlackPrice_, signedMoneyness_;
378  CumulativeNormalDistribution N_;
379  };

382  Real blackFormulaImpliedStdDev(Option::Type optionType,
383  Real strike,
384  Real forward,
385  Real blackPrice,
386  Real discount,
387  Real displacement,
388  Real guess,
389  Real accuracy,
390  Natural maxIterations)
391  {
392  checkParameters(strike, forward, displacement);
393 
394  QL_REQUIRE(discount>0.0,
395  "discount (" << discount << ") must be positive");
396 
397  QL_REQUIRE(blackPrice>=0.0,
398  "option price (" << blackPrice << ") must be non-negative");
399  // check the price of the "other" option implied by put-call paity
400  Real otherOptionPrice = blackPrice - optionType*(forward-strike)*discount;
401  QL_REQUIRE(otherOptionPrice>=0.0,
402  "negative " << Option::Type(-1*optionType) <<
403  " price (" << otherOptionPrice <<
404  ") implied by put-call parity. No solution exists for " <<
405  optionType << " strike " << strike <<
406  ", forward " << forward <<
407  ", price " << blackPrice <<
408  ", deflator " << discount);
409 
410  // solve for the out-of-the-money option which has
411  // greater vega/price ratio, i.e.
412  // it is numerically more robust for implied vol calculations
413  if (optionType==Option::Put && strike>forward) {
414  optionType = Option::Call;
415  blackPrice = otherOptionPrice;
416  }
417  if (optionType==Option::Call && strike<forward) {
418  optionType = Option::Put;
419  blackPrice = otherOptionPrice;
420  }
421 
422  strike = strike + displacement;
423  forward = forward + displacement;
424 
425  if (guess==Null<Real>())
426  guess = blackFormulaImpliedStdDevApproximation(
427  optionType, strike, forward, blackPrice, discount, displacement);
428  else
429  QL_REQUIRE(guess>=0.0,
430  "stdDev guess (" << guess << ") must be non-negative");
431  BlackImpliedStdDevHelper f(optionType, strike, forward,
432  blackPrice/discount);
433  NewtonSafe solver;
434  solver.setMaxEvaluations(maxIterations);
435  Real minSdtDev = 0.0, maxStdDev = 24.0; // 24 = 300% * sqrt(60)
436  Real stdDev = solver.solve(f, accuracy, guess, minSdtDev, maxStdDev);
437  QL_ENSURE(stdDev>=0.0,
438  "stdDev (" << stdDev << ") must be non-negative");
439  return stdDev;
440  }

457  Real Np(Real x, Real v) {
458  return CumulativeNormalDistribution()(x/v + 0.5*v);
459  }
460  Real Nm(Real x, Real v) {
461  return std::exp(-x)*CumulativeNormalDistribution()(x/v - 0.5*v);
462  }
463  Real phi(Real x, Real v) {
464  const Real ax = 2*std::fabs(x);
465  const Real v2 = v*v;
466  return (v2-ax)/(v2+ax);
467  }
468  Real F(Real v, Real x, Real cs, Real w) {
469  return cs+Nm(x,v)+w*Np(x,v);
470  }
471  Real G(Real v, Real x, Real cs, Real w) {
472  const Real q = F(v,x,cs,w)/(1+w);
473 
474  // Acklam's inverse w/o Halley's refinement step
475  // does not provide enough accuracy. But both together are
476  // slower than the boost replacement.
477  const Real k = MaddockInverseCumulativeNormal()(q);
478 
479  return k + std::sqrt(k*k + 2*std::fabs(x));
480  }
481  }

483  Real blackFormulaImpliedStdDevLiRS(
484  Option::Type optionType,
485  Real strike,
486  Real forward,
487  Real blackPrice,
488  Real discount,
489  Real displacement,
490  Real guess,
491  Real w,
492  Real accuracy,
493  Natural maxIterations) {
494 
495  QL_REQUIRE(discount>0.0,
496  "discount (" << discount << ") must be positive");
497 
498  QL_REQUIRE(blackPrice>=0.0,
499  "option price (" << blackPrice << ") must be non-negative");
500 
501  strike = strike + displacement;
502  forward = forward + displacement;
503 
504  if (guess == Null<Real>()) {
505  guess = blackFormulaImpliedStdDevApproximationRS(
506  optionType, strike, forward,
507  blackPrice, discount, displacement);
508  }
509  else {
510  QL_REQUIRE(guess>=0.0,
511  "stdDev guess (" << guess << ") must be non-negative");
512  }
513 
514  Real x = std::log(forward/strike);
515  Real cs = (optionType == Option::Call)
516  ? blackPrice / (forward*discount)
517  : (blackPrice/ (forward*discount) + 1.0 - strike/forward);
518 
519  QL_REQUIRE(cs >= 0.0, "normalized call price (" << cs
520  << ") must be positive");
521 
522  if (x > 0) {
523  // use in-out duality
524  cs = forward/strike*cs + 1.0 - forward/strike;
525  QL_REQUIRE(cs >= 0.0, "negative option price from in-out duality");
526  x = -x;
527  }
528 
529  Size nIter = 0;
530  Real dv, vk, vkp1 = guess;
531 
532  do {
533  vk = vkp1;
534  const Real alphaK = (1+w)/(1+phi(x,vk));
535  vkp1 = alphaK*G(vk,x,cs,w) + (1-alphaK)*vk;
536  dv = std::fabs(vkp1 - vk);
537  } while (dv > accuracy && ++nIter < maxIterations);
538 
539  QL_REQUIRE(dv <= accuracy, "max iterations exceeded");
540  QL_REQUIRE(vk >= 0.0, "stdDev (" << vk << ") must be non-negative");
541 
542  return vk;
543  }

633  Rate forward,
634  Real stdDev,

651  return discount * forward *

791  Real bachelierBlackFormulaImpliedVol(Option::Type optionType,
792  Real strike,
793  Real forward,
794  Real tte,
795  Real bachelierPrice,
796  Real discount) {
797 
798  const static Real SQRT_QL_EPSILON = std::sqrt(QL_EPSILON);
799 
800  QL_REQUIRE(tte>0.0,
801  "tte (" << tte << ") must be positive");
802 
803  Real forwardPremium = bachelierPrice/discount;
804 
805  Real straddlePremium;
806  if (optionType==Option::Call){
807  straddlePremium = 2.0 * forwardPremium - (forward - strike);
808  } else {
809  straddlePremium = 2.0 * forwardPremium + (forward - strike);
810  }
811 
812  Real nu = (forward - strike) / straddlePremium;
813  QL_REQUIRE(nu<1.0 || close_enough(nu,1.0),
814  "nu (" << nu << ") must be <= 1.0");
815  QL_REQUIRE(nu>-1.0 || close_enough(nu,-1.0),
816  "nu (" << nu << ") must be >= -1.0");
817 
818  nu = std::max(-1.0 + QL_EPSILON, std::min(nu,1.0 - QL_EPSILON));
819 
820  // nu / arctanh(nu) -> 1 as nu -> 0
821  Real eta = (std::fabs(nu) < SQRT_QL_EPSILON) ? 1.0 : nu / boost::math::atanh(nu);
822 
823  Real heta = h(eta);
824 
825  Real impliedBpvol = std::sqrt(M_PI / (2 * tte)) * straddlePremium * heta;
826 
827  return impliedBpvol;
