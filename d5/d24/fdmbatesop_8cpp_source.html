70  Real FdmBatesOp::IntegroIntegrand::operator()(Real y) const {
71  const Real x = x_ + M_SQRT2*delta_*y + nu_;
72  Real valueOfDerivative = (*interpl_)(x, true);
73 
74  for (auto iter = bcSet_.begin(); iter < bcSet_.end(); ++iter) {
75 
76  const ext::shared_ptr<FdmDirichletBoundary> dirichlet
77  = ext::dynamic_pointer_cast<FdmDirichletBoundary>(*iter);
78 
79  QL_REQUIRE(dirichlet, "FdmBatesOp can only deal with Dirichlet "
80  "boundary conditions.");
81 
82  valueOfDerivative
83  = dirichlet->applyAfterApplying(x, valueOfDerivative);
84  }
85 
86  return std::exp(-y*y)*valueOfDerivative;
87  }

89  Disposable<Array> FdmBatesOp::integro(const Array& r) const {
90  const ext::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
91 
92  QL_REQUIRE(layout->dim().size() == 2, "invalid layout dimension");
93 
94  Array x(layout->dim()[0]);
95  Matrix f(layout->dim()[1], layout->dim()[0]);
96 
97  const FdmLinearOpIterator endIter = layout->end();
98  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
99  ++iter) {
100  const Size i = iter.coordinates()[0];
101  const Size j = iter.coordinates()[1];
102 
103  x[i] = mesher_->location(iter, 0);
104  f[j][i] = r[iter.index()];
105 
106  }
107  std::vector<ext::shared_ptr<LinearInterpolation> > interpl(f.rows());
108  for (Size i=0; i < f.rows(); ++i) {
109  interpl[i] = ext::make_shared<LinearInterpolation>(
110  x.begin(), x.end(), f.row_begin(i));
111  }
112 
113  Array integral(r.size());
114  for (FdmLinearOpIterator iter=layout->begin(); iter!=endIter; ++iter) {
115  const Size i = iter.coordinates()[0];
116  const Size j = iter.coordinates()[1];
117 
118  integral[iter.index()] = M_1_SQRTPI*
119  gaussHermiteIntegration_(
120  IntegroIntegrand(interpl[j], bcSet_, x[i], delta_, nu_));
121  }
122 
123  return lambda_*(integral-r);
124  }
