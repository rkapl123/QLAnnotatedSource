158  template <class RNG>
159  inline ext::shared_ptr<LongstaffSchwartzPathPricer<MultiPath> >
160  MCAmericanBasketEngine<RNG>::lsmPathPricer() const {
161 
162  ext::shared_ptr<StochasticProcessArray> processArray =
163  ext::dynamic_pointer_cast<StochasticProcessArray>(
164  this->process_);
165  QL_REQUIRE(processArray && processArray->size()>0,
166  "Stochastic process array required");
167 
168  ext::shared_ptr<GeneralizedBlackScholesProcess> process =
169  ext::dynamic_pointer_cast<GeneralizedBlackScholesProcess>(
170  processArray->process(0));
171  QL_REQUIRE(process, "generalized Black-Scholes process required");
172 
173  ext::shared_ptr<EarlyExercise> exercise =
174  ext::dynamic_pointer_cast<EarlyExercise>(
175  this->arguments_.exercise);
176  QL_REQUIRE(exercise, "wrong exercise given");
177  QL_REQUIRE(!exercise->payoffAtExpiry(),
178  "payoff at expiry not handled");
179 
180  ext::shared_ptr<AmericanBasketPathPricer> earlyExercisePathPricer(
181  new AmericanBasketPathPricer(processArray->size(),
182  this->arguments_.payoff,
183  polynomOrder_,
184  polynomType_));
185 
186  return ext::make_shared<LongstaffSchwartzPathPricer<MultiPath> > (
187 
188  this->timeGrid(),
189  earlyExercisePathPricer,
190  *(process->riskFreeRate()));
191  }
