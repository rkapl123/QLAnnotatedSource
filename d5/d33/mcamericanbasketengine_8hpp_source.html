143  template <class RNG>
144  inline boost::shared_ptr<LongstaffSchwartzPathPricer<MultiPath> >
145  MCAmericanBasketEngine<RNG>::lsmPathPricer() const {
146 
147  boost::shared_ptr<StochasticProcessArray> processArray =
148  boost::dynamic_pointer_cast<StochasticProcessArray>(
149  this->process_);
150  QL_REQUIRE(processArray && processArray->size()>0,
151  "Stochastic process array required");
152 
153  boost::shared_ptr<GeneralizedBlackScholesProcess> process =
154  boost::dynamic_pointer_cast<GeneralizedBlackScholesProcess>(
155  processArray->process(0));
156  QL_REQUIRE(process, "generalized Black-Scholes process required");
157 
158  boost::shared_ptr<EarlyExercise> exercise =
159  boost::dynamic_pointer_cast<EarlyExercise>(
160  this->arguments_.exercise);
161  QL_REQUIRE(exercise, "wrong exercise given");
162  QL_REQUIRE(!exercise->payoffAtExpiry(),
163  "payoff at expiry not handled");
164 
165  boost::shared_ptr<AmericanBasketPathPricer> earlyExercisePathPricer(
166  new AmericanBasketPathPricer(processArray->size(),
167  this->arguments_.payoff));
168 
169  return boost::shared_ptr<LongstaffSchwartzPathPricer<MultiPath> > (
170  new LongstaffSchwartzPathPricer<MultiPath>(
171  this->timeGrid(),
172  earlyExercisePathPricer,
173  *(process->riskFreeRate())));
174  }
