118  void NonstandardSwap::init() {
119 
120  QL_REQUIRE(fixedNominal_.size() == fixedRate_.size(),
121  "Fixed nominal size ("
122  << fixedNominal_.size()
123  << ") does not match fixed rate size ("
124  << fixedRate_.size() << ")");
125 
126  QL_REQUIRE(fixedNominal_.size() == fixedSchedule_.size() - 1,
127  "Fixed nominal size (" << fixedNominal_.size()
128  << ") does not match schedule size ("
129  << fixedSchedule_.size() << ") - 1");
130 
131  QL_REQUIRE(floatingNominal_.size() == floatingSchedule_.size() - 1,
132  "Floating nominal size ("
133  << floatingNominal_.size()
134  << ") does not match schedule size ("
135  << floatingSchedule_.size() << ") - 1");
136 
137  QL_REQUIRE(floatingNominal_.size() == spread_.size(),
138  "Floating nominal size (" << floatingNominal_.size()
139  << ") does not match spread size ("
140  << spread_.size() << ")");
141 
142  QL_REQUIRE(floatingNominal_.size() == gearing_.size(),
143  "Floating nominal size ("
144  << floatingNominal_.size()
145  << ") does not match gearing size (" << gearing_.size()
146  << ")");
147 
148  // if the gearing is zero then the ibor leg will be set up with fixed
149  // coupons which makes trouble here in this context. We therefore use
150  // a dirty trick and enforce the gearing to be non zero.
151  for (double& i : gearing_) {
152  if (close(i, 0.0))
153  i = QL_EPSILON;
154  }
155 
156  legs_[0] = FixedRateLeg(fixedSchedule_)
157  .withNotionals(fixedNominal_)
158  .withCouponRates(fixedRate_, fixedDayCount_)
159  .withPaymentAdjustment(paymentConvention_);
160 
161  legs_[1] = IborLeg(floatingSchedule_, iborIndex_)
162  .withNotionals(floatingNominal_)
163  .withPaymentDayCounter(floatingDayCount_)
164  .withPaymentAdjustment(paymentConvention_)
165  .withSpreads(spread_)
166  .withGearings(gearing_);
167 
168  if (intermediateCapitalExchange_) {
169  for (Size i = 0; i < legs_[0].size() - 1; i++) {
170  Real cap = fixedNominal_[i] - fixedNominal_[i + 1];
171  if (!close(cap, 0.0)) {
172  auto it1 = legs_[0].begin();
173  std::advance(it1, i + 1);
174  legs_[0].insert(
175  it1, ext::shared_ptr<CashFlow>(
176  new Redemption(cap, legs_[0][i]->date())));
177  auto it2 = fixedNominal_.begin();
178  std::advance(it2, i + 1);
179  fixedNominal_.insert(it2, fixedNominal_[i]);
180  auto it3 = fixedRate_.begin();
181  std::advance(it3, i + 1);
182  fixedRate_.insert(it3, 0.0);
183  i++;
184  }
185  }
186  for (Size i = 0; i < legs_[1].size() - 1; i++) {
187  Real cap = floatingNominal_[i] - floatingNominal_[i + 1];
188  if (!close(cap, 0.0)) {
189  auto it1 = legs_[1].begin();
190  std::advance(it1, i + 1);
191  legs_[1].insert(
192  it1, ext::shared_ptr<CashFlow>(
193  new Redemption(cap, legs_[1][i]->date())));
194  auto it2 = floatingNominal_.begin();
195  std::advance(it2, i + 1);
196  floatingNominal_.insert(it2, floatingNominal_[i]);
197  i++;
198  }
199  }
200  }
201 
202  if (finalCapitalExchange_) {
203  legs_[0].push_back(ext::shared_ptr<CashFlow>(
204  new Redemption(fixedNominal_.back(), legs_[0].back()->date())));
205  fixedNominal_.push_back(fixedNominal_.back());
206  fixedRate_.push_back(0.0);
207  legs_[1].push_back(ext::shared_ptr<CashFlow>(new Redemption(
208  floatingNominal_.back(), legs_[1].back()->date())));
209  floatingNominal_.push_back(floatingNominal_.back());
210  }
211 
212  for (Leg::const_iterator i = legs_[1].begin(); i < legs_[1].end(); ++i)
213  registerWith(*i);
214 
215  switch (type_) {
216  case VanillaSwap::Payer:
217  payer_[0] = -1.0;
218  payer_[1] = +1.0;
219  break;
220  case VanillaSwap::Receiver:
221  payer_[0] = +1.0;
222  payer_[1] = -1.0;
223  break;
224  default:
225  QL_FAIL("Unknown nonstandard-swap type");
226  }
227  }

229  void NonstandardSwap::setupArguments(PricingEngine::arguments *args) const {
230 
231  Swap::setupArguments(args);
232 
233  auto* arguments = dynamic_cast<NonstandardSwap::arguments*>(args);
234 
235  if (arguments == nullptr)
236  return; // swap engine ...
237 
238  arguments->type = type_;
239  arguments->fixedNominal = fixedNominal_;
240  arguments->floatingNominal = floatingNominal_;
241  arguments->fixedRate = fixedRate_;
242 
243  const Leg &fixedCoupons = fixedLeg();
244 
245  arguments->fixedResetDates = arguments->fixedPayDates =
246  std::vector<Date>(fixedCoupons.size());
247  arguments->fixedCoupons = std::vector<Real>(fixedCoupons.size());
248  arguments->fixedIsRedemptionFlow =
249  std::vector<bool>(fixedCoupons.size(), false);
250 
251  for (Size i = 0; i < fixedCoupons.size(); ++i) {
252  ext::shared_ptr<FixedRateCoupon> coupon =
253  ext::dynamic_pointer_cast<FixedRateCoupon>(fixedCoupons[i]);
254  if (coupon != nullptr) {
255  arguments->fixedPayDates[i] = coupon->date();
256  arguments->fixedResetDates[i] = coupon->accrualStartDate();
257  arguments->fixedCoupons[i] = coupon->amount();
258  } else {
259  ext::shared_ptr<CashFlow> cashflow =
260  ext::dynamic_pointer_cast<CashFlow>(fixedCoupons[i]);
261  std::vector<Date>::const_iterator j =
262  std::find(arguments->fixedPayDates.begin(),
263  arguments->fixedPayDates.end(), cashflow->date());
264  QL_REQUIRE(j != arguments->fixedPayDates.end(),
265  "nominal redemption on "
266  << cashflow->date()
267  << "has no corresponding coupon");
268  Size jIdx = j - arguments->fixedPayDates.begin();
269  arguments->fixedIsRedemptionFlow[i] = true;
270  arguments->fixedCoupons[i] = cashflow->amount();
271  arguments->fixedResetDates[i] =
272  arguments->fixedResetDates[jIdx];
273  arguments->fixedPayDates[i] = cashflow->date();
274  }
275  }
276 
277  const Leg &floatingCoupons = floatingLeg();
278 
279  arguments->floatingResetDates = arguments->floatingPayDates =
280  arguments->floatingFixingDates =
281  std::vector<Date>(floatingCoupons.size());
282  arguments->floatingAccrualTimes =
283  std::vector<Time>(floatingCoupons.size());
284  arguments->floatingSpreads =
285  std::vector<Spread>(floatingCoupons.size());
286  arguments->floatingGearings = std::vector<Real>(floatingCoupons.size());
287  arguments->floatingCoupons = std::vector<Real>(floatingCoupons.size());
288  arguments->floatingIsRedemptionFlow =
289  std::vector<bool>(floatingCoupons.size(), false);
290 
291  for (Size i = 0; i < floatingCoupons.size(); ++i) {
292  ext::shared_ptr<IborCoupon> coupon =
293  ext::dynamic_pointer_cast<IborCoupon>(floatingCoupons[i]);
294  if (coupon != nullptr) {
295  arguments->floatingResetDates[i] = coupon->accrualStartDate();
296  arguments->floatingPayDates[i] = coupon->date();
297  arguments->floatingFixingDates[i] = coupon->fixingDate();
298  arguments->floatingAccrualTimes[i] = coupon->accrualPeriod();
299  arguments->floatingSpreads[i] = coupon->spread();
300  arguments->floatingGearings[i] = coupon->gearing();
301  try {
302  arguments->floatingCoupons[i] = coupon->amount();
303  }
304  catch (Error &) {
305  arguments->floatingCoupons[i] = Null<Real>();
306  }
307  } else {
308  ext::shared_ptr<CashFlow> cashflow =
309  ext::dynamic_pointer_cast<CashFlow>(floatingCoupons[i]);
310  std::vector<Date>::const_iterator j = std::find(
311  arguments->floatingPayDates.begin(),
312  arguments->floatingPayDates.end(), cashflow->date());
313  QL_REQUIRE(j != arguments->floatingPayDates.end(),
314  "nominal redemption on "
315  << cashflow->date()
316  << "has no corresponding coupon");
317  Size jIdx = j - arguments->floatingPayDates.begin();
318  arguments->floatingIsRedemptionFlow[i] = true;
319  arguments->floatingCoupons[i] = cashflow->amount();
320  arguments->floatingResetDates[i] =
321  arguments->floatingResetDates[jIdx];
322  arguments->floatingFixingDates[i] =
323  arguments->floatingFixingDates[jIdx];
324  arguments->floatingAccrualTimes[i] = 0.0;
325  arguments->floatingSpreads[i] = 0.0;
326  arguments->floatingGearings[i] = 1.0;
327  arguments->floatingPayDates[i] = cashflow->date();
328  }
329  }
330 
331  arguments->iborIndex = iborIndex();
332  }

341  void NonstandardSwap::arguments::validate() const {
342  Swap::arguments::validate();
343  QL_REQUIRE(fixedNominal.size() == fixedPayDates.size(),
344  "number of fixed leg nominals plus redemption flows "
345  "different from number of payment dates");
346  QL_REQUIRE(fixedRate.size() == fixedPayDates.size(),
347  "number of fixed rates plus redemption flows different from "
348  "number of payment dates");
349  QL_REQUIRE(floatingNominal.size() == floatingPayDates.size(),
350  "number of float leg nominals different from number of "
351  "payment dates");
352  QL_REQUIRE(fixedResetDates.size() == fixedPayDates.size(),
353  "number of fixed start dates different from "
354  "number of fixed payment dates");
355  QL_REQUIRE(fixedPayDates.size() == fixedCoupons.size(),
356  "number of fixed payment dates different from "
357  "number of fixed coupon amounts");
358  QL_REQUIRE(floatingResetDates.size() == floatingPayDates.size(),
359  "number of floating start dates different from "
360  "number of floating payment dates");
361  QL_REQUIRE(floatingFixingDates.size() == floatingPayDates.size(),
362  "number of floating fixing dates different from "
363  "number of floating payment dates");
364  QL_REQUIRE(floatingAccrualTimes.size() == floatingPayDates.size(),
365  "number of floating accrual Times different from "
366  "number of floating payment dates");
367  QL_REQUIRE(floatingSpreads.size() == floatingPayDates.size(),
368  "number of floating spreads different from "
369  "number of floating payment dates");
370  QL_REQUIRE(floatingPayDates.size() == floatingCoupons.size(),
371  "number of floating payment dates different from "
372  "number of floating coupon amounts");
373  }
