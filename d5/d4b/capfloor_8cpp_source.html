55  ImpliedCapVolHelper::ImpliedCapVolHelper(const CapFloor& cap,
56  Handle<YieldTermStructure> discountCurve,
57  Real targetValue,
58  Real displacement,
59  VolatilityType type)
60  : discountCurve_(std::move(discountCurve)), targetValue_(targetValue),
61  vol_(ext::make_shared<SimpleQuote>(-1.0)) {
62 
63  // vol_ is set an implausible value, so that calculation is forced
64  // at first ImpliedCapVolHelper::operator()(Volatility x) call
65  Handle<Quote> h(vol_);
66 
67  switch (type) {
68  case ShiftedLognormal:
69  engine_ = ext::shared_ptr<PricingEngine>(new
70  BlackCapFloorEngine(discountCurve_, h, Actual365Fixed(),
71  displacement));
72  break;
73  case Normal:
74  engine_ = ext::shared_ptr<PricingEngine>(new
75  BachelierCapFloorEngine(discountCurve_, h,
76  Actual365Fixed()));
77  break;
78  default:
79  QL_FAIL("unknown VolatilityType (" << type << ")");
80  break;
81  }
82 
83  cap.setupArguments(engine_->getArguments());
84 
85  results_ =
86  dynamic_cast<const Instrument::results*>(engine_->getResults());
87  }

89  Real ImpliedCapVolHelper::operator()(Volatility x) const {
90  if (x!=vol_->value()) {
91  vol_->setValue(x);
92  engine_->calculate();
93  }
94  return results_->value-targetValue_;
95  }

123  CapFloor::CapFloor(CapFloor::Type type,
124  Leg floatingLeg,
125  std::vector<Rate> capRates,
126  std::vector<Rate> floorRates)
127  : type_(type), floatingLeg_(std::move(floatingLeg)), capRates_(std::move(capRates)),
128  floorRates_(std::move(floorRates)) {
129  if (type_ == Cap || type_ == Collar) {
130  QL_REQUIRE(!capRates_.empty(), "no cap rates given");
131  capRates_.reserve(floatingLeg_.size());
132  while (capRates_.size() < floatingLeg_.size())
133  capRates_.push_back(capRates_.back());
134  }
135  if (type_ == Floor || type_ == Collar) {
136  QL_REQUIRE(!floorRates_.empty(), "no floor rates given");
137  floorRates_.reserve(floatingLeg_.size());
138  while (floorRates_.size() < floatingLeg_.size())
139  floorRates_.push_back(floorRates_.back());
140  }
141  Leg::const_iterator i;
142  for (i = floatingLeg_.begin(); i != floatingLeg_.end(); ++i)
143  registerWith(*i);
144 
145  registerWith(Settings::instance().evaluationDate());
146  }

148  CapFloor::CapFloor(CapFloor::Type type, Leg floatingLeg, const std::vector<Rate>& strikes)
149  : type_(type), floatingLeg_(std::move(floatingLeg)) {
150  QL_REQUIRE(!strikes.empty(), "no strikes given");
151  if (type_ == Cap) {
152  capRates_ = strikes;
153  capRates_.reserve(floatingLeg_.size());
154  while (capRates_.size() < floatingLeg_.size())
155  capRates_.push_back(capRates_.back());
156  } else if (type_ == Floor) {
157  floorRates_ = strikes;
158  floorRates_.reserve(floatingLeg_.size());
159  while (floorRates_.size() < floatingLeg_.size())
160  floorRates_.push_back(floorRates_.back());
161  } else
162  QL_FAIL("only Cap/Floor types allowed in this constructor");
163 
164  Leg::const_iterator i;
165  for (i = floatingLeg_.begin(); i != floatingLeg_.end(); ++i)
166  registerWith(*i);
167 
168  registerWith(Settings::instance().evaluationDate());
169  }

195  QL_REQUIRE(i < floatingLeg().size(),

209  void CapFloor::setupArguments(PricingEngine::arguments* args) const {
210  auto* arguments = dynamic_cast<CapFloor::arguments*>(args);
211  QL_REQUIRE(arguments != nullptr, "wrong argument type");
212 
213  Size n = floatingLeg_.size();
214 
215  arguments->startDates.resize(n);
216  arguments->fixingDates.resize(n);
217  arguments->endDates.resize(n);
218  arguments->accrualTimes.resize(n);
219  arguments->forwards.resize(n);
220  arguments->nominals.resize(n);
221  arguments->gearings.resize(n);
222  arguments->capRates.resize(n);
223  arguments->floorRates.resize(n);
224  arguments->spreads.resize(n);
225  arguments->indexes.resize(n);
226 
227  arguments->type = type_;
228 
229  Date today = Settings::instance().evaluationDate();
230 
231  for (Size i=0; i<n; ++i) {
232  ext::shared_ptr<FloatingRateCoupon> coupon =
233  ext::dynamic_pointer_cast<FloatingRateCoupon>(
234  floatingLeg_[i]);
235  QL_REQUIRE(coupon, "non-FloatingRateCoupon given");
236  arguments->startDates[i] = coupon->accrualStartDate();
237  arguments->fixingDates[i] = coupon->fixingDate();
238  arguments->endDates[i] = coupon->date();
239 
240  // this is passed explicitly for precision
241  arguments->accrualTimes[i] = coupon->accrualPeriod();
242 
243  // this is passed explicitly for precision...
244  if (arguments->endDates[i] >= today) { // ...but only if needed
245  arguments->forwards[i] = coupon->adjustedFixing();
246  } else {
247  arguments->forwards[i] = Null<Rate>();
248  }
249 
250  arguments->nominals[i] = coupon->nominal();
251  Spread spread = coupon->spread();
252  Real gearing = coupon->gearing();
253  arguments->gearings[i] = gearing;
254  arguments->spreads[i] = spread;
255 
256  if (type_ == Cap || type_ == Collar)
257  arguments->capRates[i] = (capRates_[i]-spread)/gearing;
258  else
259  arguments->capRates[i] = Null<Rate>();
260 
261  if (type_ == Floor || type_ == Collar)
262  arguments->floorRates[i] = (floorRates_[i]-spread)/gearing;
263  else
264  arguments->floorRates[i] = Null<Rate>();
265 
266  arguments->indexes[i] = coupon->index();
267  }
268  }

279  void CapFloor::arguments::validate() const {
280  QL_REQUIRE(endDates.size() == startDates.size(),
281  "number of start dates (" << startDates.size()

284  QL_REQUIRE(accrualTimes.size() == startDates.size(),
285  "number of start dates (" << startDates.size()
286  << ") different from that of accrual times ("
287  << accrualTimes.size() << ")");
288  QL_REQUIRE(type == CapFloor::Floor ||
289  capRates.size() == startDates.size(),
290  "number of start dates (" << startDates.size()
291  << ") different from that of cap rates ("
292  << capRates.size() << ")");
293  QL_REQUIRE(type == CapFloor::Cap ||
294  floorRates.size() == startDates.size(),
295  "number of start dates (" << startDates.size()
296  << ") different from that of floor rates ("
297  << floorRates.size() << ")");
298  QL_REQUIRE(gearings.size() == startDates.size(),
299  "number of start dates (" << startDates.size()
300  << ") different from that of gearings ("
301  << gearings.size() << ")");
302  QL_REQUIRE(spreads.size() == startDates.size(),
303  "number of start dates (" << startDates.size()
304  << ") different from that of spreads ("
305  << spreads.size() << ")");
306  QL_REQUIRE(nominals.size() == startDates.size(),
307  "number of start dates (" << startDates.size()
308  << ") different from that of nominals ("
309  << nominals.size() << ")");
310  QL_REQUIRE(forwards.size() == startDates.size(),
311  "number of start dates (" << startDates.size()
312  << ") different from that of forwards ("
