38  bool SingleProductComposite::nextTimeStep(
39  const CurveState& currentState,
40  std::vector<Size>& numberCashFlowsThisStep,
41  std::vector<std::vector<CashFlow> >& cashFlowsGenerated) {
42  QL_REQUIRE(finalized_, "composite not finalized");
43  bool done = true;
44  Size n = 0, totalCashflows = 0;
45  // for each sub-product...
46  for (iterator i=components_.begin(); i!=components_.end(); ++i, ++n) {
47  if (isInSubset_[n][currentIndex_] && !i->done) {
48  // ...make it evolve...
49  bool thisDone = i->product->nextTimeStep(currentState,
50  i->numberOfCashflows,
51  i->cashflows);
52  // ...and copy the results. Time indices need to be remapped
53  // so that they point into all cash-flow times. Amounts need
54  // to be adjusted by the corresponding multiplier.
55  for (Size j=0; j<i->product->numberOfProducts(); ++j) {
56  Size offset = totalCashflows;
57  totalCashflows += i->numberOfCashflows[j];
58  for (Size k=0; k<i->numberOfCashflows[j]; ++k) {
59  CashFlow& from = i->cashflows[j][k];
60  CashFlow& to = cashFlowsGenerated[0][k+offset];
61  to.timeIndex = i->timeIndices[from.timeIndex];
62  to.amount = from.amount * i->multiplier;
63  }
64  numberCashFlowsThisStep[0] = totalCashflows;
65  }
66  // finally, set done to false if this product isn't done
67  done = done && thisDone;
68  }
69  }
70  ++currentIndex_;
71  return done;
72  }
