57  void FdKlugeExtOUSpreadEngine::calculate() const {
58  // 1. Mesher
59  const Time maturity
60  = rTS_->dayCounter().yearFraction(rTS_->referenceDate(),
61  arguments_.exercise->lastDate());
62  const boost::shared_ptr<ExtOUWithJumpsProcess> klugeProcess
63  = klugeOUProcess_->getKlugeProcess();
64  const boost::shared_ptr<StochasticProcess1D> ouProcess
65  = klugeProcess->getExtendedOrnsteinUhlenbeckProcess();
66  const boost::shared_ptr<Fdm1dMesher> xMesher(
67  new FdmSimpleProcess1dMesher(xGrid_, ouProcess,maturity));
68 
69  const boost::shared_ptr<Fdm1dMesher> yMesher(
70  new ExponentialJump1dMesher(yGrid_,
71  klugeProcess->beta(),
72  klugeProcess->jumpIntensity(),
73  klugeProcess->eta()));
74 
75  const boost::shared_ptr<Fdm1dMesher> uMesher(
76  new FdmSimpleProcess1dMesher(uGrid_,
77  klugeOUProcess_->getExtOUProcess(),
78  maturity));
79 
80  const boost::shared_ptr<FdmMesher> mesher(
81  new FdmMesherComposite(xMesher, yMesher, uMesher));
82 
83  // 2. Calculator
84  boost::shared_ptr<BasketPayoff> basketPayoff =
85  boost::dynamic_pointer_cast<BasketPayoff>(arguments_.payoff);
86  QL_REQUIRE(basketPayoff," basket payoff expected");
87 
88  const boost::shared_ptr<Payoff> zeroStrikeCall(
89  new PlainVanillaPayoff(Option::Call, 0.0));
90 
91  const boost::shared_ptr<FdmInnerValueCalculator> gasPrice(
92  new FdmExpExtOUInnerValueCalculator(zeroStrikeCall,
93  mesher, gasShape_, 2));
94 
95  const boost::shared_ptr<FdmInnerValueCalculator> powerPrice(
96  new FdmExtOUJumpModelInnerValue(zeroStrikeCall,mesher,powerShape_));
97 
98  const boost::shared_ptr<FdmInnerValueCalculator> calculator(
99  new FdmSpreadPayoffInnerValue(basketPayoff, powerPrice, gasPrice));
100 
101  // 3. Step conditions
102  const boost::shared_ptr<FdmStepConditionComposite> conditions =
103  FdmStepConditionComposite::vanillaComposite(
104  DividendSchedule(), arguments_.exercise,
105  mesher, calculator,
106  rTS_->referenceDate(), rTS_->dayCounter());
107 
108  // 4. Boundary conditions
109  const FdmBoundaryConditionSet boundaries;
110 
111  // 5. set-up solver
112  FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
113  calculator, maturity, tGrid_, 0 };
114 
115  const boost::shared_ptr<FdmKlugeExtOUSolver<3> > solver(
116  new FdmKlugeExtOUSolver<3>(
117  Handle<KlugeExtOUProcess>(klugeOUProcess_),
118  rTS_, solverDesc, schemeDesc_));
119 
120  std::vector<Real> x(3);
121  x[0] = klugeOUProcess_->initialValues()[0];
122  x[1] = klugeOUProcess_->initialValues()[1];
123  x[2] = klugeOUProcess_->initialValues()[2];
124 
125  results_.value = solver->valueAt(x);
126  }
127 }
