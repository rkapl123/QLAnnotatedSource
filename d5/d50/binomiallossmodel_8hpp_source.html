156  /* The algorithm to compute the prob. of n defaults in the basket is
157  recursive. For this reason theres no sense in returning the prob
158  distribution of a given number of defaults.
159  */
160  template< class LLM>
161  Disposable<std::vector<Real> > BinomialLossModel<LLM>::lossProbability(
162  const Date& date,
163  const std::vector<Real>& bsktNots,
164  const std::vector<Real>& uncondDefProbInv,
165  const std::vector<Real>& mktFactors) const
166  { // the model as it is does not model the exposures conditional to the
167  // mkt factr, otherwise this needs revision
169  Size bsktSize = basket_->remainingSize();
170  /* The conditional loss per unit notional of each name at time 'date'
171  The spot recovery model is returning for all i's:
172  \frac{\int_0^t [1-rr_i(\tau; \xi)] P_{def-i}(0, \tau; \xi) d\tau}
173  {P_{def-i}(0,t;\xi)}
174  and the constant recovery model is simply returning:
175  1-RR_i
176  */
177  // conditional fractional LGD expected as given by the recovery model
178  // for the ramaining(live) names at the current eval date.
179  std::vector<Real> fractionalEL = expConditionalLgd(date, mktFactors);
180  std::vector<Real> lgdsLeft;
181  std::transform(fractionalEL.begin(), fractionalEL.end(),
182  bsktNots.begin(), std::back_inserter(lgdsLeft),
183  std::multiplies<Real>());
184  Real avgLgd =
185  std::accumulate(lgdsLeft.begin(), lgdsLeft.end(), Real(0.)) /
186  bsktSize;
187 
188  std::vector<Probability> condDefProb(bsktSize, 0.);
189  for(Size j=0; j<bsktSize; j++)//transform
190  condDefProb[j] =
191  copula_->conditionalDefaultProbabilityInvP(uncondDefProbInv[j],
192  j, mktFactors);
193  // of full portfolio:
194  Real avgProb = avgLgd <= QL_EPSILON ? 0. : // only if all are 0
195  std::inner_product(condDefProb.begin(),
196  condDefProb.end(), lgdsLeft.begin(), 0.)
197  / (avgLgd * bsktSize);
198  // model parameters:
199  Real m = avgProb * bsktSize;
200  Real floorAveProb = std::min(Real(bsktSize-1), std::floor(Real(m)));
201  Real ceilAveProb = floorAveProb + 1.;
202  // nu_A
203  Real varianceBinom = avgProb * (1. - avgProb)/bsktSize;
204  // nu_E
205  std::vector<Probability> oneMinusDefProb;//: 1.-condDefProb[j]
206  std::transform(condDefProb.begin(), condDefProb.end(),
207  std::back_inserter(oneMinusDefProb),
208  subtract_from<Real>(1.0));
209 
210  //breaks condDefProb and lgdsLeft to spare memory
211  std::transform(condDefProb.begin(), condDefProb.end(),
212  oneMinusDefProb.begin(), condDefProb.begin(),
213  std::multiplies<Real>());
214  std::transform(lgdsLeft.begin(), lgdsLeft.end(),
215  lgdsLeft.begin(), lgdsLeft.begin(), std::multiplies<Real>());
216  Real variance = std::inner_product(condDefProb.begin(),
217  condDefProb.end(), lgdsLeft.begin(), 0.);
218 
219  variance = avgLgd <= QL_EPSILON ? 0. :
220  variance / (bsktSize * bsktSize * avgLgd * avgLgd );
221  Real sumAves = -std::pow(ceilAveProb-m, 2)
222  - (std::pow(floorAveProb-m, 2) - std::pow(ceilAveProb,2.))
223  * (ceilAveProb-m);
224  Real alpha = (variance * bsktSize + sumAves)
225  / (varianceBinom * bsktSize + sumAves);
226  // Full distribution:
227  // ....DO SOMETHING CHEAPER at least go up to the loss tranche limit.
228  std::vector<Probability> lossProbDensity(bsktSize+1, 0.);
229  if(avgProb >= 1.-QL_EPSILON) {
230  lossProbDensity[bsktSize] = 1.;
231  }else if(avgProb <= QL_EPSILON) {
232  lossProbDensity[0] = 1.;
233  }else{
234  /* FIX ME: With high default probabilities one only gets tiny values
235  at the end and the sum of probabilities in the
236  conditional distribution does not add up to one. It might be due to
237  the fact that recursion should be done in the other direction as
238  pointed out in the book. This is numerical.
239  */
240  Probability probsRatio = avgProb/(1.-avgProb);
241  lossProbDensity[0] = std::pow(1.-avgProb,
242  static_cast<Real>(bsktSize));
243  for(Size i=1; i<bsktSize+1; i++) // recursive to avoid factorial
244  lossProbDensity[i] = lossProbDensity[i-1] * probsRatio
245  * (bsktSize-i+1.)/i;
246  // redistribute probability:
247  for(Size i=0; i<bsktSize+1; i++)
248  lossProbDensity[i] *= alpha;
249  // adjust average
250  Real epsilon = (1.-alpha)*(ceilAveProb-m);
251  Real epsilonPlus = 1.-alpha-epsilon;
252  lossProbDensity[static_cast<Size>(floorAveProb)] += epsilon;
253  lossProbDensity[static_cast<Size>(ceilAveProb)] += epsilonPlus;
254  }
255  return lossProbDensity;
256  }

260  template< class LLM>
261  Real BinomialLossModel<LLM>::averageLoss(
262  const Date& d,
263  const std::vector<Real>& reminingNots,
264  const std::vector<Real>& mktFctrs) const
265  {
266  Size bsktSize = basket_->remainingSize();
267  /* The conditional loss per unit notional of each name at time 'date'
268  The spot recovery model is returning for all i's:
269  \frac{\int_0^t [1-rr_i(\tau; \xi)] P_{def-i}(0, \tau; \xi) d\tau}
270  {P_{def-i}(0,t;\xi)}
271  and the constant recovery model is simply returning:
272  1-RR_i
273  */
274  std::vector<Real> fractionalEL = expConditionalLgd(d, mktFctrs);
275  Real notBskt = std::accumulate(reminingNots.begin(),
276  reminingNots.end(), Real(0.));
277  std::vector<Real> lgdsLeft;
278  std::transform(fractionalEL.begin(), fractionalEL.end(),
279  reminingNots.begin(), std::back_inserter(lgdsLeft),
280  std::multiplies<Real>());
281  return std::accumulate(lgdsLeft.begin(), lgdsLeft.end(), Real(0.))
282  / (bsktSize*notBskt);
283  }

375  template< class LLM>
376  Real BinomialLossModel<LLM>::percentile(const Date& d, Real perc) const {
377  std::map<Real, Probability> dist = lossDistribution(d);
378  // \todo: Use some of the library interpolators instead
379  if(// included in test below-> (dist.begin()->second >=1.) ||
380  (dist.begin()->second >= perc))return dist.begin()->first;
381 
382  // deterministic case (e.g. date requested is todays date)
383  if(dist.size() == 1) return dist.begin()->first;
384 
385  if(perc == 1.) return dist.rbegin()->first;
386  if(perc == 0.) return dist.begin()->first;
387  std::map<Real, Probability>::const_iterator itdist = dist.begin();
388  while (itdist->second <= perc) ++itdist;
389  Real valPlus = itdist->second;
390  Real xPlus = itdist->first;
391  --itdist; //we're never 1st or last, because of tests above
392  Real valMin = itdist->second;
393  Real xMin = itdist->first;
394 
395  Real portfLoss = xPlus-(xPlus-xMin)*(valPlus-perc)/(valPlus-valMin);
396 
397  return
398  std::min(std::max(portfLoss - attachAmount_, 0.),
399  detachAmount_ - attachAmount_);
400  }

402  template< class LLM>
403  Real BinomialLossModel<LLM>::expectedShortfall(const Date&d,
404  Real perctl) const
405  {
406  //taken from recursive since we have the distribution in both cases.
407  if(d == Settings::instance().evaluationDate()) return 0.;
408  std::map<Real, Probability> distrib = lossDistribution(d);
409 
410  std::map<Real, Probability>::iterator
411  itNxt, itDist = distrib.begin();
412  for(; itDist != distrib.end(); ++itDist)
413  if(itDist->second >= perctl) break;
414  itNxt = itDist;
415  --itDist;
416 
417  // \todo: I could linearly triangulate the exact point and get
418  // extra precission on the first(broken) period.
419  if(itNxt != distrib.end()) {
420  Real lossNxt = std::min(std::max(itNxt->first - attachAmount_,
421  0.), detachAmount_ - attachAmount_);
422  Real lossHere = std::min(std::max(itDist->first - attachAmount_,
423  0.), detachAmount_ - attachAmount_);
424 
425  Real val = lossNxt - (itNxt->second - perctl) *
426  (lossNxt - lossHere) / (itNxt->second - itDist->second);
427  Real suma = (itNxt->second - perctl) * (lossNxt + val) * .5;
428  ++itDist; ++itNxt;
429  do{
430  lossNxt = std::min(std::max(itNxt->first - attachAmount_,
431  0.), detachAmount_ - attachAmount_);
432  lossHere = std::min(std::max(itDist->first - attachAmount_,
433  0.), detachAmount_ - attachAmount_);
434  suma += .5 * (lossHere + lossNxt)
435  * (itNxt->second - itDist->second);
436  ++itDist; ++itNxt;
437  }while(itNxt != distrib.end());
438  return suma / (1.-perctl);
439  }
440  QL_FAIL("Binomial model fails to calculate ESF.");
441  }
