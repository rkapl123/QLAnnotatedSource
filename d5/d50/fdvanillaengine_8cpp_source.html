48  void FDVanillaEngine::setGridLimits(Real center, Time t) const {
49  QL_REQUIRE(center > 0.0, "negative or null underlying given");
50  QL_REQUIRE(t > 0.0, "negative or zero residual time");
51  center_ = center;
52  Size newGridPoints = safeGridPoints(gridPoints_, t);
53  if (newGridPoints > intrinsicValues_.size()) {
54  intrinsicValues_ = SampledCurve(newGridPoints);
55  }
56 
57  Real volSqrtTime = std::sqrt(process_->blackVolatility()
58  ->blackVariance(t, center_));
59 
60  // the prefactor fine tunes performance at small volatilities
61  Real prefactor = 1.0 + 0.02/volSqrtTime;
62  Real minMaxFactor = std::exp(4.0 * prefactor * volSqrtTime);
63  sMin_ = center_/minMaxFactor; // underlying grid min value
64  sMax_ = center_*minMaxFactor; // underlying grid max value
65  }

67  void FDVanillaEngine::ensureStrikeInGrid() const {
68  // ensure strike is included in the grid
69  boost::shared_ptr<StrikedTypePayoff> striked_payoff =
70  boost::dynamic_pointer_cast<StrikedTypePayoff>(payoff_);
71  if (!striked_payoff)
72  return;
73  Real requiredGridValue = striked_payoff->strike();
74 
75  if(sMin_ > requiredGridValue/safetyZoneFactor_){
76  sMin_ = requiredGridValue/safetyZoneFactor_;
77  // enforce central placement of the underlying
78  sMax_ = center_/(sMin_/center_);
79  }
80  if(sMax_ < requiredGridValue*safetyZoneFactor_){
81  sMax_ = requiredGridValue*safetyZoneFactor_;
82  // enforce central placement of the underlying
83  sMin_ = center_/(sMax_/center_);
84  }
85  }

115  // safety check to be sure we have enough grid points.
116  Size FDVanillaEngine::safeGridPoints(Size gridPoints,
117  Time residualTime) const {
118  static const Size minGridPoints = 10;
119  static const Size minGridPointsPerYear = 2;
120  return std::max(gridPoints,
121  residualTime > 1.0 ?
122  static_cast<Size>((minGridPoints +
123  (residualTime-1.0) *
124  minGridPointsPerYear))
125  : minGridPoints);
126  }
