49  void FDVanillaEngine::setGridLimits(Real center, Time t) const {
50  QL_REQUIRE(center > 0.0, "negative or null underlying given");
51  QL_REQUIRE(t > 0.0, "negative or zero residual time");
52  center_ = center;
53  Size newGridPoints = safeGridPoints(gridPoints_, t);
54  if (newGridPoints > intrinsicValues_.size()) {
55  intrinsicValues_ = SampledCurve(newGridPoints);
56  }
57 
58  Real volSqrtTime = std::sqrt(process_->blackVolatility()
59  ->blackVariance(t, center_));
60 
61  // the prefactor fine tunes performance at small volatilities
62  Real prefactor = 1.0 + 0.02/volSqrtTime;
63  Real minMaxFactor = std::exp(4.0 * prefactor * volSqrtTime);
64  sMin_ = center_/minMaxFactor; // underlying grid min value
65  sMax_ = center_*minMaxFactor; // underlying grid max value
66  }

68  void FDVanillaEngine::ensureStrikeInGrid() const {
69  // ensure strike is included in the grid
70  ext::shared_ptr<StrikedTypePayoff> striked_payoff =
71  ext::dynamic_pointer_cast<StrikedTypePayoff>(payoff_);
72  if (!striked_payoff)
73  return;
74  Real requiredGridValue = striked_payoff->strike();
75 
76  if(sMin_ > requiredGridValue/safetyZoneFactor_){
77  sMin_ = requiredGridValue/safetyZoneFactor_;
78  // enforce central placement of the underlying
79  sMax_ = center_/(sMin_/center_);
80  }
81  if(sMax_ < requiredGridValue*safetyZoneFactor_){
82  sMax_ = requiredGridValue*safetyZoneFactor_;
83  // enforce central placement of the underlying
84  sMin_ = center_/(sMax_/center_);
85  }
86  }

93  void FDVanillaEngine::initializeOperator() const {
94  if (timeDependent_) {
95  finiteDifferenceOperator_ = BSMTermOperator(intrinsicValues_.grid(),
96  process_, getResidualTime());
97  } else {
98  const YieldTermStructure& R = **(process_->riskFreeRate());
99  Rate r = R.zeroRate(exerciseDate_, R.dayCounter(), Continuous);
100  const YieldTermStructure& Q = **(process_->dividendYield());
101  Rate q = Q.zeroRate(exerciseDate_, Q.dayCounter(), Continuous);
102 
103  ext::shared_ptr<StrikedTypePayoff> striked_payoff =
104  ext::dynamic_pointer_cast<StrikedTypePayoff>(payoff_);
105  Real K = striked_payoff != nullptr ? striked_payoff->strike() : process_->x0();
106  Volatility sigma =
107  process_->blackVolatility()->blackVol(exerciseDate_, K);
108 
109  finiteDifferenceOperator_ = BSMOperator(intrinsicValues_.grid(),
110  r, q, sigma);
111  }
112  }

129  // safety check to be sure we have enough grid points.
130  Size FDVanillaEngine::safeGridPoints(Size gridPoints,
131  Time residualTime) const {
132  static const Size minGridPoints = 10;
133  static const Size minGridPointsPerYear = 2;
134  return std::max(gridPoints,
135  residualTime > 1.0 ?
136  static_cast<Size>((minGridPoints +
137  (residualTime-1.0) *
138  minGridPointsPerYear))
139  : minGridPoints);
140  }
