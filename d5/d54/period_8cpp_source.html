29  Period::Period(Frequency f) {
30  switch (f) {
31  case NoFrequency:
32  // same as Period()
33  units_ = Days;
34  length_ = 0;
35  break;
36  case Once:
37  units_ = Years;
38  length_ = 0;
39  break;
40  case Annual:
41  units_ = Years;
42  length_ = 1;
43  break;
44  case Semiannual:
45  case EveryFourthMonth:
46  case Quarterly:
47  case Bimonthly:
48  case Monthly:
49  units_ = Months;
50  length_ = 12/f;
51  break;
52  case EveryFourthWeek:
53  case Biweekly:
54  case Weekly:
55  units_ = Weeks;
56  length_ = 52/f;
57  break;
58  case Daily:
59  units_ = Days;
60  length_ = 1;
61  break;
62  case OtherFrequency:
63  QL_FAIL("unknown frequency"); // no point in showing 999...
64  default:
65  QL_FAIL("unknown frequency (" << Integer(f) << ")");
66  }

69  Frequency Period::frequency() const {
70  // unsigned version
71  Size length = std::abs(length_);
72 
73  if (length==0) {
74  if (units_==Years) return Once;
75  return NoFrequency;
76  }
77 
78  switch (units_) {
79  case Years:
80  if (length == 1)
81  return Annual;
82  else
83  return OtherFrequency;
84  case Months:
85  if (12%length == 0 && length <= 12)
86  return Frequency(12/length);
87  else
88  return OtherFrequency;
89  case Weeks:
90  if (length==1)
91  return Weekly;
92  else if (length==2)
93  return Biweekly;
94  else if (length==4)
95  return EveryFourthWeek;
96  else
97  return OtherFrequency;
98  case Days:
99  if (length==1)
100  return Daily;
101  else
102  return OtherFrequency;
103  default:
104  QL_FAIL("unknown time unit (" << Integer(units_) << ")");
105  }

108  void Period::normalize() {
109  if (length_!=0)
110  switch (units_) {
111  case Months:
112  if ((length_ % 12) == 0) {
113  length_ /= 12;
114  units_ = Years;
115  }
116  break;
117  case Days:
118  case Weeks:
119  case Years:
120  break;
121  default:
122  QL_FAIL("unknown time unit (" << Integer(units_) << ")");

126  Period& Period::operator+=(const Period& p) {
127 
128  if (length_==0) {
129  length_ = p.length();
130  units_ = p.units();
131  } else if (units_==p.units()) {
132  // no conversion needed
133  length_ += p.length();
134  } else {
135  switch (units_) {
136 
137  case Years:
138  switch (p.units()) {
139  case Months:
140  units_ = Months;
141  length_ = length_*12 + p.length();
142  break;
143  case Weeks:
144  case Days:
145  QL_REQUIRE(p.length()==0,
146  "impossible addition between " << *this <<
147  " and " << p);
148  break;
149  default:
150  QL_FAIL("unknown time unit (" << Integer(p.units()) << ")");
151  }
152  break;
153 
154  case Months:
155  switch (p.units()) {
156  case Years:
157  length_ += p.length()*12;
158  break;
159  case Weeks:
160  case Days:
161  QL_REQUIRE(p.length()==0,
162  "impossible addition between " << *this <<
163  " and " << p);
164  break;

166  QL_FAIL("unknown time unit (" << Integer(p.units()) << ")");

177  case Months:

184  }
185  break;

189  case Weeks:
190  length_ += p.length()*7;
191  break;
192  case Years:
193  case Months:
194  QL_REQUIRE(p.length()==0,
195  "impossible addition between " << *this <<
196  " and " << p);

200  }
201  break;

216  Period& Period::operator/=(Integer n) {
217  QL_REQUIRE(n != 0, "cannot be divided by zero");
218  if (length_ % n == 0) {
219  // keep the original units. If the user created a
220  // 24-months period, he'll probably want a 12-months one
221  // when he halves it.
222  length_ /= n;
223  } else {
224  // try
225  TimeUnit units = units_;
226  Integer length = length_;
227  switch (units) {
228  case Years:
229  length *= 12;
230  units = Months;
231  break;
232  case Weeks:
233  length *= 7;
234  units = Days;
235  break;
236  default:
237  ;
238  }
239  QL_REQUIRE(length % n == 0,
240  *this << " cannot be divided by " << n);
241  length_ = length/n;
242  units_ = units;
243  // if normalization were possible, we wouldn't be
244  // here---the "if" branch would have been executed
245  // instead.
246  // result.normalize();
247  }
248  return *this;
249  }

339  bool operator<(const Period& p1, const Period& p2) {
340 
341  // special cases
342  if (p1.length() == 0)
343  return p2.length() > 0;
344  if (p2.length() == 0)
345  return p1.length() < 0;
346 
347  // exact comparisons
348  if (p1.units() == p2.units())
349  return p1.length() < p2.length();
350  if (p1.units() == Months && p2.units() == Years)
351  return p1.length() < 12*p2.length();
352  if (p1.units() == Years && p2.units() == Months)
353  return 12*p1.length() < p2.length();
354  if (p1.units() == Days && p2.units() == Weeks)
355  return p1.length() < 7*p2.length();
356  if (p1.units() == Weeks && p2.units() == Days)
357  return 7*p1.length() < p2.length();
358 
359  // inexact comparisons (handled by converting to days and using limits)
360  std::pair<Integer, Integer> p1lim = daysMinMax(p1);
361  std::pair<Integer, Integer> p2lim = daysMinMax(p2);
362 
363  if (p1lim.second < p2lim.first)
364  return true;
365  else if (p1lim.first > p2lim.second)
366  return false;
367  else
368  QL_FAIL("undecidable comparison between " << p1 << " and " << p2);
369  }
