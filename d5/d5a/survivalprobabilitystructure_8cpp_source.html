46 Real SurvivalProbabilityStructure::defaultDensityImpl(Time t) const {
47 Time dt = 0.0001;
48 Time t1 = std::max(t-dt, 0.0);
49 Time t2 = t+dt;
50
51 Probability p1 = survivalProbabilityImpl(t1);
52 Probability p2 = survivalProbabilityImpl(t2);
53
54 return (p1-p2)/(t2-t1);
55 }
