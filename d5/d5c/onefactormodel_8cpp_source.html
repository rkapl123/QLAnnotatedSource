27  class OneFactorModel::ShortRateTree::Helper {

43  Real operator()(Real theta) const {
44  Real value = discountBondPrice_;
45  theta_->change(theta);
46  for (Size j=0; j<size_; j++)
47  value -= statePrices_[j]*tree_.discount(i_,j);
48  return value;
49  }

60  OneFactorModel::ShortRateTree::ShortRateTree(
61  const ext::shared_ptr<TrinomialTree>& tree,
62  const ext::shared_ptr<ShortRateDynamics>& dynamics,
63  const ext::shared_ptr
64  <TermStructureFittingParameter::NumericalImpl>& theta,
65  const TimeGrid& timeGrid)
66  : TreeLattice1D<OneFactorModel::ShortRateTree>(timeGrid, tree->size(1)),
67  tree_(tree), dynamics_(dynamics), spread_(0.0) {
68 
69  theta->reset();
70  Real value = 1.0;
71  Real vMin = -100.0;
72  Real vMax = 100.0;
73  for (Size i=0; i<(timeGrid.size() - 1); i++) {
74  Real discountBond = theta->termStructure()->discount(t_[i+1]);
75  Helper finder(i, discountBond, theta, *this);
76  Brent s1d;
77  s1d.setMaxEvaluations(1000);
78  value = s1d.solve(finder, 1e-7, value, vMin, vMax);
79  // vMin = value - 1.0;
80  // vMax = value + 1.0;
81  theta->change(value);
82  }
83  }
