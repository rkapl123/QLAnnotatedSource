39  EndCriteria::Type LevenbergMarquardt::minimize(Problem& P,
40  const EndCriteria& endCriteria) {
41  using namespace ext::placeholders;
42 
43  EndCriteria::Type ecType = EndCriteria::None;
44  P.reset();
45  Array x_ = P.currentValue();
46  currentProblem_ = &P;
47  initCostValues_ = P.costFunction().values(x_);
48  int m = initCostValues_.size();
49  int n = x_.size();
50  if(useCostFunctionsJacobian_) {
51  initJacobian_ = Matrix(m,n);
52  P.costFunction().jacobian(initJacobian_, x_);
53  }
54  boost::scoped_array<Real> xx(new Real[n]);
55  std::copy(x_.begin(), x_.end(), xx.get());
56  boost::scoped_array<Real> fvec(new Real[m]);
57  boost::scoped_array<Real> diag(new Real[n]);
58  int mode = 1;
59  Real factor = 1;
60  int nprint = 0;
61  int info = 0;
62  int nfev =0;
63  boost::scoped_array<Real> fjac(new Real[m*n]);
64  int ldfjac = m;
65  boost::scoped_array<int> ipvt(new int[n]);
66  boost::scoped_array<Real> qtf(new Real[n]);
67  boost::scoped_array<Real> wa1(new Real[n]);
68  boost::scoped_array<Real> wa2(new Real[n]);
69  boost::scoped_array<Real> wa3(new Real[n]);
70  boost::scoped_array<Real> wa4(new Real[m]);
71  // requirements; check here to get more detailed error messages.
72  QL_REQUIRE(n > 0, "no variables given");
73  QL_REQUIRE(m >= n,
74  "less functions (" << m <<
75  ") than available variables (" << n << ")");
76  QL_REQUIRE(endCriteria.functionEpsilon() >= 0.0,
77  "negative f tolerance");
78  QL_REQUIRE(xtol_ >= 0.0, "negative x tolerance");
79  QL_REQUIRE(gtol_ >= 0.0, "negative g tolerance");
80  QL_REQUIRE(endCriteria.maxIterations() > 0,
81  "null number of evaluations");
82 
83  // call lmdif to minimize the sum of the squares of m functions
84  // in n variables by the Levenberg-Marquardt algorithm.
85  MINPACK::LmdifCostFunction lmdifCostFunction =
86  ext::bind(&LevenbergMarquardt::fcn, this, _1, _2, _3, _4, _5);
87  MINPACK::LmdifCostFunction lmdifJacFunction =
88  useCostFunctionsJacobian_
89  ? ext::bind(&LevenbergMarquardt::jacFcn, this, _1, _2, _3,
90  _4, _5)
91  : MINPACK::LmdifCostFunction();
92  MINPACK::lmdif(m, n, xx.get(), fvec.get(),
93  endCriteria.functionEpsilon(),
94  xtol_,
95  gtol_,
96  endCriteria.maxIterations(),
97  epsfcn_,
98  diag.get(), mode, factor,
99  nprint, &info, &nfev, fjac.get(),
100  ldfjac, ipvt.get(), qtf.get(),
101  wa1.get(), wa2.get(), wa3.get(), wa4.get(),
102  lmdifCostFunction,
103  lmdifJacFunction);
104  info_ = info;
105  // check requirements & endCriteria evaluation
106  QL_REQUIRE(info != 0, "MINPACK: improper input parameters");
107  //QL_REQUIRE(info != 6, "MINPACK: ftol is too small. no further "
108  // "reduction in the sum of squares "
109  // "is possible.");
110  if (info != 6) ecType = QuantLib::EndCriteria::StationaryFunctionValue;
111  //QL_REQUIRE(info != 5, "MINPACK: number of calls to fcn has "
112  // "reached or exceeded maxfev.");
113  endCriteria.checkMaxIterations(nfev, ecType);
114  QL_REQUIRE(info != 7, "MINPACK: xtol is too small. no further "
115  "improvement in the approximate "
116  "solution x is possible.");
117  QL_REQUIRE(info != 8, "MINPACK: gtol is too small. fvec is "
118  "orthogonal to the columns of the "
119  "jacobian to machine precision.");
120  // set problem
121  std::copy(xx.get(), xx.get()+n, x_.begin());
122  P.setCurrentValue(x_);
123  P.setFunctionValue(P.costFunction().value(x_));
124 
125  return ecType;
126  }

128  void LevenbergMarquardt::fcn(int, int n, Real* x, Real* fvec, int*) {
129  Array xt(n);
130  std::copy(x, x+n, xt.begin());
131  // constraint handling needs some improvement in the future:
132  // starting point should not be close to a constraint violation
133  if (currentProblem_->constraint().test(xt)) {
134  const Array& tmp = currentProblem_->values(xt);
135  std::copy(tmp.begin(), tmp.end(), fvec);
136  } else {
137  std::copy(initCostValues_.begin(), initCostValues_.end(), fvec);
138  }
139  }

141  void LevenbergMarquardt::jacFcn(int m, int n, Real* x, Real* fjac, int*) {
142  Array xt(n);
143  std::copy(x, x+n, xt.begin());
144  // constraint handling needs some improvement in the future:
145  // starting point should not be close to a constraint violation
146  if (currentProblem_->constraint().test(xt)) {
147  Matrix tmp(m,n);
148  currentProblem_->costFunction().jacobian(tmp, xt);
149  Matrix tmpT = transpose(tmp);
150  std::copy(tmpT.begin(), tmpT.end(), fjac);
151  } else {
152  Matrix tmpT = transpose(initJacobian_);
153  std::copy(tmpT.begin(), tmpT.end(), fjac);
154  }
155  }
