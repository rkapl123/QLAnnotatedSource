52 MakeSwaption::operator ext::shared_ptr<Swaption>() const {
53
54 const Calendar& fixingCalendar = swapIndex_->fixingCalendar();
55 Date refDate = Settings::instance().evaluationDate();
56 // if the evaluation date is not a business day
57 // then move to the next business day
58 refDate = fixingCalendar.adjust(refDate);
59 if (fixingDate_ == Null<Date>())
60 fixingDate_ = fixingCalendar.advance(refDate, optionTenor_,
61 optionConvention_);
62 if (exerciseDate_ == Null<Date>()) {
63 exercise_ = ext::shared_ptr<Exercise>(new
64 EuropeanExercise(fixingDate_));
65 } else {
66 QL_REQUIRE(exerciseDate_ <= fixingDate_,
67 "exercise date (" << exerciseDate_ << ") must be less "
68 "than or equal to fixing date (" << fixingDate_ << ")");
69 exercise_ = ext::shared_ptr<Exercise>(new
70 EuropeanExercise(exerciseDate_));
71 }
72
73 Rate usedStrike = strike_;
74 if (strike_ == Null<Rate>()) {
75 // ATM on curve(s) attached to index
76 QL_REQUIRE(!swapIndex_->forwardingTermStructure().empty(),
77 "null term structure set to this instance of " <<
78 swapIndex_->name());
79 ext::shared_ptr<VanillaSwap> temp =
80 swapIndex_->underlyingSwap(fixingDate_);
81 temp->setPricingEngine(
82 ext::shared_ptr<PricingEngine>(new DiscountingSwapEngine(
83 swapIndex_->exogenousDiscount()
84 ? swapIndex_->discountingTermStructure()
85 : swapIndex_->forwardingTermStructure(),
86 false)));
87 usedStrike = temp->fairRate();
88 }
89
90 BusinessDayConvention bdc = swapIndex_->fixedLegConvention();
91 underlyingSwap_ =
92 MakeVanillaSwap(swapIndex_->tenor(),
93 swapIndex_->iborIndex(), usedStrike)
94 .withEffectiveDate(swapIndex_->valueDate(fixingDate_))
95 .withFixedLegCalendar(swapIndex_->fixingCalendar())
96 .withFixedLegDayCount(swapIndex_->dayCounter())
97 .withFixedLegTenor(swapIndex_->fixedLegTenor())
98 .withFixedLegConvention(bdc)
99 .withFixedLegTerminationDateConvention(bdc)
100 .withType(underlyingType_)
101 .withNominal(nominal_)
102 .withIndexedCoupons(useIndexedCoupons_);
103
104 ext::shared_ptr<Swaption> swaption(new Swaption(
105 underlyingSwap_, exercise_, delivery_, settlementMethod_));
106 swaption->setPricingEngine(engine_);
107 return swaption;
108 }
