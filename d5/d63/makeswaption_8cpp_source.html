57  MakeSwaption::operator boost::shared_ptr<Swaption>() const {
58 
59  const Calendar& fixingCalendar = swapIndex_->fixingCalendar();
60  Date refDate = Settings::instance().evaluationDate();
61  // if the evaluation date is not a business day
62  // then move to the next business day
63  refDate = fixingCalendar.adjust(refDate);
64  if (fixingDate_ == Null<Date>())
65  fixingDate_ = fixingCalendar.advance(refDate, optionTenor_,
66  optionConvention_);
67  if (exerciseDate_ == Null<Date>()) {
68  exercise_ = boost::shared_ptr<Exercise>(new
69  EuropeanExercise(fixingDate_));
70  } else {
71  QL_REQUIRE(exerciseDate_ <= fixingDate_,
72  "exercise date (" << exerciseDate_ << ") must be less "
73  "than or equal to fixing date (" << fixingDate_ << ")");
74  exercise_ = boost::shared_ptr<Exercise>(new
75  EuropeanExercise(exerciseDate_));
76  }
77 
78  Rate usedStrike = strike_;
79  if (strike_ == Null<Rate>()) {
80  // ATM on curve(s) attached to index
81  QL_REQUIRE(!swapIndex_->forwardingTermStructure().empty(),
82  "null term structure set to this instance of " <<
83  swapIndex_->name());
84  boost::shared_ptr<VanillaSwap> temp =
85  swapIndex_->underlyingSwap(fixingDate_);
86  temp->setPricingEngine(
87  boost::shared_ptr<PricingEngine>(new DiscountingSwapEngine(
88  swapIndex_->exogenousDiscount()
89  ? swapIndex_->discountingTermStructure()
90  : swapIndex_->forwardingTermStructure(),
91  false)));
92  usedStrike = temp->fairRate();
93  }
94 
95  BusinessDayConvention bdc = swapIndex_->fixedLegConvention();
96  underlyingSwap_ =
97  MakeVanillaSwap(swapIndex_->tenor(),
98  swapIndex_->iborIndex(), usedStrike)
99  .withEffectiveDate(swapIndex_->valueDate(fixingDate_))
100  .withFixedLegCalendar(swapIndex_->fixingCalendar())
101  .withFixedLegDayCount(swapIndex_->dayCounter())
102  .withFixedLegTenor(swapIndex_->fixedLegTenor())
103  .withFixedLegConvention(bdc)
104  .withFixedLegTerminationDateConvention(bdc)
105  .withType(underlyingType_);
106 
107  boost::shared_ptr<Swaption> swaption(new
108  Swaption(underlyingSwap_, exercise_, delivery_));
109  swaption->setPricingEngine(engine_);
110  return swaption;
111  }
