39 void DiscountingSwapEngine::calculate() const {
40 QL_REQUIRE(!discountCurve_.empty(),
41 "discounting term structure handle is empty");
42
43 results_.value = 0.0;
44 results_.errorEstimate = Null<Real>();
45
46 Date refDate = discountCurve_->referenceDate();
47
48 Date settlementDate = settlementDate_;
49 if (settlementDate_==Date()) {
50 settlementDate = refDate;
51 } else {
52 QL_REQUIRE(settlementDate>=refDate,
53 "settlement date (" << settlementDate << ") before "
54 "discount curve reference date (" << refDate << ")");
55 }
56
57 results_.valuationDate = npvDate_;
58 if (npvDate_==Date()) {
59 results_.valuationDate = refDate;
60 } else {
61 QL_REQUIRE(npvDate_>=refDate,
62 "npv date (" << npvDate_ << ") before "
63 "discount curve reference date (" << refDate << ")");
64 }
65 results_.npvDateDiscount = discountCurve_->discount(results_.valuationDate);
66
67 Size n = arguments_.legs.size();
68 results_.legNPV.resize(n);
69 results_.legBPS.resize(n);
70 results_.startDiscounts.resize(n);
71 results_.endDiscounts.resize(n);
72
73 bool includeRefDateFlows = includeSettlementDateFlows_ ? // NOLINT(readability-implicit-bool-conversion)
74 *includeSettlementDateFlows_ :
75 Settings::instance().includeReferenceDateEvents();
76
77 for (Size i=0; i<n; ++i) {
78 try {
79 const YieldTermStructure& discount_ref = **discountCurve_;
80 CashFlows::npvbps(arguments_.legs[i],
81 discount_ref,
82 includeRefDateFlows,
83 settlementDate,
84 results_.valuationDate,
85 results_.legNPV[i],
86 results_.legBPS[i]);
87 results_.legNPV[i] *= arguments_.payer[i];
88 results_.legBPS[i] *= arguments_.payer[i];
89
90 if (!arguments_.legs[i].empty()) {
91 Date d1 = CashFlows::startDate(arguments_.legs[i]);
92 if (d1>=refDate)
93 results_.startDiscounts[i] = discountCurve_->discount(d1);
94 else
95 results_.startDiscounts[i] = Null<DiscountFactor>();
96
97 Date d2 = CashFlows::maturityDate(arguments_.legs[i]);
98 if (d2>=refDate)
99 results_.endDiscounts[i] = discountCurve_->discount(d2);
100 else
101 results_.endDiscounts[i] = Null<DiscountFactor>();
102 } else {
103 results_.startDiscounts[i] = Null<DiscountFactor>();
104 results_.endDiscounts[i] = Null<DiscountFactor>();
105 }
106
107 } catch (std::exception &e) {
108 QL_FAIL(io::ordinal(i+1) << " leg: " << e.what());
109 }
110 results_.value += results_.legNPV[i];
111 }
112 }
