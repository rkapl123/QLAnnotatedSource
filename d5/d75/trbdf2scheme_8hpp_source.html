106  template <class TrapezoidalScheme>
107  inline void TrBDF2Scheme<TrapezoidalScheme>::step(array_type& fn, Time t) {
108  QL_REQUIRE(t-dt_ > -1e-8, "a step towards negative time given");
109 
110  const Time intermediateTimeStep = dt_*alpha_;
111 
112  array_type fStar = fn;
113  trapezoidalScheme_->setStep(intermediateTimeStep);
114  trapezoidalScheme_->step(fStar, t);
115 
116  bcSet_.setTime(std::max(0.0, t-dt_));
117  bcSet_.applyBeforeSolving(*map_, fn);
118 
119  const array_type f =
120  (1/alpha_*fStar - square<Real>()(1-alpha_)/alpha_*fn)/(2-alpha_);
121 
122  if (map_->size() == 1) {
123  fn = map_->solve_splitting(0, f, -beta_);
124  }
125  else {
126  auto preconditioner = [&](const Array& _a){ return map_->preconditioner(_a, -beta_); };
127  auto applyF = [&](const Array& _a){ return apply(_a); };
128 
129  if (solverType_ == BiCGstab) {
130  const BiCGStabResult result =
131  QuantLib::BiCGstab(applyF, std::max(Size(10), fn.size()),
132  relTol_, preconditioner).solve(f, f);
133 
134  (*iterations_) += result.iterations;
135  fn = result.x;
136  } else if (solverType_ == GMRES) {
137  const GMRESResult result =
138  QuantLib::GMRES(applyF, std::max(Size(10), fn.size() / 10U), relTol_,
139  preconditioner)
140  .solve(f, f);
141 
142  (*iterations_) += result.errors.size();
143  fn = result.x;
144  }
145  else
146  QL_FAIL("unknown/illegal solver type");
147  }
148 
149  bcSet_.applyAfterSolving(fn);
150  }
151 }
