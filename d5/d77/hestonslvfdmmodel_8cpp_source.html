57  ext::shared_ptr<Fdm1dMesher> varianceMesher(
58  const SquareRootProcessRNDCalculator& rnd,
59  Time t0, Time t1, Size vGrid,
60  Real v0, const HestonSLVFokkerPlanckFdmParams& params) {
61 
62  std::vector<ext::tuple<Real, Real, bool> > cPoints;
63 
64  const Real v0Density = params.v0Density;
65  const Real upperBoundDensity = params.vUpperBoundDensity;
66  const Real lowerBoundDensity = params.vLowerBoundDensity;
67 
68  Real lowerBound = Null<Real>(), upperBound = -Null<Real>();
69 
70  for (Size i=0; i <= 10; ++i) {
71  const Time t = t0 + i/10.0*(t1-t0);
72  lowerBound = std::min(
73  lowerBound, rnd.invcdf(params.vLowerEps, t));
74  upperBound = std::max(
75  upperBound, rnd.invcdf(1.0-params.vUpperEps, t));
76  }
77 
78  lowerBound = std::max(lowerBound, params.vMin);
79  switch (params.trafoType) {
80  case FdmSquareRootFwdOp::Log:
81  {
82  lowerBound = std::log(lowerBound);
83  upperBound = std::log(upperBound);
84 
85  const Real v0Center = std::log(v0);
86 
87  cPoints +=
88  ext::make_tuple(lowerBound, lowerBoundDensity, false),
89  ext::make_tuple(v0Center, v0Density, true),
90  ext::make_tuple(upperBound, upperBoundDensity, false);
91 
92  return ext::make_shared<Concentrating1dMesher>(
93  lowerBound, upperBound, vGrid, cPoints, 1e-8);
94  }
95  break;
96  case FdmSquareRootFwdOp::Plain:
97  {
98  const Real v0Center = v0;
99 
100  cPoints +=
101  ext::make_tuple(lowerBound, lowerBoundDensity, false),
102  ext::make_tuple(v0Center, v0Density, true),
103  ext::make_tuple(upperBound, upperBoundDensity, false);
104 
105  return ext::make_shared<Concentrating1dMesher>(
106  lowerBound, upperBound, vGrid, cPoints, 1e-8);
107  }
108  break;
109  case FdmSquareRootFwdOp::Power:
110  {
111  const Real v0Center = v0;
112 
113  cPoints +=
114  ext::make_tuple(lowerBound, lowerBoundDensity, false),
115  ext::make_tuple(v0Center, v0Density, true),
116  ext::make_tuple(upperBound, upperBoundDensity, false);
117 
118  return ext::make_shared<Concentrating1dMesher>(
119  lowerBound, upperBound, vGrid, cPoints, 1e-8);
120  }
121  break;
122  default:
123  QL_FAIL("transformation type is not implemented");
124  }
125  }

127  Real integratePDF(const Array& p,
128  const ext::shared_ptr<FdmMesherComposite>& mesher,
129  FdmSquareRootFwdOp::TransformationType trafoType,
130  Real alpha) {
131 
132  if (trafoType != FdmSquareRootFwdOp::Power) {
133  return FdmMesherIntegral(
134  mesher, DiscreteSimpsonIntegral()).integrate(p);
135  }
136  else {
137  Array tp(p.size());
138  const FdmLinearOpIterator end = mesher->layout()->end();
139  for (FdmLinearOpIterator iter = mesher->layout()->begin();
140  iter != end; ++iter) {
141  const Size idx = iter.index();
142  const Real nu = mesher->location(iter, 1);
143 
144  tp[idx] = p[idx]*std::pow(nu, alpha-1);
145  }
146 
147  return FdmMesherIntegral(
148  mesher, DiscreteSimpsonIntegral()).integrate(tp);
149  }
150  }

164  template <class Interpolator>
165  Disposable<Array> reshapePDF(
166  const Array& p,
167  const ext::shared_ptr<FdmMesherComposite>& oldMesher,
168  const ext::shared_ptr<FdmMesherComposite>& newMesher,
169  const Interpolator& interp = Interpolator()) {
170 
171  const ext::shared_ptr<FdmLinearOpLayout> oldLayout
172  = oldMesher->layout();
173  const ext::shared_ptr<FdmLinearOpLayout> newLayout
174  = newMesher->layout();
175 
176  QL_REQUIRE( oldLayout->size() == newLayout->size()
177  && oldLayout->size() == p.size(),
178  "inconsistent mesher or vector size given");
179 
180  Matrix m(oldLayout->dim()[1], oldLayout->dim()[0]);
181  for (Size i=0; i < m.rows(); ++i) {
182  std::copy(p.begin() + i*m.columns(),
183  p.begin() + (i+1)*m.columns(), m.row_begin(i));
184  }
185  const Interpolation2D interpol = interp.interpolate(
186  oldMesher->getFdm1dMeshers()[0]->locations().begin(),
187  oldMesher->getFdm1dMeshers()[0]->locations().end(),
188  oldMesher->getFdm1dMeshers()[1]->locations().begin(),
189  oldMesher->getFdm1dMeshers()[1]->locations().end(), m);
190 
191  Array pNew(p.size());
192  const FdmLinearOpIterator endIter = newLayout->end();
193  for (FdmLinearOpIterator iter = newLayout->begin();
194  iter != endIter; ++iter) {
195  const Real x = newMesher->location(iter, 0);
196  const Real v = newMesher->location(iter, 1);
197 
198  if ( x > interpol.xMax() || x < interpol.xMin()
199  || v > interpol.yMax() || v < interpol.yMin() ) {
200  pNew[iter.index()] = 0;
201  }
202  else {
203  pNew[iter.index()] = interpol(x, v);
204  }
205  }
206 
207  return pNew;
208  }

234  ext::shared_ptr<FdmScheme> fdmSchemeFactory(
235  const FdmSchemeDesc desc,
236  const ext::shared_ptr<FdmLinearOpComposite>& op) {
237 
238  switch (desc.type) {
239  case FdmSchemeDesc::HundsdorferType:
240  return ext::shared_ptr<FdmScheme>(
241  new FdmSchemeWrapper<HundsdorferScheme>(
242  new HundsdorferScheme(desc.theta, desc.mu, op)));
243  case FdmSchemeDesc::DouglasType:
244  return ext::shared_ptr<FdmScheme>(
245  new FdmSchemeWrapper<DouglasScheme>(
246  new DouglasScheme(desc.theta, op)));
247  case FdmSchemeDesc::CraigSneydType:
248  return ext::shared_ptr<FdmScheme>(
249  new FdmSchemeWrapper<CraigSneydScheme>(
250  new CraigSneydScheme(desc.theta, desc.mu, op)));
251  case FdmSchemeDesc::ModifiedCraigSneydType:
252  return ext::shared_ptr<FdmScheme>(
253  new FdmSchemeWrapper<ModifiedCraigSneydScheme>(
254  new ModifiedCraigSneydScheme(
255  desc.theta, desc.mu, op)));
256  case FdmSchemeDesc::ImplicitEulerType:
257  return ext::shared_ptr<FdmScheme>(
258  new FdmSchemeWrapper<ImplicitEulerScheme>(
259  new ImplicitEulerScheme(op)));
260  case FdmSchemeDesc::ExplicitEulerType:
261  return ext::shared_ptr<FdmScheme>(
262  new FdmSchemeWrapper<ExplicitEulerScheme>(
263  new ExplicitEulerScheme(op)));
264  default:
265  QL_FAIL("Unknown scheme type");
266  }
267  }

305  void HestonSLVFDMModel::performCalculations() const {
306  logEntries_.clear();
307 
308  const ext::shared_ptr<HestonProcess> hestonProcess
309  = hestonModel_->process();
310  const ext::shared_ptr<Quote> spot
311  = hestonProcess->s0().currentLink();
312  const ext::shared_ptr<YieldTermStructure> rTS
313  = hestonProcess->riskFreeRate().currentLink();
314  const ext::shared_ptr<YieldTermStructure> qTS
315  = hestonProcess->dividendYield().currentLink();
316 
317  const Real v0 = hestonProcess->v0();
318  const Real kappa = hestonProcess->kappa();
319  const Real theta = hestonProcess->theta();
320  const Real sigma = hestonProcess->sigma();
321  const Real mixedSigma = mixingFactor_ * sigma;
322  const Real alpha = 2*kappa*theta/(mixedSigma*mixedSigma);
323 
324  const Size xGrid = params_.xGrid;
325  const Size vGrid = params_.vGrid;
326 
327  const DayCounter dc = rTS->dayCounter();
328  const Date referenceDate = rTS->referenceDate();
329 
330  const Time T = dc.yearFraction(referenceDate, endDate_);
331 
332  QL_REQUIRE(referenceDate < endDate_,
333  "reference date must be smaller than final calibration date");
334 
335  QL_REQUIRE(localVol_->maxTime() >= T,
336  "final calibration maturity exceeds local volatility surface");
337 
338  // set-up exponential time step scheme
339  const Time maxDt = 1.0/params_.tMaxStepsPerYear;
340  const Time minDt = 1.0/params_.tMinStepsPerYear;
341 
342  Time tIdx=0.0;
343  std::vector<Time> times(1, tIdx);
344  times.reserve(Size(T*params_.tMinStepsPerYear));
345  while (tIdx < T) {
346  const Real decayFactor = std::exp(-params_.tStepNumberDecay*tIdx);
347  const Time dt = maxDt*decayFactor + minDt*(1.0-decayFactor);
348 
349  times.push_back(std::min(T, tIdx+=dt));
350  }
351 
352  for (Size i=0; i < mandatoryDates_.size(); ++i) {
353  times.push_back(
354  dc.yearFraction(referenceDate, mandatoryDates_[i]));
355  }
356 
357  const ext::shared_ptr<TimeGrid> timeGrid(
358  new TimeGrid(times.begin(), times.end()));
359 
360  // build 1d meshers
361  const LocalVolRNDCalculator localVolRND(
362  spot, rTS, qTS, localVol_.currentLink(),
363  timeGrid, xGrid,
364  params_.x0Density,
365  params_.localVolEpsProb,
366  params_.maxIntegrationIterations);
367 
368  const std::vector<Size> rescaleSteps
369  = localVolRND.rescaleTimeSteps();
370 
371  const SquareRootProcessRNDCalculator squareRootRnd(
372  v0, kappa, theta, mixedSigma);
373 
374  const FdmSquareRootFwdOp::TransformationType trafoType
375  = params_.trafoType;
376 
377  std::vector<ext::shared_ptr<Fdm1dMesher> > xMesher, vMesher;
378  xMesher.reserve(timeGrid->size());
379  vMesher.reserve(timeGrid->size());
380 
381  xMesher.push_back(localVolRND.mesher(0.0));
382  vMesher.push_back(ext::make_shared<Predefined1dMesher>(
383  std::vector<Real>(vGrid, v0)));
384 
385  Size rescaleIdx = 0;
386  for (Size i=1; i < timeGrid->size(); ++i) {
387  xMesher.push_back(localVolRND.mesher(timeGrid->at(i)));
388 
389  if ((rescaleIdx < rescaleSteps.size())
390  && (i == rescaleSteps[rescaleIdx])) {
391  ++rescaleIdx;
392  vMesher.push_back(varianceMesher(squareRootRnd,
393  timeGrid->at(rescaleSteps[rescaleIdx-1]),
394  (rescaleIdx < rescaleSteps.size())
395  ? timeGrid->at(rescaleSteps[rescaleIdx])
396  : timeGrid->back(),
397  vGrid, v0, params_));
398  }
399  else
400  vMesher.push_back(vMesher.back());
401  }
402 
403  // start probability distribution
404  ext::shared_ptr<FdmMesherComposite> mesher
405  = ext::make_shared<FdmMesherComposite>(
406  xMesher.at(1), vMesher.at(1));
407 
408  const Volatility lv0
409  = localVol_->localVol(0.0, spot->value())/std::sqrt(v0);
410 
411  ext::shared_ptr<Matrix> L(new Matrix(xGrid, timeGrid->size()));
412 
413  const Real l0 = lv0;
414  std::fill(L->column_begin(0),L->column_end(0), l0);
415  std::fill(L->column_begin(1),L->column_end(1), l0);
416 
417  // create strikes from meshers
418  std::vector<ext::shared_ptr<std::vector<Real> > > vStrikes(
419  timeGrid->size());
420 
421  for (Size i=0; i < timeGrid->size(); ++i) {
422  vStrikes[i] = ext::make_shared<std::vector<Real> >(xGrid);
423  if (xMesher[i]->locations().front()
424  == xMesher[i]->locations().back()) {
425  std::fill(vStrikes[i]->begin(), vStrikes[i]->end(),
426  std::exp(xMesher[i]->locations().front()));
427  }
428  else {
429  std::transform(xMesher[i]->locations().begin(),
430  xMesher[i]->locations().end(),
431  vStrikes[i]->begin(),
432  static_cast<Real(*)(Real)>(std::exp));
433  }
434  }
435 
436  const ext::shared_ptr<FixedLocalVolSurface> leverageFct(
437  new FixedLocalVolSurface(referenceDate, times, vStrikes, L, dc));
438 
439  ext::shared_ptr<FdmLinearOpComposite> hestonFwdOp(
440  new FdmHestonFwdOp(mesher, hestonProcess, trafoType, leverageFct, mixingFactor_));
441 
442  Array p = FdmHestonGreensFct(mesher, hestonProcess, trafoType, lv0)
443  .get(timeGrid->at(1), params_.greensAlgorithm);
444 
445  if (logging_) {
446  const LogEntry entry = { timeGrid->at(1),
447  ext::make_shared<Array>(p), mesher };
448  logEntries_.push_back(entry);
449  }
450 
451  for (Size i=2; i < times.size(); ++i) {
452  const Time t = timeGrid->at(i);
453  const Time dt = t - timeGrid->at(i-1);
454 
455  if ( mesher->getFdm1dMeshers()[0] != xMesher[i]
456  || mesher->getFdm1dMeshers()[1] != vMesher[i]) {
457  const ext::shared_ptr<FdmMesherComposite> newMesher(
458  new FdmMesherComposite(xMesher[i], vMesher[i]));
459 
460  p = reshapePDF<Bilinear>(p, mesher, newMesher);
461  mesher = newMesher;
462 
463  p = rescalePDF(p, mesher, trafoType, alpha);
464 
465  hestonFwdOp = ext::shared_ptr<FdmLinearOpComposite>(
466  new FdmHestonFwdOp(mesher, hestonProcess,
467  trafoType, leverageFct));
468  }
469 
470  Array pn = p;
471  const Array x(Exp(
472  Array(mesher->getFdm1dMeshers()[0]->locations().begin(),
473  mesher->getFdm1dMeshers()[0]->locations().end())));
474  const Array v(
475  mesher->getFdm1dMeshers()[1]->locations().begin(),
476  mesher->getFdm1dMeshers()[1]->locations().end());
477 
478  // predictor corrector steps
479  for (Size r=0; r < params_.predictionCorretionSteps; ++r) {
480  const FdmSchemeDesc fdmSchemeDesc
481  = (i < params_.nRannacherTimeSteps + 2)
482  ? FdmSchemeDesc::ImplicitEuler()
483  : params_.schemeDesc;
484 
485  const ext::shared_ptr<FdmScheme> fdmScheme(
486  fdmSchemeFactory(fdmSchemeDesc, hestonFwdOp));
487 
488  for (Size j=0; j < x.size(); ++j) {
489  Array pSlice(vGrid);
490  for (Size k=0; k < vGrid; ++k)
491  pSlice[k] = pn[j + k*xGrid];
492 
493  const Real pInt = (trafoType == FdmSquareRootFwdOp::Power)
494  ? DiscreteSimpsonIntegral()(v, Pow(v, alpha-1)*pSlice)
495  : DiscreteSimpsonIntegral()(v, pSlice);
496 
497  const Real vpInt = (trafoType == FdmSquareRootFwdOp::Log)
498  ? DiscreteSimpsonIntegral()(v, Exp(v)*pSlice)
499  : (trafoType == FdmSquareRootFwdOp::Power)
500  ? DiscreteSimpsonIntegral()(v, Pow(v, alpha)*pSlice)
501  : DiscreteSimpsonIntegral()(v, v*pSlice);
502 
503  const Real scale = pInt/vpInt;
504  const Volatility localVol = localVol_->localVol(t, x[j]);
505 
506  const Real l = (scale >= 0.0)
507  ? localVol*std::sqrt(scale) : 1.0;
508 
509  (*L)[j][i] = std::min(50.0, std::max(0.001, l));
510 
511  leverageFct->setInterpolation(Linear());
512  }
513 
514  const Real sLowerBound = std::max(x.front(),
515  std::exp(localVolRND.invcdf(
516  params_.leverageFctPropEps, t)));
517  const Real sUpperBound = std::min(x.back(),
518  std::exp(localVolRND.invcdf(
519  1.0-params_.leverageFctPropEps, t)));
520 
521  const Real lowerL = leverageFct->localVol(t, sLowerBound);
522  const Real upperL = leverageFct->localVol(t, sUpperBound);
523 
524  for (Size j=0; j < x.size(); ++j) {
525  if (x[j] < sLowerBound)
526  std::fill(L->row_begin(j)+i,
527  std::min(L->row_begin(j)+i+1, L->row_end(j)),
528  lowerL);
529  else if (x[j] > sUpperBound)
530  std::fill(L->row_begin(j)+i,
531  std::min(L->row_begin(j)+i+1, L->row_end(j)),
532  upperL);
533  else if ((*L)[j][i] == Null<Real>())
534  QL_FAIL("internal error");
535  }
536  leverageFct->setInterpolation(Linear());
537 
538  pn = p;
539 
540  fdmScheme->setStep(dt);
541  fdmScheme->step(pn, t);
542  }
543  p = pn;
544  p = rescalePDF(p, mesher, trafoType, alpha);
545 
546  if (logging_) {
547  const LogEntry entry
548  = { t, ext::make_shared<Array>(p), mesher };
549  logEntries_.push_back(entry);
550  }
551  }
552 
553  leverageFunction_ = leverageFct;
554  }
