25  void CappedFlooredYoYInflationCoupon::setCommon(
26  Rate cap, Rate floor) {
27 
28  isCapped_ = false;
29  isFloored_ = false;
30 
31  if (gearing_ > 0) {
32  if (cap != Null<Rate>()) {
33  isCapped_ = true;
34  cap_ = cap;
35  }
36  if (floor != Null<Rate>()) {
37  floor_ = floor;
38  isFloored_ = true;
39  }
40  } else {
41  if (cap != Null<Rate>()){
42  floor_ = cap;
43  isFloored_ = true;
44  }
45  if (floor != Null<Rate>()){
46  isCapped_ = true;
47  cap_ = floor;
48  }
49  }
50  if (isCapped_ && isFloored_) {
51  QL_REQUIRE(cap >= floor, "cap level (" << cap <<
52  ") less than floor level (" << floor << ")");
53  }
54  }

88  Rate CappedFlooredYoYInflationCoupon::rate() const {
89  Rate swapletRate = underlying_ != 0 ? underlying_->rate() : YoYInflationCoupon::rate();
90 
91  if(isFloored_ || isCapped_) {
92  if (underlying_ != 0) {
93  QL_REQUIRE(underlying_->pricer(), "pricer not set");
94  } else {
95  QL_REQUIRE(pricer_, "pricer not set");
96  }
97  }
98 
99  Rate floorletRate = 0.;
100  if(isFloored_) {
101  floorletRate = underlying_ != 0 ?
102  underlying_->pricer()->floorletRate(effectiveFloor()) :
103  pricer()->floorletRate(effectiveFloor());
104  }
105  Rate capletRate = 0.;
106  if(isCapped_) {
107  capletRate = underlying_ != 0 ? underlying_->pricer()->capletRate(effectiveCap()) :
108  pricer()->capletRate(effectiveCap());
109  }
110 
111  return swapletRate + floorletRate - capletRate;
112  }
