41  void BlackIborCouponPricer::initialize(const FloatingRateCoupon& coupon) {
42 
43  gearing_ = coupon.gearing();
44  spread_ = coupon.spread();
45  accrualPeriod_ = coupon.accrualPeriod();
46  QL_REQUIRE(accrualPeriod_ != 0.0, "null accrual period");
47 
48  index_ = ext::dynamic_pointer_cast<IborIndex>(coupon.index());
49  if (!index_) {
50  // check if the coupon was right
51  const IborCoupon* c = dynamic_cast<const IborCoupon*>(&coupon);
52  QL_REQUIRE(c, "IborCoupon required");
53  // coupon was right, index is not
54  QL_FAIL("IborIndex required");
55  }
56  Handle<YieldTermStructure> rateCurve =
57  index_->forwardingTermStructure();
58 
59  Date paymentDate = coupon.date();
60  if (paymentDate > rateCurve->referenceDate())
61  discount_ = rateCurve->discount(paymentDate);
62  else
63  discount_ = 1.0;
64 
65  spreadLegValue_ = spread_ * accrualPeriod_ * discount_;
66 
67  coupon_ = &coupon;
68  }

70  Real BlackIborCouponPricer::optionletPrice(Option::Type optionType,
71  Real effStrike) const {
72  Date fixingDate = coupon_->fixingDate();
73  if (fixingDate <= Settings::instance().evaluationDate()) {
74  // the amount is determined
75  Real a, b;
76  if (optionType==Option::Call) {
77  a = coupon_->indexFixing();
78  b = effStrike;
79  } else {
80  a = effStrike;
81  b = coupon_->indexFixing();
82  }
83  return std::max(a - b, 0.0)* accrualPeriod_*discount_;
84  } else {
85  // not yet determined, use Black model
86  QL_REQUIRE(!capletVolatility().empty(),
87  "missing optionlet volatility");
88  Real stdDev =
89  std::sqrt(capletVolatility()->blackVariance(fixingDate,
90  effStrike));
91  Real shift = capletVolatility()->displacement();
92  bool shiftedLn =
93  capletVolatility()->volatilityType() == ShiftedLognormal;
94  Rate fixing =
95  shiftedLn
96  ? blackFormula(optionType, effStrike, adjustedFixing(),
97  stdDev, 1.0, shift)
98  : bachelierBlackFormula(optionType, effStrike,
99  adjustedFixing(), stdDev, 1.0);
100  return fixing * accrualPeriod_ * discount_;
101  }
102  }

104  Rate BlackIborCouponPricer::adjustedFixing(Rate fixing) const {
105 
106  if (fixing == Null<Rate>())
107  fixing = coupon_->indexFixing();
108 
109  // if the pay date is equal to the index estimation end date
110  // there is no convexity; in all other cases in principle an
111  // adjustment has to be applied, but the Black76 method only
112  // applies the standard in arrears adjustment; the bivariate
113  // lognormal method is more accurate in this regard.
114  if ((!coupon_->isInArrears() && timingAdjustment_ == Black76))
115  return fixing;
116  Date d1 = coupon_->fixingDate();
117  Date d2 = index_->valueDate(d1);
118  Date d3 = index_->maturityDate(d2);
119  if (coupon_->date() == d3)
120  return fixing;
121 
122  QL_REQUIRE(!capletVolatility().empty(),
123  "missing optionlet volatility");
124  Date referenceDate = capletVolatility()->referenceDate();
125  // no variance has accumulated, so the convexity is zero
126  if (d1 <= referenceDate)
127  return fixing;
128  Time tau = index_->dayCounter().yearFraction(d2, d3);
129  Real variance = capletVolatility()->blackVariance(d1, fixing);
130 
131  Real shift = capletVolatility()->displacement();
132  bool shiftedLn =
133  capletVolatility()->volatilityType() == ShiftedLognormal;
134 
135  Spread adjustment = shiftedLn
136  ? (fixing + shift) * (fixing + shift) *
137  variance * tau / (1.0 + fixing * tau)
138  : variance * tau / (1.0 + fixing * tau);
139 
140  if (timingAdjustment_ == BivariateLognormal) {
141  QL_REQUIRE(!correlation_.empty(), "no correlation given");
142  Date d4 = coupon_->date();
143  Date d5 = d4 >= d3 ? d3 : d2;
144  Time tau2 = index_->dayCounter().yearFraction(d5, d4);
145  if (d4 >= d3)
146  adjustment = 0.0;
147  // if d4 < d2 (payment before index start) we just apply the
148  // Black76 in arrears adjustment
149  if (tau2 > 0.0) {
150  Real fixing2 =
151  (index_->forwardingTermStructure()->discount(d5) /
152  index_->forwardingTermStructure()->discount(d4) -
153  1.0) /
154  tau2;
155  adjustment -= shiftedLn
156  ? correlation_->value() * tau2 * variance *
157  (fixing + shift) * (fixing2 + shift) /
158  (1.0 + fixing2 * tau2)
159  : correlation_->value() * tau2 * variance /
160  (1.0 + fixing2 * tau2);
161  }
162  }
163  return fixing + adjustment;
164  }

225  // we might end up here because a CappedFlooredCoupon

236  "pricer not compatible with CMS spread Coupon");

329  void setCouponPricersFirstMatching(const Leg& leg,
330  const std::vector<ext::shared_ptr<FloatingRateCouponPricer> >& p) {
331  std::vector<PricerSetter> setter;
332  for (Size i = 0; i < p.size(); ++i) {
333  setter.push_back(PricerSetter(p[i]));
334  }
335  for (Size i = 0; i < leg.size(); ++i) {
336  Size j = 0;
337  do {
338  try {
339  leg[i]->accept(setter[j]);
340  j = p.size();
341  } catch (...) {
342  ++j;
343  }
344  } while (j < p.size());
345  }
346  }

357  void setCouponPricers(
358  const Leg& leg,
359  const std::vector<ext::shared_ptr<FloatingRateCouponPricer> >&
360  pricers) {
361  Size nCashFlows = leg.size();
362  QL_REQUIRE(nCashFlows>0, "no cashflows");
363 
364  Size nPricers = pricers.size();
365  QL_REQUIRE(nCashFlows >= nPricers,
366  "mismatch between leg size (" << nCashFlows <<
367  ") and number of pricers (" << nPricers << ")");
368 
369  for (Size i=0; i<nCashFlows; ++i) {
370  PricerSetter setter(i<nPricers ? pricers[i] : pricers[nPricers-1]);
371  leg[i]->accept(setter);
372  }
373  }
