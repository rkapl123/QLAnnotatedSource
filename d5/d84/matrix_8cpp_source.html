65  Disposable<Matrix> inverse(const Matrix& m) {
66  #if !defined(QL_NO_UBLAS_SUPPORT)
67 
68  QL_REQUIRE(m.rows() == m.columns(), "matrix is not square");
69 
70  boost::numeric::ublas::matrix<Real> a(m.rows(), m.columns());
71 
72  std::copy(m.begin(), m.end(), a.data().begin());
73 
74  boost::numeric::ublas::permutation_matrix<Size> pert(m.rows());
75 
76  // lu decomposition
77  Size singular = 1;
78  try {
79  singular = lu_factorize(a, pert);
80  } catch (const boost::numeric::ublas::internal_logic& e) {
81  QL_FAIL("lu_factorize error: " << e.what());
82  } catch (const boost::numeric::ublas::external_logic& e) {
83  QL_FAIL("lu_factorize error: " << e.what());
84  }
85  QL_REQUIRE(singular == 0, "singular matrix given");
86 
87  boost::numeric::ublas::matrix<Real>
88  inverse = boost::numeric::ublas::identity_matrix<Real>(m.rows());
89 
90  // backsubstitution
91  try {
92  boost::numeric::ublas::lu_substitute(a, pert, inverse);
93  } catch (const boost::numeric::ublas::internal_logic& e) {
94  QL_FAIL("lu_substitute error: " << e.what());
95  }
96 
97  Matrix retVal(m.rows(), m.columns());
98  std::copy(inverse.data().begin(), inverse.data().end(),
99  retVal.begin());
100 
101  return retVal;
102 
103  #else
104  QL_FAIL("this version of gcc does not support "
105  "the Boost uBLAS library");
106  #endif
107  }

109  Real determinant(const Matrix& m) {
110  #if !defined(QL_NO_UBLAS_SUPPORT)
111  QL_REQUIRE(m.rows() == m.columns(), "matrix is not square");
112 
113  boost::numeric::ublas::matrix<Real> a(m.rows(), m.columns());
114  std::copy(m.begin(), m.end(), a.data().begin());
115 
116 
117  // lu decomposition
118  boost::numeric::ublas::permutation_matrix<Size> pert(m.rows());
119  /* const Size singular = */ lu_factorize(a, pert);
120 
121  Real retVal = 1.0;
122 
123  for (Size i=0; i < m.rows(); ++i) {
124  if (pert[i] != i)
125  retVal *= -a(i,i);
126  else
127  retVal *= a(i,i);
128  }
129  return retVal;
130 
131  #else
