28  Disposable<Matrix>
29  ForwardForwardMappings::ForwardForwardJacobian(const CurveState& cs,
30  Size multiplier,
31  Size offset)
32  {
33 
34  Size n = cs.numberOfRates();
35 
36  QL_REQUIRE(offset < multiplier, "offset must be less than period in"
37  " forward forward mappings");
38  Size k = (n-offset)/multiplier;
39 
40  const std::vector<Time>& tau = cs.rateTaus();
41 
42  Matrix jacobian = Matrix(k, n, 0.0);
43 
44  Size m=offset;
45  for (Size l=0; l < k; ++l)
46  {
47  Real df = cs.discountRatio(m,m+multiplier);
48  Real bigTau = cs.rateTimes()[m+multiplier]
49  - cs.rateTimes()[m];
50 
51  for (Size r=0; r < multiplier; ++r, ++m)
52  {
53  Real value = df * tau[m]*cs.discountRatio(m+1,m)-1;
54  value /= bigTau;
55  jacobian[l][m]=-value;
56 
57  }
58  }
59 
60  return jacobian;
61  }

90
