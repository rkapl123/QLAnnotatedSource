27  TCopulaPolicy::TCopulaPolicy(
28  const std::vector<std::vector<Real> >& factorWeights,
29  const initTraits& vals)
30  {
31  for(Size iFactor=0; iFactor<vals.tOrders.size(); iFactor++) {
32  // require no T is of order 2 (finite variance)
33  QL_REQUIRE(vals.tOrders[iFactor] > 2,
34  "Non finite variance T in latent model.");
35 
36  distributions_.push_back(boost::math::students_t_distribution<>(
37  vals.tOrders[iFactor]));
38  // inverses T variaces used in normalization of the random factors
39  // For low values of the T order this number is very close to zero
40  // and it enters the expressions dividing them, which introduces
41  // numerical errors.
42  varianceFactors_.push_back(std::sqrt(
43  (vals.tOrders[iFactor]-2.)/vals.tOrders[iFactor]));
44  }
45 
46  for(Size iLVar=0; iLVar<factorWeights.size(); iLVar++) {
47  // This ensures the latent model is 'canonical'
48  QL_REQUIRE(vals.tOrders.size() == factorWeights[iLVar].size()+1,
49  // num factors plus one
50  "Incompatible number of T functions and number of factors.");
51 
52  Real factorsNorm = std::inner_product(factorWeights[iLVar].begin(),
53  factorWeights[iLVar].end(), factorWeights[iLVar].begin(), 0.);
54  QL_REQUIRE(factorsNorm < 1.,
55  "Non normal random factor combination.");
56  Real idiosyncFctr = std::sqrt(1.-factorsNorm);
57 
58  // linear comb factors ajusted for the variance renormalization:
59  std::vector<Real> normFactorWeights;
60  for(Size iFactor=0; iFactor<factorWeights[iLVar].size(); iFactor++)
61  normFactorWeights.push_back(factorWeights[iLVar][iFactor] *
62  varianceFactors_[iFactor]);
63  // idiosincratic term, all Z factors are assumed identical.
64  normFactorWeights.push_back(idiosyncFctr * varianceFactors_.back());
65  latentVarsCumul_.push_back(
66  CumulativeBehrensFisher(vals.tOrders, normFactorWeights));
67  latentVarsInverters_.push_back(
68  InverseCumulativeBehrensFisher(vals.tOrders,
69  normFactorWeights));
70  }
71  }

73  Disposable<std::vector<Real> >
74  TCopulaPolicy::allFactorCumulInverter(
75  const std::vector<Real>& probs) const
76  {
77  #if defined(QL_EXTRA_SAFETY_CHECKS)
78  QL_REQUIRE(probs.size()-latentVarsCumul_.size()
79  == distributions_.size()-1,
80  "Incompatible sample and latent model sizes");
81  #endif
82 
83  using namespace ext::placeholders;
84 
85  std::vector<Real> result(probs.size());
86  Size indexSystemic = 0;
87  std::transform(probs.begin(), probs.begin() + varianceFactors_.size()-1,
88  result.begin(),
89  ext::bind(&TCopulaPolicy::inverseCumulativeDensity,
90  this, _1, indexSystemic++));
91  std::transform(probs.begin() + varianceFactors_.size()-1, probs.end(),
92  result.begin()+ varianceFactors_.size()-1,
93  ext::bind(&TCopulaPolicy::inverseCumulativeZ, this, _1));
94  return result;
95  }
