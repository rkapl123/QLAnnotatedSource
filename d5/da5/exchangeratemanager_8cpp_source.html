51  ExchangeRate ExchangeRateManager::lookup(const Currency& source,
52  const Currency& target,
53  Date date,
54  ExchangeRate::Type type) const {
55 
56  if (source == target)
57  return ExchangeRate(source,target,1.0);
58 
59  if (date == Date())
60  date = Settings::instance().evaluationDate();
61 
62  if (type == ExchangeRate::Direct) {
63  return directLookup(source,target,date);
64  } else if (!source.triangulationCurrency().empty()) {
65  const Currency& link = source.triangulationCurrency();
66  if (link == target)
67  return directLookup(source,link,date);
68  else
69  return ExchangeRate::chain(directLookup(source,link,date),
70  lookup(link,target,date));
71  } else if (!target.triangulationCurrency().empty()) {
72  const Currency& link = target.triangulationCurrency();
73  if (source == link)
74  return directLookup(link,target,date);
75  else
76  return ExchangeRate::chain(lookup(source,link,date),
77  directLookup(link,target,date));
78  } else {
79  return smartLookup(source,target,date);
80  }
81  }

147  ExchangeRate ExchangeRateManager::smartLookup(
148  const Currency& source,
149  const Currency& target,
150  const Date& date,
151  std::list<Integer> forbidden) const {
152  // direct exchange rates are preferred.
153  if (const ExchangeRate* direct = fetch(source,target,date))
154  return *direct;
155 
156  // if none is found, turn to smart lookup. The source currency
157  // is forbidden to subsequent lookups in order to avoid cycles.
158  forbidden.push_back(source.numericCode());
159  std::map<Key, std::list<Entry> >::const_iterator i;
160  for (i = data_.begin(); i != data_.end(); ++i) {
161  // we look for exchange-rate data which involve our source
162  // currency...
163  if (hashes(i->first, source) && !(i->second.empty())) {
164  // ...whose other currency is not forbidden...
165  const Entry& e = i->second.front();
166  const Currency& other =
167  source == e.rate.source() ?
168  e.rate.target() : e.rate.source();
169  if (std::find(forbidden.begin(),forbidden.end(),
170  other.numericCode()) == forbidden.end()) {
171  // ...and which carries information for the requested date.
172  if (const ExchangeRate* head = fetch(source,other,date)) {
173  // if we can get to the target from here...
174  try {
175  ExchangeRate tail = smartLookup(other,target,date,
176  forbidden);
177  // ..we're done.
178  return ExchangeRate::chain(*head,tail);
179  } catch (Error&) {
180  // otherwise, we just discard this rate.
181  ;
182  }
183  }
184  }
185  }
186  }
187  // if the loop completed, we have no way to return the requested rate.
188  QL_FAIL("no conversion available from "
189  << source.code() << " to " << target.code()
190  << " for " << date);
191  }

193  const ExchangeRate* ExchangeRateManager::fetch(const Currency& source,
194  const Currency& target,
195  const Date& date) const {
196  const std::list<Entry>& rates = data_[hash(source,target)];
197  std::list<Entry>::const_iterator i =
198  std::find_if(rates.begin(), rates.end(), valid_at(date));
199  return i == rates.end() ?
200  (const ExchangeRate*) 0 :
201  &(i->rate);
202  }
