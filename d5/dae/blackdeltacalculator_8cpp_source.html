54  Real BlackDeltaCalculator::deltaFromStrike(Real strike) const {
55 
56  QL_REQUIRE(strike >=0.0,
57  "positive strike value required: " <<
58  strike << " not allowed");
59 
60  Real res=0.0;
61 
62  switch(dt_){
63  case DeltaVolQuote::Spot:
64  res=phi_*fDiscount_*cumD1(strike);
65  break;
66 
67  case DeltaVolQuote::Fwd:
68  res=phi_*cumD1(strike);
69  break;
70 
71  case DeltaVolQuote::PaSpot:
72  res=phi_*fDiscount_*cumD2(strike)*strike/forward_;
73  break;
74 
75  case DeltaVolQuote::PaFwd:
76  res=phi_*cumD2(strike)*strike/forward_;
77  break;
78 
79  default:
80  QL_FAIL("invalid delta type");
81  }
82  return res;
83  }

89  Real BlackDeltaCalculator::strikeFromDelta(Real delta,
90  DeltaVolQuote::DeltaType dt)
91  const{
92  Real res=0.0;
93  Real arg=0.0;
94  InverseCumulativeNormal f;
95 
96  QL_REQUIRE(delta*phi_>=0.0, "Option type and delta are incoherent.");
97 
98  switch (dt) {
99  case DeltaVolQuote::Spot:
100  QL_REQUIRE(std::fabs(delta)<=fDiscount_,
101  "Spot delta out of range.");
102 
103  arg=-phi_*f(phi_*delta/fDiscount_)*stdDev_+0.5*stdDev_*stdDev_;
104  res=forward_*std::exp(arg);
105  break;
106 
107  case DeltaVolQuote::Fwd:
108  QL_REQUIRE(std::fabs(delta)<=1.0,
109  "Forward delta out of range.");
110 
111  arg=-phi_*f(phi_*delta)*stdDev_+0.5*stdDev_*stdDev_;
112  res=forward_*std::exp(arg);
113  break;
114 
115  case DeltaVolQuote::PaSpot:
116  case DeltaVolQuote::PaFwd: {
117  // This has to be solved numerically. One of the
118  // problems is that the premium adjusted call delta is
119  // not monotonic in strike, such that two solutions
120  // might occur. The one right to the max of the delta is
121  // considered to be the correct strike. Some proper
122  // interval bounds for the strike need to be chosen, the
123  // numerics can otherwise be very unreliable and
124  // unstable. I've chosen Brent over Newton, since the
125  // interval can be specified explicitly and we can not
126  // run into the area on the left of the maximum. The
127  // put delta doesn't have this property and can be
128  // solved without any problems, but also numerically.
129 
130  BlackDeltaPremiumAdjustedSolverClass f(
131  ot_, dt , spot_,dDiscount_, fDiscount_, stdDev_,delta);
132 
133  Brent solver;
134  solver.setMaxEvaluations(1000);
135  Real accuracy = 1.0e-10;
136 
137  Real rightLimit=0.0;
138  Real leftLimit=0.0;
139 
140  // Strike of not premium adjusted is always to the right of premium adjusted
141  if (dt==DeltaVolQuote::PaSpot) {
142  rightLimit=strikeFromDelta(delta, DeltaVolQuote::Spot);
143  } else {
144  rightLimit=strikeFromDelta(delta, DeltaVolQuote::Fwd);
145  }
146 
147  if (phi_<0) { // if put
148  res=solver.solve(f, accuracy, rightLimit, 0.0, spot_*100.0);
149  break;
150  } else {
151 
152  // find out the left limit which is the strike
153  // corresponding to the value where premium adjusted
154  // deltas have their maximum.
155 
156  BlackDeltaPremiumAdjustedMaxStrikeClass g(
157  ot_,dt, spot_,dDiscount_, fDiscount_, stdDev_);
158 
159  leftLimit=solver.solve(g, accuracy, rightLimit*0.5,
160  0.0, rightLimit);
161 
162  Real guess=leftLimit+(rightLimit-leftLimit)*0.5;
163 
164  res=solver.solve(f, accuracy, guess, leftLimit, rightLimit);
165  } // end if phi<0 else
166 
167  break;
168  }
169 
170  default:
171  QL_FAIL("invalid delta type");
172  }
173 
174  return res;
175  }

177  Real BlackDeltaCalculator::atmStrike(DeltaVolQuote::AtmType atmT) const {
178 
179  Real res=0.0;
180 
181  switch(atmT) {
182  case DeltaVolQuote::AtmDeltaNeutral:
183  if(dt_==DeltaVolQuote::Spot || dt_==DeltaVolQuote::Fwd){
184  res=fExpPos_;
185  } else {
186  res=fExpNeg_;
187  }
188  break;
189 
190  case DeltaVolQuote::AtmFwd:
191  res=forward_;
192  break;
193 
194  case DeltaVolQuote::AtmGammaMax: case DeltaVolQuote::AtmVegaMax:
195  res=fExpPos_;
196  break;
197 
198  case DeltaVolQuote::AtmPutCall50:
199  QL_REQUIRE(dt_==DeltaVolQuote::Fwd,
200  "|PutDelta|=CallDelta=0.50 only possible for forward delta.");
201  res=fExpPos_;
202  break;
203 
204  default:
205  QL_FAIL("invalid atm type");
206  }
207 
208  return res;
209  }

212  Real BlackDeltaCalculator::cumD1(Real strike) const {
213 
214  Real d1_=0.0;
215  Real cum_d1_pos_ = 1.0; // N(d1)
216  Real cum_d1_neg_ = 0.0; // N(-d1)
217 
218  CumulativeNormalDistribution f;
219 
220  if (stdDev_>=QL_EPSILON) {
221  if(strike>0) {
222  d1_ = std::log(forward_/strike)/stdDev_ + 0.5*stdDev_;
223  return f(phi_*d1_);
224  }
225  } else {
226  if (forward_<strike) {
227  cum_d1_pos_ = 0.0;
228  cum_d1_neg_ = 1.0;
229  } else if(forward_==strike){
230  d1_ = 0.5*stdDev_;
231  return f(phi_*d1_);
232  }
233  }
234 
235  if (phi_>0) { // if Call
236  return cum_d1_pos_;
237  } else {
238  return cum_d1_neg_;
239  }
240  }

279  cum_d2_neg_= 1.0;
280  } else if (forward_==strike) {
281  d2_ = -0.5*stdDev_;
282  return(f(phi_*d2_));
283  }
284
