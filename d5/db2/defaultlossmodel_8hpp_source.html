146  void setBasket(Basket* bskt) {
147  /* After this; if the model modifies its internal status/caches (if
148  any) it should notify the prior basket to recognise that basket is
149  not in a calculated=true state. Since we dont know at this level if
150  the model keeps caches it is the children responsibility. Typically
151  this is done at the first call to calculate to the loss model, there
152  it notifies the basket. The old basket is still registered with us
153  until the basket takes in a new model....
154  ..alternatively both old basket and model could be forced reset here
155  */
156  basket_.linkTo(ext::shared_ptr<Basket>(bskt, null_deleter()),
157  false);
