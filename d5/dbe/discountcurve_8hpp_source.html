159  template <class T>
160  DiscountFactor InterpolatedDiscountCurve<T>::discountImpl(Time t) const {
161  if (t <= this->times_.back())
162  return this->interpolation_(t, true);
163 
164  // flat fwd extrapolation
165  Time tMax = this->times_.back();
166  DiscountFactor dMax = this->data_.back();
167  Rate instFwdMax = - this->interpolation_.derivative(tMax) / dMax;
168  return dMax * std::exp(- instFwdMax * (t-tMax));
169  }

246  template <class T>
247  void InterpolatedDiscountCurve<T>::initialize()
248  {
249  QL_REQUIRE(dates_.size() >= T::requiredPoints,
250  "not enough input dates given");
251  QL_REQUIRE(this->data_.size() == dates_.size(),
252  "dates/data count mismatch");
253  QL_REQUIRE(this->data_[0] == 1.0,
254  "the first discount must be == 1.0 "
255  "to flag the corresponding date as reference date");
256 
257  this->times_.resize(dates_.size());
258  this->times_[0] = 0.0;
259  for (Size i=1; i<dates_.size(); ++i) {
260  QL_REQUIRE(dates_[i] > dates_[i-1],
261  "invalid date (" << dates_[i] << ", vs "
262  << dates_[i-1] << ")");
263  this->times_[i] = dayCounter().yearFraction(dates_[0], dates_[i]);
264  QL_REQUIRE(!close(this->times_[i],this->times_[i-1]),
265  "two dates correspond to the same time "
266  "under this curve's day count convention");
267  QL_REQUIRE(this->data_[i] > 0.0, "negative discount");
268  #if !defined(QL_NEGATIVE_RATES)
269  QL_REQUIRE(this->data_[i] <= this->data_[i-1],
270  "negative forward rate implied by the discount " <<
271  this->data_[i] << " at " << dates_[i] <<
272  " (t=" << this->times_[i] << ") after the discount " <<
273  this->data_[i-1] << " at " << dates_[i-1] <<
274  " (t=" << this->times_[i-1] << ")");
275  #endif
276  }
277 
278  this->interpolation_ =
279  this->interpolator_.interpolate(this->times_.begin(),
280  this->times_.end(),
281  this->data_.begin());
282  this->interpolation_.update();
283  }
