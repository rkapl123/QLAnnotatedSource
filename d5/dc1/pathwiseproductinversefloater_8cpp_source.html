64  bool MarketModelPathwiseInverseFloater::nextTimeStep(
65  const CurveState& currentState,
66  std::vector<Size>& numberCashFlowsThisStep,
67  std::vector<std::vector<MarketModelPathwiseMultiProduct::CashFlow> >& cashFlowsGenerated)
68  {
69  numberCashFlowsThisStep[0] =1 ;
70  for (Size i=1; i <= lastIndex_; ++i)
71  cashFlowsGenerated[0][0].amount[i] =0;
72 
73  Rate liborRate = currentState.forwardRate(currentIndex_);
74  Real inverseFloatingCoupon = std::max((fixedStrikes_[currentIndex_] - fixedMultipliers_[currentIndex_]*liborRate),0.0)*fixedAccruals_[currentIndex_] ;
75  Real floatingCoupon = (liborRate+floatingSpreads_[currentIndex_])*floatingAccruals_[currentIndex_];
76  cashFlowsGenerated[0][0].timeIndex = currentIndex_;
77  cashFlowsGenerated[0][0].amount[0] =multiplier_*(inverseFloatingCoupon - floatingCoupon);
78 
79 
80 
81  if (inverseFloatingCoupon > 0.0)
82  {
83 
84  cashFlowsGenerated[0][0].amount[currentIndex_+1] =multiplier_*( - fixedMultipliers_[currentIndex_]*fixedAccruals_[currentIndex_] - floatingAccruals_[currentIndex_]);
85 
86 
87  }
88  else
89  {
90 
91  cashFlowsGenerated[0][0].amount[currentIndex_+1] =-multiplier_*floatingAccruals_[currentIndex_];
92 
93  }
94 
95  ++currentIndex_;
96 
97  return (currentIndex_ == lastIndex_);
