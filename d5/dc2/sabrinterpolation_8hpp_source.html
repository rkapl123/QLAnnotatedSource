65 struct SABRSpecs {
66  Size dimension() { return 4; }
67  void defaultValues(std::vector<Real> &params, std::vector<bool> &,
68  const Real &forward, const Real expiryTime,
69  const std::vector<Real> &addParams) {
70  if (params[1] == Null<Real>())
71  params[1] = 0.5;
72  if (params[0] == Null<Real>())
73  // adapt alpha to beta level
74  params[0] = 0.2 * (params[1] < 0.9999 ?
75  std::pow(forward + (addParams.empty() ? 0.0 : addParams[0]),
76  1.0 - params[1]) :
77  1.0);
78  if (params[2] == Null<Real>())
79  params[2] = std::sqrt(0.4);
80  if (params[3] == Null<Real>())
81  params[3] = 0.0;
82  }
83  void guess(Array &values, const std::vector<bool> &paramIsFixed,
84  const Real &forward, const Real expiryTime,
85  const std::vector<Real> &r, const std::vector<Real> &addParams) {
86  Size j = 0;
87  if (!paramIsFixed[1])
88  values[1] = (1.0 - 2E-6) * r[j++] + 1E-6;
89  if (!paramIsFixed[0]) {
90  values[0] = (1.0 - 2E-6) * r[j++] + 1E-6; // lognormal vol guess
91  // adapt this to beta level
92  if (values[1] < 0.999)
93  values[0] *=
94  std::pow(forward + (addParams.empty() ? 0.0 : addParams[0]), 1.0 - values[1]);
95  }
96  if (!paramIsFixed[2])
97  values[2] = 1.5 * r[j++] + 1E-6;
98  if (!paramIsFixed[3])
99  values[3] = (2.0 * r[j++] - 1.0) * (1.0 - 1E-6);
100  }
101  Real eps1() { return .0000001; }
102  Real eps2() { return .9999; }
103  Real dilationFactor() { return 0.001; }
104  Array inverse(const Array &y, const std::vector<bool> &,
105  const std::vector<Real> &, const Real) {
106  Array x(4);
107  x[0] = y[0] < 25.0 + eps1() ? std::sqrt(y[0] - eps1())
108  : (y[0] - eps1() + 25.0) / 10.0;
109  // y_[1] = std::tan(M_PI*(x[1] - 0.5))/dilationFactor();
110  x[1] = std::sqrt(-std::log(y[1]));
111  x[2] = y[2] < 25.0 + eps1() ? std::sqrt(y[2] - eps1())
112  : (y[2] - eps1() + 25.0) / 10.0;
113  x[3] = std::asin(y[3] / eps2());
114  return x;
115  }
116  Array direct(const Array &x, const std::vector<bool> &,
117  const std::vector<Real> &, const Real) {
118  Array y(4);
119  y[0] = std::fabs(x[0]) < 5.0 ? x[0] * x[0] + eps1()
120  : (10.0 * std::fabs(x[0]) - 25.0) + eps1();
121  // y_[1] = std::atan(dilationFactor_*x[1])/M_PI + 0.5;
122  y[1] = std::fabs(x[1]) < std::sqrt(-std::log(eps1()))
123  ? std::exp(-(x[1] * x[1]))
124  : eps1();
125  y[2] = std::fabs(x[2]) < 5.0 ? x[2] * x[2] + eps1()
126  : (10.0 * std::fabs(x[2]) - 25.0) + eps1();
127  y[3] = std::fabs(x[3]) < 2.5 * M_PI
128  ? eps2() * std::sin(x[3])
129  : eps2() * (x[3] > 0.0 ? 1.0 : (-1.0));
130  return y;
131  }
132  Real weight(const Real strike, const Real forward, const Real stdDev,
133  const std::vector<Real> &addParams) {
134  return blackFormulaStdDevDerivative(strike, forward, stdDev, 1.0,
135  addParams[0]);
136  }
137  typedef SABRWrapper type;
138  ext::shared_ptr<type> instance(const Time t, const Real &forward,
139  const std::vector<Real> &params,
140  const std::vector<Real> &addParams) {
141  return ext::make_shared<type>(t, forward, params, addParams);
142  }
143 };
144 }

154  SABRInterpolation(const I1 &xBegin, // x = strikes
155  const I1 &xEnd,

161  const ext::shared_ptr<EndCriteria> &endCriteria =

168 
169  impl_ = ext::shared_ptr<Interpolation::Impl>(

173  boost::assign::list_of(alphaIsFixed)(betaIsFixed)(nuIsFixed)(
174  rhoIsFixed),
175  vegaWeighted, endCriteria, optMethod, errorAccept, useMaxError,
176  maxGuesses, boost::assign::list_of(shift)));
177  coeffs_ = ext::dynamic_pointer_cast<
178  detail::XABRCoeffHolder<detail::SABRSpecs> >(impl_);
179  }

181  Real forward() const { return coeffs_->forward_; }

183  Real beta() const { return coeffs_->params_[1]; }
184  Real nu() const { return coeffs_->params_[2]; }
185  Real rho() const { return coeffs_->params_[3]; }
186  Real rmsError() const { return coeffs_->error_; }
187  Real maxError() const { return coeffs_->maxError_; }
188  const std::vector<Real> &interpolationWeights() const {
189  return coeffs_->weights_;
190  }
191  EndCriteria::Type endCriteria() { return coeffs_->XABREndCriteria_; }
192 
193  private:
194  ext::shared_ptr<detail::XABRCoeffHolder<detail::SABRSpecs> > coeffs_;
195 };
