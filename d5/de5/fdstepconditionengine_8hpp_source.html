63 template <template <class> class Scheme>
64 void FDStepConditionEngine<Scheme>::calculate(
65 PricingEngine::results* r) const {
66 auto* results = dynamic_cast<OneAssetOption::results*>(r);
67 setGridLimits();
68 initializeInitialCondition();
69 initializeOperator();
70 initializeBoundaryConditions();
71 initializeStepCondition();
72
73 typedef FiniteDifferenceModel<ParallelEvolver<
74 Scheme<TridiagonalOperator> > > model_type;
75
76 typename model_type::operator_type operatorSet;
77 typename model_type::array_type arraySet;
78 typename model_type::bc_set bcSet;
79 typename model_type::condition_type conditionSet;
80
81 prices_ = intrinsicValues_;
82
83 controlPrices_ = intrinsicValues_;
84 controlOperator_ = finiteDifferenceOperator_;
85 controlBCs_[0] = BCs_[0];
86 controlBCs_[1] = BCs_[1];
87
88 operatorSet.push_back(finiteDifferenceOperator_);
89 operatorSet.push_back(controlOperator_);
90
91 arraySet.push_back(prices_.values());
92 arraySet.push_back(controlPrices_.values());
93
94 bcSet.push_back(BCs_);
95 bcSet.push_back(controlBCs_);
96
97 conditionSet.push_back(stepCondition_);
98 conditionSet.push_back(ext::shared_ptr<StandardStepCondition>(
99 new NullCondition<Array>));
100
101 model_type model(operatorSet, bcSet);
102
103 model.rollback(arraySet, getResidualTime(),
104 0.0, timeSteps_, conditionSet);
105
106 prices_.values() = arraySet[0];
107 controlPrices_.values() = arraySet[1];
108
109 ext::shared_ptr<StrikedTypePayoff> striked_payoff =
110 ext::dynamic_pointer_cast<StrikedTypePayoff>(payoff_);
111 QL_REQUIRE(striked_payoff, "non-striked payoff given");
112
113 Real variance =
114 process_->blackVolatility()->blackVariance(
115 exerciseDate_, striked_payoff->strike());
116 DiscountFactor dividendDiscount =
117 process_->dividendYield()->discount(exerciseDate_);
118 DiscountFactor riskFreeDiscount =
119 process_->riskFreeRate()->discount(exerciseDate_);
120 Real spot = process_->stateVariable()->value();
121 Real forwardPrice = spot * dividendDiscount / riskFreeDiscount;
122
123 BlackCalculator black(striked_payoff, forwardPrice,
124 std::sqrt(variance), riskFreeDiscount);
125
126 results->value = prices_.valueAtCenter()
127 - controlPrices_.valueAtCenter()
128 + black.value();
129 results->delta = prices_.firstDerivativeAtCenter()
130 - controlPrices_.firstDerivativeAtCenter()
131 + black.delta(spot);
132 results->gamma = prices_.secondDerivativeAtCenter()
133 - controlPrices_.secondDerivativeAtCenter()
134 + black.gamma(spot);
135 results->additionalResults["priceCurve"] = prices_;
136 }
