58  Real G2::sigmaP(Time t, Time s) const {
59  Real temp = 1.0 - std::exp(-(a()+b())*t);
60  Real temp1 = 1.0 - std::exp(-a()*(s-t));
61  Real temp2 = 1.0 - std::exp(-b()*(s-t));
62  Real a3 = a()*a()*a();
63  Real b3 = b()*b()*b();
64  Real sigma2 = sigma()*sigma();
65  Real eta2 = eta()*eta();
66  Real value =
67  0.5*sigma2*temp1*temp1*(1.0 - std::exp(-2.0*a()*t))/a3 +
68  0.5*eta2*temp2*temp2*(1.0 - std::exp(-2.0*b()*t))/b3 +
69  2.0*rho()*sigma()*eta()/(a()*b()*(a()+b()))*
70  temp1*temp2*temp;
71  return std::sqrt(value);
72  }

88  Real G2::V(Time t) const {
89  Real expat = std::exp(-a()*t);
90  Real expbt = std::exp(-b()*t);
91  Real cx = sigma()/a();
92  Real cy = eta()/b();
93  Real valuex = cx*cx*(t + (2.0*expat-0.5*expat*expat-1.5)/a());
94  Real valuey = cy*cy*(t + (2.0*expbt-0.5*expbt*expbt-1.5)/b());
95  Real value = 2.0*rho()*cx*cy* (t + (expat - 1.0)/a()
96  + (expbt - 1.0)/b()
97  - (expat*expbt-1.0)/(a()+b()));
98  return valuex + valuey + value;
99  }

110  class G2::SwaptionPricingFunction {
111  public:
112  SwaptionPricingFunction(Real a,
113  Real sigma,
114  Real b,
115  Real eta,
116  Real rho,
117  Real w,
118  Real start,
119  std::vector<Time> payTimes,
120  Rate fixedRate,
121  const G2& model)
122  : a_(a), sigma_(sigma), b_(b), eta_(eta), rho_(rho), w_(w), T_(start),
123  t_(std::move(payTimes)), rate_(fixedRate), size_(t_.size()), A_(size_), Ba_(size_),
124  Bb_(size_) {
125 
126 
127  sigmax_ = sigma_*std::sqrt(0.5*(1.0-std::exp(-2.0*a_*T_))/a_);
128  sigmay_ = eta_*std::sqrt(0.5*(1.0-std::exp(-2.0*b_*T_))/b_);
129  rhoxy_ = rho_*eta_*sigma_*(1.0 - std::exp(-(a_+b_)*T_))/
130  ((a_+b_)*sigmax_*sigmay_);
131 
132  Real temp = sigma_*sigma_/(a_*a_);
133  mux_ = -((temp+rho_*sigma_*eta_/(a_*b_))*(1.0 - std::exp(-a*T_)) -
134  0.5*temp*(1.0 - std::exp(-2.0*a_*T_)) -
135  rho_*sigma_*eta_/(b_*(a_+b_))*
136  (1.0- std::exp(-(b_+a_)*T_)));
137 
138  temp = eta_*eta_/(b_*b_);
139  muy_ = -((temp+rho_*sigma_*eta_/(a_*b_))*(1.0 - std::exp(-b*T_)) -
140  0.5*temp*(1.0 - std::exp(-2.0*b_*T_)) -
141  rho_*sigma_*eta_/(a_*(a_+b_))*
142  (1.0- std::exp(-(b_+a_)*T_)));
143 
144  for (Size i=0; i<size_; i++) {
145  A_[i] = model.A(T_, t_[i]);
146  Ba_[i] = model.B(a_, t_[i]-T_);
147  Bb_[i] = model.B(b_, t_[i]-T_);
148  }
149  }
150 
151  Real mux() const { return mux_; }
152  Real sigmax() const { return sigmax_; }
153  Real operator()(Real x) const {
154  CumulativeNormalDistribution phi;
155  Real temp = (x - mux_)/sigmax_;
156  Real txy = std::sqrt(1.0 - rhoxy_*rhoxy_);
157 
158  Array lambda(size_);
159  Size i;
160  for (i=0; i<size_; i++) {
161  Real tau = (i==0 ? t_[0] - T_ : t_[i] - t_[i-1]);
162  Real c = (i==size_-1 ? (1.0+rate_*tau) : rate_*tau);
163  lambda[i] = c*A_[i]*std::exp(-Ba_[i]*x);
164  }
165 
166  SolvingFunction function(lambda, Bb_) ;
167  Brent s1d;
168  s1d.setMaxEvaluations(1000);
169  Real yb = s1d.solve(function, 1e-6, 0.00, -100.0, 100.0);
170 
171  Real h1 = (yb - muy_)/(sigmay_*txy) -
172  rhoxy_*(x - mux_)/(sigmax_*txy);
173  Real value = phi(-w_*h1);
174 
175 
176  for (i=0; i<size_; i++) {
177  Real h2 = h1 +
178  Bb_[i]*sigmay_*std::sqrt(1.0-rhoxy_*rhoxy_);
179  Real kappa = - Bb_[i] *
180  (muy_ - 0.5*txy*txy*sigmay_*sigmay_*Bb_[i] +
181  rhoxy_*sigmay_*(x-mux_)/sigmax_);
182  value -= lambda[i] *std::exp(kappa)*phi(-w_*h2);
183  }
184 
185  return std::exp(-0.5*temp*temp)*value/
186  (sigmax_*std::sqrt(2.0*M_PI));
187  }

191  class SolvingFunction {
192  public:
193  SolvingFunction(const Array& lambda, const Array& Bb)
194  : lambda_(lambda), Bb_(Bb) {}
195  Real operator()(Real y) const {
196  Real value = 1.0;
197  for (Size i=0; i<lambda_.size(); i++) {
198  value -= lambda_[i]*std::exp(-Bb_[i]*y);
199  }
200  return value;
201  }
202  private:
203  const Array& lambda_;
204  const Array& Bb_;
205  };

216  Real G2::swaption(const Swaption::arguments& arguments,
217  Rate fixedRate, Real range, Size intervals) const {
218 
219  Date settlement = termStructure()->referenceDate();
220  DayCounter dayCounter = termStructure()->dayCounter();
221  Time start = dayCounter.yearFraction(settlement,
222  arguments.floatingResetDates[0]);
223  Real w = (arguments.type==VanillaSwap::Payer ? 1 : -1 );
224 
225  std::vector<Time> fixedPayTimes(arguments.fixedPayDates.size());
226  for (Size i=0; i<fixedPayTimes.size(); ++i)
227  fixedPayTimes[i] =
228  dayCounter.yearFraction(settlement,
229  arguments.fixedPayDates[i]);
230 
231  SwaptionPricingFunction function(a(), sigma(), b(), eta(), rho(),
232  w, start,
233  fixedPayTimes,
234  fixedRate, (*this));
235 
236  Real upper = function.mux() + range*function.sigmax();
237  Real lower = function.mux() - range*function.sigmax();
238  SegmentIntegral integrator(intervals);
239  return arguments.nominal*w*termStructure()->discount(start)*
240  integrator(function, lower, upper);
241  }
