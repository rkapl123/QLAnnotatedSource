101  class InverseCumulativeNormal
102  : public std::unary_function<Real,Real> {
103  public:
104  InverseCumulativeNormal(Real average = 0.0,
105  Real sigma = 1.0);
106  // function
107  Real operator()(Real x) const {
108  return average_ + sigma_*standard_value(x);
109  }
110  // value for average=0, sigma=1
111  /* Compared to operator(), this method avoids 2 floating point
112  operations (we use average=0 and sigma=1 most of the
113  time). The speed difference is noticeable.
114  */
115  static Real standard_value(Real x) {
116  Real z;
117  if (x < x_low_ || x_high_ < x) {
118  z = tail_value(x);
119  } else {
120  z = x - 0.5;
121  Real r = z*z;
122  z = (((((a1_*r+a2_)*r+a3_)*r+a4_)*r+a5_)*r+a6_)*z /
123  (((((b1_*r+b2_)*r+b3_)*r+b4_)*r+b5_)*r+1.0);
124  }
125 
126  // The relative error of the approximation has absolute value less
127  // than 1.15e-9. One iteration of Halley's rational method (third
128  // order) gives full machine precision.
129  // #define REFINE_TO_FULL_MACHINE_PRECISION_USING_HALLEYS_METHOD
130  #ifdef REFINE_TO_FULL_MACHINE_PRECISION_USING_HALLEYS_METHOD
131  // error (f_(z) - x) divided by the cumulative's derivative
132  const Real r = (f_(z) - x) * M_SQRT2 * M_SQRTPI * exp(0.5 * z*z);
133  // Halley's method
134  z -= r/(1+0.5*z*r);
135  #endif
136 
137  return z;
138  }
139  private:
140  /* Handling tails moved into a separate method, which should
141  make the inlining of operator() and standard_value method
142  easier. tail_value is called rarely and doesn't need to be
143  inlined.
144  */
145  static Real tail_value(Real x);
146  #if defined(QL_PATCH_SOLARIS)
147  CumulativeNormalDistribution f_;
148  #else
149  static const CumulativeNormalDistribution f_;
150  #endif

266  inline NormalDistribution::NormalDistribution(Real average,
267  Real sigma)
268  : average_(average), sigma_(sigma) {
269 
270  QL_REQUIRE(sigma_>0.0,
271  "sigma must be greater than 0.0 ("
272  << sigma_ << " not allowed)");
273 
274  normalizationFactor_ = M_SQRT_2*M_1_SQRTPI/sigma_;
275  derNormalizationFactor_ = sigma_*sigma_;
276  denominator_ = 2.0*derNormalizationFactor_;
277  }

279  inline Real NormalDistribution::operator()(Real x) const {
280  Real deltax = x-average_;
281  Real exponent = -(deltax*deltax)/denominator_;
282  // debian alpha had some strange problem in the very-low range
283  return exponent <= -690.0 ? 0.0 : // exp(x) < 1.0e-300 anyway
284  normalizationFactor_*std::exp(exponent);
285  }
