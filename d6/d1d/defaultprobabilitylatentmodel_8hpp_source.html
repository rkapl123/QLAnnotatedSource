149  Probability conditionalDefaultProbabilityInvP(Real invCumYProb,
150  Size iName,
151  const std::vector<Real>& m) const {
152  Real sumMs =
153  std::inner_product(factorWeights_[iName].begin(),
154  factorWeights_[iName].end(), m.begin(), 0.);
155  Real res = cumulativeZ((invCumYProb - sumMs) /
156  idiosyncFctrs_[iName] );
157  #if defined(QL_EXTRA_SAFETY_CHECKS)
158  QL_REQUIRE (res >= 0. && res <= 1.,
159  "conditional probability " << res << "out of range");
160  #endif
161 
162  return res;
163  }

256  template<class CP>
257  Real DefaultLatentModel<CP>::defaultCorrelation(const Date& d,
258  Size iNamei, Size iNamej) const
259  {
260  QL_REQUIRE(basket_, "No portfolio basket set.");
261 
262  const boost::shared_ptr<Pool>& pool = basket_->pool();
263  // unconditionals:
264  Probability pi = pool->get(pool->names()[iNamei]).
265  defaultProbability(basket_->defaultKeys()[iNamei])
266  ->defaultProbability(d);
267  Probability pj = pool->get(pool->names()[iNamej]).
268  defaultProbability(basket_->defaultKeys()[iNamej])
269  ->defaultProbability(d);
270  Real pipj = pi * pj;
271  Real invPi = inverseCumulativeY(pi, iNamei);
272  Real invPj = inverseCumulativeY(pj, iNamej);
273  // avoid repetitive calls when i=j?
274  Real E1i1j; // joint default covariance term
275  if(iNamei !=iNamej) {
276  E1i1j = integratedExpectedValue(
277  boost::function<Real (const std::vector<Real>& v1)>(
278  boost::bind(
279  &DefaultLatentModel<CP>::condProbProduct,
280  this, invPi, invPj, iNamei, iNamej, _1) ));
281  }else{
282  E1i1j = pi;
283  }
284  return (E1i1j - pipj )/std::sqrt(pipj*(1.-pi)*(1.-pj));
285  }

288  template<class CP>
289  Real DefaultLatentModel<CP>::conditionalProbAtLeastNEvents(Size n,
290  const Date& date,
291  const std::vector<Real>& mktFactors) const {
292  QL_REQUIRE(basket_, "No portfolio basket set.");
293 
294  /* \todo
295  This algorithm traverses all permutations starting form the
296  lowest one. This is inneficient, there shouldnt be any need to
297  go through the invalid ones. Use combinations of n elements.
298 
299  See integration in O'Kane for homogeneous ntds.
300  */
301  // first position with as many defaults as desired:
302  Size poolSize = basket_->size();//move to 'livesize'
303  const boost::shared_ptr<Pool>& pool = basket_->pool();
304 
305  BigNatural limit =
306  static_cast<BigNatural>(std::pow(2., (int)(poolSize)));
307 
308  // Precalc conditional probabilities
309  std::vector<Probability> pDefCond;
310  for(Size i=0; i<poolSize; i++)
311  pDefCond.push_back(conditionalDefaultProbability(
312  pool->get(pool->names()[i]).
313  defaultProbability(basket_->defaultKeys()[i])->
314  defaultProbability(date), i, mktFactors));
315 
316  Probability probNEventsOrMore = 0.;
317  for(BigNatural mask =
318  static_cast<BigNatural>(std::pow(2., (int)(n))-1);
319  mask < limit; mask++)
320  {
321  // cheap permutations
322  boost::dynamic_bitset<> bsetMask(poolSize, mask);
323  if(bsetMask.count() >= n) {
324  Probability pConfig = 1;
325  for(Size i=0; i<bsetMask.size(); i++)
326  pConfig *=
327  (bsetMask[i] ? pDefCond[i] : (1.- pDefCond[i]));
328  probNEventsOrMore += pConfig;
329  }
330  }
331  return probNEventsOrMore;
332  }
