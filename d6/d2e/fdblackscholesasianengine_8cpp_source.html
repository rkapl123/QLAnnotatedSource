44  void FdBlackScholesAsianEngine::calculate() const {
45 
46  QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
47  "European exercise supported only");
48  QL_REQUIRE(arguments_.averageType == Average::Arithmetic,
49  "Arithmetic averaging supported only");
50  QL_REQUIRE( arguments_.runningAccumulator == 0
51  || arguments_.pastFixings > 0,
52  "Running average requires at least one past fixing");
53 
54  // 1. Mesher
55  const ext::shared_ptr<StrikedTypePayoff> payoff =
56  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
57  const Time maturity = process_->time(arguments_.exercise->lastDate());
58  const ext::shared_ptr<Fdm1dMesher> equityMesher(
59  new FdmBlackScholesMesher(xGrid_, process_, maturity,
60  payoff->strike()));
61 
62  const Real spot = process_->x0();
63  QL_REQUIRE(spot > 0.0, "negative or null underlying given");
64 
65  const Real avg = (arguments_.runningAccumulator == 0)
66  ? spot : arguments_.runningAccumulator/arguments_.pastFixings;
67 
68  const Real normInvEps = InverseCumulativeNormal()(1-0.0001);
69  const Real sigmaSqrtT
70  = process_->blackVolatility()->blackVol(maturity, payoff->strike())
71  *std::sqrt(maturity);
72  const Real r = sigmaSqrtT*normInvEps;
73 
74  Real xMin = std::min(std::log(avg) - 0.25*r, std::log(spot) - 1.5*r);
75  Real xMax = std::max(std::log(avg) + 0.25*r, std::log(spot) + 1.5*r);
76 
77  const ext::shared_ptr<Fdm1dMesher> averageMesher(
78  new FdmBlackScholesMesher(aGrid_, process_, maturity,
79  payoff->strike(), xMin, xMax));
80 
81  const ext::shared_ptr<FdmMesher> mesher (
82  new FdmMesherComposite(equityMesher, averageMesher));
83 
84  // 2. Calculator
85  ext::shared_ptr<FdmInnerValueCalculator> calculator(
86  new FdmLogInnerValue(payoff, mesher, 1));
87 
88  // 3. Step conditions
89  std::list<ext::shared_ptr<StepCondition<Array> > > stepConditions;
90  std::list<std::vector<Time> > stoppingTimes;
91 
92  // 3.1 Arithmetic average step conditions
93  std::vector<Time> averageTimes;
94  for (Size i=0; i<arguments_.fixingDates.size(); ++i) {
95  Time t = process_->time(arguments_.fixingDates[i]);
96  QL_REQUIRE(t >= 0, "Fixing dates must not contain past date");
97  averageTimes.push_back(t);
98  }
99  stoppingTimes.push_back(std::vector<Time>(averageTimes));
100  stepConditions.push_back(ext::shared_ptr<StepCondition<Array> >(
101  new FdmArithmeticAverageCondition(
102  averageTimes, arguments_.runningAccumulator,
103  arguments_.pastFixings, mesher, 0)));
104 
105  ext::shared_ptr<FdmStepConditionComposite> conditions(
106  new FdmStepConditionComposite(stoppingTimes, stepConditions));
107 
108  // 4. Boundary conditions
109  const FdmBoundaryConditionSet boundaries;
110 
111  // 5. Solver
112  FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
113  calculator, maturity, tGrid_, 0 };
114  ext::shared_ptr<FdmSimple2dBSSolver> solver(
115  new FdmSimple2dBSSolver(
116  Handle<GeneralizedBlackScholesProcess>(process_),
117  payoff->strike(), solverDesc, schemeDesc_));
118 
119  results_.value = solver->valueAt(spot, avg);
120  results_.delta = solver->deltaAt(spot, avg, spot*0.01);
121  results_.gamma = solver->gammaAt(spot, avg, spot*0.01);
122  }
123 }
