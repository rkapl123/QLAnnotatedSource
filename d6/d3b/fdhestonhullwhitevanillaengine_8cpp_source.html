54  void FdHestonHullWhiteVanillaEngine::calculate() const {
55 
56  // 1. cache lookup for precalculated results
57  for (auto& cachedArgs2result : cachedArgs2results_) {
58  if (cachedArgs2result.first.exercise->type() == arguments_.exercise->type() &&
59  cachedArgs2result.first.exercise->dates() == arguments_.exercise->dates()) {
60  ext::shared_ptr<PlainVanillaPayoff> p1 =
61  ext::dynamic_pointer_cast<PlainVanillaPayoff>(
62  arguments_.payoff);
63  ext::shared_ptr<PlainVanillaPayoff> p2 =
64  ext::dynamic_pointer_cast<PlainVanillaPayoff>(cachedArgs2result.first.payoff);
65 
66  if ((p1 != nullptr) && p1->strike() == p2->strike() &&
67  p1->optionType() == p2->optionType()) {
68  QL_REQUIRE(arguments_.cashFlow.empty(),
69  "multiple strikes engine does "
70  "not work with discrete dividends");
71  results_ = cachedArgs2result.second;
72  return;
73  }
74  }
75  }
76 
77  // 2. Mesher
78  const ext::shared_ptr<HestonProcess> hestonProcess=model_->process();
79  const Time maturity=hestonProcess->time(arguments_.exercise->lastDate());
80 
81  // 2.1 The variance mesher
82  const Size tGridMin = 5;
83  const ext::shared_ptr<FdmHestonVarianceMesher> varianceMesher(
84  new FdmHestonVarianceMesher(vGrid_, hestonProcess,
85  maturity,std::max(tGridMin,tGrid_/50)));
86 
87  // 2.2 The equity mesher
88  const ext::shared_ptr<StrikedTypePayoff> payoff =
89  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
90  QL_REQUIRE(payoff, "wrong payoff type given");
91 
92  ext::shared_ptr<Fdm1dMesher> equityMesher;
93  if (strikes_.empty()) {
94  equityMesher = ext::shared_ptr<Fdm1dMesher>(
95  new FdmBlackScholesMesher(
96  xGrid_,
97  FdmBlackScholesMesher::processHelper(
98  hestonProcess->s0(), hestonProcess->dividendYield(),
99  hestonProcess->riskFreeRate(),
100  varianceMesher->volaEstimate()),
101  maturity, payoff->strike(),
102  Null<Real>(), Null<Real>(), 0.0001, 1.5,
103  std::pair<Real, Real>(payoff->strike(), 0.1),
104  arguments_.cashFlow));
105  }
106  else {
107  QL_REQUIRE(arguments_.cashFlow.empty(),"multiple strikes engine "
108  "does not work with discrete dividends");
109  equityMesher = ext::shared_ptr<Fdm1dMesher>(
110  new FdmBlackScholesMultiStrikeMesher(
111  xGrid_,
112  FdmBlackScholesMesher::processHelper(
113  hestonProcess->s0(), hestonProcess->dividendYield(),
114  hestonProcess->riskFreeRate(),
115  varianceMesher->volaEstimate()),
116  maturity, strikes_, 0.0001, 1.5,
117  std::pair<Real, Real>(payoff->strike(), 0.075)));
118  }
119 
120  //2.3 The short rate mesher
121  const ext::shared_ptr<OrnsteinUhlenbeckProcess> ouProcess(
122  new OrnsteinUhlenbeckProcess(hwProcess_->a(),hwProcess_->sigma()));
123  const ext::shared_ptr<Fdm1dMesher> shortRateMesher(
124  new FdmSimpleProcess1dMesher(rGrid_, ouProcess, maturity));
125 
126  const ext::shared_ptr<FdmMesher> mesher(
127  new FdmMesherComposite(equityMesher, varianceMesher,
128  shortRateMesher));
129 
130  // 3. Calculator
131  const ext::shared_ptr<FdmInnerValueCalculator> calculator(
132  new FdmLogInnerValue(arguments_.payoff, mesher, 0));
133 
134  // 4. Step conditions
135  const ext::shared_ptr<FdmStepConditionComposite> conditions =
136  FdmStepConditionComposite::vanillaComposite(
137  arguments_.cashFlow, arguments_.exercise,
138  mesher, calculator,
139  hestonProcess->riskFreeRate()->referenceDate(),
140  hestonProcess->riskFreeRate()->dayCounter());
141 
142  // 5. Boundary conditions
143  const FdmBoundaryConditionSet boundaries;
144 
145  // 6. Solver
146  const FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
147  calculator, maturity,
148  tGrid_, dampingSteps_ };
149 
150  const ext::shared_ptr<FdmHestonHullWhiteSolver> solver(
151  new FdmHestonHullWhiteSolver(Handle<HestonProcess>(hestonProcess),
152  Handle<HullWhiteProcess>(hwProcess_),
153  corrEquityShortRate_,
154  solverDesc, schemeDesc_));
155 
156  const Real spot = hestonProcess->s0()->value();
157  const Real v0 = hestonProcess->v0();
158  results_.value = solver->valueAt(spot, v0, 0);
159  results_.delta = solver->deltaAt(spot, v0, 0, spot*0.01);
160  results_.gamma = solver->gammaAt(spot, v0, 0, spot*0.01);
161  results_.theta = solver->thetaAt(spot, v0, 0);
162 
163  cachedArgs2results_.resize(strikes_.size());
164  for (Size i=0; i < strikes_.size(); ++i) {
165  cachedArgs2results_[i].first.exercise = arguments_.exercise;
166  cachedArgs2results_[i].first.payoff =
167  ext::make_shared<PlainVanillaPayoff>(
168  payoff->optionType(), strikes_[i]);
169  const Real d = payoff->strike()/strikes_[i];
170 
171  DividendVanillaOption::results&
172  results = cachedArgs2results_[i].second;
173  results.value = solver->valueAt(spot*d, v0, 0)/d;
174  results.delta = solver->deltaAt(spot*d, v0, 0, spot*d*0.01);
175  results.gamma = solver->gammaAt(spot*d, v0, 0, spot*d*0.01)*d;
176  results.theta = solver->thetaAt(spot*d, v0, 0)/d;
177  }
178 
179  if (controlVariate_) {
180  ext::shared_ptr<PricingEngine> analyticEngine(
181  new AnalyticHestonEngine(*model_, 164));
182  ext::shared_ptr<Exercise> exercise(
183  new EuropeanExercise(arguments_.exercise->lastDate()));
184 
185  VanillaOption option(payoff, exercise);
186  option.setPricingEngine(analyticEngine);
187  Real analyticNPV = option.NPV();
188 
189  ext::shared_ptr<FdHestonVanillaEngine> fdEngine(
190  new FdHestonVanillaEngine(*model_, tGrid_, xGrid_,
191  vGrid_, dampingSteps_,
192  schemeDesc_));
193  fdEngine->enableMultipleStrikesCaching(strikes_);
194  option.setPricingEngine(fdEngine);
195 
196  Real fdNPV = option.NPV();
197  results_.value += analyticNPV - fdNPV;
198  for (Size i=0; i < strikes_.size(); ++i) {
199  VanillaOption controlVariateOption(
200  ext::shared_ptr<StrikedTypePayoff>(
201  new PlainVanillaPayoff(payoff->optionType(),
202  strikes_[i])), exercise);
203  controlVariateOption.setPricingEngine(analyticEngine);
204  analyticNPV = controlVariateOption.NPV();
205 
206  controlVariateOption.setPricingEngine(fdEngine);
207  fdNPV = controlVariateOption.NPV();
208  cachedArgs2results_[i].second.value += analyticNPV - fdNPV;
209  }
210  }
211  }
