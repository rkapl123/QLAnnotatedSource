75  void YieldTermStructure::setJumps(const Date& referenceDate) {
76  if (jumpDates_.empty() && !jumps_.empty()) { // turn of year dates
77  jumpDates_.resize(nJumps_);
78  jumpTimes_.resize(nJumps_);
79  Year y = referenceDate.year();
80  for (Size i=0; i<nJumps_; ++i)
81  jumpDates_[i] = Date(31, December, y+i);
82  } else { // fixed dates
83  QL_REQUIRE(jumpDates_.size()==nJumps_,
84  "mismatch between number of jumps (" << nJumps_ <<
85  ") and jump dates (" << jumpDates_.size() << ")");
86  }
87  for (Size i=0; i<nJumps_; ++i)
88  jumpTimes_[i] = timeFromReference(jumpDates_[i]);
89  latestReference_ = referenceDate;
90  }

92  DiscountFactor YieldTermStructure::discount(Time t,
93  bool extrapolate) const {
94  checkRange(t, extrapolate);
95 
96  if (jumps_.empty())
97  return discountImpl(t);
98 
99  DiscountFactor jumpEffect = 1.0;
100  for (Size i=0; i<nJumps_; ++i) {
101  if (jumpTimes_[i]>0 && jumpTimes_[i]<t) {
102  QL_REQUIRE(jumps_[i]->isValid(),
103  "invalid " << io::ordinal(i+1) << " jump quote");
104  DiscountFactor thisJump = jumps_[i]->value();
105  QL_REQUIRE(thisJump > 0.0,
106  "invalid " << io::ordinal(i+1) << " jump value: " <<
107  thisJump);
108  jumpEffect *= thisJump;
109  }
110  }
111  return jumpEffect * discountImpl(t);
112  }

114  InterestRate YieldTermStructure::zeroRate(const Date& d,
115  const DayCounter& dayCounter,
116  Compounding comp,
117  Frequency freq,
118  bool extrapolate) const {
119  if (d==referenceDate()) {
120  Real compound = 1.0/discount(dt, extrapolate);
121  // t has been calculated with a possibly different daycounter
122  // but the difference should not matter for very small times
123  return InterestRate::impliedRate(compound,
124  dayCounter, comp, freq,
125  dt);
126  }
127  Real compound = 1.0/discount(d, extrapolate);
128  return InterestRate::impliedRate(compound,
129  dayCounter, comp, freq,
130  referenceDate(), d);
131  }

144  InterestRate YieldTermStructure::forwardRate(const Date& d1,
145  const Date& d2,
146  const DayCounter& dayCounter,
147  Compounding comp,
148  Frequency freq,
149  bool extrapolate) const {
150  if (d1==d2) {
151  checkRange(d1, extrapolate);
152  Time t1 = std::max(timeFromReference(d1) - dt/2.0, 0.0);
153  Time t2 = t1 + dt;
154  Real compound =
155  discount(t1, true)/discount(t2, true);
156  // times have been calculated with a possibly different daycounter
157  // but the difference should not matter for very small times
158  return InterestRate::impliedRate(compound,
159  dayCounter, comp, freq,
160  dt);
161  }
162  QL_REQUIRE(d1 < d2, d1 << " later than " << d2);
163  Real compound = discount(d1, extrapolate)/discount(d2, extrapolate);
164  return InterestRate::impliedRate(compound,
165  dayCounter, comp, freq,
166  d1, d2);
167  }

169  InterestRate YieldTermStructure::forwardRate(Time t1,
170  Time t2,
171  Compounding comp,
172  Frequency freq,
173  bool extrapolate) const {
174  Real compound;
175  if (t2==t1) {
176  checkRange(t1, extrapolate);
177  t1 = std::max(t1 - dt/2.0, 0.0);
178  t2 = t1 + dt;
179  compound = discount(t1, true)/discount(t2, true);
180  } else {
181  QL_REQUIRE(t2>t1, "t2 (" << t2 << ") < t1 (" << t2 << ")");
182  compound = discount(t1, extrapolate)/discount(t2, extrapolate);
183  }
184  return InterestRate::impliedRate(compound,
185  dayCounter(), comp, freq,
186  t2-t1);
187  }

189  void YieldTermStructure::update() {
190  TermStructure::update();
191  Date newReference = Date();
192  try {
193  newReference = referenceDate();
194  if (newReference != latestReference_)
195  setJumps(newReference);
196  } catch (Error&) {
197  if (newReference == Date()) {
198  // the curve couldn't calculate the reference
199  // date. Most of the times, this is because some
200  // underlying handle wasn't set, so we can just absorb
201  // the exception and continue; the jumps will be set
202  // correctly when a valid underlying is set.
203  return;
204  } else {
205  // something else happened during the call to
206  // setJumps(), so we let the exception bubble up.
207  throw;
208  }
209  }
210  }
