59 void YieldTermStructure::setJumps(const Date& referenceDate) {
60 if (jumpDates_.empty() && !jumps_.empty()) { // turn of year dates
61 jumpDates_.resize(nJumps_);
62 jumpTimes_.resize(nJumps_);
63 Year y = referenceDate.year();
64 for (Size i=0; i<nJumps_; ++i)
65 jumpDates_[i] = Date(31, December, y+i);
66 } else { // fixed dates
67 QL_REQUIRE(jumpDates_.size()==nJumps_,
68 "mismatch between number of jumps (" << nJumps_ <<
69 ") and jump dates (" << jumpDates_.size() << ")");
70 }
71 for (Size i=0; i<nJumps_; ++i)
72 jumpTimes_[i] = timeFromReference(jumpDates_[i]);
73 latestReference_ = referenceDate;
74 }

76 DiscountFactor YieldTermStructure::discount(Time t,
77 bool extrapolate) const {
78 checkRange(t, extrapolate);
79
80 if (jumps_.empty())
81 return discountImpl(t);
82
83 DiscountFactor jumpEffect = 1.0;
84 for (Size i=0; i<nJumps_; ++i) {
85 if (jumpTimes_[i]>0 && jumpTimes_[i]<t) {
86 QL_REQUIRE(jumps_[i]->isValid(),
87 "invalid " << io::ordinal(i+1) << " jump quote");
88 DiscountFactor thisJump = jumps_[i]->value();
89 QL_REQUIRE(thisJump > 0.0,
90 "invalid " << io::ordinal(i+1) << " jump value: " <<
91 thisJump);
92 jumpEffect *= thisJump;
93 }
94 }
95 return jumpEffect * discountImpl(t);
96 }

98 InterestRate YieldTermStructure::zeroRate(const Date& d,
99 const DayCounter& dayCounter,
100 Compounding comp,
101 Frequency freq,
102 bool extrapolate) const {
103 if (d==referenceDate()) {
104 Real compound = 1.0/discount(dt, extrapolate);
105 // t has been calculated with a possibly different daycounter
106 // but the difference should not matter for very small times
107 return InterestRate::impliedRate(compound,
108 dayCounter, comp, freq,
109 dt);
110 }
111 Real compound = 1.0/discount(d, extrapolate);
112 return InterestRate::impliedRate(compound,
113 dayCounter, comp, freq,
114 referenceDate(), d);
115 }

128 InterestRate YieldTermStructure::forwardRate(const Date& d1,
129 const Date& d2,
130 const DayCounter& dayCounter,
131 Compounding comp,
132 Frequency freq,
133 bool extrapolate) const {
134 if (d1==d2) {
135 checkRange(d1, extrapolate);
136 Time t1 = std::max(timeFromReference(d1) - dt/2.0, 0.0);
137 Time t2 = t1 + dt;
138 Real compound =
139 discount(t1, true)/discount(t2, true);
140 // times have been calculated with a possibly different daycounter
141 // but the difference should not matter for very small times
142 return InterestRate::impliedRate(compound,
143 dayCounter, comp, freq,
144 dt);
145 }
146 QL_REQUIRE(d1 < d2, d1 << " later than " << d2);
147 Real compound = discount(d1, extrapolate)/discount(d2, extrapolate);
148 return InterestRate::impliedRate(compound,
149 dayCounter, comp, freq,
150 d1, d2);
151 }

153 InterestRate YieldTermStructure::forwardRate(Time t1,
154 Time t2,
155 Compounding comp,
156 Frequency freq,
157 bool extrapolate) const {
158 Real compound;
159 if (t2==t1) {
160 checkRange(t1, extrapolate);
161 t1 = std::max(t1 - dt/2.0, 0.0);
162 t2 = t1 + dt;
163 compound = discount(t1, true)/discount(t2, true);
164 } else {
165 QL_REQUIRE(t2>t1, "t2 (" << t2 << ") < t1 (" << t2 << ")");
166 compound = discount(t1, extrapolate)/discount(t2, extrapolate);
167 }
168 return InterestRate::impliedRate(compound,
169 dayCounter(), comp, freq,
170 t2-t1);
171 }

173 void YieldTermStructure::update() {
174 TermStructure::update();
175 Date newReference = Date();
176 try {
177 newReference = referenceDate();
178 if (newReference != latestReference_)
179 setJumps(newReference);
180 } catch (Error&) {
181 if (newReference == Date()) {
182 // the curve couldn't calculate the reference
183 // date. Most of the times, this is because some
184 // underlying handle wasn't set, so we can just absorb
185 // the exception and continue; the jumps will be set
186 // correctly when a valid underlying is set.
187 return;
188 } else {
189 // something else happened during the call to
190 // setJumps(), so we let the exception bubble up.
191 throw;
192 }
193 }
194 }
