73 FloatingRateBond::FloatingRateBond(
74 Natural settlementDays,
75 Real faceAmount,
76 const Date& startDate,
77 const Date& maturityDate,
78 Frequency couponFrequency,
79 const Calendar& calendar,
80 const ext::shared_ptr<IborIndex>& iborIndex,
81 const DayCounter& accrualDayCounter,
82 BusinessDayConvention accrualConvention,
83 BusinessDayConvention paymentConvention,
84 Natural fixingDays,
85 const std::vector<Real>& gearings,
86 const std::vector<Spread>& spreads,
87 const std::vector<Rate>& caps,
88 const std::vector<Rate>& floors,
89 bool inArrears,
90 Real redemption,
91 const Date& issueDate,
92 const Date& stubDate,
93 DateGeneration::Rule rule,
94 bool endOfMonth,
95 const Period& exCouponPeriod,
96 const Calendar& exCouponCalendar,
97 const BusinessDayConvention exCouponConvention,
98 bool exCouponEndOfMonth)
99 : Bond(settlementDays, calendar, issueDate) {
100
101 maturityDate_ = maturityDate;
102
103 Date firstDate, nextToLastDate;
104 switch (rule) {
105 case DateGeneration::Backward:
106 firstDate = Date();
107 nextToLastDate = stubDate;
108 break;
109 case DateGeneration::Forward:
110 firstDate = stubDate;
111 nextToLastDate = Date();
112 break;
113 case DateGeneration::Zero:
114 case DateGeneration::ThirdWednesday:
115 case DateGeneration::Twentieth:
116 case DateGeneration::TwentiethIMM:
117 QL_FAIL("stub date (" << stubDate << ") not allowed with " <<
118 rule << " DateGeneration::Rule");
119 default:
120 QL_FAIL("unknown DateGeneration::Rule (" << Integer(rule) << ")");
121 }
122
123 Schedule schedule(startDate, maturityDate_, Period(couponFrequency),
124 calendar_, accrualConvention, accrualConvention,
125 rule, endOfMonth,
126 firstDate, nextToLastDate);
127
128 cashflows_ = IborLeg(schedule, iborIndex)
129 .withNotionals(faceAmount)
130 .withPaymentDayCounter(accrualDayCounter)
131 .withPaymentAdjustment(paymentConvention)
132 .withFixingDays(fixingDays)
133 .withGearings(gearings)
134 .withSpreads(spreads)
135 .withCaps(caps)
136 .withFloors(floors)
137 .inArrears(inArrears)
138 .withExCouponPeriod(exCouponPeriod,exCouponCalendar, exCouponConvention, exCouponEndOfMonth);
139
140 addRedemptionsToCashflows(std::vector<Real>(1, redemption));
141
142 QL_ENSURE(!cashflows().empty(), "bond with no cashflows!");
143 QL_ENSURE(redemptions_.size() == 1, "multiple redemptions created");
144
145 registerWith(iborIndex);
146 }
