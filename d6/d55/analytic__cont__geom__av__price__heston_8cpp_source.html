25  class AnalyticContinuousGeometricAveragePriceAsianHestonEngine::Integrand {
26  private:
27  Real t_, T_, K_, logK_;
28  Size cutoff_;
29  const AnalyticContinuousGeometricAveragePriceAsianHestonEngine* const parent_;
30  Real xiRightLimit_;
31  std::complex<Real> i_;
32 
33  public:
34  Integrand(Real T,
35  Size cutoff,
36  Real K,
37  const AnalyticContinuousGeometricAveragePriceAsianHestonEngine* const parent,
38  Real xiRightLimit) : t_(0.0), T_(T), K_(K), logK_(std::log(K)), cutoff_(cutoff),
39  parent_(parent), xiRightLimit_(xiRightLimit), i_(std::complex<Real>(0.0, 1.0)) {}
40 
41  double operator()(double xi) const {
42  double xiDash = (0.5+1e-8+0.5*xi) * xiRightLimit_; // Map xi to full range
43 
44  std::complex<Real> inner1 = parent_->Phi(1.0 + xiDash*i_, 0, T_, t_, cutoff_);
45  std::complex<Real> inner2 = - K_*parent_->Phi(xiDash*i_, 0, T_, t_, cutoff_);
46 
47  return 0.5*xiRightLimit_*std::real((inner1 + inner2) * std::exp(-xiDash*logK_*i_) / (xiDash*i_));
48  }
49  };

51  class AnalyticContinuousGeometricAveragePriceAsianHestonEngine::DcfIntegrand {
52  private:
53  Real t_, T_, denominator_;
54  const Handle<YieldTermStructure> riskFreeRate_;
55  const Handle<YieldTermStructure> dividendYield_;
56  public:
57  DcfIntegrand(Real t,
58  Real T,
59  Handle<YieldTermStructure> riskFreeRate,
60  Handle<YieldTermStructure> dividendYield)
61  : t_(t), T_(T), riskFreeRate_(std::move(riskFreeRate)),
62  dividendYield_(std::move(dividendYield)) {
63  denominator_ = std::log(riskFreeRate_->discount(t_)) - std::log(dividendYield_->discount(t_));
64  }
65 
66  double operator()(double u) const {
67  double uDash = (0.5+1e-8+0.5*u) * (T_ - t_) + t_; // Map u to full range
68  return 0.5*(T_ - t_)*(-std::log(riskFreeRate_->discount(uDash))
69  + std::log(dividendYield_->discount(uDash)) + denominator_);
70  }
71  };

75  AnalyticContinuousGeometricAveragePriceAsianHestonEngine(
76  ext::shared_ptr<HestonProcess> process, Size summationCutoff, Real xiRightLimit)
77  : process_(std::move(process)), a3_(0.0), a4_(0.0), a5_(0.0), summationCutoff_(summationCutoff),
78  xiRightLimit_(xiRightLimit), integrator_(128) {
79  registerWith(process_);
80 
81  v0_ = process_->v0();
82  rho_ = process_->rho();
83  kappa_ = process_->kappa();
84  theta_ = process_->theta();
85  sigma_ = process_->sigma();
86  s0_ = process_->s0();
87 
88  riskFreeRate_ = process_->riskFreeRate();
89  dividendYield_ = process_->dividendYield();
90 
91  // Some of the required constant intermediate variables can be calculated now
92  // (although anything depending on T will need to be calculated dynamically later)
93  a1_ = 2.0 * v0_ / (sigma_*sigma_);
94  a2_ = 2.0 * kappa_ * theta_ / (sigma_*sigma_);
95  }

108  const std::complex<Real>& s, const std::complex<Real>& w, Real T) const {

117  std::complex<Real> AnalyticContinuousGeometricAveragePriceAsianHestonEngine::f(const std::complex<Real>& z1,
118  const std::complex<Real>& z2,
119  const std::complex<Real>& z3,
120  const std::complex<Real>& z4,
121  int n, // Can't use Size here as n can be negative
122  Real tau) const {;
123  std::complex<Real> result;
124 
125  // This equation is highly recursive, use dynamic programming with a mutable variable
126  // to record the results of previous calls
127  if (n<2) {
128  if (n<0) {
129  result = 0.0;
130  } else if (n==0) {
131  result = 1.0;
132  } else {
133  result = 0.5*(kappa_ - z4*sigma_*sigma_)*tau;
134  }
135  } else {
136  std::complex<Real> fMinusN[4];
137  Real prefactor = -0.5*sigma_*sigma_*tau*tau / (n*(n-1));
138 
139  // For each offset, look up the value in the map and only evaluate function if it's not there
140  for (int offset=1; offset<5; offset++) {
141  int location = n-offset;
142  std::map<int, std::complex<Real> >::const_iterator position = fLookupTable_.find(location);
143  if (position != fLookupTable_.end()) {
144  std::complex<Real> value = position->second;
145  fMinusN[offset-1] = value;
146  } else {
147  fMinusN[offset-1] = f(z1, z2, z3, z4, location, tau);
148  }
149  }
150 
151  result = prefactor * (z1*tau*tau*fMinusN[3] + z2*tau*fMinusN[2] + (z3 - 0.5*kappa_*kappa_/(sigma_*sigma_))*fMinusN[1]);
152  }
153 
154  // Store this value in our mutable lookup map
155  fLookupTable_[n] = result;
156 
157  return result;
158  };

160  std::pair<std::complex<Real>, std::complex<Real> >
161  AnalyticContinuousGeometricAveragePriceAsianHestonEngine::F_F_tilde(
162  const std::complex<Real>& z1,
163  const std::complex<Real>& z2,
164  const std::complex<Real>& z3,
165  const std::complex<Real>& z4,
166  Real tau,
167  Size cutoff) const {
168  std::complex<Real> temp = 0.0;
169  std::complex<Real> runningSum1 = 0.0;
170  std::complex<Real> runningSum2 = 0.0;
171 
172  for (Size i=0; i<cutoff; i++) {
173  temp = f(z1, z2, z3, z4, i, tau);
174  runningSum1 += temp;
175  runningSum2 += temp*double(i)/tau;
176  }
177 
178  std::pair<std::complex<Real>, std::complex<Real> > result(runningSum1, runningSum2);
179 
180  return result;
181  };

183  std::complex<Real> AnalyticContinuousGeometricAveragePriceAsianHestonEngine::Phi(
184  const std::complex<Real>& s,
185  const std::complex<Real>& w,
186  Real T,
187  Real t,
188  Size cutoff) const {
189  Real tau = T - t;
190 
191  std::complex<Real> z1 = z1_f(s, w, T);
192  std::complex<Real> z2 = z2_f(s, w, T);
193  std::complex<Real> z3 = z3_f(s, w, T);
194  std::complex<Real> z4 = z4_f(s, w);
195 
196  // Clear the mutable lookup map before calling fLookupTable
197  fLookupTable_ = std::map<int, std::complex<Real> >();
198  std::pair<std::complex<Real>, std::complex<Real> > temp = F_F_tilde(z1, z2, z3, z4, tau, cutoff);
199 
200  std::complex<Real> F, F_tilde;
201  F = temp.first;
202  F_tilde = temp.second;
203 
204  return std::exp(-a1_*F_tilde/F - a2_*std::log(F) + a3_*s + a4_*w + a5_);
205  }

207  void AnalyticContinuousGeometricAveragePriceAsianHestonEngine::calculate() const {
208  QL_REQUIRE(arguments_.averageType == Average::Geometric,
209  "not a geometric average option");
210  QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
211  "not an European Option");
212 
213  ext::shared_ptr<PlainVanillaPayoff> payoff =
214  ext::dynamic_pointer_cast<PlainVanillaPayoff>(arguments_.payoff);
215  QL_REQUIRE(payoff, "non-plain payoff given");
216 
217  Real strike = payoff->strike();
218  Date exercise = arguments_.exercise->lastDate();
219 
220  Time expiryTime = this->process_->time(exercise);
221  QL_REQUIRE(expiryTime >= 0.0, "Expiry Date cannot be in the past");
222 
223  Real expiryDcf = riskFreeRate_->discount(expiryTime);
224  Real expiryDividendDiscount = dividendYield_->discount(expiryTime);
225 
226  // TODO: extend to cover seasoned options (discussed in paper)
227  Time startTime = 0.0;
228 
229  // These parameters only need to be calculated once per pricing, but are
230  // functions of t and T so need to be reset in calculate()
231  Time t = startTime;
232  Time T = expiryTime;
233  Time tau = T - t;
234  Real logS0 = std::log(s0_->value());
235 
236  // To deal with non-constant rates and dividends, we reformulate Eq.s (14) to (17) with
237  // r_ --> (r(t) - q(t)), which gives the new expressions for a3 and a4 used below
238  Real dcf = riskFreeRate_->discount(T) / riskFreeRate_->discount(t);
239  Real qdcf = dividendYield_->discount(T) / dividendYield_->discount(t);
240  DcfIntegrand dcfIntegrand = DcfIntegrand(t, T, riskFreeRate_, dividendYield_);
241  Real integratedDcf = integrator_(dcfIntegrand);
242 
243  a3_ = (tau*logS0 + integratedDcf)/T - kappa_*theta_*rho_*tau*tau/(2*sigma_*T) - rho_*tau*v0_/(sigma_*T);
244  a4_ = logS0*qdcf/dcf - rho_*v0_/sigma_ + rho_*kappa_*theta_*tau/sigma_;
245  a5_ = (kappa_*v0_ + kappa_*kappa_*theta_*tau) / (sigma_*sigma_);
246 
247  // Calculate the two terms in eq (29) - Phi(1,0) is real (asian forward) but need to type convert
248  Real term1 = 0.5 * (std::real(Phi(1,0, T, t, summationCutoff_)) - strike);
249 
250  Integrand integrand(T, summationCutoff_, strike, this, xiRightLimit_);
251  Real term2 = integrator_(integrand) / M_PI;
252 
253  // Apply the payoff functions
254  Real value = 0.0;
255  switch (payoff->optionType()){
256  case Option::Call:
257  value = expiryDcf * (term1 + term2);
258  break;
259  case Option::Put:
260  value = expiryDcf * (-term1 + term2);
261  break;
262  default:
263  QL_FAIL("unknown option type");
264  }
265 
266  results_.value = value;
267 
268  results_.additionalResults["dcf"] = expiryDcf;
269  results_.additionalResults["qf"] = expiryDividendDiscount;
270  results_.additionalResults["s0"] = s0_->value();
271  results_.additionalResults["strike"] = strike;
272  results_.additionalResults["expiryTime"] = expiryTime;
273  results_.additionalResults["exercise"] = exercise;
274 
275  results_.additionalResults["term1"] = term1;
276  results_.additionalResults["term2"] = term2;
277  results_.additionalResults["xiRightLimit"] = xiRightLimit_;
278  results_.additionalResults["summationCutoff"] = summationCutoff_;
279 
280  results_.additionalResults["a1"] = a1_;
281  results_.additionalResults["a2"] = a2_;
282  results_.additionalResults["a3"] = a3_;
283  results_.additionalResults["a4"] = a4_;
284  results_.additionalResults["a5"] = a5_;
285  }
