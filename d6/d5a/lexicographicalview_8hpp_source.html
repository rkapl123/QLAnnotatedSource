86 template <class RandomAccessIterator>
87 inline
88 LexicographicalView<RandomAccessIterator>::LexicographicalView(
89 const RandomAccessIterator& begin,
90 const RandomAccessIterator& end,
91 Size xSize)
92 : begin_(begin), end_(end), xSize_(xSize),
93 ySize_((end-begin)/xSize) {
94 QL_REQUIRE((end_-begin_) % xSize_ == 0,
95 "The x size of the view is not an exact divisor"
96 "of the size of the underlying sequence");
