49  void IsdaCdsEngine::calculate() const {
50 
51  QL_REQUIRE(numericalFix_ == None || numericalFix_ == Taylor,
52  "numerical fix must be None or Taylor");
53  QL_REQUIRE(accrualBias_ == HalfDayBias || accrualBias_ == NoBias,
54  "accrual bias must be HalfDayBias or NoBias");
55  QL_REQUIRE(forwardsInCouponPeriod_ == Flat ||
56  forwardsInCouponPeriod_ == Piecewise,
57  "forwards in coupon period must be Flat or Piecewise");
58 
59  // it would be possible to handle the cases which are excluded below,
60  // but the ISDA engine is not explicitly specified to handle them,
61  // so we just forbid them too
62 
63  Actual365Fixed dc;
64  Actual360 dc1;
65  Actual360 dc2(true);
66 
67  Date evalDate = Settings::instance().evaluationDate();
68 
69  // check if given curves are ISDA compatible
70  // (the interpolation is checked below)
71 
72  QL_REQUIRE(!discountCurve_.empty(), "no discount term structure set");
73  QL_REQUIRE(!probability_.empty(), "no probability term structure set");
74  QL_REQUIRE(discountCurve_->dayCounter() == dc,
75  "yield term structure day counter ("
76  << discountCurve_->dayCounter()
77  << ") should be Act/365(Fixed)");
78  QL_REQUIRE(probability_->dayCounter() == dc,
79  "probability term structure day counter ("
80  << probability_->dayCounter() << ") should be "
81  << "Act/365(Fixed)");
82  QL_REQUIRE(discountCurve_->referenceDate() == evalDate,
83  "yield term structure reference date ("
84  << discountCurve_->referenceDate()
85  << " should be evaluation date (" << evalDate << ")");
86  QL_REQUIRE(probability_->referenceDate() == evalDate,
87  "probability term structure reference date ("
88  << probability_->referenceDate()
89  << " should be evaluation date (" << evalDate << ")");
90  QL_REQUIRE(arguments_.settlesAccrual,
91  "ISDA engine not compatible with non accrual paying CDS");
92  QL_REQUIRE(arguments_.paysAtDefaultTime,
93  "ISDA engine not compatible with end period payment");
94  QL_REQUIRE(ext::dynamic_pointer_cast<FaceValueClaim>(
95  arguments_.claim) != NULL,
96  "ISDA engine not compatible with non face value claim");
97 
98  Date maturity = arguments_.maturity;
99  Date effectiveProtectionStart =
100  std::max<Date>(arguments_.protectionStart, evalDate + 1);
101 
102  // collect nodes from both curves and sort them
103  std::vector<Date> yDates, cDates;
104 
105  if(ext::shared_ptr<InterpolatedDiscountCurve<LogLinear> > castY1 =
106  ext::dynamic_pointer_cast<
107  InterpolatedDiscountCurve<LogLinear> >(*discountCurve_)) {
108  yDates = castY1->dates();
109  } else if(ext::shared_ptr<InterpolatedForwardCurve<BackwardFlat> >
110  castY2 = ext::dynamic_pointer_cast<
111  InterpolatedForwardCurve<BackwardFlat> >(*discountCurve_)) {
112  yDates = castY2->dates();
113  } else if(ext::shared_ptr<InterpolatedForwardCurve<ForwardFlat> >
114  castY3 = ext::dynamic_pointer_cast<
115  InterpolatedForwardCurve<ForwardFlat> >(*discountCurve_)) {
116  yDates = castY3->dates();
117  } else if(ext::shared_ptr<FlatForward> castY4 =
118  ext::dynamic_pointer_cast<FlatForward>(*discountCurve_)) {
119  // no dates to extract
120  } else {
121  QL_FAIL("Yield curve must be flat forward interpolated");
122  }
123 
124  if(ext::shared_ptr<InterpolatedSurvivalProbabilityCurve<LogLinear> >
125  castC1 = ext::dynamic_pointer_cast<
126  InterpolatedSurvivalProbabilityCurve<LogLinear> >(
127  *probability_)) {
128  cDates = castC1->dates();
129  } else if(
130  ext::shared_ptr<InterpolatedHazardRateCurve<BackwardFlat> > castC2 =
131  ext::dynamic_pointer_cast<
132  InterpolatedHazardRateCurve<BackwardFlat> >(*probability_)) {
133  cDates = castC2->dates();
134  } else if(
135  ext::shared_ptr<FlatHazardRate> castC3 =
136  ext::dynamic_pointer_cast<FlatHazardRate>(*probability_)) {
137  // no dates to extract
138  } else{
139  QL_FAIL("Credit curve must be flat forward interpolated");
140  }
141 
142  std::vector<Date> nodes;
143  std::set_union(yDates.begin(), yDates.end(), cDates.begin(), cDates.end(), std::back_inserter(nodes));
144 
145 
146  if(nodes.empty()){
147  nodes.push_back(maturity);
148  }
149  const Real nFix = (numericalFix_ == None ? 1E-50 : 0.0);
150 
151  // protection leg pricing (npv is always negative at this stage)
152  Real protectionNpv = 0.0;
153 
154  Date d0 = effectiveProtectionStart-1;
155  Real P0 = discountCurve_->discount(d0);
156  Real Q0 = probability_->survivalProbability(d0);
157  Date d1;
158  std::vector<Date>::const_iterator it =
159  std::upper_bound(nodes.begin(), nodes.end(), effectiveProtectionStart);
160 
161  for(;it != nodes.end(); ++it) {
162  if(*it > maturity) {
163  d1 = maturity;
164  it = nodes.end() - 1; //early exit
165  } else {
166  d1 = *it;
167  }
168  Real P1 = discountCurve_->discount(d1);
169  Real Q1 = probability_->survivalProbability(d1);
170 
171  Real fhat = std::log(P0) - std::log(P1);
172  Real hhat = std::log(Q0) - std::log(Q1);
173  Real fhphh = fhat + hhat;
174 
175  if (fhphh < 1E-4 && numericalFix_ == Taylor) {
176  Real fhphhq = fhphh * fhphh;
177  protectionNpv +=
178  P0 * Q0 * hhat * (1.0 - 0.5 * fhphh + 1.0 / 6.0 * fhphhq -
179  1.0 / 24.0 * fhphhq * fhphh +
180  1.0 / 120 * fhphhq * fhphhq);
181  } else {
182  protectionNpv += hhat / (fhphh + nFix) * (P0 * Q0 - P1 * Q1);
183  }
184  d0 = d1;
185  P0 = P1;
186  Q0 = Q1;
187  }
188  protectionNpv *= arguments_.claim->amount(
189  Null<Date>(), arguments_.notional, recoveryRate_);
190 
191  results_.defaultLegNPV = protectionNpv;
192 
193  // premium leg pricing (npv is always positive at this stage)
194 
195  Real premiumNpv = 0.0, defaultAccrualNpv = 0.0;
196  for (Size i = 0; i < arguments_.leg.size(); ++i) {
197  ext::shared_ptr<FixedRateCoupon> coupon =
198  ext::dynamic_pointer_cast<FixedRateCoupon>(arguments_.leg[i]);
199 
200  QL_REQUIRE(coupon->dayCounter() == dc ||
201  coupon->dayCounter() == dc1 ||
202  coupon->dayCounter() == dc2,
203  "ISDA engine requires a coupon day counter Act/365Fixed "
204  << "or Act/360 (" << coupon->dayCounter() << ")");
205 
206  // premium coupons
207  if (!arguments_.leg[i]->hasOccurred(effectiveProtectionStart,
208  includeSettlementDateFlows_)) {
209  premiumNpv +=
210  coupon->amount() *
211  discountCurve_->discount(coupon->date()) *
212  probability_->survivalProbability(coupon->date()-1);
213  }
214 
215  // default accruals
216 
217  if (!detail::simple_event(coupon->accrualEndDate())
218  .hasOccurred(effectiveProtectionStart, false)) {
219  Date start = std::max<Date>(coupon->accrualStartDate(),
220  effectiveProtectionStart)-1;
221  Date end = coupon->date()-1;
222  Real tstart =
223  discountCurve_->timeFromReference(coupon->accrualStartDate()-1) -
224  (accrualBias_ == HalfDayBias ? 1.0 / 730.0 : 0.0);
225  std::vector<Date> localNodes;
226  localNodes.push_back(start);
227  //add intermediary nodes, if any
228  if (forwardsInCouponPeriod_ == Piecewise) {
229  std::vector<Date>::const_iterator it0 =
230  std::upper_bound(nodes.begin(), nodes.end(), start);
231  std::vector<Date>::const_iterator it1 =
232  std::lower_bound(nodes.begin(), nodes.end(), end);
233  localNodes.insert(localNodes.end(), it0, it1);
234  }
235  localNodes.push_back(end);
236 
237  Real defaultAccrThisNode = 0.;
238  std::vector<Date>::const_iterator node = localNodes.begin();
239  Real t0 = discountCurve_->timeFromReference(*node);
240  Real P0 = discountCurve_->discount(*node);
241  Real Q0 = probability_->survivalProbability(*node);
242 
243  for (++node; node != localNodes.end(); ++node) {
244  Real t1 = discountCurve_->timeFromReference(*node);
245  Real P1 = discountCurve_->discount(*node);
246  Real Q1 = probability_->survivalProbability(*node);
247  Real fhat = std::log(P0) - std::log(P1);
248  Real hhat = std::log(Q0) - std::log(Q1);
249  Real fhphh = fhat + hhat;
250  if (fhphh < 1E-4 && numericalFix_ == Taylor) {
251  // see above, terms up to (f+h)^3 seem more than enough,
252  // what exactly is implemented in the standard isda C
253  // code ?
254  Real fhphhq = fhphh * fhphh;
255  defaultAccrThisNode +=
256  hhat * P0 * Q0 *
257  ((t0 - tstart) *
258  (1.0 - 0.5 * fhphh + 1.0 / 6.0 * fhphhq -
259  1.0 / 24.0 * fhphhq * fhphh) +
260  (t1 - t0) *
261  (0.5 - 1.0 / 3.0 * fhphh + 1.0 / 8.0 * fhphhq -
262  1.0 / 30.0 * fhphhq * fhphh));
263  } else {
264  defaultAccrThisNode +=
265  (hhat / (fhphh + nFix)) *
266  ((t1 - t0) * ((P0 * Q0 - P1 * Q1) / (fhphh + nFix) -
267  P1 * Q1) +
268  (t0 - tstart) * (P0 * Q0 - P1 * Q1));
269  }
270 
271  t0 = t1;
272  P0 = P1;
273  Q0 = Q1;
274  }
275  defaultAccrualNpv += defaultAccrThisNode * arguments_.notional *
276  coupon->rate() * 365. / 360.;
277  }
278  }
279 
280 
281  results_.couponLegNPV = premiumNpv + defaultAccrualNpv;
282 
283  // upfront flow npv
284 
285  Real upfPVO1 = 0.0;
286  results_.upfrontNPV = 0.0;
287  if (!arguments_.upfrontPayment->hasOccurred(
288  evalDate, includeSettlementDateFlows_)) {
289  upfPVO1 =
290  discountCurve_->discount(arguments_.upfrontPayment->date());
291  if(arguments_.upfrontPayment->amount() != 0.) {
292  results_.upfrontNPV = upfPVO1 * arguments_.upfrontPayment->amount();
293  }
294  }
295 
296  results_.accrualRebateNPV = 0.;
297  // NOLINTNEXTLINE(readability-implicit-bool-conversion)
298  if (arguments_.accrualRebate && arguments_.accrualRebate->amount() != 0. &&
299  !arguments_.accrualRebate->hasOccurred(evalDate, includeSettlementDateFlows_)) {
300  results_.accrualRebateNPV =
301  discountCurve_->discount(arguments_.accrualRebate->date()) *
302  arguments_.accrualRebate->amount();
303  }
304 
305  Real upfrontSign = Protection::Seller != 0U ? 1.0 : -1.0;
306 
307  if (arguments_.side == Protection::Seller) {
308  results_.defaultLegNPV *= -1.0;
309  results_.accrualRebateNPV *= -1.0;
310  } else {
311  results_.couponLegNPV *= -1.0;
312  results_.upfrontNPV *= -1.0;
313  }
314 
315  results_.value = results_.defaultLegNPV + results_.couponLegNPV +
316  results_.upfrontNPV + results_.accrualRebateNPV;
317 
318  results_.errorEstimate = Null<Real>();
319 
320  if (results_.couponLegNPV != 0.0) {
321  results_.fairSpread =
322  -results_.defaultLegNPV * arguments_.spread /
323  (results_.couponLegNPV + results_.accrualRebateNPV);
324  } else {
325  results_.fairSpread = Null<Rate>();
326  }
327 
328  Real upfrontSensitivity = upfPVO1 * arguments_.notional;
329  if (upfrontSensitivity != 0.0) {
330  results_.fairUpfront =
331  -upfrontSign * (results_.defaultLegNPV + results_.couponLegNPV +
332  results_.accrualRebateNPV) /
333  upfrontSensitivity;
334  } else {
335  results_.fairUpfront = Null<Rate>();
336  }
337 
338  static const Rate basisPoint = 1.0e-4;
339 
340  if (arguments_.spread != 0.0) {
341  results_.couponLegBPS =
342  results_.couponLegNPV * basisPoint / arguments_.spread;
343  } else {
344  results_.couponLegBPS = Null<Rate>();
345  }
346 
347  // NOLINTNEXTLINE(readability-implicit-bool-conversion)
348  if (arguments_.upfront && *arguments_.upfront != 0.0) {
349  results_.upfrontBPS =
350  results_.upfrontNPV * basisPoint / (*arguments_.upfront);
351  } else {
352  results_.upfrontBPS = Null<Rate>();
353  }
354  }
355 }
