99 void Gsr::updateTimes() const {
100  volsteptimes_.clear();
101  int j = 0;
102  for (std::vector<Date>::const_iterator i = volstepdates_.begin();
103  i != volstepdates_.end(); ++i, ++j) {
104  volsteptimes_.push_back(termStructure()->timeFromReference(*i));
105  volsteptimesArray_[j] = volsteptimes_[j];
106  if (j == 0)
107  QL_REQUIRE(volsteptimes_[0] > 0.0, "volsteptimes must be positive ("
108  << volsteptimes_[0] << ")");
109  else
110  QL_REQUIRE(volsteptimes_[j] > volsteptimes_[j - 1],
111  "volsteptimes must be strictly increasing ("
112  << volsteptimes_[j - 1] << "@" << (j - 1) << ", "
113  << volsteptimes_[j] << "@" << j << ")");
114  }
115  if (stateProcess_ != NULL)
116  ext::static_pointer_cast<GsrProcess>(stateProcess_)->flushCache();
117 }

133 void Gsr::initialize(Real T) {
134 
135  volsteptimesArray_ = Array(volstepdates_.size());
136 
137  updateTimes();
138 
139  QL_REQUIRE(volatilities_.size() == volsteptimes_.size() + 1,
140  "there must be n+1 volatilities ("
141  << volatilities_.size() << ") for n volatility step times ("
142  << volsteptimes_.size() << ")");
143 
144  QL_REQUIRE(reversions_.size() == 1 ||
145  reversions_.size() == volsteptimes_.size() + 1,
146  "there must be 1 or n+1 reversions ("
147  << reversions_.size() << ") for n volatility step times ("
148  << volsteptimes_.size() << ")");
149  if (reversions_.size() == 1) {
150  reversion_ = ConstantParameter(reversions_[0]->value(), NoConstraint());
151  } else {
152  reversion_ = PiecewiseConstantParameter(volsteptimes_, NoConstraint());
153  for (Size i = 0; i < reversion_.size(); i++) {
154  reversion_.setParam(i, reversions_[i]->value());
155  }
156  }
157 
158  // sigma_ =
159  // PiecewiseConstantParameter(volsteptimes_,PositiveConstraint());
160  sigma_ = PiecewiseConstantParameter(volsteptimes_, NoConstraint());
161  for (Size i = 0; i < sigma_.size(); i++) {
162  sigma_.setParam(i, volatilities_[i]->value());
163  }
164 
165  stateProcess_ = ext::make_shared<GsrProcess>(
166  volsteptimesArray_, sigma_.params(), reversion_.params(), T);
167 
168  registerWith(termStructure());
169 
170  registerWith(stateProcess_);
171 
172  volatilityObserver_ = ext::make_shared<VolatilityObserver>(this);
173  reversionObserver_ = ext::make_shared<ReversionObserver>(this);
174 
175  for (Size i = 0; i < reversions_.size(); ++i)
176  reversionObserver_->registerWith(reversions_[i]);
177 
178  for (Size i = 0; i < volatilities_.size(); ++i)
179  volatilityObserver_->registerWith(volatilities_[i]);
180 }

182 Real Gsr::zerobondImpl(const Time T, const Time t, const Real y,
183  const Handle<YieldTermStructure> &yts) const {
184 
185  calculate();
186 
187  if (t == 0.0)
188  return yts.empty() ? this->termStructure()->discount(T, true)
189  : yts->discount(T, true);
190 
191  ext::shared_ptr<GsrProcess> p =
192  ext::dynamic_pointer_cast<GsrProcess>(stateProcess_);
193 
194  Real x = y * stateProcess_->stdDeviation(0.0, 0.0, t) +
195  stateProcess_->expectation(0.0, 0.0, t);
196  Real gtT = p->G(t, T, x);
197 
198  Real d = yts.empty()
199  ? termStructure()->discount(T, true) /
200  termStructure()->discount(t, true)
201  : yts->discount(T, true) / yts->discount(t, true);
202 
203  return d * exp(-x * gtT - 0.5 * p->y(t) * gtT * gtT);
204 }
