36  class MCLookbackEngine : public I::engine,
37  public McSimulation<SingleVariate,RNG,S> {
38  public:

46  Size timeSteps,

48  bool brownianBridge,
49  bool antithetic,

54  void calculate() const {
55  Real spot = process_->x0();
56  QL_REQUIRE(spot >= 0.0, "negative or null underlying given");
57  McSimulation<SingleVariate,RNG,S>::calculate(requiredTolerance_,
58  requiredSamples_,
59  maxSamples_);
60  this->results_.value = this->mcModel_->sampleAccumulator().mean();
61  if (RNG::allowsErrorEstimate)
62  this->results_.errorEstimate =
63  this->mcModel_->sampleAccumulator().errorEstimate();
64  }
65  protected:
66  // McSimulation implementation
67  TimeGrid timeGrid() const;
68  ext::shared_ptr<path_generator_type> pathGenerator() const {
69  TimeGrid grid = timeGrid();
70  typename RNG::rsg_type gen =
71  RNG::make_sequence_generator(grid.size()-1,seed_);
72  return ext::shared_ptr<path_generator_type>(
73  new path_generator_type(process_,
74  grid, gen, brownianBridge_));
75  }
76  ext::shared_ptr<path_pricer_type> pathPricer() const;
77  // data members

79  Size timeSteps_, timeStepsPerYear_;
80  Size requiredSamples_, maxSamples_;
81  Real requiredTolerance_;
82  bool antithetic_;
83  bool brownianBridge_;
84  BigNatural seed_;
85  };

116  template <class I, class RNG, class S>
117  inline MCLookbackEngine<I, RNG,S>::MCLookbackEngine(
118  const ext::shared_ptr<GeneralizedBlackScholesProcess>& process,
119  Size timeSteps,
120  Size timeStepsPerYear,
121  bool brownianBridge,
122  bool antitheticVariate,
123  Size requiredSamples,

125  Size maxSamples,
126  BigNatural seed)
127  : McSimulation<SingleVariate,RNG,S>(antitheticVariate, false),

129  timeStepsPerYear_(timeStepsPerYear),
130  requiredSamples_(requiredSamples), maxSamples_(maxSamples),
131  requiredTolerance_(requiredTolerance),

133  QL_REQUIRE(timeSteps != Null<Size>() ||
134  timeStepsPerYear != Null<Size>(),

137  timeStepsPerYear == Null<Size>(),

139  QL_REQUIRE(timeSteps != 0,
140  "timeSteps must be positive, " << timeSteps <<
141  " not allowed");
142  QL_REQUIRE(timeStepsPerYear != 0,
143  "timeStepsPerYear must be positive, " << timeStepsPerYear <<
144  " not allowed");
145  this->registerWith(process_);
146  }

149  template <class I, class RNG, class S>
150  inline TimeGrid MCLookbackEngine<I,RNG,S>::timeGrid() const {
151 
152  Time residualTime = process_->time(this->arguments_.exercise->lastDate());
153  if (timeSteps_ != Null<Size>()) {
154  return TimeGrid(residualTime, timeSteps_);
155  } else if (timeStepsPerYear_ != Null<Size>()) {
156  Size steps = static_cast<Size>(timeStepsPerYear_*residualTime);
157  return TimeGrid(residualTime, std::max<Size>(steps, 1));
158  } else {
159  QL_FAIL("time steps not specified");
160  }
161  }
