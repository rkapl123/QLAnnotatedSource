41  FdmHestonFwdOp::FdmHestonFwdOp(const ext::shared_ptr<FdmMesher>& mesher,
42  const ext::shared_ptr<HestonProcess>& process,
43  FdmSquareRootFwdOp::TransformationType type,
44  ext::shared_ptr<LocalVolTermStructure> leverageFct,
45  const Real mixingFactor)
46  : type_(type), kappa_(process->kappa()), theta_(process->theta()), sigma_(process->sigma()),
47  rho_(process->rho()), v0_(process->v0()), mixedSigma_(mixingFactor * sigma_),
48  rTS_(process->riskFreeRate().currentLink()), qTS_(process->dividendYield().currentLink()),
49  varianceValues_(0.5 * mesher->locations(1)),
50  dxMap_(ext::make_shared<FirstDerivativeOp>(0, mesher)),
51  dxxMap_(ext::make_shared<ModTripleBandLinearOp>(TripleBandLinearOp(
52  type == FdmSquareRootFwdOp::Log ?
53  SecondDerivativeOp(0, mesher).mult(0.5 * Exp(mesher->locations(1))) :
54  SecondDerivativeOp(0, mesher).mult(0.5 * mesher->locations(1))))),
55  boundary_(ext::make_shared<ModTripleBandLinearOp>(TripleBandLinearOp(
56  SecondDerivativeOp(0, mesher).mult(Array(mesher->locations(0).size(), 0.0))))),
57  mapX_(ext::make_shared<TripleBandLinearOp>(0, mesher)),
58  mapY_(ext::make_shared<FdmSquareRootFwdOp>(mesher, kappa_, theta_, mixedSigma_, 1, type)),
59  correlation_(ext::make_shared<NinePointLinearOp>(
60  type == FdmSquareRootFwdOp::Log ?
61  SecondOrderMixedDerivativeOp(0, 1, mesher)
62  .mult(Array(mesher->layout()->size(), rho_ * mixedSigma_)) :
63  SecondOrderMixedDerivativeOp(0, 1, mesher)
64  .mult(rho_ * mixedSigma_ * mesher->locations(1)))),
65  leverageFct_(std::move(leverageFct)), mesher_(mesher) {
66  const ext::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
67  // zero flux boundary condition
68  const Size n = layout->dim()[1];
69  const Real lowerBoundaryFactor = mapY_->lowerBoundaryFactor(type);
70  const Real upperBoundaryFactor = mapY_->upperBoundaryFactor(type);
71 
72  const Real logFacLow = type == FdmSquareRootFwdOp::Log ? exp(mapY_->v(0)) : 1.0;
73  const Real logFacUpp = type == FdmSquareRootFwdOp::Log ? exp(mapY_->v(n+1)) : 1.0;
74 
75  const Real alpha = -2*rho_/mixedSigma_*lowerBoundaryFactor*logFacLow;
76  const Real beta = -2*rho_/mixedSigma_*upperBoundaryFactor*logFacUpp;
77 
78  ModTripleBandLinearOp fDx(FirstDerivativeOp(0, mesher));
79 
80  const FdmLinearOpIterator endIter = layout->end();
81  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter; ++iter) {
82  if (iter.coordinates()[1] == 0) {
83  const Size idx = iter.index();
84  if (!leverageFct_) {
85  dxxMap_->upper(idx) += alpha*fDx.upper(idx);
86  dxxMap_->diag(idx) += alpha*fDx.diag(idx);
87  dxxMap_->lower(idx) += alpha*fDx.lower(idx);
88  }
89  boundary_->upper(idx)= alpha*fDx.upper(idx);
90  boundary_->diag(idx) = alpha*fDx.diag(idx);
91  boundary_->lower(idx) = alpha*fDx.lower(idx);
92  }
93  else if (iter.coordinates()[1] == n-1) {
94  const Size idx = iter.index();
95 
96  if (!leverageFct_) {
97  dxxMap_->upper(idx)+= beta*fDx.upper(idx);
98  dxxMap_->diag(idx) += beta*fDx.diag(idx);
99  dxxMap_->lower(idx) += beta*fDx.lower(idx);
100  }
101  boundary_->upper(idx)= beta*fDx.upper(idx);
102  boundary_->diag(idx) = beta*fDx.diag(idx);
103  boundary_->lower(idx) = beta*fDx.lower(idx);
104  }
105  }
106  }

112  void FdmHestonFwdOp::setTime(Time t1, Time t2){
113  const Rate r = rTS_->forwardRate(t1, t2, Continuous).rate();
114  const Rate q = qTS_->forwardRate(t1, t2, Continuous).rate();
115  if (leverageFct_ != nullptr) {
116  L_ = getLeverageFctSlice(t1, t2);
117  Array Lsquare = L_*L_;
118  if (type_ == FdmSquareRootFwdOp::Plain) {
119  mapX_->axpyb( Array(1, -r + q), *dxMap_,
120  dxxMap_->multR(Lsquare).add(boundary_->multR(L_))
121  .add(dxMap_->multR(rho_*mixedSigma_*L_))
122  .add(dxMap_->mult(varianceValues_).multR(Lsquare)),
123  Array());
124  } else if (type_ == FdmSquareRootFwdOp::Power) {
125  mapX_->axpyb( Array(1, -r + q), *dxMap_,
126  dxxMap_->multR(Lsquare).add(boundary_->multR(L_))
127  .add(dxMap_->multR(rho_*2.0*kappa_*theta_/(mixedSigma_)*L_))
128  .add(dxMap_->mult(varianceValues_).multR(Lsquare)), Array());
129  } else if (type_ == FdmSquareRootFwdOp::Log) {
130  mapX_->axpyb( Array(1, -r + q), *dxMap_,
131  dxxMap_->multR(Lsquare).add(boundary_->multR(L_))
132  .add(dxMap_->mult(0.5*Exp(2.0*varianceValues_)).multR(Lsquare)),
133  Array());
134  }
135  } else {
136  if (type_ == FdmSquareRootFwdOp::Plain) {
137  mapX_->axpyb( - r + q + rho_*mixedSigma_ + varianceValues_, *dxMap_,
138  *dxxMap_, Array());
139  } else if (type_ == FdmSquareRootFwdOp::Power) {
140  mapX_->axpyb( - r + q + rho_*2.0*kappa_*theta_/(mixedSigma_) + varianceValues_,
141  *dxMap_, *dxxMap_, Array());
142  } else if (type_ == FdmSquareRootFwdOp::Log) {
143  mapX_->axpyb( - r + q + 0.5*Exp(2.0*varianceValues_), *dxMap_,
144  *dxxMap_, Array());
145  }
146  }
147  }

197  Disposable<Array> FdmHestonFwdOp::getLeverageFctSlice(Time t1, Time t2)
198  const {
199  const ext::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
200  Array v(layout->size(), 1.0);
201 
202  if (!leverageFct_)
203  return v;
204 
205  const Real t = 0.5*(t1+t2);
206  const Time time = std::min(leverageFct_->maxTime(), t);
207  //std::max(leverageFct_->minTime(), t));
208 
209  const FdmLinearOpIterator endIter = layout->end();
210  for (FdmLinearOpIterator iter = layout->begin();
211  iter!=endIter; ++iter) {
212  const Size nx = iter.coordinates()[0];
213 
214  if (iter.coordinates()[1] == 0) {
215  const Real x = std::exp(mesher_->location(iter, 0));
216  const Real spot = std::min(leverageFct_->maxStrike(),
217  std::max(leverageFct_->minStrike(), x));
218  v[nx] = std::max(0.01, leverageFct_->localVol(time, spot, true));
219  }
220  else {
221  v[iter.index()] = v[nx];
222  }
223  }
224  return v;
225  }
