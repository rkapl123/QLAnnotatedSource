41  FdmHestonFwdOp::FdmHestonFwdOp(
42  const boost::shared_ptr<FdmMesher>& mesher,
43  const boost::shared_ptr<HestonProcess>& process,
44  FdmSquareRootFwdOp::TransformationType type,
45  const boost::shared_ptr<LocalVolTermStructure>& leverageFct)
46  : type_(type),
47  kappa_(process->kappa()),
48  theta_(process->theta()),
49  sigma_(process->sigma()),
50  rho_ (process->rho()),
51  v0_ (process->v0()),
52  rTS_ (process->riskFreeRate().currentLink()),
53  qTS_ (process->dividendYield().currentLink()),
54  varianceValues_(0.5*mesher->locations(1)),
55  dxMap_ (boost::make_shared<FirstDerivativeOp>(0, mesher)),
56  dxxMap_(boost::make_shared<ModTripleBandLinearOp>(TripleBandLinearOp(
57  type == FdmSquareRootFwdOp::Log ?
58  SecondDerivativeOp(0, mesher).mult(0.5*Exp(mesher->locations(1)))
59  : SecondDerivativeOp(0, mesher).mult(0.5*mesher->locations(1))
60  ))),
61  boundary_(boost::make_shared<ModTripleBandLinearOp>(TripleBandLinearOp(SecondDerivativeOp(0, mesher).mult(Array(mesher->locations(0).size(), 0.0))))),
62  mapX_ (boost::make_shared<TripleBandLinearOp>(0, mesher)),
63  mapY_ (boost::make_shared<FdmSquareRootFwdOp>(mesher,kappa_,theta_,sigma_, 1, type)),
64  correlation_(boost::make_shared<NinePointLinearOp>(
65  type == FdmSquareRootFwdOp::Log ?
66  SecondOrderMixedDerivativeOp(0, 1, mesher)
67  .mult(Array(mesher->layout()->size(), rho_*sigma_))
68  : SecondOrderMixedDerivativeOp(0, 1, mesher)
69  .mult(rho_*sigma_*mesher->locations(1))
70  )),
71  leverageFct_(leverageFct),
72  mesher_(mesher)
73  {
74  const boost::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
75  // zero flux boundary condition
76  const Size n = layout->dim()[1];
77  const Real lowerBoundaryFactor = mapY_->lowerBoundaryFactor(type);
78  const Real upperBoundaryFactor = mapY_->upperBoundaryFactor(type);
79 
80  const Real logFacLow = type == FdmSquareRootFwdOp::Log ? exp(mapY_->v(0)) : 1.0;
81  const Real logFacUpp = type == FdmSquareRootFwdOp::Log ? exp(mapY_->v(n+1)) : 1.0;
82 
83  const Real alpha = -2*rho_/sigma_*lowerBoundaryFactor*logFacLow;
84  const Real beta = -2*rho_/sigma_*upperBoundaryFactor*logFacUpp;
85 
86  ModTripleBandLinearOp fDx(FirstDerivativeOp(0, mesher));
87 
88  const FdmLinearOpIterator endIter = layout->end();
89  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter; ++iter) {
90  if (iter.coordinates()[1] == 0) {
91  const Size idx = iter.index();
92  if (!leverageFct_) {
93  dxxMap_->upper()[idx]+= alpha*fDx.upper()[idx];
94  dxxMap_->diag()[idx] += alpha*fDx.diag()[idx];
95  dxxMap_->lower()[idx] += alpha*fDx.lower()[idx];
96  }
97  boundary_->upper()[idx]= alpha*fDx.upper()[idx];
98  boundary_->diag()[idx] = alpha*fDx.diag()[idx];
99  boundary_->lower()[idx] = alpha*fDx.lower()[idx];
100  }
101  else if (iter.coordinates()[1] == n-1) {
102  const Size idx = iter.index();
103 
104  if (!leverageFct_) {
105  dxxMap_->upper()[idx]+= beta*fDx.upper()[idx];
106  dxxMap_->diag()[idx] += beta*fDx.diag()[idx];
107  dxxMap_->lower()[idx] += beta*fDx.lower()[idx];
108  }
109  boundary_->upper()[idx]= beta*fDx.upper()[idx];
110  boundary_->diag()[idx] = beta*fDx.diag()[idx];
111  boundary_->lower()[idx] = beta*fDx.lower()[idx];
112  }
113  }
114  }

120  void FdmHestonFwdOp::setTime(Time t1, Time t2){
121  const Rate r = rTS_->forwardRate(t1, t2, Continuous).rate();
122  const Rate q = qTS_->forwardRate(t1, t2, Continuous).rate();
123  if (leverageFct_) {
124  L_ = getLeverageFctSlice(t1, t2);
125  Array Lsquare = L_*L_;
126  if (type_ == FdmSquareRootFwdOp::Plain) {
127  mapX_->axpyb( Array(1, -r + q), *dxMap_,
128  dxxMap_->multR(Lsquare).add(boundary_->multR(L_))
129  .add(dxMap_->multR(rho_*sigma_*L_))
130  .add(dxMap_->mult(varianceValues_).multR(Lsquare)),
131  Array());
132  } else if (type_ == FdmSquareRootFwdOp::Power) {
133  mapX_->axpyb( Array(1, -r + q), *dxMap_,
134  dxxMap_->multR(Lsquare).add(boundary_->multR(L_))
135  .add(dxMap_->multR(rho_*2.0*kappa_*theta_/(sigma_)*L_))
136  .add(dxMap_->mult(varianceValues_).multR(Lsquare)), Array());
137  } else if (type_ == FdmSquareRootFwdOp::Log) {
138  mapX_->axpyb( Array(1, -r + q), *dxMap_,
139  dxxMap_->multR(Lsquare).add(boundary_->multR(L_))
140  .add(dxMap_->mult(0.5*Exp(2.0*varianceValues_)).multR(Lsquare)),
141  Array());
142  }
143  }
144  else {
145  if (type_ == FdmSquareRootFwdOp::Plain) {
146  mapX_->axpyb( - r + q + rho_*sigma_ + varianceValues_, *dxMap_,
147  *dxxMap_, Array());
148  } else if (type_ == FdmSquareRootFwdOp::Power) {
149  mapX_->axpyb( - r + q + rho_*2.0*kappa_*theta_/(sigma_) + varianceValues_,
150  *dxMap_, *dxxMap_, Array());
151  } else if (type_ == FdmSquareRootFwdOp::Log) {
152  mapX_->axpyb( - r + q + 0.5*Exp(2.0*varianceValues_), *dxMap_,
153  *dxxMap_, Array());
154  }
155  }
156  }

209  Disposable<Array> FdmHestonFwdOp::getLeverageFctSlice(Time t1, Time t2)
210  const {
211  const boost::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
212  Array v(layout->size(), 1.0);
213 
214  if (!leverageFct_)
215  return v;
216 
217  const Real t = 0.5*(t1+t2);
218  const Time time = std::min(leverageFct_->maxTime(), t);
219  //std::max(leverageFct_->minTime(), t));
220 
221  const FdmLinearOpIterator endIter = layout->end();
222  for (FdmLinearOpIterator iter = layout->begin();
223  iter!=endIter; ++iter) {
224  const Size nx = iter.coordinates()[0];
225 
226  if (iter.coordinates()[1] == 0) {
227  const Real x = std::exp(mesher_->location(iter, 0));
228  const Real spot = std::min(leverageFct_->maxStrike(),
229  std::max(leverageFct_->minStrike(), x));
230  v[nx] = std::max(0.01, leverageFct_->localVol(time, spot, true));
231  }
232  else {
233  v[iter.index()] = v[nx];
234  }
235  }
236  return v;
237  }
