60  Real FixedRateBondForward::spotIncome(const Handle<YieldTermStructure>&
61  incomeDiscountCurve) const {
62 
63  Real income = 0.0;
64  Date settlement = settlementDate();
65  Leg cf =
66  fixedCouponBond_->cashflows();
67 
68  /*
69  the following assumes
70  1. cashflows are in ascending order !
71  2. considers as income: all coupons paid between settlementDate()
72  and contract delivery/maturity date
73  */
74  for (Size i = 0; i < cf.size(); ++i) {
75  if (!cf[i]->hasOccurred(settlement, false)) {
76  if (cf[i]->hasOccurred(maturityDate_, false)) {
77  income += cf[i]->amount() *
78  incomeDiscountCurve->discount(cf[i]->date()) ;
79  } else {
80  break;
81  }
82  }
83  }
84 
85  return income;
86  }
