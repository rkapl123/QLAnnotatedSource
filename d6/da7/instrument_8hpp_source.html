163 inline void Instrument::performCalculations() const {
164 QL_REQUIRE(engine_, "null pricing engine");
165 engine_->reset();
166 setupArguments(engine_->getArguments());
167 engine_->getArguments()->validate();
168 engine_->calculate();
169 fetchResults(engine_->getResults());
170 }

172 inline void Instrument::fetchResults(
173 const PricingEngine::results* r) const {
174 const auto* results = dynamic_cast<const Instrument::results*>(r);
175 QL_ENSURE(results != nullptr, "no results returned from pricing engine");
176
177 NPV_ = results->value;
178 errorEstimate_ = results->errorEstimate;
179 valuationDate_ = results->valuationDate;
180
181 additionalResults_ = results->additionalResults;
182 }

204 template <class T>
205 inline T Instrument::result(const std::string& tag) const {
206 calculate();
207 std::map<std::string, ext::any>::const_iterator value =
208 additionalResults_.find(tag);
209 QL_REQUIRE(value != additionalResults_.end(),
210 tag << " not provided");
211 return ext::any_cast<T>(value->second);
212 }
