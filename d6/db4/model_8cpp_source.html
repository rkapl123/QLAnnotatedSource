37  class CalibratedModel::CalibrationFunction : public CostFunction {
38  public:
39  CalibrationFunction(CalibratedModel* model,
40  const vector<ext::shared_ptr<CalibrationHelper> >& h,
41  vector<Real> weights,
42  const Projection& projection)
43  : model_(model, null_deleter()), instruments_(h), weights_(std::move(weights)),
44  projection_(projection) {}

46  ~CalibrationFunction() override = default;
47 
48  Real value(const Array& params) const override {
49  model_->setParams(projection_.include(params));
50  Real value = 0.0;
51  for (Size i=0; i<instruments_.size(); i++) {
52  Real diff = instruments_[i]->calibrationError();
53  value += diff*diff*weights_[i];
54  }
55  return std::sqrt(value);
56  }

58  Disposable<Array> values(const Array& params) const override {
59  model_->setParams(projection_.include(params));
60  Array values(instruments_.size());
61  for (Size i=0; i<instruments_.size(); i++) {
62  values[i] = instruments_[i]->calibrationError()
63  *std::sqrt(weights_[i]);
64  }
65  return values;
66  }

77  void CalibratedModel::calibrate(
78  const vector<ext::shared_ptr<CalibrationHelper> >& instruments,
79  OptimizationMethod& method,
80  const EndCriteria& endCriteria,
81  const Constraint& additionalConstraint,
82  const vector<Real>& weights,
83  const vector<bool>& fixParameters) {
84 
85  QL_REQUIRE(!instruments.empty(), "no instruments provided");
86 
87  Constraint c;
88  if (additionalConstraint.empty())
89  c = *constraint_;
90  else
91  c = CompositeConstraint(*constraint_,additionalConstraint);
92 
93  QL_REQUIRE(weights.empty() || weights.size() == instruments.size(),
94  "mismatch between number of instruments (" <<
95  instruments.size() << ") and weights (" <<
96  weights.size() << ")");
97  vector<Real> w =
98  weights.empty() ? vector<Real>(instruments.size(), 1.0): weights;
99 
100  Array prms = params();
101  QL_REQUIRE(fixParameters.empty() || fixParameters.size() == prms.size(),
102  "mismatch between number of parameters (" <<
103  prms.size() << ") and fixed-parameter specs (" <<
104  fixParameters.size() << ")");
105  vector<bool> all(prms.size(), false);
106  Projection proj(prms, !fixParameters.empty() ? fixParameters : all);
107  CalibrationFunction f(this,instruments,w,proj);
108  ProjectedConstraint pc(c,proj);
109  Problem prob(f, pc, proj.project(prms));
110  shortRateEndCriteria_ = method.minimize(prob, endCriteria);
111  Array result(prob.currentValue());
112  setParams(proj.include(result));
113  problemValues_ = prob.values(result);
114  functionEvaluation_ = prob.functionEvaluation();
115 
116  notifyObservers();
117  }
