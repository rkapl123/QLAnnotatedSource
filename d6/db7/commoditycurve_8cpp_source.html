37 
38  QL_REQUIRE(dates_.size()>1, "too few dates");
39  QL_REQUIRE(data_.size()==dates_.size(), "dates/prices count mismatch");

41  times_.resize(dates_.size());
42  times_[0]=0.0;

47  times_[i] = dayCounter.yearFraction(dates_[0], dates_[i]);
48  }
49 
50  interpolation_ =
51  interpolator_.interpolate(times_.begin(), times_.end(),
52  data_.begin());
53  interpolation_.update();
54  }

67  void CommodityCurve::setPrices(std::map<Date, Real>& prices) {
68  QL_REQUIRE(prices.size()>1, "too few prices");
69 
70  dates_.clear();
71  data_.clear();
72  for (std::map<Date, Real>::const_iterator i = prices.begin(); i != prices.end(); ++i) {
73  dates_.push_back(i->first);
74  data_.push_back(i->second);
75  }
76 
77  times_.resize(dates_.size());
78  times_[0]=0.0;
79  for (Size i = 1; i < dates_.size(); i++)
80  times_[i] = dayCounter().yearFraction(dates_[0], dates_[i]);
81 
82  interpolation_ =
83  interpolator_.interpolate(times_.begin(), times_.end(),
84  data_.begin());
85  interpolation_.update();
86  }
