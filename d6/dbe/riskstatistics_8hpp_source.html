152  template <class S>
153  Real GenericRiskStatistics<S>::regret(Real target) const {
154  // average over the range below the target
155  std::pair<Real,Size> result =
156  this->expectationValue(compose(square<Real>(),
157  subtract<Real>(target)),
158  less_than<Real>(target));
159  Real x = result.first;
160  Size N = result.second;
161  QL_REQUIRE(N > 1,
162  "samples under target <= 1, unsufficient");
163  return (N/(N-1.0))*x;
164  }

189  template <class S>
190  Real GenericRiskStatistics<S>::expectedShortfall(Real centile) const {
191  QL_REQUIRE(centile>=0.9 && centile<1.0,
192  "percentile (" << centile << ") out of range [0.9, 1.0)");
193 
194  QL_ENSURE(this->samples() != 0, "empty sample set");
195  Real target = -valueAtRisk(centile);
196  std::pair<Real,Size> result =
197  this->expectationValue(identity<Real>(),
198  less_than<Real>(target));
199  Real x = result.first;
200  Size N = result.second;
201  QL_ENSURE(N != 0, "no data below the target");
202  // must be a loss, i.e., capped at 0.0 and negated
203  return -std::min<Real>(x, 0.0);
204  }

206  template <class S>
207  Real GenericRiskStatistics<S>::shortfall(Real target) const {
208  QL_ENSURE(this->samples() != 0, "empty sample set");
209  return this->expectationValue(clip(constant<Real,Real>(1.0),
210  less_than<Real>(target)),
211  everywhere()).first;
