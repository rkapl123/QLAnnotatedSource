105  void FloatFloatSwap::init(
106  boost::optional<BusinessDayConvention> paymentConvention1,
107  boost::optional<BusinessDayConvention> paymentConvention2) {
108 
109  QL_REQUIRE(nominal1_.size() == schedule1_.size() - 1,
110  "nominal1 size (" << nominal1_.size()
111  << ") does not match schedule1 size ("
112  << schedule1_.size() << ")");
113  QL_REQUIRE(nominal2_.size() == schedule2_.size() - 1,
114  "nominal2 size (" << nominal2_.size()
115  << ") does not match schedule2 size ("
116  << nominal2_.size() << ")");
117  QL_REQUIRE(gearing1_.empty() || gearing1_.size() == nominal1_.size(),
118  "nominal1 size (" << nominal1_.size() << ") does not match gearing1 size ("
119  << gearing1_.size() << ")");
120  QL_REQUIRE(gearing2_.empty() || gearing2_.size() == nominal2_.size(),
121  "nominal2 size (" << nominal2_.size() << ") does not match gearing2 size ("
122  << gearing2_.size() << ")");
123  QL_REQUIRE(cappedRate1_.empty() || cappedRate1_.size() == nominal1_.size(),
124  "nominal1 size (" << nominal1_.size() << ") does not match cappedRate1 size ("
125  << cappedRate1_.size() << ")");
126  QL_REQUIRE(cappedRate2_.empty() || cappedRate2_.size() == nominal2_.size(),
127  "nominal2 size (" << nominal2_.size() << ") does not match cappedRate2 size ("
128  << cappedRate2_.size() << ")");
129  QL_REQUIRE(flooredRate1_.empty() || flooredRate1_.size() == nominal1_.size(),
130  "nominal1 size (" << nominal1_.size() << ") does not match flooredRate1 size ("
131  << flooredRate1_.size() << ")");
132  QL_REQUIRE(flooredRate2_.empty() || flooredRate2_.size() == nominal2_.size(),
133  "nominal2 size (" << nominal2_.size() << ") does not match flooredRate2 size ("
134  << flooredRate2_.size() << ")");
135 
136  if (paymentConvention1) // NOLINT(readability-implicit-bool-conversion)
137  paymentConvention1_ = *paymentConvention1;
138  else
139  paymentConvention1_ = schedule1_.businessDayConvention();
140 
141  if (paymentConvention2) // NOLINT(readability-implicit-bool-conversion)
142  paymentConvention2_ = *paymentConvention2;
143  else
144  paymentConvention2_ = schedule2_.businessDayConvention();
145 
146  if (gearing1_.empty())
147  gearing1_ = std::vector<Real>(nominal1_.size(), 1.0);
148  if (gearing2_.empty())
149  gearing2_ = std::vector<Real>(nominal2_.size(), 1.0);
150  if (spread1_.empty())
151  spread1_ = std::vector<Real>(nominal1_.size(), 0.0);
152  if (spread2_.empty())
153  spread2_ = std::vector<Real>(nominal2_.size(), 0.0);
154  if (cappedRate1_.empty())
155  cappedRate1_ = std::vector<Real>(nominal1_.size(), Null<Real>());
156  if (cappedRate2_.empty())
157  cappedRate2_ = std::vector<Real>(nominal2_.size(), Null<Real>());
158  if (flooredRate1_.empty())
159  flooredRate1_ = std::vector<Real>(nominal1_.size(), Null<Real>());
160  if (flooredRate2_.empty())
161  flooredRate2_ = std::vector<Real>(nominal2_.size(), Null<Real>());
162 
163  bool isNull;
164  isNull = cappedRate1_[0] == Null<Real>();
165  for (Size i = 0; i < cappedRate1_.size(); i++) {
166  if (isNull)
167  QL_REQUIRE(cappedRate1_[i] == Null<Real>(),
168  "cappedRate1 must be null for all or none entry ("
169  << (i + 1) << "th is " << cappedRate1_[i]
170  << ")");
171  else
172  QL_REQUIRE(cappedRate1_[i] != Null<Real>(),
173  "cappedRate 1 must be null for all or none entry ("
174  << "1st is " << cappedRate1_[0] << ")");
175  }
176  isNull = cappedRate2_[0] == Null<Real>();
177  for (Size i = 0; i < cappedRate2_.size(); i++) {
178  if (isNull)
179  QL_REQUIRE(cappedRate2_[i] == Null<Real>(),
180  "cappedRate2 must be null for all or none entry ("
181  << (i + 1) << "th is " << cappedRate2_[i]
182  << ")");
183  else
184  QL_REQUIRE(cappedRate2_[i] != Null<Real>(),
185  "cappedRate2 must be null for all or none entry ("
186  << "1st is " << cappedRate2_[0] << ")");
187  }
188  isNull = flooredRate1_[0] == Null<Real>();
189  for (Size i = 0; i < flooredRate1_.size(); i++) {
190  if (isNull)
191  QL_REQUIRE(flooredRate1_[i] == Null<Real>(),
192  "flooredRate1 must be null for all or none entry ("
193  << (i + 1) << "th is " << flooredRate1_[i]
194  << ")");
195  else
196  QL_REQUIRE(flooredRate1_[i] != Null<Real>(),
197  "flooredRate 1 must be null for all or none entry ("
198  << "1st is " << flooredRate1_[0] << ")");
199  }
200  isNull = flooredRate2_[0] == Null<Real>();
201  for (Size i = 0; i < flooredRate2_.size(); i++) {
202  if (isNull)
203  QL_REQUIRE(flooredRate2_[i] == Null<Real>(),
204  "flooredRate2 must be null for all or none entry ("
205  << (i + 1) << "th is " << flooredRate2_[i]
206  << ")");
207  else
208  QL_REQUIRE(flooredRate2_[i] != Null<Real>(),
209  "flooredRate2 must be null for all or none entry ("
210  << "1st is " << flooredRate2_[0] << ")");
211  }
212 
213  // if the gearing is zero then the ibor / cms leg will be set up with
214  // fixed coupons which makes trouble here in this context. We therefore
215  // use a dirty trick and enforce the gearing to be non zero.
216  for (Size i = 0; i < gearing1_.size(); i++)
217  if (close(gearing1_[i], 0.0))
218  gearing1_[i] = QL_EPSILON;
219  for (Size i = 0; i < gearing2_.size(); i++)
220  if (close(gearing2_[i], 0.0))
221  gearing2_[i] = QL_EPSILON;
222 
223  ext::shared_ptr<IborIndex> ibor1 =
224  ext::dynamic_pointer_cast<IborIndex>(index1_);
225  ext::shared_ptr<IborIndex> ibor2 =
226  ext::dynamic_pointer_cast<IborIndex>(index2_);
227  ext::shared_ptr<SwapIndex> cms1 =
228  ext::dynamic_pointer_cast<SwapIndex>(index1_);
229  ext::shared_ptr<SwapIndex> cms2 =
230  ext::dynamic_pointer_cast<SwapIndex>(index2_);
231  ext::shared_ptr<SwapSpreadIndex> cmsspread1 =
232  ext::dynamic_pointer_cast<SwapSpreadIndex>(index1_);
233  ext::shared_ptr<SwapSpreadIndex> cmsspread2 =
234  ext::dynamic_pointer_cast<SwapSpreadIndex>(index2_);
235 
236  QL_REQUIRE(ibor1 != NULL || cms1 != NULL || cmsspread1 != NULL,
237  "index1 must be ibor or cms or cms spread");
238  QL_REQUIRE(ibor2 != NULL || cms2 != NULL || cmsspread2 != NULL,
239  "index2 must be ibor or cms");
240 
241  if (ibor1 != 0) {
242  IborLeg leg(schedule1_, ibor1);
243  leg = leg.withNotionals(nominal1_)
244  .withPaymentDayCounter(dayCount1_)
245  .withPaymentAdjustment(paymentConvention1_)
246  .withSpreads(spread1_)
247  .withGearings(gearing1_);
248  if (cappedRate1_[0] != Null<Real>())
249  leg = leg.withCaps(cappedRate1_);
250  if (flooredRate1_[0] != Null<Real>())
251  leg = leg.withFloors(flooredRate1_);
252  legs_[0] = leg;
253  }
254 
255  if (ibor2 != 0) {
256  IborLeg leg(schedule2_, ibor2);
257  leg = leg.withNotionals(nominal2_)
258  .withPaymentDayCounter(dayCount2_)
259  .withPaymentAdjustment(paymentConvention2_)
260  .withSpreads(spread2_)
261  .withGearings(gearing2_);
262  if (cappedRate2_[0] != Null<Real>())
263  leg = leg.withCaps(cappedRate2_);
264  if (flooredRate2_[0] != Null<Real>())
265  leg = leg.withFloors(flooredRate2_);
266  legs_[1] = leg;
267  }
268 
269  if (cms1 != 0) {
270  CmsLeg leg(schedule1_, cms1);
271  leg = leg.withNotionals(nominal1_)
272  .withPaymentDayCounter(dayCount1_)
273  .withPaymentAdjustment(paymentConvention1_)
274  .withSpreads(spread1_)
275  .withGearings(gearing1_);
276  if (cappedRate1_[0] != Null<Real>())
277  leg = leg.withCaps(cappedRate1_);
278  if (flooredRate1_[0] != Null<Real>())
279  leg = leg.withFloors(flooredRate1_);
280  legs_[0] = leg;
281  }
282 
283  if (cms2 != 0) {
284  CmsLeg leg(schedule2_, cms2);
285  leg = leg.withNotionals(nominal2_)
286  .withPaymentDayCounter(dayCount2_)
287  .withPaymentAdjustment(paymentConvention2_)
288  .withSpreads(spread2_)
289  .withGearings(gearing2_);
290  if (cappedRate2_[0] != Null<Real>())
291  leg = leg.withCaps(cappedRate2_);
292  if (flooredRate2_[0] != Null<Real>())
293  leg = leg.withFloors(flooredRate2_);
294  legs_[1] = leg;
295  }
296 
297  if (cmsspread1 != 0) {
298  CmsSpreadLeg leg(schedule1_, cmsspread1);
299  leg = leg.withNotionals(nominal1_)
300  .withPaymentDayCounter(dayCount1_)
301  .withPaymentAdjustment(paymentConvention1_)
302  .withSpreads(spread1_)
303  .withGearings(gearing1_);
304  if (cappedRate1_[0] != Null<Real>())
305  leg = leg.withCaps(cappedRate1_);
306  if (flooredRate1_[0] != Null<Real>())
307  leg = leg.withFloors(flooredRate1_);
308  legs_[0] = leg;
309  }
310 
311  if (cmsspread2 != 0) {
312  CmsSpreadLeg leg(schedule2_, cmsspread2);
313  leg = leg.withNotionals(nominal2_)
314  .withPaymentDayCounter(dayCount2_)
315  .withPaymentAdjustment(paymentConvention2_)
316  .withSpreads(spread2_)
317  .withGearings(gearing2_);
318  if (cappedRate2_[0] != Null<Real>())
319  leg = leg.withCaps(cappedRate2_);
320  if (flooredRate2_[0] != Null<Real>())
321  leg = leg.withFloors(flooredRate2_);
322  legs_[1] = leg;
323  }
324 
325  if (intermediateCapitalExchange_) {
326  for (Size i = 0; i < legs_[0].size() - 1; i++) {
327  Real cap = nominal1_[i] - nominal1_[i + 1];
328  if (!close(cap, 0.0)) {
329  std::vector<ext::shared_ptr<CashFlow> >::iterator it1 =
330  legs_[0].begin();
331  std::advance(it1, i + 1);
332  legs_[0].insert(
333  it1, ext::shared_ptr<CashFlow>(
334  new Redemption(cap, legs_[0][i]->date())));
335  std::vector<Real>::iterator it2 = nominal1_.begin();
336  std::advance(it2, i + 1);
337  nominal1_.insert(it2, nominal1_[i]);
338  i++;
339  }
340  }
341  for (Size i = 0; i < legs_[1].size() - 1; i++) {
342  Real cap = nominal2_[i] - nominal2_[i + 1];
343  if (!close(cap, 0.0)) {
344  std::vector<ext::shared_ptr<CashFlow> >::iterator it1 =
345  legs_[1].begin();
346  std::advance(it1, i + 1);
347  legs_[1].insert(
348  it1, ext::shared_ptr<CashFlow>(
349  new Redemption(cap, legs_[1][i]->date())));
350  std::vector<Real>::iterator it2 = nominal2_.begin();
351  std::advance(it2, i + 1);
352  nominal2_.insert(it2, nominal2_[i]);
353  i++;
354  }
355  }
356  }
357 
358  if (finalCapitalExchange_) {
359  legs_[0].push_back(ext::shared_ptr<CashFlow>(
360  new Redemption(nominal1_.back(), legs_[0].back()->date())));
361  nominal1_.push_back(nominal1_.back());
362  legs_[1].push_back(ext::shared_ptr<CashFlow>(
363  new Redemption(nominal2_.back(), legs_[1].back()->date())));
364  nominal2_.push_back(nominal2_.back());
365  }
366 
367  for (Leg::const_iterator i = legs_[0].begin(); i < legs_[0].end(); ++i)
368  registerWith(*i);
369 
370  for (Leg::const_iterator i = legs_[1].begin(); i < legs_[1].end(); ++i)
371  registerWith(*i);
372 
373  switch (type_) {
374  case VanillaSwap::Payer:
375  payer_[0] = -1.0;
376  payer_[1] = +1.0;
377  break;
378  case VanillaSwap::Receiver:
379  payer_[0] = +1.0;
380  payer_[1] = -1.0;
381  break;
382  default:
383  QL_FAIL("Unknown float float - swap type");
384  }
385  }

387  void FloatFloatSwap::setupArguments(PricingEngine::arguments *args) const {
388 
389  Swap::setupArguments(args);
390 
391  FloatFloatSwap::arguments *arguments =
392  dynamic_cast<FloatFloatSwap::arguments *>(args);
393 
394  if (arguments == 0)
395  return; // swap engine ... // QL_REQUIRE(arguments != 0, "argument type does not match");
396 
397  arguments->type = type_;
398  arguments->nominal1 = nominal1_;
399  arguments->nominal2 = nominal2_;
400  arguments->index1 = index1_;
401  arguments->index2 = index2_;
402 
403  const Leg &leg1Coupons = leg1();
404  const Leg &leg2Coupons = leg2();
405 
406  arguments->leg1ResetDates = arguments->leg1PayDates =
407  arguments->leg1FixingDates = std::vector<Date>(leg1Coupons.size());
408  arguments->leg2ResetDates = arguments->leg2PayDates =
409  arguments->leg2FixingDates = std::vector<Date>(leg2Coupons.size());
410 
411  arguments->leg1Spreads = arguments->leg1AccrualTimes =
412  arguments->leg1Gearings = std::vector<Real>(leg1Coupons.size());
413  arguments->leg2Spreads = arguments->leg2AccrualTimes =
414  arguments->leg2Gearings = std::vector<Real>(leg2Coupons.size());
415 
416  arguments->leg1Coupons =
417  std::vector<Real>(leg1Coupons.size(), Null<Real>());
418  arguments->leg2Coupons =
419  std::vector<Real>(leg2Coupons.size(), Null<Real>());
420 
421  arguments->leg1IsRedemptionFlow =
422  std::vector<bool>(leg1Coupons.size(), false);
423  arguments->leg2IsRedemptionFlow =
424  std::vector<bool>(leg2Coupons.size(), false);
425 
426  arguments->leg1CappedRates = arguments->leg1FlooredRates =
427  std::vector<Real>(leg1Coupons.size(), Null<Real>());
428  arguments->leg2CappedRates = arguments->leg2FlooredRates =
429  std::vector<Real>(leg2Coupons.size(), Null<Real>());
430 
431  for (Size i = 0; i < leg1Coupons.size(); ++i) {
432  ext::shared_ptr<FloatingRateCoupon> coupon =
433  ext::dynamic_pointer_cast<FloatingRateCoupon>(leg1Coupons[i]);
434  if (coupon != 0) {
435  arguments->leg1AccrualTimes[i] = coupon->accrualPeriod();
436  arguments->leg1PayDates[i] = coupon->date();
437  arguments->leg1ResetDates[i] = coupon->accrualStartDate();
438  arguments->leg1FixingDates[i] = coupon->fixingDate();
439  arguments->leg1Spreads[i] = coupon->spread();
440  arguments->leg1Gearings[i] = coupon->gearing();
441  try {
442  arguments->leg1Coupons[i] = coupon->amount();
443  }
444  catch (Error &) {
445  arguments->leg1Coupons[i] = Null<Real>();
446  }
447  ext::shared_ptr<CappedFlooredCoupon> cfcoupon =
448  ext::dynamic_pointer_cast<CappedFlooredCoupon>(
449  leg1Coupons[i]);
450  if (cfcoupon != 0) {
451  arguments->leg1CappedRates[i] = cfcoupon->cap();
452  arguments->leg1FlooredRates[i] = cfcoupon->floor();
453  }
454  } else {
455  ext::shared_ptr<CashFlow> cashflow =
456  ext::dynamic_pointer_cast<CashFlow>(leg1Coupons[i]);
457  std::vector<Date>::const_iterator j =
458  std::find(arguments->leg1PayDates.begin(),
459  arguments->leg1PayDates.end(), cashflow->date());
460  QL_REQUIRE(j != arguments->leg1PayDates.end(),
461  "nominal redemption on "
462  << cashflow->date()
463  << "has no corresponding coupon");
464  Size jIdx = j - arguments->leg1PayDates.begin();
465  arguments->leg1IsRedemptionFlow[i] = true;
466  arguments->leg1Coupons[i] = cashflow->amount();
467  arguments->leg1ResetDates[i] = arguments->leg1ResetDates[jIdx];
468  arguments->leg1FixingDates[i] =
469  arguments->leg1FixingDates[jIdx];
470  arguments->leg1AccrualTimes[i] = 0.0;
471  arguments->leg1Spreads[i] = 0.0;
472  arguments->leg1Gearings[i] = 1.0;
473  arguments->leg1PayDates[i] = cashflow->date();
474  }
475  }
476 
477  for (Size i = 0; i < leg2Coupons.size(); ++i) {
478  ext::shared_ptr<FloatingRateCoupon> coupon =
479  ext::dynamic_pointer_cast<FloatingRateCoupon>(leg2Coupons[i]);
480  if (coupon != 0) {
481  arguments->leg2AccrualTimes[i] = coupon->accrualPeriod();
482  arguments->leg2PayDates[i] = coupon->date();
483  arguments->leg2ResetDates[i] = coupon->accrualStartDate();
484  arguments->leg2FixingDates[i] = coupon->fixingDate();
485  arguments->leg2Spreads[i] = coupon->spread();
486  arguments->leg2Gearings[i] = coupon->gearing();
487  try {
488  arguments->leg2Coupons[i] = coupon->amount();
489  }
490  catch (Error &) {
491  arguments->leg2Coupons[i] = Null<Real>();
492  }
493  ext::shared_ptr<CappedFlooredCoupon> cfcoupon =
494  ext::dynamic_pointer_cast<CappedFlooredCoupon>(
495  leg2Coupons[i]);
496  if (cfcoupon != 0) {
497  arguments->leg2CappedRates[i] = cfcoupon->cap();
498  arguments->leg2FlooredRates[i] = cfcoupon->floor();
499  }
500  } else {
501  ext::shared_ptr<CashFlow> cashflow =
502  ext::dynamic_pointer_cast<CashFlow>(leg2Coupons[i]);
503  std::vector<Date>::const_iterator j =
504  std::find(arguments->leg2PayDates.begin(),
505  arguments->leg2PayDates.end(), cashflow->date());
506  QL_REQUIRE(j != arguments->leg2PayDates.end(),
507  "nominal redemption on "
508  << cashflow->date()
509  << "has no corresponding coupon");
510  Size jIdx = j - arguments->leg2PayDates.begin();
511  arguments->leg2IsRedemptionFlow[i] = true;
512  arguments->leg2Coupons[i] = cashflow->amount();
513  arguments->leg2ResetDates[i] = arguments->leg2ResetDates[jIdx];
514  arguments->leg2FixingDates[i] =
515  arguments->leg2FixingDates[jIdx];
516  arguments->leg2AccrualTimes[i] = 0.0;
