81  // very basic choice for correlation structure
82  class TwoParameterCorrelation : public CorrelationStructure {
83  protected:
84  ext::shared_ptr<Interpolation> rhoInf_;
85  ext::shared_ptr<Interpolation> beta_;
86 
87  public:
88  TwoParameterCorrelation(const ext::shared_ptr<Interpolation>& rhoInf,
89  const ext::shared_ptr<Interpolation>& beta)
90  : rhoInf_(rhoInf), beta_(beta) {}
91  Real operator()(const Time& start1, const Time& start2) const {
92  Real rhoInf = (*rhoInf_)(start1);
93  Real beta = (*beta_)(start1);
94  Real rho = rhoInf + (1.0 - rhoInf) * exp(-beta * fabs(start2 - start1));
95  return rho;
96  }
97  };
