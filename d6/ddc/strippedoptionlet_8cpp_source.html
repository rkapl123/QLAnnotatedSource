32 StrippedOptionlet::StrippedOptionlet(Natural settlementDays,
33 const Calendar& calendar,
34 BusinessDayConvention bdc,
35 ext::shared_ptr<IborIndex> iborIndex,
36 const std::vector<Date>& optionletDates,
37 const vector<Rate>& strikes,
38 vector<vector<Handle<Quote> > > v,
39 DayCounter dc,
40 VolatilityType type,
41 Real displacement)
42 : calendar_(calendar), settlementDays_(settlementDays), businessDayConvention_(bdc),
43 dc_(std::move(dc)), iborIndex_(std::move(iborIndex)), type_(type),
44 displacement_(displacement), nOptionletDates_(optionletDates.size()),
45 optionletDates_(optionletDates), optionletTimes_(nOptionletDates_),
46 optionletAtmRates_(nOptionletDates_), optionletStrikes_(nOptionletDates_, strikes),
47 nStrikes_(strikes.size()), optionletVolQuotes_(std::move(v)),
48 optionletVolatilities_(nOptionletDates_, vector<Volatility>(nStrikes_))
49
50 {
51 checkInputs();
52 registerWith(Settings::instance().evaluationDate());
53 registerWithMarketData();
54
55 Date refDate = calendar.advance(Settings::instance().evaluationDate(),
56 settlementDays, Days);
57
58 for (Size i=0; i<nOptionletDates_; ++i)
59 optionletTimes_[i] = dc_.yearFraction(refDate, optionletDates_[i]);
60 }

62 void StrippedOptionlet::checkInputs() const {
63
64 QL_REQUIRE(!optionletDates_.empty(), "empty optionlet tenor vector");
65 QL_REQUIRE(nOptionletDates_==optionletVolQuotes_.size(),
66 "mismatch between number of option tenors (" <<
67 nOptionletDates_ << ") and number of volatility rows (" <<
68 optionletVolQuotes_.size() << ")");
69 QL_REQUIRE(optionletDates_[0]>Settings::instance().evaluationDate(),
70 "first option date (" << optionletDates_[0] << ") is in the past");
71 for (Size i=1; i<nOptionletDates_; ++i)
72 QL_REQUIRE(optionletDates_[i]>optionletDates_[i-1],
73 "non increasing option dates: " << io::ordinal(i) <<
74 " is " << optionletDates_[i-1] << ", " <<
75 io::ordinal(i+1) << " is " << optionletDates_[i]);
76
77 QL_REQUIRE(nStrikes_==optionletVolQuotes_[0].size(),
78 "mismatch between strikes(" << optionletStrikes_[0].size() <<
79 ") and vol columns (" << optionletVolQuotes_[0].size() << ")");
80 for (Size j=1; j<nStrikes_; ++j)
81 QL_REQUIRE(optionletStrikes_[0][j-1]<optionletStrikes_[0][j],
82 "non increasing strikes: " << io::ordinal(j) <<
83 " is " << io::rate(optionletStrikes_[0][j-1]) << ", " <<
84 io::ordinal(j+1) << " is " << io::rate(optionletStrikes_[0][j]));
85 }

94 void StrippedOptionlet::performCalculations() const {
95 for (Size i=0; i<nOptionletDates_; ++i)
96 for (Size j=0; j<nStrikes_; ++j)
97 optionletVolatilities_[i][j] = optionletVolQuotes_[i][j]->value();
