71  Real SmileSection::optionPrice(Rate strike,
72  Option::Type type,
73  Real discount) const {
74  Real atm = atmLevel();
75  QL_REQUIRE(atm != Null<Real>(),
76  "smile section must provide atm level to compute option price");
77  // if lognormal or shifted lognormal,
78  // for strike at -shift, return option price even if outside
79  // minstrike, maxstrike interval
80  if (volatilityType() == ShiftedLognormal)
81  return blackFormula(type,strike,atm, std::fabs(strike+shift()) < QL_EPSILON ?
82  0.2 : sqrt(variance(strike)),discount,shift());
83  else
84  return bachelierBlackFormula(type,strike,atm,sqrt(variance(strike)),discount);
85  }

87  Real SmileSection::digitalOptionPrice(Rate strike,
88  Option::Type type,
89  Real discount,
90  Real gap) const {
91  Real m = volatilityType() == ShiftedLognormal ? -shift() : -QL_MAX_REAL;
92  Real kl = std::max(strike-gap/2.0,m);
93  Real kr = kl+gap;
94  return (type==Option::Call ? 1.0 : -1.0) *
95  (optionPrice(kl,type,discount)-optionPrice(kr,type,discount)) / gap;
96  }

98  Real SmileSection::density(Rate strike, Real discount, Real gap) const {
99  Real m = volatilityType() == ShiftedLognormal ? -shift() : -QL_MAX_REAL;
100  Real kl = std::max(strike-gap/2.0,m);
101  Real kr = kl+gap;
102  return (digitalOptionPrice(kl,Option::Call,discount,gap) -
103  digitalOptionPrice(kr,Option::Call,discount,gap)) / gap;
104  }

118  Real SmileSection::volatility(Rate strike, VolatilityType volatilityType,
119  Real shift) const {
120  if(volatilityType == volatilityType_ && close(shift,this->shift()))
121  return volatility(strike);
122  Real atm = atmLevel();
123  QL_REQUIRE(atm != Null<Real>(),
124  "smile section must provide atm level to compute converted volatilties");
125  Option::Type type = strike >= atm ? Option::Call : Option::Put;
126  Real premium = optionPrice(strike,type);
127  Real premiumAtm = optionPrice(atm,type);
128  if (volatilityType == ShiftedLognormal) {
129  try {
130  return blackFormulaImpliedStdDev(type, strike, atm, premium,
131  1.0, shift) /
132  std::sqrt(exerciseTime());
133  } catch(...) {
134  return blackFormulaImpliedStdDevChambers(
135  type, strike, atm, premium, premiumAtm, 1.0, shift) /
136  std::sqrt(exerciseTime());
137  }
138  } else {
139  return bachelierBlackFormulaImpliedVol(type, strike, atm,
140  exerciseTime(), premium);
141  }
142  }
