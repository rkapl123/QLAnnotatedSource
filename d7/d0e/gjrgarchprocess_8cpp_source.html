56  Disposable<Array> GJRGARCHProcess::drift(Time t, const Array& x) const {
57  Array tmp(2);
58  const Real N = CumulativeNormalDistribution()(lambda_);
59  const Real n = std::exp(-lambda_*lambda_/2.0)/std::sqrt(2*M_PI);
60  const Real q2 = 1.0 + lambda_*lambda_;
61  const Real q3 = lambda_*n + N + lambda_*lambda_*N;
62  const Real vol = (x[1] > 0.0) ? std::sqrt(x[1])
63  : (discretization_ == Reflection) ? - std::sqrt(-x[1])
64  : 0.0;
65 
66  tmp[0] = riskFreeRate_->forwardRate(t, t, Continuous)
67  - dividendYield_->forwardRate(t, t, Continuous)
68  - 0.5 * vol * vol;
69 
70  tmp[1] = daysPerYear_*daysPerYear_*omega_ + daysPerYear_*(beta_
71  + alpha_*q2 + gamma_*q3 - 1.0) *
72  ((discretization_==PartialTruncation) ? x[1] : vol*vol);
73  return tmp;
74  }

76  Disposable<Matrix> GJRGARCHProcess::diffusion(Time, const Array& x) const {
77  /* the correlation matrix is
78  | 1 rho |
79  | rho 1 |
80  whose square root (which is used here) is
81  | 1 0 |
82  | rho std::sqrt(1-rho^2) |
83  */
84  Matrix tmp(2,2);
85  const Real N = CumulativeNormalDistribution()(lambda_);
86  const Real n = std::exp(-lambda_*lambda_/2.0)/std::sqrt(2*M_PI);
87  const Real sigma2 = 2.0 + 4.0*lambda_*lambda_;
88  const Real q3 = lambda_*n + N + lambda_*lambda_*N;
89  const Real Eml_e4 = lambda_*lambda_*lambda_*n + 5.0*lambda_*n
90  + 3.0*N + lambda_*lambda_*lambda_*lambda_*N
91  + 6.0*lambda_*lambda_*N;
92  const Real sigma3 = Eml_e4 - q3*q3;
93  const Real sigma12 = -2.0*lambda_;
94  const Real sigma13 = -2.0*n - 2*lambda_*N;
95  const Real sigma23 = 2.0*N + sigma12*sigma13;
96  const Real vol = (x[1] > 0.0) ? std::sqrt(x[1])
97  : (discretization_ == Reflection) ? - std::sqrt(-x[1])
98  : 1e-8; // set vol to (almost) zero but still
99  // expose some correlation information
100  const Real rho1 = std::sqrt(daysPerYear_)*(alpha_*sigma12
101  + gamma_*sigma13) * vol * vol;
102  const Real rho2 = vol*vol*std::sqrt(daysPerYear_)
103  *std::sqrt(alpha_*alpha_*(sigma2 - sigma12*sigma12)
104  + gamma_*gamma_*(sigma3 - sigma13*sigma13)
105  + 2.0*alpha_*gamma_*(sigma23 - sigma12*sigma13));
106 
107  // tmp[0][0], tmp[0][1] are the coefficients of dW_1 and dW_2
108  // in asset return stochastic process
109  tmp[0][0] = vol; tmp[0][1] = 0.0;
110  tmp[1][0] = rho1; tmp[1][1] = rho2;
111  return tmp;
112  }

122  Disposable<Array> GJRGARCHProcess::evolve(Time t0, const Array& x0,
123  Time dt, const Array& dw) const {
124  Array retVal(2);
125  Real vol, mu, nu;
126 
127  const Real sdt = std::sqrt(dt);
128  const Real N = CumulativeNormalDistribution()(lambda_);
129  const Real n = std::exp(-lambda_*lambda_/2.0)/std::sqrt(2*M_PI);
130  const Real sigma2 = 2.0 + 4.0*lambda_*lambda_;
131  const Real q2 = 1.0 + lambda_*lambda_;
132  const Real q3 = lambda_*n + N + lambda_*lambda_*N;
133  const Real Eml_e4 = lambda_*lambda_*lambda_*n + 5.0*lambda_*n
134  + 3.0*N + lambda_*lambda_*lambda_*lambda_*N
135  + 6.0*lambda_*lambda_*N;
136  const Real sigma3 = Eml_e4 - q3*q3;
137  const Real sigma12 = -2.0*lambda_;
138  const Real sigma13 = -2.0*n - 2*lambda_*N;
139  const Real sigma23 = 2.0*N + sigma12*sigma13;
140  const Real rho1 = std::sqrt(daysPerYear_)*(alpha_*sigma12 + gamma_*sigma13);
141  const Real rho2 = std::sqrt(daysPerYear_)
142  *std::sqrt(alpha_*alpha_*(sigma2 - sigma12*sigma12)
143  + gamma_*gamma_*(sigma3 - sigma13*sigma13)
144  + 2.0*alpha_*gamma_*(sigma23 - sigma12*sigma13));
145 
146  switch (discretization_) {
147  // For the definition of PartialTruncation, FullTruncation
148  // and Reflection see Lord, R., R. Koekkoek and D. van Dijk (2006),
149  // "A Comparison of biased simulation schemes for
150  // stochastic volatility models",
151  // Working Paper, Tinbergen Institute
152  case PartialTruncation:
153  vol = (x0[1] > 0.0) ? std::sqrt(x0[1]) : 0.0;
154  mu = riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
155  - dividendYield_->forwardRate(t0, t0+dt, Continuous)
156  - 0.5 * vol * vol;
157  nu = daysPerYear_*daysPerYear_*omega_
158  + daysPerYear_*(beta_ + alpha_*q2 + gamma_*q3 - 1.0) * x0[1];
159 
160  retVal[0] = x0[0] * std::exp(mu*dt+vol*dw[0]*sdt);
161  retVal[1] = x0[1] + nu*dt + sdt*vol*vol*(rho1*dw[0] + rho2*dw[1]);
162  break;
163  case FullTruncation:
164  vol = (x0[1] > 0.0) ? std::sqrt(x0[1]) : 0.0;
165  mu = riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
166  - dividendYield_->forwardRate(t0, t0+dt, Continuous)
167  - 0.5 * vol * vol;
168  nu = daysPerYear_*daysPerYear_*omega_
169  + daysPerYear_*(beta_ + alpha_*q2 + gamma_*q3 - 1.0) * vol *vol;
170 
171  retVal[0] = x0[0] * std::exp(mu*dt+vol*dw[0]*sdt);
172  retVal[1] = x0[1] + nu*dt + sdt*vol*vol*(rho1*dw[0] + rho2*dw[1]);
173  break;
174  case Reflection:
175  vol = std::sqrt(std::fabs(x0[1]));
176  mu = riskFreeRate_->forwardRate(t0, t0+dt, Continuous)
177  - dividendYield_->forwardRate(t0, t0+dt, Continuous)
178  - 0.5 * vol*vol;
179  nu = daysPerYear_*daysPerYear_*omega_
180  + daysPerYear_*(beta_ + alpha_*q2 + gamma_*q3 - 1.0) * vol * vol;
181 
182  retVal[0] = x0[0]*std::exp(mu*dt+vol*dw[0]*sdt);
183  retVal[1] = vol*vol
184  +nu*dt + sdt*vol*vol*(rho1*dw[0] + rho2*dw[1]);
185  break;
186  default:
187  QL_FAIL("unknown discretization schema");
188  }
189 
190  return retVal;
191  }
