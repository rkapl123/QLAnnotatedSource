55  void FdmSimpleStorageCondition::applyTo(Array& a, Time t) const {
56  const std::vector<Time>::const_iterator iter
57  = std::find(exerciseTimes_.begin(), exerciseTimes_.end(), t);
58 
59  if (iter != exerciseTimes_.end()) {
60  Array retVal(a.size());
61 
62  Matrix m(y_.size(), x_.size());
63  std::copy(a.begin(), a.end(), m.begin());
64  BilinearInterpolation interpl(x_.begin(), x_.end(),
65  y_.begin(), y_.end(), m);
66 
67  const boost::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
68  const FdmLinearOpIterator endIter = layout->end();
69 
70  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
71  ++iter) {
72 
73  const std::vector<Size>& coor = iter.coordinates();
74  const Real x = x_[coor[0]];
75  const Real y = y_[coor[1]];
76 
77  const Real price = calculator_->innerValue(iter, t);
78 
79  const Real maxWithDraw = std::min(y-y_.front(), changeRate_);
80  const Real sellPrice = interpl(x, y-maxWithDraw);
81 
82  const Real maxInject = std::min(y_.back()-y, changeRate_);
83  const Real buyPrice = interpl(x, y+maxInject);
84 
85  // bang-bang-wait strategy
86  Real currentValue = std::max(a[iter.index()],
87  std::max(buyPrice - price*maxInject,
88  sellPrice + price*maxWithDraw));
89 
90  // check if intermediate grid points give a better value
91  std::vector<Real>::const_iterator yIter =
92  std::upper_bound(y_.begin(), y_.end(), y - maxWithDraw);
93 
94  while (yIter != y_.end() && *yIter < y + maxInject) {
95  if (*yIter != y) {
96  const Real change = *yIter - y;
97  const Real storagePrice(interpl(x, *yIter));
98 
99  currentValue = std::max(currentValue,
100  storagePrice - change*price);
101  }
102  ++yIter;
103  }
104 
105  retVal[iter.index()] = currentValue;
106  }
107  a = retVal;
108  }
109  }
