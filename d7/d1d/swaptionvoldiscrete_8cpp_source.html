120  void SwaptionVolatilityDiscrete::checkOptionDates(const Date& reference) const {
121  QL_REQUIRE(optionDates_[0] > reference,
122  "first option date (" << optionDates_[0] <<
123  ") must be greater than reference date (" << reference << ")");
124  for (Size i=1; i<nOptionTenors_; ++i) {
125  QL_REQUIRE(optionDates_[i]>optionDates_[i-1],
126  "non increasing option dates: " << io::ordinal(i) <<
127  " is " << optionDates_[i-1] << ", " << io::ordinal(i+1) <<
128  " is " << optionDates_[i]);
129  }
130  }

154  void SwaptionVolatilityDiscrete::initializeOptionDatesAndTimes() const {
