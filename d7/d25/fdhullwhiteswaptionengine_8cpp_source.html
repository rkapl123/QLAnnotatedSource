53  void FdHullWhiteSwaptionEngine::calculate() const {
54  // 1. Term structure
55  const Handle<YieldTermStructure> ts = model_->termStructure();
56 
57  // 2. Mesher
58  const DayCounter dc = ts->dayCounter();
59  const Date referenceDate = ts->referenceDate();
60  const Time maturity = dc.yearFraction(referenceDate,
61  arguments_.exercise->lastDate());
62 
63 
64  const boost::shared_ptr<OrnsteinUhlenbeckProcess> process(
65  new OrnsteinUhlenbeckProcess(model_->a(), model_->sigma()));
66 
67  const boost::shared_ptr<Fdm1dMesher> shortRateMesher(
68  new FdmSimpleProcess1dMesher(xGrid_, process, maturity,1,invEps_));
69 
70  const boost::shared_ptr<FdmMesher> mesher(
71  new FdmMesherComposite(shortRateMesher));
72 
73  // 3. Inner Value Calculator
74  const std::vector<Date>& exerciseDates = arguments_.exercise->dates();
75  std::map<Time, Date> t2d;
76 
77  for (Size i=0; i < exerciseDates.size(); ++i) {
78  const Time t = dc.yearFraction(referenceDate, exerciseDates[i]);
79  QL_REQUIRE(t >= 0, "exercise dates must not contain past date");
80 
81  t2d[t] = exerciseDates[i];
82  }
83 
84  const Handle<YieldTermStructure> disTs = model_->termStructure();
85  const Handle<YieldTermStructure> fwdTs
86  = arguments_.swap->iborIndex()->forwardingTermStructure();
87 
88  QL_REQUIRE(fwdTs->dayCounter() == disTs->dayCounter(),
89  "day counter of forward and discount curve must match");
90  QL_REQUIRE(fwdTs->referenceDate() == disTs->referenceDate(),
91  "reference date of forward and discount curve must match");
92 
93  const boost::shared_ptr<HullWhite> fwdModel(
94  new HullWhite(fwdTs, model_->a(), model_->sigma()));
95 
96  const boost::shared_ptr<FdmInnerValueCalculator> calculator(
97  new FdmAffineModelSwapInnerValue<HullWhite>(
98  model_.currentLink(), fwdModel,
99  arguments_.swap, t2d, mesher, 0));
100 
101  // 4. Step conditions
102  const boost::shared_ptr<FdmStepConditionComposite> conditions =
103  FdmStepConditionComposite::vanillaComposite(
104  DividendSchedule(), arguments_.exercise,
105  mesher, calculator, referenceDate, dc);
106 
107  // 5. Boundary conditions
108  const FdmBoundaryConditionSet boundaries;
109 
110  // 6. Solver
111  FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
112  calculator, maturity,
113  tGrid_, dampingSteps_ };
114 
115  const boost::scoped_ptr<FdmHullWhiteSolver> solver(
116  new FdmHullWhiteSolver(model_, solverDesc, schemeDesc_));
117 
118  results_.value = solver->valueAt(0.0);
119  }
120 }
