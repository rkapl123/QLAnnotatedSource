100  // With fixed reversion calibrate the volatilities one by one
101  // to the given helpers. It is assumed that that volatility step
102  // dates are suitable for this, i.e. they should be identical to
103  // the fixing dates of the helpers (except for the last one where
104  // we do not need a step). Also note that the endcritera reflect
105  // only the status of the last calibration when using this method.
106  void calibrateVolatilitiesIterative(
107  const std::vector<boost::shared_ptr<CalibrationHelper> > &helpers,
108  OptimizationMethod &method, const EndCriteria &endCriteria,
109  const Constraint &constraint = Constraint(),
110  const std::vector<Real> &weights = std::vector<Real>()) {
111 
112  for (Size i = 0; i < helpers.size(); i++) {
113  std::vector<boost::shared_ptr<CalibrationHelper> > h(1, helpers[i]);
114  calibrate(h, method, endCriteria, constraint, weights,
115  MoveVolatility(i));
116  }
117  }
