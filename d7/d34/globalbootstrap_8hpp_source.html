108 template <class Curve> void GlobalBootstrap<Curve>::initialize() const {
109 
110  // ensure helpers are sorted
111  std::sort(ts_->instruments_.begin(), ts_->instruments_.end(), detail::BootstrapHelperSorter());
112  std::sort(additionalHelpers_.begin(), additionalHelpers_.end(), detail::BootstrapHelperSorter());
113 
114  // skip expired helpers
115  Date firstDate = Traits::initialDate(ts_);
116 
117  firstHelper_ = 0;
118  if (!ts_->instruments_.empty()) {
119  while (firstHelper_ < ts_->instruments_.size() && ts_->instruments_[firstHelper_]->pillarDate() <= firstDate)
120  ++firstHelper_;
121  }
122  numberHelpers_ = ts_->instruments_.size() - firstHelper_;
123 
124  // skip expired additional helpers
125  firstAdditionalHelper_ = 0;
126  if (!additionalHelpers_.empty()) {
127  while (firstAdditionalHelper_ < additionalHelpers_.size() &&
128  additionalHelpers_[firstAdditionalHelper_]->pillarDate() <= firstDate)
129  ++firstAdditionalHelper_;
130  }
131  numberAdditionalHelpers_ = additionalHelpers_.size() - firstAdditionalHelper_;
132 
133  // skip expired additional dates
134  std::vector<Date> additionalDates;
135  if (!(additionalDates_ == QL_NULL_FUNCTION))
136  additionalDates = additionalDates_();
137  firstAdditionalDate_ = 0;
138  if (!additionalDates.empty()) {
139  while (firstAdditionalDate_ < additionalDates.size() && additionalDates[firstAdditionalDate_] <= firstDate)
140  ++firstAdditionalDate_;
141  }
142  numberAdditionalDates_ = additionalDates.size() - firstAdditionalDate_;
143 
144  QL_REQUIRE(numberHelpers_ + numberAdditionalDates_ >= Interpolator::requiredPoints - 1,
145  "not enough alive instruments (" << numberHelpers_ << ") + additional dates (" << numberAdditionalDates_
146  << ") = " << numberHelpers_ + numberAdditionalDates_ << " provided, "
147  << Interpolator::requiredPoints - 1 << " required");
148 
149  // calculate dates and times
150  std::vector<Date> &dates = ts_->dates_;
151  std::vector<Time> &times = ts_->times_;
152 
153  // first populate the dates vector and make sure they are sorted and there are no duplicates
154  dates.clear();
155  dates.push_back(firstDate);
156  for (Size j = 0; j < numberHelpers_; ++j)
157  dates.push_back(ts_->instruments_[firstHelper_ + j]->pillarDate());
158  for (Size j = firstAdditionalDate_; j < numberAdditionalDates_; ++j)
159  dates.push_back(additionalDates[firstAdditionalDate_ + j]);
160  std::sort(dates.begin(), dates.end());
161  auto it = std::unique(dates.begin(), dates.end());
162  QL_REQUIRE(it == dates.end(), "duplicate dates among alive instruments and additional dates");
163 
164  // build times vector
165  times.clear();
166  for (auto& date : dates)
167  times.push_back(ts_->timeFromReference(date));
168 
169  // determine maxDate
170  Date maxDate = firstDate;
171  for (Size j = 0; j < numberHelpers_; ++j) {
172  maxDate = std::max(ts_->instruments_[firstHelper_ + j]->latestRelevantDate(), maxDate);
173  }
174  for (Size j = 0; j < numberAdditionalDates_; ++j) {
175  maxDate = std::max(additionalDates[firstAdditionalDate_ + j], maxDate);
176  }
177  ts_->maxDate_ = maxDate;
178 
179  // set initial guess only if the current curve cannot be used as guess
180  if (!validCurve_ || ts_->data_.size() != dates.size()) {
181  // ts_->data_[0] is the only relevant item,
182  // but reasonable numbers might be needed for the whole data vector
183  // because, e.g., of interpolation's early checks
184  ts_->data_ = std::vector<Real>(dates.size(), Traits::initialValue(ts_));
185  }
186  initialized_ = true;
187 }

189 template <class Curve> void GlobalBootstrap<Curve>::calculate() const {
190 
191  // we might have to call initialize even if the curve is initialized
192  // and not moving, just because helpers might be date relative and change
193  // with evaluation date change.
194  // anyway it makes little sense to use date relative helpers with a
195  // non-moving curve if the evaluation date changes
196  if (!initialized_ || ts_->moving_)
197  initialize();
198 
199  // setup helpers
200  for (Size j = 0; j < numberHelpers_; ++j) {
201  const ext::shared_ptr<typename Traits::helper> &helper = ts_->instruments_[firstHelper_ + j];
202  // check for valid quote
203  QL_REQUIRE(helper->quote()->isValid(), io::ordinal(j + 1)
204  << " instrument (maturity: " << helper->maturityDate()
205  << ", pillar: " << helper->pillarDate() << ") has an invalid quote");
206  // don't try this at home!
207  // This call creates helpers, and removes "const".
208  // There is a significant interaction with observability.
209  helper->setTermStructure(const_cast<Curve *>(ts_));
210  }
211 
212  // setup additional helpers
213  for (Size j = 0; j < numberAdditionalHelpers_; ++j) {
214  const ext::shared_ptr<typename Traits::helper> &helper = additionalHelpers_[firstAdditionalHelper_ + j];
215  QL_REQUIRE(helper->quote()->isValid(), io::ordinal(j + 1)
216  << " additional instrument (maturity: " << helper->maturityDate()
217  << ") has an invalid quote");
218  helper->setTermStructure(const_cast<Curve *>(ts_));
219  }
220 
221  Real accuracy = accuracy_ != Null<Real>() ? accuracy_ : ts_->accuracy_;
222 
223  // setup optimizer and EndCriteria
224  Real optEps = accuracy;
225  LevenbergMarquardt optimizer(optEps, optEps, optEps); // FIXME hardcoded tolerances
226  EndCriteria ec(1000, 10, optEps, optEps, optEps); // FIXME hardcoded values here as well
227 
228  // setup interpolation
229  if (!validCurve_) {
230  ts_->interpolation_ =
231  ts_->interpolator_.interpolate(ts_->times_.begin(), ts_->times_.end(), ts_->data_.begin());
232  }
233 
234  // determine bounds, we use an unconstrained optimisation transforming the free variables to [lowerBound,upperBound]
235  std::vector<Real> lowerBounds(numberHelpers_ + numberAdditionalDates_),
236  upperBounds(numberHelpers_ + numberAdditionalDates_);
237  for (Size i = 0; i < numberHelpers_ + numberAdditionalDates_; ++i) {
238  // just pass zero as the first alive helper, it's not used in the standard QL traits anyway
239  lowerBounds[i] = Traits::minValueAfter(i + 1, ts_, validCurve_, 0);
240  upperBounds[i] = Traits::maxValueAfter(i + 1, ts_, validCurve_, 0);
241  }

243  // setup cost function
244  class TargetFunction : public CostFunction {
245  public:
246  TargetFunction(const Size firstHelper,
247  const Size numberHelpers,
248  ext::function<Array()> additionalErrors,
249  Curve* ts,
250  std::vector<Real> lowerBounds,
251  std::vector<Real> upperBounds)
252  : firstHelper_(firstHelper), numberHelpers_(numberHelpers),
253  additionalErrors_(std::move(additionalErrors)), ts_(ts),
254  lowerBounds_(std::move(lowerBounds)), upperBounds_(std::move(upperBounds)) {}
255 
256  Real transformDirect(const Real x, const Size i) const {
257  return (std::atan(x) + M_PI_2) / M_PI * (upperBounds_[i] - lowerBounds_[i]) + lowerBounds_[i];
258  }
259 
260  Real transformInverse(const Real y, const Size i) const {
261  return std::tan((y - lowerBounds_[i]) * M_PI / (upperBounds_[i] - lowerBounds_[i]) - M_PI_2);
262  }
263 
264  Real value(const Array& x) const override {
265  Array v = values(x);
266  std::transform(v.begin(), v.end(), v.begin(), square<Real>());
267  return std::sqrt(std::accumulate(v.begin(), v.end(), 0.0) / static_cast<Real>(v.size()));
268  }
269 
270  Disposable<Array> values(const Array& x) const override {
271  for (Size i = 0; i < x.size(); ++i) {
272  Traits::updateGuess(ts_->data_, transformDirect(x[i], i), i + 1);
273  }
274  ts_->interpolation_.update();
275  std::vector<Real> result(numberHelpers_);
276  for (Size i = 0; i < numberHelpers_; ++i) {
277  result[i] = ts_->instruments_[firstHelper_ + i]->quote()->value() -
278  ts_->instruments_[firstHelper_ + i]->impliedQuote();
279  }
280  if (!(additionalErrors_ == QL_NULL_FUNCTION)) {
281  Array tmp = additionalErrors_();
282  result.resize(numberHelpers_ + tmp.size());
283  for (Size i = 0; i < tmp.size(); ++i) {
284  result[numberHelpers_ + i] = tmp[i];
285  }
286  }
287  Array asArray(result.begin(), result.end());
288  return asArray;
289  }
290 
291  private:
292  Size firstHelper_, numberHelpers_;
293  ext::function<Array()> additionalErrors_;
294  Curve *ts_;
295  const std::vector<Real> lowerBounds_, upperBounds_;
296  };
297  TargetFunction cost(firstHelper_, numberHelpers_, additionalErrors_, ts_, lowerBounds, upperBounds);
298 
299  // setup guess
300  Array guess(numberHelpers_ + numberAdditionalDates_);
301  for (Size i = 0; i < guess.size(); ++i) {
302  // just pass zero as the first alive helper, it's not used in the standard QL traits anyway
303  guess[i] = cost.transformInverse(Traits::guess(i + 1, ts_, validCurve_, 0), i);
304  }
305 
306  // setup problem
307  NoConstraint noConstraint;
308  Problem problem(cost, noConstraint, guess);
309 
310  // run optimization
311  optimizer.minimize(problem, ec);
312 
313  // evaluate target function on best value found to ensure that data_ contains the optimal value
314  Real finalTargetError = cost.value(problem.currentValue());
315 
316  // check final error
317  QL_REQUIRE(finalTargetError <= accuracy,
318  "global bootstrap failed, error is " << finalTargetError << ", accuracy is " << accuracy);
319 
320  // set valid flag
321  validCurve_ = true;
322 }
