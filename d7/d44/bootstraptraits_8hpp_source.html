61  template <class C>
62  static Real guess(Size i,
63  const C* c,
64  bool validData,
65  Size) // firstAliveHelper
66  {
67  if (validData) // previous iteration value
68  return c->data()[i];
69 
70  if (i==1) // first pillar
71  return 1.0/(1.0+detail::avgRate*c->times()[1]);
72 
73  // flat rate extrapolation
74  Real r = -std::log(c->data()[i-1])/c->times()[i-1];
75  return std::exp(-r * c->times()[i]);
76  }

78  // possible constraints based on previous values
79  template <class C>
80  static Real minValueAfter(Size i,
81  const C* c,
82  bool validData,
83  Size) // firstAliveHelper
84  {
85  if (validData) {
86  #if defined(QL_NEGATIVE_RATES)
87  return *(std::min_element(c->data().begin(),
88  c->data().end()))/2.0;
89  #else
90  return c->data().back()/2.0;
91  #endif
92  }
93  Time dt = c->times()[i] - c->times()[i-1];
94  return c->data()[i-1] * std::exp(- detail::maxRate * dt);
95  }
