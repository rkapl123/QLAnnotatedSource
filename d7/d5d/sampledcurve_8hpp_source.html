107  template <class T>
108  void regrid(const Array &new_grid,
109  T func) {
110  Array transformed_grid(grid_.size());
111 
112  std::transform(grid_.begin(), grid_.end(),
113  transformed_grid.begin(), func);
114  CubicInterpolation priceSpline(transformed_grid.begin(),
115  transformed_grid.end(),
116  values_.begin(),
117  CubicInterpolation::Spline, false,
118  CubicInterpolation::SecondDerivative, 0.0,
119  CubicInterpolation::SecondDerivative, 0.0);
120  priceSpline.update();
121 
122  Array newValues = new_grid;
123  std::transform(newValues.begin(), newValues.end(),
124  newValues.begin(), func);
125  for (Array::iterator j = newValues.begin();
126  j != newValues.end(); ++j) {
127  *j = priceSpline(*j, true);
128  }
129  values_.swap(newValues);
130  grid_ = new_grid;
131  }
