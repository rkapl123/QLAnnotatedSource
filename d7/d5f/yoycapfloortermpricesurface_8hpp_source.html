330  template<class I2D, class I1D>
331  void InterpolatedYoYCapFloorTermPriceSurface<I2D,I1D>::
332  intersect() const {
333 
334 
335  // TODO: define the constants outside the code
336  const Real maxSearchRange = 0.0201;
337  const Real maxExtrapolationMaturity = 5.01;
338  const Real searchStep = 0.0050;
339  const Real intrinsicValueAddOn = 0.001;
340 
341  std::vector<bool> validMaturity(cfMaturities_.size(),false);
342 
343  cfMaturityTimes_.clear();
344  for (Size i=0; i<cfMaturities_.size();i++) {
345  cfMaturityTimes_.push_back(timeFromReference(
346  yoyOptionDateFromTenor(cfMaturities_[i])));
347  }
348 
349  capPrice_ = interpolator2d_.interpolate(
350  cfMaturityTimes_.begin(),cfMaturityTimes_.end(),
351  cStrikes_.begin(), cStrikes_.end(),
352  cPrice_
353  );
354  capPrice_.enableExtrapolation();
355 
356  floorPrice_ = interpolator2d_.interpolate(
357  cfMaturityTimes_.begin(),cfMaturityTimes_.end(),
358  fStrikes_.begin(), fStrikes_.end(),
359  fPrice_
360  );
361  floorPrice_.enableExtrapolation();
362 
363  atmYoYSwapDateRates_.first.clear();
364  atmYoYSwapDateRates_.second.clear();
365  atmYoYSwapTimeRates_.first.clear();
366  atmYoYSwapTimeRates_.second.clear();
367  Brent solver;
368  Real solverTolerance_ = 1e-7;
369  Real lo,hi,guess;
370  std::vector<Real> minSwapRateIntersection(cfMaturityTimes_.size());
371  std::vector<Real> maxSwapRateIntersection(cfMaturityTimes_.size());
372  std::vector<Time> tmpSwapMaturities;
373  std::vector<Rate> tmpSwapRates;
374  for (Size i = 0; i < cfMaturities_.size(); i++) {
375  Time t = cfMaturityTimes_[i];
376  // determine the sum of discount factors
377  Size numYears = (Size)(t + 0.5);
378  Real sumDiscount = 0.0;
379  for (Size j=0; j<numYears; ++j)
380  sumDiscount += nominalTermStructure()->discount(j + 1.0);
381  // determine the minimum value of the ATM swap point
382  Real tmpMinSwapRateIntersection = -1.e10;
383  Real tmpMaxSwapRateIntersection = 1.e10;
384  for (Size j=0; j<fStrikes_.size(); ++j) {
385  Real price = floorPrice_(t,fStrikes_[j]);
386  Real minSwapRate = fStrikes_[j] - price / (sumDiscount * 10000);
387  if (minSwapRate > tmpMinSwapRateIntersection)
388  tmpMinSwapRateIntersection = minSwapRate;
389  }
390  for (Size j=0; j<cStrikes_.size(); ++j) {
391  Real price = capPrice_(t,cStrikes_[j]);
392  Real maxSwapRate = cStrikes_[j] + price / (sumDiscount * 10000);
393  if (maxSwapRate < tmpMaxSwapRateIntersection)
394  tmpMaxSwapRateIntersection = maxSwapRate;
395  }
396  maxSwapRateIntersection[i] = tmpMaxSwapRateIntersection;
397  minSwapRateIntersection[i] = tmpMinSwapRateIntersection;
398 
399  // find the interval where the intersection lies
400  bool trialsExceeded = false;
401  int numTrials = (int)(maxSearchRange / searchStep);
402  if ( floorPrice_(t,fStrikes_.back()) > capPrice_(t,fStrikes_.back()) ) {
403  int counter = 1;
404  bool stop = false;
405  Real strike = 0.0;
406  while (stop == false) {
407  strike = fStrikes_.back() - counter * searchStep;
408  if (floorPrice_(t, strike) < capPrice_(t, strike))
409  stop = true;
410  counter++;
411  if (counter == numTrials + 1) {
412  if (stop == false) {
413  stop = true;
414  trialsExceeded = true;
415  }
416  }
417  }
418  lo = strike;
419  hi = strike + searchStep;
420  } else {
421  int counter = 1;
422  bool stop = false;
423  Real strike = 0.0;
424  while (stop == false) {
425  strike = fStrikes_.back() + counter * searchStep;
426  if (floorPrice_(t, strike) > capPrice_(t, strike))
427  stop = true;
428  counter++;
429  if (counter == numTrials + 1) {
430  if (stop == false) {
431  stop = true;
432  trialsExceeded = true;
433  }
434  }
435  }
436  lo = strike - searchStep;
437  hi = strike;
438  }
439 
440  guess = (hi+lo)/2.0;
441  Rate kI = -999.999;
442 
443  if (trialsExceeded == false) {
444  try{
445  kI = solver.solve( ObjectiveFunction(t, capPrice_, floorPrice_), solverTolerance_, guess, lo, hi );
446  } catch( std::exception &e) {
447  QL_FAIL("cap/floor intersection finding failed at t = " << t << ", error msg: "<< e.what());
448  }
449  // error message if kI is economically nonsensical (only if t is large)
450  if (kI <= minSwapRateIntersection[i]) {
451  if (t > maxExtrapolationMaturity)
452  QL_FAIL("cap/floor intersection finding failed at t = " << t <<
453  ", error msg: intersection value is below the arbitrage free lower bound "
454  << minSwapRateIntersection[i]);
455  }
456  else
457  {
458  tmpSwapMaturities.push_back(t);
459  tmpSwapRates.push_back(kI);
460  validMaturity[i] = true;
461  }
462  }
463  else
464  {
465  // error message if t is too large
466  if (t > maxExtrapolationMaturity)
467  QL_FAIL("cap/floor intersection finding failed at t = " << t <<
468  ", error msg: no interection found inside the admissible range");
469  }
470  }
471 
472  // extrapolation of swap rates if necessary
473  //Polynomial2D tmpInterpol;
474  //Interpolation interpol = tmpInterpol.interpolate(tmpSwapMaturities.begin(), tmpSwapMaturities.end(), tmpSwapRates.begin());
475  //interpol.enableExtrapolation();
476  int counter = 0;
477  for (Size i=0; i<cfMaturities_.size(); ++i) {
478  if ( !validMaturity[i] ) {
479  atmYoYSwapDateRates_.first.push_back(referenceDate()+cfMaturities_[i]);
480  atmYoYSwapTimeRates_.first.push_back(timeFromReference(referenceDate()+cfMaturities_[i]));
481  // atmYoYSwapRates_->second.push_back(interpol((*cfMaturities_)[i]));
482  // Heuristic: overwrite the the swap rate with a value that guarantees that the
483  // intrinsic value of all options is lower than the price
484  Real newSwapRate = minSwapRateIntersection[i] + intrinsicValueAddOn;
485  if (newSwapRate > maxSwapRateIntersection[i])
486  newSwapRate = 0.5 * (minSwapRateIntersection[i] + maxSwapRateIntersection[i]);
487  atmYoYSwapTimeRates_.second.push_back(newSwapRate);
488  atmYoYSwapDateRates_.second.push_back(newSwapRate);
489  } else {
490  atmYoYSwapTimeRates_.first.push_back(tmpSwapMaturities[counter]);
491  atmYoYSwapTimeRates_.second.push_back(tmpSwapRates[counter]);
492  atmYoYSwapDateRates_.first.push_back(
493  yoyOptionDateFromTenor(cfMaturities_.at(counter)));
494  atmYoYSwapDateRates_.second.push_back(tmpSwapRates[counter]);
495  counter++;
496  }
497  }
498 
499  // create the swap curve using the factory
500  atmYoYSwapRateCurve_ =
501  interpolator1d_.interpolate(atmYoYSwapTimeRates_.first.begin(),
502  atmYoYSwapTimeRates_.first.end(),
503  atmYoYSwapTimeRates_.second.begin());
504  }

507  template<class I2D, class I1D>
508  void InterpolatedYoYCapFloorTermPriceSurface<I2D,I1D>::
509  calculateYoYTermStructure() const {
510 
511  // which yoy-swap points to use in building the yoy-fwd curve?
512  // for now pick every year
513  Size nYears = (Size)(0.5+timeFromReference(referenceDate()+cfMaturities_.back()));
514 
515  std::vector<boost::shared_ptr<BootstrapHelper<YoYInflationTermStructure> > > YYhelpers;
516  for (Size i=1; i<=nYears; i++) {
517  Date maturity = nominalTermStructure()->referenceDate() + Period(i,Years);
518  Handle<Quote> quote(boost::shared_ptr<Quote>(
519  new SimpleQuote( atmYoYSwapRate( maturity ) )));
520  boost::shared_ptr<BootstrapHelper<YoYInflationTermStructure> >
521  anInstrument(
522  new YearOnYearInflationSwapHelper(
523  quote, observationLag(), maturity,
524  calendar(), bdc_, dayCounter(),
525  yoyIndex()));
526  YYhelpers.push_back (anInstrument);
527  }
528 
529  // usually this base rate is known
530  // however for the data to be self-consistent
531  // we pick this as the end of the curve
532  Rate baseYoYRate = atmYoYSwapRate( referenceDate() );
533 
534  Handle<YieldTermStructure> nominalH( nominalTermStructure() );
535  // Linear is OK because we have every year
536  boost::shared_ptr<PiecewiseYoYInflationCurve<Linear> > pYITS(
537  new PiecewiseYoYInflationCurve<Linear>(
538  nominalTermStructure()->referenceDate(),
539  calendar(), dayCounter(), observationLag(), yoyIndex()->frequency(),
540  yoyIndex()->interpolated(), baseYoYRate,
541  nominalH, YYhelpers));
542  pYITS->recalculate();
543  yoy_ = pYITS; // store
544 
545  // check that helpers are repriced
546  const Real eps = 1e-5;
547  for (Size i=0; i<YYhelpers.size(); i++) {
548  Rate original = atmYoYSwapRate( yoyOptionDateFromTenor(Period(i+1,Years)) );
549  QL_REQUIRE(fabs(YYhelpers[i]->impliedQuote() - original) <eps,
550  "could not reprice helper "<< i
551  << ", data " << original
552  << ", implied quote " << YYhelpers[i]->impliedQuote()
553  );
554  }
555  }
556 
557 }
