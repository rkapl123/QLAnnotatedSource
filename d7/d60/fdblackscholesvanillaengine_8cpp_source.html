70  void FdBlackScholesVanillaEngine::calculate() const {
71 
72  // 0. Cash dividend model
73  const Date exerciseDate = arguments_.exercise->lastDate();
74  const Time maturity = process_->time(exerciseDate);
75  const Date settlementDate = process_->riskFreeRate()->referenceDate();
76 
77  Real spotAdjustment = 0.0;
78  DividendSchedule dividendSchedule = DividendSchedule();
79 
80  switch (cashDividendModel_) {
81  case Spot:
82  dividendSchedule = arguments_.cashFlow;
83  break;
84  case Escrowed:
85  for (DividendSchedule::const_iterator
86  divIter = arguments_.cashFlow.begin();
87  divIter != arguments_.cashFlow.end(); ++divIter) {
88 
89  const Date divDate = (*divIter)->date();
90 
91  if (divDate <= exerciseDate && divDate >= settlementDate) {
92  const Real divAmount = (*divIter)->amount();
93 
94  if (divAmount != 0.0) {
95  QL_REQUIRE(arguments_.exercise->type()
96  == Exercise::European,
97  "Escrowed dividend model expects an European option");
98 
99  const DiscountFactor discount =
100  process_->riskFreeRate()->discount(divDate) /
101  process_->dividendYield()->discount(divDate);
102 
103  spotAdjustment -= divAmount * discount;
104  }
105  }
106  }
107 
108  QL_REQUIRE(process_->x0() + spotAdjustment > 0.0,
109  "spot minus dividends becomes negative");
110  break;
111  default:
112  QL_FAIL("unknwon cash dividend model");
113  }
114 
115  // 1. Mesher
116  const ext::shared_ptr<StrikedTypePayoff> payoff =
117  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
118 
119  const ext::shared_ptr<Fdm1dMesher> equityMesher(
120  new FdmBlackScholesMesher(
121  xGrid_, process_, maturity, payoff->strike(),
122  Null<Real>(), Null<Real>(), 0.0001, 1.5,
123  std::pair<Real, Real>(payoff->strike(), 0.1),
124  dividendSchedule, quantoHelper_,
125  spotAdjustment));
126 
127  const ext::shared_ptr<FdmMesher> mesher (
128  new FdmMesherComposite(equityMesher));
129 
130  // 2. Calculator
131  const ext::shared_ptr<FdmInnerValueCalculator> calculator(
132  new FdmLogInnerValue(payoff, mesher, 0));
133 
134  // 3. Step conditions
135  const ext::shared_ptr<FdmStepConditionComposite> conditions =
136  FdmStepConditionComposite::vanillaComposite(
137  dividendSchedule, arguments_.exercise, mesher, calculator,
138  process_->riskFreeRate()->referenceDate(),
139  process_->riskFreeRate()->dayCounter());
140 
141  // 4. Boundary conditions
142  const FdmBoundaryConditionSet boundaries;
143 
144  // 5. Solver
145  FdmSolverDesc solverDesc = { mesher, boundaries, conditions, calculator,
146  maturity, tGrid_, dampingSteps_ };
147 
148  const ext::shared_ptr<FdmBlackScholesSolver> solver(
149  ext::make_shared<FdmBlackScholesSolver>(
150  Handle<GeneralizedBlackScholesProcess>(process_),
151  payoff->strike(), solverDesc, schemeDesc_,
152  localVol_, illegalLocalVolOverwrite_,
153  Handle<FdmQuantoHelper>(quantoHelper_)));
154 
155  const Real spot = process_->x0() + spotAdjustment;
156 
157  results_.value = solver->valueAt(spot);
158  results_.delta = solver->deltaAt(spot);
159  results_.gamma = solver->gammaAt(spot);
160  results_.theta = solver->thetaAt(spot);
161  }
