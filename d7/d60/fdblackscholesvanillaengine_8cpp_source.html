72 void FdBlackScholesVanillaEngine::calculate() const {
73 // 0. Cash dividend model
74 const Date exerciseDate = arguments_.exercise->lastDate();
75 const Time maturity = process_->time(exerciseDate);
76 const Date settlementDate = process_->riskFreeRate()->referenceDate();
77
78 Real spotAdjustment = 0.0;
79 DividendSchedule dividendSchedule = DividendSchedule();
80
81 ext::shared_ptr<EscrowedDividendAdjustment> escrowedDivAdj;
82
83 switch (cashDividendModel_) {
84 case Spot:
85 dividendSchedule = arguments_.cashFlow;
86 break;
87 case Escrowed:
88 if (arguments_.exercise->type() != Exercise::European)
89 // add dividend dates as stopping times
90 for (const auto& cf: arguments_.cashFlow)
91 dividendSchedule.push_back(
92 ext::make_shared<FixedDividend>(0.0, cf->date()));
93
94 QL_REQUIRE(quantoHelper_ == nullptr,
95 "Escrowed dividend model is not supported for Quanto-Options");
96
97 escrowedDivAdj = ext::make_shared<EscrowedDividendAdjustment>(
98 arguments_.cashFlow,
99 process_->riskFreeRate(),
100 process_->dividendYield(),
101 [&](Date d){ return process_->time(d); },
102 maturity
103 );
104
105 spotAdjustment =
106 escrowedDivAdj->dividendAdjustment(process_->time(settlementDate));
107
108 QL_REQUIRE(process_->x0() + spotAdjustment > 0.0,
109 "spot minus dividends becomes negative");
110
111 break;
112 default:
113 QL_FAIL("unknwon cash dividend model");
114 }
115
116 // 1. Mesher
117 const ext::shared_ptr<StrikedTypePayoff> payoff =
118 ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
119
120 const ext::shared_ptr<Fdm1dMesher> equityMesher =
121 ext::make_shared<FdmBlackScholesMesher>(
122 xGrid_, process_, maturity, payoff->strike(),
123 Null<Real>(), Null<Real>(), 0.0001, 1.5,
124 std::pair<Real, Real>(payoff->strike(), 0.1),
125 dividendSchedule, quantoHelper_,
126 spotAdjustment);
127
128 const ext::shared_ptr<FdmMesher> mesher =
129 ext::make_shared<FdmMesherComposite>(equityMesher);
130
131 // 2. Calculator
132 ext::shared_ptr<FdmInnerValueCalculator> calculator;
133 switch (cashDividendModel_) {
134 case Spot:
135 calculator = ext::make_shared<FdmLogInnerValue>(
136 payoff, mesher, 0);
137 break;
138 case Escrowed:
139 calculator = ext::make_shared<FdmEscrowedLogInnerValueCalculator>(
140 escrowedDivAdj, payoff, mesher, 0);
141 break;
142 default:
143 QL_FAIL("unknwon cash dividend model");
144 }
145
146 // 3. Step conditions
147 const ext::shared_ptr<FdmStepConditionComposite> conditions =
148 FdmStepConditionComposite::vanillaComposite(
149 dividendSchedule, arguments_.exercise, mesher, calculator,
150 process_->riskFreeRate()->referenceDate(),
151 process_->riskFreeRate()->dayCounter());
152
153 // 4. Boundary conditions
154 const FdmBoundaryConditionSet boundaries;
155
156 // 5. Solver
157 FdmSolverDesc solverDesc = { mesher, boundaries, conditions, calculator,
158 maturity, tGrid_, dampingSteps_ };
159
160 const ext::shared_ptr<FdmBlackScholesSolver> solver(
161 ext::make_shared<FdmBlackScholesSolver>(
162 Handle<GeneralizedBlackScholesProcess>(process_),
163 payoff->strike(), solverDesc, schemeDesc_,
164 localVol_, illegalLocalVolOverwrite_,
165 Handle<FdmQuantoHelper>(quantoHelper_)));
166
167 const Real spot = process_->x0() + spotAdjustment;
168
169 results_.value = solver->valueAt(spot);
170 results_.delta = solver->deltaAt(spot);
171 results_.gamma = solver->gammaAt(spot);
172 results_.theta = solver->thetaAt(spot);
173 }
