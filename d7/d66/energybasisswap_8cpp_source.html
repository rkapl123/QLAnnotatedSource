53  void EnergyBasisSwap::performCalculations() const {
54 
55  try {
56 
57  if (payIndex_->empty()) {
58  if (payIndex_->forwardCurveEmpty()) {
59  QL_FAIL("index [" + payIndex_->name() +
60  "] does not have any quotes or forward prices");
61  } else {
62  addPricingError(PricingError::Warning,
63  "index [" + payIndex_->name() +
64  "] does not have any quotes; "
65  "using forward prices from [" +
66  payIndex_->forwardCurve()->name() + "]");
67  }
68  }
69  if (receiveIndex_->empty()) {
70  if (receiveIndex_->forwardCurveEmpty()) {
71  QL_FAIL("index [" + receiveIndex_->name() +
72  "] does not have any quotes or forward prices");
73  } else {
74  addPricingError(PricingError::Warning,
75  "index [" + receiveIndex_->name() +
76  "] does not have any quotes; "
77  "using forward prices from [" +
78  receiveIndex_->forwardCurve()->name() +
79  "]");
80  }
81  }
82 
83  NPV_ = 0.0;
84  additionalResults_.clear();
85  dailyPositions_.clear();
86  paymentCashFlows_.clear();
87 
88  Date evaluationDate = Settings::instance().evaluationDate();
89 
90  const Currency& baseCurrency =
91  CommoditySettings::instance().currency();
92  const UnitOfMeasure baseUnitOfMeasure =
93  CommoditySettings::instance().unitOfMeasure();
94 
95  Real quantityUomConversionFactor =
96  calculateUomConversionFactor(
97  pricingPeriods_[0]->quantity().commodityType(),
98  baseUnitOfMeasure,
99  pricingPeriods_[0]->quantity().unitOfMeasure());
100  Real payIndexUomConversionFactor =
101  calculateUomConversionFactor(payIndex_->commodityType(),
102  payIndex_->unitOfMeasure(),
103  baseUnitOfMeasure);
104  Real receiveIndexUomConversionFactor =
105  calculateUomConversionFactor(receiveIndex_->commodityType(),
106  receiveIndex_->unitOfMeasure(),
107  baseUnitOfMeasure);
108 
109  Real payIndexFxConversionFactor =
110  calculateFxConversionFactor(payIndex_->currency(),
111  baseCurrency, evaluationDate);
112  Real receiveIndexFxConversionFactor =
113  calculateFxConversionFactor(receiveIndex_->currency(),
114  baseCurrency, evaluationDate);
115  Real payLegFxConversionFactor =
116  calculateFxConversionFactor(baseCurrency, payCurrency_,
117  evaluationDate);
118  Real receiveLegFxConversionFactor =
119  calculateFxConversionFactor(baseCurrency, receiveCurrency_,
120  evaluationDate);
121 
122  Real basisUomConversionFactor =
123  calculateUomConversionFactor(
124  pricingPeriods_[0]->quantity().commodityType(),
125  basis_.unitOfMeasure(), baseUnitOfMeasure);
126  Real basisFxConversionFactor =
127  calculateFxConversionFactor(baseCurrency,
128  basis_.amount().currency(),
129  evaluationDate);
130 
131  Real basisValue = basis_.amount().value() *
132  basisUomConversionFactor * basisFxConversionFactor;
133 
134  Date lastPayIndexQuoteDate = payIndex_->lastQuoteDate();
135  Date lastReceiveIndexQuoteDate = receiveIndex_->lastQuoteDate();
136 
137  if (lastPayIndexQuoteDate < evaluationDate - 1) {
138  std::ostringstream message;
139  message << "index [" << payIndex_->name()
140  << "] has last quote date of "
141  << io::iso_date(lastPayIndexQuoteDate);
142  addPricingError(PricingError::Warning, message.str());
143  }
144  if (lastReceiveIndexQuoteDate < evaluationDate - 1) {
145  std::ostringstream message;
146  message << "index [" << receiveIndex_->name()
147  << "] has last quote date of "
148  << io::iso_date(lastReceiveIndexQuoteDate);
149  addPricingError(PricingError::Warning, message.str());
150  }
151 
152  Date lastQuoteDate = std::min(lastPayIndexQuoteDate,
153  lastReceiveIndexQuoteDate);
154 
155  Real totalQuantityAmount = 0;
156 
157  // price each period
158  for (PricingPeriods::const_iterator pi = pricingPeriods_.begin();
159  pi != pricingPeriods_.end(); ++pi) {
160  const ext::shared_ptr<PricingPeriod>& pricingPeriod = *pi;
161 
162  Integer periodDayCount = 0;
163 
164  // get the index quotes
165  Date periodStartDate =
166  calendar_.adjust(pricingPeriod->startDate());
167  for (Date stepDate = periodStartDate;
168  stepDate <= pricingPeriod->endDate();
169  stepDate = calendar_.advance(stepDate, 1*Days)) {
170 
171  bool unrealized = stepDate > evaluationDate;
172  Real payQuoteValue = 0;
173  Real receiveQuoteValue = 0;
174 
175  if (stepDate <= lastQuoteDate) {
176  payQuoteValue = payIndex_->price(stepDate);
177  receiveQuoteValue = receiveIndex_->price(stepDate);
178  } else {
179  payQuoteValue = payIndex_->forwardPrice(stepDate);
180  receiveQuoteValue =
181  receiveIndex_->forwardPrice(stepDate);
182  }
183 
184  if (payQuoteValue == 0) {
185  std::ostringstream message;
186  message << "pay quote value for curve ["
187  << payIndex_->name() << "] is 0 for date "
188  << io::iso_date(stepDate);
189  addPricingError(PricingError::Warning, message.str());
190  }
191  if (receiveQuoteValue == 0) {
192  std::ostringstream message;
193  message << "receive quote value for curve ["
194  << receiveIndex_->name() << "] is 0 for date "
195  << io::iso_date(stepDate);
196  addPricingError(PricingError::Warning, message.str());
197  }
198 
199  QL_REQUIRE(payQuoteValue != Null<Real>(),
200  "curve [" << payIndex_->name() <<
201  "] missing value for pricing date: "
202  << stepDate);
203  QL_REQUIRE(receiveQuoteValue != Null<Real>(),
204  "curve [" << receiveIndex_->name() <<
205  "] missing value for pricing date: "
206  << stepDate);
207 
208  Real payLegPriceValue =
209  payQuoteValue * payIndexUomConversionFactor *
210  payIndexFxConversionFactor;
211  Real receiveLegPriceValue =
212  receiveQuoteValue * receiveIndexUomConversionFactor *
213  receiveIndexFxConversionFactor;
214 
215  if (spreadToPayLeg_)
216  payLegPriceValue += basisValue;
217  else
218  receiveLegPriceValue += basisValue;
219 
220  dailyPositions_[stepDate] =
221  EnergyDailyPosition(stepDate, payLegPriceValue,
222  receiveLegPriceValue, unrealized);
223  periodDayCount++;
224  }
225 
226  Real periodQuantityAmount =
227  pricingPeriod->quantity().amount() *
228  quantityUomConversionFactor;
229  totalQuantityAmount += periodQuantityAmount;
230 
231  Real avgDailyQuantityAmount =
232  periodDayCount == 0 ? 0 :
233  periodQuantityAmount / periodDayCount;
234 
235  Real payLegValue = 0;
236  Real receiveLegValue = 0;
237  for (std::map<Date, EnergyDailyPosition>::iterator dpi =
238  dailyPositions_.find(periodStartDate);
239  dpi != dailyPositions_.end() &&
240  dpi->first <= pricingPeriod->endDate(); ++dpi) {
241  EnergyDailyPosition& dailyPosition = dpi->second;
242  dailyPosition.quantityAmount = avgDailyQuantityAmount;
243  dailyPosition.riskDelta =
244  (-dailyPosition.payLegPrice + dailyPosition.receiveLegPrice) * avgDailyQuantityAmount;
245  payLegValue += -dailyPosition.payLegPrice * avgDailyQuantityAmount;
246  receiveLegValue += dailyPosition.receiveLegPrice * avgDailyQuantityAmount;
247  }
248 
249  Real discountFactor = 1;
250  Real payLegDiscountFactor = 1;
251  Real receiveLegDiscountFactor = 1;
252  if (pricingPeriod->paymentDate() >= evaluationDate + 2 /* settlement days*/) {
253  discountFactor =
254  discountTermStructure_->discount(
255  pricingPeriod->paymentDate());
256  payLegDiscountFactor =
257  payLegTermStructure_->discount(
258  pricingPeriod->paymentDate());
259  receiveLegDiscountFactor =
260  receiveLegTermStructure_->discount(
261  pricingPeriod->paymentDate());
262  }
263 
264  Real uDelta = receiveLegValue + payLegValue;
265  Real dDelta = (receiveLegValue * receiveLegDiscountFactor) +
266  (payLegValue * payLegDiscountFactor);
267  Real pmtFxConversionFactor =
268  (dDelta > 0) ? payLegFxConversionFactor : receiveLegFxConversionFactor;
269  Currency pmtCurrency =
270  (dDelta > 0) ? receiveCurrency_ : payCurrency_;
271  Real pmtDiscountFactor =
272  (dDelta > 0) ? receiveLegDiscountFactor : payLegDiscountFactor;
273 
274  paymentCashFlows_[pricingPeriod->paymentDate()] =
275  ext::make_shared<CommodityCashFlow> (
276  pricingPeriod->paymentDate(),
277  Money(baseCurrency,
278  uDelta * discountFactor),
279  Money(baseCurrency, uDelta),
280  Money(pmtCurrency,
281  dDelta * pmtFxConversionFactor),
282  Money(pmtCurrency,
283  uDelta * pmtFxConversionFactor),
284  discountFactor,
285  pmtDiscountFactor,
286  pricingPeriod->paymentDate() <= evaluationDate);
287 
288  calculateSecondaryCostAmounts(
289  pricingPeriods_[0]->quantity().commodityType(),
290  totalQuantityAmount, evaluationDate);
291 
292  NPV_ += dDelta;
293  }
294 
295  QL_REQUIRE(!paymentCashFlows_.empty(), "no cashflows");
296 
297  for (SecondaryCostAmounts::const_iterator i =
298  secondaryCostAmounts_.begin();
299  i != secondaryCostAmounts_.end(); ++i) {
300  Real amount = i->second.value();
301  NPV_ -= amount;
302  }
303 
304  additionalResults_["dailyPositions"] = dailyPositions_;
305 
306  } catch (const QuantLib::Error& e) {
307  addPricingError(PricingError::Error, e.what());
308  throw;
309  } catch (const std::exception& e) {
310  addPricingError(PricingError::Error, e.what());
311  throw;
312  }
313  }
314 
315 }
