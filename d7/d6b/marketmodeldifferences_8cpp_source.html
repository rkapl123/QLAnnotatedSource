28  std::vector<Volatility> rateVolDifferences(
29  const MarketModel& marketModel1,
30  const MarketModel& marketModel2) {
31  QL_ENSURE(marketModel1.initialRates() == marketModel2.initialRates(),
32  "initialRates do not match");
33  const EvolutionDescription& evolutionDescription1
34  = marketModel1.evolution();
35  const EvolutionDescription& evolutionDescription2
36  = marketModel2.evolution();
37  QL_ENSURE(evolutionDescription1.evolutionTimes()
38  == evolutionDescription2.evolutionTimes(),
39  "Evolution times do not match");
40 
41  const Matrix& totalCovariance1
42  = marketModel1.totalCovariance(marketModel1.numberOfSteps()-1);
43  const Matrix& totalCovariance2
44  = marketModel2.totalCovariance(marketModel2.numberOfSteps()-1);
45  const std::vector<Time>& maturities =
46  evolutionDescription1.evolutionTimes();
47 
48  std::vector<Volatility> result(totalCovariance1.columns());
49  for (Size i=0; i<totalCovariance1.columns(); ++i) {
50  Real diff = totalCovariance1[i][i]-totalCovariance2[i][i];
51  result[i] = std::sqrt(diff/maturities[i]);
52  }
53  return result;
54  }

56  std::vector<Spread> rateInstVolDifferences(
57  const MarketModel& marketModel1,
58  const MarketModel& marketModel2,
59  Size index) {
60  QL_ENSURE(marketModel1.initialRates() == marketModel2.initialRates(),
61  "initialRates do not match");
62  const EvolutionDescription& evolutionDescription1
63  = marketModel1.evolution();
64  const EvolutionDescription& evolutionDescription2
65  = marketModel2.evolution();
66  QL_ENSURE(evolutionDescription1.evolutionTimes()
67  == evolutionDescription2.evolutionTimes(),
68  "Evolution times do not match");
69  QL_ENSURE(index<evolutionDescription1.numberOfSteps(),
70  "the index given is greater than the number of steps");
71 
72  const std::vector<Time>& evolutionTimes
73  = evolutionDescription1.evolutionTimes();
74  std::vector<Spread> result(evolutionTimes.size());
75 
76  Time previousEvolutionTime = 0;
77  for (Size i=0; i<evolutionTimes.size(); ++i) {
78  Time currentEvolutionTime = evolutionTimes[i];
79  Time dt = currentEvolutionTime - previousEvolutionTime;
80  const Matrix& covariance1 = marketModel1.covariance(i);
81  const Matrix& covariance2 = marketModel2.covariance(i);
82  Real diff = covariance1[index][index] - covariance2[index][index];
83  result[i] = std::sqrt(diff/dt);
84  previousEvolutionTime = currentEvolutionTime;
85  }
86  return result;
87  }

89  std::vector<Matrix> coterminalSwapPseudoRoots(
90  const PiecewiseConstantCorrelation& piecewiseConstantCorrelation,
91  const std::vector<ext::shared_ptr<PiecewiseConstantVariance> >&
92  piecewiseConstantVariances) {
93  QL_ENSURE(piecewiseConstantCorrelation.times()
94  == piecewiseConstantVariances.front()->rateTimes(),
95  "correlations and volatilities intertave");
96  std::vector<Matrix> peudoRoots;
97  const std::vector<Time>& rateTimes
98  = piecewiseConstantVariances.front()->rateTimes();
99  for (Size i=1; i<rateTimes.size(); ++i) {
100  Time sqrtTau = std::sqrt(rateTimes[i]-rateTimes[i-1]);
101  const Matrix& correlations
102  = piecewiseConstantCorrelation.correlation(i);
103  Matrix pseudoRoot(correlations.rows(), correlations.rows());
104  for (Size j=0; j<correlations.rows(); ++j) {
105  Real volatility
106  = piecewiseConstantVariances[j]->volatility(i)*sqrtTau;
107  std::transform(correlations.row_begin(j),
108  correlations.row_end(j),
109  pseudoRoot.row_begin(j),
110  multiply_by<Real>(volatility));
111  }
112  peudoRoots.push_back(pseudoRoot);
113  }
114  return peudoRoots;
115  }
