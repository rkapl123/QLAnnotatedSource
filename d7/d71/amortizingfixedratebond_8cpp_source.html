61  std::pair<Integer,Integer> daysMinMax(const Period& p) {
62  switch (p.units()) {
63  case Days:
64  return std::make_pair(p.length(), p.length());
65  case Weeks:
66  return std::make_pair(7*p.length(), 7*p.length());
67  case Months:
68  return std::make_pair(28*p.length(), 31*p.length());
69  case Years:
70  return std::make_pair(365*p.length(), 366*p.length());
71  default:
72  QL_FAIL("unknown time unit (" << Integer(p.units()) << ")");
73  }
74  }

76  bool isSubPeriod(const Period& subPeriod,
77  const Period& superPeriod,
78  Integer& numSubPeriods) {
79 
80  std::pair<Integer, Integer> superDays(daysMinMax(superPeriod));
81  std::pair<Integer, Integer> subDays(daysMinMax(subPeriod));
82 
83  //obtain the approximate time ratio
84  Real minPeriodRatio =
85  ((Real)superDays.first)/((Real)subDays.second);
86  Real maxPeriodRatio =
87  ((Real)superDays.second)/((Real)subDays.first);
88  auto lowRatio = static_cast<Integer>(std::floor(minPeriodRatio));
89  auto highRatio = static_cast<Integer>(std::ceil(maxPeriodRatio));
90 
91  try {
92  for(Integer i=lowRatio; i <= highRatio; ++i) {
93  Period testPeriod = subPeriod * i;
94  if(testPeriod == superPeriod) {
95  numSubPeriods = i;
96  return true;
97  }
98  }
99  } catch(Error&) {
100  return false;
101  }
102 
103  return false;
104  }

118  std::vector<Real> sinkingNotionals(const Period& maturityTenor,
119  const Frequency& sinkingFrequency,
120  Rate couponRate,
121  Real initialNotional) {
122  Period freqPeriod(sinkingFrequency);
123  Integer nPeriods;
124  QL_REQUIRE(isSubPeriod(freqPeriod, maturityTenor, nPeriods),
125  "Bond frequency is incompatible with the maturity tenor");
126 
127  std::vector<Real> notionals(nPeriods+1);
128  notionals.front() = initialNotional;
129  Real coupon = couponRate / static_cast<Real>(sinkingFrequency);
130  Real compoundedInterest = 1.0;
131  Real totalValue = std::pow(1.0+coupon, nPeriods);
132  for(Size i = 0; i < (Size)nPeriods-1; ++i) {
133  compoundedInterest *= (1.0 + coupon);
134  Real currentNotional = 0.0;
135  if(coupon < 1.0e-12) {
136  currentNotional =
137  initialNotional*(1.0 - (i+1.0)/nPeriods);
138  }
139  else {
140  currentNotional =
141  initialNotional*(compoundedInterest - (compoundedInterest-1.0)/(1.0 - 1.0/totalValue));
142  }
143  notionals[i+1] = currentNotional;
144  }
145  notionals.back() = 0.0;
146  return notionals;
147  }
