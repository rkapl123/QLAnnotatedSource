140 std::pair<Integer,Integer> daysMinMax(const Period& p) {
141 switch (p.units()) {
142 case Days:
143 return std::make_pair(p.length(), p.length());
144 case Weeks:
145 return std::make_pair(7*p.length(), 7*p.length());
146 case Months:
147 return std::make_pair(28*p.length(), 31*p.length());
148 case Years:
149 return std::make_pair(365*p.length(), 366*p.length());
150 default:
151 QL_FAIL("unknown time unit (" << Integer(p.units()) << ")");
152 }
153 }

155 bool isSubPeriod(const Period& subPeriod,
156 const Period& superPeriod,
157 Integer& numSubPeriods) {
158
159 std::pair<Integer, Integer> superDays(daysMinMax(superPeriod));
160 std::pair<Integer, Integer> subDays(daysMinMax(subPeriod));
161
162 //obtain the approximate time ratio
163 Real minPeriodRatio =
164 ((Real)superDays.first)/((Real)subDays.second);
165 Real maxPeriodRatio =
166 ((Real)superDays.second)/((Real)subDays.first);
167 auto lowRatio = static_cast<Integer>(std::floor(minPeriodRatio));
168 auto highRatio = static_cast<Integer>(std::ceil(maxPeriodRatio));
169
170 try {
171 for(Integer i=lowRatio; i <= highRatio; ++i) {
172 Period testPeriod = subPeriod * i;
173 if(testPeriod == superPeriod) {
174 numSubPeriods = i;
175 return true;
176 }
177 }
178 } catch(Error&) {
179 return false;
180 }
181
182 return false;
183 }

187 std::vector<Real> sinkingNotionals(const Period& bondLength,
188 const Frequency& sinkingFrequency,
189 Rate couponRate,
190 Real initialNotional) {
191 Integer nPeriods;
192 QL_REQUIRE(isSubPeriod(Period(sinkingFrequency), bondLength, nPeriods),
193 "Bond frequency is incompatible with the maturity tenor");
194
195 std::vector<Real> notionals(nPeriods+1);
196 notionals.front() = initialNotional;
197 Real coupon = couponRate / static_cast<Real>(sinkingFrequency);
198 Real compoundedInterest = 1.0;
199 Real totalValue = std::pow(1.0+coupon, nPeriods);
200 for (Size i = 0; i < (Size)nPeriods-1; ++i) {
201 compoundedInterest *= (1.0 + coupon);
202 Real currentNotional = 0.0;
203 if(coupon < 1.0e-12) {
204 currentNotional = initialNotional*(1.0 - (i+1.0)/nPeriods);
205 } else {
206 currentNotional =
207 initialNotional*(compoundedInterest - (compoundedInterest-1.0)/(1.0 - 1.0/totalValue));
208 }
209 notionals[i+1] = currentNotional;
210 }
211 notionals.back() = 0.0;
212
213 return notionals;
214 }
