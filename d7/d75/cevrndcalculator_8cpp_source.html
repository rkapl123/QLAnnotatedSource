44  Real CEVRNDCalculator::massAtZero(Time t) const {
45  if (delta_ < 2.0)
46  return 1.0-boost::math::gamma_p(-0.5*delta_+1.0,x0_/(2.0*t));

48  return 0.0;

60  Real CEVRNDCalculator::pdf(Real f, Time t) const {
61  const Real y = X(f);
62 
63  if (delta_ < 2.0) {
64  return boost::math::pdf(
65  boost::math::non_central_chi_squared_distribution<Real>(
66  4.0-delta_, y/t), x0_/t)/t * 2.0*(1.0-beta_)*y/f;
67  }
68  else {
69  return boost::math::pdf(
70  boost::math::non_central_chi_squared_distribution<Real>(
71  delta_, x0_/t), y/t)/t * 2.0*(beta_-1.0)*y/f;
72  }
73  }

75  Real CEVRNDCalculator::cdf(Real f, Time t) const {
76  const Real y = X(f);
77 
78  if (delta_ < 2.0)
79  return 1.0 - boost::math::cdf(
80  boost::math::non_central_chi_squared_distribution<Real>(
81  2.0-delta_, y/t), x0_/t);
82  else
83  return 1.0 - boost::math::cdf(
84  boost::math::non_central_chi_squared_distribution<Real>(
85  delta_, x0_/t), y/t);

88  Real CEVRNDCalculator::sankaranApprox(Real c, Time t, Real x) const {
89  const Real a = x0_/t;
90  const Real b = 2.0 - delta_;
91 
92  c = std::max(c, -0.45*b);
93 
94  const Real h = 1 - 2*(b+c)*(b+3*c)/(3*square<Real>()(b+2*c));
95  const Real p = (b+2*c)/square<Real>()(b+c);
96  const Real m = (h-1)*(1-3*h);
97 
98  const Real u = (std::pow(a/(b+c), h) - (1 + h*p*(h-1-0.5*(2-h)*m*p)))/
99  (h*std::sqrt(2*p)*(1+0.5*m*p));
100 
101  return u - x;
102  }

104  Real CEVRNDCalculator::invcdf(Real q, Time t) const {
105  using namespace ext::placeholders;
106 
107  if (delta_ < 2.0) {
108  if (f0_ < QL_EPSILON || q < massAtZero(t))
109  return 0.0;
110 
111  const Real x = InverseCumulativeNormal()(1-q);
112 
113  const ext::function<Real(Real)> cdfApprox
114  = ext::bind(&CEVRNDCalculator::sankaranApprox,
115  this, _1, t, x);
116 
117  const Real y0 = X(f0_)/t;
118 
119  try {
120  Brent brent;
121  brent.setMaxEvaluations(20);
122  const Real guess =
123  invX(brent.solve(cdfApprox, 1e-8, y0, 0.02*y0) * t);
124 
125  return InvCDFHelper(this, guess, 1e-8, 100).inverseCDF(q, t);
126  }
127  catch (...) {
128  return InvCDFHelper(this, f0_, 1e-8, 100).inverseCDF(q, t);
129  }
130  }
131  else {
132  const Real x = t * boost::math::quantile(
133  boost::math::non_central_chi_squared_distribution<Real>(
134  delta_, x0_/t), 1-q);
135  return invX(x);
136  }
137  }
