63  Real FixedRateCoupon::accruedAmount(const Date& d) const {
64  if (d <= accrualStartDate_ || d > paymentDate_) {
65  // out of coupon range
66  return 0.0;
67  } else if (tradingExCoupon(d)) {
68  return -nominal()*(rate_.compoundFactor(d,
69  std::max(d, accrualEndDate_),
70  refPeriodStart_,
71  refPeriodEnd_) - 1.0);
72  } else {
73  // usual case
74  return nominal()*(rate_.compoundFactor(accrualStartDate_,
75  std::min(d,accrualEndDate_),
76  refPeriodStart_,
77  refPeriodEnd_) - 1.0);
78  }
79  }

167  FixedRateLeg::operator Leg() const {
168 
169  QL_REQUIRE(!couponRates_.empty(), "no coupon rates given");
170  QL_REQUIRE(!notionals_.empty(), "no notional given");
171 
172  Leg leg;
173  leg.reserve(schedule_.size()-1);
174 
175  // first period might be short or long
176  Date start = schedule_.date(0), end = schedule_.date(1);
177  Date paymentDate = paymentCalendar_.advance(end, paymentLag_, Days, paymentAdjustment_);
178  Date exCouponDate;
179  InterestRate rate = couponRates_[0];
180  Real nominal = notionals_[0];
181 
182  if (exCouponPeriod_ != Period())
183  {
184  exCouponDate = exCouponCalendar_.advance(paymentDate,
185  -exCouponPeriod_,
186  exCouponAdjustment_,
187  exCouponEndOfMonth_);
188  }
189  Date ref = schedule_.hasTenor() &&
190  schedule_.hasIsRegular() && !schedule_.isRegular(1) ?
191  schedule_.calendar().advance(end,
192  -schedule_.tenor(),
193  schedule_.businessDayConvention(),
194  schedule_.endOfMonth())
195  : start;
196  InterestRate r(rate.rate(),
197  firstPeriodDC_.empty() ? rate.dayCounter()
198  : firstPeriodDC_,
199  rate.compounding(), rate.frequency());
200  leg.push_back(ext::shared_ptr<CashFlow>(new
201  FixedRateCoupon(paymentDate, nominal, r,
202  start, end, ref, end, exCouponDate)));
203  // regular periods
204  for (Size i=2; i<schedule_.size()-1; ++i) {
205  start = end; end = schedule_.date(i);
206  Date paymentDate = paymentCalendar_.advance(end, paymentLag_, Days, paymentAdjustment_);
207  if (exCouponPeriod_ != Period())
208  {
209  exCouponDate = exCouponCalendar_.advance(paymentDate,
210  -exCouponPeriod_,
211  exCouponAdjustment_,
212  exCouponEndOfMonth_);
213  }
214  if ((i-1) < couponRates_.size())
215  rate = couponRates_[i-1];
216  else
217  rate = couponRates_.back();
218  if ((i-1) < notionals_.size())
219  nominal = notionals_[i-1];
220  else
221  nominal = notionals_.back();
222  leg.push_back(ext::shared_ptr<CashFlow>(new
223  FixedRateCoupon(paymentDate, nominal, rate,
224  start, end, start, end, exCouponDate)));
225  }
226  if (schedule_.size() > 2) {
227  // last period might be short or long
228  Size N = schedule_.size();
229  start = end; end = schedule_.date(N-1);
230  Date paymentDate = paymentCalendar_.advance(end, paymentLag_, Days, paymentAdjustment_);
231  if (exCouponPeriod_ != Period())
232  {
233  exCouponDate = exCouponCalendar_.advance(paymentDate,
234  -exCouponPeriod_,
235  exCouponAdjustment_,
236  exCouponEndOfMonth_);
237  }
238  if ((N-2) < couponRates_.size())
239  rate = couponRates_[N-2];
240  else
241  rate = couponRates_.back();
242  if ((N-2) < notionals_.size())
243  nominal = notionals_[N-2];
244  else
245  nominal = notionals_.back();
246  InterestRate r( rate.rate(), lastPeriodDC_.empty() ?
247  rate.dayCounter() :
248  lastPeriodDC_ , rate.compounding(), rate.frequency() );
249  if ((schedule_.hasIsRegular() && schedule_.isRegular(N - 1)) ||
250  !schedule_.hasTenor()) {
251  leg.push_back(ext::shared_ptr<CashFlow>(new
252  FixedRateCoupon(paymentDate, nominal, r,
253  start, end, start, end, exCouponDate)));
254  } else {
255  Date ref = schedule_.calendar().advance(
256  start,
257  schedule_.tenor(),
258  schedule_.businessDayConvention(),
259  schedule_.endOfMonth());
260  leg.push_back(ext::shared_ptr<CashFlow>(new
261  FixedRateCoupon(paymentDate, nominal, r,
262  start, end, start, ref, exCouponDate)));
263  }
264  }
265  return leg;
266  }
