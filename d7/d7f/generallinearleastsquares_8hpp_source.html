115  template <class xIterator, class yIterator, class vIterator>
116  void GeneralLinearLeastSquares::calculate(xIterator xBegin, xIterator xEnd,
117  yIterator yBegin, yIterator yEnd,
118  vIterator vBegin) {
119 
120  const Size n = residuals_.size();
121  const Size m = err_.size();
122 
123  QL_REQUIRE( n == Size(std::distance(yBegin, yEnd)),
124  "sample set need to be of the same size");
125  QL_REQUIRE(n >= m, "sample set is too small");
126 
127  Size i;
128 
129  Matrix A(n, m);
130  for (i=0; i<m; ++i)
131  std::transform(xBegin, xEnd, A.column_begin(i), *vBegin++);
132 
133  const SVD svd(A);
134  const Matrix& V = svd.V();
135  const Matrix& U = svd.U();
136  const Array& w = svd.singularValues();
137  const Real threshold = n * QL_EPSILON * svd.singularValues()[0];
138 
139  for (i=0; i<m; ++i) {
140  if (w[i] > threshold) {
141  const Real u = std::inner_product(U.column_begin(i),
142  U.column_end(i),
143  yBegin, 0.0)/w[i];
144 
145  for (Size j=0; j<m; ++j) {
146  a_[j] +=u*V[j][i];
147  err_[j]+=V[j][i]*V[j][i]/(w[i]*w[i]);
148  }
149  }
150  }
151  err_ = Sqrt(err_);
152  Array tmp = A*a_;
153  std::transform(tmp.begin(), tmp.end(),
154  yBegin, residuals_.begin(), std::minus<Real>());
155 
156  const Real chiSq
157  = std::inner_product(residuals_.begin(), residuals_.end(),
158  residuals_.begin(), 0.0);
159  std::transform(err_.begin(), err_.end(), standardErrors_.begin(),
160  std::bind1st(std::multiplies<Real>(),
161  std::sqrt(chiSq/(n-2))));
162  }
