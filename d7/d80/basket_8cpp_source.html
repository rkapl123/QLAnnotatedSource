31  Basket::Basket(const Date& refDate,
32  const vector<string>& names,
33  const vector<Real>& notionals,
34  const ext::shared_ptr<Pool>& pool,
35  Real attachment,
36  Real detachment,
37  const ext::shared_ptr<Claim>& claim)
38  : notionals_(notionals), pool_(pool), claim_(claim), attachmentRatio_(attachment),
39  detachmentRatio_(detachment), basketNotional_(0.0), attachmentAmount_(0.0),
40  detachmentAmount_(0.0), trancheNotional_(0.0), refDate_(refDate) {
41  QL_REQUIRE(!notionals_.empty(), "notionals empty");
42  QL_REQUIRE (attachmentRatio_ >= 0 &&
43  attachmentRatio_ <= detachmentRatio_ &&
44  detachmentRatio_ <= 1,
45  "invalid attachment/detachment ratio");
46  QL_REQUIRE(pool_, "Empty pool pointer.");
47  QL_REQUIRE(notionals_.size() == pool_->size(),
48  "unmatched data entry sizes in basket");
49 
50  // registrations relevant to the loss status, not to the expected
51  // loss values; those are through models.
52  registerWith(Settings::instance().evaluationDate());
53  registerWith(claim_);
54 
55  computeBasket();
56 
57  // At this point Issuers in the pool might or might not have
58  // probability term structures for the defultKeys(eventType+
59  // currency+seniority) entering in this basket. This is not
60  // necessarily a problem.
61  for (Size i = 0; i < notionals_.size(); i++) {
62  basketNotional_ += notionals_[i];
63  attachmentAmount_ += notionals_[i] * attachmentRatio_;
64  detachmentAmount_ += notionals_[i] * detachmentRatio_;
65  }
66  trancheNotional_ = detachmentAmount_ - attachmentAmount_;
67  }

117  Real Basket::cumulatedLoss(const Date& endDate) const {
118  QL_REQUIRE(endDate >= refDate_,
119  "Target date lies before basket inception");
120  Real loss = 0.0;
121  for (Size i = 0; i < size(); i++) {
122  ext::shared_ptr<DefaultEvent> credEvent =
123  pool_->get(pool_->names()[i]).defaultedBetween(refDate_,
124  endDate, pool_->defaultKeys()[i]);
125  if (credEvent != 0) {
126  /* \todo If the event has not settled one would need to
127  introduce some model recovery rate (independently of a loss
128  model) This remains to be done.
129  */
130  if(credEvent->hasSettled())
131  loss += claim_->amount(credEvent->date(),
132  // notionals_[i],
133  exposure(pool_->names()[i], credEvent->date()),
134  credEvent->settlement().recoveryRate(
135  pool_->defaultKeys()[i].seniority()));
136  }
137  }
138  return loss;
139  }

149  endDate, pool_->defaultKeys()[i]);
150  if (credEvent != 0) {

152  loss += claim_->amount(credEvent->date(),
153  //notionals_[i],

155  //NOtice I am requesting an exposure in the past...

159  pool_->defaultKeys()[i].seniority()));
160  }
161  }
162  }
163  return loss;

299  Probability Basket::probOverLoss(const Date& d, Real lossFraction) const {
300  // convert initial basket fraction to remaining basket fraction
301  calculate();
302  // if eaten up all the tranche the prob of losing any amount is 1
303  // (we have already lost it)
304  if(evalDateRemainingNot_ == 0.) return 1.;
305 
306  // Turn to live (remaining) tranche units to feed into the model request
307  Real xPtfl = attachmentAmount_ +
308  (detachmentAmount_-attachmentAmount_)*lossFraction;
309  Real xPrim = (xPtfl- evalDateAttachAmount_)/
310  (detachmentAmount_-evalDateAttachAmount_);
311  // in live tranche fractional units
312  // if the level falls within realized losses the prob is 1.
313  if(xPtfl < 0.) return 1.;
314 
315  return lossModel_->probOverLoss(d, xPrim);
316  }
