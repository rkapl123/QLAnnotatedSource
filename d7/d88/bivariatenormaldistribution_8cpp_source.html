56  Real BivariateCumulativeNormalDistributionDr78::operator()(Real a,
57  Real b) const {
58 
59  CumulativeNormalDistribution cumNormalDist;
60  Real CumNormDistA = cumNormalDist(a);
61  Real CumNormDistB = cumNormalDist(b);
62  Real MaxCumNormDistAB = std::max(CumNormDistA, CumNormDistB);
63  Real MinCumNormDistAB = std::min(CumNormDistA, CumNormDistB);
64 
65  if (1.0-MaxCumNormDistAB<1e-15)
66  return MinCumNormDistAB;
67 
68  if (MinCumNormDistAB<1e-15)
69  return MinCumNormDistAB;
70 
71  Real a1 = a / std::sqrt(2.0 * (1.0 - rho2_));
72  Real b1 = b / std::sqrt(2.0 * (1.0 - rho2_));
73 
74  Real result=-1.0;
75 
76  if (a<=0.0 && b<=0 && rho_<=0) {
77  Real sum=0.0;
78  for (Size i=0; i<5; i++) {
79  for (Size j=0;j<5; j++) {
80  sum += x_[i]*x_[j]*
81  std::exp(a1*(2.0*y_[i]-a1)+b1*(2.0*y_[j]-b1)
82  +2.0*rho_*(y_[i]-a1)*(y_[j]-b1));
83  }
84  }
85  result = std::sqrt(1.0 - rho2_)/M_PI*sum;
86  } else if (a<=0 && b>=0 && rho_>=0) {
87  BivariateCumulativeNormalDistributionDr78 bivCumNormalDist(-rho_);
88  result= CumNormDistA - bivCumNormalDist(a, -b);
89  } else if (a>=0.0 && b<=0.0 && rho_>=0.0) {
90  BivariateCumulativeNormalDistributionDr78 bivCumNormalDist(-rho_);
91  result= CumNormDistB - bivCumNormalDist(-a, b);
92  } else if (a>=0.0 && b>=0.0 && rho_<=0.0) {
93  result= CumNormDistA + CumNormDistB -1.0 + (*this)(-a, -b);
94  } else if (a*b*rho_>0.0) {
95  Real rho1 = (rho_*a-b)*(a>0.0 ? 1.0: -1.0)/
96  std::sqrt(a*a-2.0*rho_*a*b+b*b);
97  BivariateCumulativeNormalDistributionDr78 bivCumNormalDist(rho1);
98 
99  Real rho2 = (rho_*b-a)*(b>0.0 ? 1.0: -1.0)/
100  std::sqrt(a*a-2.0*rho_*a*b+b*b);
101  BivariateCumulativeNormalDistributionDr78 CBND2(rho2);
102 
103  Real delta = (1.0-(a>0.0 ? 1.0: -1.0)*(b>0.0 ? 1.0: -1.0))/4.0;
104 
105  result= bivCumNormalDist(a, 0.0) + CBND2(b, 0.0) - delta;
106  } else {
107  QL_FAIL("case not handled");
108  }
109 
110  return result;
111  }

118  class eqn3 { /* Relates to eqn3 Genz 2004 */
119  public:
120  eqn3(Real h, Real k, Real asr) {
121  hk_ = h * k;
122  hs_ = (h * h + k * k) / 2;
123  asr_ = asr;
124  }
125  Real operator()(Real x) const {
126  Real sn = std::sin(asr_ * (-x + 1) * 0.5);
127  return std::exp((sn * hk_ - hs_) / (1.0 - sn * sn));
128  }
129  private:
130  Real hk_, asr_, hs_;
131  };
132 
133  class eqn6 { /* Relates to eqn6 Genz 2004 */
134  public:
135  eqn6(Real a, Real c, Real d, Real bs, Real hk)
136  : a_(a), c_(c), d_(d), bs_(bs), hk_(hk) {}
137  Real operator()(Real x) const {
138  Real xs = a_ * (-x + 1);
139  xs = std::fabs(xs*xs);
140  Real rs = std::sqrt(1 - xs);
141  Real asr = -(bs_ / xs + hk_) / 2;
142  if (asr > -100.0) {
143  return (a_ * std::exp(asr) *
144  (std::exp(-hk_ * (1 - rs) / (2 * (1 + rs))) / rs -
145  (1 + c_ * xs * (1 + d_ * xs))));
146  } else {
147  return 0.0;
148  }
149  }
150  private:
151  Real a_, c_, d_, bs_, hk_;
152  };

167  Real BivariateCumulativeNormalDistributionWe04DP::operator()(
168  Real x, Real y) const {
169 
170  /* The implementation is described at section 2.4 "Hybrid
171  Numerical Integration Algorithms" of "Numerical Computation
172  of Rectangular Bivariate an Trivariate Normal and t
173  Probabilities", Genz (2004), Statistics and Computing 14,
174  151-160. (available at
175  www.sci.wsu.edu/math/faculty/henz/homepage)
176 
177  The Gauss-Legendre quadrature have been extracted to
178  TabulatedGaussLegendre (x,w zero-based)
179 
180  Tthe functions ot be integrated numerically have been moved
181  to classes eqn3 and eqn6
182 
183  Change some magic numbers to M_PI */
184 
185  TabulatedGaussLegendre gaussLegendreQuad(20);
186  if (std::fabs(correlation_) < 0.3) {
187  gaussLegendreQuad.order(6);
188  } else if (std::fabs(correlation_) < 0.75) {
189  gaussLegendreQuad.order(12);
190  }
191 
192  Real h = -x;
193  Real k = -y;
194  Real hk = h * k;
195  Real BVN = 0.0;
196 
197  if (std::fabs(correlation_) < 0.925)
198  {
199  if (std::fabs(correlation_) > 0)
200  {
201  Real asr = std::asin(correlation_);
202  eqn3 f(h,k,asr);
203  BVN = gaussLegendreQuad(f);
204  BVN *= asr * (0.25 / M_PI);
205  }
206  BVN += cumnorm_(-h) * cumnorm_(-k);
207  }
208  else
209  {
210  if (correlation_ < 0)
211  {
212  k *= -1;
213  hk *= -1;
214  }
215  if (std::fabs(correlation_) < 1)
216  {
217  Real Ass = (1 - correlation_) * (1 + correlation_);
218  Real a = std::sqrt(Ass);
219  Real bs = (h-k)*(h-k);
220  Real c = (4 - hk) / 8;
221  Real d = (12 - hk) / 16;
222  Real asr = -(bs / Ass + hk) / 2;
223  if (asr > -100)
224  {
225  BVN = a * std::exp(asr) *
226  (1 - c * (bs - Ass) * (1 - d * bs / 5) / 3 +
227  c * d * Ass * Ass / 5);
228  }
229  if (-hk < 100)
230  {
231  Real B = std::sqrt(bs);
232  BVN -= std::exp(-hk / 2) * 2.506628274631 *
233  cumnorm_(-B / a) * B *
234  (1 - c * bs * (1 - d * bs / 5) / 3);
235  }
236  a /= 2;
237  eqn6 f(a,c,d,bs,hk);
238  BVN += gaussLegendreQuad(f);
239  BVN /= (-2.0 * M_PI);
240  }
241 
242  if (correlation_ > 0) {
243  BVN += cumnorm_(-std::max(h, k));
244  } else {
245  BVN *= -1;
246  if (k > h) {
247  // evaluate cumnorm where it is most precise, that
248  // is in the lower tail because of double accuracy
249  // around 0.0 vs around 1.0
250  if (h >= 0) {
251  BVN += cumnorm_(-h) - cumnorm_(-k);
252  } else {
253  BVN += cumnorm_(k) - cumnorm_(h);
254  }
255  }
256  }
257  }
258  return BVN;
259  }
