63 void GaussianRandomDefaultModel::nextSequence(Real tmax) {
64 const std::vector<Real>& values = rsg_.nextSequence().value;
65 Real a = sqrt(copula_->correlation());
66 for (Size j = 0; j < pool_->size(); j++) {
67 const string name = pool_->names()[j];
68 const Handle<DefaultProbabilityTermStructure>&
69 dts = pool_->get(name).defaultProbability(defaultKeys_[j]);
70
71 Real y = a * values[0] + sqrt(1-a*a) * values[j+1];
72 Real p = CumulativeNormalDistribution()(y);
73
74 if (dts->defaultProbability(tmax) < p)
75 pool_->setTime(name, tmax+1);
76 else
77 pool_->setTime(name, Brent().solve(Root(dts,p),accuracy_,0,1));
78 }
79 }
