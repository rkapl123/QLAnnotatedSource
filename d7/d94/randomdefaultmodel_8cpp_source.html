62  void GaussianRandomDefaultModel::nextSequence(Real tmax) {
63  const std::vector<Real>& values = rsg_.nextSequence().value;
64  Real a = sqrt(copula_->correlation());
65  for (Size j = 0; j < pool_->size(); j++) {
66  const string name = pool_->names()[j];
67  const Handle<DefaultProbabilityTermStructure>&
68  dts = pool_->get(name).defaultProbability(defaultKeys_[j]);
69 
70  Real y = a * values[0] + sqrt(1-a*a) * values[j+1];
71  Real p = CumulativeNormalDistribution()(y);
72 
73  if (dts->defaultProbability(tmax) < p)
74  pool_->setTime(name, tmax+1);
75  else
76  pool_->setTime(name, Brent().solve(Root(dts,p),accuracy_,0,1));
77  }
78  }
