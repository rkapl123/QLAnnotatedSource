58  Real ValueEstimate::value(const Array& parameters) const {
59  std::copy(parameters.begin(), parameters.end(),
60  parameters_.begin());
61  Real sum = 0.0;
62  Size n = 0;
63  for (Size i=0; i<simulationData_.size(); ++i) {
64  if (simulationData_[i].isValid) {
65  ++n;
66  if (exercise_.exercise(exerciseIndex_,
67  parameters_,
68  simulationData_[i].values))
69  sum += simulationData_[i].exerciseValue;
70  else
71  sum += simulationData_[i].cumulatedCashFlows;
72  }
73  }
74  return -sum/n;

81  Real genericEarlyExerciseOptimization(
82  std::vector<std::vector<NodeData> >& simulationData,
83  const ParametricExercise& exercise,
84  std::vector<std::vector<Real> >& parameters,
85  const EndCriteria& endCriteria,
86  OptimizationMethod& method) {
87 
88  Size steps = simulationData.size();
89  parameters.resize(steps-1);
90 
91  for (Size i=steps-1; i!=0; --i) {
92  const std::vector<NodeData>& exerciseData = simulationData[i];
93 
94  parameters[i-1].resize(exercise.numberOfParameters()[i-1]);
95 
96 
97  // optimize
98  ValueEstimate f(exerciseData, exercise, i-1);
99 
100  Array guess(parameters[i-1].size());
101  exercise.guess(i-1, parameters[i-1]);
102  std::copy(parameters[i-1].begin(), parameters[i-1].end(),
103  guess.begin());
104 
105  NoConstraint c;
106 
107  Problem p(f, c, guess);
108  method.minimize(p, endCriteria);
109 
110  Array result = p.currentValue();
111  std::copy(result.begin(), result.end(),
112  parameters[i-1].begin());
113 
114  std::vector<NodeData>& previousData = simulationData[i-1];
115  for (Size j=0; j<previousData.size(); ++j) {
116  if (exerciseData[j].isValid) {
117  if (exercise.exercise(i-1,
118  parameters[i-1],
119  exerciseData[j].values))
120  previousData[j].cumulatedCashFlows +=
121  exerciseData[j].exerciseValue;
122  else
123  previousData[j].cumulatedCashFlows +=
124  exerciseData[j].cumulatedCashFlows;
125  }
126  }
127  }
128 
129  Real sum = 0.0;
130  const std::vector<NodeData>& initialData = simulationData.front();
131  for (Size i=0; i<initialData.size(); ++i)
132  sum += initialData[i].cumulatedCashFlows;
133  return sum/initialData.size();
134  }
