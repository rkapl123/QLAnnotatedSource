102  void update() {
103  std::vector<Real>::const_iterator x = this->xBegin_;
104  std::vector<Real>::const_iterator y = this->yBegin_;
105  std::vector<Real> times, blackVols;
106  for ( ; x!=this->xEnd_; ++x, ++y) {
107  times.push_back(*x);
108  blackVols.push_back(*y);
109  }
110  abcdCalibrator_ = ext::shared_ptr<AbcdCalibration>(
111  new AbcdCalibration(times, blackVols,
112  a_, b_, c_, d_,
113  aIsFixed_, bIsFixed_,
114  cIsFixed_, dIsFixed_,
115  vegaWeighted_,
116  endCriteria_,
117  optMethod_));
118  abcdCalibrator_->compute();
119  a_ = abcdCalibrator_->a();
120  b_ = abcdCalibrator_->b();
121  c_ = abcdCalibrator_->c();
122  d_ = abcdCalibrator_->d();
123  k_ = abcdCalibrator_->k(times, blackVols);
124  error_ = abcdCalibrator_->error();
125  maxError_ = abcdCalibrator_->maxError();
126  abcdEndCriteria_ = abcdCalibrator_->endCriteria();
127  }
128  Real value(Real x) const {
129  QL_REQUIRE(x>=0.0, "time must be non negative: " <<
130  x << " not allowed");
131  return abcdCalibrator_->value(x);
132  }

134  QL_FAIL("Abcd primitive not implemented");
135  }

137  QL_FAIL("Abcd derivative not implemented");

139  Real secondDerivative(Real) const {

141  }
142  Real k(Time t) const {
