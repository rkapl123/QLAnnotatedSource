105  void update() override {
106  auto x = this->xBegin_;
107  auto y = this->yBegin_;
108  std::vector<Real> times, blackVols;
109  for ( ; x!=this->xEnd_; ++x, ++y) {
110  times.push_back(*x);
111  blackVols.push_back(*y);
112  }
113  abcdCalibrator_ = ext::shared_ptr<AbcdCalibration>(
114  new AbcdCalibration(times, blackVols,
115  a_, b_, c_, d_,
116  aIsFixed_, bIsFixed_,
117  cIsFixed_, dIsFixed_,
118  vegaWeighted_,
119  endCriteria_,
120  optMethod_));
121  abcdCalibrator_->compute();
122  a_ = abcdCalibrator_->a();
123  b_ = abcdCalibrator_->b();
124  c_ = abcdCalibrator_->c();
125  d_ = abcdCalibrator_->d();
126  k_ = abcdCalibrator_->k(times, blackVols);
127  error_ = abcdCalibrator_->error();
128  maxError_ = abcdCalibrator_->maxError();
129  abcdEndCriteria_ = abcdCalibrator_->endCriteria();
130  }
131  Real value(Real x) const override {
132  QL_REQUIRE(x>=0.0, "time must be non negative: " <<
133  x << " not allowed");
134  return abcdCalibrator_->value(x);
135  }

140  }
141  Real k(Time t) const {
