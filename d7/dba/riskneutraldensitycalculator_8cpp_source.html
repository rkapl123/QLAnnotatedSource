46  Real RiskNeutralDensityCalculator::InvCDFHelper::inverseCDF(Real p, Time t)
47  const {
48  const Real guessCDF = calculator_->cdf(guess_, t);
49 
50  Size evaluations = maxEvaluations_;
51  Real upper = guess_, lower = guess_;
52 
53  if (guessCDF < p)
54  while (calculator_->cdf(upper*=1.5, t) < p && evaluations > 0) {
55  --evaluations;
56  }
57  else
58  while (calculator_->cdf(lower*=0.75, t) > p && evaluations > 0) {
59  --evaluations;
60  }
61 
62  QL_REQUIRE(evaluations, "could not calculate interval");
63 
64  const boost::function<Real(Real)> cdf
65  = boost::bind(&RiskNeutralDensityCalculator::cdf,
66  calculator_, _1, t);
67 
68  Brent solver;
69  solver.setMaxEvaluations(evaluations);
70  return solver.solve(compose(std::bind2nd(std::minus<Real>(), p), cdf),
71  accuracy_, 0.5*(lower + upper), lower, upper);
72  }
