43  Date
44  YoYOptionletVolatilitySurface::baseDate() const {
45 
46  // Depends on interpolation, or not, of observed index
47  // and observation lag with which it was built.
48  // We want this to work even if the index does not
49  // have a yoy term structure.
50  if (indexIsInterpolated()) {
51  return referenceDate() - observationLag();
52  } else {
53  return inflationPeriod(referenceDate() - observationLag(),
54  frequency()).first;
55  }
56  }

59  void YoYOptionletVolatilitySurface::checkRange(const Date& d, Rate strike,
60  bool extrapolate) const {
61  QL_REQUIRE(d >= baseDate(),
62  "date (" << d << ") is before base date");
63  QL_REQUIRE(extrapolate || allowsExtrapolation() || d <= maxDate(),
64  "date (" << d << ") is past max curve date ("
65  << maxDate() << ")");
66  QL_REQUIRE(extrapolate || allowsExtrapolation() ||
67  (strike >= minStrike() && strike <= maxStrike()),
68  "strike (" << strike << ") is outside the curve domain ["
69  << minStrike() << "," << maxStrike()<< "]] at date = " << d);
70  }

87  Volatility
88  YoYOptionletVolatilitySurface::volatility(const Date& maturityDate,
89  Rate strike,
90  const Period &obsLag,
91  bool extrapolate) const {
92 
93  Period useLag = obsLag;
94  if (obsLag==Period(-1,Days)) {
95  useLag = observationLag();
96  }
97 
98  if (indexIsInterpolated()) {
99  YoYOptionletVolatilitySurface::checkRange(maturityDate-useLag, strike, extrapolate);
100  Time t = timeFromReference(maturityDate-useLag);
101  return volatilityImpl(t,strike);
102  } else {
103  std::pair<Date,Date> dd = inflationPeriod(maturityDate-useLag, frequency());
104  YoYOptionletVolatilitySurface::checkRange(dd.first, strike, extrapolate);
105  Time t = timeFromReference(dd.first);
106  return volatilityImpl(t,strike);
107  }
108  }

122  Time
123  YoYOptionletVolatilitySurface::timeFromBase(const Date &maturityDate,
124  const Period& obsLag) const {
125 
126  Period useLag = obsLag;
127  if (obsLag==Period(-1,Days)) {
128  useLag = observationLag();
129  }
130 
131  Date useDate;
132  if (indexIsInterpolated()) {
133  useDate = maturityDate - useLag;
134  } else {
135  useDate = inflationPeriod(maturityDate - useLag,
136  frequency()).first;
137  }
138 
139  // This assumes that the inflation term structure starts
140  // as late as possible given the inflation index definition,
141  // which is the usual case.
142  return dayCounter().yearFraction(baseDate(), useDate);
143  }
