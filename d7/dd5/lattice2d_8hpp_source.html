103  template <class Impl, class T>
104  Size TreeLattice2D<Impl,T>::descendant(Size i, Size index,
105  Size branch) const {
106  Size modulo = tree1_->size(i);
107 
108  Size index1 = index % modulo;
109  Size index2 = index / modulo;
110  Size branch1 = branch % T::branches;
111  Size branch2 = branch / T::branches;
112 
113  modulo = tree1_->size(i+1);
114  return tree1_->descendant(i, index1, branch1) +
115  tree2_->descendant(i, index2, branch2)*modulo;
116  }

118  template <class Impl, class T>
119  Real TreeLattice2D<Impl,T>::probability(Size i, Size index,
120  Size branch) const {
121  Size modulo = tree1_->size(i);
122 
123  Size index1 = index % modulo;
124  Size index2 = index / modulo;
125  Size branch1 = branch % T::branches;
126  Size branch2 = branch / T::branches;
127 
128  Real prob1 = tree1_->probability(i, index1, branch1);
129  Real prob2 = tree2_->probability(i, index2, branch2);
130  // does the 36 below depend on T::branches?
131  return prob1*prob2 + rho_*(m_[branch1][branch2])/36.0;
132  }
