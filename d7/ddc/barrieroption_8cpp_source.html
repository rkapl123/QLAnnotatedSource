52  Volatility BarrierOption::impliedVolatility(
53  Real targetValue,
54  const ext::shared_ptr<GeneralizedBlackScholesProcess>& process,
55  Real accuracy,
56  Size maxEvaluations,
57  Volatility minVol,
58  Volatility maxVol) const {
59 
60  QL_REQUIRE(!isExpired(), "option expired");
61 
62  ext::shared_ptr<SimpleQuote> volQuote(new SimpleQuote);
63 
64  ext::shared_ptr<GeneralizedBlackScholesProcess> newProcess =
65  detail::ImpliedVolatilityHelper::clone(process, volQuote);
66 
67  // engines are built-in for the time being
68  boost::scoped_ptr<PricingEngine> engine;
69  switch (exercise_->type()) {
70  case Exercise::European:
71  engine.reset(new AnalyticBarrierEngine(newProcess));
72  break;
73  case Exercise::American:
74  case Exercise::Bermudan:
75  QL_FAIL("engine not available for non-European barrier option");
76  break;
77  default:
78  QL_FAIL("unknown exercise type");
79  }
80 
81  return detail::ImpliedVolatilityHelper::calculate(*this,
82  *engine,
83  *volQuote,
84  targetValue,
85  accuracy,
86  maxEvaluations,
87  minVol, maxVol);
88  }

112  bool BarrierOption::engine::triggered(Real underlying) const {
113  switch (arguments_.barrierType) {
114  case Barrier::DownIn:
115  case Barrier::DownOut:
116  return underlying < arguments_.barrier;
117  case Barrier::UpIn:
118  case Barrier::UpOut:

121  QL_FAIL("unknown type");
