51 Volatility BarrierOption::impliedVolatility(
52 Real targetValue,
53 const ext::shared_ptr<GeneralizedBlackScholesProcess>& process,
54 Real accuracy,
55 Size maxEvaluations,
56 Volatility minVol,
57 Volatility maxVol) const {
58
59 QL_REQUIRE(!isExpired(), "option expired");
60
61 ext::shared_ptr<SimpleQuote> volQuote(new SimpleQuote);
62
63 ext::shared_ptr<GeneralizedBlackScholesProcess> newProcess =
64 detail::ImpliedVolatilityHelper::clone(process, volQuote);
65
66 // engines are built-in for the time being
67 std::unique_ptr<PricingEngine> engine;
68 switch (exercise_->type()) {
69 case Exercise::European:
70 engine = std::make_unique<AnalyticBarrierEngine>(newProcess);
71 break;
72 case Exercise::American:
73 case Exercise::Bermudan:
74 QL_FAIL("engine not available for non-European barrier option");
75 break;
76 default:
77 QL_FAIL("unknown exercise type");
78 }
79
80 return detail::ImpliedVolatilityHelper::calculate(*this,
81 *engine,
82 *volQuote,
83 targetValue,
84 accuracy,
85 maxEvaluations,
86 minVol, maxVol);
87 }

94 void BarrierOption::arguments::validate() const {
95 OneAssetOption::arguments::validate();
96
97 switch (barrierType) {
98 case Barrier::DownIn:
99 case Barrier::UpIn:
100 case Barrier::DownOut:
101 case Barrier::UpOut:
102 break;
103 default:
104 QL_FAIL("unknown type");
105 }
106
107 QL_REQUIRE(barrier != Null<Real>(), "no barrier given");
108 QL_REQUIRE(rebate != Null<Real>(), "no rebate given");
109 }

111 bool BarrierOption::engine::triggered(Real underlying) const {
112 switch (arguments_.barrierType) {
113 case Barrier::DownIn:
114 case Barrier::DownOut:
115 return underlying < arguments_.barrier;
116 case Barrier::UpIn:
117 case Barrier::UpOut:
118 return underlying > arguments_.barrier;
119 default:
120 QL_FAIL("unknown type");
121 }
122 }
