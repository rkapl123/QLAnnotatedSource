86  void CreditDefaultSwap::init(const Schedule& schedule, BusinessDayConvention paymentConvention,
87  const DayCounter& dayCounter, const DayCounter& lastPeriodDayCounter,
88  bool rebatesAccrual, const Date& upfrontDate) {
89 
90  QL_REQUIRE(!schedule.empty(), "CreditDefaultSwap needs a non-empty schedule.");
91 
92  bool postBigBang = false;
93  if (schedule.hasRule()) {
94  DateGeneration::Rule rule = schedule.rule();
95  postBigBang = rule == DateGeneration::CDS || rule == DateGeneration::CDS2015;
96  }
97 
98  if (!postBigBang) {
99  QL_REQUIRE(protectionStart_ <= schedule[0], "protection can not start after accrual");
100  }
101 
102  leg_ = FixedRateLeg(schedule)
103  .withNotionals(notional_)
104  .withCouponRates(runningSpread_, dayCounter)
105  .withPaymentAdjustment(paymentConvention)
106  .withLastPeriodDayCounter(lastPeriodDayCounter);
107 
108  // Deduce the trade date if not given.
109  if (tradeDate_ == Date()) {
110  if (postBigBang) {
111  tradeDate_ = protectionStart_;
112  } else {
113  tradeDate_ = protectionStart_ - 1;
114  }
115  }
116 
117  // Deduce the cash settlement date if not given.
118  Date effectiveUpfrontDate = upfrontDate;
119  if (effectiveUpfrontDate == Date()) {
120  effectiveUpfrontDate = schedule.calendar().advance(tradeDate_,
121  cashSettlementDays_, Days, paymentConvention);
122  }
123  QL_REQUIRE(effectiveUpfrontDate >= protectionStart_, "The cash settlement date must not " <<
124  "be before the protection start date.");
125 
126  // Create the upfront payment, if one is provided.
127  Real upfrontAmount = 0.0;
128  if (upfront_) // NOLINT(readability-implicit-bool-conversion)
129  upfrontAmount = *upfront_ * notional_;
130  upfrontPayment_ = ext::make_shared<SimpleCashFlow>(upfrontAmount, effectiveUpfrontDate);
131 
132  // Set the maturity date.
133  maturity_ = schedule.dates().back();
134 
135  // Deal with the accrual rebate. We use the standard conventions for accrual calculation introduced with the
136  // CDS Big Bang in 2009.
137  if (rebatesAccrual) {
138 
139  Real rebateAmount = 0.0;
140  Date refDate = tradeDate_ + 1;
141 
142  if (tradeDate_ >= schedule.dates().front()) {
143  for (Size i = 0; i < leg_.size(); ++i) {
144  const ext::shared_ptr<CashFlow>& cf = leg_[i];
145  if (refDate < cf->date()) {
146  // Calculate the accrual. The most likely scenario.
147  ext::shared_ptr<FixedRateCoupon> frc = ext::dynamic_pointer_cast<FixedRateCoupon>(cf);
148  rebateAmount = frc->accruedAmount(refDate);
149  break;
150  } else if (refDate == cf->date() && i < leg_.size() - 1) {
151  // If not the last coupon and trade date + 1 is the next coupon payment date,
152  // the accrual is 0 so do nothing.
153  break;
154  } else {
155  // Must have trade date + 1 >= last coupon's payment date. '>' here probably does not make
156  // sense - should possibly have an exception above if trade date >= last coupon's date.
157  ext::shared_ptr<FixedRateCoupon> frc = ext::dynamic_pointer_cast<FixedRateCoupon>(cf);
158  rebateAmount = frc->amount();
159  break;
160  }
161  }
162  }
163 
164  accrualRebate_ = ext::make_shared<SimpleCashFlow>(rebateAmount, effectiveUpfrontDate);
165  }
166 
167  if (!claim_)
168  claim_ = ext::make_shared<FaceValueClaim>();
169  registerWith(claim_);
170  }

334  Rate CreditDefaultSwap::impliedHazardRate(
335  Real targetNPV,
336  const Handle<YieldTermStructure>& discountCurve,
337  const DayCounter& dayCounter,
338  Real recoveryRate,
339  Real accuracy,
340  PricingModel model) const {
341 
342  ext::shared_ptr<SimpleQuote> flatRate = ext::make_shared<SimpleQuote>(0.0);
343 
344  Handle<DefaultProbabilityTermStructure> probability =
345  Handle<DefaultProbabilityTermStructure>(
346  ext::make_shared<FlatHazardRate>(0, WeekendsOnly(),
347  Handle<Quote>(flatRate), dayCounter));
348 
349  ext::shared_ptr<PricingEngine> engine;
350  switch (model) {
351  case Midpoint:
352  engine = ext::make_shared<MidPointCdsEngine>(
353  probability, recoveryRate, discountCurve);
354  break;
355  case ISDA:
356  engine = ext::make_shared<IsdaCdsEngine>(
357  probability, recoveryRate, discountCurve,
358  boost::none,
359  IsdaCdsEngine::Taylor,
360  IsdaCdsEngine::HalfDayBias,
361  IsdaCdsEngine::Piecewise);
362  break;
363  default:
364  QL_FAIL("unknown CDS pricing model: " << model);
365  }
366 
367  setupArguments(engine->getArguments());
368  const auto* results = dynamic_cast<const CreditDefaultSwap::results*>(engine->getResults());
369 
370  ObjectiveFunction f(targetNPV, *flatRate, *engine, results);
371  //very close guess if targetNPV = 0.
372  Rate guess = runningSpread_ / (1 - recoveryRate) * 365./360.;
373  Real step = 0.1 * guess;
374  return Brent().solve(f, accuracy, guess, step);
375  }

377  Rate CreditDefaultSwap::conventionalSpread(
378  Real conventionalRecovery,
379  const Handle<YieldTermStructure>& discountCurve,
380  const DayCounter& dayCounter,
381  PricingModel model) const {
382 
383  ext::shared_ptr<SimpleQuote> flatRate = ext::make_shared<SimpleQuote>(0.0);
384 
385  Handle<DefaultProbabilityTermStructure> probability =
386  Handle<DefaultProbabilityTermStructure>(
387  ext::make_shared<FlatHazardRate>(0, WeekendsOnly(),
388  Handle<Quote>(flatRate), dayCounter));
389 
390  ext::shared_ptr<PricingEngine> engine;
391  switch (model) {
392  case Midpoint:
393  engine = ext::make_shared<MidPointCdsEngine>(
394  probability, conventionalRecovery, discountCurve);
395  break;
396  case ISDA:
397  engine = ext::make_shared<IsdaCdsEngine>(
398  probability, conventionalRecovery, discountCurve,
399  boost::none,
400  IsdaCdsEngine::Taylor,
401  IsdaCdsEngine::HalfDayBias,
402  IsdaCdsEngine::Piecewise);
403  break;
404  default:
405  QL_FAIL("unknown CDS pricing model: " << model);
406  }
407 
408  setupArguments(engine->getArguments());
409  const auto* results = dynamic_cast<const CreditDefaultSwap::results*>(engine->getResults());
410 
411  ObjectiveFunction f(0., *flatRate, *engine, results);
412  Rate guess = runningSpread_ / (1 - conventionalRecovery) * 365./360.;
413  Real step = guess * 0.1;
414 
415  Brent().solve(f, 1e-9, guess, step);
416  return results->fairSpread;
417  }

475  Date cdsMaturity(const Date& tradeDate, const Period& tenor, DateGeneration::Rule rule) {
476 
477  QL_REQUIRE(rule == DateGeneration::CDS2015 || rule == DateGeneration::CDS || rule == DateGeneration::OldCDS,
478  "cdsMaturity should only be used with date generation rule CDS2015, CDS or OldCDS");
479 
480  QL_REQUIRE(tenor.units() == Years || (tenor.units() == Months && tenor.length() % 3 == 0),
481  "cdsMaturity expects a tenor that is a multiple of 3 months.");
482 
483  if (rule == DateGeneration::OldCDS) {
484  QL_REQUIRE(tenor != 0 * Months, "A tenor of 0M is not supported for OldCDS.");
485  }
486 
487  Date anchorDate = previousTwentieth(tradeDate, rule);
488  if (rule == DateGeneration::CDS2015 && (anchorDate == Date(20, Dec, anchorDate.year()) ||
489  anchorDate == Date(20, Jun, anchorDate.year()))) {
490  if (tenor.length() == 0) {
491  return Null<Date>();
492  } else {
493  anchorDate -= 3 * Months;
494  }
495  }
496 
497  Date maturity = anchorDate + tenor + 3 * Months;
498  QL_REQUIRE(maturity > tradeDate, "error calculating CDS maturity. Tenor is " << tenor << ", trade date is " <<
499  io::iso_date(tradeDate) << " generating a maturity of " << io::iso_date(maturity) << " <= trade date.");
500 
501  return maturity;
502  }
