37  CreditDefaultSwap::CreditDefaultSwap(Protection::Side side,
38  Real notional,
39  Rate spread,
40  const Schedule& schedule,
41  BusinessDayConvention convention,
42  const DayCounter& dayCounter,
43  bool settlesAccrual,
44  bool paysAtDefaultTime,
45  const Date& protectionStart,
46  const boost::shared_ptr<Claim>& claim,
47  const DayCounter& lastPeriodDayCounter,
48  const bool rebatesAccrual)
49  : side_(side), notional_(notional), upfront_(boost::none),
50  runningSpread_(spread), settlesAccrual_(settlesAccrual),
51  paysAtDefaultTime_(paysAtDefaultTime), claim_(claim),
52  protectionStart_(protectionStart == Null<Date>() ? schedule[0] :
53  protectionStart) {
54  QL_REQUIRE((protectionStart_ <= schedule[0]) ||
55  (schedule.rule() == DateGeneration::CDS) ||
56  (schedule.rule() == DateGeneration::CDS2015),
57  "protection can not start after accrual");
58 
59  leg_ = FixedRateLeg(schedule)
60  .withNotionals(notional)
61  .withCouponRates(spread, dayCounter)
62  .withPaymentAdjustment(convention)
63  .withLastPeriodDayCounter(lastPeriodDayCounter);
64 
65  Date effectiveUpfrontDate = schedule.calendar().advance(
66  protectionStart_ - 1, 3, Days, convention);
67  // the upfront valuation date is trade_date + 3 (using calendar) and
68  // protection start is assumed to be T+1 (independent of the calendar)
69 
70  if(rebatesAccrual) {
71  Size i = 0;
72  while (leg_[i]->hasOccurred(protectionStart_, false)) ++i;
73  boost::shared_ptr<FixedRateCoupon> coupon =
74  boost::dynamic_pointer_cast<FixedRateCoupon>(leg_[i]);
75  QL_REQUIRE(coupon->accrualStartDate() <= protectionStart_,
76  "contract cannot start before accrual");
77  const Date& rebateDate = effectiveUpfrontDate;
78  accrualRebate_ = boost::make_shared<SimpleCashFlow>(
79  coupon->accruedAmount(protectionStart_),
80  rebateDate);
81  }
82 
83  upfrontPayment_ = boost::make_shared<SimpleCashFlow>(0.0, effectiveUpfrontDate);
84  if (!claim_)
85  claim_ = boost::make_shared<FaceValueClaim>();
86  registerWith(claim_);
87 
88 
89  maturity_ = schedule.dates().back();
90  }

92  CreditDefaultSwap::CreditDefaultSwap(Protection::Side side,
93  Real notional,
94  Rate upfront,
95  Rate runningSpread,
96  const Schedule& schedule,
97  BusinessDayConvention convention,
98  const DayCounter& dayCounter,
99  bool settlesAccrual,
100  bool paysAtDefaultTime,
101  const Date& protectionStart,
102  const Date& upfrontDate,
103  const boost::shared_ptr<Claim>& claim,
104  const DayCounter& lastPeriodDayCounter,
105  const bool rebatesAccrual)
106  : side_(side), notional_(notional), upfront_(upfront),
107  runningSpread_(runningSpread), settlesAccrual_(settlesAccrual),
108  paysAtDefaultTime_(paysAtDefaultTime), claim_(claim),
109  protectionStart_(protectionStart == Null<Date>() ? schedule[0] :
110  protectionStart) {
111  QL_REQUIRE((protectionStart_ <= schedule[0])
112  || (schedule.rule() == DateGeneration::CDS)
113  || (schedule.rule() == DateGeneration::CDS2015),
114  "protection can not start after accrual");
115 
116  leg_ = FixedRateLeg(schedule)
117  .withNotionals(notional)
118  .withCouponRates(runningSpread, dayCounter)
119  .withPaymentAdjustment(convention)
120  .withLastPeriodDayCounter(lastPeriodDayCounter);
121 
122  // If empty, adjust to T+3 standard settlement, alternatively add
123  // an arbitrary date to the constructor
124  Date effectiveUpfrontDate = upfrontDate == Null<Date>() ?
125  schedule.calendar().advance(protectionStart_ -1 , 3, Days, convention) :
126  upfrontDate;
127  // protection start is assumed to be on trade_date + 1 (no calendar adjustment)
128  upfrontPayment_ = boost::make_shared<SimpleCashFlow>(notional*upfront,
129  effectiveUpfrontDate);
130 
131  QL_REQUIRE(effectiveUpfrontDate >= protectionStart_,
132  "upfront can not be due before contract start");
133 
134  if(rebatesAccrual) {
135  Size i = 0;
136  while (leg_[i]->hasOccurred(protectionStart_, false)) ++i;
137  boost::shared_ptr<FixedRateCoupon> coupon =
138  boost::dynamic_pointer_cast<FixedRateCoupon>(leg_[i]);
139  QL_REQUIRE(coupon->accrualStartDate() <= protectionStart_,
140  "contract cannot start before accrual");
141  const Date& rebateDate = effectiveUpfrontDate;
142  accrualRebate_ = boost::make_shared<SimpleCashFlow>(
143  coupon->accruedAmount(protectionStart_),
144  rebateDate);
145  }
146 
147  if (!claim_)
148  claim_ = boost::make_shared<FaceValueClaim>();
149  registerWith(claim_);
150 
151  maturity_ = schedule.dates().back();
152  }

319  Rate CreditDefaultSwap::impliedHazardRate(
320  Real targetNPV,
321  const Handle<YieldTermStructure>& discountCurve,
322  const DayCounter& dayCounter,
323  Real recoveryRate,
324  Real accuracy,
325  PricingModel model) const {
326 
327  boost::shared_ptr<SimpleQuote> flatRate = boost::make_shared<SimpleQuote>(0.0);
328 
329  Handle<DefaultProbabilityTermStructure> probability =
330  Handle<DefaultProbabilityTermStructure>(
331  boost::make_shared<FlatHazardRate>(0, WeekendsOnly(),
332  Handle<Quote>(flatRate), dayCounter));
333 
334  boost::shared_ptr<PricingEngine> engine;
335  switch (model) {
336  case Midpoint:
337  engine = boost::make_shared<MidPointCdsEngine>(
338  probability, recoveryRate, discountCurve);
339  break;
340  case ISDA:
341  engine = boost::make_shared<IsdaCdsEngine>(
342  probability, recoveryRate, discountCurve,
343  boost::none,
344  IsdaCdsEngine::Taylor,
345  IsdaCdsEngine::HalfDayBias,
346  IsdaCdsEngine::Piecewise);
347  break;
348  default:
349  QL_FAIL("unknown CDS pricing model: " << model);
350  }
351 
352  setupArguments(engine->getArguments());
353  const CreditDefaultSwap::results* results =
354  dynamic_cast<const CreditDefaultSwap::results*>(
355  engine->getResults());
356 
357  ObjectiveFunction f(targetNPV, *flatRate, *engine, results);
358  //very close guess if targetNPV = 0.
359  Rate guess = runningSpread_ / (1 - recoveryRate) * 365./360.;
360  Real step = 0.1 * guess;
361  return Brent().solve(f, accuracy, guess, step);
362  }

364  Rate CreditDefaultSwap::conventionalSpread(
365  Real conventionalRecovery,
366  const Handle<YieldTermStructure>& discountCurve,
367  const DayCounter& dayCounter,
368  PricingModel model) const {
369 
370  boost::shared_ptr<SimpleQuote> flatRate = boost::make_shared<SimpleQuote>(0.0);
371 
372  Handle<DefaultProbabilityTermStructure> probability =
373  Handle<DefaultProbabilityTermStructure>(
374  boost::make_shared<FlatHazardRate>(0, WeekendsOnly(),
375  Handle<Quote>(flatRate), dayCounter));
376 
377  boost::shared_ptr<PricingEngine> engine;
378  switch (model) {
379  case Midpoint:
380  engine = boost::make_shared<MidPointCdsEngine>(
381  probability, conventionalRecovery, discountCurve);
382  break;
383  case ISDA:
384  engine = boost::make_shared<IsdaCdsEngine>(
385  probability, conventionalRecovery, discountCurve,
386  boost::none,
387  IsdaCdsEngine::Taylor,
388  IsdaCdsEngine::HalfDayBias,
389  IsdaCdsEngine::Piecewise);
390  break;
391  default:
392  QL_FAIL("unknown CDS pricing model: " << model);
393  }
394 
395  setupArguments(engine->getArguments());
396  const CreditDefaultSwap::results* results =
397  dynamic_cast<const CreditDefaultSwap::results*>(
398  engine->getResults());
399 
400  ObjectiveFunction f(0., *flatRate, *engine, results);
401  Rate guess = runningSpread_ / (1 - conventionalRecovery) * 365./360.;
402  Real step = guess * 0.1;
403 
404  Brent().solve(f, 1e-9, guess, step);
405  return results->fairSpread;
406  }
