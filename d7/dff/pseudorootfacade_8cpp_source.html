35  PseudoRootFacade::PseudoRootFacade(
36  const std::vector<Matrix>& covariancePseudoRoots,
37  const std::vector<Rate>& rateTimes,
38  const std::vector<Rate>& initialRates,
39  const std::vector<Spread>& displacements)
40  : numberOfFactors_(covariancePseudoRoots.front().columns()),
41  numberOfRates_(covariancePseudoRoots.front().rows()),
42  numberOfSteps_(covariancePseudoRoots.size()),
43  initialRates_(initialRates),
44  displacements_(displacements),
45  evolution_(rateTimes),
46  covariancePseudoRoots_(covariancePseudoRoots)
47  {
48  checkIncreasingTimes(rateTimes);
49  QL_REQUIRE(rateTimes.size()>1,
50  "Rate times must contain at least two values");
51  QL_REQUIRE(numberOfRates_==rateTimes.size()-1,
52  "mismatch between number of rates (" << numberOfRates_ <<
53  ") and rate times");
54  QL_REQUIRE(numberOfRates_==displacements.size(),
55  "mismatch between number of rates (" << numberOfRates_ <<
56  ") and displacements (" << displacements.size() << ")");
57  QL_REQUIRE(numberOfRates_<=numberOfFactors_*numberOfSteps_,
58  "number of rates (" << numberOfRates_ <<
59  ") greater than number of factors (" << numberOfFactors_
60  << ") times number of steps (" << numberOfSteps_ << ")");
61 
62  // evolutionTimes are not given for the time being
63  QL_REQUIRE(numberOfRates_==covariancePseudoRoots.size(),
64  "number of rates (" << numberOfRates_ <<
65  ") must be equal to covariancePseudoRoots.size() (" <<
66  covariancePseudoRoots.size() << ")");
67 
68  for (Size k=0; k<numberOfSteps_; ++k) {
69  QL_REQUIRE(covariancePseudoRoots_[k].rows()==numberOfRates_,
70  "step " << k
71  << ": pseudoRoot has wrong number of rows: "
72  << covariancePseudoRoots_[k].rows()
73  << " instead of " << numberOfRates_);
74  // not really mandatory
75  QL_REQUIRE(covariancePseudoRoots_[k].columns()==numberOfFactors_,
76  "step " << k
77  << ": pseudoRoot has wrong number of columns: "
78  << covariancePseudoRoots_[k].columns()
79  << " instead of " << numberOfFactors_);
80  }
81  }
