124  void
125  ObjectiveFunction::updateVolatilityCubeAndCmsMarket(const Array &x) const {
126  const std::vector<Period> &swapTenors = cmsMarket_->swapTenors();
127  Size nSwapTenors = swapTenors.size();
128  QL_REQUIRE(nSwapTenors + 1 == x.size(),
129  "bad calibration guess nSwapTenors+1 != x.size()");
130  const boost::shared_ptr<SwaptionVolCube1> volCubeBySabr =
131  boost::dynamic_pointer_cast<SwaptionVolCube1>(*volCube_);
132  for (Size i = 0; i < nSwapTenors; ++i)
133  volCubeBySabr->recalibration(
134  smileAndCms_->betaTransformDirect(x[i]), swapTenors[i]);
135  Real meanReversion =
136  smileAndCms_->reversionTransformDirect(x[nSwapTenors]);
137  cmsMarket_->reprice(volCube_, meanReversion);
138  }

193  void
194  ObjectiveFunction3::updateVolatilityCubeAndCmsMarket(const Array &x) const {
195  const std::vector<Period> &swapTenors = cmsMarket_->swapTenors();
196  const std::vector<Period> &swapLengths = cmsMarket_->swapLengths();
197  Size nSwapTenors = swapTenors.size();
198  Size nSwapLengths = swapLengths.size();
199  QL_REQUIRE(
200  (nSwapLengths * nSwapTenors) + 1 == x.size(),
201  "bad calibration guess (nSwapLengths*nSwapTenors)+1 != x.size()");
202  const boost::shared_ptr<SwaptionVolCube1> volCubeBySabr =
203  boost::dynamic_pointer_cast<SwaptionVolCube1>(*volCube_);
204  for (Size i = 0; i < nSwapTenors; ++i) {
205  std::vector<Real> beta(x.begin() + (i * nSwapLengths),
206  x.begin() + ((i + 1) * nSwapLengths));
207  for (Size j = 0; j < beta.size(); ++j)
208  beta[j] = smileAndCms_->betaTransformDirect(beta[j]);
209  volCubeBySabr->recalibration(swapLengths, beta, swapTenors[i]);
210  }
211  Real meanReversion = smileAndCms_->reversionTransformDirect(
212  x[nSwapLengths + nSwapTenors]);
213  cmsMarket_->reprice(volCube_, meanReversion);
214  }

249  void
250  ObjectiveFunction5::updateVolatilityCubeAndCmsMarket(const Array &x) const {
251  const std::vector<Period> &swapTenors = cmsMarket_->swapTenors();
252  const std::vector<Period> &swapLengths = cmsMarket_->swapLengths();
253  Size nSwapTenors = swapTenors.size();
254  Size nSwapLengths = swapLengths.size();
255  QL_REQUIRE((3 * nSwapTenors) == x.size(),
256  "bad calibration guess (3*nSwapTenors) != x.size()");
257  const boost::shared_ptr<SwaptionVolCube1> volCubeBySabr =
258  boost::dynamic_pointer_cast<SwaptionVolCube1>(*volCube_);
259  for (Size i = 0; i < nSwapTenors; ++i) {
260  Real betaInf = smileAndCms_->betaTransformDirect(x[0 + 3 * i]);
261  Real beta0 = smileAndCms_->betaTransformDirect(x[1 + 3 * i]);
262  Real decay = x[2 + 3 * i] * x[2 + 3 * i];
263  std::vector<Real> beta(nSwapLengths);
264  for (Size j = 0; j < beta.size(); ++j) {
265  Real t = smileAndCms_->volCube_->timeFromReference(
266  smileAndCms_->volCube_->optionDateFromTenor(swapLengths[j]));
267  beta[j] = betaInf + (beta0 - betaInf) * std::exp(-decay * t);
268  }
269  volCubeBySabr->recalibration(swapLengths, beta, swapTenors[i]);
270  }
271  cmsMarket_->reprice(
272  volCube_,
273  fixedMeanReversion_ == Null<Real>()
274  ? Null<Real>()
275  : smileAndCms_->reversionTransformDirect(fixedMeanReversion_));
276  }

335  Array CmsMarketCalibration::compute(
336  const boost::shared_ptr<EndCriteria> &endCriteria,
337  const boost::shared_ptr<OptimizationMethod> &method, const Array &guess,
338  bool isMeanReversionFixed) {
339  Size nSwapTenors = cmsMarket_->swapTenors().size();
340  QL_REQUIRE(isMeanReversionFixed || guess.size() == nSwapTenors + 1,
341  "if mean reversion is not fixed, a guess must be provided");
342  QL_REQUIRE(nSwapTenors == guess.size() ||
343  nSwapTenors == guess.size() - 1,
344  "guess size (" << guess.size()
345  << ") must be equal to swap tenors size ("
346  << nSwapTenors
347  << ") or greater by one if mean reversion is "
348  "given as last element");
349  bool isMeanReversionGiven = (nSwapTenors == guess.size() - 1);
350  Size nBeta = guess.size() - (isMeanReversionGiven ? 1 : 0);
351  Array result;
352  if (isMeanReversionFixed) {
353  NoConstraint constraint;
354  Real fixedMeanReversion =
355  isMeanReversionGiven ? guess[nBeta] : Null<Real>();
356  Array betasGuess(nBeta);
357  for (Size i = 0; i < nBeta; ++i)
358  betasGuess[i] = guess[i];
359  ObjectiveFunction2 costFunction(
360  this, fixedMeanReversion == Null<Real>()
361  ? Null<Real>()
362  : reversionTransformInverse(fixedMeanReversion));
363  Problem problem(costFunction, constraint, betasGuess);
364  endCriteria_ = method->minimize(problem, *endCriteria);
365  Array tmp = problem.currentValue();
366  error_ = costFunction.value(tmp);
367  result = Array(nBeta + (isMeanReversionGiven ? 1 : 0));
368  for (Size i = 0; i < nBeta; ++i)
369  result[i] = betaTransformDirect(tmp[i]);
370  if (isMeanReversionGiven)
371  result[nBeta] = fixedMeanReversion;
372  } else {
373  NoConstraint constraint;
374  ObjectiveFunction costFunction(this);
375  Array betaReversionGuess(nBeta + 1);
376  for (Size i = 0; i < nBeta; ++i)
377  betaReversionGuess[i] = betaTransformInverse(guess[i]);
378  betaReversionGuess[nBeta] = reversionTransformInverse(guess[nBeta]);
379  Problem problem(costFunction, constraint, betaReversionGuess);
380  endCriteria_ = method->minimize(problem, *endCriteria);
381  result = problem.currentValue();
382  error_ = costFunction.value(result);
383  for (Size i = 0; i < nBeta; ++i)
384  result[i] = betaTransformDirect(result[i]);
385  result[nBeta] = reversionTransformDirect(result[nBeta]);
386  }
387  const boost::shared_ptr<SwaptionVolCube1> volCubeBySabr =
388  boost::dynamic_pointer_cast<SwaptionVolCube1>(*volCube_);
389  volCubeBySabr->updateAfterRecalibration();
390  sparseSabrParameters_ = volCubeBySabr->sparseSabrParameters();
391  denseSabrParameters_ = volCubeBySabr->denseSabrParameters();
392  browseCmsMarket_ = cmsMarket_->browse();
393 
394  return result;
395  }

397  Matrix CmsMarketCalibration::compute(
398  const boost::shared_ptr<EndCriteria> &endCriteria,
399  const boost::shared_ptr<OptimizationMethod> &method,
400  const Matrix &guess, bool isMeanReversionFixed,
401  const Real meanReversionGuess) {
402  Size nSwapTenors = cmsMarket_->swapTenors().size();
403  Size nSwapLengths = cmsMarket_->swapLengths().size();
404  QL_REQUIRE(isMeanReversionFixed || meanReversionGuess != Null<Real>(),
405  "if mean reversion is not fixed, a guess must be provided");
406  QL_REQUIRE(nSwapTenors == guess.columns(),
407  "number of swap tenors ("
408  << nSwapTenors
409  << ") must be equal to number of guess columns ("
410  << guess.columns() << ")");
411  QL_REQUIRE(nSwapLengths == guess.rows(),
412  "number of swap lengths ("
413  << nSwapLengths
414  << ") must be equal to number of guess rows ("
415  << guess.rows() << ")");
416  Matrix result;
417  Size nBeta = nSwapTenors * nSwapLengths;
418  if (isMeanReversionFixed) {
419  NoConstraint constraint;
420  Array betasGuess(nBeta);
421  for (Size i = 0; i < nSwapTenors; ++i) {
422  for (Size j = 0; j < nSwapLengths; ++j) {
423  betasGuess[i * nSwapLengths + j] =
424  betaTransformInverse(guess[j][i]);
425  }
426  }
427  ObjectiveFunction4 costFunction(
428  this, meanReversionGuess == Null<Real>()
429  ? meanReversionGuess
430  : reversionTransformInverse(meanReversionGuess));
431  Problem problem(costFunction, constraint, betasGuess);
432  endCriteria_ = method->minimize(problem, *endCriteria);
433  Array tmp = problem.currentValue();
434  error_ = costFunction.value(tmp);
435  result = Matrix(nSwapLengths,
436  nSwapTenors +
437  (meanReversionGuess != Null<Real>() ? 1 : 0));
438  for (Size i = 0; i < nSwapTenors; ++i) {
439  for (Size j = 0; j < nSwapLengths; ++j) {
440  result[j][i] =
441  betaTransformDirect(tmp[i * nSwapLengths + j]);
442  }
443  }
444  if (meanReversionGuess != Null<Real>()) {
445  for (Size j = 0; j < nSwapLengths; ++j) {
446  result[j][nSwapTenors] = meanReversionGuess;
447  }
448  }
449  } else {
450  NoConstraint constraint;
451  Array betasReversionGuess(nBeta + 1);
452  for (Size i = 0; i < nSwapTenors; ++i) {
453  for (Size j = 0; j < nSwapLengths; ++j) {
454  betasReversionGuess[i * nSwapLengths + j] =
455  betaTransformInverse(guess[j][i]);
456  }
457  }
458  betasReversionGuess[nBeta] =
459  reversionTransformInverse(meanReversionGuess);
460  ObjectiveFunction3 costFunction(this);
461  Problem problem(costFunction, constraint, betasReversionGuess);
462  endCriteria_ = method->minimize(problem, *endCriteria);
463  Array tmp = problem.currentValue();
464  error_ = costFunction.value(tmp);
465  result = Matrix(nSwapLengths, nSwapTenors + 1);
466  for (Size i = 0; i < nSwapTenors; ++i) {
467  for (Size j = 0; j < nSwapLengths; ++j) {
468  result[j][i] =
469  betaTransformDirect(tmp[i * nSwapLengths + j]);
470  }
471  }
472  for (Size j = 0; j < nSwapLengths; ++j) {
473  result[j][nSwapTenors] = reversionTransformDirect(tmp[nBeta]);
474  }
475  }
476  const boost::shared_ptr<SwaptionVolCube1> volCubeBySabr =
477  boost::dynamic_pointer_cast<SwaptionVolCube1>(*volCube_);
478  volCubeBySabr->updateAfterRecalibration();
479  sparseSabrParameters_ = volCubeBySabr->sparseSabrParameters();
480  denseSabrParameters_ = volCubeBySabr->denseSabrParameters();
481  browseCmsMarket_ = cmsMarket_->browse();
482 
483  return result;

486  Matrix CmsMarketCalibration::computeParametric(
487  const boost::shared_ptr<EndCriteria> &endCriteria,
488  const boost::shared_ptr<OptimizationMethod> &method,
489  const Matrix &guess, bool isMeanReversionFixed,
490  const Real meanReversionGuess) {
491 
492  Size nSwapTenors = cmsMarket_->swapTenors().size();
493  Size nSwapLengths = cmsMarket_->swapLengths().size();
494  QL_REQUIRE(isMeanReversionFixed || meanReversionGuess != Null<Real>(),
495  "if mean reversion is not fixed, a guess must be provided");
496  QL_REQUIRE(nSwapTenors == guess.columns(),
497  "number of swap tenors ("
498  << nSwapTenors
499  << ") must be equal to number of guess columns ("
500  << guess.columns() << ")");
501  QL_REQUIRE(3 == guess.rows(),
502  "number of parameters ("
503  << 3 << ") must be equal to number of guess rows ("
504  << guess.rows() << ")");
505 
506  Matrix result;
507  Size nParams = nSwapTenors * 3;
508  if (isMeanReversionFixed) {
509  NoConstraint constraint;
510  Array betasGuess(nParams);
511  for (Size i = 0; i < nSwapTenors; ++i) {
512  for (Size j = 0; j < nParams; ++j) {
513  betasGuess[i * 3 + j] =
514  (j == 0 || j == 1) ? betaTransformInverse(guess[j][i])
515  : std::sqrt(guess[j][i]);
516  }
517  }
518  ObjectiveFunction5 costFunction(
519  this, meanReversionGuess == Null<Real>()
520  ? meanReversionGuess
521  : reversionTransformInverse(meanReversionGuess));
522  Problem problem(costFunction, constraint, betasGuess);
523  endCriteria_ = method->minimize(problem, *endCriteria);
524  Array tmp = problem.currentValue();
525  error_ = costFunction.value(tmp);
526  result = Matrix(
527  3, nSwapTenors + (meanReversionGuess != Null<Real>() ? 1 : 0));
528  for (Size i = 0; i < nSwapTenors; ++i) {
529  for (Size j = 0; j < 3; ++j) {
530  result[j][i] = (j == 0 || j == 1)
531  ? betaTransformDirect(tmp[i * 3 + j])
532  : tmp[i * 3 + j] * tmp[i * 3 + j];
533  }
534  }
535  if (meanReversionGuess != Null<Real>()) {
536  for (Size j = 0; j < nSwapLengths; ++j) {
537  result[j][nSwapTenors] = meanReversionGuess;
538  }
539  }
540  } else {
541  NoConstraint constraint;
542  Array betasReversionGuess(nParams + 1);
543  for (Size i = 0; i < nSwapTenors; ++i) {
544  for (Size j = 0; j < nParams; ++j) {
545  betasReversionGuess[i * nSwapLengths + j] =
546  (j == 0 || j == 1) ? betaTransformInverse(guess[j][i])
547  : std::sqrt(guess[j][i]);
548  }
549  }
550  betasReversionGuess[nParams] =
