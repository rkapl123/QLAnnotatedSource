32  Fdm2DimSolver::Fdm2DimSolver(
33  const FdmSolverDesc& solverDesc,
34  const FdmSchemeDesc& schemeDesc,
35  const boost::shared_ptr<FdmLinearOpComposite>& op)
36  : solverDesc_(solverDesc),
37  schemeDesc_(schemeDesc),
38  op_(op),
39  thetaCondition_(boost::make_shared<FdmSnapshotCondition>(
40  0.99*std::min(1.0/365.0,
41  solverDesc.condition->stoppingTimes().empty()
42  ? solverDesc.maturity
43  : solverDesc.condition->stoppingTimes().front()))),
44  conditions_(FdmStepConditionComposite::joinConditions(thetaCondition_,
45  solverDesc.condition)),
46  initialValues_(solverDesc.mesher->layout()->size()),
47  resultValues_ (solverDesc.mesher->layout()->dim()[1],
48  solverDesc.mesher->layout()->dim()[0]) {
49 
50  const boost::shared_ptr<FdmMesher> mesher = solverDesc.mesher;
51  const boost::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
52 
53  x_.reserve(layout->dim()[0]);
54  y_.reserve(layout->dim()[1]);
55 
56  const FdmLinearOpIterator endIter = layout->end();
57  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
58  ++iter) {
59  initialValues_[iter.index()]
60  = solverDesc_.calculator->avgInnerValue(iter,
61  solverDesc.maturity);
62 
63  if (!iter.coordinates()[1]) {
64  x_.push_back(mesher->location(iter, 0));
65  }
66  if (!iter.coordinates()[0]) {
67  y_.push_back(mesher->location(iter, 1));
68  }
69  }
70  }

73  void Fdm2DimSolver::performCalculations() const {
74  Array rhs(initialValues_.size());
75  std::copy(initialValues_.begin(), initialValues_.end(), rhs.begin());
76 
77  FdmBackwardSolver(op_, solverDesc_.bcSet, conditions_, schemeDesc_)
78  .rollback(rhs, solverDesc_.maturity, 0.0,
79  solverDesc_.timeSteps, solverDesc_.dampingSteps);
80 
81  std::copy(rhs.begin(), rhs.end(), resultValues_.begin());
82  interpolation_ = boost::make_shared<BicubicSpline>(x_.begin(), x_.end(),
83  y_.begin(), y_.end(),
84  resultValues_);
85  }

92  Real Fdm2DimSolver::thetaAt(Real x, Real y) const {
93  QL_REQUIRE(conditions_->stoppingTimes().front() > 0.0,
94  "stopping time at zero-> can't calculate theta");
95 
96  calculate();
97  Matrix thetaValues(resultValues_.rows(), resultValues_.columns());
98 
99  const Array& rhs = thetaCondition_->getValues();
100  std::copy(rhs.begin(), rhs.end(), thetaValues.begin());
101 
102  return (BicubicSpline(x_.begin(), x_.end(), y_.begin(), y_.end(),
103  thetaValues)(x, y) - interpolateAt(x, y))
104  / thetaCondition_->getTime();
105  }
