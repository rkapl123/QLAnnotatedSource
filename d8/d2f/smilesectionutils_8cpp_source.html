25 SmileSectionUtils::SmileSectionUtils(const SmileSection &section,
26 const std::vector<Real> &moneynessGrid,
27 const Real atm,
28 const bool deleteArbitragePoints) {
29
30 if (!moneynessGrid.empty()) {
31 QL_REQUIRE(
32 section.volatilityType() == Normal || moneynessGrid[0] >= 0.0,
33 "moneyness grid should only contain non negative values ("
34 << moneynessGrid[0] << ")");
35 for (Size i = 0; i < moneynessGrid.size() - 1; i++) {
36 QL_REQUIRE(moneynessGrid[i] < moneynessGrid[i + 1],
37 "moneyness grid should contain strictly increasing "
38 "values ("
39 << moneynessGrid[i] << ","
40 << moneynessGrid[i + 1] << " at indices " << i
41 << ", " << i + 1 << ")");
42 }
43 }
44
45 if (atm == Null<Real>()) {
46 f_ = section.atmLevel();
47 QL_REQUIRE(f_ != Null<Real>(),
48 "atm level must be provided by source section or given "
49 "in the constructor");
50 } else {
51 f_ = atm;
52 }
53
54 std::vector<Real> tmp;
55
56 static const Real defaultMoney[] = { 0.0, 0.01, 0.05, 0.10, 0.25, 0.40,
57 0.50, 0.60, 0.70, 0.80, 0.90, 1.0,
58 1.25, 1.5, 1.75, 2.0, 5.0, 7.5,
59 10.0, 15.0, 20.0 };
60 static const Real defaultMoneyNormal[] = {
61 -0.20, -0.15, -0.10, -0.075, -0.05, -0.04, -0.03,
62 -0.02, -0.015, -0.01, -0.0075, -0.0050, -0.0025, 0.0,
63 0.0025, 0.0050, 0.0075, 0.01, 0.015, 0.02, 0.03,
64 0.04, 0.05, 0.075, 0.10, 0.15, 0.20
65 };
66
67 if (moneynessGrid.empty()) {
68 tmp = section.volatilityType() == Normal
69 ? std::vector<Real>(defaultMoneyNormal,
70 defaultMoneyNormal + 27)
71 : std::vector<Real>(defaultMoney, defaultMoney + 21);
72 }
73 else
74 tmp = std::vector<Real>(moneynessGrid);
75
76 Real shift = section.shift();
77
78 if (section.volatilityType() == ShiftedLognormal && tmp[0] > QL_EPSILON) {
79 m_.push_back(0.0);
80 k_.push_back(-shift);
81 }
82
83 bool minStrikeAdded = false, maxStrikeAdded = false;
84 for (Real& i : tmp) {
85 Real k = section.volatilityType() == Normal ? Real(f_ + i) : Real(i * (f_ + shift) - shift);
86 if ((section.volatilityType() == ShiftedLognormal && i <= QL_EPSILON) ||
87 (k >= section.minStrike() && k <= section.maxStrike())) {
88 if (!minStrikeAdded || !close(k, section.minStrike())) {
89 m_.push_back(i);
90 k_.push_back(k);
91 }
92 if (close(k, section.maxStrike()))
93 maxStrikeAdded = true;
94 } else { // if the section provides a limited strike range
95 // we put the respective endpoint in our grid
96 // in order to not loose too much information
97 if (k < section.minStrike() && !minStrikeAdded) {
98 m_.push_back(section.volatilityType() == Normal
99 ? Real(section.minStrike() - f_)
100 : Real((section.minStrike() + shift) / f_));
101 k_.push_back(section.minStrike());
102 minStrikeAdded = true;
103 }
104 if (k > section.maxStrike() && !maxStrikeAdded) {
105 m_.push_back(section.volatilityType() == Normal
106 ? Real(section.maxStrike() - f_)
107 : Real((section.maxStrike() + shift) / f_));
108 k_.push_back(section.maxStrike());
109 maxStrikeAdded = true;
110 }
111 }
112 }
113
114 // only known for shifted lognormal vols, otherwise we include
115 // the lower strike in the loop below
116 if(section.volatilityType() == ShiftedLognormal)
117 c_.push_back(f_ + shift);
118
119 for (Size i = (section.volatilityType() == Normal ? 0 : 1);
120 i < k_.size(); i++) {
121 c_.push_back(section.optionPrice(k_[i], Option::Call, 1.0));
122 }
123
124 Size centralIndex =
125 std::upper_bound(m_.begin(), m_.end(),
126 (section.volatilityType() == Normal ? 0.0 : 1.0) -
127 QL_EPSILON) -
128 m_.begin();
129 QL_REQUIRE(centralIndex < k_.size() - 1 && centralIndex > 1,
130 "Atm point in moneyness grid ("
131 << centralIndex << ") too close to boundary.");
132
133 // shift central index to the right if necessary
134 // (sometimes even the atm point lies in an arbitrageable area)
135
136 while (!af(centralIndex, centralIndex, centralIndex + 1) &&
137 centralIndex < k_.size() - 1)
138 centralIndex++;
139
140 QL_REQUIRE(centralIndex < k_.size(),
141 "central index is at right boundary");
142
143 leftIndex_ = centralIndex;
144 rightIndex_ = centralIndex;
145
146 bool done = false;
147 while (!done) {
148
149 bool isAf = true;
150 done = true;
151
152 while (isAf && rightIndex_ < k_.size() - 1) {
153 rightIndex_++;
154 isAf = af(leftIndex_, rightIndex_, rightIndex_) &&
155 af(leftIndex_, rightIndex_ - 1, rightIndex_);
156 }
157 if (!isAf)
158 rightIndex_--;
159
160 isAf = true;
161 while (isAf && leftIndex_ > 1) {
162 leftIndex_--;
163 isAf = af(leftIndex_, leftIndex_, rightIndex_) &&
164 af(leftIndex_, leftIndex_ + 1, rightIndex_);
165 }
166 if (!isAf)
167 leftIndex_++;
168
169 if (rightIndex_ < leftIndex_)
170 rightIndex_ = leftIndex_;
171
172 if (deleteArbitragePoints && leftIndex_ > 1) {
173 m_.erase(m_.begin() + leftIndex_ - 1);
174 k_.erase(k_.begin() + leftIndex_ - 1);
175 c_.erase(c_.begin() + leftIndex_ - 1);
176 leftIndex_--;
177 rightIndex_--;
178 done = false;
179 }
180 if (deleteArbitragePoints && rightIndex_ < k_.size() - 1) {
181 m_.erase(m_.begin() + rightIndex_ + 1);
182 k_.erase(k_.begin() + rightIndex_ + 1);
183 c_.erase(c_.begin() + rightIndex_ + 1);
184 rightIndex_--;
185 done = false;
186 }
187 }
188
189 QL_REQUIRE(rightIndex_ > leftIndex_,
190 "arbitrage free region must at least contain two "
191 "points (only index is "
192 << leftIndex_ << ")");
193
194 }

196 std::pair<Real, Real> SmileSectionUtils::arbitragefreeRegion() const {
197 return {k_[leftIndex_], k_[rightIndex_]};
198 }

200 std::pair<Size, Size> SmileSectionUtils::arbitragefreeIndices() const {
201 return {leftIndex_, rightIndex_};

204 bool SmileSectionUtils::af(const Size i0, const Size i,
205 const Size i1) const {
206 if (i == 0)
207 return true;
208 Size im = i - 1 >= i0 ? i - 1 : 0;
209 Real q1 = (c_[i] - c_[im]) / (k_[i] - k_[im]);
210 if (q1 < -1.0 || q1 > 0.0)
211 return false;
212 if (i >= i1)
213 return true;
214 Real q2 = (c_[i + 1] - c_[i]) / (k_[i + 1] - k_[i]);
215 return q1 <= q2 && q2 <= 0.0;
216 }
