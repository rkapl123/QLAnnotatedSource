86  template <class PathType> inline
87  LongstaffSchwartzPathPricer<PathType>::LongstaffSchwartzPathPricer(
88  const TimeGrid& times,
89  const ext::shared_ptr<EarlyExercisePathPricer<PathType> >&
90  pathPricer,
91  const ext::shared_ptr<YieldTermStructure>& termStructure)
92  : calibrationPhase_(true),
93  pathPricer_(pathPricer),
94  coeff_ (new Array[times.size()-2]),
95  dF_ (new DiscountFactor[times.size()-1]),
96  v_ (pathPricer_->basisSystem()),
97  len_ (times.size()) {
98 
99  for (Size i=0; i<times.size()-1; ++i) {
100  dF_[i] = termStructure->discount(times[i+1])
101  / termStructure->discount(times[i]);
102  }
103  }

105  template <class PathType> inline
106  Real LongstaffSchwartzPathPricer<PathType>::operator()
107  (const PathType& path) const {
108  if (calibrationPhase_) {
109  // store paths for the calibration
110  paths_.push_back(path);
111  // result doesn't matter
112  return 0.0;
113  }
114 
115  Real price = (*pathPricer_)(path, len_-1);
116 
117  // Initialize with exercise on last date
118  bool exercised = (price > 0.0);
119 
120  for (Size i=len_-2; i>0; --i) {
121  price*=dF_[i];
122 
123  const Real exercise = (*pathPricer_)(path, i);
124  if (exercise > 0.0) {
125  const StateType regValue = pathPricer_->state(path, i);
126 
127  Real continuationValue = 0.0;
128  for (Size l=0; l<v_.size(); ++l) {
129  continuationValue += coeff_[i-1][l] * v_[l](regValue);
130  }
131 
132  if (continuationValue < exercise) {
133  price = exercise;
134 
135  // Exercised
136  exercised = true;
137  }
138  }
139  }
140 
141  exerciseProbability_.add(exercised ? 1.0 : 0.0);
142 
143  return price*dF_[0];
144  }

146  template <class PathType> inline
147  void LongstaffSchwartzPathPricer<PathType>::calibrate() {
148  const Size n = paths_.size();
149  Array prices(n), exercise(n);
150  std::vector<StateType> p_state(n);
151  std::vector<Real> p_price(n), p_exercise(n);
152 
153  for (Size i=0; i<n; ++i) {
154  p_state[i] = pathPricer_->state(paths_[i],len_-1);
155  prices[i] = p_price[i] = (*pathPricer_)(paths_[i], len_-1);
156  p_exercise[i] = prices[i];
157  }
158 
159  post_processing(len_ - 1, p_state, p_price, p_exercise);
160 
161  std::vector<Real> y;
162  std::vector<StateType> x;
163  for (Size i=len_-2; i>0; --i) {
164  y.clear();
165  x.clear();
166 
167  //roll back step
168  for (Size j=0; j<n; ++j) {
169  exercise[j]=(*pathPricer_)(paths_[j], i);
170  if (exercise[j]>0.0) {
171  x.push_back(pathPricer_->state(paths_[j], i));
172  y.push_back(dF_[i]*prices[j]);
173  }
174  }
175 
176  if (v_.size() <= x.size()) {
177  coeff_[i-1] = GeneralLinearLeastSquares(x, y, v_).coefficients();
178  }
179  else {
180  // if number of itm paths is smaller then the number of
181  // calibration functions then early exercise if exerciseValue > 0
182  coeff_[i-1] = Array(v_.size(), 0.0);
183  }
184 
185  for (Size j=0, k=0; j<n; ++j) {
186  prices[j]*=dF_[i];
187  if (exercise[j]>0.0) {
188  Real continuationValue = 0.0;
189  for (Size l=0; l<v_.size(); ++l) {
190  continuationValue += coeff_[i-1][l] * v_[l](x[k]);
191  }
192  if (continuationValue < exercise[j]) {
193  prices[j] = exercise[j];
194  }
195  ++k;
196  }
197  p_state[j] = pathPricer_->state(paths_[j],i);
198  p_price[j] = prices[j];
199  p_exercise[j] = exercise[j];
200  }
201 
202  post_processing(i, p_state, p_price, p_exercise);
203  }
204 
205  // remove calibration paths and release memory
206  std::vector<PathType> empty;
207  paths_.swap(empty);
208  // entering the calculation phase
209  calibrationPhase_ = false;
210  }
