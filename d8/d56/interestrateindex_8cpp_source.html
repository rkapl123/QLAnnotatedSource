61  Rate InterestRateIndex::fixing(const Date& fixingDate,
62  bool forecastTodaysFixing) const {
63 
64  QL_REQUIRE(isValidFixingDate(fixingDate),
65  "Fixing date " << fixingDate << " is not valid");
66 
67  Date today = Settings::instance().evaluationDate();
68 
69  if (fixingDate>today ||
70  (fixingDate==today && forecastTodaysFixing))
71  return forecastFixing(fixingDate);
72 
73  if (fixingDate<today ||
74  Settings::instance().enforcesTodaysHistoricFixings()) {
75  // must have been fixed
76  // do not catch exceptions
77  Rate result = pastFixing(fixingDate);
78  QL_REQUIRE(result != Null<Real>(),
79  "Missing " << name() << " fixing for " << fixingDate);
80  return result;
81  }
82 
83  try {
84  // might have been fixed
85  Rate result = pastFixing(fixingDate);
86  if (result!=Null<Real>())
87  return result;
88  else
89  ; // fall through and forecast
90  } catch (Error&) {
91  ; // fall through and forecast
92  }
93  return forecastFixing(fixingDate);
94  }
