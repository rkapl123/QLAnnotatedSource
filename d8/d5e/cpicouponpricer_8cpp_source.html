83  Real CPICouponPricer::optionletRate(Option::Type optionType,
84  Real effStrike) const {
85  Date fixingDate = coupon_->fixingDate();
86  if (fixingDate <= Settings::instance().evaluationDate()) {
87  // the amount is determined
88  Real a, b;
89  if (optionType==Option::Call) {
90  a = coupon_->indexFixing();
91  b = effStrike;
92  } else {
93  a = effStrike;
94  b = coupon_->indexFixing();
95  }
96  return std::max(a - b, 0.0);
97  } else {
98  // not yet determined, use Black/DD1/Bachelier/whatever from Impl
99  QL_REQUIRE(!capletVolatility().empty(),
100  "missing optionlet volatility");
101  Real stdDev =
102  std::sqrt(capletVolatility()->totalVariance(fixingDate,
103  effStrike));
104  return optionletPriceImp(optionType,
105  effStrike,
106  adjustedFixing(),
107  stdDev);
108  }
109  }

122  void CPICouponPricer::initialize(const InflationCoupon& coupon) {
123  coupon_ = dynamic_cast<const CPICoupon*>(&coupon);
124  gearing_ = coupon_->fixedRate();
125  spread_ = coupon_->spread();
126  paymentDate_ = coupon_->date();
127  rateCurve_ =
128  !nominalTermStructure_.empty() ?
129  nominalTermStructure_ :
130  ext::dynamic_pointer_cast<ZeroInflationIndex>(coupon.index())
131  ->zeroInflationTermStructure()
132  ->nominalTermStructure();
133 
134  // past or future fixing is managed in YoYInflationIndex::fixing()
135  // use yield curve from index (which sets discount)
136 
137  discount_ = 1.0;
138  if (paymentDate_ > rateCurve_->referenceDate()) {
139  if (rateCurve_.empty()) {
140  // allow to extract rates, but mark the discount as invalid for prices
141  discount_ = Null<Real>();
142  } else {
143  discount_ = rateCurve_->discount(paymentDate_);
144  }
145  }
146  }

155  Rate CPICouponPricer::swapletRate() const {
156  // This way we do not require the index to have
157  // a yield curve, i.e. we do not get the problem
158  // that a discounting-instrument-pricer is used
159  // with a different yield curve
160  return gearing_ * adjustedFixing() + spread_;
