69  Real CPICouponPricer::optionletPrice(Option::Type optionType,
70  Real effStrike) const {
71  Date fixingDate = coupon_->fixingDate();
72  if (fixingDate <= Settings::instance().evaluationDate()) {
73  // the amount is determined
74  Real a, b;
75  if (optionType==Option::Call) {
76  a = coupon_->indexFixing();
77  b = effStrike;
78  } else {
79  a = effStrike;
80  b = coupon_->indexFixing();
81  }
82  return std::max(a - b, 0.0)* coupon_->accrualPeriod()*discount_;
83  } else {
84  // not yet determined, use Black/DD1/Bachelier/whatever from Impl
85  QL_REQUIRE(!capletVolatility().empty(),
86  "missing optionlet volatility");
87  Real stdDev =
88  std::sqrt(capletVolatility()->totalVariance(fixingDate,
89  effStrike));
90  Rate fixing = optionletPriceImp(optionType,
91  effStrike,
92  adjustedFixing(),
93  stdDev);
94  return fixing * coupon_->accrualPeriod() * discount_;
95  }
96  }

109  void CPICouponPricer::initialize(const InflationCoupon& coupon) {
110  coupon_ = dynamic_cast<const CPICoupon*>(&coupon);
111  gearing_ = coupon_->fixedRate();
112  spread_ = coupon_->spread();
113  paymentDate_ = coupon_->date();
114  rateCurve_ = boost::dynamic_pointer_cast<ZeroInflationIndex>(coupon.index())
115  ->zeroInflationTermStructure()
116  ->nominalTermStructure();
117 
118  // past or future fixing is managed in YoYInflationIndex::fixing()
119  // use yield curve from index (which sets discount)
120 
121  discount_ = 1.0;
122  if (paymentDate_ > rateCurve_->referenceDate())
123  discount_ = rateCurve_->discount(paymentDate_);
124 
125  spreadLegValue_ = spread_ * coupon_->accrualPeriod()* discount_;
126 
127  }
