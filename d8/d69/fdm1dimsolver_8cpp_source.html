32  Fdm1DimSolver::Fdm1DimSolver(
33  const FdmSolverDesc& solverDesc,
34  const FdmSchemeDesc& schemeDesc,
35  const boost::shared_ptr<FdmLinearOpComposite>& op)
36  : solverDesc_(solverDesc),
37  schemeDesc_(schemeDesc),
38  op_(op),
39  thetaCondition_(boost::make_shared<FdmSnapshotCondition>(
40  0.99*std::min(1.0/365.0,
41  solverDesc.condition->stoppingTimes().empty()
42  ? solverDesc.maturity
43  : solverDesc.condition->stoppingTimes().front()))),
44  conditions_(FdmStepConditionComposite::joinConditions(thetaCondition_,
45  solverDesc.condition)),
46  x_ (solverDesc.mesher->layout()->size()),
47  initialValues_(solverDesc.mesher->layout()->size()),
48  resultValues_ (solverDesc.mesher->layout()->size()) {
49 
50  const boost::shared_ptr<FdmMesher> mesher = solverDesc.mesher;
51  const boost::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
52 
53  const FdmLinearOpIterator endIter = layout->end();
54  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
55  ++iter) {
56  initialValues_[iter.index()]
57  = solverDesc_.calculator->avgInnerValue(iter,
58  solverDesc.maturity);
59  x_[iter.index()] = mesher->location(iter, 0);
60  }
61  }

64  void Fdm1DimSolver::performCalculations() const {
65  Array rhs(initialValues_.size());
66  std::copy(initialValues_.begin(), initialValues_.end(), rhs.begin());
67 
68  FdmBackwardSolver(op_, solverDesc_.bcSet, conditions_, schemeDesc_)
69  .rollback(rhs, solverDesc_.maturity, 0.0,
70  solverDesc_.timeSteps, solverDesc_.dampingSteps);
71 
72  std::copy(rhs.begin(), rhs.end(), resultValues_.begin());
73  interpolation_ = boost::make_shared<MonotonicCubicNaturalSpline>(x_.begin(), x_.end(),
74  resultValues_.begin());
75  }

82  Real Fdm1DimSolver::thetaAt(Real x) const {
83  QL_REQUIRE(conditions_->stoppingTimes().front() > 0.0,
84  "stopping time at zero-> can't calculate theta");
85 
86  calculate();
87  Array thetaValues(resultValues_.size());
88 
89  const Array& rhs = thetaCondition_->getValues();
90  std::copy(rhs.begin(), rhs.end(), thetaValues.begin());
91 
92  Real temp = MonotonicCubicNaturalSpline(
93  x_.begin(), x_.end(), thetaValues.begin())(x);
94  return ( temp - interpolateAt(x) ) / thetaCondition_->getTime();
95  }
