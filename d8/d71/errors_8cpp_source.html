57 msg << "\n" << file << ":" << line << ": ";
58 #endif
59 #ifdef QL_ERROR_FUNCTIONS
60 if (function != "(unknown)")
61 msg << "In function `" << function << "': \n";
62 #endif
63 msg << message;
64 return msg.str();
