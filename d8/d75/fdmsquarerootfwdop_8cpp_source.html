40  FdmSquareRootFwdOp::FdmSquareRootFwdOp(
41  const ext::shared_ptr<FdmMesher>& mesher,
42  Real kappa, Real theta, Real sigma,
43  Size direction, TransformationType transform)
44  : direction_(direction),
45  kappa_(kappa),
46  theta_(theta),
47  sigma_(sigma),
48  transform_(transform),
49  mapX_(transform == Plain ?
50  new ModTripleBandLinearOp(FirstDerivativeOp(direction_, mesher)
51  .mult(kappa*(mesher->locations(direction_)-theta) + sigma*sigma)
52  .add(SecondDerivativeOp(direction_, mesher)
53  .mult(0.5*sigma*sigma*mesher->locations(direction_)))
54  .add(Array(mesher->layout()->size(), kappa)))
55 
56  : transform == Power ? new ModTripleBandLinearOp(
57  SecondDerivativeOp(direction_, mesher)
58  .mult(0.5*sigma*sigma*mesher->locations(direction_))
59  .add(FirstDerivativeOp(direction_, mesher)
60  .mult(kappa*(mesher->locations(direction_)+theta)))
61  .add(Array(mesher->layout()->size(),
62  2*kappa*kappa*theta/(sigma*sigma))))
63 
64  : new ModTripleBandLinearOp(FirstDerivativeOp(direction_, mesher)
65  .mult(Exp(-mesher->locations(direction))
66  *( -0.5*sigma*sigma - kappa*theta) + kappa)
67  .add(SecondDerivativeOp(direction_, mesher)
68  .mult(0.5*sigma*sigma*Exp(-mesher->locations(direction))))
69  .add(kappa*theta*Exp(-mesher->locations(direction))))
70  ),
71  v_ (mesher->layout()->dim()[direction_]) {
72 
73  const FdmLinearOpIterator endIter = mesher->layout()->end();
74  for (FdmLinearOpIterator iter = mesher->layout()->begin();
75  iter != endIter; ++iter) {
76  const Real v = mesher->location(iter, direction_);
77  v_[iter.coordinates()[direction_]] = v;
78  }
79 
80  // zero flux boundary condition
81  setLowerBC(mesher);
82  setUpperBC(mesher);
83  }

85  void FdmSquareRootFwdOp::setLowerBC(
86  const ext::shared_ptr<FdmMesher>& mesher) {
87  const Size n = 1;
88  Real alpha, beta, gamma;
89 
90  getCoeff(alpha, beta, gamma, n);
91  const Real f = lowerBoundaryFactor(transform_);
92 
93  const Real b = -(h(n-1)+h(n))/zeta(n);
94  const Real c = h(n-1)/zetap(n);
95 
96  const FdmLinearOpIterator endIter = mesher->layout()->end();
97  for (FdmLinearOpIterator iter = mesher->layout()->begin();
98  iter != endIter; ++iter) {
99  if (iter.coordinates()[direction_] == 0) {
100  const Size idx = iter.index();
101  mapX_->diag()[idx] = beta + f*b; //*v(n-1);
102  mapX_->upper()[idx] = gamma + f*c; //*v(n-1);
103  }
104  }
105  }

107  void FdmSquareRootFwdOp::setUpperBC(
108  const ext::shared_ptr<FdmMesher>& mesher) {
109  const Size n = v_.size();
110  Real alpha, beta, gamma;
111 
112  getCoeff(alpha, beta, gamma, n);
113  const Real f = upperBoundaryFactor(transform_);
114 
115  const Real b = (h(n)+h(n-1))/zeta(n);
116  const Real c = -h(n)/zetam(n);
117 
118  const FdmLinearOpIterator endIter = mesher->layout()->end();
119  for (FdmLinearOpIterator iter = mesher->layout()->begin();
120  iter != endIter; ++iter) {
121  if (iter.coordinates()[direction_] == n-1) {
122  const Size idx = iter.index();
123  mapX_->diag()[idx] = beta + f*b; //*v(n+1);
124  mapX_->lower()[idx] = alpha + f*c; //*v(n+1);
125  }
126  }
127  }

157  Real FdmSquareRootFwdOp::f0Plain() const {
158  const Size n = 1;
159  const Real a = -(2*h(n-1)+h(n))/zetam(n);
160  const Real alpha = sigma_*sigma_*v(n)/zetam(n) - mu(n)*h(n)/zetam(n);
161  const Real nu = a*v(n-1) + (2*kappa_*(v(n-1)-theta_) + sigma_*sigma_)
162  /(sigma_*sigma_);
163 
164  return alpha/nu*v(n-1);
165  }

168  const Size n = v_.size();

171  const Real nu = a*v(n+1) + (2*kappa_*(v(n+1)-theta_) + sigma_*sigma_)
172  /(sigma_*sigma_);

177  Real FdmSquareRootFwdOp::f0Power() const {
178  const Size n = 1;
179  const Real mu = kappa_*(v(n)+theta_);
180  const Real a = -(2*h(n-1)+h(n))/zetam(n);
181  const Real alpha = sigma_*sigma_*v(n)/zetam(n) - mu*h(n)/zetam(n);
182  const Real nu = a*v(n-1) +2*(kappa_*v(n-1)/(sigma_*sigma_));

219  Real FdmSquareRootFwdOp::v(Size i) const {
220  if (i > 0 && i <= v_.size()) {
221  return v_[i-1];
222  }
223  else if (i == 0) {
224  if (transform_ == Log) {
225  return 2*v_[0] - v_[1];
226 // log(std::max(0.5*exp(v_[0]), exp(v_[0] - 0.01 * (v_[1] - v_[0]))));
227  } else {
228  return std::max(0.5*v_[0], v_[0] - 0.01 * (v_[1] - v_[0]));
229  }
230  }
231  else if (i == v_.size()+1) {
232  return v_.back() + (v_.back() - *(v_.end()-2));
233  }
234  else {
235  QL_FAIL("unknown index");

274  void FdmSquareRootFwdOp::getCoeffPlain(Real& alpha, Real& beta,
275  Real& gamma, Size n) const {
276  alpha = sigma_*sigma_*v(n)/zetam(n) - mu(n)*h(n)/zetam(n);
277  beta = - sigma_*sigma_*v(n)/zeta(n)
278  + mu(n)*(h(n)-h(n-1))/zeta(n) + kappa_;
279  gamma = sigma_*sigma_*v(n)/zetap(n) + mu(n)*h(n-1)/zetap(n);
280

283  void FdmSquareRootFwdOp::getCoeffLog(Real& alpha, Real& beta,
284  Real& gamma, Size n) const {
285  const Real mu = ((-kappa_*theta_-sigma_*sigma_/2.0)*exp(-v(n))+kappa_);
286  alpha = sigma_*sigma_*exp(-v(n))/zetam(n) - mu*h(n)/zetam(n);
287  beta = - sigma_*sigma_*exp(-v(n))/zeta(n)
288  + mu*(h(n)-h(n-1))/zeta(n) + kappa_*theta_*exp(-v(n));
289  gamma = sigma_*sigma_*exp(-v(n))/zetap(n) + mu*h(n-1)/zetap(n);

292  void FdmSquareRootFwdOp::getCoeffPower(Real& alpha, Real& beta,
293  Real& gamma, Size n) const {
294  const Real mu = kappa_*(theta_+v(n));
295  alpha = (sigma_*sigma_*v(n) - mu*h(n))/zetam(n);
296  beta = (-sigma_*sigma_*v(n) + mu*(h(n)-h(n-1)))/zeta(n)
297  + 2*kappa_*kappa_*theta_/(sigma_*sigma_);
298  gamma= (sigma_*sigma_*v(n) + mu*h(n-1))/zetap(n);
299  }
