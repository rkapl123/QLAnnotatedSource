33  LiborForwardModelProcess::LiborForwardModelProcess(
34  Size size,
35  const boost::shared_ptr<IborIndex>& index)
36  : StochasticProcess(boost::shared_ptr<discretization>(
37  new EulerDiscretization)),
38  size_ (size),
39  index_ (index),
40  initialValues_ (size_),
41  fixingTimes_ (size_),
42  fixingDates_ (size_),
43  accrualStartTimes_(size),
44  accrualEndTimes_ (size),
45  accrualPeriod_ (size_),
46  m1(size_), m2(size_) {
47 
48  const DayCounter dayCounter = index_->dayCounter();
49  const Leg flows = cashFlows();
50 
51  QL_REQUIRE(size_ == flows.size(), "wrong number of cashflows");
52 
53  Date settlement = index_->forwardingTermStructure()->referenceDate();
54  const Date startDate =
55  boost::dynamic_pointer_cast<IborCoupon>(flows[0])->fixingDate();
56 
57  for (Size i = 0; i < size_; ++i) {
58  const boost::shared_ptr<IborCoupon> coupon =
59  boost::dynamic_pointer_cast<IborCoupon>(flows[i]);
60 
61  QL_REQUIRE(coupon->date() == coupon->accrualEndDate(),
62  "irregular coupon types are not suppported");
63 
64  initialValues_[i] = coupon->rate();
65  accrualPeriod_[i] = coupon->accrualPeriod();
66 
67  fixingDates_[i] = coupon->fixingDate();
68  fixingTimes_[i] =
69  dayCounter.yearFraction(startDate, coupon->fixingDate());
70  accrualStartTimes_[i] =
71  dayCounter.yearFraction(settlement,coupon->accrualStartDate());
72  accrualEndTimes_[i] =
73  dayCounter.yearFraction(settlement,coupon->accrualEndDate());
74  }
75  }

77  Disposable<Array> LiborForwardModelProcess::drift(Time t,
78  const Array& x) const {
79  Array f(size_, 0.0);
80  Matrix covariance(lfmParam_->covariance(t, x));
81 
82  const Size m = nextIndexReset(t);
83 
84  for (Size k=m; k<size_; ++k) {
85  m1[k] = accrualPeriod_[k]*x[k]/(1+accrualPeriod_[k]*x[k]);
86  f[k] = std::inner_product(m1.begin()+m, m1.begin()+k+1,
87  covariance.column_begin(k)+m,0.0)
88  - 0.5*covariance[k][k];
89  }
90 
91  return f;
92  }

115  Disposable<Array> LiborForwardModelProcess::evolve(
116  Time t0, const Array& x0,
117  Time dt, const Array& dw) const {
118  /* predictor-corrector step to reduce discretization errors.
119 
120  Short - but slow - solution would be
121 
122  Array rnd_0 = stdDeviation(t0, x0, dt)*dw;
123  Array drift_0 = discretization_->drift(*this, t0, x0, dt);
124 
125  return apply(x0, ( drift_0 + discretization_
126  ->drift(*this,t0,apply(x0, drift_0 + rnd_0),dt) )*0.5 + rnd_0);
127 
128  The following implementation does the same but is faster.
129  */
130 
131  const Size m = nextIndexReset(t0);
132  const Real sdt = std::sqrt(dt);
133 
134  Array f(x0);
135  Matrix diff = lfmParam_->diffusion(t0, x0);
136  Matrix covariance = lfmParam_->covariance(t0, x0);
137 
138  for (Size k=m; k<size_; ++k) {
139  const Real y = accrualPeriod_[k]*x0[k];
140  m1[k] = y/(1+y);
141  const Real d = (
142  std::inner_product(m1.begin()+m, m1.begin()+k+1,
143  covariance.column_begin(k)+m,0.0)
144  -0.5*covariance[k][k]) * dt;
145 
146  const Real r = std::inner_product(
147  diff.row_begin(k), diff.row_end(k), dw.begin(), 0.0)*sdt;
148 
149  const Real x = y*std::exp(d + r);
150  m2[k] = x/(1+x);
151  f[k] = x0[k] * std::exp(0.5*(d+
152  (std::inner_product(m2.begin()+m, m2.begin()+k+1,
153  covariance.column_begin(k)+m,0.0)
154  -0.5*covariance[k][k])*dt)+ r);
155  }
156 
157  return f;
158  }

228  std::vector<DiscountFactor> LiborForwardModelProcess::discountBond(
229  const std::vector<Rate> & rates) const {
230 
231  std::vector<DiscountFactor> discountFactors(size_);
232  discountFactors[0] = 1.0/(1.0 + rates[0]*accrualPeriod_[0]);
233 
234  for (Size i = 1; i < size_; ++i) {
235  discountFactors[i] =
236  discountFactors[i-1]/(1.0 + rates[i]*accrualPeriod_[i]);
237  }
238
