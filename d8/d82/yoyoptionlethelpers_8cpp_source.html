88 void YoYOptionletHelper::setTermStructure(
89 YoYOptionletVolatilitySurface* v) {
90
91 BootstrapHelper<YoYOptionletVolatilitySurface>::setTermStructure(v);
92 // set up a new yoyCapFloor
93 // but this one does NOT own its inflation term structure
94 const bool own = false;
95 // create a handle to the new vol surface
96 Handle<YoYOptionletVolatilitySurface> volSurf(
97 ext::shared_ptr<YoYOptionletVolatilitySurface>(v, null_deleter()),
98 own);
99 // in this case all we need to do is reset the vol in the pricer
100 // we must do it because the surface is a different one each time
101 // i.e. the pointer (handle) changes, not just what it points to
102 pricer_->setVolatility(volSurf);
103 }
