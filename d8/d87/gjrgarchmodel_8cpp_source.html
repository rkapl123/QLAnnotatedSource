42  GJRGARCHModel::GJRGARCHModel(
43  const boost::shared_ptr<GJRGARCHProcess> & process)
44  : CalibratedModel(6), process_(process) {
45  arguments_[0] = ConstantParameter(process->omega(),
46  PositiveConstraint());
47  arguments_[1] = ConstantParameter(process->alpha(),
48  BoundaryConstraint( 0.0, 1.0));
49  arguments_[2] = ConstantParameter(process->beta(),
50  BoundaryConstraint( 0.0, 1.0));
51  arguments_[3] = ConstantParameter(process->gamma(),
52  BoundaryConstraint(-1.0, 1.0));
53  arguments_[4] = ConstantParameter(process->lambda(), NoConstraint());
54  arguments_[5] = ConstantParameter(process->v0(),
55  PositiveConstraint());
56 
57  constraint_ = boost::shared_ptr<Constraint>(
58  new CompositeConstraint(*constraint_, VolatilityConstraint()));
59 
60  generateArguments();
61 
62  registerWith(process_->riskFreeRate());
63  registerWith(process_->dividendYield());
64  registerWith(process_->s0());
65  }
