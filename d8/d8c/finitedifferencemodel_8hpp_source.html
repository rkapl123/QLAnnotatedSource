89  void rollbackImpl(array_type& a,
90  Time from,
91  Time to,
92  Size steps,
93  const condition_type* condition) {
94 
95  QL_REQUIRE(from >= to,
96  "trying to roll back from " << from << " to " << to);
97 
98  Time dt = (from-to)/steps, t = from;
99  evolver_.setStep(dt);
100 
101  if(!stoppingTimes_.empty() && stoppingTimes_.back() == from) {
102  if (condition)
103  condition->applyTo(a,from);
104  }
105  for (Size i=0; i<steps; ++i, t -= dt) {
106  Time now = t;
107  // make sure last step ends exactly on "to" in order to not
108  // miss a stopping time at "to" due to numerical issues
109  Time next = (i < steps -1)? t-dt : to;
110 
111  if (std::fabs(to-next) < std::sqrt(QL_EPSILON)) next = to;
112  bool hit = false;
113  for (Integer j = static_cast<Integer>(stoppingTimes_.size())-1; j >= 0 ; --j) {
114  if (next <= stoppingTimes_[j] && stoppingTimes_[j] < now) {
115  // a stopping time was hit
116  hit = true;
117 
118  // perform a small step to stoppingTimes_[j]...
119  evolver_.setStep(now-stoppingTimes_[j]);
120  evolver_.step(a,now);
121  if (condition)
122  condition->applyTo(a,stoppingTimes_[j]);
123  // ...and continue the cycle
124  now = stoppingTimes_[j];
125  }
126  }
127  // if we did hit...
128  if (hit) {
129  // ...we might have to make a small step to
130  // complete the big one...
131  if (now > next) {
132  evolver_.setStep(now - next);
133  evolver_.step(a,now);
134  if (condition)
135  condition->applyTo(a,next);
136  }
137  // ...and in any case, we have to reset the
138  // evolver to the default step.
139  evolver_.setStep(dt);
140  } else {
141  // if we didn't, the evolver is already set to the
142  // default step, which is ok for us.
143  evolver_.step(a,now);
144  if (condition)
145  condition->applyTo(a, next);
146  }
147  }
148  }
149  Evolver evolver_;
150  std::vector<Time> stoppingTimes_;
151  };
