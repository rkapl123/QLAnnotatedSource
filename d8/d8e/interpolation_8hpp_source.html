100  bool isInRange(Real x) const {
101  #if defined(QL_EXTRA_SAFETY_CHECKS)
102  for (I1 i=xBegin_, j=xBegin_+1; j!=xEnd_; ++i, ++j)
103  QL_REQUIRE(*j > *i, "unsorted x values");
104  #endif
105  Real x1 = xMin(), x2 = xMax();
106  return (x >= x1 && x <= x2) || close(x,x1) || close(x,x2);
107  }
108  protected:
109  Size locate(Real x) const {
110  #if defined(QL_EXTRA_SAFETY_CHECKS)
111  for (I1 i=xBegin_, j=xBegin_+1; j!=xEnd_; ++i, ++j)
112  QL_REQUIRE(*j > *i, "unsorted x values");
113  #endif
114  if (x < *xBegin_)
115  return 0;
116  else if (x > *(xEnd_-1))
117  return xEnd_-xBegin_-2;
118  else
119  return std::upper_bound(xBegin_,xEnd_-1,x)-xBegin_-1;
120  }
