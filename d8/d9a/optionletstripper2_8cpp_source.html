60  void OptionletStripper2::performCalculations() const {
61 
63  optionletDates_ = stripper1_->optionletFixingDates();
64  optionletPaymentDates_ = stripper1_->optionletPaymentDates();
65  optionletAccrualPeriods_ = stripper1_->optionletAccrualPeriods();
66  optionletTimes_ = stripper1_->optionletFixingTimes();
67  atmOptionletRate_ = stripper1_->atmOptionletRates();
68  for (Size i=0; i<optionletTimes_.size(); ++i) {
69  optionletStrikes_[i] = stripper1_->optionletStrikes(i);
70  optionletVolatilities_[i] = stripper1_->optionletVolatilities(i);
71  }
72 
73  // atmCapFloorTermVolCurve data
74  const std::vector<Period>& optionExpiriesTenors =
75  atmCapFloorTermVolCurve_->optionTenors();
76  const std::vector<Time>& optionExpiriesTimes =
77  atmCapFloorTermVolCurve_->optionTimes();
78 
79  for (Size j=0; j<nOptionExpiries_; ++j) {
80  Volatility atmOptionVol = atmCapFloorTermVolCurve_->volatility(
81  optionExpiriesTimes[j], 33.3333); // dummy strike
82  ext::shared_ptr<BlackCapFloorEngine> engine(new
83  BlackCapFloorEngine(iborIndex_->forwardingTermStructure(),
84  atmOptionVol, dc_));
85  caps_[j] = MakeCapFloor(CapFloor::Cap,
86  optionExpiriesTenors[j],
87  iborIndex_,
88  Null<Rate>(),
89  0*Days).withPricingEngine(engine);
90  atmCapFloorStrikes_[j] =
91  caps_[j]->atmRate(**iborIndex_->forwardingTermStructure());
92  atmCapFloorPrices_[j] = caps_[j]->NPV();
93  }
94 
95  spreadsVolImplied_ = spreadsVolImplied();
96 
97  StrippedOptionletAdapter adapter(stripper1_);
98  adapter.enableExtrapolation();
99 
100  Volatility unadjustedVol, adjustedVol;
101  for (Size j=0; j<nOptionExpiries_; ++j) {
102  for (Size i=0; i<optionletVolatilities_.size(); ++i) {
103  if (i<=caps_[j]->floatingLeg().size()) {
104  unadjustedVol = adapter.volatility(optionletTimes_[i],
105  atmCapFloorStrikes_[j]);
106  adjustedVol = unadjustedVol + spreadsVolImplied_[j];
107 
108  // insert adjusted volatility
109  std::vector<Rate>::const_iterator previous =
110  std::lower_bound(optionletStrikes_[i].begin(),
111  optionletStrikes_[i].end(),
112  atmCapFloorStrikes_[j]);
113  Size insertIndex = previous - optionletStrikes_[i].begin();
114 
115  optionletStrikes_[i].insert(
116  optionletStrikes_[i].begin() + insertIndex,
117  atmCapFloorStrikes_[j]);
118  optionletVolatilities_[i].insert(
119  optionletVolatilities_[i].begin() + insertIndex,
120  adjustedVol);
121  }
122  }
123  }
124  }

126  std::vector<Volatility> OptionletStripper2::spreadsVolImplied() const {
127 
128  Brent solver;
129  std::vector<Volatility> result(nOptionExpiries_);
130  Volatility guess = 0.0001, minSpread = -0.1, maxSpread = 0.1;
131  for (Size j=0; j<nOptionExpiries_; ++j) {
132  ObjectiveFunction f(stripper1_, caps_[j], atmCapFloorPrices_[j]);
133  solver.setMaxEvaluations(maxEvaluations_);
134  Volatility root = solver.solve(f, accuracy_, guess,
135  minSpread, maxSpread);
136  result[j] = root;
137  }
138  return result;
139  }

160  OptionletStripper2::ObjectiveFunction::ObjectiveFunction(
161  const ext::shared_ptr<OptionletStripper1>& optionletStripper1,
162  const ext::shared_ptr<CapFloor>& cap,
163  Real targetValue)
164  : cap_(cap),
165  targetValue_(targetValue)
166  {
167  ext::shared_ptr<OptionletVolatilityStructure> adapter(new
168  StrippedOptionletAdapter(optionletStripper1));
169  adapter->enableExtrapolation();
170 
171  // set an implausible value, so that calculation is forced
172  // at first operator()(Volatility x) call
173  spreadQuote_ = ext::make_shared<SimpleQuote>(-1.0);
174 
175  ext::shared_ptr<OptionletVolatilityStructure> spreadedAdapter(new
176  SpreadedOptionletVolatility(Handle<OptionletVolatilityStructure>(
177  adapter), Handle<Quote>(spreadQuote_)));
178 
179  ext::shared_ptr<BlackCapFloorEngine> engine(new
180  BlackCapFloorEngine(
181  optionletStripper1->iborIndex()->forwardingTermStructure(),
182  Handle<OptionletVolatilityStructure>(spreadedAdapter)));
183 
184  cap_->setPricingEngine(engine);
185  }

187  Real OptionletStripper2::ObjectiveFunction::operator()(Volatility s) const

192  }
