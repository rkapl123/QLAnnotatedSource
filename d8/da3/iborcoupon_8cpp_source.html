86 Rate IborCoupon::indexFixing() const {
87
88 initializeCachedData();
89
90 /* instead of just returning index_->fixing(fixingValueDate_)
91 its logic is duplicated here using a specialized iborIndex
92 forecastFixing overload which
93 1) allows to save date/time recalculations, and
94 2) takes into account par coupon needs
95 */
96 Date today = QuantLib::Settings::instance().evaluationDate();
97
98 if (fixingDate_>today)
99 return iborIndex_->forecastFixing(fixingValueDate_,
100 fixingEndDate_,
101 spanningTime_);
102
103 if (fixingDate_<today ||
104 QuantLib::Settings::instance().enforcesTodaysHistoricFixings()) {
105 // do not catch exceptions
106 Rate result = index_->pastFixing(fixingDate_);
107 QL_REQUIRE(result != Null<Real>(),
108 "Missing " << index_->name() << " fixing for " << fixingDate_);
109 return result;
110 }
111
112 try {
113 Rate result = index_->pastFixing(fixingDate_);
114 if (result!=Null<Real>())
115 return result;
116 else
117 ; // fall through and forecast
118 } catch (Error&) {
119 ; // fall through and forecast
120 }
121 return iborIndex_->forecastFixing(fixingValueDate_,
122 fixingEndDate_,
123 spanningTime_);
124 }

268 IborLeg::operator Leg() const {
269
270 Leg leg = FloatingLeg<IborIndex, IborCoupon, CappedFlooredIborCoupon>(
271 schedule_, notionals_, index_, paymentDayCounter_,
272 paymentAdjustment_, fixingDays_, gearings_, spreads_,
273 caps_, floors_, inArrears_, zeroPayments_, paymentLag_, paymentCalendar_,
274 exCouponPeriod_, exCouponCalendar_, exCouponAdjustment_, exCouponEndOfMonth_);
275
276 if (caps_.empty() && floors_.empty() && !inArrears_) {
277 ext::shared_ptr<IborCouponPricer> pricer = ext::make_shared<BlackIborCouponPricer>(
278 Handle<OptionletVolatilityStructure>(),
279 BlackIborCouponPricer::TimingAdjustment::Black76,
280 Handle<Quote>(ext::make_shared<SimpleQuote>(1.0)), useIndexedCoupons_);
281 setCouponPricer(leg, pricer);
282 }
283
284 return leg;
285 }
