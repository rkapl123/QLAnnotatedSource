54  IborCoupon::IborCoupon(const Date& paymentDate,
55  Real nominal,
56  const Date& startDate,
57  const Date& endDate,
58  Natural fixingDays,
59  const ext::shared_ptr<IborIndex>& iborIndex,
60  Real gearing,
61  Spread spread,
62  const Date& refPeriodStart,
63  const Date& refPeriodEnd,
64  const DayCounter& dayCounter,
65  bool isInArrears,
66  const Date& exCouponDate)
67  : FloatingRateCoupon(paymentDate, nominal, startDate, endDate,
68  fixingDays, iborIndex, gearing, spread,
69  refPeriodStart, refPeriodEnd,
70  dayCounter, isInArrears, exCouponDate),
71  iborIndex_(iborIndex) {
72  constructorWasNotCalled_ = false;
73 
74  fixingDate_ = fixingDate();
75 
76  const Calendar& fixingCalendar = index_->fixingCalendar();
77  Natural indexFixingDays = index_->fixingDays();
78 
79  fixingValueDate_ = fixingCalendar.advance(
80  fixingDate_, indexFixingDays, Days);
81 
82  if (usingAtParCoupons_) {
83  if (isInArrears_)
84  fixingEndDate_ = index_->maturityDate(fixingValueDate_);
85  else { // par coupon approximation
86  Date nextFixingDate = fixingCalendar.advance(
87  accrualEndDate_, -static_cast<Integer>(fixingDays_), Days);
88  fixingEndDate_ = fixingCalendar.advance(
89  nextFixingDate, indexFixingDays, Days);
90  // make sure the estimation period contains at least one day
91  fixingEndDate_ = std::max(fixingEndDate_, fixingValueDate_ + 1);
92  }
93  } else {
94  fixingEndDate_ = index_->maturityDate(fixingValueDate_);
95  }
96 
97  const DayCounter& dc = index_->dayCounter();
98  spanningTime_ = dc.yearFraction(fixingValueDate_,
99  fixingEndDate_);
100  QL_REQUIRE(spanningTime_>0.0,
101  "\n cannot calculate forward rate between " <<
102  fixingValueDate_ << " and " << fixingEndDate_ <<
103  ":\n non positive time (" << spanningTime_ <<
104  ") using " << dc.name() << " daycounter");
105  }

107  Rate IborCoupon::indexFixing() const {
108 
109  /* instead of just returning index_->fixing(fixingValueDate_)
110  its logic is duplicated here using a specialized iborIndex
111  forecastFixing overload which
112  1) allows to save date/time recalculations, and
113  2) takes into account par coupon needs
114  */
115  Date today = Settings::instance().evaluationDate();
116 
117  if (fixingDate_>today)
118  return iborIndex_->forecastFixing(fixingValueDate_,
119  fixingEndDate_,
120  spanningTime_);
121 
122  if (fixingDate_<today ||
123  Settings::instance().enforcesTodaysHistoricFixings()) {
124  // do not catch exceptions
125  Rate result = index_->pastFixing(fixingDate_);
126  QL_REQUIRE(result != Null<Real>(),
127  "Missing " << index_->name() << " fixing for " << fixingDate_);
128  return result;
129  }
130 
131  try {
132  Rate result = index_->pastFixing(fixingDate_);
133  if (result!=Null<Real>())
134  return result;
135  else
136  ; // fall through and forecast
137  } catch (Error&) {
138  ; // fall through and forecast
139  }
140  return iborIndex_->forecastFixing(fixingValueDate_,
141  fixingEndDate_,
142  spanningTime_);
143  }
