35  IborCoupon::IborCoupon(const Date& paymentDate,
36  Real nominal,
37  const Date& startDate,
38  const Date& endDate,
39  Natural fixingDays,
40  const shared_ptr<IborIndex>& iborIndex,
41  Real gearing,
42  Spread spread,
43  const Date& refPeriodStart,
44  const Date& refPeriodEnd,
45  const DayCounter& dayCounter,
46  bool isInArrears)
47  : FloatingRateCoupon(paymentDate, nominal, startDate, endDate,
48  fixingDays, iborIndex, gearing, spread,
49  refPeriodStart, refPeriodEnd,
50  dayCounter, isInArrears),
51  iborIndex_(iborIndex) {
52 
53  fixingDate_ = fixingDate();
54 
55  const Calendar& fixingCalendar = index_->fixingCalendar();
56  Natural indexFixingDays = index_->fixingDays();
57 
58  fixingValueDate_ = fixingCalendar.advance(
59  fixingDate_, indexFixingDays, Days);
60 
61  #ifdef QL_USE_INDEXED_COUPON
62  fixingEndDate_ = index_->maturityDate(fixingValueDate_);
63  #else
64  if (isInArrears_)
65  fixingEndDate_ = index_->maturityDate(fixingValueDate_);
66  else { // par coupon approximation
67  Date nextFixingDate = fixingCalendar.advance(
68  accrualEndDate_, -static_cast<Integer>(fixingDays_), Days);
69  fixingEndDate_ = fixingCalendar.advance(
70  nextFixingDate, indexFixingDays, Days);
71  }
72  #endif
73 
74  const DayCounter& dc = index_->dayCounter();
75  spanningTime_ = dc.yearFraction(fixingValueDate_,
76  fixingEndDate_);
77  QL_REQUIRE(spanningTime_>0.0,
78  "\n cannot calculate forward rate between " <<
79  fixingValueDate_ << " and " << fixingEndDate_ <<
80  ":\n non positive time (" << spanningTime_ <<
81  ") using " << dc.name() << " daycounter");
82  }

84  Rate IborCoupon::indexFixing() const {
85 
86  /* instead of just returning index_->fixing(fixingValueDate_)
87  its logic is duplicated here using a specialized iborIndex
88  forecastFixing overload which
89  1) allows to save date/time recalculations, and
90  2) takes into account par coupon needs
91  */
92  Date today = Settings::instance().evaluationDate();
93 
94  if (fixingDate_>today)
95  return iborIndex_->forecastFixing(fixingValueDate_,
96  fixingEndDate_,
97  spanningTime_);
98 
99  if (fixingDate_<today ||
100  Settings::instance().enforcesTodaysHistoricFixings()) {
101  // do not catch exceptions
102  Rate result = index_->pastFixing(fixingDate_);
103  QL_REQUIRE(result != Null<Real>(),
104  "Missing " << index_->name() << " fixing for " << fixingDate_);
105  return result;
106  }
107 
108  try {
109  Rate result = index_->pastFixing(fixingDate_);
110  if (result!=Null<Real>())
111  return result;
112  else
113  ; // fall through and forecast
114  } catch (Error&) {
115  ; // fall through and forecast
116  }
117  return iborIndex_->forecastFixing(fixingValueDate_,
118  fixingEndDate_,
119  spanningTime_);
120  }

230  IborLeg::operator Leg() const {
231 
232  Leg leg = FloatingLeg<IborIndex, IborCoupon, CappedFlooredIborCoupon>(
233  schedule_, notionals_, index_, paymentDayCounter_,
234  paymentAdjustment_, fixingDays_, gearings_, spreads_,
235  caps_, floors_, inArrears_, zeroPayments_, paymentLag_, paymentCalendar_);
236 
237  if (caps_.empty() && floors_.empty() && !inArrears_) {
238  shared_ptr<IborCouponPricer> pricer(new BlackIborCouponPricer);
239  setCouponPricer(leg, pricer);
240  }
241 
242  return leg;
