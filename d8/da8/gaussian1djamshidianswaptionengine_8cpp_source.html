27 class Gaussian1dJamshidianSwaptionEngine::rStarFinder {
28 public:
29 rStarFinder(const ext::shared_ptr<Gaussian1dModel>& model,
30 Real nominal,
31 const Date& maturityDate,
32 const Date& valueDate,
33 std::vector<Date> fixedPayDates,
34 const std::vector<Real>& amounts,
35 const Size startIndex)
36 : strike_(nominal), maturityDate_(maturityDate), valueDate_(valueDate),
37 startIndex_(startIndex), times_(std::move(fixedPayDates)), amounts_(amounts),
38 model_(model) {}
39
40 Real operator()(Rate y) const {
41 Real value = strike_;
42 Size size = times_.size();
43 for (Size i = startIndex_; i < size; i++) {
44 Real dbValue = model_->zerobond(times_[i], maturityDate_, y) /
45 model_->zerobond(valueDate_, maturityDate_, y);
46 value -= amounts_[i] * dbValue;
47 }
48 return value;
49 }
50
51 private:
52 Real strike_;
53 Date maturityDate_, valueDate_;
54 Size startIndex_;
55 std::vector<Date> times_;
56 const std::vector<Real> &amounts_;
57 const ext::shared_ptr<Gaussian1dModel> &model_;
58 };

60 void Gaussian1dJamshidianSwaptionEngine::calculate() const {
61
62 QL_REQUIRE(arguments_.settlementMethod != Settlement::ParYieldCurve,
63 "cash settled (ParYieldCurve) swaptions not priced with "
64 "Gaussian1dJamshidianSwaptionEngine");
65
66 QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
67 "cannot use the Jamshidian decomposition "
68 "on exotic swaptions");
69
70 QL_REQUIRE(arguments_.swap->spread() == 0.0,
71 "non zero spread (" << arguments_.swap->spread()
72 << ") not allowed"); // PC
73
74 Date referenceDate;
75 DayCounter dayCounter;
76
77 referenceDate = model_->termStructure()->referenceDate();
78 dayCounter = model_->termStructure()->dayCounter();
79
80 std::vector<Real> amounts(arguments_.fixedCoupons);
81 amounts.back() += arguments_.nominal;
82
83 Size startIndex = std::upper_bound(arguments_.fixedResetDates.begin(),
84 arguments_.fixedResetDates.end(),
85 arguments_.exercise->date(0) - 1) -
86 arguments_.fixedResetDates.begin();
87 // only consider coupons with start date >= exercise dates
88
89 rStarFinder finder(*model_, arguments_.nominal,
90 arguments_.exercise->date(0),
91 arguments_.fixedResetDates[startIndex],
92 arguments_.fixedPayDates, amounts, startIndex);
93 Brent s1d;
94 Rate minStrike = -8.0;
95 Rate maxStrike = 8.0;
96 s1d.setMaxEvaluations(10000);
97 s1d.setLowerBound(minStrike);
98 s1d.setUpperBound(maxStrike);
99 Rate rStar = s1d.solve(finder, 1e-8, 0.00, minStrike,
100 maxStrike); // this is actually yStar
101
102 Option::Type w =
103 arguments_.type == Swap::Payer ? Option::Put : Option::Call;
104 Size size = arguments_.fixedCoupons.size();
105
106 Real value = 0.0;
107 for (Size i = startIndex; i < size; i++) {
108 // Real fixedPayTime =
109 // dayCounter.yearFraction(referenceDate,arguments_.fixedPayDates[i]);
110 Real strike =
111 model_->zerobond(arguments_.fixedPayDates[i],
112 arguments_.exercise->date(0), rStar) /
113 model_->zerobond(arguments_.fixedResetDates[startIndex],
114 arguments_.exercise->date(0), rStar);
115 Real dboValue =
116 model_->zerobondOption(w, arguments_.exercise->date(0),
117 arguments_.fixedResetDates[startIndex],
118 arguments_.fixedPayDates[i], strike);
119 value += amounts[i] * dboValue;
120 }
121 results_.value = value;
122 }
