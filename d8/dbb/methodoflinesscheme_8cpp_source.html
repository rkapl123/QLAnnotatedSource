46  void MethodOfLinesScheme::step(array_type& a, Time t) {
47  using namespace ext::placeholders;
48  QL_REQUIRE(t-dt_ > -1e-8, "a step towards negative time given");
49 
50  const std::vector<Real> v =
51  AdaptiveRungeKutta<Real>(eps_, relInitStepSize_*dt_)(
52  ext::bind(&MethodOfLinesScheme::apply, this, _1, _2),
53  std::vector<Real>(a.begin(), a.end()),
54  t, std::max(0.0, t-dt_));
55 
56  Array y(v.begin(), v.end());
57 
58  bcSet_.applyAfterSolving(y);
59 
60  a = y;
61  }
