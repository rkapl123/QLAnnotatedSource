98 inline Rate DriftTermStructure::zeroYieldImpl(Time t) const {
99 // warning: here it is assumed that
100 // a) all TS have the same daycount.
101 // b) all TS have the same referenceDate
102 // It should be QL_REQUIREd
103 return riskFreeTS_->zeroRate(t, Continuous, NoFrequency, true).rate()
104 - dividendTS_->zeroRate(t, Continuous, NoFrequency, true).rate()
105 - 0.5 * blackVolTS_->blackVol(t, underlyingLevel_, true)
106 * blackVolTS_->blackVol(t, underlyingLevel_, true);
107 }
