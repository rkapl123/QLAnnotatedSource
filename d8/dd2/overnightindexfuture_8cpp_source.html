45  Real OvernightIndexFuture::averagedSpotValue() const {
46  Date today = Settings::instance().evaluationDate();
47  Real avg = 0;
48  Date d1 = valueDate_;
49  const TimeSeries<Real>& history = IndexManager::instance()
50  .getHistory(overnightIndex_->name());
51  Real fwd;
52  while (d1 < maturityDate_) {
53  Date d2 = calendar_.advance(d1, 1, Days);
54  if (d1 < today) {
55  fwd = history[d1];
56  QL_REQUIRE(fwd != Null<Real>(), "missing rate on " <<
57  d1 << " for index " << overnightIndex_->name());
58  } else {
59  fwd = discountCurve_->forwardRate(d1, d2, dayCounter_, Simple).rate();
60  }
61  avg += fwd * dayCounter_.yearFraction(d1, d2);
62  d1 = d2;
63  }
64 
65  Real convAdj = convexityAdjustment_.empty() ? 0.0 :
66  convexityAdjustment_->value();
67  Real R = convAdj + avg /
68  dayCounter_.yearFraction(valueDate_, maturityDate_);
69  return 100.0 * (1.0 - R);
70  }

72  Real OvernightIndexFuture::compoundedSpotValue() const {
73  Date today = Settings::instance().evaluationDate();
74  Real prod = 1;
75  if (today > valueDate_) {
76  // can't value on a weekend inside reference period because we
77  // won't know the reset rate until start of next business day.
78  // user can supply an estimate if they really want to do this
79  today = calendar_.adjust(today, businessDayConvention_);
80  // for valuations inside the reference period, index quotes
81  // must have been populated in the history
82  const TimeSeries<Real>& history = IndexManager::instance()
83  .getHistory(overnightIndex_->name());
84  Date d1 = valueDate_;
85  while (d1 < today) {
86  Real r = history[d1];
87  QL_REQUIRE(r != Null<Real>(), "missing rate on " <<
88  d1 << " for index " << overnightIndex_->name());
89  Date d2 = calendar_.advance(d1, 1, Days);
90  prod *= 1 + r * dayCounter_.yearFraction(d1, d2);
91  d1 = d2;
92  }
93  }
94  DiscountFactor forwardDiscount = discountCurve_->discount(maturityDate_);
95  if (valueDate_ > today) {
96  forwardDiscount /= discountCurve_->discount(valueDate_);
97  }
98  prod /= forwardDiscount;
99 
100  Real convAdj = convexityAdjustment_.empty() ? 0.0 :
101  convexityAdjustment_->value();
102  Real R = convAdj + (prod - 1) /
103  dayCounter_.yearFraction(valueDate_, maturityDate_);
104  return 100.0 * (1.0 - R);
105  }

107  Real OvernightIndexFuture::spotValue() const {
