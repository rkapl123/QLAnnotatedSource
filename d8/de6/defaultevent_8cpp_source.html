133  bool DefaultEvent::matchesDefaultKey(
134  const DefaultProbKey& contractKey) const {
135  if(bondsCurrency_ != contractKey.currency()) return false;
136  // a contract with NoSeniority matches all events
137  if((bondsSeniority_ != contractKey.seniority())
138  && (contractKey.seniority() != NoSeniority))
139  return false;
140  // loop on all event types in the contract and chek if we match any,
141  // calls derived types
142  for(Size i=0; i<contractKey.size(); i++) {
143  if(this->matchesEventType(contractKey.eventTypes()[i])) return true;
144  }
145  return false;
146  }

158  bool FailureToPayEvent::matchesEventType(
159  const boost::shared_ptr<DefaultType>& contractEvType) const {
160  boost::shared_ptr<FailureToPay> eveType =
161  boost::dynamic_pointer_cast<FailureToPay>(contractEvType);
162  // this chekcs the atomic types, no need to call parents method
163  if(!eveType) return false;
164  if(defaultedAmount_ < eveType->amountRequired()) return false;
165  Date today = Settings::instance().evaluationDate();
166  if(!this->hasOccurred(today - eveType->gracePeriod(), true))
167  return false;
168  return true;
169  }
