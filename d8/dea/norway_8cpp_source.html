30  bool Norway::Impl::isBusinessDay(const Date& date) const {
31  Weekday w = date.weekday();
32  Day d = date.dayOfMonth(), dd = date.dayOfYear();
33  Month m = date.month();
34  Year y = date.year();
35  Day em = easterMonday(y);
36  if (isWeekend(w)
37  // Holy Thursday
38  || (dd == em-4)
39  // Good Friday
40  || (dd == em-3)
41  // Easter Monday
42  || (dd == em)
43  // Ascension Thursday
44  || (dd == em+38)
45  // Whit Monday
46  || (dd == em+49)
47  // New Year's Day
48  || (d == 1 && m == January)
49  // May Day
50  || (d == 1 && m == May)
51  // National Independence Day
52  || (d == 17 && m == May)
53  // Christmas
54  || (d == 25 && m == December)
55  // Boxing Day
56  || (d == 26 && m == December))
57  return false; // NOLINT(readability-simplify-boolean-expr)
58  return true;
