102  void ContinuousPartialFloatingLookbackOption::arguments::validate() const {
103 
104  ContinuousFloatingLookbackOption::arguments::validate();
105 
106  boost::shared_ptr<EuropeanExercise> europeanExercise =
107  boost::dynamic_pointer_cast<EuropeanExercise>(exercise);
108  QL_REQUIRE(lookbackPeriodEnd <= europeanExercise->lastDate(),
109  "lookback start date must be earlier than exercise date");
110 
111  boost::shared_ptr<FloatingTypePayoff> floatingTypePayoff =
112  boost::dynamic_pointer_cast<FloatingTypePayoff>(payoff);
113 
114  if (floatingTypePayoff->optionType() == Option::Call) {
115  QL_REQUIRE(lambda >= 1.0,
116  "lambda should be greater than or equal to 1 for calls");
117  }
118  if (floatingTypePayoff->optionType() == Option::Put) {
119  QL_REQUIRE(lambda <= 1.0,
120  "lambda should be smaller than or equal to 1 for puts");
121  }
122  }
