52  void FdmExtendedOrnsteinUhlenbeckOp::setTime(Time t1, Time t2) {
53  const Rate r = rTS_->forwardRate(t1, t2, Continuous).rate();
54 
55  const ext::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
56  const FdmLinearOpIterator endIter = layout->end();
57 
58  Array drift(layout->size());
59  for (FdmLinearOpIterator iter = layout->begin();
60  iter!=endIter; ++iter) {
61  const Size i = iter.index();
62  drift[i] = process_->drift(0.5*(t1+t2), x_[i]);
63  }
64  mapX_.axpyb(drift, dxMap_, dxxMap_, Array(1, -r));
65  }
