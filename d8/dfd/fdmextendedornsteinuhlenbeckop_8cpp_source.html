57  void FdmExtendedOrnsteinUhlenbackOp::setTime(Time t1, Time t2) {
58  const Rate r = rTS_->forwardRate(t1, t2, Continuous).rate();
59 
60  const boost::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
61  const FdmLinearOpIterator endIter = layout->end();
62 
63  Array drift(layout->size());
64  for (FdmLinearOpIterator iter = layout->begin();
65  iter!=endIter; ++iter) {
66  const Size i = iter.index();
67  drift[i] = process_->drift(0.5*(t1+t2), x_[i]);
68  }
69  mapX_.axpyb(drift, dxMap_, dxxMap_, Array(1, -r));
70  }
