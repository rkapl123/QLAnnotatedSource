58  Real GeneralizedOrnsteinUhlenbeckProcess::variance(
59  Time t, Real, Time dt) const {
60  Real speed = speed_(t);
61  Volatility vol = volatility_(t);
62 
63  if (speed < std::sqrt(QL_EPSILON)) {
64  // algebraic limit for small speed
65  return vol*vol*dt;
66  } else {
67  return 0.5*vol*vol/speed*(1.0 - std::exp(-2.0*speed*dt));
68  }
69  }
