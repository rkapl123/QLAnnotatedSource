108  template <typename ForwardIterator>
109  void calibrate(ForwardIterator begin, ForwardIterator end) {
110  std::vector<Volatility> r2;
111  Real mean_r2 = to_r2(begin, end, r2);
112  ext::shared_ptr<Problem> p =
113  calibrate_r2(mode_, r2, mean_r2, alpha_, beta_, vl_);
114  gamma_ = 1 - alpha_ - beta_;
115  vl_ /= gamma_;
116  logLikelihood_ = p ? -p->functionValue() :
117  -costFunction(begin, end);
118  }

155  // a helper for calculation of r^2 and <r^2>
156  template <typename InputIterator>
157  static Real to_r2(InputIterator begin, InputIterator end,
158  std::vector<Volatility>& r2) {
159  Real u2(0.0), mean_r2(0.0), w(1.0);
160  for (; begin != end; ++begin) {
161  u2 = *begin; u2 *= u2;
162  mean_r2 = (1.0 - w) * mean_r2 + w * u2;
163  r2.push_back(u2);
164  w /= (w + 1.0);
165  }
166  return mean_r2;
167  }

236  template<class InputIterator>
237  static Real costFunction(InputIterator begin, InputIterator end,
238  Real alpha, Real beta, Real omega) {
239  Real retval(0.0);
240  Real u2(0.0), sigma2(0.0);
241  Size N = 0;
242  for (; begin != end; ++begin, ++N) {
243  sigma2 = omega + alpha * u2 + beta * sigma2;
244  u2 = *begin; u2 *= u2;
245  retval += std::log(sigma2) + u2 / sigma2;
246  }
247  return N > 0 ? retval / (2*N) : 0.0;
248  }
