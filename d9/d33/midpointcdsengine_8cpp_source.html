39  void MidPointCdsEngine::calculate() const {
40  QL_REQUIRE(!discountCurve_.empty(),
41  "no discount term structure set");
42  QL_REQUIRE(!probability_.empty(),
43  "no probability term structure set");
44 
45  Date today = Settings::instance().evaluationDate();
46  Date settlementDate = discountCurve_->referenceDate();
47 
48  // Upfront Flow NPV and accrual rebate NPV. Either we are on-the-run (no flow)
49  // or we are forward start
50  //
51 
52  // date determining the probability survival so we have to pay
53  // the upfront flows (did not knock out)
54  Date effectiveProtectionStart =
55  arguments_.protectionStart > probability_->referenceDate() ?
56  arguments_.protectionStart : probability_->referenceDate();
57 
58  Probability nonKnockOut =
59  probability_->survivalProbability(effectiveProtectionStart);
60 
61  Real upfPVO1 = 0.0;
62  results_.upfrontNPV = 0.0;
63  if (!arguments_.upfrontPayment->hasOccurred(
64  settlementDate,
65  includeSettlementDateFlows_)) {
66  upfPVO1 =
67  nonKnockOut *
68  discountCurve_->discount(arguments_.upfrontPayment->date());
69  if(arguments_.upfrontPayment->amount() != 0.) {
70  results_.upfrontNPV =
71  upfPVO1 * arguments_.upfrontPayment->amount();
72  }
73  }
74 
75  results_.accrualRebateNPV = 0.;
76  // NOLINTNEXTLINE(readability-implicit-bool-conversion)
77  if (arguments_.accrualRebate && arguments_.accrualRebate->amount() != 0. &&
78  !arguments_.accrualRebate->hasOccurred(settlementDate, includeSettlementDateFlows_)) {
79  results_.accrualRebateNPV = nonKnockOut *
80  discountCurve_->discount(arguments_.accrualRebate->date()) *
81  arguments_.accrualRebate->amount();
82 
83  }
84 
85  results_.couponLegNPV = 0.0;
86  results_.defaultLegNPV = 0.0;
87  for (Size i=0; i<arguments_.leg.size(); ++i) {
88  if (arguments_.leg[i]->hasOccurred(settlementDate,
89  includeSettlementDateFlows_))
90  continue;
91 
92  ext::shared_ptr<FixedRateCoupon> coupon =
93  ext::dynamic_pointer_cast<FixedRateCoupon>(arguments_.leg[i]);
94 
95  // In order to avoid a few switches, we calculate the NPV
96  // of both legs as a positive quantity. We'll give them
97  // the right sign at the end.
98 
99  Date paymentDate = coupon->date(),
100  startDate = coupon->accrualStartDate(),
101  endDate = coupon->accrualEndDate();
102  // this is the only point where it might not coincide
103  if (i==0)
104  startDate = arguments_.protectionStart;
105  Date effectiveStartDate =
106  (startDate <= today && today <= endDate) ? today : startDate;
107  Date defaultDate = // mid-point
108  effectiveStartDate + (endDate-effectiveStartDate)/2;
109 
110  Probability S = probability_->survivalProbability(paymentDate);
111  Probability P = probability_->defaultProbability(
112  effectiveStartDate,
113  endDate);
114 
115  // on one side, we add the fixed rate payments in case of
116  // survival...
117  results_.couponLegNPV +=
118  S * coupon->amount() *
119  discountCurve_->discount(paymentDate);
120  // ...possibly including accrual in case of default.
121  if (arguments_.settlesAccrual) {
122  if (arguments_.paysAtDefaultTime) {
123  results_.couponLegNPV +=
124  P * coupon->accruedAmount(defaultDate) *
125  discountCurve_->discount(defaultDate);
126  } else {
127  // pays at the end
128  results_.couponLegNPV +=
129  P * coupon->amount() *
130  discountCurve_->discount(paymentDate);
131  }
132  }
133 
134  // on the other side, we add the payment in case of default.
135  Real claim = arguments_.claim->amount(defaultDate,
136  arguments_.notional,
137  recoveryRate_);
138  if (arguments_.paysAtDefaultTime) {
139  results_.defaultLegNPV +=
140  P * claim * discountCurve_->discount(defaultDate);
141  } else {
142  results_.defaultLegNPV +=
143  P * claim * discountCurve_->discount(paymentDate);
144  }
145  }
146 
147  Real upfrontSign = 1.0;
148  switch (arguments_.side) {
149  case Protection::Seller:
150  results_.defaultLegNPV *= -1.0;
151  results_.accrualRebateNPV *= -1.0;
152  break;
153  case Protection::Buyer:
154  results_.couponLegNPV *= -1.0;
155  results_.upfrontNPV *= -1.0;
156  upfrontSign = -1.0;
157  break;
158  default:
159  QL_FAIL("unknown protection side");
160  }
161 
162  results_.value =
163  results_.defaultLegNPV + results_.couponLegNPV +
164  results_.upfrontNPV + results_.accrualRebateNPV;
165  results_.errorEstimate = Null<Real>();
166 
167  if (results_.couponLegNPV != 0.0) {
168  results_.fairSpread =
169  -results_.defaultLegNPV*arguments_.spread/
170  (results_.couponLegNPV + results_.accrualRebateNPV);
171  } else {
172  results_.fairSpread = Null<Rate>();
173  }
174 
175  Real upfrontSensitivity = upfPVO1 * arguments_.notional;
176  if (upfrontSensitivity != 0.0) {
177  results_.fairUpfront =
178  -upfrontSign*(results_.defaultLegNPV + results_.couponLegNPV +
179  results_.accrualRebateNPV)
180  / upfrontSensitivity;
181  } else {
182  results_.fairUpfront = Null<Rate>();
183  }
184 
185  static const Rate basisPoint = 1.0e-4;
186 
187  if (arguments_.spread != 0.0) {
188  results_.couponLegBPS =
189  results_.couponLegNPV*basisPoint/arguments_.spread;
190  } else {
191  results_.couponLegBPS = Null<Rate>();
192  }
193 
194  // NOLINTNEXTLINE(readability-implicit-bool-conversion)
195  if (arguments_.upfront && *arguments_.upfront != 0.0) {
196  results_.upfrontBPS =
197  results_.upfrontNPV*basisPoint/(*arguments_.upfront);
198  } else {
199  results_.upfrontBPS = Null<Rate>();
200  }
201  }
202 
203 }
