40  void MultiplicativePriceSeasonality::validate() const
41  {
42  switch (this->frequency()) {
43  case Semiannual: //2
44  case EveryFourthMonth: //3
45  case Quarterly: //4
46  case Bimonthly: //6
47  case Monthly: //12
48  case Biweekly: // etc.
49  case Weekly:
50  case Daily:
51  QL_REQUIRE(!this->seasonalityFactors().empty(), "no seasonality factors given");
52  QL_REQUIRE( (this->seasonalityFactors().size() %
53  this->frequency()) == 0,
54  "For frequency " << this->frequency()
55  << " require multiple of " << ((int)this->frequency()) << " factors "
56  << this->seasonalityFactors().size() << " were given.");
57  break;
58  default:
59  QL_FAIL("bad frequency specified: " << this->frequency()
60  << ", only semi-annual through daily permitted.");
61  break;
62  }

66  bool MultiplicativePriceSeasonality::isConsistent(const InflationTermStructure& iTS) const
67  {
68  // If multi-year is the specification consistent with the term structure start date?
69  // We do NOT test daily seasonality because this will, in general, never be consistent
70  // given weekends, holidays, leap years, etc.
71  if(this->frequency() == Daily) return true;
72  if(Size(this->frequency()) == seasonalityFactors().size()) return true;
73 
74  // how many years do you need to test?
75  Size nTest = seasonalityFactors().size() / this->frequency();
76  // ... relative to the start of the inflation curve
77  std::pair<Date,Date> lim = inflationPeriod(iTS.baseDate(), iTS.frequency());
78  Date curveBaseDate = lim.second;
79  Real factorBase = this->seasonalityFactor(curveBaseDate);
80 
81  Real eps = 0.00001;
82  for (Size i = 1; i < nTest; i++) {
83  Real factorAt = this->seasonalityFactor(curveBaseDate+Period(i,Years));
84  QL_REQUIRE(std::fabs(factorAt-factorBase)<eps,"seasonality is inconsistent with inflation term structure, factors "
85  << factorBase << " and later factor " << factorAt << ", " << i << " years later from inflation curve "
86  <<" with base date at " << curveBaseDate);
87  }
88 
89  return true;
90  }

125  const Rate r,
126  const InflationTermStructure& iTS) const {
127  // Mimic the logic in ZeroInflationIndex::forecastFixing for choosing the

147  Real MultiplicativePriceSeasonality::seasonalityFactor(const Date &to) const {
148 
149  Date from = seasonalityBaseDate();
150  Frequency factorFrequency = frequency();
151  Size nFactors = seasonalityFactors().size();
152  Period factorPeriod(factorFrequency);
153  Size which = 0;
154  if (from==to) {
155  which = 0;
156  } else {
157  // days, weeks, months, years are the only time unit possibilities
158  Integer diffDays = std::abs(to - from); // in days
159  Integer dir = 1;
160  if(from > to)dir = -1;
161  Integer diff;
162  if (factorPeriod.units() == Days) {
163  diff = dir*diffDays;
164  } else if (factorPeriod.units() == Weeks) {
165  diff = dir * (diffDays / 7);
166  } else if (factorPeriod.units() == Months) {
167  std::pair<Date,Date> lim = inflationPeriod(to, factorFrequency);
168  diff = diffDays / (31*factorPeriod.length());
169  Date go = from + dir*diff*factorPeriod;
170  while ( !(lim.first <= go && go <= lim.second) ) {
171  go += dir*factorPeriod;
172  diff++;
173  }
174  diff=dir*diff;
175  } else if (factorPeriod.units() == Years) {
176  QL_FAIL("seasonality period time unit is not allowed to be : " << factorPeriod.units());
177  } else {
178  QL_FAIL("Unknown time unit: " << factorPeriod.units());
179  }
180  // now adjust to the available number of factors, direction dependent
181 
182  if (dir==1) {
183  which = diff % nFactors;
184  } else {
185  which = (nFactors - (-diff % nFactors)) % nFactors;
186  }
187  }
188 
189  return seasonalityFactors()[which];
190  }

193  Rate MultiplicativePriceSeasonality::seasonalityCorrection(Rate rate,
194  const Date& atDate,
195  const DayCounter& dc,
196  const Date& curveBaseDate,
197  const bool isZeroRate) const {
198  // need _two_ corrections in order to get: seasonality = factor[atDate-seasonalityBase] / factor[reference-seasonalityBase]
199  // i.e. for ZERO inflation rates you have the true fixing at the curve base so this factor must be normalized to one
200  // for YoY inflation rates your reference point is the year before
201 
202  Real factorAt = this->seasonalityFactor(atDate);
203 
204  //Getting seasonality correction for either ZC or YoY
205  Rate f;
206  if (isZeroRate) {
207  Rate factorBase = this->seasonalityFactor(curveBaseDate);
208  Real seasonalityAt = factorAt / factorBase;
209  Time timeFromCurveBase = dc.yearFraction(curveBaseDate, atDate);
210  f = std::pow(seasonalityAt, 1/timeFromCurveBase);
211  }
212  else {
213  Rate factor1Ybefore = this->seasonalityFactor(atDate - Period(1,Years));
214  f = factorAt / factor1Ybefore;
215  }
216 
217  return (rate + 1)*f - 1;
218  }

221  Real KerkhofSeasonality::seasonalityFactor(const Date &to) const {
222 
223  Integer dir = 1;
224  Date from = seasonalityBaseDate();
225  Size fromMonth = from.month();
226  Size toMonth = to.month();
227 
228  Period factorPeriod(frequency());
229 
230  if (toMonth < fromMonth)
231  {
232  Size dummy = fromMonth;
233  fromMonth = toMonth;
234  toMonth = dummy;
235  dir = 0; // We calculate invers Factor in loop
236  }
237 
238  QL_REQUIRE(seasonalityFactors().size() == 12 &&
239  factorPeriod.units() == Months,
240  "12 monthly seasonal factors needed for Kerkhof Seasonality:"
241  << " got " << seasonalityFactors().size());
242 
243  Real seasonalCorrection = 1.0;
244  for (Size i = fromMonth ; i<toMonth; i++)
245  {
246  seasonalCorrection *= seasonalityFactors()[i];
247 
248  }
249 
250  if (dir == 0) // invers Factor required
251  {
252  seasonalCorrection = 1/seasonalCorrection;
253  }
254 
255  return seasonalCorrection;
256  }

258  Rate KerkhofSeasonality::seasonalityCorrection(Rate rate,
259  const Date& atDate,
260  const DayCounter& dc,
261  const Date& curveBaseDate,
262  const bool isZeroRate) const {
263 
264  Real indexFactor = this->seasonalityFactor(atDate);
265 
266  // Getting seasonality correction
267  Rate f;
268  if (isZeroRate) {
269  std::pair<Date,Date> lim = inflationPeriod(curveBaseDate, Monthly);
270  Time timeFromCurveBase = dc.yearFraction(lim.first, atDate);
271  f = std::pow(indexFactor, 1/timeFromCurveBase);
272  }
273  else {
274  QL_FAIL("Seasonal Kerkhof model is not defined on YoY rates");
275  }
276 
277  return (rate + 1)*f - 1;
278  }
