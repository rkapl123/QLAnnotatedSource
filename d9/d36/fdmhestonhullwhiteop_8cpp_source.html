32  FdmHestonHullWhiteEquityPart::FdmHestonHullWhiteEquityPart(
33  const ext::shared_ptr<FdmMesher>& mesher,
34  ext::shared_ptr<HullWhite> hwModel,
35  ext::shared_ptr<YieldTermStructure> qTS)
36  : x_(mesher->locations(2)), varianceValues_(0.5 * mesher->locations(1)),
37  dxMap_(FirstDerivativeOp(0, mesher)),
38  dxxMap_(SecondDerivativeOp(0, mesher).mult(0.5 * mesher->locations(1))), mapT_(0, mesher),
39  hwModel_(std::move(hwModel)), mesher_(mesher), qTS_(std::move(qTS)) {
40 
41  // on the boundary s_min and s_max the second derivative
42  // d²V/dS² is zero and due to Ito's Lemma the variance term
43  // in the drift should vanish.
44  const ext::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
45  const FdmLinearOpIterator endIter = layout->end();
46  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
47  ++iter) {
48  if ( iter.coordinates()[0] == 0
49  || iter.coordinates()[0] == layout->dim()[0]-1) {
50  varianceValues_[iter.index()] = 0.0;
51  }
52  }
53  volatilityValues_ = Sqrt(2*varianceValues_);
54  }

56  void FdmHestonHullWhiteEquityPart::setTime(Time t1, Time t2) {
57  const ext::shared_ptr<OneFactorModel::ShortRateDynamics> dynamics =
58  hwModel_->dynamics();
59 
60  const Real phi = 0.5*( dynamics->shortRate(t1, 0.0)
61  + dynamics->shortRate(t2, 0.0));
62 
63  const Rate q = qTS_->forwardRate(t1, t2, Continuous).rate();
64 
65  mapT_.axpyb(x_+phi-varianceValues_-q, dxMap_, dxxMap_, Array());
66  }

72  FdmHestonHullWhiteOp::FdmHestonHullWhiteOp(const ext::shared_ptr<FdmMesher>& mesher,
73  const ext::shared_ptr<HestonProcess>& hestonProcess,
74  const ext::shared_ptr<HullWhiteProcess>& hwProcess,
75  Real equityShortRateCorrelation)
76  : v0_(hestonProcess->v0()), kappa_(hestonProcess->kappa()), theta_(hestonProcess->theta()),
77  sigma_(hestonProcess->sigma()), rho_(hestonProcess->rho()),
78  hwModel_(ext::make_shared<HullWhite>(
79  hestonProcess->riskFreeRate(), hwProcess->a(), hwProcess->sigma())),
80  hestonCorrMap_(
81  SecondOrderMixedDerivativeOp(0, 1, mesher).mult(rho_ * sigma_ * mesher->locations(1))),
82  equityIrCorrMap_(
83  SecondOrderMixedDerivativeOp(0, 2, mesher)
84  .mult(Sqrt(mesher->locations(1)) * hwProcess->sigma() * equityShortRateCorrelation)),
85  dyMap_(SecondDerivativeOp(1U, mesher)
86  .mult(0.5 * sigma_ * sigma_ * mesher->locations(1))
87  .add(FirstDerivativeOp(1, mesher).mult(kappa_ * (theta_ - mesher->locations(1))))),
88  dxMap_(mesher, hwModel_, hestonProcess->dividendYield().currentLink()),
89  hullWhiteOp_(mesher, hwModel_, 2) {
90 
91  QL_REQUIRE( equityShortRateCorrelation*equityShortRateCorrelation
92  + hestonProcess->rho()*hestonProcess->rho() <= 1.0,
93  "correlation matrix has negative eigenvalues");
94  }
