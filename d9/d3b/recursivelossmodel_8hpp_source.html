213  template<class CP>
214  void RecursiveLossModel<CP>::resetModel() {
215  // basket update:
216  notionals_ = basket_->remainingNotionals();
217  notional_ = basket_->remainingNotional();
218  attachAmount_ = basket_->remainingAttachmentAmount();
219  detachAmount_ = basket_->remainingDetachmentAmount();
220  // model parameters:
221  remainingBsktSize_ = notionals_.size();
222 
223  copula_->resetBasket(basket_.currentLink());
224 
225  std::vector<Real> lgdsTmp, lgds;
226  for(Size i=0; i<remainingBsktSize_; ++i)
227  lgds.push_back(notionals_[i]*(1.-copula_->recoveries()[i]));
228  lgdsTmp = lgds;
230  lgds.erase(std::remove(lgds.begin(), lgds.end(), 0.), lgds.end());
231  lossUnit_ = *(std::min_element(lgds.begin(), lgds.end()))
232  / nBuckets_;
233  for(Size i=0; i<remainingBsktSize_; ++i)
234  wk_.push_back(std::floor(lgdsTmp[i]/lossUnit_ + .5));
235  }

257  Real RecursiveLossModel<CP>::percentile(const Date& d,
258  Real percentile) const
259  {
260  std::map<Real, Probability> dist = lossDistribution(d);
261 
262  if(dist.begin()->second >=1.) return dist.begin()->first;
263 
264  // deterministic case (e.g. date requested is todays date)
265  if(dist.size() == 1) return dist.begin()->first;
266 
267  if(percentile == 1.) return dist.rbegin()->second;
268  if(percentile == 0.) return dist.begin()->second;
269  std::map<Real, Probability>::const_iterator itdist = dist.begin();
270  while (itdist->second <= percentile) ++itdist;
271  Real valPlus = itdist->second;
272  Real xPlus = itdist->first;
273  --itdist; //we're never 1st or last, because of tests above
274  Real valMin = itdist->second;
275  Real xMin = itdist->first;
276 
277  // return xPlus-(xPlus-xMin)*(valPlus-percentile)/(valPlus-valMin);
278  Real portfLoss = xPlus-(xPlus-xMin)*(valPlus-percentile)
279  /(valPlus-valMin);
280  return //remainingNotional_ *
281  std::min(std::max(portfLoss - attachAmount_, 0.),
282  detachAmount_ - attachAmount_);
283  }

326  Disposable<std::map<Real, Probability> >
327  RecursiveLossModel<CP>::conditionalLossDistrib(
328  const std::vector<Probability>& pDefDate,
329  //const Date& date,
330  const std::vector<Real>& mktFactor) const
331  {
332  //eq. 10 p.68
333  //attainable losses distribution, recursive algorithm
334  const std::vector<Probability>& uncDefProb = pDefDate;// alias, remove
335 
336  std::map<Real, Probability> pIndepDistrib;
338  pIndepDistrib.insert(std::make_pair(0., 1.));
339  for(Size iName=0; iName<remainingBsktSize_; ++iName) {
340  Probability pDef =
341  copula_->conditionalDefaultProbability(uncDefProb[iName], iName,
342  mktFactor);
344  std::map<Real, Probability> pDistTemp;
345  std::map<Real, Probability>::iterator distIt =
346  pIndepDistrib.begin();
347  while(distIt != pIndepDistrib.end()) {
349  std::map<Real, Probability>::iterator matchIt
350  = pDistTemp.find(distIt->first);
351  if(matchIt != pDistTemp.end()) {
352  matchIt->second += distIt->second * (1.-pDef);
353  }else{
354  pDistTemp.insert(std::make_pair(distIt->first,
355  distIt->second * (1.-pDef)));
356  }
358  matchIt = pDistTemp.find(distIt->first + wk_[iName]);
359  if(matchIt != pDistTemp.end()) {
360  matchIt->second += distIt->second * pDef;
361  }else{
362  pDistTemp.insert(std::make_pair(
363  distIt->first+wk_[iName], distIt->second * pDef));
364  }
365  ++distIt;
366  }
368  pIndepDistrib = pDistTemp;
369  }
370  /* Apply tranche limits now .... mind you this could be done outside*/
372  return pIndepDistrib;
373  }
