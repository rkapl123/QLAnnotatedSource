57  void RiskyBond::performCalculations() const {
58  NPV_ = 0;
59  Date today = Settings::instance().evaluationDate();
60  Date npvDate = calendar_.advance(today, settlementDays_, Days);
61  std::vector<ext::shared_ptr<CashFlow> > cf = cashflows();
62  Date d1 = effectiveDate();
63  for (auto& i : cf) {
64  Date d2 = i->date();
65  if (d2 > npvDate) {
66  d1 = max(npvDate , d1);
67  Date defaultDate = d1 + (d2-d1)/2;
68 
69  Real coupon = i->amount() * defaultTS_->survivalProbability(d2);
70  Real recovery = notional(defaultDate) * recoveryRate_
71  * (defaultTS_->survivalProbability(d1)
72  -defaultTS_->survivalProbability(d2));
73  NPV_ += coupon * yieldTS()->discount(d2);
74  NPV_ += recovery * yieldTS()->discount(defaultDate);
75  }
76  d1 = d2;
77  }
78  valuationDate_ = npvDate;
79  }

106  std::vector<ext::shared_ptr<CashFlow> > RiskyBond::expectedCashflows() {
107  std::vector<ext::shared_ptr<CashFlow> > expected;

109  Date today = Settings::instance().evaluationDate();
110  Date npvDate = calendar_.advance(today, settlementDays_, Days);
111  Date d1 = effectiveDate();
112  for (auto& i : cf) {
113  Date d2 = i->date();
114  if (d2 > npvDate) {
115  d1 = max(npvDate, d1);
116  Date defaultDate = d1 + (d2-d1)/2;
117 
118  Real coupon = i->amount() * defaultTS_->survivalProbability(d2);
119  Real recovery = notional(defaultDate) * recoveryRate_
120  * (defaultTS_->survivalProbability(d1)
121  -defaultTS_->survivalProbability(d2));
122  ext::shared_ptr<CashFlow>
123  flow1(new SimpleCashFlow(coupon, d2));
124  expected.push_back(flow1);
125 
126  ext::shared_ptr<CashFlow>
127  flow2(new SimpleCashFlow(recovery, defaultDate));
128  expected.push_back(flow2);
129  }
130  d1 = d2;
131  }
132  return expected;
133  }

149  // paymentConvention_(paymentConvention),

151  // FIXME: Take paymentConvention into account
152  std::vector<Date> dates = schedule_.dates();
153  Real previousNotional = notionals_.front();

157  notionals_.back());

160  rate_, dayCounter_, dates[i-1], dates[i]));
161  ext::shared_ptr<CashFlow> amortization(new
162  AmortizingPayment(previousNotional - currentNotional, dates[i]));
163  previousNotional = currentNotional;
164 
165  leg_.push_back(interest);

168  leg_.push_back(amortization);
169  redemptionLeg_.push_back(amortization);

173  ext::shared_ptr<CashFlow> redemption(new
