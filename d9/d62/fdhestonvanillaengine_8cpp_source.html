52  FdmSolverDesc FdHestonVanillaEngine::getSolverDesc(Real) const {
53  // 1. Mesher
54  const boost::shared_ptr<HestonProcess> process = model_->process();
55  const Time maturity = process->time(arguments_.exercise->lastDate());
56 
57  // 1.1 The variance mesher
58  const Size tGridMin = 5;
59  const boost::shared_ptr<FdmHestonVarianceMesher> varianceMesher(
60  new FdmHestonVarianceMesher(vGrid_, process,
61  maturity,std::max(tGridMin,tGrid_/50)));
62 
63  // 1.2 The equity mesher
64  const boost::shared_ptr<StrikedTypePayoff> payoff =
65  boost::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
66 
67  boost::shared_ptr<Fdm1dMesher> equityMesher;
68  if (strikes_.empty()) {
69  equityMesher = boost::shared_ptr<Fdm1dMesher>(
70  new FdmBlackScholesMesher(
71  xGrid_,
72  FdmBlackScholesMesher::processHelper(
73  process->s0(), process->dividendYield(),
74  process->riskFreeRate(), varianceMesher->volaEstimate()),
75  maturity, payoff->strike(),
76  Null<Real>(), Null<Real>(), 0.0001, 2.0,
77  std::pair<Real, Real>(payoff->strike(), 0.1),
78  arguments_.cashFlow));
79  }
80  else {
81  QL_REQUIRE(arguments_.cashFlow.empty(),"multiple strikes engine "
82  "does not work with discrete dividends");
83  equityMesher = boost::shared_ptr<Fdm1dMesher>(
84  new FdmBlackScholesMultiStrikeMesher(
85  xGrid_,
86  FdmBlackScholesMesher::processHelper(
87  process->s0(), process->dividendYield(),
88  process->riskFreeRate(), varianceMesher->volaEstimate()),
89  maturity, strikes_, 0.0001, 1.5,
90  std::pair<Real, Real>(payoff->strike(), 0.075)));
91  }
92 
93  const boost::shared_ptr<FdmMesher> mesher(
94  new FdmMesherComposite(equityMesher, varianceMesher));
95 
96  // 2. Calculator
97  const boost::shared_ptr<FdmInnerValueCalculator> calculator(
98  new FdmLogInnerValue(arguments_.payoff, mesher, 0));
99 
100  // 3. Step conditions
101  const boost::shared_ptr<FdmStepConditionComposite> conditions =
102  FdmStepConditionComposite::vanillaComposite(
103  arguments_.cashFlow, arguments_.exercise,
104  mesher, calculator,
105  process->riskFreeRate()->referenceDate(),
106  process->riskFreeRate()->dayCounter());
107 
108  // 4. Boundary conditions
109  const FdmBoundaryConditionSet boundaries;
110 
111  // 5. Solver
112  FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
113  calculator, maturity,
114  tGrid_, dampingSteps_ };
115 
116  return solverDesc;
117  }

119  void FdHestonVanillaEngine::calculate() const {
120 
121  // cache lookup for precalculated results
122  for (Size i=0; i < cachedArgs2results_.size(); ++i) {
123  if ( cachedArgs2results_[i].first.exercise->type()
124  == arguments_.exercise->type()
125  && cachedArgs2results_[i].first.exercise->dates()
126  == arguments_.exercise->dates()) {
127  boost::shared_ptr<PlainVanillaPayoff> p1 =
128  boost::dynamic_pointer_cast<PlainVanillaPayoff>(
129  arguments_.payoff);
130  boost::shared_ptr<PlainVanillaPayoff> p2 =
131  boost::dynamic_pointer_cast<PlainVanillaPayoff>(
132  cachedArgs2results_[i].first.payoff);
133 
134  if (p1 && p1->strike() == p2->strike()
135  && p1->optionType() == p2->optionType()) {
136  QL_REQUIRE(arguments_.cashFlow.empty(),
137  "multiple strikes engine does "
138  "not work with discrete dividends");
139  results_ = cachedArgs2results_[i].second;
140  return;
141  }
142  }
143  }
144 
145  const boost::shared_ptr<HestonProcess> process = model_->process();
146 
147  boost::shared_ptr<FdmHestonSolver> solver(new FdmHestonSolver(
148  Handle<HestonProcess>(process),
149  getSolverDesc(1.5), schemeDesc_,
150  Handle<FdmQuantoHelper>(), leverageFct_));
151 
152  const Real v0 = process->v0();
153  const Real spot = process->s0()->value();
154 
155  results_.value = solver->valueAt(spot, v0);
156  results_.delta = solver->deltaAt(spot, v0);
157  results_.gamma = solver->gammaAt(spot, v0);
158  results_.theta = solver->thetaAt(spot, v0);
159 
160  cachedArgs2results_.resize(strikes_.size());
161  const boost::shared_ptr<StrikedTypePayoff> payoff =
162  boost::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
163  for (Size i=0; i < strikes_.size(); ++i) {
164  cachedArgs2results_[i].first.exercise = arguments_.exercise;
165  cachedArgs2results_[i].first.payoff =
166  boost::shared_ptr<PlainVanillaPayoff>(
167  new PlainVanillaPayoff(payoff->optionType(), strikes_[i]));
168  const Real d = payoff->strike()/strikes_[i];
169 
170  DividendVanillaOption::results&
171  results = cachedArgs2results_[i].second;
172  results.value = solver->valueAt(spot*d, v0)/d;
173  results.delta = solver->deltaAt(spot*d, v0);
174  results.gamma = solver->gammaAt(spot*d, v0)*d;
175  results.theta = solver->thetaAt(spot*d, v0)/d;
176  }
177  }
