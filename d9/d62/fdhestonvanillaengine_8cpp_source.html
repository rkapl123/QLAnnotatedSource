70 FdmSolverDesc FdHestonVanillaEngine::getSolverDesc(Real) const {
71 // 1. Mesher
72 const ext::shared_ptr<HestonProcess> process = model_->process();
73 const Time maturity = process->time(arguments_.exercise->lastDate());
74
75 // 1.1 The variance mesher
76 const Size tGridMin = 5;
77 const Size tGridAvgSteps = std::max(tGridMin, tGrid_/50);
78 const ext::shared_ptr<FdmHestonLocalVolatilityVarianceMesher> vMesher
79 = ext::make_shared<FdmHestonLocalVolatilityVarianceMesher>(
80 vGrid_, process, leverageFct_, maturity, tGridAvgSteps, 0.0001, mixingFactor_);
81
82 const Volatility avgVolaEstimate = vMesher->volaEstimate();
83
84 // 1.2 The equity mesher
85 const ext::shared_ptr<StrikedTypePayoff> payoff =
86 ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
87
88 ext::shared_ptr<Fdm1dMesher> equityMesher;
89 if (strikes_.empty()) {
90 equityMesher = ext::shared_ptr<Fdm1dMesher>(
91 new FdmBlackScholesMesher(
92 xGrid_,
93 FdmBlackScholesMesher::processHelper(
94 process->s0(), process->dividendYield(),
95 process->riskFreeRate(), avgVolaEstimate),
96 maturity, payoff->strike(),
97 Null<Real>(), Null<Real>(), 0.0001, 2.0,
98 std::pair<Real, Real>(payoff->strike(), 0.1),
99 arguments_.cashFlow,
100 quantoHelper_));
101 }
102 else {
103 QL_REQUIRE(arguments_.cashFlow.empty(),"multiple strikes engine "
104 "does not work with discrete dividends");
105 equityMesher = ext::shared_ptr<Fdm1dMesher>(
106 new FdmBlackScholesMultiStrikeMesher(
107 xGrid_,
108 FdmBlackScholesMesher::processHelper(
109 process->s0(), process->dividendYield(),
110 process->riskFreeRate(), avgVolaEstimate),
111 maturity, strikes_, 0.0001, 1.5,
112 std::pair<Real, Real>(payoff->strike(), 0.075)));
113 }
114
115 const ext::shared_ptr<FdmMesher> mesher(
116 new FdmMesherComposite(equityMesher, vMesher));
117
118 // 2. Calculator
119 const ext::shared_ptr<FdmInnerValueCalculator> calculator(
120 new FdmLogInnerValue(arguments_.payoff, mesher, 0));
121
122 // 3. Step conditions
123 const ext::shared_ptr<FdmStepConditionComposite> conditions =
124 FdmStepConditionComposite::vanillaComposite(
125 arguments_.cashFlow, arguments_.exercise,
126 mesher, calculator,
127 process->riskFreeRate()->referenceDate(),
128 process->riskFreeRate()->dayCounter());
129
130 // 4. Boundary conditions
131 const FdmBoundaryConditionSet boundaries;
132
133 // 5. Solver
134 FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
135 calculator, maturity,
136 tGrid_, dampingSteps_ };
137
138 return solverDesc;
139 }

141 void FdHestonVanillaEngine::calculate() const {
142
143 // cache lookup for precalculated results
144 for (auto& cachedArgs2result : cachedArgs2results_) {
145 if (cachedArgs2result.first.exercise->type() == arguments_.exercise->type() &&
146 cachedArgs2result.first.exercise->dates() == arguments_.exercise->dates()) {
147 ext::shared_ptr<PlainVanillaPayoff> p1 =
148 ext::dynamic_pointer_cast<PlainVanillaPayoff>(
149 arguments_.payoff);
150 ext::shared_ptr<PlainVanillaPayoff> p2 =
151 ext::dynamic_pointer_cast<PlainVanillaPayoff>(cachedArgs2result.first.payoff);
152
153 if ((p1 != nullptr) && p1->strike() == p2->strike() &&
154 p1->optionType() == p2->optionType()) {
155 QL_REQUIRE(arguments_.cashFlow.empty(),
156 "multiple strikes engine does "
157 "not work with discrete dividends");
158 results_ = cachedArgs2result.second;
159 return;
160 }
161 }
162 }
163
164 const ext::shared_ptr<HestonProcess> process = model_->process();
165
166 ext::shared_ptr<FdmHestonSolver> solver(new FdmHestonSolver(
167 Handle<HestonProcess>(process),
168 getSolverDesc(1.5), schemeDesc_,
169 Handle<FdmQuantoHelper>(quantoHelper_), leverageFct_,
170 mixingFactor_));
171
172 const Real v0 = process->v0();
173 const Real spot = process->s0()->value();
174
175 results_.value = solver->valueAt(spot, v0);
176 results_.delta = solver->deltaAt(spot, v0);
177 results_.gamma = solver->gammaAt(spot, v0);
178 results_.theta = solver->thetaAt(spot, v0);
179
180 cachedArgs2results_.resize(strikes_.size());
181 const ext::shared_ptr<StrikedTypePayoff> payoff =
182 ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
183 for (Size i=0; i < strikes_.size(); ++i) {
184 cachedArgs2results_[i].first.exercise = arguments_.exercise;
185 cachedArgs2results_[i].first.payoff =
186 ext::make_shared<PlainVanillaPayoff>(
187 payoff->optionType(), strikes_[i]);
188 const Real d = payoff->strike()/strikes_[i];
189
190 DividendVanillaOption::results&
191 results = cachedArgs2results_[i].second;
192 results.value = solver->valueAt(spot*d, v0)/d;
193 results.delta = solver->deltaAt(spot*d, v0);
194 results.gamma = solver->gammaAt(spot*d, v0)*d;
195 results.theta = solver->thetaAt(spot*d, v0)/d;
196 }
197 }
