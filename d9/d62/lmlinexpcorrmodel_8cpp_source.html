59 void LmLinearExponentialCorrelationModel::generateArguments() {
60 const Real rho = arguments_[0](0.0);
61 const Real beta= arguments_[1](0.0);
62
63 for (Size i=0; i<size_; ++i) {
64 for (Size j=i; j<size_; ++j) {
65 corrMatrix_[i][j] = corrMatrix_[j][i]
66 = rho + (1-rho)*std::exp(-beta*std::fabs(Real(i)-Real(j)));
67 }
68 }
69
70 pseudoSqrt_ = rankReducedSqrt(corrMatrix_, factors_,
71 1.0, SalvagingAlgorithm::None);
72
73 corrMatrix_ = pseudoSqrt_ * transpose(pseudoSqrt_);
74 }
75}
