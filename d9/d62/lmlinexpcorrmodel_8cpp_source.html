61  void LmLinearExponentialCorrelationModel::generateArguments() {
62  const Real rho = arguments_[0](0.0);
63  const Real beta= arguments_[1](0.0);
64 
65  for (Size i=0; i<size_; ++i) {
66  for (Size j=i; j<size_; ++j) {
67  corrMatrix_[i][j] = corrMatrix_[j][i]
68  = rho + (1-rho)*std::exp(-beta*std::fabs(Real(i)-Real(j)));
69  }
70  }
71 
72  pseudoSqrt_ = rankReducedSqrt(corrMatrix_, factors_,
73  1.0, SalvagingAlgorithm::None);
74 
75  corrMatrix_ = pseudoSqrt_ * transpose(pseudoSqrt_);
76  }
77 }
