53 Gaussian1dSmileSection::Gaussian1dSmileSection(
54  const Date &fixingDate, const boost::shared_ptr<IborIndex> &iborIndex,
55  const boost::shared_ptr<Gaussian1dModel> &model,
56  const DayCounter &dc,
57  const boost::shared_ptr<Gaussian1dCapFloorEngine> capEngine)
58  : SmileSection(fixingDate, dc, model->termStructure()->referenceDate()),
59  fixingDate_(fixingDate), swapIndex_(boost::shared_ptr<SwapIndex>()),
60  iborIndex_(iborIndex), model_(model), engine_(capEngine) {
61 
62  atm_ = model_->forwardRate(fixingDate_, Null<Date>(), 0.0, iborIndex_);
63  CapFloor c = MakeCapFloor(CapFloor::Cap, iborIndex_->tenor(), iborIndex_,
64  Null<Real>(),
65  0 * Days).withEffectiveDate(fixingDate_, false);
66  annuity_ =
67  iborIndex_->dayCounter().yearFraction(c.startDate(), c.maturityDate()) *
68  model_->zerobond(c.maturityDate());
69 
70  if (engine_ == NULL) {
71  engine_ = boost::make_shared<Gaussian1dCapFloorEngine>(
72  model_, 64, 7.0, true,
73  false); // use model curve as discounting curve
74  }
75 }

79 Real Gaussian1dSmileSection::optionPrice(Rate strike, Option::Type type,
80  Real discount) const {
81 
82  if (swapIndex_ != NULL) {
83  Swaption s = MakeSwaption(swapIndex_, fixingDate_, strike)
84  .withUnderlyingType(type == Option::Call
85  ? VanillaSwap::Payer
86  : VanillaSwap::Receiver)
87  .withPricingEngine(engine_);
88  Real tmp = s.NPV();
89  return tmp / annuity_ * discount;
90  } else {
91  CapFloor c =
92  MakeCapFloor(type == Option::Call ? CapFloor::Cap : CapFloor::Floor,
93  iborIndex_->tenor(), iborIndex_, strike, 0 * Days)
94  .withEffectiveDate(fixingDate_, false)
95  .withPricingEngine(engine_);
96  Real tmp = c.NPV();
97  return tmp / annuity_ * discount;
98  }
99 }

101 Real Gaussian1dSmileSection::volatilityImpl(Rate strike) const {
102  Real vol = 0.0;
103  try {
104  Option::Type type = strike >= atm_ ? Option::Call : Option::Put;
105  Real o = optionPrice(strike, type);
106  vol = blackFormulaImpliedStdDev(type, strike, atm_, o) /
107  sqrt(exerciseTime());
108  } catch (...) {
109  }
110  return vol;
111 }
