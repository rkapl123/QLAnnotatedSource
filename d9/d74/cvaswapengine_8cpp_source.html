108 void CounterpartyAdjSwapEngine::calculate() const {
109 /* both DTS, YTS ref dates and pricing date consistency
110 checks? settlement... */
111 QL_REQUIRE(!discountCurve_.empty(),
112 "no discount term structure set");
113 QL_REQUIRE(!defaultTS_.empty(),
114 "no ctpty default term structure set");
115 QL_REQUIRE(!swaptionletEngine_.empty(),
116 "no swap option engine set");
117
118 Date priceDate = defaultTS_->referenceDate();
119
120 Real cumOptVal = 0.,
121 cumPutVal = 0.;
122 // Vanilla swap so 0 leg is floater
123
124 std::vector<Date>::const_iterator nextFD =
125 arguments_.fixedPayDates.begin();
126 Date swapletStart = priceDate;
127 while (*nextFD < priceDate) ++nextFD;
128
129 // Compute fair spread for strike value:
130 // copy args into the non risky engine
131 auto* noCVAArgs = dynamic_cast<Swap::arguments*>(baseSwapEngine_->getArguments());
132 QL_REQUIRE(noCVAArgs != nullptr, "wrong argument type");
133
134 noCVAArgs->legs = this->arguments_.legs;
135 noCVAArgs->payer = this->arguments_.payer;
136
137 baseSwapEngine_->calculate();
138
139 ext::shared_ptr<FixedRateCoupon> coupon = ext::dynamic_pointer_cast<FixedRateCoupon>(arguments_.legs[0][0]);
140 QL_REQUIRE(coupon,"dynamic cast of fixed leg coupon failed.");
141 Rate baseSwapRate = coupon->rate();
142
143 const auto* vSResults = dynamic_cast<const Swap::results*>(baseSwapEngine_->getResults());
144 QL_REQUIRE(vSResults != nullptr, "wrong result type");
145
146 Rate baseSwapFairRate = -baseSwapRate * vSResults->legNPV[1] /
147 vSResults->legNPV[0];
148 Real baseSwapNPV = vSResults->value;
149
150 Swap::Type reversedType = arguments_.type == Swap::Payer ? Swap::Receiver : Swap::Payer;
151
152 // Swaplet options summatory:
153 while(nextFD != arguments_.fixedPayDates.end()) {
154 // iFD coupon not fixed, create swaptionlet:
155 ext::shared_ptr<FloatingRateCoupon> floatCoupon = ext::dynamic_pointer_cast<FloatingRateCoupon>(arguments_.legs[1][0]);
156 QL_REQUIRE(floatCoupon,"dynamic cast of floating leg coupon failed.");
157 ext::shared_ptr<IborIndex> swapIndex = ext::dynamic_pointer_cast<IborIndex>(floatCoupon->index());
158 QL_REQUIRE(swapIndex,"dynamic cast of floating leg index failed.");
159
160 // Alternatively one could cap this period to, say, 1M
161 // Period swapPeriod = ext::dynamic_pointer_cast<FloatingRateCoupon>(
162 // arguments_.legs[1][0])->index()->tenor();
163
164 Period baseSwapsTenor(arguments_.fixedPayDates.back().serialNumber()
165 - swapletStart.serialNumber(), Days);
166 ext::shared_ptr<VanillaSwap> swaplet = MakeVanillaSwap(
167 baseSwapsTenor,
168 swapIndex,
169 baseSwapFairRate // strike
170 )
171 .withType(arguments_.type)
172 .withNominal(arguments_.nominal)
174 .withEffectiveDate(swapletStart)
175 .withTerminationDate(arguments_.fixedPayDates.back());
176 ext::shared_ptr<VanillaSwap> revSwaplet = MakeVanillaSwap(
177 baseSwapsTenor,
178 swapIndex,
179 baseSwapFairRate // strike
180 )
181 .withType(reversedType)
182 .withNominal(arguments_.nominal)
184 .withEffectiveDate(swapletStart)
185 .withTerminationDate(arguments_.fixedPayDates.back());
186
187 Swaption swaptionlet(swaplet,
188 ext::make_shared<EuropeanExercise>(swapletStart));
189 Swaption putSwaplet(revSwaplet,
190 ext::make_shared<EuropeanExercise>(swapletStart));
191 swaptionlet.setPricingEngine(swaptionletEngine_.currentLink());
192 putSwaplet.setPricingEngine(swaptionletEngine_.currentLink());
193
194 // atm underlying swap means that the value of put = value
195 // call so this double pricing is not needed
196 cumOptVal += swaptionlet.NPV() * defaultTS_->defaultProbability(
197 swapletStart, *nextFD);
198 cumPutVal += putSwaplet.NPV() * invstDTS_->defaultProbability(
199 swapletStart, *nextFD);
200
201 swapletStart = *nextFD;
202 ++nextFD;
203 }
204
205 results_.value = baseSwapNPV - (1.-ctptyRecoveryRate_) * cumOptVal
206 + (1.-invstRecoveryRate_) * cumPutVal;
207
208 results_.fairRate = -baseSwapRate * (vSResults->legNPV[1]
209 - (1.-ctptyRecoveryRate_) * cumOptVal +
210 (1.-invstRecoveryRate_) * cumPutVal )
211 / vSResults->legNPV[0];
212
213 }
