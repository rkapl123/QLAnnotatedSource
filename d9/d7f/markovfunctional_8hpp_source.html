156  void validate() {
157 
158  if ((adjustments_ & KahaleInterpolation) != 0)
159  addAdjustment(KahaleSmile);
160 
161  if ((adjustments_ & KahaleSmile) != 0 &&
162  (adjustments_ & SmileDeleteArbitragePoints) != 0) {
163  addAdjustment(KahaleInterpolation);
164  }
165 
166  QL_REQUIRE((adjustments_ & SabrSmile) == 0 ||
167  (adjustments_ & KahaleSmile) == 0 ||
168  (adjustments_ & CustomSmile) == 0
169  ,
170  "Only one of KahaleSmile, SabrSmile and CustomSmile"
171  "can be specified at the same time");
172  QL_REQUIRE(yGridPoints_ > 0, "At least one grid point ("
173  << yGridPoints_
174  << ") for the state process "
175  "discretization must be "
176  "given");
177  QL_REQUIRE(yStdDevs_ > 0.0,
178  "Multiple of standard deviations covered by state "
179  "process discretization ("
180  << yStdDevs_ << ") must be positive");
181  QL_REQUIRE(gaussHermitePoints_ > 0,
182  "Number of gauss hermite integration points ("
183  << gaussHermitePoints_ << ") must be positive");
184  QL_REQUIRE(digitalGap_ > 0.0, "Digital gap ("
185  << digitalGap_
186  << ") must be positive");
187  QL_REQUIRE(marketRateAccuracy_ > 0.0,
188  "Market rate accuracy (" << marketRateAccuracy_
189  << ") must be positive");
190  QL_REQUIRE(
191  (adjustments_ & KahaleSmile) == 0 || lowerRateBound_ == 0.0,
192  "If Kahale extrapolation is used, the lower rate bound ("
193  << lowerRateBound_ << ") must be zero.");
194  QL_REQUIRE(
195  lowerRateBound_ < upperRateBound_,
196  "Lower rate bound ("
197  << lowerRateBound_
198  << ") must be strictly less than upper rate bound ("
199  << upperRateBound_ << ")");
200  QL_REQUIRE(((adjustments_ & CustomSmile) == 0) ||
201  customSmileFactory_,
202  "missing CustomSmileFactoy");
203  }
