134  void validate() {
135 
136  if ((adjustments_ & KahaleInterpolation) != 0)
137  addAdjustment(KahaleSmile);
138 
139  if ((adjustments_ & KahaleSmile) != 0 &&
140  (adjustments_ & SmileDeleteArbitragePoints) != 0) {
141  addAdjustment(KahaleInterpolation);
142  }
143 
144  QL_REQUIRE((adjustments_ & SabrSmile) == 0 ||
145  (adjustments_ & KahaleSmile) == 0,
146  "KahaleSmile and SabrSmile can not specified at the "
147  "same time");
148  QL_REQUIRE(yGridPoints_ > 0, "At least one grid point ("
149  << yGridPoints_
150  << ") for the state process "
151  "discretization must be "
152  "given");
153  QL_REQUIRE(yStdDevs_ > 0.0,
154  "Multiple of standard deviations covered by state "
155  "process discretization ("
156  << yStdDevs_ << ") must be positive");
157  QL_REQUIRE(gaussHermitePoints_ > 0,
158  "Number of gauss hermite integration points ("
159  << gaussHermitePoints_ << ") must be positive");
160  QL_REQUIRE(digitalGap_ > 0.0, "Digital gap ("
161  << digitalGap_
162  << ") must be positive");
163  QL_REQUIRE(marketRateAccuracy_ > 0.0,
164  "Market rate accuracy (" << marketRateAccuracy_
165  << ") must be positive");
166  QL_REQUIRE(
167  (adjustments_ & KahaleSmile) == 0 || lowerRateBound_ == 0.0,
168  "If Kahale extrapolation is used, the lower rate bound ("
169  << lowerRateBound_ << ") must be zero.");
170  QL_REQUIRE(
171  lowerRateBound_ < upperRateBound_,
172  "Lower rate bound ("
173  << lowerRateBound_
174  << ") must be strictly less than upper rate bound ("
175  << upperRateBound_ << ")");
176  }
