97  template <template <class> class Scheme>
98  void FDMultiPeriodEngine<Scheme>::calculate(
99  PricingEngine::results* r) const {
100  OneAssetOption::results *results =
101  dynamic_cast<OneAssetOption::results *>(r);
102  QL_REQUIRE(results, "incorrect argument type");
103  Time beginDate, endDate;
104  Size dateNumber = stoppingTimes_.size();
105  bool lastDateIsResTime = false;
106  Integer firstIndex = -1;
107  Integer lastIndex = static_cast<Integer>(dateNumber) - 1;
108  bool firstDateIsZero = false;
109  Time firstNonZeroDate = getResidualTime();
110 
111  Real dateTolerance = 1e-6;
112 
113  if (dateNumber > 0) {
114  QL_REQUIRE(getDividendTime(0) >= 0,
115  "first date (" << getDividendTime(0)
116  << ") cannot be negative");
117  if(getDividendTime(0) < getResidualTime() * dateTolerance ){
118  firstDateIsZero = true;
119  firstIndex = 0;
120  if(dateNumber >= 2)
121  firstNonZeroDate = getDividendTime(1);
122  }
123 
124  if (std::fabs(getDividendTime(lastIndex) - getResidualTime())
125  < dateTolerance) {
126  lastDateIsResTime = true;
127  lastIndex = Integer(dateNumber) - 2;
128  }
129 
130  if (!firstDateIsZero)
131  firstNonZeroDate = getDividendTime(0);
132 
133  if (dateNumber >= 2) {
134  for (Size j = 1; j < dateNumber; j++)
135  QL_REQUIRE(getDividendTime(j-1) < getDividendTime(j),
136  "dates must be in increasing order: "
137  << getDividendTime(j-1)
138  << " is not strictly smaller than "
139  << getDividendTime(j));
140  }
141  }
142 
143  Time dt = getResidualTime()/(timeStepPerPeriod_*(dateNumber+1));
144 
145  // Ensure that dt is always smaller than the first non-zero date
146  if (firstNonZeroDate <= dt)
147  dt = firstNonZeroDate/2.0;
148 
149  setGridLimits();
150  initializeInitialCondition();
151  initializeOperator();
152  initializeBoundaryConditions();
153  initializeModel();
154  initializeStepCondition();
155 
156  prices_ = intrinsicValues_;
157  if(lastDateIsResTime)
158  executeIntermediateStep(dateNumber - 1);
159 
160  Integer j = lastIndex;
161  do {
162  if (j == Integer(dateNumber) - 1)
163  beginDate = getResidualTime();
164  else
165  beginDate = getDividendTime(j+1);
166 
167  if (j >= 0)
168  endDate = getDividendTime(j);
169  else
170  endDate = dt;
171 
172  model_->rollback(prices_.values(),
173  beginDate, endDate,
174  timeStepPerPeriod_, *stepCondition_);
175  if (j >= 0)
176  executeIntermediateStep(j);
177  } while (--j >= firstIndex);
178 
179  model_->rollback(prices_.values(), dt, 0, 1, *stepCondition_);
180 
181  if(firstDateIsZero)
182  executeIntermediateStep(0);
183 
184  results->value = prices_.valueAtCenter();
185  results->delta = prices_.firstDerivativeAtCenter();
186  results->gamma = prices_.secondDerivativeAtCenter();
187  results->additionalResults["priceCurve"] = prices_;
188  }
