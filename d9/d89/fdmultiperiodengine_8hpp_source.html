50 virtual void setupArguments(

52 const std::vector<ext::shared_ptr<Event> >& schedule) const {
53 FDVanillaEngine::setupArguments(args);
54 events_ = schedule;
55 stoppingTimes_.clear();
56 Size n = schedule.size();
57 stoppingTimes_.reserve(n);
58 for (Size i=0; i<n; ++i)
59 stoppingTimes_.push_back(process_->time(events_[i]->date()));

62 void setupArguments(const PricingEngine::arguments* a) const override {
63 FDVanillaEngine::setupArguments(a);
64 const auto* args = dynamic_cast<const OneAssetOption::arguments*>(a);
65 QL_REQUIRE(args, "incorrect argument type");
66 events_.clear();
67
68 Size n = args->exercise->dates().size();
69 stoppingTimes_.resize(n);
70 for (Size i=0; i<n; ++i)
71 stoppingTimes_[i] =
72 process_->time(args->exercise->date(i));
73 }

96 template <template <class> class Scheme>
97 void FDMultiPeriodEngine<Scheme>::calculate(
98 PricingEngine::results* r) const {
99 auto* results = dynamic_cast<OneAssetOption::results*>(r);
100 QL_REQUIRE(results, "incorrect argument type");
101 Time beginDate, endDate;
102 Size dateNumber = stoppingTimes_.size();
103 bool lastDateIsResTime = false;
104 Integer firstIndex = -1;
105 Integer lastIndex = static_cast<Integer>(dateNumber) - 1;
106 bool firstDateIsZero = false;
107 Time firstNonZeroDate = getResidualTime();
108
109 Real dateTolerance = 1e-6;
110
111 if (dateNumber > 0) {
112 QL_REQUIRE(getDividendTime(0) >= 0,
113 "first date (" << getDividendTime(0)
114 << ") cannot be negative");
115 if(getDividendTime(0) < getResidualTime() * dateTolerance ){
116 firstDateIsZero = true;
117 firstIndex = 0;
118 if(dateNumber >= 2)
119 firstNonZeroDate = getDividendTime(1);
120 }
121
122 if (std::fabs(getDividendTime(lastIndex) - getResidualTime())
123 < dateTolerance) {
124 lastDateIsResTime = true;
125 lastIndex = Integer(dateNumber) - 2;
126 }
127
128 if (!firstDateIsZero)
129 firstNonZeroDate = getDividendTime(0);
130
131 if (dateNumber >= 2) {
132 for (Size j = 1; j < dateNumber; j++)
133 QL_REQUIRE(getDividendTime(j-1) < getDividendTime(j),
134 "dates must be in increasing order: "
135 << getDividendTime(j-1)
136 << " is not strictly smaller than "
137 << getDividendTime(j));
138 }
139 }
140
141 Time dt = getResidualTime()/(timeStepPerPeriod_*(dateNumber+1));
142
143 // Ensure that dt is always smaller than the first non-zero date
144 if (firstNonZeroDate <= dt)
145 dt = firstNonZeroDate/2.0;
146
147 setGridLimits();
148 initializeInitialCondition();
149 initializeOperator();
150 initializeBoundaryConditions();
151 initializeModel();
152 initializeStepCondition();
153
154 prices_ = intrinsicValues_;
155 if(lastDateIsResTime)
156 executeIntermediateStep(dateNumber - 1);
157
158 Integer j = lastIndex;
159 do {
160 if (j == Integer(dateNumber) - 1)
161 beginDate = getResidualTime();
162 else
163 beginDate = getDividendTime(j+1);
164
165 if (j >= 0)
166 endDate = getDividendTime(j);
167 else
168 endDate = dt;
169
170 model_->rollback(prices_.values(),
171 beginDate, endDate,
172 timeStepPerPeriod_, *stepCondition_);
173 if (j >= 0)
174 executeIntermediateStep(j);
175 } while (--j >= firstIndex);
176
177 model_->rollback(prices_.values(), dt, 0, 1, *stepCondition_);
178
179 if(firstDateIsZero)
180 executeIntermediateStep(0);
181
182 results->value = prices_.valueAtCenter();
183 results->delta = prices_.firstDerivativeAtCenter();
184 results->gamma = prices_.secondDerivativeAtCenter();
185 results->additionalResults["priceCurve"] = prices_;
186 }
