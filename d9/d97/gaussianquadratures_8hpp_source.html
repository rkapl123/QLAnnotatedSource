59  Real sum = 0.0;

212  class TabulatedGaussLegendre {
213  public:
214  explicit TabulatedGaussLegendre(Size n = 20) { order(n); }
215  template <class F>
216  Real operator() (const F& f) const {
217  QL_ASSERT(w_ != nullptr, "Null weights");
218  QL_ASSERT(x_ != nullptr, "Null abscissas");
219  Size startIdx;
220  Real val;
221 
222  const Size isOrderOdd = order_ & 1;
223 
224  if (isOrderOdd) {
225  QL_ASSERT((n_>0), "assume at least 1 point in quadrature");
226  val = w_[0]*f(x_[0]);
227  startIdx=1;
228  } else {
229  val = 0.0;
230  startIdx=0;
231  }
232 
233  for (Size i=startIdx; i<n_; ++i) {
234  val += w_[i]*f( x_[i]);
235  val += w_[i]*f(-x_[i]);
236  }
237  return val;
238  }
