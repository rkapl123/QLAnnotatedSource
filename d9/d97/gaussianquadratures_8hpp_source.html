203  class TabulatedGaussLegendre {
204  public:
205  explicit TabulatedGaussLegendre(Size n = 20) { order(n); }
206  template <class F>
207  Real operator() (const F& f) const {
208  QL_ASSERT(w_!=0, "Null weights" );
209  QL_ASSERT(x_!=0, "Null abscissas");
210  Size startIdx;
211  Real val;
212 
213  const Size isOrderOdd = order_ & 1;
214 
215  if (isOrderOdd) {
216  QL_ASSERT((n_>0), "assume at least 1 point in quadrature");
217  val = w_[0]*f(x_[0]);
218  startIdx=1;
219  } else {
220  val = 0.0;
221  startIdx=0;
222  }
223 
224  for (Size i=startIdx; i<n_; ++i) {
225  val += w_[i]*f( x_[i]);
226  val += w_[i]*f(-x_[i]);
227  }
228  return val;
229  }
