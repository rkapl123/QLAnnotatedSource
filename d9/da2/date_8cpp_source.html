62  Date::Date(Day d, Month m, Year y) {
63  QL_REQUIRE(y > 1900 && y < 2200,

78  Month Date::month() const {
79  Day d = dayOfYear(); // dayOfYear is 1 based
80  Integer m = d/30 + 1;
81  bool leap = isLeap(year());
82  while (d <= monthOffset(Month(m),leap))
83  --m;
84  while (d > monthOffset(Month(m+1),leap))
85  ++m;
86  return Month(m);

135  Date Date::advance(const Date& date, Integer n, TimeUnit units) {
136  switch (units) {
137  case Days:
138  return date + n;
139  case Weeks:
140  return date + 7*n;
141  case Months: {
142  Day d = date.dayOfMonth();
143  Integer m = Integer(date.month())+n;
144  Year y = date.year();
145  while (m > 12) {
146  m -= 12;
147  y += 1;
148  }
149  while (m < 1) {
150  m += 12;
151  y -= 1;
152  }
153 
154  QL_ENSURE(y >= 1900 && y <= 2199,
155  "year " << y << " out of bounds. "
156  << "It must be in [1901,2199]");
157 
158  Integer length = monthLength(Month(m), isLeap(y));
159  if (d > length)
160  d = length;
161 
162  return Date(d, Month(m), y);
163  }
164  case Years: {
165  Day d = date.dayOfMonth();
166  Month m = date.month();
167  Year y = date.year()+n;
168 
169  QL_ENSURE(y >= 1900 && y <= 2199,
170  "year " << y << " out of bounds. "
171  << "It must be in [1901,2199]");
172 
173  if (d == 29 && m == February && !isLeap(y))
174  d = 28;
175 
176  return Date(d,m,y);
177  }
178  default:
179  QL_FAIL("undefined time units");
180  }

450  void advance(ptime& dt, Integer n, TimeUnit units) {
451  using boost::gregorian::gregorian_calendar;
452 
453  switch (units) {
454  case Days:
455  dt += boost::gregorian::days(n);
456  break;
457  case Weeks:
458  dt += boost::gregorian::weeks(n);
459  break;
460  case Months:
461  case Years : {
462  const boost::gregorian::date date = dt.date();
463  const Day eoM = gregorian_calendar::end_of_month_day(
464  date.year(), date.month());
465 
466  if (units == Months) {
467  dt += boost::gregorian::months(n);
468  }
469  else {
470  dt += boost::gregorian::years(n);
471  }
472 
473  if (date.day() == eoM) {
474  // avoid snap-to-end-of-month
475  // behavior of boost::date_time
476  const Day newEoM
477  = gregorian_calendar::end_of_month_day(
478  dt.date().year(), dt.date().month());
479 
480  if (newEoM > eoM) {
481  dt -= boost::gregorian::days(newEoM - eoM);
482  }
483  }
484  }
485  break;
486  default:
487  QL_FAIL("undefined time units");
488  }
489  }
