32  void AnalyticDividendEuropeanEngine::calculate() const {
33 
34  QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
35  "not an European option");
36 
37  ext::shared_ptr<StrikedTypePayoff> payoff =
38  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
39  QL_REQUIRE(payoff, "non-striked payoff given");
40 
41  Date settlementDate = process_->riskFreeRate()->referenceDate();
42  Real riskless = 0.0;
43  Size i;
44  for (i=0; i<arguments_.cashFlow.size(); i++) {
45  const Date cashFlowDate = arguments_.cashFlow[i]->date();
46 
47  if ( cashFlowDate >= settlementDate
48  && cashFlowDate <= arguments_.exercise->lastDate()) {
49 
50  riskless += arguments_.cashFlow[i]->amount() *
51  process_->riskFreeRate()->discount(cashFlowDate) /
52  process_->dividendYield()->discount(cashFlowDate);
53  }
54  }
55 
56  Real spot = process_->stateVariable()->value() - riskless;
57  QL_REQUIRE(spot > 0.0,
58  "negative or null underlying after subtracting dividends");
59 
60  DiscountFactor dividendDiscount =
61  process_->dividendYield()->discount(
62  arguments_.exercise->lastDate());
63  DiscountFactor riskFreeDiscount =
64  process_->riskFreeRate()->discount(arguments_.exercise->lastDate());
65  Real forwardPrice = spot * dividendDiscount / riskFreeDiscount;
66 
67  Real variance =
68  process_->blackVolatility()->blackVariance(
69  arguments_.exercise->lastDate(),
70  payoff->strike());
71 
72  BlackCalculator black(payoff, forwardPrice, std::sqrt(variance),
73  riskFreeDiscount);
74 
75  results_.value = black.value();
76  results_.delta = black.delta(spot);
77  results_.gamma = black.gamma(spot);
78 
79  DayCounter rfdc = process_->riskFreeRate()->dayCounter();
80  DayCounter dydc = process_->dividendYield()->dayCounter();
81  DayCounter voldc = process_->blackVolatility()->dayCounter();
82  Time t = voldc.yearFraction(
83  process_->blackVolatility()->referenceDate(),
84  arguments_.exercise->lastDate());
85  results_.vega = black.vega(t);
86 
87  Real delta_theta = 0.0, delta_rho = 0.0;
88  for (i = 0; i < arguments_.cashFlow.size(); i++) {
89  Date d = arguments_.cashFlow[i]->date();
90 
91  if ( d >= settlementDate
92  && d <= arguments_.exercise->lastDate()) {
93 
94  delta_theta -= arguments_.cashFlow[i]->amount() *
95  ( process_->riskFreeRate()->zeroRate(d,rfdc,Continuous,Annual)
96  - process_->dividendYield()->zeroRate(d,dydc,Continuous,Annual)) *
97  process_->riskFreeRate()->discount(d) /
98  process_->dividendYield()->discount(d);
99 
100  Time t = process_->time(d);
101  delta_rho += arguments_.cashFlow[i]->amount() * t *
102  process_->riskFreeRate()->discount(t) /
103  process_->dividendYield()->discount(t);
104  }
105  }
106  t = process_->time(arguments_.exercise->lastDate());
107  try {
108  results_.theta = black.theta(spot, t) +
109  delta_theta * black.delta(spot);
110  } catch (Error&) {
111  results_.theta = Null<Real>();
112  }
113 
114  results_.rho = black.rho(t) +
115  delta_rho * black.delta(spot);
116  }
117 
118 }
