27 Real Constraint::update(Array& params, const Array& direction, Real beta) const {
28
29 Real diff=beta;
30 Array newParams = params + diff*direction;
31 bool valid = test(newParams);
32 Integer icount = 0;
33 while (!valid) {
34 if (icount > 200)
35 QL_FAIL("can't update parameter vector");
36 diff *= 0.5;
37 icount ++;
38 newParams = params + diff*direction;
39 valid = test(newParams);
40 }
41 params += diff*direction;
42 return diff;
43 }
44}
