28  SecondDerivativeOp::SecondDerivativeOp(
29  Size direction,
30  const boost::shared_ptr<FdmMesher>& mesher)
31  : TripleBandLinearOp(direction, mesher) {
32 
33  const boost::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
34  const FdmLinearOpIterator endIter = layout->end();
35 
36  for (FdmLinearOpIterator iter = layout->begin(); iter!=endIter; ++iter) {
37  const Size i = iter.index();
38  const Real hm = mesher->dminus(iter, direction_);
39  const Real hp = mesher->dplus(iter, direction_);
40 
41  const Real zetam1 = hm*(hm+hp);
42  const Real zeta0 = hm*hp;
43  const Real zetap1 = hp*(hm+hp);
44 
45  const Size co = iter.coordinates()[direction_];
46  if (co == 0 || co == layout->dim()[direction]-1) {
47  lower_[i] = diag_[i] = upper_[i] = 0.0;
48  }
49  else {
50  lower_[i] = 2.0/zetam1;
51  diag_[i] = -2.0/zeta0;
52  upper_[i] = 2.0/zetap1;
53  }
54  }
55  }
56 }
