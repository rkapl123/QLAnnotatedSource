35 //! Two-additive-factor gaussian model class.
36 /*! This class implements a two-additive-factor model defined by
37 \f[
38 dr_t = \varphi(t) + x_t + y_t
39 \f]
40 where \f$ x_t \f$ and \f$ y_t \f$ are defined by
41 \f[
42 dx_t = -a x_t dt + \sigma dW^1_t, x_0 = 0
43 \f]
44 \f[
45 dy_t = -b y_t dt + \sigma dW^2_t, y_0 = 0
46 \f]
47 and \f$ dW^1_t dW^2_t = \rho dt \f$.
48
49 \bug This class was not tested enough to guarantee
50 its functionality.
51
52 \ingroup shortrate
53 */

132 //! Analytical term-structure fitting parameter \f$ \varphi(t) \f$.
133 /*! \f$ \varphi(t) \f$ is analytically defined by
134 \f[
135 \varphi(t) = f(t) +
136 \frac{1}{2}(\frac{\sigma(1-e^{-at})}{a})^2 +
137 \frac{1}{2}(\frac{\eta(1-e^{-bt})}{b})^2 +
138 \rho\frac{\sigma(1-e^{-at})}{a}\frac{\eta(1-e^{-bt})}{b},
139 \f]
140 where \f$ f(t) \f$ is the instantaneous forward rate at \f$ t \f$.
141 */
142 class G2::FittingParameter : public TermStructureFittingParameter {
143 private:
144 class Impl final : public Parameter::Impl {
145 public:
146 Impl(Handle<YieldTermStructure> termStructure,
147 Real a,
148 Real sigma,
149 Real b,
150 Real eta,
151 Real rho)
152 : termStructure_(std::move(termStructure)), a_(a), sigma_(sigma), b_(b), eta_(eta),
153 rho_(rho) {}
154
155 Real value(const Array&, Time t) const override {
156 Rate forward = termStructure_->forwardRate(t, t,
157 Continuous,
158 NoFrequency);
159 Real temp1 = sigma_*(1.0-std::exp(-a_*t))/a_;
160 Real temp2 = eta_*(1.0-std::exp(-b_*t))/b_;
161 Real value = 0.5*temp1*temp1 + 0.5*temp2*temp2 +
162 rho_*temp1*temp2 + forward;
163 return value;
164 }
165
166 private:
167 Handle<YieldTermStructure> termStructure_;
168 Real a_, sigma_, b_, eta_, rho_;
169 };
170 public:
171 FittingParameter(const Handle<YieldTermStructure>& termStructure,
172 Real a,
173 Real sigma,
174 Real b,
175 Real eta,
176 Real rho)
177 : TermStructureFittingParameter(ext::shared_ptr<Parameter::Impl>(
178 new FittingParameter::Impl(termStructure, a, sigma,
179 b, eta, rho))) {}
180 };
