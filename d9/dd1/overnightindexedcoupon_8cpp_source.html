39  public:
40  void initialize(const FloatingRateCoupon& coupon) {

44  Rate swapletRate() const {
45 
46  shared_ptr<OvernightIndex> index =
47  dynamic_pointer_cast<OvernightIndex>(coupon_->index());
48 
49  const vector<Date>& fixingDates = coupon_->fixingDates();
50  const vector<Time>& dt = coupon_->dt();
51 
52  Size n = dt.size(),
53  i = 0;
54 
55  Real compoundFactor = 1.0;
56 
57  // already fixed part
58  Date today = Settings::instance().evaluationDate();
59  while (i<n && fixingDates[i]<today) {
60  // rate must have been fixed
61  Rate pastFixing = IndexManager::instance().getHistory(
62  index->name())[fixingDates[i]];
63  QL_REQUIRE(pastFixing != Null<Real>(),
64  "Missing " << index->name() <<
65  " fixing for " << fixingDates[i]);
66  compoundFactor *= (1.0 + pastFixing*dt[i]);
67  ++i;
68  }
69 
70  // today is a border case
71  if (i<n && fixingDates[i] == today) {
72  // might have been fixed
73  try {
74  Rate pastFixing = IndexManager::instance().getHistory(
75  index->name())[fixingDates[i]];
76  if (pastFixing != Null<Real>()) {
77  compoundFactor *= (1.0 + pastFixing*dt[i]);
78  ++i;
79  } else {
80  ; // fall through and forecast
81  }
82  } catch (Error&) {
83  ; // fall through and forecast
84  }
85  }
86 
87  // forward part using telescopic property in order
88  // to avoid the evaluation of multiple forward fixings
89  if (i<n) {
90  Handle<YieldTermStructure> curve =
91  index->forwardingTermStructure();
92  QL_REQUIRE(!curve.empty(),
93  "null term structure set to this instance of "<<
94  index->name());
95 
96  const vector<Date>& dates = coupon_->valueDates();
97  DiscountFactor startDiscount = curve->discount(dates[i]);
98  DiscountFactor endDiscount = curve->discount(dates[n]);
99 
100  compoundFactor *= startDiscount/endDiscount;
101  }
102 
103  Rate rate = (compoundFactor - 1.0) / coupon_->accrualPeriod();
104  return coupon_->gearing() * rate + coupon_->spread();
105  }

117  OvernightIndexedCoupon::OvernightIndexedCoupon(
118  const Date& paymentDate,
119  Real nominal,
120  const Date& startDate,
121  const Date& endDate,
122  const shared_ptr<OvernightIndex>& overnightIndex,
123  Real gearing,
124  Spread spread,
125  const Date& refPeriodStart,
126  const Date& refPeriodEnd,
127  const DayCounter& dayCounter,
128  bool telescopicValueDates)
129  : FloatingRateCoupon(paymentDate, nominal, startDate, endDate,
130  overnightIndex->fixingDays(), overnightIndex,
131  gearing, spread,
132  refPeriodStart, refPeriodEnd,
133  dayCounter, false) {
134 
135  // value dates
136  Date tmpEndDate = endDate;
137 
138  /* For the coupon's valuation only the first and last future valuation
139  dates matter, therefore we can avoid to construct the whole series
140  of valuation dates, a front and back stub will do. However notice
141  that if the global evaluation date moves forward it might run past
142  the front stub of valuation dates we build here (which incorporates
143  a grace period of 7 business after the evluation date). This will
144  lead to false coupon projections (see the warning the class header). */
145 
146  if (telescopicValueDates) {
147  // build optimised value dates schedule: front stub goes
148  // from start date to max(evalDate,startDate) + 7bd
149  Date evalDate = Settings::instance().evaluationDate();
150  tmpEndDate = overnightIndex->fixingCalendar().advance(
151  std::max(startDate, evalDate), 7, Days, Following);
152  tmpEndDate = std::min(tmpEndDate, endDate);
153  }
154  Schedule sch =
155  MakeSchedule()
156  .from(startDate)
157  // .to(endDate)
158  .to(tmpEndDate)
159  .withTenor(1 * Days)
160  .withCalendar(overnightIndex->fixingCalendar())
161  .withConvention(overnightIndex->businessDayConvention())
162  .backwards();
163  valueDates_ = sch.dates();
164 
165  if (telescopicValueDates) {
166  // build optimised value dates schedule: back stub
167  // contains at least two dates
168  Date tmp = overnightIndex->fixingCalendar().advance(
169  endDate, -1, Days, Preceding);
170  if (tmp != valueDates_.back())
171  valueDates_.push_back(tmp);
172  tmp = overnightIndex->fixingCalendar().adjust(
173  endDate, overnightIndex->businessDayConvention());
174  if (tmp != valueDates_.back())
175  valueDates_.push_back(tmp);
176  }
177 
178  QL_ENSURE(valueDates_.size()>=2, "degenerate schedule");
179 
180  // fixing dates
181  n_ = valueDates_.size()-1;
182  if (overnightIndex->fixingDays()==0) {
183  fixingDates_ = vector<Date>(valueDates_.begin(),
184  valueDates_.end()-1);
185  } else {
186  fixingDates_.resize(n_);
187  for (Size i=0; i<n_; ++i)
188  fixingDates_[i] = overnightIndex->fixingDate(valueDates_[i]);
189  }
190 
191  // accrual (compounding) periods
192  dt_.resize(n_);
193  const DayCounter& dc = overnightIndex->dayCounter();
194  for (Size i=0; i<n_; ++i)
195  dt_[i] = dc.yearFraction(valueDates_[i], valueDates_[i+1]);
196 
197  setPricer(shared_ptr<FloatingRateCouponPricer>(new
198  OvernightIndexedCouponPricer));
199  }

279  OvernightLeg::operator Leg() const {
280 
281  QL_REQUIRE(!notionals_.empty(), "no notional given");
282 
283  Leg cashflows;
284 
285  // the following is not always correct
286  Calendar calendar = schedule_.calendar();
287 
288  Date refStart, start, refEnd, end;
289  Date paymentDate;
290 
291  Size n = schedule_.size()-1;
292  for (Size i=0; i<n; ++i) {
293  refStart = start = schedule_.date(i);
294  refEnd = end = schedule_.date(i+1);
295  paymentDate = paymentCalendar_.advance(end, paymentLag_, Days, paymentAdjustment_);
296 
297  if (i == 0 && schedule_.hasIsRegular() && !schedule_.isRegular(i+1))
298  refStart = calendar.adjust(end - schedule_.tenor(),
299  paymentAdjustment_);
300  if (i == n-1 && schedule_.hasIsRegular() && !schedule_.isRegular(i+1))
301  refEnd = calendar.adjust(start + schedule_.tenor(),
302  paymentAdjustment_);
303 
304  cashflows.push_back(shared_ptr<CashFlow>(new
305  OvernightIndexedCoupon(paymentDate,
306  detail::get(notionals_, i,
307  notionals_.back()),
308  start, end,
309  overnightIndex_,
310  detail::get(gearings_, i, 1.0),
311  detail::get(spreads_, i, 0.0),
312  refStart, refEnd,
313  paymentDayCounter_,
314  telescopicValueDates_)));
315  }
316  return cashflows;
317  }
