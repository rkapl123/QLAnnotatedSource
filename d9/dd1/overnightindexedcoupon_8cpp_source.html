37  public:
38  void initialize(const FloatingRateCoupon& coupon) {

42  Rate swapletRate() const {
43 
44  ext::shared_ptr<OvernightIndex> index =
45  ext::dynamic_pointer_cast<OvernightIndex>(coupon_->index());
46 
47  const vector<Date>& fixingDates = coupon_->fixingDates();
48  const vector<Time>& dt = coupon_->dt();
49 
50  Size n = dt.size(),
51  i = 0;
52 
53  Real compoundFactor = 1.0;
54 
55  // already fixed part
56  Date today = Settings::instance().evaluationDate();
57  while (i<n && fixingDates[i]<today) {
58  // rate must have been fixed
59  Rate pastFixing = IndexManager::instance().getHistory(
60  index->name())[fixingDates[i]];
61  QL_REQUIRE(pastFixing != Null<Real>(),
62  "Missing " << index->name() <<
63  " fixing for " << fixingDates[i]);
64  compoundFactor *= (1.0 + pastFixing*dt[i]);
65  ++i;
66  }
67 
68  // today is a border case
69  if (i<n && fixingDates[i] == today) {
70  // might have been fixed
71  try {
72  Rate pastFixing = IndexManager::instance().getHistory(
73  index->name())[fixingDates[i]];
74  if (pastFixing != Null<Real>()) {
75  compoundFactor *= (1.0 + pastFixing*dt[i]);
76  ++i;
77  } else {
78  ; // fall through and forecast
79  }
80  } catch (Error&) {
81  ; // fall through and forecast
82  }
83  }
84 
85  // forward part using telescopic property in order
86  // to avoid the evaluation of multiple forward fixings
87  if (i<n) {
88  Handle<YieldTermStructure> curve =
89  index->forwardingTermStructure();
90  QL_REQUIRE(!curve.empty(),
91  "null term structure set to this instance of "<<
92  index->name());
93 
94  const vector<Date>& dates = coupon_->valueDates();
95  DiscountFactor startDiscount = curve->discount(dates[i]);
96  DiscountFactor endDiscount = curve->discount(dates[n]);
97 
98  compoundFactor *= startDiscount/endDiscount;
99  }
100 
101  Rate rate = (compoundFactor - 1.0) / coupon_->accrualPeriod();
102  return coupon_->gearing() * rate + coupon_->spread();
103  }

115  OvernightIndexedCoupon::OvernightIndexedCoupon(
116  const Date& paymentDate,
117  Real nominal,
118  const Date& startDate,
119  const Date& endDate,
120  const ext::shared_ptr<OvernightIndex>& overnightIndex,
121  Real gearing,
122  Spread spread,
123  const Date& refPeriodStart,
124  const Date& refPeriodEnd,
125  const DayCounter& dayCounter,
126  bool telescopicValueDates)
127  : FloatingRateCoupon(paymentDate, nominal, startDate, endDate,
128  overnightIndex->fixingDays(), overnightIndex,
129  gearing, spread,
130  refPeriodStart, refPeriodEnd,
131  dayCounter, false) {
132 
133  // value dates
134  Date tmpEndDate = endDate;
135 
136  /* For the coupon's valuation only the first and last future valuation
137  dates matter, therefore we can avoid to construct the whole series
138  of valuation dates, a front and back stub will do. However notice
139  that if the global evaluation date moves forward it might run past
140  the front stub of valuation dates we build here (which incorporates
141  a grace period of 7 business after the evluation date). This will
142  lead to false coupon projections (see the warning the class header). */
143 
144  if (telescopicValueDates) {
145  // build optimised value dates schedule: front stub goes
146  // from start date to max(evalDate,startDate) + 7bd
147  Date evalDate = Settings::instance().evaluationDate();
148  tmpEndDate = overnightIndex->fixingCalendar().advance(
149  std::max(startDate, evalDate), 7, Days, Following);
150  tmpEndDate = std::min(tmpEndDate, endDate);
151  }
152  Schedule sch =
153  MakeSchedule()
154  .from(startDate)
155  // .to(endDate)
156  .to(tmpEndDate)
157  .withTenor(1 * Days)
158  .withCalendar(overnightIndex->fixingCalendar())
159  .withConvention(overnightIndex->businessDayConvention())
160  .backwards();
161  valueDates_ = sch.dates();
162 
163  if (telescopicValueDates) {
164  // build optimised value dates schedule: back stub
165  // contains at least two dates
166  Date tmp = overnightIndex->fixingCalendar().advance(
167  endDate, -1, Days, Preceding);
168  if (tmp != valueDates_.back())
169  valueDates_.push_back(tmp);
170  tmp = overnightIndex->fixingCalendar().adjust(
171  endDate, overnightIndex->businessDayConvention());
172  if (tmp != valueDates_.back())
173  valueDates_.push_back(tmp);
174  }
175 
176  QL_ENSURE(valueDates_.size()>=2, "degenerate schedule");
177 
178  // fixing dates
179  n_ = valueDates_.size()-1;
180  if (overnightIndex->fixingDays()==0) {
181  fixingDates_ = vector<Date>(valueDates_.begin(),
182  valueDates_.end()-1);
183  } else {
184  fixingDates_.resize(n_);
185  for (Size i=0; i<n_; ++i)
186  fixingDates_[i] = overnightIndex->fixingDate(valueDates_[i]);
187  }
188 
189  // accrual (compounding) periods
190  dt_.resize(n_);
191  const DayCounter& dc = overnightIndex->dayCounter();
192  for (Size i=0; i<n_; ++i)
193  dt_[i] = dc.yearFraction(valueDates_[i], valueDates_[i+1]);
194 
195  setPricer(ext::shared_ptr<FloatingRateCouponPricer>(new
196  OvernightIndexedCouponPricer));
197  }

277  OvernightLeg::operator Leg() const {
278 
279  QL_REQUIRE(!notionals_.empty(), "no notional given");
280 
281  Leg cashflows;
282 
283  // the following is not always correct
284  Calendar calendar = schedule_.calendar();
285 
286  Date refStart, start, refEnd, end;
287  Date paymentDate;
288 
289  Size n = schedule_.size()-1;
290  for (Size i=0; i<n; ++i) {
291  refStart = start = schedule_.date(i);
292  refEnd = end = schedule_.date(i+1);
293  paymentDate = paymentCalendar_.advance(end, paymentLag_, Days, paymentAdjustment_);
294 
295  if (i == 0 && schedule_.hasIsRegular() && !schedule_.isRegular(i+1))
296  refStart = calendar.adjust(end - schedule_.tenor(),
297  paymentAdjustment_);
298  if (i == n-1 && schedule_.hasIsRegular() && !schedule_.isRegular(i+1))
299  refEnd = calendar.adjust(start + schedule_.tenor(),
300  paymentAdjustment_);
301 
302  cashflows.push_back(ext::shared_ptr<CashFlow>(new
303  OvernightIndexedCoupon(paymentDate,
304  detail::get(notionals_, i,
305  notionals_.back()),
306  start, end,
307  overnightIndex_,
308  detail::get(gearings_, i, 1.0),
309  detail::get(spreads_, i, 0.0),
310  refStart, refEnd,
311  paymentDayCounter_,
312  telescopicValueDates_)));
313  }
314  return cashflows;
315  }
