132  inline void Handle<T>::Link::linkTo(const ext::shared_ptr<T>& h,
133  bool registerAsObserver) {
134  if ((h != h_) || (isObserver_ != registerAsObserver)) {
135  if (h_ && isObserver_)
136  unregisterWith(h_);
137  h_ = h;
138  isObserver_ = registerAsObserver;

140  registerWith(h_);
