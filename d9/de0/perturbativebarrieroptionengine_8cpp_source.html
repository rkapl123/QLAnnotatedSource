71  Real stock,
72  Real hbarr,
73  Real taumin,
74  Real taumax,
75  int iord,
76  int igm,
77  const ext::function<Real(Real, Real)>& integr,
78  const ext::function<Real(Real, Real)>& integalpha,
79  const ext::function<Real(Real, Real)>& integs,
80  const ext::function<Real(Real)>& alpha,
81  const ext::function<Real(Real)>& sigmaq) {
82  Real v0=0.0, v1=0.0, v1p=0.0, v2p=0.0, v2pp=0.0, gm=0.0;
83  int i=0,j=0;
84  Real tmp=0.0, e1=0.0, e2=0.0, e3=0.0, e4=0.0;
85  Real xstar=0.0, s0=0.0;
86  Real sigmat=0.0, disc=0.0, d1=0.0,d2=0.0,d3=0.0,d4=0.0;
87  Real et=0.0,tt=0.0, dt=0.0,p=0.0;
88  int npoint,npoint2;
89  static double pi= 3.14159265358979324;
90  Real dsqpi;
91  Real caux=0.0,ccaux=0.0;
92  Real auxnew=0.0;
93  Real x=0.0,b=0.0,c=0.0;
94 
95  if(igm==0) {
96  gm=0.0;
97  } else if(igm==1) {
98  gm=integalpha(taumin,taumax)/(0.5*integs(taumin,taumax));
99  } else {
100  gm=0.0;
101  }
102 
103  /*
104  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
105  !! !!
106  !! xstar=min(0,log(kprice/hbarr)) !!
107  !! !!
108  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
109  */
110 
111  xstar=log(kprice/hbarr);
112 
113  if(xstar>0.0) xstar=0.0;
114  sigmat=integs(taumin,taumax);
115  disc=-integr(taumin,taumax);
116 
117  /*
118  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
119  !! Change of variable !!
120  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
121  */
122  s0=stock/hbarr;
123 
124  /*
125  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
126  !! !!
127  !! Compute the zero-th order term P_0 !!
128  !! !!
129  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
130  */
131  d1=(xstar-log(s0)+(1.0-gm)*0.5*sigmat)/sqrt(sigmat);
132  d2=(xstar+log(s0)+(1.0-gm)*0.5*sigmat)/sqrt(sigmat);
133  d3=(xstar-log(s0)-(1.0+gm)*0.5*sigmat)/sqrt(sigmat);
134  d4=(xstar+log(s0)-(1.0+gm)*0.5*sigmat)/sqrt(sigmat);
135 
136  e1=PHID(d1);
137  e2=PHID(d2);
138  e3=PHID(d3);
139  e4=PHID(d4);
140 
141  v0=kprice*e1-kprice*POW(s0,(1.0-gm))*e2;
142  v0=v0+exp(gm*0.5*sigmat)*(-hbarr*s0*e3+hbarr*POW(s0,-gm)*e4);
143  v0=v0*exp(disc);
144 
145  if(iord==0) return v0;
146 
147  /*
148  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
149  !! !!
150  !! Compute the first order term P_1 !!
151  !! !!
152  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
153  */
154 
155  npoint=1000;
156  npoint2=100;
157 
158  dt=(taumax-taumin)/double(npoint);
159 
160  tt=0.5*integs(taumin,taumax);
161 
162  x=log(s0);
163  et=exp(0.5*(1.0-gm)*x);
164 
165  dsqpi=POW(pi,0.5);
166 
167  v1=0.0;
168  for( i=1;i<=npoint;i++) {
169  v1p=0.0;
170  tmp=taumin+dt*double(2*i-1)*0.5;
171  p=0.5*integs(tmp,taumax);
172  /*
173  !!
174  !! Function E(p,tt,a,b,gm)
175  !!
176  */
177  ccaux=v(p,tt,x,xstar,gm)+v(p,tt,x,-xstar,gm)-v(p,tt,-x,xstar,gm)-v(p,tt,-x,-xstar,gm);
178  auxnew=ccaux*(-kprice*exp(-xstar*0.5*(1.0-gm))+hbarr*exp(xstar*0.5*(1.0+gm)));
179  v1p=v1p+auxnew;
180  /*
181  !!
182  !! Function L(p,tt,a,b,c,gm)
183  !!
184  */
185  b=gm-1.0;
186  c=-xstar;
187  ccaux=llold(p,tt,x,b,c,gm)-llold(p,tt,-x,b,c,gm);
188  auxnew=kprice*(1.0-gm)*ccaux;
189  v1p=v1p+auxnew;
190 
191  b=-(gm+1.0);
192  c=xstar;
193  ccaux=llold(p,tt,x,b,c,gm)-llold(p,tt,-x,b,c,gm);
194  auxnew=-exp(gm*p)*hbarr*ccaux;
195  v1p=v1p+auxnew;
196 
197  b=(gm+1.0);
198  c=-xstar;
199  ccaux=llold(p,tt,x,b,c,gm)-llold(p,tt,-x,b,c,gm);
200  auxnew=exp(gm*p)*hbarr*gm*ccaux;
201  v1p=v1p+auxnew;
202  /*
203  !!
204  !! Function F(p,tt,a,b,c,gm)
205  !!
206  */
207  b=gm-1.0;
208  auxnew=-kprice*(1.0-gm)*(ff(p,tt,x,b,gm)-ff(p,tt,-x,b,gm));
209  v1p=v1p+auxnew;
210 
211  b=gm+1.0;
212  auxnew=-exp(gm*p)*gm*hbarr*(ff(p,tt,x,b,gm)-ff(p,tt,-x,b,gm));
213  v1p=v1p+auxnew;
214 
215  v1=v1+(alpha(tmp)-gm*0.5*sigmaq(tmp))*v1p;
216  }
217 
218  v1=exp(disc)*et*v1*dt/(dsqpi*2.0);
219 
220  if(iord==1) return v0+v1;
221 
222  /*
223  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
224  !! !!
225  !! Compute the second order term P_2 !!
226  !! !!
227  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
228  */
229 
230  Real v2,dtp, tmp1,s,caux2;
231  v2=0.0;
232 
233  for(i=1;i<=npoint;i++) {
234  v2p=0.0;
235  tmp=taumin+dt*(double)(2*i-1)*0.5;
236  p=0.5*integs(tmp,taumax);
237 
238  dtp=(taumax-tmp)/(double)(npoint2);
239 
240  for(j=1;j<=npoint2; j++) {
241  tmp1=tmp+dtp*(double)(2*j-1)*0.50;
242  s=0.50*integs(tmp1,taumax);
243 
244  caux=dll(s,p,tt,-x,-1.0+gm,-xstar,gm)-dll(s,p,tt,x,-1.0+gm,-xstar,gm);
245  v2pp=caux*kprice*(1.0-gm);
246 
247  caux=dll(s,p,tt,-x,-1.0-gm,xstar,gm)-dll(s,p,tt,x,-1.0-gm,xstar,gm);
248  v2pp=v2pp-exp(gm*s)*hbarr*caux;
249 
250  caux=dll(s,p,tt,-x,1.0+gm,-xstar,gm)-dll(s,p,tt,x,1.0+gm,-xstar,gm);
251  v2pp=v2pp+exp(gm*s)*gm*hbarr*caux;
252 
253  caux=+dvv(s,p,tt,-x,xstar,gm)-dvv(s,p,tt,x,xstar,gm);
254  caux=caux+(dvv(s,p,tt,-x,-xstar,gm)-dvv(s,p,tt,x,-xstar,gm));
255  caux2=hbarr*exp(0.5*(1.0+gm)*xstar)-kprice*exp(-0.5*(1.0-gm)*xstar);
256  v2pp=v2pp+caux2*caux;
257 
258  caux=dff(s,p,tt,-x,-1.0+gm,gm)-dff(s,p,tt,x,-1.0+gm,gm);
259  v2pp=v2pp-(1.0-gm)*kprice*caux;
260 
261  caux=dff(s,p,tt,-x,1.0+gm,gm)-dff(s,p,tt,x,1.0+gm,gm);
262  v2pp=v2pp-exp(gm*s)*gm*hbarr*caux;
263 
264  v2pp=v2pp*0.5*(1.0-gm);
265 
266  caux=-ddll(s,p,tt,-x,-1.0+gm,-xstar,gm)+ddll(s,p,tt,x,-1.0+gm,-xstar,gm);
267  v2pp=v2pp+caux*kprice*(1.0-gm);
268 
269  caux=-ddll(s,p,tt,-x,-1.0-gm,xstar,gm)+ddll(s,p,tt,x,-1.0-gm,xstar,gm);
270  v2pp=v2pp-exp(gm*s)*hbarr*caux;
271 
272  caux=-ddll(s,p,tt,-x,1.0+gm,-xstar,gm)+ddll(s,p,tt,x,1.0+gm,-xstar,gm);
273  v2pp=v2pp+exp(gm*s)*gm*hbarr*caux;
274 
275  caux=-ddvv(s,p,tt,-x,xstar,gm)+ddvv(s,p,tt,x,xstar,gm);
276  caux=caux+(-dvv(s,p,tt,-x,-xstar,gm)+dvv(s,p,tt,x,-xstar,gm));
277  caux2=hbarr*exp(0.5*(1.0+gm)*xstar)-kprice*exp(-0.5*(1-gm)*xstar);
278 
279  v2pp=v2pp+caux2*caux;
280 
281  caux=-ddff(s,p,tt,-x,-1+gm,gm)+ddff(s,p,tt,x,-1+gm,gm);
282  v2pp=v2pp-(1.0-gm)*kprice*caux;
283 
284  caux=-ddff(s,p,tt,-x,1.0+gm,gm)+ddff(s,p,tt,x,1.0+gm,gm);
285  v2pp=v2pp-exp(gm*s)*gm*hbarr*caux;
286 
287  v2p=v2p+(alpha(tmp1)-gm*0.5*sigmaq(tmp1))*v2pp;
288  }
289 
290  v2=v2+v2p*(alpha(tmp)-gm*0.5*sigmaq(tmp))*dtp;
291  }
292 
293  v2=exp(disc)*et*v2*dt;
294 
295  return v0+v1+v2;
296  }

299  Real PHID(Real Z){
300  /*
301  * Normal distribution probabilities accurate to 1D-15.
302  * Z = number of standard deviations from the mean.
303  *
304  * The software that computes the normal distribution
305  * probabilities has been developed by M.C. Recchioni
306  * based upon algorithm 5666 (Programmer Alan Miller)
307  * for the error function, taken from:
308  * Hart, J.F. et al, 'Computer Approximations', Wiley, 1968
309  *
310  */
311  Real P0, P1, P2, P3, P4, P5, P6;
312  Real Q0, Q1, Q2, Q3, Q4, Q5, Q6, Q7;
313  Real P, EXPNTL, CUTOFF, ROOTPI, ZABS;
314 
315  P0 = 220.2068679123761;
316  P1 = 221.2135961699311;
317  P2 = 112.0792914978709;
318  P3 = 33.91286607838300;
319  P4 = 6.373962203531650;
320  P5 = 0.7003830644436881;
321  P6 = 0.03526249659989109;
322 
323  Q0 = 440.4137358247522;
324  Q1 = 793.8265125199484;
325  Q2 = 637.3336333788311;
326  Q3 = 296.5642487796737;
327  Q4 = 86.78073220294608;
328  Q5 = 16.064177579206950;
329  Q6 = 1.7556671631826420;
330  Q7 = 0.088388347648318440;
331  ROOTPI = 2.506628274631001;
332  CUTOFF = 7.071067811865475;
333 
334  ZABS = fabs(Z);
335  /*
336  |Z| > 37
337  */
338  if (ZABS > 37)
339  P = 0;
340  else
341  {
342  /*
343  |Z| <= 37
344  */
345  EXPNTL =exp(-ZABS*ZABS/2);
346  /*
347  |Z| < CUTOFF = 10/SQRT(2)
348  */
349  if ( ZABS < CUTOFF )
350  P = EXPNTL*((((((P6*ZABS + P5)*ZABS + P4)*ZABS + P3)*ZABS+ P2)*ZABS + P1)*ZABS + P0)/(((((((Q7*ZABS + Q6)*ZABS + Q5)*ZABS + Q4)*ZABS + Q3)*ZABS + Q2)*ZABS + Q1)*ZABS + Q0);
351  /*
352  |Z| >= CUTOFF.
353  */
354  else
355  P = EXPNTL/(ZABS + 1/(ZABS + 2/(ZABS + 3/(ZABS + 4/(ZABS + 0.65)))))/ROOTPI;
356 
357  }
358  if ( Z > 0 ) P = 1 - P;
359 
360  return(P);
361  }

441  /*
442  !!
443  !! Function D_E(s,p,tt,a,b,gm)
444  !!
445  */
446  Real dvv(Real s,Real p,Real tt,Real a,Real b,Real gm)
447  {
448  double ppi= 3.14159265358979324;
449  Real result;
450  Real aa,caux,caux1,caux2;
451  Real xx,yy,rho;
452 
453  aa=(a*p+b*(tt-p))/POW(2.0*p*tt*(tt-p),0.5);
454  caux=PHID(aa);
455 
456  aa=exp((a-b)*(a-b)/(4.0*tt))*exp(POW((1.0-gm),2)*tt/4.0)*POW(tt,0.5);
457  caux=-caux/aa;
458 
459  xx=(a*p+b*(tt-p))/POW(2.0*tt*p*(tt-p),0.5);
460  yy=(a*s+b*(tt-s))/POW(2.0*tt*s*(tt-s),0.5);
461  rho=POW((s*(tt-p))/(p*(tt-s)),0.5);
462  caux1=ND2(-xx,-yy,rho);
463  caux1=caux1/aa;
464 
465 
466  aa=exp((a+b)*(a+b)/(4.0*tt))*exp(POW((1.0-gm),2)*tt/4.0)*POW(tt,0.5);
467 
468  xx=(a*p-b*(tt-p))/POW(2.0*tt*p*(tt-p),0.5);
469  yy=(a*s-b*(tt-s))/POW(2.0*tt*s*(tt-s),0.5);
470  rho=POW((s*(tt-p))/(p*(tt-s)),0.5);
471  caux2=ND2(-xx,-yy,rho);
472  caux2=caux2/aa;
473  result=(caux+caux1+caux2)/(2.0*POW(ppi,0.5));
474  return(result);
475  }

478  !!
479  !! Function D_F(s,p,tt,a,b,gm)
480  !!
481  */
482  Real dff(Real s, Real p,Real tt,Real a,Real b,Real gm)
483  {
484  Real result;
485  Real aa,caux,caux1,caux2;
486  Real xx,yy,rho;
487 
488  xx=(a-b*(tt-p))/POW(2.0*(tt-p),0.5);
489  caux=-PHID(xx)*exp(-0.5*a*b);
490 
491  xx=(a+b*(tt-p))/POW(2.0*(tt-p),0.5);
492  yy=(a+b*(tt-s))/POW(2.0*(tt-s),0.5);
493  rho=POW((tt-p)/(tt-s),0.5);
494  caux1=ND2(-xx,-yy,rho);
495  caux1=exp(0.5*a*b)*caux1;
496 
497  xx=(a-b*(tt-p))/POW(2.0*(tt-p),0.5);
498  yy=(a-b*(tt-s))/POW(2.0*(tt-s),0.5);
499  rho=POW((tt-p)/(tt-s),0.5);
500  caux2=ND2(-xx,-yy,rho);
501  caux2=exp(-0.5*a*b)*caux2;
502 
503  aa=exp((b*b-(1.0-gm)*(1.0-gm))*(tt-s)/4.0);
504 
505  result=(caux+caux1+caux2)*aa;
506  return(result);
507  }

510  /*
511  !!
512  !! Function D_L(s,p,a,b,c,gm)
513  !!
514  */
515  Real dll(Real s,Real p,Real tt,Real a,Real b,Real c,Real gm)
516  {
517  Real result;
518  Real aa,caux,caux1;
519  Real sigmarho[4],limit[4],epsi;
520 
521  epsi=1.e-12;
522  limit[1]=(a+b*(tt-p))/POW(2.0*(tt-p),0.5);
523  limit[2]=(a+b*(tt-s))/POW(2.0*(tt-s),0.5);
524  limit[3]=(a+b*tt+c)/POW(2.0*tt,0.5);
525  sigmarho[1]=POW((tt-p)/(tt-s),0.5);
526  sigmarho[2]=POW((tt-p)/tt,0.5);
527  sigmarho[3]=POW((tt-s)/tt,0.5);
528 
529  caux=exp(0.5*a*b)*tvtl(0,limit,sigmarho,epsi);
530 
531  limit[1]=(a-b*(tt-p))/POW(2.0*(tt-p),0.5);
532  limit[2]=(-a+b*(tt-s))/POW(2.0*(tt-s),0.5);
533  limit[3]=(-a+b*tt+c)/POW(2.0*tt,0.5);
534  sigmarho[1]=-POW((tt-p)/(tt-s),0.5);
535  sigmarho[2]=-POW((tt-p)/tt,0.5);
536  sigmarho[3]=POW((tt-s)/tt,0.5);
537 
538  caux1=-exp(-0.5*a*b)*tvtl(0,limit,sigmarho,epsi);
539 
540  aa=exp((b*b-(1.0-gm)*(1.0-gm))*(tt-s)/4.0);
541 
542 
543  result=(caux+caux1)*aa;
544 
545  return(result);
546  }

791  Real tvtl(int NU, const Real limit[], const Real sigmarho[], Real epsi) {
792  /*
793  A function for computing trivariate normal and t-probabilities.
794 
795  This function uses algorithms developed from the ideas
796  described in the papers:
797  R.L. Plackett, Biometrika 41(1954), pp. 351-360.
798  Z. Drezner, Math. Comp. 62(1994), pp. 289-294.
799  and uses adaptive integration.
800 
801  The software given here is based on algorithms described in
802  the paper A. Genz: "Numerical Computation of Rectangular
803  Bivariate and Trivariate Normal and t Probabilities",
804  Statistics and Computing 14 (2004) 251-260.
805 
806  This software has been developed by M.C. Recchioni based on
807  previous software developed by
808  Alan Genz
809  Department of Mathematics
810  Washington State University
811  Pullman, WA 99164-3113
812  Email : alangenz@wsu.edu
813  The software developed by A. Genz is available free of
814  charge in the website:
815  www.math.wsu.edu/faculty/genz/software/software.html
816 
817  The software calculates the probability that
818  X(I) < H(I), for I = 1,2,3
819 
820  NU INTEGER degrees of freedom; use NU = 0 for normal cases.
821  LIMIT REAL array of uppoer limits for probability distribution
822  SIGMARHO REAL array of three correlation coefficients, should
823  contain the lower left portion of the correlation matrix.
824  SIGMARHO should contains the values r21, r31, r23 in that order.
825  EPSI REAL required absolute accuracy; maximum accuracy for most
826  computations is approximately 1D-14
827 
828  */
829 
830  static Real result;
831  static Real ONE=1.0, ZRO=0.0, EPS, TVT;
832  static Real PT, R12, R13;
833  static double ppi= 3.14159265358979324;
834  EPS = max( 1.e-14, epsi );
835  PT=ppi/2.0;
836 
837  NUC = NU;
838  H1 = limit[1];
839  H2 = limit[2];
840  H3 = limit[3];
841  R12 = sigmarho[1];
842  R13 = sigmarho[2];
843  R23 = sigmarho[3];
844  /*
845  * Sort R's and check for special cases
846  */
847  if ( fabs(R12) > fabs(R13) ) {
848  H2 = H3;
849  H3 = limit[2];
850  R12 = R13;
851  R13 = sigmarho[1];
852  }
853 
854  if ( fabs(R13) > fabs(R23) ) {
855  H1 = H2;
856  H2 = limit[1];
857  R23 = R13;
858  R13 = sigmarho[3];
859  }
860 
861  TVT = 0.0;
862  if ( (fabs(H1) + fabs(H2) + fabs(H3)) < EPS ) TVT = ( 1 + ( asin(R12) + asin(R13) + asin(R23) )/PT )/8.0;
863 
864  else if ( (NU < 1) && ( (fabs(R12) + fabs(R13)) < EPS) ) TVT = PHID(H1)*BVTL( NU, H2, H3, R23 );
865 
866  else if ( (NU < 1) && ((fabs(R13) + fabs(R23))< EPS) ) TVT = PHID(H3)*BVTL( NU, H1, H2, R12 );
867 
868  else if( (NU < 1) && ((fabs(R12) + fabs(R23))< EPS) ) TVT = PHID(H2)*BVTL( NU, H1, H3, R13 );
869 
870  else if ( (1.0 - R23)< EPS ) TVT = BVTL( NU, H1, min( H2, H3 ), R12 );
871 
872  else if ( (R23 + 1.0) <EPS ) {
873  if ( H2 > -H3 ) TVT = BVTL( NU, H1, H2, R12 ) - BVTL( NU, H1, -H3, R12 );}
874  else
875  {
876  /*
877  * Compute singular TVT value
878  */
879  if ( NU < 1 ) TVT = BVTL( NU, H2, H3, R23 )*PHID(H1);
880 
881  else if ( R23 > 0 ) TVT = BVTL( NU, H1, min( H2, H3 ), ZRO );
882 
883  else if ( H2 > -H3 ) TVT = BVTL( NU, H1, H2, ZRO ) - BVTL( NU, H1, -H3, ZRO );
884 
885  /*
886  * Use numerical integration to compute probability
887  *
888  */
889  RUA = asin( R12 );
890  RUB = asin( R13 );
891  AR = asin( R23);
892  RUC = SIGN( PT, AR ) - AR;
893  TVT = TVT + ADONET(ZRO, ONE, EPS, TVTMFN) / (4.0 * PT);
894  }
895  result = max( ZRO, min( TVT, ONE ) );
896 
897  return(result);
898  }
