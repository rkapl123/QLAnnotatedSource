49  Real fixing,
50  bool forceOverwrite) {
51 
52  std::pair<Date,Date> lim = inflationPeriod(fixingDate, frequency_);
53  Size n = lim.second - lim.first + 1;
54  std::vector<Date> dates(n);
55  std::vector<Rate> rates(n);
56  for (Size i=0; i<n; ++i) {
57  dates[i] = lim.first + i;
58  rates[i] = fixing;

61  Index::addFixings(dates.begin(), dates.end(),
62  rates.begin(), forceOverwrite);

80  Rate ZeroInflationIndex::fixing(const Date& aFixingDate,
81  bool /*forecastTodaysFixing*/) const {
82  if (!needsForecast(aFixingDate)) {
83  std::pair<Date,Date> lim = inflationPeriod(aFixingDate, frequency_);
84  const TimeSeries<Real>& ts = timeSeries();
85  Real pastFixing = ts[lim.first];
86  QL_REQUIRE(pastFixing != Null<Real>(),
87  "Missing " << name() << " fixing for " << lim.first);
88  Real theFixing = pastFixing;
89  if (interpolated_) {
90  // fixings stored on first day of every period
91  if (aFixingDate == lim.first) {
92  // we don't actually need the next fixing
93  theFixing = pastFixing;
94  } else {
95  Real pastFixing2 = ts[lim.second+1];
96  QL_REQUIRE(pastFixing2 != Null<Real>(),
97  "Missing " << name() << " fixing for " << lim.second+1);
98 
99  // Use lagged period for interpolation
100  std::pair<Date, Date> reference_period_lim = inflationPeriod(aFixingDate + zeroInflationTermStructure()->observationLag(), frequency_);
101  // now linearly interpolate
102  Real daysInPeriod = reference_period_lim.second + 1 - reference_period_lim.first;
103  theFixing = pastFixing
104  + (pastFixing2 - pastFixing)*(aFixingDate - lim.first) / daysInPeriod;
105  }
106  }
107  return theFixing;
108  } else {
109  return forecastFixing(aFixingDate);
110  }
111  }

114  bool ZeroInflationIndex::needsForecast(const Date& fixingDate) const {
115 
116  // Stored fixings are always non-interpolated.
117  // If an interpolated fixing is required then
118  // the availability lag + one inflation period
119  // must have passed to use historical fixings
120  // (because you need the next one to interpolate).
121  // The interpolation is calculated (linearly) on demand.
122 
123  Date today = Settings::instance().evaluationDate();
124  Date todayMinusLag = today - availabilityLag_;
125 
126  Date historicalFixingKnown =
127  inflationPeriod(todayMinusLag, frequency_).first-1;
128  Date latestNeededDate = fixingDate;
129 
130  if (interpolated_) { // might need the next one too
131  std::pair<Date,Date> p = inflationPeriod(fixingDate, frequency_);
132  if (fixingDate > p.first)
133  latestNeededDate += Period(frequency_);
134  }
135 
136  if (latestNeededDate <= historicalFixingKnown) {
137  // the fixing date is well before the availability lag, so
138  // we know that fixings were provided.
139  return false;
140  } else if (latestNeededDate > today) {
141  // the fixing can't be available, no matter what's in the
142  // time series
143  return true;
144  } else {
145  // we're not sure, but the fixing might be there so we
146  // check. Todo: check which fixings are not possible, to
147  // avoid using fixings in the future
148  Real f = timeSeries()[latestNeededDate];
149  return (f == Null<Real>());
150  }
151  }

154  Rate ZeroInflationIndex::forecastFixing(const Date& fixingDate) const {
155  // the term structure is relative to the fixing value at the base date.
156  Date baseDate = zeroInflation_->baseDate();
157  QL_REQUIRE(!needsForecast(baseDate),
158  name() << " index fixing at base date is not available");
159  Real baseFixing = fixing(baseDate);
160  Date effectiveFixingDate;
161  if (interpolated()) {
162  effectiveFixingDate = fixingDate;
163  } else {
164  // start of period is the convention
165  // so it's easier to do linear interpolation on fixings
166  effectiveFixingDate = inflationPeriod(fixingDate, frequency()).first;
167  }
168 
169  // no observation lag because it is the fixing for the date
170  // but if index is not interpolated then that fixing is constant
171  // for each period, hence the t uses the effectiveFixingDate
172  // However, it's slightly safe to get the zeroRate with the
173  // fixingDate to avoid potential problems at the edges of periods
174  Time t = zeroInflation_->dayCounter().yearFraction(baseDate, effectiveFixingDate);
175  bool forceLinearInterpolation = false;
176  Rate zero = zeroInflation_->zeroRate(fixingDate, Period(0,Days), forceLinearInterpolation);
177  // Annual compounding is the convention for zero inflation rates (or quotes)
178  return baseFixing * std::pow(1.0 + zero, t);
179  }

209  Rate YoYInflationIndex::fixing(const Date& fixingDate,
210  bool /*forecastTodaysFixing*/) const {
211 
212  Date today = Settings::instance().evaluationDate();
213  Date todayMinusLag = today - availabilityLag_;
214  std::pair<Date,Date> lim = inflationPeriod(todayMinusLag, frequency_);
215  Date lastFix = lim.first-1;
216 
217  Date flatMustForecastOn = lastFix+1;
218  Date interpMustForecastOn = lastFix+1 - Period(frequency_);
219 
220 
221  if (interpolated() && fixingDate >= interpMustForecastOn) {
222  return forecastFixing(fixingDate);
223  }
224 
225  if (!interpolated() && fixingDate >= flatMustForecastOn) {
226  return forecastFixing(fixingDate);
227  }
228 
229  // four cases with ratio() and interpolated()
230 
231  const TimeSeries<Real>& ts = timeSeries();
232  if (ratio()) {
233 
234  if(interpolated()){ // IS ratio, IS interpolated
235 
236  std::pair<Date,Date> lim = inflationPeriod(fixingDate, frequency_);
237  Date fixMinus1Y=NullCalendar().advance(fixingDate, -1*Years, ModifiedFollowing);
238  std::pair<Date,Date> limBef = inflationPeriod(fixMinus1Y, frequency_);
239  Real dp= lim.second + 1 - lim.first;
240  Real dpBef=limBef.second + 1 - limBef.first;
241  Real dl = fixingDate-lim.first;
242  // potentially does not work on 29th Feb
243  Real dlBef = fixMinus1Y - limBef.first;
244  // get the four relevant fixings
245  // recall that they are stored flat for every day
246  Rate limFirstFix = ts[lim.first];
247  QL_REQUIRE(limFirstFix != Null<Rate>(),
248  "Missing " << name() << " fixing for "
249  << lim.first );
250  Rate limSecondFix = ts[lim.second+1];
251  QL_REQUIRE(limSecondFix != Null<Rate>(),
252  "Missing " << name() << " fixing for "
253  << lim.second+1 );
254  Rate limBefFirstFix = ts[limBef.first];
255  QL_REQUIRE(limBefFirstFix != Null<Rate>(),
256  "Missing " << name() << " fixing for "
257  << limBef.first );
258  Rate limBefSecondFix =
259  IndexManager::instance().getHistory(name())[limBef.second+1];
260  QL_REQUIRE(limBefSecondFix != Null<Rate>(),
261  "Missing " << name() << " fixing for "
262  << limBef.second+1 );
263 
264  Real linearNow = limFirstFix + (limSecondFix-limFirstFix)*dl/dp;
265  Real linearBef = limBefFirstFix + (limBefSecondFix-limBefFirstFix)*dlBef/dpBef;
266  Rate wasYES = linearNow / linearBef - 1.0;
267 
268  return wasYES;
269 
270  } else { // IS ratio, NOT interpolated
271  Rate pastFixing = ts[fixingDate];
272  QL_REQUIRE(pastFixing != Null<Rate>(),
273  "Missing " << name() << " fixing for "
274  << fixingDate);
275  Date previousDate = fixingDate - 1*Years;
276  Rate previousFixing = ts[previousDate];
277  QL_REQUIRE(previousFixing != Null<Rate>(),
278  "Missing " << name() << " fixing for "
279  << previousDate );
280 
281  return pastFixing/previousFixing - 1.0;
282  }
283 
284  } else { // NOT ratio
285 
286  if (interpolated()) { // NOT ratio, IS interpolated
287 
288  std::pair<Date,Date> lim = inflationPeriod(fixingDate, frequency_);
289  Real dp= lim.second + 1 - lim.first;
290  Real dl = fixingDate-lim.first;
291  Rate limFirstFix = ts[lim.first];
292  QL_REQUIRE(limFirstFix != Null<Rate>(),
293  "Missing " << name() << " fixing for "
294  << lim.first );
295  Rate limSecondFix = ts[lim.second+1];
296  QL_REQUIRE(limSecondFix != Null<Rate>(),
297  "Missing " << name() << " fixing for "
298  << lim.second+1 );
299  Real linearNow = limFirstFix + (limSecondFix-limFirstFix)*dl/dp;
300 
301  return linearNow;
302 
303  } else { // NOT ratio, NOT interpolated
304  // so just flat
305 
306  Rate pastFixing = ts[fixingDate];
307  QL_REQUIRE(pastFixing != Null<Rate>(),
308  "Missing " << name() << " fixing for "
309  << fixingDate);
310  return pastFixing;
311 
312  }
313  }
