115 inline void LazyObject::update() {
116 // forwards notifications only the first time
117 if (calculated_ || alwaysForward_) {
118 // set to false early
119 // 1) to prevent infinite recursion
120 // 2) otherways non-lazy observers would be served obsolete
121 // data because of calculated_ being still true
122 calculated_ = false;
123 // observers don't expect notifications from frozen objects
124 if (!frozen_)
125 notifyObservers();
126 // exiting notifyObservers() calculated_ could be
127 // already true because of non-lazy observers
128 }
129 }
130
131 inline void LazyObject::recalculate() {
132 bool wasFrozen = frozen_;
133 calculated_ = frozen_ = false;
134 try {
135 calculate();
136 } catch (...) {
137 frozen_ = wasFrozen;
138 notifyObservers();
139 throw;
140 }
141 frozen_ = wasFrozen;
142 notifyObservers();
143 }
144
145 inline void LazyObject::freeze() {
146 frozen_ = true;
147 }
148
149 inline void LazyObject::unfreeze() {
150 // send notifications, just in case we lost any,
151 // but only once, i.e. if it was frozen
152 if (frozen_) {
153 frozen_ = false;
154 notifyObservers();
155 }
156 }
157
158 inline void LazyObject::alwaysForwardNotifications() {
159 alwaysForward_ = true;
160 }
161
162 inline void LazyObject::calculate() const {
163 if (!calculated_ && !frozen_) {
164 calculated_ = true; // prevent infinite recursion in
165 // case of bootstrapping
166 try {
167 performCalculations();
168 } catch (...) {
169 calculated_ = false;
170 throw;
171 }
172 }
173 }
