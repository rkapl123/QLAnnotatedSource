136 inline void LazyObject::update() {
137 if (updating_) {
138 #ifdef QL_THROW_IN_CYCLES
139 QL_FAIL("recursive notification loop detected; you probably created an object cycle");
140 #else
141 return;
142 #endif
143 }
144
145 // This sets updating to true (so the above check breaks the
146 // infinite loop if we enter this method recursively) and will
147 // set it back to false when we exit this scope, either
148 // successfully or because of an exception.
149 UpdateChecker checker(this);
150
151 // forwards notifications only the first time
152 if (calculated_ || alwaysForward_) {
153 // set to false early
154 // 1) to prevent infinite recursion
155 // 2) otherways non-lazy observers would be served obsolete
156 // data because of calculated_ being still true
157 calculated_ = false;
158 // observers don't expect notifications from frozen objects
159 if (!frozen_)
160 notifyObservers();
161 // exiting notifyObservers() calculated_ could be
162 // already true because of non-lazy observers
163 }
164 }

166 inline void LazyObject::recalculate() {
167 bool wasFrozen = frozen_;
168 calculated_ = frozen_ = false;
169 try {
170 calculate();
171 } catch (...) {
172 frozen_ = wasFrozen;
173 notifyObservers();
174 throw;
175 }
176 frozen_ = wasFrozen;
177 notifyObservers();
178 }

201 inline void LazyObject::calculate() const {
