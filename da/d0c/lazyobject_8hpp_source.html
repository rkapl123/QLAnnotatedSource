118  inline void LazyObject::update() {
119  // forwards notifications only the first time
120  if (calculated_ || alwaysForward_) {
121  // set to false early
122  // 1) to prevent infinite recursion
123  // 2) otherways non-lazy observers would be served obsolete
124  // data because of calculated_ being still true
125  calculated_ = false;
126  // observers don't expect notifications from frozen objects
127  if (!frozen_)
128  notifyObservers();
129  // exiting notifyObservers() calculated_ could be
130  // already true because of non-lazy observers
131  }
132  }

134  inline void LazyObject::recalculate() {
135  bool wasFrozen = frozen_;
136  calculated_ = frozen_ = false;
137  try {
138  calculate();
139  } catch (...) {
140  frozen_ = wasFrozen;
141  notifyObservers();
142  throw;
143  }
144  frozen_ = wasFrozen;
145  notifyObservers();
146  }

152  inline void LazyObject::unfreeze() {
153  // send notifications, just in case we lost any,
154  // but only once, i.e. if it was frozen
