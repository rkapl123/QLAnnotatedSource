47  Concentrating1dMesher::Concentrating1dMesher(
48  Real start, Real end, Size size, const std::pair<Real, Real>& cPoints,
49  const bool requireCPoint)
50  : Fdm1dMesher(size) {
51 
52  QL_REQUIRE(end > start, "end must be larger than start");
53 
54  const Real cPoint = cPoints.first;
55  const Real density = cPoints.second == Null<Real>() ?
56  Null<Real>() : cPoints.second*(end - start);
57 
58  QL_REQUIRE(cPoint == Null<Real>() || (cPoint >= start && cPoint <= end),
59  "cPoint must be between start and end");
60  QL_REQUIRE(density == Null<Real>() || density > 0.0,
61  "density > 0 required");
62  QL_REQUIRE(cPoint == Null<Real>() || density != Null<Real>(),
63  "density must be given if cPoint is given");
64  QL_REQUIRE(!requireCPoint || cPoint != Null<Real>(),
65  "cPoint is required in grid but not given");
66 
67  const Real dx = 1.0 / (size - 1);
68 
69  if (cPoint != Null<Real>()) {
70  std::vector<Real> u, z;
71  ext::shared_ptr<Interpolation> transform;
72  const Real c1 = asinh((start - cPoint) / density);
73  const Real c2 = asinh((end - cPoint) / density);
74  if (requireCPoint) {
75  u.push_back(0.0);
76  z.push_back(0.0);
77  if (!close(cPoint, start) && !close(cPoint, end)) {
78  const Real z0 = -c1 / (c2 - c1);
79  const Real u0 =
80  std::max(
81  std::min(static_cast<int>(z0 * (size - 1) + 0.5),
82  static_cast<int>(size) - 2),
83  1) /
84  ((Real)(size - 1));
85  u.push_back(u0);
86  z.push_back(z0);
87  }
88  u.push_back(1.0);
89  z.push_back(1.0);
90  transform = ext::shared_ptr<Interpolation>(
91  new LinearInterpolation(u.begin(), u.end(), z.begin()));
92  }
93 
94  for (Size i = 1; i < size - 1; ++i) {
95  const Real li = requireCPoint ? (*transform)(i*dx) : i*dx;
96  locations_[i] = cPoint
97  + density*std::sinh(c1*(1.0 - li) + c2*li);
98  }
99  }
100  else {
101  for (Size i = 1; i < size - 1; ++i) {
102  locations_[i] = start + i*dx*(end - start);
103  }
104  }
105 
106  locations_.front() = start;
107  locations_.back() = end;
108 
109  for (Size i = 0; i < size - 1; ++i) {
110  dplus_[i] = dminus_[i + 1] = locations_[i + 1] - locations_[i];
111  }
112  dplus_.back() = dminus_.front() = Null<Real>();
113  }

123  Real solve(Real a, Real y0, Real x0, Real x1) {
124  using namespace ext::placeholders;

126  ext::bind(&OdeIntegrationFct::jac, this, a, _1, _2));

132  Real jac(Real a, Real, Real y) const {
133  Real s=0.0;
134  for (Size i=0; i < points_.size(); ++i) {
135  s+=1.0/(betas_[i] + square<Real>()(y - points_[i]));
136  }
137  return a/std::sqrt(s);
138  }

151  Concentrating1dMesher::Concentrating1dMesher(
152  Real start, Real end, Size size,
153  const std::vector<ext::tuple<Real, Real, bool> >& cPoints,
154  Real tol)
155  : Fdm1dMesher(size) {
156  using namespace ext::placeholders;
157 
158  QL_REQUIRE(end > start, "end must be larger than start");
159 
160  std::vector<Real> points, betas;
161  for (std::vector<ext::tuple<Real, Real, bool> >::const_iterator
162  iter = cPoints.begin(); iter != cPoints.end(); ++iter) {
163  points.push_back(ext::get<0>(*iter));
164  betas.push_back(square<Real>()(ext::get<1>(*iter)*(end-start)));
165  }
166 
167  // get scaling factor a so that y(1) = end
168  Real aInit = 0.0;
169  for (Size i=0; i < points.size(); ++i) {
170  const Real c1 = asinh((start-points[i])/betas[i]);
171  const Real c2 = asinh((end-points[i])/betas[i]);
172  aInit+=(c2-c1)/points.size();
173  }
174 
175  OdeIntegrationFct fct(points, betas, tol);
176  const Real a = Brent().solve(
177  ext::bind(std::minus<Real>(),
178  ext::bind(&OdeIntegrationFct::solve,
179  &fct, _1, start, 0.0, 1.0), end),
180  tol, aInit, 0.1*aInit);
181 
182  // solve ODE for all grid points
183  Array x(size), y(size);
184  x[0] = 0.0; y[0] = start;
185  const Real dx = 1.0/(size-1);
186  for (Size i=1; i < size; ++i) {
187  x[i] = i*dx;
188  y[i] = fct.solve(a, y[i-1], x[i-1], x[i]);
189  }
190 
191  // eliminate numerical noise and ensure y(1) = end
192  const Real dy = y.back() - end;
193  for (Size i=1; i < size; ++i) {
194  y[i] -= i*dx*dy;
195  }
196 
197  LinearInterpolation odeSolution(x.begin(), x.end(), y.begin());
198 
199  // ensure required points are part of the grid
200  std::vector<std::pair<Real, Real> > w(1, std::make_pair(0.0, 0.0));
201 
202  for (Size i=0; i < points.size(); ++i) {
203  if (ext::get<2>(cPoints[i]) && points[i] > start && points[i] < end) {
204 
205  const Size j = std::distance(y.begin(),
206  std::lower_bound(y.begin(), y.end(), points[i]));
207 
208  const Real e = Brent().solve(
209  ext::bind(std::minus<Real>(),
210  ext::bind(&LinearInterpolation::operator(),
211  odeSolution, _1, true), points[i]),
212  QL_EPSILON, x[j], 0.5/size);
213 
214  w.push_back(std::make_pair(std::min(x[size-2], x[j]), e));
215  }
216  }
217  w.push_back(std::make_pair(1.0, 1.0));
218  std::sort(w.begin(), w.end());
219  w.erase(std::unique(w.begin(), w.end(), equal_on_first), w.end());
220 
221  std::vector<Real> u(w.size()), z(w.size());
222  for (Size i=0; i < w.size(); ++i) {
223  u[i] = w[i].first;
224  z[i] = w[i].second;
225  }
226  LinearInterpolation transform(u.begin(), u.end(), z.begin());
227 
228  for (Size i=0; i < size; ++i) {
229  locations_[i] = odeSolution(transform(i*dx));
230  }
231 
232  for (Size i=0; i < size-1; ++i) {
233  dplus_[i] = dminus_[i+1] = locations_[i+1] - locations_[i];
234  }
235  dplus_.back() = dminus_.front() = Null<Real>();
236  }
