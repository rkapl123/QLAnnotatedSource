31 bool ExerciseAdapter::nextTimeStep(
32 const CurveState& currentState,
33 std::vector<Size>& numberCashFlowsThisStep,
34 std::vector<std::vector<MarketModelMultiProduct::CashFlow> >&
35 generatedCashFlows) {
36 std::fill(numberCashFlowsThisStep.begin(),
37 numberCashFlowsThisStep.end(), 0);
38 bool done = false;
39
40 exercise_->nextStep(currentState);
41 if (isExerciseTime_[currentIndex_]) {
42 MarketModelMultiProduct::CashFlow cashflow =
43 exercise_->value(currentState);
44 numberCashFlowsThisStep[0] = 1;
45 generatedCashFlows[0][0] = cashflow;
46 done = true;
47 }
48 ++currentIndex_;
49 return done || currentIndex_ == isExerciseTime_.size();
50 }
