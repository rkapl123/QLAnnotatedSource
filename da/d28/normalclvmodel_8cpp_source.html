39  NormalCLVModel::NormalCLVModel(
40  const ext::shared_ptr<GeneralizedBlackScholesProcess>& bsProcess,
41  const ext::shared_ptr<OrnsteinUhlenbeckProcess>& ouProcess,
42  const std::vector<Date>& maturityDates,
43  Size lagrangeOrder, Real pMax, Real pMin)
44  : x_(M_SQRT2*GaussHermiteIntegration(lagrangeOrder).x()),
45  sigma_( (pMax != Null<Real>())
46  ? x_.back() / InverseCumulativeNormal()(pMax)
47  : (pMin != Null<Real>())
48  ? x_.front() / InverseCumulativeNormal()(pMin)
49  : 1.0),
50  bsProcess_ (bsProcess),
51  ouProcess_ (ouProcess),
52  maturityDates_(maturityDates),
53  rndCalculator_(ext::make_shared<GBSMRNDCalculator>(bsProcess)),
54  maturityTimes_(maturityDates.size()) {
55 
56  registerWith(bsProcess_);
57  registerWith(ouProcess_);
58 
59  for (Size i=0; i < maturityTimes_.size(); ++i) {
60  maturityTimes_[i] = bsProcess_->time(maturityDates[i]);
61  QL_REQUIRE(i==0 || maturityTimes_[i-1] < maturityTimes_[i],
62  "dates must be sorted");
63  }
64  }

75  Disposable<Array> NormalCLVModel::collocationPointsX(const Date& d) const {
76  const Time t = bsProcess_->time(d);

78  const Real expectation
79  = ouProcess_->expectation(0.0, ouProcess_->x0(), t);

103  NormalCLVModel::MappingFunction::MappingFunction(

105  : y_(model.x_.size()),

107  ouProcess_(model.ouProcess_),
108  data_(ext::make_shared<InterpolationData>(model)) {

110  for (Size i=0; i < data_->s_.columns(); ++i) {
111  const Array y = model.collocationPointsY(model.maturityDates_[i]);
112  std::copy(y.begin(), y.end(), data_->s_.column_begin(i));

118  data_->s_.row_begin(i)));

123  Real NormalCLVModel::MappingFunction::operator()(Time t, Real x) const {
124  for (Size i=0; i < y_.size(); ++i) {
125  y_[i] = data_->interpl_[i](t, true);
126  }
127 
128  const Real expectation
129  = ouProcess_->expectation(0.0, ouProcess_->x0(), t);
130  const Real stdDeviation
131  = ouProcess_->stdDeviation(0.0, ouProcess_->x0(), t);
132 
133  const Real r = sigma_*(x-expectation)/stdDeviation;
134 
135  return data_->lagrangeInterpl_.value(y_, r);
136  }
