76 Real Integrator::operator()(const std::function<Real (Real)>& f,
77 Real a,
78 Real b) const {
79 evaluations_ = 0;
80 if (a == b)
81 return 0.0;
82 if (b > a)
83 return integrate(f, a, b);
84 else
85 return -integrate(f, b, a);
