108  bool LongstaffSchwartzExerciseStrategy::exercise(
109  const CurveState& currentState) const {
110 
111  Size exerciseIndex = exerciseIndex_[currentIndex_-1];
112 
113  CashFlow exerciseCF = exercise_->value(currentState);
114  Real exerciseValue = exerciseCF.amount *
115  rebateDiscounters_[exerciseCF.timeIndex]
116  .numeraireBonds(currentState,
117  numeraires_[currentIndex_-1]) /
118  principalInNumerairePortfolio_;
119 
120  CashFlow controlCF = control_->value(currentState);
121  Real controlValue = controlCF.amount *
122  controlDiscounters_[controlCF.timeIndex]
123  .numeraireBonds(currentState,
124  numeraires_[currentIndex_-1]) /
125  principalInNumerairePortfolio_;
126 
127  basisSystem_->values(currentState,
128  basisValues_[exerciseIndex]);
129 
130  const std::vector<Real>& alphas = basisCoefficients_[exerciseIndex];
131  Real continuationValue =
132  std::inner_product(alphas.begin(), alphas.end(),
133  basisValues_[exerciseIndex].begin(),
134  controlValue);
135 
136  return exerciseValue >= continuationValue;
137  }

139  void LongstaffSchwartzExerciseStrategy::nextStep(
140  const CurveState& currentState) {
141  principalInNumerairePortfolio_ = newPrincipal_;
142 
143  if (isRebateTime_[currentIndex_])
144  exercise_->nextStep(currentState);
145  if (isControlTime_[currentIndex_])
146  control_->nextStep(currentState);
147  if (isBasisTime_[currentIndex_])
148  basisSystem_->nextStep(currentState);
149 
150  if (currentIndex_ < numeraires_.size()-1) {
151  Size numeraire = numeraires_[currentIndex_];
152  Size nextNumeraire = numeraires_[currentIndex_+1];
153  newPrincipal_ *=
154  currentState.discountRatio(numeraire, nextNumeraire);
155  }
156 
157  ++currentIndex_;
158  }
