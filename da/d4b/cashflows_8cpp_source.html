38  Date CashFlows::startDate(const Leg& leg) {
39  QL_REQUIRE(!leg.empty(), "empty leg");

41  Date d = Date::maxDate();
42  for (const auto& i : leg) {
43  ext::shared_ptr<Coupon> c = ext::dynamic_pointer_cast<Coupon>(i);
44  if (c != nullptr)
45  d = std::min(d, c->accrualStartDate());
46  else
47  d = std::min(d, i->date());
48  }
49  return d;

178  template<typename Iter>
179  Rate aggregateRate(const Leg& leg,
180  Iter first,
181  Iter last) {
182  if (first==last) return 0.0;
183 
184  Date paymentDate = (*first)->date();
185  bool firstCouponFound = false;
186  Real nominal = 0.0;
187  Time accrualPeriod = 0.0;
188  DayCounter dc;
189  Rate result = 0.0;
190  for (; first<last && (*first)->date()==paymentDate; ++first) {
191  ext::shared_ptr<Coupon> cp = ext::dynamic_pointer_cast<Coupon>(*first);
192  if (cp) {
193  if (firstCouponFound) {
194  QL_REQUIRE(nominal == cp->nominal() &&
195  accrualPeriod == cp->accrualPeriod() &&
196  dc == cp->dayCounter(),
197  "cannot aggregate two different coupons on "
198  << paymentDate);
199  } else {
200  firstCouponFound = true;
201  nominal = cp->nominal();
202  accrualPeriod = cp->accrualPeriod();
203  dc = cp->dayCounter();
204  }
205  result += cp->rate();
206  }
207  }
208  QL_ENSURE(firstCouponFound,
209  "no coupon paid at cashflow date " << paymentDate);
210  return result;
211  }

399  class BPSCalculator : public AcyclicVisitor,
400  public Visitor<CashFlow>,
401  public Visitor<Coupon> {
402  public:
403  explicit BPSCalculator(const YieldTermStructure& discountCurve)
404  : discountCurve_(discountCurve), bps_(0.0), nonSensNPV_(0.0) {}
405  void visit(Coupon& c) override {
406  Real bps = c.nominal() *
407  c.accrualPeriod() *
408  discountCurve_.discount(c.date());
409  bps_ += bps;
410  }
411  void visit(CashFlow& cf) override {
412  nonSensNPV_ += cf.amount() *
413  discountCurve_.discount(cf.date());
414  }
415  Real bps() const { return bps_; }
416  Real nonSensNPV() const { return nonSensNPV_; }
417  private:
418  const YieldTermStructure& discountCurve_;
419  Real bps_, nonSensNPV_;
420  };

425  Real CashFlows::npv(const Leg& leg,
426  const YieldTermStructure& discountCurve,
427  bool includeSettlementDateFlows,
428  Date settlementDate,
429  Date npvDate) {
430 
431  if (leg.empty())
432  return 0.0;
433 
434  if (settlementDate == Date())
435  settlementDate = Settings::instance().evaluationDate();
436 
437  if (npvDate == Date())
438  npvDate = settlementDate;
439 
440  Real totalNPV = 0.0;
441  for (const auto& i : leg) {
442  if (!i->hasOccurred(settlementDate, includeSettlementDateFlows) &&
443  !i->tradingExCoupon(settlementDate))
444  totalNPV += i->amount() * discountCurve.discount(i->date());
445  }
446 
447  return totalNPV/discountCurve.discount(npvDate);
448  }

473  void CashFlows::npvbps(const Leg& leg,
474  const YieldTermStructure& discountCurve,
475  bool includeSettlementDateFlows,
476  Date settlementDate,
477  Date npvDate,
478  Real& npv,
479  Real& bps) {
480 
481  npv = 0.0;
482  if (leg.empty()) {
483  bps = 0.0;
484  return;
485  }
486 
487  for (const auto& i : leg) {
488  CashFlow& cf = *i;
489  if (!cf.hasOccurred(settlementDate,
490  includeSettlementDateFlows) &&
491  !cf.tradingExCoupon(settlementDate)) {
492  ext::shared_ptr<Coupon> cp = ext::dynamic_pointer_cast<Coupon>(i);
493  Real df = discountCurve.discount(cf.date());
494  npv += cf.amount() * df;
495  if (cp != nullptr)
496  bps += cp->nominal() * cp->accrualPeriod() * df;
497  }
498  }
499  DiscountFactor d = discountCurve.discount(npvDate);
500  npv /= d;
501  bps = basisPoint_ * bps / d;
502  }

504  Rate CashFlows::atmRate(const Leg& leg,
505  const YieldTermStructure& discountCurve,
506  bool includeSettlementDateFlows,
507  Date settlementDate,
508  Date npvDate,
509  Real targetNpv) {
510  if (leg.empty())
511  return 0.0;
512 
513  if (settlementDate == Date())
514  settlementDate = Settings::instance().evaluationDate();
515 
516  if (npvDate == Date())
517  npvDate = settlementDate;
518 
519  Real npv = 0.0;
520  BPSCalculator calc(discountCurve);
521  for (const auto& i : leg) {
522  CashFlow& cf = *i;
523  if (!cf.hasOccurred(settlementDate,
524  includeSettlementDateFlows) &&
525  !cf.tradingExCoupon(settlementDate)) {
526  npv += cf.amount() *
527  discountCurve.discount(cf.date());
528  cf.accept(calc);
529  }
530  }
531 
532  if (targetNpv==Null<Real>())
533  targetNpv = npv - calc.nonSensNPV();
534  else {
535  targetNpv *= discountCurve.discount(npvDate);
536  targetNpv -= calc.nonSensNPV();
537  }
538 
539  if (targetNpv==0.0)
540  return 0.0;
541 
542  Real bps = calc.bps();
543  QL_REQUIRE(bps!=0.0, "null bps: impossible atm rate");
544 
545  return targetNpv/bps;
546  }

562  // helper fucntion used to calculate Time-To-Discount for each stage when calculating discount factor stepwisely
563  Time getStepwiseDiscountTime(const ext::shared_ptr<QuantLib::CashFlow>& cashFlow,
564  const DayCounter& dc,
565  Date npvDate,
566  Date lastDate) {
567  Date cashFlowDate = cashFlow->date();
568  Date refStartDate, refEndDate;
569  ext::shared_ptr<Coupon> coupon =
570  ext::dynamic_pointer_cast<Coupon>(cashFlow);
571  if (coupon != nullptr) {
572  refStartDate = coupon->referencePeriodStart();
573  refEndDate = coupon->referencePeriodEnd();
574  } else {
575  if (lastDate == npvDate) {
576  // we don't have a previous coupon date,
577  // so we fake it
578  refStartDate = cashFlowDate - 1*Years;
579  } else {
580  refStartDate = lastDate;
581  }
582  refEndDate = cashFlowDate;
583  }
584 
585  if ((coupon != nullptr) && lastDate != coupon->accrualStartDate()) {
586  Time couponPeriod = dc.yearFraction(coupon->accrualStartDate(),
587  cashFlowDate, refStartDate, refEndDate);
588  Time accruedPeriod = dc.yearFraction(coupon->accrualStartDate(),
589  lastDate, refStartDate, refEndDate);
590  return couponPeriod - accruedPeriod;
591  } else {
592  return dc.yearFraction(lastDate, cashFlowDate,
593  refStartDate, refEndDate);
594  }
595  }

597  Real simpleDuration(const Leg& leg,
598  const InterestRate& y,
599  bool includeSettlementDateFlows,
600  Date settlementDate,
601  Date npvDate) {
602  if (leg.empty())
603  return 0.0;
604 
605  if (settlementDate == Date())
606  settlementDate = Settings::instance().evaluationDate();
607 
608  if (npvDate == Date())
609  npvDate = settlementDate;
610 
611  Real P = 0.0;
612  Real dPdy = 0.0;
613  Time t = 0.0;
614  Date lastDate = npvDate;
615  const DayCounter& dc = y.dayCounter();
616  for (const auto& i : leg) {
617  if (i->hasOccurred(settlementDate, includeSettlementDateFlows))
618  continue;
619 
620  Real c = i->amount();
621  if (i->tradingExCoupon(settlementDate)) {
622  c = 0.0;
623  }
624 
625  t += getStepwiseDiscountTime(i, dc, npvDate, lastDate);
626  DiscountFactor B = y.discountFactor(t);
627  P += c * B;
628  dPdy += t * c * B;
629 
630  lastDate = i->date();
631  }
632  if (P == 0.0) // no cashflows
633  return 0.0;
634  return dPdy/P;
635  }

637  Real modifiedDuration(const Leg& leg,
638  const InterestRate& y,
639  bool includeSettlementDateFlows,
640  Date settlementDate,
641  Date npvDate) {
642  if (leg.empty())
643  return 0.0;
644 
645  if (settlementDate == Date())
646  settlementDate = Settings::instance().evaluationDate();
647 
648  if (npvDate == Date())
649  npvDate = settlementDate;
650 
651  Real P = 0.0;
652  Time t = 0.0;
653  Real dPdy = 0.0;
654  Rate r = y.rate();
655  Natural N = y.frequency();
656  Date lastDate = npvDate;
657  const DayCounter& dc = y.dayCounter();
658  for (const auto& i : leg) {
659  if (i->hasOccurred(settlementDate, includeSettlementDateFlows))
660  continue;
661 
662  Real c = i->amount();
663  if (i->tradingExCoupon(settlementDate)) {
664  c = 0.0;
665  }
666 
667  t += getStepwiseDiscountTime(i, dc, npvDate, lastDate);
668  DiscountFactor B = y.discountFactor(t);
669  P += c * B;
670  switch (y.compounding()) {
671  case Simple:
672  dPdy -= c * B*B * t;
673  break;
674  case Compounded:
675  dPdy -= c * t * B/(1+r/N);
676  break;
677  case Continuous:
678  dPdy -= c * B * t;
679  break;
680  case SimpleThenCompounded:
681  if (t<=1.0/N)
682  dPdy -= c * B*B * t;
683  else
684  dPdy -= c * t * B/(1+r/N);
685  break;
686  case CompoundedThenSimple:
687  if (t>1.0/N)
688  dPdy -= c * B*B * t;
689  else
690  dPdy -= c * t * B/(1+r/N);
691  break;
692  default:
693  QL_FAIL("unknown compounding convention (" <<
694  Integer(y.compounding()) << ")");
695  }
696  lastDate = i->date();
697  }
698 
699  if (P == 0.0) // no cashflows

968  Real CashFlows::convexity(const Leg& leg,
969  const InterestRate& y,
970  bool includeSettlementDateFlows,
971  Date settlementDate,
972  Date npvDate) {
973  if (leg.empty())
974  return 0.0;
975 
976  if (settlementDate == Date())
977  settlementDate = Settings::instance().evaluationDate();
978 
979  if (npvDate == Date())
980  npvDate = settlementDate;
981 
982  const DayCounter& dc = y.dayCounter();
983 
984  Real P = 0.0;
985  Time t = 0.0;
986  Real d2Pdy2 = 0.0;
987  Rate r = y.rate();
988  Natural N = y.frequency();
989  Date lastDate = npvDate;
990  for (const auto& i : leg) {
991  if (i->hasOccurred(settlementDate, includeSettlementDateFlows))
992  continue;
993 
994  Real c = i->amount();
995  if (i->tradingExCoupon(settlementDate)) {
996  c = 0.0;
997  }
998 
999  t += getStepwiseDiscountTime(i, dc, npvDate, lastDate);
1000  DiscountFactor B = y.discountFactor(t);
1001  P += c * B;
1002  switch (y.compounding()) {
1003  case Simple:
1004  d2Pdy2 += c * 2.0*B*B*B*t*t;
1005  break;
1006  case Compounded:
1007  d2Pdy2 += c * B*t*(N*t+1)/(N*(1+r/N)*(1+r/N));
1008  break;
1009  case Continuous:
1010  d2Pdy2 += c * B*t*t;
1011  break;
1012  case SimpleThenCompounded:
1013  if (t<=1.0/N)
1014  d2Pdy2 += c * 2.0*B*B*B*t*t;
1015  else
1016  d2Pdy2 += c * B*t*(N*t+1)/(N*(1+r/N)*(1+r/N));
1017  break;
1018  case CompoundedThenSimple:
1019  if (t>1.0/N)
1020  d2Pdy2 += c * 2.0*B*B*B*t*t;
1021  else
1022  d2Pdy2 += c * B*t*(N*t+1)/(N*(1+r/N)*(1+r/N));
1023  break;
1024  default:
1025  QL_FAIL("unknown compounding convention (" <<
1026  Integer(y.compounding()) << ")");
1027  }
1028  lastDate = i->date();
1029  }
1030 
1031  if (P == 0.0)
1032  // no cashflows
1033  return 0.0;
1034 
1035  return d2Pdy2/P;
1036  }
