83  template <class T>
84  void TsiveriotisFernandesLattice<T>::stepback(
85  Size i,
86  const Array& values,
87  const Array& conversionProbability,
88  const Array& spreadAdjustedRate,
89  Array& newValues,
90  Array& newConversionProbability,
91  Array& newSpreadAdjustedRate) const {
92 
93  for (Size j=0; j<this->size(i); j++) {
94 
95  // new conversion probability is calculated via backward
96  // induction using up and down probabilities on tree on
97  // previous conversion probabilities, ie weighted average
98  // of previous probabilities.
99  newConversionProbability[j] =
100  this->pd_*conversionProbability[j] +
101  this->pu_*conversionProbability[j+1];
102 
103  // Use blended discounting rate
104  newSpreadAdjustedRate[j] =
105  newConversionProbability[j] * this->riskFreeRate_ +
106  (1-newConversionProbability[j])*(this->riskFreeRate_+creditSpread_);
107 
108  newValues[j] =
109  (this->pd_*values[j]/(1+(spreadAdjustedRate[j]*this->dt_)))
110  + (this->pu_*values[j+1]/(1+(spreadAdjustedRate[j+1]*this->dt_)));
111 
112  }
113  }

123  template <class T>
124  void TsiveriotisFernandesLattice<T>::partialRollback(DiscretizedAsset& asset,
125  Time to) const {
126 
127  Time from = asset.time();
128 
129  if (close(from,to))
130  return;
131 
132  QL_REQUIRE(from > to,
133  "cannot roll the asset back to" << to
134  << " (it is already at t = " << from << ")");
135 
136  auto& convertible = dynamic_cast<DiscretizedConvertible&>(asset);
137 
138  auto iFrom = Integer(this->t_.index(from));
139  auto iTo = Integer(this->t_.index(to));
140 
141  for (Integer i=iFrom-1; i>=iTo; --i) {
142 
143  Array newValues(this->size(i));
144  Array newSpreadAdjustedRate(this->size(i));
145  Array newConversionProbability(this->size(i));
146 
147  stepback(i, convertible.values(),
148  convertible.conversionProbability(),
149  convertible.spreadAdjustedRate(), newValues,
150  newConversionProbability,newSpreadAdjustedRate);
151 
152  convertible.time() = this->t_[i];
153  convertible.values() = newValues;
154  convertible.spreadAdjustedRate() = newSpreadAdjustedRate;
155  convertible.conversionProbability() = newConversionProbability;
156 
157  // skip the very last adjustment
158  if (i != iTo)
159  convertible.adjustValues();
160  }
161  }
