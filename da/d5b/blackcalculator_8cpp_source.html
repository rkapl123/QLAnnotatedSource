67  void BlackCalculator::initialize(const shared_ptr<StrikedTypePayoff>& p) {
68  QL_REQUIRE(strike_>=0.0,
69  "strike (" << strike_ << ") must be non-negative");
70  QL_REQUIRE(forward_>0.0,
71  "forward (" << forward_ << ") must be positive");
72  //QL_REQUIRE(displacement_>=0.0,
73  // "displacement (" << displacement_ << ") must be non-negative");
74  QL_REQUIRE(stdDev_>=0.0,
75  "stdDev (" << stdDev_ << ") must be non-negative");
76  QL_REQUIRE(discount_>0.0,
77  "discount (" << discount_ << ") must be positive");
78 
79  if (stdDev_>=QL_EPSILON) {
80  if (close(strike_, 0.0)) {
81  d1_ = QL_MAX_REAL;
82  d2_ = QL_MAX_REAL;
83  cum_d1_ = 1.0;
84  cum_d2_ = 1.0;
85  n_d1_ = 0.0;
86  n_d2_ = 0.0;
87  } else {
88  d1_ = std::log(forward_/strike_)/stdDev_ + 0.5*stdDev_;
89  d2_ = d1_-stdDev_;
90  CumulativeNormalDistribution f;
91  cum_d1_ = f(d1_);
92  cum_d2_ = f(d2_);
93  n_d1_ = f.derivative(d1_);
94  n_d2_ = f.derivative(d2_);
95  }
96  } else {
97  if (close(forward_, strike_)) {
98  d1_ = 0;
99  d2_ = 0;
100  cum_d1_ = 0.5;
101  cum_d2_ = 0.5;
102  n_d1_ = M_SQRT_2 * M_1_SQRTPI;
103  n_d2_ = M_SQRT_2 * M_1_SQRTPI;
104  } else if (forward_>strike_) {
105  d1_ = QL_MAX_REAL;
106  d2_ = QL_MAX_REAL;
107  cum_d1_ = 1.0;
108  cum_d2_ = 1.0;
109  n_d1_ = 0.0;
110  n_d2_ = 0.0;
111  } else {
112  d1_ = QL_MIN_REAL;
113  d2_ = QL_MIN_REAL;
114  cum_d1_ = 0.0;
115  cum_d2_ = 0.0;
116  n_d1_ = 0.0;
117  n_d2_ = 0.0;
118  }
119  }
120 
121  x_ = strike_;
122  DxDstrike_ = 1.0;
123 
124  // the following one will probably disappear as soon as
125  // super-share will be properly handled
126  DxDs_ = 0.0;
127 
128  // this part is always executed.
129  // in case of plain-vanilla payoffs, it is also the only part
130  // which is executed.
131  switch (p->optionType()) {
132  case Option::Call:
133  alpha_ = cum_d1_;// N(d1)
134  DalphaDd1_ = n_d1_;// n(d1)
135  beta_ = -cum_d2_;// -N(d2)
136  DbetaDd2_ = - n_d2_;// -n(d2)
137  break;
138  case Option::Put:
139  alpha_ = -1.0+cum_d1_;// -N(-d1)
140  DalphaDd1_ = n_d1_;// n( d1)
141  beta_ = 1.0-cum_d2_;// N(-d2)
142  DbetaDd2_ = - n_d2_;// -n( d2)
143  break;
144  default:
145  QL_FAIL("invalid option type");
146  }
147 
148  // now dispatch on type.
149 
150  Calculator calc(*this);
151  p->accept(calc);
152  }

160  void BlackCalculator::Calculator::visit(CashOrNothingPayoff& payoff) {
161  black_.alpha_ = black_.DalphaDd1_ = 0.0;
162  black_.x_ = payoff.cashPayoff();
163  black_.DxDstrike_ = 0.0;
164  switch (payoff.optionType()) {
165  case Option::Call:
166  black_.beta_ = black_.cum_d2_;
167  black_.DbetaDd2_ = black_.n_d2_;
168  break;
169  case Option::Put:
170  black_.beta_ = 1.0-black_.cum_d2_;
171  black_.DbetaDd2_ = -black_.n_d2_;
172  break;
173  default:
174  QL_FAIL("invalid option type");
175  }
176  }

204  Real BlackCalculator::delta(Real spot) const {
205 
206  QL_REQUIRE(spot > 0.0, "positive spot value required: " <<
207  spot << " not allowed");
208 
209  Real DforwardDs = forward_ / spot;
210 
211  Real temp = stdDev_*spot;
212  Real DalphaDs = DalphaDd1_/temp;
213  Real DbetaDs = DbetaDd2_/temp;
214  Real temp2 = DalphaDs * forward_ + alpha_ * DforwardDs
215  +DbetaDs * x_ + beta_ * DxDs_;
216 
217  return discount_ * temp2;
218  }

224  Real DbetaDforward = DbetaDd2_/temp;

226  +DbetaDforward * x_; // DXDforward = 0.0

228  return discount_ * temp2;

231  Real BlackCalculator::elasticity(Real spot) const {
232  Real val = value();
233  Real del = delta(spot);
234  if (val>QL_EPSILON)
235  return del/val*spot;
236  else if (std::fabs(del)<QL_EPSILON)
237  return 0.0;
238  else if (del>0.0)
239  return QL_MAX_REAL;
240  else
241  return QL_MIN_REAL;

257  Real BlackCalculator::gamma(Real spot) const {
258 
259  QL_REQUIRE(spot > 0.0, "positive spot value required: " <<
260  spot << " not allowed");
261 
262  Real DforwardDs = forward_ / spot;
263 
264  Real temp = stdDev_*spot;
265  Real DalphaDs = DalphaDd1_/temp;
266  Real DbetaDs = DbetaDd2_/temp;
267 
268  Real D2alphaDs2 = - DalphaDs/spot*(1+d1_/stdDev_);
269  Real D2betaDs2 = - DbetaDs /spot*(1+d2_/stdDev_);
270 
271  Real temp2 = D2alphaDs2 * forward_ + 2.0 * DalphaDs * DforwardDs
272  +D2betaDs2 * x_ + 2.0 * DbetaDs * DxDs_;
273 
274  return discount_ * temp2;
275  }

277  Real BlackCalculator::gammaForward() const {
278 
279  Real temp = stdDev_*forward_;
280  Real DalphaDforward = DalphaDd1_/temp;
281  Real DbetaDforward = DbetaDd2_/temp;
282 
283  Real D2alphaDforward2 = - DalphaDforward/forward_*(1+d1_/stdDev_);
284  Real D2betaDforward2 = - DbetaDforward /forward_*(1+d2_/stdDev_);
285 
286  Real temp2 = D2alphaDforward2 * forward_ + 2.0 * DalphaDforward
287  +D2betaDforward2 * x_; // DXDforward = 0.0
288 
289  return discount_ * temp2;

292  Real BlackCalculator::theta(Real spot,
293  Time maturity) const {
294 
295  QL_REQUIRE(maturity>=0.0,
296  "maturity (" << maturity << ") must be non-negative");
297  if (close(maturity, 0.0)) return 0.0;
298  return -( std::log(discount_) * value()
299  +std::log(forward_/spot) * spot * delta(spot)
300  +0.5*variance_ * spot * spot * gamma(spot))/maturity;
301  }

303  Real BlackCalculator::vega(Time maturity) const {
304  QL_REQUIRE(maturity>=0.0,
305  "negative maturity not allowed");
306 
307  Real temp = std::log(strike_/forward_)/variance_;
308  // actually DalphaDsigma / SQRT(T)
309  Real DalphaDsigma = DalphaDd1_*(temp+0.5);
310  Real DbetaDsigma = DbetaDd2_ *(temp-0.5);
311 
312  Real temp2 = DalphaDsigma * forward_ + DbetaDsigma * x_;
313 
314  return discount_ * std::sqrt(maturity) * temp2;
315 
316  }

318  Real BlackCalculator::rho(Time maturity) const {
319  QL_REQUIRE(maturity>=0.0,
320  "negative maturity not allowed");
321 
322  // actually DalphaDr / T
323  Real DalphaDr = DalphaDd1_/stdDev_;
324  Real DbetaDr = DbetaDd2_/stdDev_;
325  Real temp = DalphaDr * forward_ + alpha_ * forward_ + DbetaDr * x_;
326 
327  return maturity * (discount_ * temp - value());
