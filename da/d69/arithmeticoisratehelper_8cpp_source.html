89  Real ArithmeticOISRateHelper::impliedQuote() const {
90  QL_REQUIRE(termStructure_ != nullptr, "term structure not set");
91  // we didn't register as observers - force calculation
92  swap_->recalculate();
93  //return swap_->fairRate();
94  // weak implementation... to be improved
95  static const Spread basisPoint = 1.0e-4;
96  Real floatingLegNPV = swap_->overnightLegNPV();
97  Spread spread = spread_.empty() ? 0.0 : spread_->value();
98  Real spreadNPV = swap_->overnightLegBPS() / basisPoint*spread;
99  Real totNPV = -(floatingLegNPV + spreadNPV);
100  Real result = totNPV / (swap_->fixedLegBPS() / basisPoint);
101  return result;
102  }
