128  template <class Interpolator>
129  InterpolatedZeroInflationCurve<Interpolator>::InterpolatedZeroInflationCurve(
130  const Date& referenceDate,
131  const Calendar& calendar,
132  const DayCounter& dayCounter,
133  const Period& lag,
134  Frequency frequency,
135  bool indexIsInterpolated,
136  std::vector<Date> dates,
137  const std::vector<Rate>& rates,
138  const Interpolator& interpolator)
139  : ZeroInflationTermStructure(
140  referenceDate, calendar, dayCounter, rates[0], lag, frequency, indexIsInterpolated),
141  InterpolatedCurve<Interpolator>(std::vector<Time>(), rates, interpolator),
142  dates_(std::move(dates)) {
143 
144  QL_REQUIRE(dates_.size() > 1, "too few dates: " << dates_.size());
145 
146  // check that the data starts from the beginning,
147  // i.e. referenceDate - lag, at least must be in the relevant
148  // period
149  std::pair<Date, Date> lim =
150  inflationPeriod(referenceDate - this->observationLag(), frequency);
151  QL_REQUIRE(lim.first <= dates_[0] && dates_[0] <= lim.second,
152  "first data date is not in base period, date: "
153  << dates_[0] << " not within [" << lim.first << "," << lim.second << "]");
154 
155  // by convention, if the index is not interpolated we pull all the dates
156  // back to the start of their inflationPeriods
157  // otherwise the time calculations will be inconsistent
158  if (!indexIsInterpolated_) {
159  for (auto& date : dates_) {
160  date = inflationPeriod(date, frequency).first;
161  }
162  }
163 
164  QL_REQUIRE(this->data_.size() == dates_.size(),
165  "indices/dates count mismatch: " << this->data_.size() << " vs "
166  << dates_.size());
167 
168  this->times_.resize(dates_.size());
169  this->times_[0] = timeFromReference(dates_[0]);
170  for (Size i = 1; i < dates_.size(); i++) {
171  QL_REQUIRE(dates_[i] > dates_[i - 1], "dates not sorted");
172 
173  // but must be greater than -1
174  QL_REQUIRE(this->data_[i] > -1.0, "zero inflation data < -100 %");
175 
176  // this can be negative
177  this->times_[i] = timeFromReference(dates_[i]);
178  QL_REQUIRE(!close(this->times_[i], this->times_[i - 1]),
179  "two dates correspond to the same time "
180  "under this curve's day count convention");
181  }
182 
183  this->interpolation_ = this->interpolator_.interpolate(
184  this->times_.begin(), this->times_.end(), this->data_.begin());
185  this->interpolation_.update();
186  }
