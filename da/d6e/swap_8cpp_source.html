29  Swap::Swap(const Leg& firstLeg,
30  const Leg& secondLeg)
31  : legs_(2), payer_(2),
32  legNPV_(2, 0.0), legBPS_(2, 0.0),
33  startDiscounts_(2, 0.0), endDiscounts_(2, 0.0),

45  Swap::Swap(const std::vector<Leg>& legs,
46  const std::vector<bool>& payer)
47  : legs_(legs), payer_(legs.size(), 1.0),
48  legNPV_(legs.size(), 0.0), legBPS_(legs.size(), 0.0),
49  startDiscounts_(legs.size(), 0.0), endDiscounts_(legs.size(), 0.0),
50  npvDateDiscount_(0.0) {
51  QL_REQUIRE(payer.size()==legs_.size(),
52  "size mismatch between payer (" << payer.size() <<
53  ") and legs (" << legs_.size() << ")");
54  for (Size j=0; j<legs_.size(); ++j) {
55  if (payer[j]) payer_[j]=-1.0;
56  for (Leg::iterator i = legs_[j].begin(); i!= legs_[j].end(); ++i)
57  registerWith(*i);
58  }
59  }

68  bool Swap::isExpired() const {
69  for (Size j=0; j<legs_.size(); ++j) {
70  Leg::const_iterator i;
71  for (i = legs_[j].begin(); i!= legs_[j].end(); ++i)
72  if (!(*i)->hasOccurred())
73  return false;
74  }
75  return true;
76  }

143  Date Swap::startDate() const {
144  QL_REQUIRE(!legs_.empty(), "no legs given");
145  Date d = CashFlows::startDate(legs_[0]);
146  for (Size j=1; j<legs_.size(); ++j)
147  d = std::min(d, CashFlows::startDate(legs_[j]));
148  return d;

151  Date Swap::maturityDate() const {
152  QL_REQUIRE(!legs_.empty(), "no legs given");
153  Date d = CashFlows::maturityDate(legs_[0]);
154  for (Size j=1; j<legs_.size(); ++j)
155  d = std::max(d, CashFlows::maturityDate(legs_[j]));
