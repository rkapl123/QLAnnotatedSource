47  void FdSimpleBSSwingEngine::calculate() const {
48  QL_REQUIRE(arguments_.exercise->type() == Exercise::Bermudan,
49  "Bermudan exercise supported only");
50 
51  // 1. Mesher
52  const boost::shared_ptr<StrikedTypePayoff> payoff =
53  boost::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
54  QL_REQUIRE(payoff, "Strike type payoff expected");
55 
56  const Time maturity = process_->time(arguments_.exercise->lastDate());
57  const boost::shared_ptr<Fdm1dMesher> equityMesher(
58  new FdmBlackScholesMesher(xGrid_, process_,
59  maturity, payoff->strike()));
60 
61  const boost::shared_ptr<Fdm1dMesher> exerciseMesher(
62  new Uniform1dMesher(
63  0, static_cast<Real>(arguments_.maxExerciseRights),
64  arguments_.maxExerciseRights+1));
65 
66  const boost::shared_ptr<FdmMesher> mesher (
67  new FdmMesherComposite(equityMesher, exerciseMesher));
68 
69  // 2. Calculator
70  boost::shared_ptr<FdmInnerValueCalculator> calculator(
71  new FdmZeroInnerValue());
72 
73  // 3. Step conditions
74  std::list<boost::shared_ptr<StepCondition<Array> > > stepConditions;
75  std::list<std::vector<Time> > stoppingTimes;
76 
77  // 3.1 Bermudan step conditions
78  std::vector<Time> exerciseTimes;
79  for (Size i=0; i<arguments_.exercise->dates().size(); ++i) {
80  Time t = process_->time(arguments_.exercise->dates()[i]);
81  QL_REQUIRE(t >= 0, "exercise dates must not contain past date");
82  exerciseTimes.push_back(t);
83  }
84  stoppingTimes.push_back(exerciseTimes);
85 
86  boost::shared_ptr<FdmInnerValueCalculator> exerciseCalculator(
87  new FdmLogInnerValue(payoff, mesher, 0));
88 
89  stepConditions.push_back(boost::shared_ptr<StepCondition<Array> >(
90  new FdmSimpleSwingCondition(
91  exerciseTimes, mesher, exerciseCalculator,
92  1, arguments_.minExerciseRights)));
93 
94  boost::shared_ptr<FdmStepConditionComposite> conditions(
95  new FdmStepConditionComposite(stoppingTimes, stepConditions));
96 
97  // 4. Boundary conditions
98  const FdmBoundaryConditionSet boundaries;
99 
100  // 5. Solver
101  FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
102  calculator, maturity, tGrid_, 0 };
103  boost::shared_ptr<FdmSimple2dBSSolver> solver(
104  new FdmSimple2dBSSolver(
105  Handle<GeneralizedBlackScholesProcess>(process_),
106  payoff->strike(), solverDesc, schemeDesc_));
107 
108  const Real spot = process_->x0();
109 
110  results_.value = solver->valueAt(spot, 1);
111  results_.delta = solver->deltaAt(spot, 1, spot*0.01);
112  results_.gamma = solver->gammaAt(spot, 1, spot*0.01);
113  results_.theta = solver->thetaAt(spot, 1);
114  }
115 }
