63  Real LiborForwardModel::discountBondOption(Option::Type type,
64  Real strike, Time maturity,
65  Time bondMaturity) const {
66 
67  const std::vector<Time> & accrualStartTimes
68  = process_->accrualStartTimes();
69  const std::vector<Time> & accrualEndTimes
70  = process_->accrualEndTimes();
71 
72  QL_REQUIRE( accrualStartTimes.front()<= maturity
73  && accrualStartTimes.back() >= maturity,
74  "capet maturity does not fit to the process");
75 
76  const Size i = std::lower_bound(accrualStartTimes.begin(),
77  accrualStartTimes.end(),
78  maturity) - accrualStartTimes.begin();
79 
80  QL_REQUIRE( i<process_->size()
81  && std::fabs(maturity - accrualStartTimes[i])
82  < 100*std::numeric_limits<Real>::epsilon()
83  && std::fabs(bondMaturity - accrualEndTimes[i])
84  < 100*std::numeric_limits<Real>::epsilon(),
85  "irregular fixings are not (yet) supported");
86 
87  const Real tenor = accrualEndTimes[i] - accrualStartTimes[i];
88  const Real forward = process_->initialValues()[i];
89  const Real capRate = (1.0/strike - 1.0)/tenor;
90  const Volatility var = covarProxy_
91  ->integratedCovariance(i, i, process_->fixingTimes()[i]);
92  const DiscountFactor dis =
93  process_->index()->forwardingTermStructure()->discount(bondMaturity);
94 
95  const Real black = blackFormula(
96  (type==Option::Put ? Option::Call : Option::Put),
97  capRate, forward, std::sqrt(var));
98 
99  const Real npv = dis * tenor * black;
100 
101  return npv / (1.0 + capRate*tenor);
102  }

104  Disposable<Array> LiborForwardModel::w_0(Size alpha, Size beta) const {
105  Array omega(beta+1,0.0);
106  QL_REQUIRE(alpha<beta, "alpha needs to be smaller than beta");
107 
108  Real s=0.0;
109  for (Size k=alpha+1; k<=beta; ++k) {
110  Real b=accrualPeriod_[k];
111  for (Size j=alpha+1; j<=k; ++j) {
112  b*=f_[j];
113  }
114  s+=b;
115  }
116 
117  for (Size i=alpha+1; i<=beta; ++i) {
118  Real a=accrualPeriod_[i];
119  for (Size j=alpha+1; j<=i; ++j) {
120  a*=f_[j];
121  }
122 
123  omega[i] = a/s;
124  }
125 
126  return omega;
127  }

129  Rate LiborForwardModel::S_0(Size alpha, Size beta) const {
130  const Array w = w_0(alpha, beta);
131  const Array f = process_->initialValues();
132 
133  Rate fwdRate=0.0;
134  for (Size i=alpha+1; i <=beta; ++i) {
135  fwdRate+=w[i]*f[i];
136  }
137 
138  return fwdRate;
139  }

141  // calculating swaption volatility matrix using
142  // Rebonatos approx. formula. Be aware that this
143  // matrix is valid only for regular fixings and
144  // assumes that the fix and floating leg have the
145  // same frequency
146  boost::shared_ptr<SwaptionVolatilityMatrix>
147  LiborForwardModel::getSwaptionVolatilityMatrix() const {
148  if (swaptionVola) {
149  return swaptionVola;
150  }
151 
152  const boost::shared_ptr<IborIndex> index = process_->index();
153  const Date today = process_->fixingDates()[0];
154 
155  const Size size=process_->size()/2;
156  Matrix volatilities(size, size);
157 
158  std::vector<Date> exercises(process_->fixingDates().begin() + 1,
159  process_->fixingDates().begin() + size+1);
160 
161  std::vector<Period> lengths(size);
162  for (Size i=0; i < size; ++i) {
163  lengths[i] = (i+1)*index->tenor();
164  }
165 
166  const Array f = process_->initialValues();
167  for (Size k=0; k < size; ++k) {
168  const Size alpha =k;
169  const Time t_alpha=process_->fixingTimes()[alpha+1];
170 
171  Matrix var(size, size);
172  for (Size i=alpha+1; i <= k+size; ++i) {
173  for (Size j=i; j <= k+size; ++j) {
174  var[i-alpha-1][j-alpha-1] = var[j-alpha-1][i-alpha-1] =
175  covarProxy_->integratedCovariance(i, j, t_alpha);
176  }
177  }
178 
179  for (Size l=1; l <= size; ++l) {
180  const Size beta =l + k;
181  const Array w = w_0(alpha, beta);
182 
183  Real sum=0.0;
184  for (Size i=alpha+1; i <= beta; ++i) {
185  for (Size j=alpha+1; j <= beta; ++j) {
186  sum+=w[i]*w[j]*f[i]*f[j]*var[i-alpha-1][j-alpha-1];
187  }
188  }
189  volatilities[k][l-1] =
190  std::sqrt(sum/t_alpha)/S_0(alpha, beta);
191  }
192  }
193 
194  return swaptionVola = boost::shared_ptr<SwaptionVolatilityMatrix>(
195  new SwaptionVolatilityMatrix(today, exercises, lengths,
196  volatilities,
197  index->dayCounter()));
198  }
