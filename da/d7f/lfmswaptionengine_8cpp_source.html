37  void LfmSwaptionEngine::calculate() const {
38 
39  QL_REQUIRE(arguments_.settlementMethod != Settlement::ParYieldCurve,
40  "cash settled (ParYieldCurve) swaptions not priced with Lfm engine");
41 
42  static const Spread basisPoint = 1.0e-4;
43 
44  VanillaSwap swap = *arguments_.swap;
45  swap.setPricingEngine(ext::shared_ptr<PricingEngine>(
46  new DiscountingSwapEngine(discountCurve_, false)));
47 
48  Spread correction = swap.spread() *
49  std::fabs(swap.floatingLegBPS()/swap.fixedLegBPS());
50  Rate fixedRate = swap.fixedRate() - correction;
51  Rate fairRate = swap.fairRate() - correction;
52 
53  ext::shared_ptr<SwaptionVolatilityMatrix> volatility =
54  model_->getSwaptionVolatilityMatrix();
55 
56  Date referenceDate = volatility->referenceDate();
57  DayCounter dayCounter = volatility->dayCounter();
58 
59  Time exercise = dayCounter.yearFraction(referenceDate,
60  arguments_.exercise->date(0));
61  Time swapLength =
62  dayCounter.yearFraction(referenceDate,
63  arguments_.fixedPayDates.back())
64  - dayCounter.yearFraction(referenceDate,
65  arguments_.fixedResetDates[0]);
66 
67  Option::Type w = arguments_.type==VanillaSwap::Payer ?
68  Option::Call : Option::Put;
69  Volatility vol = volatility->volatility(exercise, swapLength,
70  fairRate, true);
71  results_.value = (swap.fixedLegBPS()/basisPoint) *
72  blackFormula(w, fixedRate, fairRate, vol*std::sqrt(exercise));
73  }
74 
75 }
