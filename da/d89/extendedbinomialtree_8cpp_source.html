110  ExtendedTian::ExtendedTian(
111  const boost::shared_ptr<StochasticProcess1D>& process,
112  Time end, Size steps, Real)
113  : ExtendedBinomialTree<ExtendedTian>(process, end, steps) {
114 
115  Real q = std::exp(process->variance(0.0, x0_, dt_));
116 
117  Real r = std::exp(this->driftStep(0.0))*std::sqrt(q);
118 
119  up_ = 0.5 * r * q * (q + 1 + std::sqrt(q * q + 2 * q - 3));
120  down_ = 0.5 * r * q * (q + 1 - std::sqrt(q * q + 2 * q - 3));
121 
122  pu_ = (r - down_) / (up_ - down_);
123  pd_ = 1.0 - pu_;
124 
125  // doesn't work
126  // treeCentering_ = (up_+down_)/2.0;
127  // up_ = up_-treeCentering_;
128 
129  QL_REQUIRE(pu_<=1.0, "negative probability");
130  QL_REQUIRE(pu_>=0.0, "negative probability");
131  }

163  ExtendedLeisenReimer::ExtendedLeisenReimer(
164  const boost::shared_ptr<StochasticProcess1D>& process,
165  Time end, Size steps, Real strike)
166  : ExtendedBinomialTree<ExtendedLeisenReimer>(process, end,
167  (steps%2 ? steps : steps+1)),
168  end_(end), oddSteps_(steps%2 ? steps : steps+1), strike_(strike) {
169 
170  QL_REQUIRE(strike>0.0, "strike " << strike << "must be positive");
171  Real variance = process->variance(0.0, x0_, end);
172 
173  Real ermqdt = std::exp(this->driftStep(0.0) + 0.5*variance/oddSteps_);
174  Real d2 = (std::log(x0_/strike) + this->driftStep(0.0)*oddSteps_ ) /
175  std::sqrt(variance);
176 
177  pu_ = PeizerPrattMethod2Inversion(d2, oddSteps_);
178  pd_ = 1.0 - pu_;
179  Real pdash = PeizerPrattMethod2Inversion(d2+std::sqrt(variance),
180  oddSteps_);
181  up_ = ermqdt * pdash / pu_;
182  down_ = (ermqdt - pu_ * up_) / (1.0 - pu_);
183 
184  }

217  Real ExtendedJoshi4::computeUpProb(Real k, Real dj) const {
218  Real alpha = dj/(std::sqrt(8.0));
219  Real alpha2 = alpha*alpha;
220  Real alpha3 = alpha*alpha2;
221  Real alpha5 = alpha3*alpha2;
222  Real alpha7 = alpha5*alpha2;
223  Real beta = -0.375*alpha-alpha3;
224  Real gamma = (5.0/6.0)*alpha5 + (13.0/12.0)*alpha3
225  +(25.0/128.0)*alpha;
226  Real delta = -0.1025 *alpha- 0.9285 *alpha3
227  -1.43 *alpha5 -0.5 *alpha7;
228  Real p =0.5;
229  Real rootk= std::sqrt(k);
230  p+= alpha/rootk;
231  p+= beta /(k*rootk);
232  p+= gamma/(k*k*rootk);
233  // delete next line to get results for j three tree
234  p+= delta/(k*k*k*rootk);
235  return p;
236  }
