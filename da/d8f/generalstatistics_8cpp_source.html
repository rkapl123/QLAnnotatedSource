43  Real GeneralStatistics::variance() const {
44  Size N = samples();
45  QL_REQUIRE(N > 1,
46  "sample number <=1, unsufficient");
47  // Subtract the mean and square. Repeat on the whole range.
48  // Hopefully, the whole thing will be inlined in a single loop.
49  Real s2 = expectationValue(compose(square<Real>(),
50  subtract<Real>(mean())),
51  everywhere()).first;
52  return s2*N/(N-1.0);
53  }

55  Real GeneralStatistics::skewness() const {
56  Size N = samples();
57  QL_REQUIRE(N > 2,
58  "sample number <=2, unsufficient");
59 
60  Real x = expectationValue(compose(cube<Real>(),
61  subtract<Real>(mean())),
62  everywhere()).first;
63  Real sigma = standardDeviation();
64 
65  return (x/(sigma*sigma*sigma))*(N/(N-1.0))*(N/(N-2.0));
66  }

68  Real GeneralStatistics::kurtosis() const {
69  Size N = samples();
70  QL_REQUIRE(N > 3,
71  "sample number <=3, unsufficient");
72 
73  Real x = expectationValue(compose(fourth_power<Real>(),
74  subtract<Real>(mean())),
75  everywhere()).first;
76  Real sigma2 = variance();
77 
78  Real c1 = (N/(N-1.0)) * (N/(N-2.0)) * ((N+1.0)/(N-3.0));
79  Real c2 = 3.0 * ((N-1.0)/(N-2.0)) * ((N-1.0)/(N-3.0));
80 
81  return c1*(x/(sigma2*sigma2))-c2;
82  }

84  Real GeneralStatistics::percentile(Real percent) const {
85 
86  QL_REQUIRE(percent > 0.0 && percent <= 1.0,
87  "percentile (" << percent << ") must be in (0.0, 1.0]");
88 
89  Real sampleWeight = weightSum();
90  QL_REQUIRE(sampleWeight>0.0,
91  "empty sample set");
92 
93  sort();
94 
95  std::vector<std::pair<Real,Real> >::iterator k, l;
96  k = samples_.begin();
97  l = samples_.end()-1;
98  /* the sum of weight is non null, therefore there's
99  at least one sample */
100  Real integral = k->second, target = percent*sampleWeight;
101  while (integral < target && k != l) {
102  ++k;
103  integral += k->second;
104  }
105  return k->first;
106  }
