39  Time Actual365Fixed::CA_Impl::yearFraction(const Date& d1,
40  const Date& d2,
41  const Date& refPeriodStart,
42  const Date& refPeriodEnd) const {
43  if (d1 == d2)
44  return 0.0;
45 
46  // We need the period to calculate the frequency
47  QL_REQUIRE(refPeriodStart != Date(), "invalid refPeriodStart");
48  QL_REQUIRE(refPeriodEnd != Date(), "invalid refPeriodEnd");
49 
50  Time dcs = daysBetween(d1,d2);
51  Time dcc = daysBetween(refPeriodStart,refPeriodEnd);
52  Integer months = Integer(0.5+12*dcc/365);
53  QL_REQUIRE(months != 0,
54  "invalid reference period for Act/365 Canadian; "
55  "must be longer than a month");
56  Integer frequency = Integer(12/months);
57 
58  if (dcs < Integer(365/frequency))
59  return dcs/365.0;
60 
61  return 1./frequency - (dcc-dcs)/365.0;
62 
63  }

65  Date::serial_type Actual365Fixed::NL_Impl::dayCount(const Date& d1,
66  const Date& d2) const {
67 
68  static const Integer MonthOffset[] = {
69  0, 31, 59, 90, 120, 151, // Jan - Jun
70  181, 212, 243, 273, 304, 334 // Jun - Dec
71  };
72 
73  Date::serial_type s1 = d1.dayOfMonth()
74  + MonthOffset[d1.month()-1] + (d1.year() * 365);
75  Date::serial_type s2 = d2.dayOfMonth()
76  + MonthOffset[d2.month()-1] + (d2.year() * 365);
77 
78  if (d1.month() == Feb && d1.dayOfMonth() == 29) {
79  --s1;
80  }
81 
82  if (d2.month() == Feb && d2.dayOfMonth() == 29) {
83  --s2;
84  }
85 
86  return s2 - s1;
87  }
