34  class integrand_vasicek {
35  private:
36  const Real sigma_s_;
37  const Real sigma_r_;
38  const Real correlation_;
39  const Real kappa_;
40  const Real T_;
41  public:
42  integrand_vasicek(Real sigma_s, Real sigma_r, Real correlation, Real kappa, Real T)
43  : sigma_s_(sigma_s), sigma_r_(sigma_r), correlation_(correlation), kappa_(kappa), T_(T){}
44  Real operator()(Real u) const {
45  Real g = g_k(T_ - u, kappa_);
46  return (sigma_s_ * sigma_s_) + (2 * correlation_ * sigma_s_ * sigma_r_ * g) + (sigma_r_ * sigma_r_ * g * g);
47  }
48  };

61  void AnalyticBlackVasicekEngine::calculate() const {
62  QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
63  "not an European option");
64 
65  ext::shared_ptr<StrikedTypePayoff> payoff =
66  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
67 
68  QL_REQUIRE(payoff, "non-striked payoff given");
69 
70  CumulativeNormalDistribution f;
71 
72  Real t = 0;
73  Real T = blackProcess_->riskFreeRate()->dayCounter().yearFraction(blackProcess_->riskFreeRate().currentLink()->referenceDate(),arguments_.exercise->lastDate());
74  Real kappa = vasicekProcess_->a();
75  Real S_t = blackProcess_->x0();
76  Real K = payoff->strike();
77  Real sigma_s = blackProcess_->blackVolatility()->blackVol(t, K);
78  Real sigma_r = vasicekProcess_->sigma();
79  Real r_t = vasicekProcess_->r0();
80 
81  Real zcb = vasicekProcess_->discountBond(t, T, r_t);
82  Real epsilon = payoff->optionType() == Option::Call ? 1 : -1;
83  Real upsilon = (*simpsonIntegral_)(integrand_vasicek(sigma_s, sigma_r, correlation_, kappa, T), t, T);
84  Real d_positive = (std::log((S_t / K) / zcb) + upsilon / 2) / std::sqrt(upsilon);
85  Real d_negative = (std::log((S_t / K) / zcb) - upsilon / 2) / std::sqrt(upsilon);
86  Real n_d1 = f(epsilon * d_positive);
87  Real n_d2 = f(epsilon * d_negative);
88 
89  results_.value = epsilon * ((S_t * n_d1) - (zcb * K * n_d2));
90  }
