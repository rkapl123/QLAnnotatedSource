86  Real ExtendedOrnsteinUhlenbeckProcess::expectation(
87  Time t0, Real x0, Time dt) const {
88  switch (discretization_) {
89  case MidPoint:
90  return ouProcess_->expectation(t0, x0, dt)
91  + b_(t0+0.5*dt)*(1.0 - std::exp(-speed_*dt));
92  break;
93  case Trapezodial:
94  {
95  const Time t = t0+dt;
96  const Time u = t0;
97  const Real bt = b_(t);
98  const Real bu = b_(u);
99  const Real ex = std::exp(-speed_*dt);
100 
101  return ouProcess_->expectation(t0, x0, dt)
102  + bt-ex*bu - (bt-bu)/(speed_*dt)*(1-ex);
103  }
104  break;
105  case GaussLobatto:
106  return ouProcess_->expectation(t0, x0, dt)
107  + speed_*std::exp(-speed_*(t0+dt))
108  * GaussLobattoIntegral(100000, intEps_)(integrand(b_, speed_),
109  t0, t0+dt);
110  break;
111  default:
112  QL_FAIL("unknown discretization scheme");
113  }
114  }
