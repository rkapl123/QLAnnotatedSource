84 Real ExtendedOrnsteinUhlenbeckProcess::expectation(
85 Time t0, Real x0, Time dt) const {
86 switch (discretization_) {
87 case MidPoint:
88 return ouProcess_->expectation(t0, x0, dt)
89 + b_(t0+0.5*dt)*(1.0 - std::exp(-speed_*dt));
90 case Trapezodial:
91 {
92 const Time t = t0+dt;
93 const Time u = t0;
94 const Real bt = b_(t);
95 const Real bu = b_(u);
96 const Real ex = std::exp(-speed_*dt);
97
98 return ouProcess_->expectation(t0, x0, dt)
99 + bt-ex*bu - (bt-bu)/(speed_*dt)*(1-ex);
100 }
101 case GaussLobatto:
102 return ouProcess_->expectation(t0, x0, dt)
103 + speed_*std::exp(-speed_*(t0+dt))
104 * GaussLobattoIntegral(100000, intEps_)(integrand(b_, speed_),
105 t0, t0+dt);
106 default:
107 QL_FAIL("unknown discretization scheme");
108 }
109 }
