39  class FDDividendEngineBase : public FDMultiPeriodEngine<Scheme> {
40  public:
41  FDDividendEngineBase(
42  const boost::shared_ptr<GeneralizedBlackScholesProcess>& process,
43  Size timeSteps = 100,
44  Size gridPoints = 100,
45  bool timeDependent = false)
46  : FDMultiPeriodEngine<Scheme>(process, timeSteps,
47  gridPoints, timeDependent) {}
48  protected:
49  virtual void setupArguments(const PricingEngine::arguments*) const;
50  void setGridLimits() const = 0;

52  Real getDividendAmount(Size i) const {
53  const Dividend *dividend =
54  dynamic_cast<const Dividend *>(this->events_[i].get());
55  if (dividend) {
56  return dividend->amount();
57  } else {
58  return 0.0;
59  }
60  }
61  Real getDiscountedDividend(Size i) const {
62  Real dividend = getDividendAmount(i);
63  Real discount =
64  this->process_->riskFreeRate()->discount(
65  this->events_[i]->date()) /
66  this->process_->dividendYield()->discount(
67  this->events_[i]->date());
68  return dividend * discount;
69  }
70  };

161  template <template <class> class Scheme>
162  void FDDividendEngineMerton73<Scheme>::setGridLimits() const {
163  Real paidDividends = 0.0;
164  for (Size i=0; i<this->events_.size(); i++) {
165  if (this->getDividendTime(i) >= 0.0)
166  paidDividends += this->getDiscountedDividend(i);
167  }
168 
169  FDVanillaEngine::setGridLimits(
170  this->process_->stateVariable()->value()-paidDividends,
171  this->getResidualTime());
172  this->ensureStrikeInGrid();
173  }

175  // TODO: Make this work for both fixed and scaled dividends
176  template <template <class> class Scheme>
177  void FDDividendEngineMerton73<Scheme>::executeIntermediateStep(
178  Size step) const{
179  Real scaleFactor =
180  this->getDiscountedDividend(step) / this->center_ + 1.0;
181  this->sMin_ *= scaleFactor;
182  this->sMax_ *= scaleFactor;
183  this->center_ *= scaleFactor;
184 
185  this->intrinsicValues_.scaleGrid(scaleFactor);
186  this->intrinsicValues_.sample(*(this->payoff_));
187  this->prices_.scaleGrid(scaleFactor);
188  this->initializeOperator();
189  this->initializeModel();
190 
191  this->initializeStepCondition();
192  this->stepCondition_ -> applyTo(this->prices_.values(),
193  this->getDividendTime(step));
194  }

210  template <template <class> class Scheme>
211  void FDDividendEngineShiftScale<Scheme>::setGridLimits() const {
212  Real underlying = this->process_->stateVariable()->value();
213  for (Size i=0; i<this->events_.size(); i++) {
214  const Dividend *dividend =
215  dynamic_cast<const Dividend *>(this->events_[i].get());
216  if (!dividend) continue;
217  if (this->getDividendTime(i) < 0.0) continue;
218  underlying -= dividend->amount(underlying);
219  }
220 
221  FDVanillaEngine::setGridLimits(underlying,
222  this->getResidualTime());
223  this->ensureStrikeInGrid();
224  }

226  template <template <class> class Scheme>
227  void FDDividendEngineShiftScale<Scheme>::executeIntermediateStep(
228  Size step) const{
229  const Dividend *dividend =
230  dynamic_cast<const Dividend *>(this->events_[step].get());
231  if (!dividend) return;
232  detail::DividendAdder adder(dividend);
233  this->sMin_ = adder(this->sMin_);
234  this->sMax_ = adder(this->sMax_);
235  this->center_ = adder(this->center_);
236  this->intrinsicValues_.transformGrid(adder);
237 
238  this->intrinsicValues_.sample(*(this->payoff_));
239  this->prices_.transformGrid(adder);
240 
241  this->initializeOperator();
242  this->initializeModel();
243 
244  this->initializeStepCondition();
245  this->stepCondition_ -> applyTo(this->prices_.values(),
246  this->getDividendTime(step));
247  }
