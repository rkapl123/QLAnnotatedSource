36 template <template <class> class Scheme = CrankNicolson>
37 class QL_DEPRECATED FDDividendEngineBase : public FDMultiPeriodEngine<Scheme> {
38 public:
39 FDDividendEngineBase(
40 const ext::shared_ptr<GeneralizedBlackScholesProcess>& process,
41 Size timeSteps = 100,
42 Size gridPoints = 100,
43 bool timeDependent = false)
44 : FDMultiPeriodEngine<Scheme>(process, timeSteps,
45 gridPoints, timeDependent) {}
46 protected:
47 virtual void setupArguments(const PricingEngine::arguments*) const;
48 virtual void setGridLimits() const = 0;
49 virtual void executeIntermediateStep(Size step) const = 0;
50 Real getDividendAmount(Size i) const {
51 const auto* dividend = dynamic_cast<const Dividend*>(this->events_[i].get());
52 if (dividend != nullptr) {
53 return dividend->amount();
54 } else {
55 return 0.0;
56 }
57 }
58 Real getDiscountedDividend(Size i) const {
59 Real dividend = getDividendAmount(i);
60 Real discount =
61 this->process_->riskFreeRate()->discount(
62 this->events_[i]->date()) /
63 this->process_->dividendYield()->discount(
64 this->events_[i]->date());
65 return dividend * discount;
66 }
67 };

138 // The value of the x axis is the NPV of the underlying minus the
139 // value of the paid dividends.
140
141 // Note that to get the PDE to work, I have to scale the values
142 // and not shift them. This means that the price curve assumes
143 // that the dividends are scaled with the value of the underlying.
144 //
145
146 template <template <class> class Scheme>
147 void FDDividendEngineMerton73<Scheme>::setGridLimits() const {
148 Real paidDividends = 0.0;
149 for (Size i=0; i<this->events_.size(); i++) {
150 if (this->getDividendTime(i) >= 0.0)
151 paidDividends += this->getDiscountedDividend(i);
152 }
153
154 FDVanillaEngine::setGridLimits(
155 this->process_->stateVariable()->value()-paidDividends,
156 this->getResidualTime());
157 this->ensureStrikeInGrid();
158 }
159
160 // TODO: Make this work for both fixed and scaled dividends
161 template <template <class> class Scheme>
162 void FDDividendEngineMerton73<Scheme>::executeIntermediateStep(
163 Size step) const{
164 Real scaleFactor =
165 this->getDiscountedDividend(step) / this->center_ + 1.0;
166 this->sMin_ *= scaleFactor;
167 this->sMax_ *= scaleFactor;
168 this->center_ *= scaleFactor;
169
170 this->intrinsicValues_.scaleGrid(scaleFactor);
171 this->intrinsicValues_.sample(*(this->payoff_));
172 this->prices_.scaleGrid(scaleFactor);
173 this->initializeOperator();
174 this->initializeModel();
175
176 this->initializeStepCondition();
177 this->stepCondition_ -> applyTo(this->prices_.values(),
178 this->getDividendTime(step));
179 }

195 template <template <class> class Scheme>
196 void FDDividendEngineShiftScale<Scheme>::setGridLimits() const {
197 Real underlying = this->process_->stateVariable()->value();
198 for (Size i=0; i<this->events_.size(); i++) {
199 const auto* dividend = dynamic_cast<const Dividend*>(this->events_[i].get());
200 if (dividend == nullptr)
201 continue;
202 if (this->getDividendTime(i) < 0.0) continue;
203 underlying -= dividend->amount(underlying);
204 }
205
206 FDVanillaEngine::setGridLimits(underlying,
207 this->getResidualTime());
208 this->ensureStrikeInGrid();
209 }

211 template <template <class> class Scheme>
212 void FDDividendEngineShiftScale<Scheme>::executeIntermediateStep(
213 Size step) const{
214 const auto* dividend = dynamic_cast<const Dividend*>(this->events_[step].get());
215 if (dividend == nullptr)
216 return;
217 detail::DividendAdder adder(dividend);
218 this->sMin_ = adder(this->sMin_);
219 this->sMax_ = adder(this->sMax_);
220 this->center_ = adder(this->center_);
221 this->intrinsicValues_.transformGrid(adder);
222
223 this->intrinsicValues_.sample(*(this->payoff_));
224 this->prices_.transformGrid(adder);
225
226 this->initializeOperator();
227 this->initializeModel();
228
229 this->initializeStepCondition();
230 this->stepCondition_ -> applyTo(this->prices_.values(),
231 this->getDividendTime(step));
232 }
