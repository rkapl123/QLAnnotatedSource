79  Real value(Real x1, Real x2) const override {
80 
81  Real res=0.0;
82 
83  Array X(2),Xn(2);
84  X[0]=x1;X[1]=x2;
85 
86  Size cnt=0; // counter
87 
88  for( Size j=0; j< ySize_;++j){
89  for( Size i=0; i< xSize_;++i){
90  Xn[0]=this->xBegin_[i];
91  Xn[1]=this->yBegin_[j];
92  res+=alphaVec_[cnt]*kernelAbs(X,Xn);
93  cnt++;
94  }
95  }
96  return res/gammaFunc(X);
97  }
98 
99  // the calculation will solve y=M*a for a. Due to
100  // singularity or rounding errors the recalculation
101  // M*a may not give y. Here, a failure will be thrown if
102  // |M*a-y|>=invPrec_
103  void setInverseResultPrecision(Real invPrec){
104  invPrec_=invPrec;
105  }
106 
107  private:
108 
109  // returns K(||X-Y||) where X,Y are vectors
110  Real kernelAbs(const Array& X, const Array& Y)const{
111  return kernel_(Norm2(X-Y));
112  }
113 
114  Real gammaFunc(const Array& X)const{
115 
116  Real res=0.0;
117  Array Xn(X.size());
118 
119  for(Size j=0; j< ySize_;++j){
120  for(Size i=0; i< xSize_;++i){
121  Xn[0]=this->xBegin_[i];
122  Xn[1]=this->yBegin_[j];
123  res+=kernelAbs(X,Xn);
124  }
125  }
126 
127  return res;
128  }

130  void updateAlphaVec(){
131  // Function calculates the alpha vector with given
132  // fixed pillars+values
133 
134  Array Xk(2),Xn(2);
135 
136  Size rowCnt=0,colCnt=0;
137  Real tmpVar=0.0;
138 
139  // write y-vector and M-Matrix
140  for(Size j=0; j< ySize_;++j){
141  for(Size i=0; i< xSize_;++i){
142 
143  yVec_[rowCnt]=this->zData_[i][j];
144  // calculate X_k
145  Xk[0]=this->xBegin_[i];
146  Xk[1]=this->yBegin_[j];
147 
148  tmpVar=1/gammaFunc(Xk);
149  colCnt=0;
150 
151  for(Size jM=0; jM< ySize_;++jM){
152  for(Size iM=0; iM< xSize_;++iM){
153  Xn[0]=this->xBegin_[iM];
154  Xn[1]=this->yBegin_[jM];
155  M_[rowCnt][colCnt]=kernelAbs(Xk,Xn)*tmpVar;
156  colCnt++; // increase column counter
157  }// end iM
158  }// end jM
159  rowCnt++; // increase row counter
160  } // end i
161  }// end j
162 
163  alphaVec_=qrSolve(M_, yVec_);
164 
165  // check if inversion worked up to a reasonable precision.
166  // I've chosen not to check determinant(M_)!=0 before solving
167 
168  Array diffVec=Abs(M_*alphaVec_ - yVec_);
169  for (double i : diffVec) {
170  QL_REQUIRE(i < invPrec_, "inversion failed in 2d kernel interpolation");
171  }
172  }

176  Real invPrec_;

178  Matrix M_;
179  Kernel kernel_;
180  };
