250  template<class I2D>
251  void InterpolatedCPICapFloorTermPriceSurface<I2D>::
252  performCalculations() const {
253 
254  cPriceB_ =
255  Matrix(cfStrikes_.size(), cfMaturities_.size(), Null<Real>());
256  fPriceB_ =
257  Matrix(cfStrikes_.size(), cfMaturities_.size(), Null<Real>());
258 
259  Handle<ZeroInflationTermStructure> zts = zii_->zeroInflationTermStructure();
260  Handle<YieldTermStructure> yts = nominalTS_;
261  QL_REQUIRE(!zts.empty(), "Zts is empty!!!");
262  QL_REQUIRE(!yts.empty(), "Yts is empty!!!");
263 
264  for (Size j = 0; j < cfMaturities_.size(); ++j) {
265  Period mat = cfMaturities_[j];
266  Real df = yts->discount(cpiOptionDateFromTenor(mat));
267  Real atm_quote = zts->zeroRate(cpiOptionDateFromTenor(mat));
268  Real atm = std::pow(1.0 + atm_quote, mat.length());
269  Real S = atm * df;
270  for (Size i = 0; i < cfStrikes_.size(); ++i) {
271  Real K_quote = cfStrikes_[i];
272  Real K = std::pow(1.0 + K_quote, mat.length());
273  Size indF = std::find_if(fStrikes_.begin(), fStrikes_.end(),
274  detail::CloseEnoughComparator(cfStrikes_[i])) -
275  fStrikes_.begin();
276  Size indC = std::find_if(cStrikes_.begin(), cStrikes_.end(),
277  detail::CloseEnoughComparator(cfStrikes_[i])) -
278  cStrikes_.begin();
279  bool isFloorStrike = indF < fStrikes_.size();
280  bool isCapStrike = indC < cStrikes_.size();
281  if (isFloorStrike) {
282  fPriceB_[i][j] = fPrice_[indF][j];
283  if (!isCapStrike) {
284  cPriceB_[i][j] = fPrice_[indF][j] + S - K * df;
285  }
286  }
287  if (isCapStrike) {
288  cPriceB_[i][j] = cPrice_[indC][j];
289  if (!isFloorStrike) {
290  fPriceB_[i][j] = cPrice_[indC][j] + K * df - S;
291  }
292  }
293  }
294  }
295 
296  // check that all cells are filled
297  for (Size i = 0; i < cPriceB_.rows(); ++i) {
298  for (Size j = 0; j < cPriceB_.columns(); ++j) {
299  QL_REQUIRE(cPriceB_[i][j] != Null<Real>(),
300  "InterpolatedCPICapFloorTermPriceSurface: did not "
301  "fill call price matrix at ("
302  << i << "," << j << "), this is unexpected");
303  QL_REQUIRE(fPriceB_[i][j] != Null<Real>(),
304  "InterpolatedCPICapFloorTermPriceSurface: did not "
305  "fill floor price matrix at ("
306  << i << "," << j << "), this is unexpected");
307  }
308  }
309 
310  cfMaturityTimes_.clear();
311  for (Size i=0; i<cfMaturities_.size();i++) {
312  cfMaturityTimes_.push_back(timeFromReference(cpiOptionDateFromTenor(cfMaturities_[i])));
313  }
314 
315  capPrice_ = interpolator2d_.interpolate(cfMaturityTimes_.begin(),cfMaturityTimes_.end(),
316  cfStrikes_.begin(), cfStrikes_.end(),
317  cPriceB_
318  );
319  capPrice_.enableExtrapolation();
320 
321  floorPrice_ = interpolator2d_.interpolate(cfMaturityTimes_.begin(),cfMaturityTimes_.end(),
322  cfStrikes_.begin(), cfStrikes_.end(),
323  fPriceB_
324  );
325  floorPrice_.enableExtrapolation();
326  }
