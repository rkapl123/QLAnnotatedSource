53  Volatility DoubleBarrierOption::impliedVolatility(
54  Real targetValue,

56  Real accuracy,
57  Size maxEvaluations,
58  Volatility minVol,
59  Volatility maxVol) const {
60

62 
63  boost::shared_ptr<SimpleQuote> volQuote(new SimpleQuote);
64 
65  boost::shared_ptr<GeneralizedBlackScholesProcess> newProcess =
66  detail::ImpliedVolatilityHelper::clone(process, volQuote);
67 
68  // engines are built-in for the time being
69  boost::scoped_ptr<PricingEngine> engine;
70  switch (exercise_->type()) {
71  case Exercise::European:
72  engine.reset(new AnalyticDoubleBarrierEngine(newProcess));
73  break;
74  case Exercise::American:
75  case Exercise::Bermudan:
76  QL_FAIL("engine not available for non-European barrier option");
77  break;
78  default:
79  QL_FAIL("unknown exercise type");
80  }
81 
82  return detail::ImpliedVolatilityHelper::calculate(*this,
83  *engine,
84  *volQuote,
85  targetValue,
86  accuracy,
87  maxEvaluations,
88  minVol, maxVol);
