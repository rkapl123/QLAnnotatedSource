82  void SubPeriodsPricer::initialize(const FloatingRateCoupon& coupon) {
83  coupon_ = dynamic_cast<const SubPeriodsCoupon*>(&coupon);
84  QL_REQUIRE(coupon_, "sub-periods coupon required");
85  gearing_ = coupon_->gearing();
86  spread_ = coupon_->spread();
87 
88  Date paymentDate = coupon_->date();
89 
90  boost::shared_ptr<IborIndex> index =
91  boost::dynamic_pointer_cast<IborIndex>(coupon_->index());
92  const Handle<YieldTermStructure>& rateCurve =
93  index->forwardingTermStructure();
94  discount_ = rateCurve->discount(paymentDate);
95  accrualFactor_ = coupon_->accrualPeriod();
96  spreadLegValue_ = spread_ * accrualFactor_* discount_;
97 
98  startTime_ = coupon_->startTime();
99  endTime_ = coupon_->endTime();
100  observationTimes_ = coupon_->observationTimes();
101  observations_ = coupon_->observations();
102 
103  const std::vector<Date>& observationDates =
104  coupon_->observationsSchedule()->dates();
105 
106  QL_REQUIRE(observationDates.size()==observations_+2,
107  "incompatible size of initialValues vector");
108 
109  initialValues_ = std::vector<Real>(observationDates.size(),0.);
110 
111  observationCvg_ = std::vector<Real>(observationDates.size(),0.);
112 
113  observationIndexStartDates_ =
114  std::vector<Date>(observationDates.size());
115  observationIndexEndDates_ =
116  std::vector<Date>(observationDates.size());
117 
118  Calendar calendar = index->fixingCalendar();
119 
120  for(Size i=0; i<observationDates.size(); i++) {
121  Date fixingDate = calendar.advance(
122  observationDates[i],
123  -static_cast<Integer>(coupon_->fixingDays()),
124  Days);
125 
126  initialValues_[i] =
127  index->fixing(fixingDate) + coupon_->rateSpread();
128 
129  Date fixingValueDate = index->valueDate(fixingDate);
130  Date endValueDate = index->maturityDate(fixingValueDate);
131 
132  observationIndexStartDates_[i] = fixingValueDate;
133  observationIndexEndDates_[i] = endValueDate;
134 
135  observationCvg_[i] =
136  index->dayCounter().yearFraction(fixingValueDate, endValueDate);
137  }
138  }

160  Real AveragingRatePricer::swapletPrice() const {
161  // past or future fixing is managed in InterestRateIndex::fixing()
162 
163  Size nCount = initialValues_.size();
164  Real dAvgRate = 0.0, dTotalCvg = 0.0, dTotalPayment = 0.0;
165  for (Size i=0; i<nCount; i++) {
166  dTotalPayment += initialValues_[i] * observationCvg_[i];
167  dTotalCvg += observationCvg_[i];
168  }
169 
170  dAvgRate = dTotalPayment/dTotalCvg;
171 
172  Real swapletPrice = dAvgRate*coupon_->accrualPeriod()*discount_;
173  return gearing_ * swapletPrice + spreadLegValue_;
174  }

176  Real CompoundingRatePricer::swapletPrice() const {
177  // past or future fixing is managed in InterestRateIndex::fixing()
178 
179  Real dNotional = 1.0;
180 
181  Size nCount = initialValues_.size();
182  Real dCompRate = 0.0, dTotalCvg = 0.0, dTotalPayment = 0.0;
183  for (Size i=0; i<nCount; i++) {
184  dTotalPayment = initialValues_[i] * observationCvg_[i]*dNotional;
185  dNotional += dTotalPayment;
186  dTotalCvg += observationCvg_[i];
187  }
188 
189  dCompRate = (dNotional - 1.0)/dTotalCvg;
190 
191  Real swapletPrice = dCompRate*coupon_->accrualPeriod()*discount_;
192  return gearing_ * swapletPrice + spreadLegValue_;
193  }
