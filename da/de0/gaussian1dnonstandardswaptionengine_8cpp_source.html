32  Real
33  Gaussian1dNonstandardSwaptionEngine::underlyingNpv(const Date &expiry,
34  const Real y) const {
35 
36  // determine the indices on both legs representing the cashflows that
37  // are part of the exercise into right
38 
39  Size fixedIdx =
40  std::upper_bound(arguments_.fixedResetDates.begin(),
41  arguments_.fixedResetDates.end(), expiry - 1) -
42  arguments_.fixedResetDates.begin();
43  Size floatingIdx =
44  std::upper_bound(arguments_.floatingResetDates.begin(),
45  arguments_.floatingResetDates.end(), expiry - 1) -
46  arguments_.floatingResetDates.begin();
47 
48  // calculate the npv of these cashflows conditional on y at expiry
49 
50  Real type = (Real)arguments_.type;
51 
52  Real npv = 0.0;
53  for (Size i = fixedIdx; i < arguments_.fixedResetDates.size(); i++) {
54  npv -=
55  arguments_.fixedCoupons[i] *
56  model_->zerobond(arguments_.fixedPayDates[i], expiry, y,
57  discountCurve_) *
58  (oas_.empty()
59  ? 1.0
60  : exp(-oas_->value() *
61  model_->termStructure()->dayCounter().yearFraction(
62  expiry, arguments_.fixedPayDates[i])));
63  }
64 
65  for (Size i = floatingIdx; i < arguments_.floatingResetDates.size();
66  i++) {
67  Real amount;
68  if (!arguments_.floatingIsRedemptionFlow[i])
69  amount = (arguments_.floatingGearings[i] *
70  model_->forwardRate(
71  arguments_.floatingFixingDates[i], expiry, y,
72  arguments_.swap->iborIndex()) +
73  arguments_.floatingSpreads[i]) *
74  arguments_.floatingAccrualTimes[i] *
75  arguments_.floatingNominal[i];
76  else
77  amount = arguments_.floatingCoupons[i];
78  npv +=
79  amount * model_->zerobond(arguments_.floatingPayDates[i],
80  expiry, y, discountCurve_) *
81  (oas_.empty()
82  ? 1.0
83  : exp(-oas_->value() *
84  model_->termStructure()->dayCounter().yearFraction(
85  expiry, arguments_.floatingPayDates[i])));
86  }
87 
88  return type * npv;
89  }

102  const Disposable<Array> Gaussian1dNonstandardSwaptionEngine::initialGuess(const Date &expiry) const {
103 
104  Size fixedIdx =
105  std::upper_bound(arguments_.fixedResetDates.begin(),
106  arguments_.fixedResetDates.end(), expiry - 1) -
107  arguments_.fixedResetDates.begin();
108 
109  Array initial(3);
110  Real nominalSum = 0.0, weightedRate = 0.0, ind = 0.0;
111  for (Size i = fixedIdx; i < arguments_.fixedResetDates.size(); i++) {
112  nominalSum += arguments_.fixedNominal[i];
113  Real rate = arguments_.fixedRate[i];
114  if (close(rate, 0.0))
115  rate = 0.03; // this value is at least better than zero
116  weightedRate += arguments_.fixedNominal[i] * rate;
117  if (arguments_.fixedNominal[i] > 1E-8) // exclude zero nominal periods
118  ind += 1.0;
119  }
120  Real nominalAvg = nominalSum / ind;
121 
122  QL_REQUIRE(nominalSum > 0.0,
123  "sum of nominals on fixed leg must be positive ("
124  << nominalSum << ")");
125 
126  weightedRate /= nominalSum;
127  initial[0] = nominalAvg;
128  initial[1] =
129  model_->termStructure()->timeFromReference(underlyingLastDate()) -
130  model_->termStructure()->timeFromReference(expiry);
131  initial[2] = weightedRate;
132 
133  return initial;
134  }

136  void Gaussian1dNonstandardSwaptionEngine::calculate() const {
137 
138  QL_REQUIRE(arguments_.settlementMethod != Settlement::ParYieldCurve,
139  "cash settled (ParYieldCurve) swaptions not priced with "
140  "Gaussian1dNonstandardSwaptionEngine");
141 
142  Date settlement = model_->termStructure()->referenceDate();
143 
144  if (arguments_.exercise->dates().back() <=
145  settlement) { // swaption is expired, possibly generated swap is not
146  // valued
147  results_.value = 0.0;
148  return;
149  }
150 
151  ext::shared_ptr<RebatedExercise> rebatedExercise =
152  ext::dynamic_pointer_cast<RebatedExercise>(arguments_.exercise);
153 
154  int idx = arguments_.exercise->dates().size() - 1;
155  int minIdxAlive = static_cast<int>(
156  std::upper_bound(arguments_.exercise->dates().begin(),
157  arguments_.exercise->dates().end(), settlement) -
158  arguments_.exercise->dates().begin());
159 
160  NonstandardSwap swap = *arguments_.swap;
161  Option::Type type =
162  arguments_.type == VanillaSwap::Payer ? Option::Call : Option::Put;
163 
164  Array npv0(2 * integrationPoints_ + 1, 0.0),
165  npv1(2 * integrationPoints_ + 1, 0.0);
166  Array z = model_->yGrid(stddevs_, integrationPoints_);
167  Array p(z.size(), 0.0);
168 
169  // for probability computation
170  std::vector<Array> npvp0, npvp1;
171  if (probabilities_ != None) {
172  for (int i = 0; i < idx - minIdxAlive + 2; ++i) {
173  Array npvTmp0(2 * integrationPoints_ + 1, 0.0);
174  Array npvTmp1(2 * integrationPoints_ + 1, 0.0);
175  npvp0.push_back(npvTmp0);
176  npvp1.push_back(npvTmp1);
177  }
178  }
179  // end probabkility computation
180 
181  Date expiry1 = Null<Date>(), expiry0;
182  Time expiry1Time = Null<Real>(), expiry0Time;
183 
184  do {
185 
186  if (idx == minIdxAlive - 1)
187  expiry0 = settlement;
188  else
189  expiry0 = arguments_.exercise->dates()[idx];
190 
191  expiry0Time = std::max(
192  model_->termStructure()->timeFromReference(expiry0), 0.0);
193 
194  Size j1 =
195  std::upper_bound(arguments_.fixedResetDates.begin(),
196  arguments_.fixedResetDates.end(), expiry0 - 1) -
197  arguments_.fixedResetDates.begin();
198  Size k1 =
199  std::upper_bound(arguments_.floatingResetDates.begin(),
200  arguments_.floatingResetDates.end(), expiry0 - 1) -
201  arguments_.floatingResetDates.begin();
202 
203  // todo add openmp support later on (as in gaussian1dswaptionengine)
204 
205  for (Size k = 0; k < (expiry0 > settlement ? npv0.size() : 1);
206  k++) {
207 
208  Real price = 0.0;
209  if (expiry1Time != Null<Real>()) {
210  Real zSpreadDf =
211  oas_.empty() ? 1.0
212  : std::exp(-oas_->value() *
213  (expiry1Time - expiry0Time));
214  Array yg = model_->yGrid(stddevs_, integrationPoints_,
215  expiry1Time, expiry0Time,
216  expiry0 > settlement ? z[k] : 0.0);
217  CubicInterpolation payoff0(
218  z.begin(), z.end(), npv1.begin(),
219  CubicInterpolation::Spline, true,
220  CubicInterpolation::Lagrange, 0.0,
221  CubicInterpolation::Lagrange, 0.0);
222  for (Size i = 0; i < yg.size(); i++) {
223  p[i] = payoff0(yg[i], true);
224  }
225  CubicInterpolation payoff1(
226  z.begin(), z.end(), p.begin(),
227  CubicInterpolation::Spline, true,
228  CubicInterpolation::Lagrange, 0.0,
229  CubicInterpolation::Lagrange, 0.0);
230  for (Size i = 0; i < z.size() - 1; i++) {
231  price += model_->gaussianShiftedPolynomialIntegral(
232  0.0, payoff1.cCoefficients()[i],
233  payoff1.bCoefficients()[i],
234  payoff1.aCoefficients()[i], p[i], z[i],
235  z[i], z[i + 1]) *
236  zSpreadDf;
237  }
238  if (extrapolatePayoff_) {
239  if (flatPayoffExtrapolation_) {
240  price +=
241  model_->gaussianShiftedPolynomialIntegral(
242  0.0, 0.0, 0.0, 0.0, p[z.size() - 2],
243  z[z.size() - 2], z[z.size() - 1], 100.0) *
244  zSpreadDf;
245  price += model_->gaussianShiftedPolynomialIntegral(
246  0.0, 0.0, 0.0, 0.0, p[0], z[0], -100.0,
247  z[0]) *
248  zSpreadDf;
249  } else {
250  if (type == Option::Call)
251  price +=
252  model_->gaussianShiftedPolynomialIntegral(
253  0.0,
254  payoff1.cCoefficients()[z.size() - 2],
255  payoff1.bCoefficients()[z.size() - 2],
256  payoff1.aCoefficients()[z.size() - 2],
257  p[z.size() - 2], z[z.size() - 2],
258  z[z.size() - 1], 100.0) *
259  zSpreadDf;
260  if (type == Option::Put)
261  price +=
262  model_->gaussianShiftedPolynomialIntegral(
263  0.0, payoff1.cCoefficients()[0],
264  payoff1.bCoefficients()[0],
265  payoff1.aCoefficients()[0], p[0], z[0],
266  -100.0, z[0]) *
267  zSpreadDf;
268  }
269  }
270  }
271 
272  npv0[k] = price;
273 
274  // for probability computation
275  if (probabilities_ != None) {
276  for (Size m = 0; m < npvp0.size(); m++) {
277  Real price = 0.0;
278  if (expiry1Time != Null<Real>()) {
279  Real zSpreadDf =
280  oas_.empty()
281  ? 1.0
282  : std::exp(-oas_->value() *
283  (expiry1Time - expiry0Time));
284  Array yg = model_->yGrid(
285  stddevs_, integrationPoints_, expiry1Time,
286  expiry0Time, expiry0 > settlement ? z[k] : 0.0);
287  CubicInterpolation payoff0(
288  z.begin(), z.end(), npvp1[m].begin(),
289  CubicInterpolation::Spline, true,
290  CubicInterpolation::Lagrange, 0.0,
291  CubicInterpolation::Lagrange, 0.0);
292  for (Size i = 0; i < yg.size(); i++) {
293  p[i] = payoff0(yg[i], true);
294  }
295  CubicInterpolation payoff1(
296  z.begin(), z.end(), p.begin(),
297  CubicInterpolation::Spline, true,
298  CubicInterpolation::Lagrange, 0.0,
299  CubicInterpolation::Lagrange, 0.0);
300  for (Size i = 0; i < z.size() - 1; i++) {
301  price +=
302  model_->gaussianShiftedPolynomialIntegral(
303  0.0, payoff1.cCoefficients()[i],
304  payoff1.bCoefficients()[i],
305  payoff1.aCoefficients()[i], p[i], z[i],
306  z[i], z[i + 1]) *
307  zSpreadDf;
308  }
309  if (extrapolatePayoff_) {
310  if (flatPayoffExtrapolation_) {
311  price +=
312  model_
313  ->gaussianShiftedPolynomialIntegral(
314  0.0, 0.0, 0.0, 0.0,
315  p[z.size() - 2],
316  z[z.size() - 2],
317  z[z.size() - 1], 100.0) *
318  zSpreadDf;
319  price +=
320  model_
321  ->gaussianShiftedPolynomialIntegral(
322  0.0, 0.0, 0.0, 0.0, p[0],
323  z[0], -100.0, z[0]) *
324  zSpreadDf;
325  } else {
326  if (type == Option::Call)
327  price +=
328  model_
329  ->gaussianShiftedPolynomialIntegral(
330  0.0,
331  payoff1.cCoefficients()
332  [z.size() - 2],
333  payoff1.bCoefficients()
334  [z.size() - 2],
335  payoff1.aCoefficients()
336  [z.size() - 2],
337  p[z.size() - 2],
338  z[z.size() - 2],
339  z[z.size() - 1], 100.0) *
340  zSpreadDf;
341  if (type == Option::Put)
342  price +=
343  model_
344  ->gaussianShiftedPolynomialIntegral(
345  0.0,
346  payoff1
347  .cCoefficients()[0],
348  payoff1
349  .bCoefficients()[0],
350  payoff1
351  .aCoefficients()[0],
352  p[0], z[0], -100.0,
353  z[0]) *
354  zSpreadDf;
355  }
356  }
357  }
358 
359  npvp0[m][k] = price;
360  }
361  }
362  // end probability computation
363 
364  if (expiry0 > settlement) {
365  Real floatingLegNpv = 0.0;
366  for (Size l = k1; l < arguments_.floatingCoupons.size();
367  l++) {
368  Real zSpreadDf =
369  oas_.empty()
370  ? 1.0
371  : std::exp(
372  -oas_->value() *
373  (model_->termStructure()
374  ->dayCounter()
375  .yearFraction(
376  expiry0,
377  arguments_
378  .floatingPayDates[l])));
379  Real amount;
380  if (arguments_.floatingIsRedemptionFlow[l])
381  amount = arguments_.floatingCoupons[l];
382  else
383  amount = arguments_.floatingNominal[l] *
384  arguments_.floatingAccrualTimes[l] *
385  (arguments_.floatingGearings[l] *
386  model_->forwardRate(
387  arguments_.floatingFixingDates[l],
388  expiry0, z[k],
389  arguments_.swap->iborIndex()) +
390  arguments_.floatingSpreads[l]);
391  floatingLegNpv +=
392  amount *
393  model_->zerobond(arguments_.floatingPayDates[l],
394  expiry0, z[k], discountCurve_) *
395  zSpreadDf;
396  }
397  Real fixedLegNpv = 0.0;
398  for (Size l = j1; l < arguments_.fixedCoupons.size(); l++) {
399  Real zSpreadDf =
400  oas_.empty()
401  ? 1.0
402  : std::exp(
403  -oas_->value() *
404  (model_->termStructure()
405  ->dayCounter()
406  .yearFraction(
407  expiry0,
408  arguments_.fixedPayDates[l])));
409  fixedLegNpv +=
410  arguments_.fixedCoupons[l] *
411  model_->zerobond(arguments_.fixedPayDates[l],
412  expiry0, z[k], discountCurve_) *
413  zSpreadDf;
414  }
415  Real rebate = 0.0;
416  Real zSpreadDf = 1.0;
417  Date rebateDate = expiry0;
418  if (rebatedExercise != NULL) {
419  rebate = rebatedExercise->rebate(idx);
420  rebateDate = rebatedExercise->rebatePaymentDate(idx);
421  zSpreadDf =
422  oas_.empty()
423  ? 1.0
424  : std::exp(
425  -oas_->value() *
426  (model_->termStructure()
427  ->dayCounter()
428  .yearFraction(expiry0, rebateDate)));
429  }
430  Real exerciseValue =
431  ((type == Option::Call ? 1.0 : -1.0) *
432  (floatingLegNpv - fixedLegNpv) +
433  rebate * model_->zerobond(rebateDate, expiry0, z[k],
434  discountCurve_) *
435  zSpreadDf) /
436  model_->numeraire(expiry0Time, z[k], discountCurve_);
437 
438  // for probability computation
439  if (probabilities_ != None) {
440  if (idx == static_cast<int>(
441  arguments_.exercise->dates().size()) -
442  1) // if true we are at the latest date,
443  // so we init
444  // the no call probability
445  npvp0.back()[k] =
446  probabilities_ == Naive
447  ? 1.0
448  : 1.0 / (model_->zerobond(expiry0Time, 0.0,
449  0.0,
450  discountCurve_) *
451  model_->numeraire(expiry0, z[k],
452  discountCurve_));
453  if (exerciseValue >= npv0[k]) {
454  npvp0[idx - minIdxAlive][k] =
455  probabilities_ == Naive
456  ? 1.0
457  : 1.0 /
458  (model_->zerobond(expiry0Time, 0.0,
459  0.0,
460  discountCurve_) *
461  model_->numeraire(expiry0Time, z[k],
462  discountCurve_));
463  for (Size ii = idx - minIdxAlive + 1;
464  ii < npvp0.size(); ii++)
465  npvp0[ii][k] = 0.0;
466  }
467  }
468  // end probability computation
469 
470  npv0[k] = std::max(npv0[k], exerciseValue);
471  }
472  }
473 
474  npv1.swap(npv0);
475 
476  // for probability computation
477  if (probabilities_ != None) {
478  for (Size i = 0; i < npvp0.size(); i++)
479  npvp1[i].swap(npvp0[i]);
480  }
481  // end probability computation
482 
483  expiry1 = expiry0;
484  expiry1Time = expiry0Time;
485 
486  } while (--idx >= minIdxAlive - 1);
487 
488  results_.value = npv1[0] * model_->numeraire(0.0, 0.0, discountCurve_);
489 
490  // for probability computation
491  if (probabilities_ != None) {
492  std::vector<Real> prob(npvp0.size());
493  for (Size i = 0; i < npvp0.size(); i++) {
494  prob[i] = npvp1[i][0] *
495  (probabilities_ == Naive
496  ? 1.0
497  : model_->numeraire(0.0, 0.0, discountCurve_));
498  }
499  results_.additionalResults["probabilities"] = prob;
500  }
501  // end probability computation
502  }
503 }
