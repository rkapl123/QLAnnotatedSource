56  void FdSimpleExtOUJumpSwingEngine::calculate() const {
57 
58  // 1. Exercise
59  ext::shared_ptr<SwingExercise> swingExercise(
60  ext::dynamic_pointer_cast<SwingExercise>(arguments_.exercise));
61 
62  QL_REQUIRE(swingExercise, "Swing exercise supported only");
63 
64  // 2. Mesher
65  const std::vector<Time> exerciseTimes
66  = swingExercise->exerciseTimes(rTS_->dayCounter(),
67  rTS_->referenceDate());
68 
69  const Time maturity = exerciseTimes.back();
70  const ext::shared_ptr<StochasticProcess1D> ouProcess(
71  process_->getExtendedOrnsteinUhlenbeckProcess());
72  const ext::shared_ptr<Fdm1dMesher> xMesher(
73  new FdmSimpleProcess1dMesher(xGrid_, ouProcess,maturity));
74 
75  const ext::shared_ptr<Fdm1dMesher> yMesher(
76  new ExponentialJump1dMesher(yGrid_,
77  process_->beta(),
78  process_->jumpIntensity(),
79  process_->eta()));
80  const ext::shared_ptr<Fdm1dMesher> exerciseMesher(
81  new Uniform1dMesher(
82  0, static_cast<Real>(arguments_.maxExerciseRights),
83  arguments_.maxExerciseRights+1));
84 
85  const ext::shared_ptr<FdmMesher> mesher(
86  new FdmMesherComposite(xMesher, yMesher, exerciseMesher));
87 
88  // 3. Calculator
89  ext::shared_ptr<FdmInnerValueCalculator> calculator(
90  new FdmZeroInnerValue());
91  // 4. Step conditions
92  std::list<ext::shared_ptr<StepCondition<Array> > > stepConditions;
93  std::list<std::vector<Time> > stoppingTimes;
94 
95  // 4.1 Bermudan step conditions
96  stoppingTimes.push_back(exerciseTimes);
97 
98  ext::shared_ptr<FdmInnerValueCalculator> exerciseCalculator(
99  new FdmExtOUJumpModelInnerValue(arguments_.payoff, mesher, shape_));
100 
101  stepConditions.push_back(ext::shared_ptr<StepCondition<Array> >(
102  new FdmSimpleSwingCondition(
103  exerciseTimes, mesher, exerciseCalculator,
104  2, arguments_.minExerciseRights)));
105 
106  ext::shared_ptr<FdmStepConditionComposite> conditions(
107  new FdmStepConditionComposite(stoppingTimes, stepConditions));
108 
109 
110  // 5. Boundary conditions
111  const FdmBoundaryConditionSet boundaries;
112 
113  // 6. set-up solver
114  FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
115  calculator, maturity, tGrid_, 0 };
116 
117  const ext::shared_ptr<FdmSimple3dExtOUJumpSolver> solver(
118  new FdmSimple3dExtOUJumpSolver(
119  Handle<ExtOUWithJumpsProcess>(process_),
120  rTS_, solverDesc, schemeDesc_));
121 
122  const Real x = process_->initialValues()[0];
123  const Real y = process_->initialValues()[1];
124 
125  results_.value = solver->valueAt(x, y, 0.0);
126  }
127 }
