51  void COSHestonEngine::calculate() const {
52 
53  // this is a european option pricer
54  QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
55  "not an European option");
56 
57  // plain vanilla
58  const ext::shared_ptr<PlainVanillaPayoff> payoff =
59  ext::dynamic_pointer_cast<PlainVanillaPayoff>(arguments_.payoff);
60  QL_REQUIRE(payoff, "non plain vanilla payoff given");
61 
62  const ext::shared_ptr<HestonProcess> process = model_->process();
63 
64  const Date maturityDate = arguments_.exercise->lastDate();
65  const Time maturity = process->time(maturityDate);
66 
67  const Real cum1 = c1(maturity);
68  const Real w = std::sqrt(std::fabs(c2(maturity))
69  // the 4th order doesn't necessarily improve the precision
70  // + std::sqrt(std::fabs(c4(maturity)))
71  );
72 
73  const Real k = payoff->strike();
74  const Real spot = process->s0()->value();
75  QL_REQUIRE(spot > 0.0, "negative or null underlying given");
76 
77  const DiscountFactor df
78  = process->riskFreeRate()->discount(maturityDate);
79  const DiscountFactor qf
80  = process->dividendYield()->discount(maturityDate);
81  const Real fwd = spot*qf/df;
82  const Real x = std::log(fwd/k);
83 
84  const Real a = x + cum1 - L_*w;
85  const Real b = x + cum1 + L_*w;
86 
87  const Real d = 1.0/(b-a);
88 
89  const Real expA = std::exp(a);
90  Real s = chF(0, maturity).real()*(expA-1-a)*d;
91 
92  for (Size n=1; n < N_; ++n) {
93  const Real r = n*M_PI*d;
94  const Real U_n = 2.0*d*( 1.0/(1.0 + r*r)
95  *(expA + r*std::sin(r*a) - std::cos(r*a)) - 1.0/r*std::sin(r*a));
96 
97  s += U_n*(chF(r, maturity)
98  *std::exp(std::complex<Real>(0, r*(x-a)))).real();
99  }
100 
101  if (payoff->optionType() == Option::Put)
102  results_.value = k*df*s;
103  else if (payoff->optionType() == Option::Call) {
104  const DiscountFactor qf
105  = process->dividendYield()->discount(maturityDate);
106  results_.value = spot*qf - k*df*(1-s);
107  }
108  else
109  QL_FAIL("unknown payoff type");
110  }

117  std::complex<Real> COSHestonEngine::chF(Real u, Real t)
118  const {
119  const Real sigma2 = sigma_*sigma_;
120 
121  const std::complex<Real> D = std::sqrt(
122  square<std::complex<Real> >()(
123  std::complex<Real>(kappa_, -rho_*sigma_*u))
124  + std::complex<Real>(u*u, u)*sigma2);
125 
126  const std::complex<Real> g(kappa_, -rho_*sigma_*u);
127 
128  const std::complex<Real> G = (g-D)/(g+D);
129 
130  return std::exp(
131  v0_/(sigma2)*(1.0-std::exp(-D*t))/(1.0-G*std::exp(-D*t))
132  *(g-D) + kappa_*theta_/sigma2*((g-D)*t
133  -2.0*std::log((1.0-G*std::exp(-D*t))/(1.0-G)))
134  );
135  }
136 
137  /*
138  Mathematica program to calculate the n-th cumulant
139 
140  d[z_] := Sqrt[(kappa -i*rho*sigma*z)^2 + (z*z+i*z)*sigma^2]
141 
142  g[z_] := (kappa -i*rho*sigma*z - d[z])/(kappa -i*rho*sigma*z + d[z])
143 
144  phi[z_] := Exp[ v0/(sigma^2)*(1-Exp[-d[z]*t])/(1-g[z]*Exp[-d[z]*t])
145  *(kappa -i*rho*sigma*z - d[z]) + kappa*theta/sigma^2
146  *((kappa -i*rho*sigma*z-d[z])*t
147  -2*Log[(1-g[z]*Exp[-d[z]*t])/(1-g[z]) ]) ]
148 
149  e[z_] := Log[phi[-i*z]]
150 
151  // for C++ formatting see
152  // http://mathematica.stackexchange.com/questions/114175/cform-not-getting-exp-pow-log-functions
153  cpp = RawBoxes[Replace[ToBoxes@#,
154  InterpretationBox[a_, b_, c___] :> With[{aa =
155  StringReplace[ a, {"Sqrt" -> "std::sqrt", "Power(E," -> "std::exp(",
156  "Power" -> "std::pow"}]}, aa], {0, Infinity}]] &;
157 
158  c[n_] := cpp@CForm[FullSimplify[Derivative[n][e][0],
159  kappa > 0 && theta > 0 && v0 > 0 && sigma > 0 &&
160  rho [Element] {-1, 1} && i^2 == -1]]
161  */

163  Real COSHestonEngine::c1(Time t) const {
164  return (-theta_ + std::exp(kappa_*t)
165  *( theta_ - kappa_*t*theta_ -
166  v0_) + v0_)/(2*std::exp(kappa_*t)*kappa_);
167  }

169  Real COSHestonEngine::c2(Time t) const {
170  const Real sigma2 = sigma_*sigma_;
171  const Real kappa2 = kappa_*kappa_;
172  const Real kappa3 = kappa2*kappa_;
173 
174  return (sigma2*(theta_ - 2*v0_) +
175  std::exp(2*kappa_*t)*(8*kappa3*t*theta_ -
176  8*kappa2*(theta_ + rho_*sigma_*t*theta_ - v0_) +
177  sigma2*(-5*theta_ + 2*v0_) + 2*kappa_*sigma_*(8*rho_*theta_ +
178  sigma_*t*theta_ - 4*rho_*v0_)) +
179  4*std::exp(kappa_*t)*(sigma2*theta_ -
180  2*kappa2*(-1 + rho_*sigma_*t)*(theta_ - v0_) +
181  kappa_*sigma_*(sigma_*t*(theta_ - v0_) + 2*rho_*(-2*theta_ +
182  v0_))))/(8.*std::exp(2*kappa_*t)*kappa3);
183  }
