34  MarketModelPathwiseSwap::MarketModelPathwiseSwap(const std::vector<Time>& rateTimes,
35  const std::vector<Real>& accruals,
36  const std::vector<Rate>& strikes,
37  Real multiplier)
38  : rateTimes_(rateTimes),
39  accruals_(accruals),
40  strikes_(strikes) ,
41  numberRates_(rateTimes.size()-1),
42  multiplier_(multiplier)
43  {
44  checkIncreasingTimes(rateTimes);
45  std::vector<Time> evolTimes(rateTimes_);
46  evolTimes.pop_back();
47 
48  QL_REQUIRE(evolTimes.size()==numberRates_,
49  "rateTimes.size()<> numberOfRates+1");
50 
51  if (strikes_.size() == 1)
52  strikes_ = std::vector<Rate>(numberRates_, strikes[0]);
53 
54  if (accruals_.size() == 1)
55  accruals_ = std::vector<Rate>(numberRates_, accruals[0]);
56 
57  QL_REQUIRE(accruals_.size()==numberRates_,
58  "accruals.size() does not equal numberOfRates or 1");
59 
60  QL_REQUIRE(strikes.size()==numberRates_,
61  "strikes.size() does not equal numberOfRates or 1" );
62 
63  evolution_ = EvolutionDescription(rateTimes,evolTimes);
64 
65  }

67  bool MarketModelPathwiseSwap::nextTimeStep(
68  const CurveState& currentState,
69  std::vector<Size>& numberCashFlowsThisStep,
70  std::vector<std::vector<MarketModelPathwiseMultiProduct::CashFlow> >& cashFlowsGenerated)
71  {
72  Rate liborRate = currentState.forwardRate(currentIndex_);
73  cashFlowsGenerated[0][0].timeIndex = currentIndex_+1;
74 
75  cashFlowsGenerated[0][0].amount[0] =
76  (liborRate-strikes_[currentIndex_])*accruals_[currentIndex_]*multiplier_;
77 
78  numberCashFlowsThisStep[0] = 1;
79 
80  for (Size i=1; i <= numberRates_; ++i)
81  cashFlowsGenerated[0][0].amount[i] =0;
82 
83  cashFlowsGenerated[0][0].amount[currentIndex_+1] = accruals_[currentIndex_]*multiplier_;
84 
85  ++currentIndex_;
86  return (currentIndex_ == strikes_.size());
87  }
