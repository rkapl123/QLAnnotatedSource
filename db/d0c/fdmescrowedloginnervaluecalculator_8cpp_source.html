38 Real FdmEscrowedLogInnerValueCalculator::innerValue(
39 const FdmLinearOpIterator& iter, Time t) {
40
41 const Real s_t = std::exp(mesher_->location(iter, direction_));
42 const Real spot = s_t - escrowedDividendAdj_->dividendAdjustment(t);
43
44 return (*payoff_)(spot);
45 }
