173  template <class SpeedInterpolationTraits,class VolInterpolationTraits>
174  void initialize(const Handle<YieldTermStructure>& yieldtermStructure,
175  const std::vector<Date>& speedstructure,
176  const std::vector<Date>& volstructure,
177  const std::vector<Real>& speed,
178  const std::vector<Real>& vol,
179  const SpeedInterpolationTraits &speedtraits,
180  const VolInterpolationTraits &voltraits,
181  const ext::function<Real(Real)>& f,
182  const ext::function<Real(Real)>& fInverse)
183  {
184  QL_REQUIRE(speedstructure.size()==speed.size(),
185  "mean reversion inputs inconsistent");
186  QL_REQUIRE(volstructure.size()==vol.size(),
187  "volatility inputs inconsistent");
188  if (!f_)
189  f_ = identity;
190  if (!fInverse_)
191  fInverse_ = identity;
192 
193  DayCounter dc = yieldtermStructure->dayCounter();
194  Date ref = yieldtermStructure->referenceDate();
195  for (Size i=0;i<speedstructure.size();i++)
196  speedperiods_.push_back(dc.yearFraction(ref,speedstructure[i]));
197  for (Size i=0;i<volstructure.size();i++)
198  volperiods_.push_back(dc.yearFraction(ref,volstructure[i]));
199 
200  // interpolator x points to *periods_ vector, y points to
201  // the internal Array in the parameter
202  InterpolationParameter atemp(speedperiods_.size(), NoConstraint());
203  a_ = atemp;
204  for (Size i=0; i<speedperiods_.size(); i++)
205  a_.setParam(i, speed[i]);
206  speed_ = speedtraits.interpolate(speedperiods_.begin(),
207  speedperiods_.end(),a_.params().begin());
208  speed_.enableExtrapolation();
209  atemp.reset(speed_);
210 
211  InterpolationParameter sigmatemp(volperiods_.size(), PositiveConstraint());
212  sigma_ = sigmatemp;
213  for (Size i=0; i<volperiods_.size(); i++)
214  sigma_.setParam(i, vol[i]);
215  vol_ = voltraits.interpolate(volperiods_.begin(),
216  volperiods_.end(),sigma_.params().begin());
217  vol_.enableExtrapolation();
218  sigmatemp.reset(vol_);
219 
220  generateArguments();
221  registerWith(yieldtermStructure);
222  }
223  };

285  class GeneralizedHullWhite::FittingParameter

290  Impl(const Handle<YieldTermStructure>& termStructure,

293 
294  Real value(const Array&, Time t) const {
295  Rate forwardRate =
296  termStructure_->forwardRate(t, t, Continuous, NoFrequency);
297  Real temp = a_ < std::sqrt(QL_EPSILON) ?
298  sigma_*t :
299  sigma_*(1.0 - std::exp(-a_*t))/a_;

301  }

353  namespace detail {
354  template <class I1, class I2>
355  class LinearFlatInterpolationImpl
356  : public Interpolation::templateImpl<I1,I2> {
357  public:
358  LinearFlatInterpolationImpl(const I1& xBegin, const I1& xEnd,
359  const I2& yBegin)
360  : Interpolation::templateImpl<I1,I2>(xBegin, xEnd, yBegin,
361  LinearFlat::requiredPoints),
362  primitiveConst_(xEnd-xBegin), s_(xEnd-xBegin) {}
363  void update() {
364  primitiveConst_[0] = 0.0;
365  for (Size i=1; i<Size(this->xEnd_-this->xBegin_); ++i) {
366  Real dx = this->xBegin_[i]-this->xBegin_[i-1];
367  s_[i-1] = (this->yBegin_[i]-this->yBegin_[i-1])/dx;
368  primitiveConst_[i] = primitiveConst_[i-1]
369  + dx*(this->yBegin_[i-1] +0.5*dx*s_[i-1]);
370  }
371  }
372  Real value(Real x) const {
373  if (x <= this->xMin())
374  return this->yBegin_[0];
375  if (x >= this->xMax())
376  return *(this->yBegin_+(this->xEnd_-this->xBegin_)-1);
377  Size i = this->locate(x);
378  return this->yBegin_[i] + (x-this->xBegin_[i])*s_[i];
379  }
380  Real primitive(Real x) const {
381  Size i = this->locate(x);
382  Real dx = x-this->xBegin_[i];
383  return primitiveConst_[i] +
384  dx*(this->yBegin_[i] + 0.5*dx*s_[i]);
385  }
386  Real derivative(Real x) const {
387  if (!this->isInRange(x))
388  return 0;
389  Size i = this->locate(x);
390  return s_[i];
391  }
392  Real secondDerivative(Real) const {
393  return 0.0;
394  }
395  private:
396  std::vector<Real> primitiveConst_, s_;
397  };
398  }
