170  template <class SpeedInterpolationTraits,class VolInterpolationTraits>
171  void initialize(const Handle<YieldTermStructure>& yieldtermStructure,
172  const std::vector<Date>& speedstructure,
173  const std::vector<Date>& volstructure,
174  const std::vector<Real>& speed,
175  const std::vector<Real>& vol,
176  const SpeedInterpolationTraits &speedtraits,
177  const VolInterpolationTraits &voltraits,
178  const ext::function<Real(Real)>& f,
179  const ext::function<Real(Real)>& fInverse)
180  {
181  QL_REQUIRE(speedstructure.size()==speed.size(),
182  "mean reversion inputs inconsistent");
183  QL_REQUIRE(volstructure.size()==vol.size(),
184  "volatility inputs inconsistent");
185  if (!f_)
186  f_ = identity;
187  if (!fInverse_)
188  fInverse_ = identity;
189 
190  DayCounter dc = yieldtermStructure->dayCounter();
191  Date ref = yieldtermStructure->referenceDate();
192  for (auto i : speedstructure)
193  speedperiods_.push_back(dc.yearFraction(ref, i));
194  for (auto i : volstructure)
195  volperiods_.push_back(dc.yearFraction(ref, i));
196 
197  // interpolator x points to *periods_ vector, y points to
198  // the internal Array in the parameter
199  InterpolationParameter atemp(speedperiods_.size(), NoConstraint());
200  a_ = atemp;
201  for (Size i=0; i<speedperiods_.size(); i++)
202  a_.setParam(i, speed[i]);
203  speed_ = speedtraits.interpolate(speedperiods_.begin(),
204  speedperiods_.end(),a_.params().begin());
205  speed_.enableExtrapolation();
206  atemp.reset(speed_);
207 
208  InterpolationParameter sigmatemp(volperiods_.size(), PositiveConstraint());
209  sigma_ = sigmatemp;
210  for (Size i=0; i<volperiods_.size(); i++)
211  sigma_.setParam(i, vol[i]);
212  vol_ = voltraits.interpolate(volperiods_.begin(),
213  volperiods_.end(),sigma_.params().begin());
214  vol_.enableExtrapolation();
215  sigmatemp.reset(vol_);
216 
217  generateArguments();
218  registerWith(yieldtermStructure);
219  }

342  namespace detail {
343  template <class I1, class I2>
344  class LinearFlatInterpolationImpl
345  : public Interpolation::templateImpl<I1,I2> {
346  public:
347  LinearFlatInterpolationImpl(const I1& xBegin, const I1& xEnd,
348  const I2& yBegin)
349  : Interpolation::templateImpl<I1,I2>(xBegin, xEnd, yBegin,
350  LinearFlat::requiredPoints),
351  primitiveConst_(xEnd-xBegin), s_(xEnd-xBegin) {}
352  void update() override {
353  primitiveConst_[0] = 0.0;
354  for (Size i=1; i<Size(this->xEnd_-this->xBegin_); ++i) {
355  Real dx = this->xBegin_[i]-this->xBegin_[i-1];
356  s_[i-1] = (this->yBegin_[i]-this->yBegin_[i-1])/dx;
357  primitiveConst_[i] = primitiveConst_[i-1]
358  + dx*(this->yBegin_[i-1] +0.5*dx*s_[i-1]);
359  }
360  }
361  Real value(Real x) const override {
362  if (x <= this->xMin())
363  return this->yBegin_[0];
364  if (x >= this->xMax())
365  return *(this->yBegin_+(this->xEnd_-this->xBegin_)-1);
366  Size i = this->locate(x);
367  return this->yBegin_[i] + (x-this->xBegin_[i])*s_[i];
368  }
369  Real primitive(Real x) const override {
370  Size i = this->locate(x);
371  Real dx = x-this->xBegin_[i];
372  return primitiveConst_[i] +
373  dx*(this->yBegin_[i] + 0.5*dx*s_[i]);
374  }
375  Real derivative(Real x) const override {
376  if (!this->isInRange(x))
377  return 0;
378  Size i = this->locate(x);
379  return s_[i];
380  }
381  Real secondDerivative(Real) const override { return 0.0; }
382 
383  private:
384  std::vector<Real> primitiveConst_, s_;
385  };
386  }
