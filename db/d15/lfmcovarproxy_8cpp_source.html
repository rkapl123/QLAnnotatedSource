51  Matrix pca = corrModel_->pseudoSqrt(t, x);
52  Array vol = volaModel_->volatility(t, x);

62  Disposable<Matrix> LfmCovarianceProxy::covariance(Time t,
63  const Array& x) const {
64 
65  Array volatility = volaModel_->volatility(t, x);
66  Matrix correlation = corrModel_->correlation(t, x);
67 
68  Matrix tmp(size_, size_);
69  for (Size i=0; i<size_; ++i) {
70  for (Size j=0; j<size_; ++j) {
71  tmp[i][j] = volatility[i]*correlation[i][j]*volatility[j];
72  }
73  }
74 
75  return tmp;
76  }

97  Real LfmCovarianceProxy::Var_Helper::operator()(Real t) const {
98  Volatility v1, v2;
99 
100  if (i_ == j_) {
101  v1 = v2 = volaModel_->volatility(i_, t);
102  } else {
103  v1 = volaModel_->volatility(i_, t);
104  v2 = volaModel_->volatility(j_, t);
105  }
106 
107  return v1 * corrModel_->correlation(i_, j_, t) * v2;
108  }

110  Real LfmCovarianceProxy::integratedCovariance(
111  Size i, Size j, Time t, const Array& x) const {
112 
113  if (corrModel_->isTimeIndependent()) {
114  try {
115  // if all objects support these methods
116  // thats by far the fastest way to get the
117  // integrated covariance
118  return corrModel_->correlation(i, j, 0.0, x)
119  * volaModel_->integratedVariance(j, i, t, x);
120  }
121  catch (Error&) {
122  // okay proceed with the
123  // slow numerical integration routine
124  }
125  }
126 
127  QL_REQUIRE(x.empty(), "can not handle given x here");
128 
129  Real tmp=0.0;
130  Var_Helper helper(this, i, j);
131 
132  GaussKronrodAdaptive integrator(1e-10, 10000);
133  for (Size k=0; k<64; ++k) {
134  tmp+=integrator(helper, k*t/64., (k+1)*t/64.);
135  }
136  return tmp;
137  }
