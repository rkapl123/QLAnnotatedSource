49  // the cost function with respect to x
50  virtual void gradient(Array& grad, const Array& x) const {
51  Real eps = finiteDifferenceEpsilon(), fp, fm;
52  Array xx(x);
53  for (Size i=0; i<x.size(); i++) {
54  xx[i] += eps;
55  fp = value(xx);
56  xx[i] -= 2.0*eps;
57  fm = value(xx);
58  grad[i] = 0.5*(fp - fm)/eps;
59  xx[i] = x[i];
60  }
61  }

73  virtual void jacobian(Matrix &jac, const Array &x) const {
74  Real eps = finiteDifferenceEpsilon();
75  Array xx(x), fp, fm;
76  for(Size i=0; i<x.size(); ++i) {
77  xx[i] += eps;
78  fp = values(xx);
79  xx[i] -= 2.0*eps;
80  fm = values(xx);
81  for(Size j=0; j<fp.size(); ++j) {
82  jac[j][i] = 0.5*(fp[j]-fm[j])/eps;
83  }
84  xx[i] = x[i];
85  }
86  }
