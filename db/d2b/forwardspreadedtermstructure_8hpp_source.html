107  inline void ForwardSpreadedTermStructure::update() {
108  if (!originalCurve_.empty()) {
109  YieldTermStructure::update();
110  } else {
111  /* The implementation inherited from YieldTermStructure
112  asks for our reference date, which we don't have since
113  the original curve is still not set. Therefore, we skip
114  over that and just call the base-class behavior. */
115  // NOLINTNEXTLINE(bugprone-parent-virtual-call)
116  TermStructure::update();
117  }
118  }
