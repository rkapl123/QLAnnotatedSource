34 //! Term structure with added spread on the instantaneous forward rate
35 /*! \note This term structure will remain linked to the original
36 structure, i.e., any changes in the latter will be
37 reflected in this structure as well.
38
39 \ingroup yieldtermstructures
40
41 \test
42 - the correctness of the returned values is tested by
43 checking them against numerical calculations.
44 - observability against changes in the underlying term
45 structure and in the added spread is checked.
46 */
47 class ForwardSpreadedTermStructure : public ForwardRateStructure {
48 public:
49 ForwardSpreadedTermStructure(Handle<YieldTermStructure>, Handle<Quote> spread);
50 //! \name TermStructure interface
51 //@{
52 DayCounter dayCounter() const override;
53 Date maxDate() const override;
54 Time maxTime() const override;
55 const Date& referenceDate() const override;
56 Calendar calendar() const override;
57 Natural settlementDays() const override;
58 //@}
59 //! \name Observer interface
60 //@{
61 void update() override;
62 //@}
63 protected:
64 //! \name ForwardRateStructure implementation
65 //@{
66 Rate forwardImpl(Time t) const override;
67 /* This method must disappear should the spread become a curve */
68 Rate zeroYieldImpl(Time t) const override;
69 //@}
70 private:
71 Handle<YieldTermStructure> originalCurve_;
72 Handle<Quote> spread_;
73 };

106 inline void ForwardSpreadedTermStructure::update() {
107 if (!originalCurve_.empty()) {
108 YieldTermStructure::update();
109 } else {
110 /* The implementation inherited from YieldTermStructure
111 asks for our reference date, which we don't have since
112 the original curve is still not set. Therefore, we skip
113 over that and just call the base-class behavior. */
114 // NOLINTNEXTLINE(bugprone-parent-virtual-call)
115 TermStructure::update();
116 }
117 }
118
119 inline Rate ForwardSpreadedTermStructure::forwardImpl(Time t) const {
120 return originalCurve_->forwardRate(t, t, Continuous, NoFrequency, true)
121 + spread_->value();
122 }
123
124 inline Rate ForwardSpreadedTermStructure::zeroYieldImpl(Time t) const {
125 return originalCurve_->zeroRate(t, Continuous, NoFrequency, true)
126 + spread_->value();
127 }
