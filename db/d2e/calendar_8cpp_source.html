30  void Calendar::addHoliday(const Date& d) {
31  QL_REQUIRE(impl_, "no calendar implementation provided");
32 
33 #ifdef QL_HIGH_RESOLUTION_DATE
34  const Date _d(d.dayOfMonth(), d.month(), d.year());
35 #else
36  const Date& _d = d;
37 #endif
38 
39  // if d was a genuine holiday previously removed, revert the change
40  impl_->removedHolidays.erase(_d);
41  // if it's already a holiday, leave the calendar alone.
42  // Otherwise, add it.
43  if (impl_->isBusinessDay(_d))
44  impl_->addedHolidays.insert(_d);
45  }

64  Date Calendar::adjust(const Date& d,
65  BusinessDayConvention c) const {
66  QL_REQUIRE(d != Date(), "null date");
67 
68  if (c == Unadjusted)
69  return d;
70 
71  Date d1 = d;
72  if (c == Following || c == ModifiedFollowing
73  || c == HalfMonthModifiedFollowing) {
74  while (isHoliday(d1))
75  ++d1;
76  if (c == ModifiedFollowing
77  || c == HalfMonthModifiedFollowing) {
78  if (d1.month() != d.month()) {
79  return adjust(d, Preceding);
80  }
81  if (c == HalfMonthModifiedFollowing) {
82  if (d.dayOfMonth() <= 15 && d1.dayOfMonth() > 15) {
83  return adjust(d, Preceding);
84  }
85  }
86  }
87  } else if (c == Preceding || c == ModifiedPreceding) {
88  while (isHoliday(d1))
89  --d1;
90  if (c == ModifiedPreceding && d1.month() != d.month()) {
91  return adjust(d,Following);
92  }
93  } else if (c == Nearest) {
94  Date d2 = d;
95  while (isHoliday(d1) && isHoliday(d2))
96  {
97  ++d1;
98  --d2;
99  }
100  if (isHoliday(d1))
101  return d2;
102  else
103  return d1;
104  } else {
105  QL_FAIL("unknown business-day convention");
106  }
107  return d1;
108  }

110  Date Calendar::advance(const Date& d,
111  Integer n, TimeUnit unit,
112  BusinessDayConvention c,
113  bool endOfMonth) const {
114  QL_REQUIRE(d!=Date(), "null date");
115  if (n == 0) {
116  return adjust(d,c);
117  } else if (unit == Days) {
118  Date d1 = d;
119  if (n > 0) {
120  while (n > 0) {
121  ++d1;
122  while (isHoliday(d1))
123  ++d1;
124  --n;
125  }
126  } else {
127  while (n < 0) {
128  --d1;
129  while(isHoliday(d1))
130  --d1;
131  ++n;
132  }
133  }
134  return d1;
135  } else if (unit == Weeks) {
136  Date d1 = d + n*unit;
137  return adjust(d1,c);
138  } else {
139  Date d1 = d + n*unit;
140 
141  // we are sure the unit is Months or Years
142  if (endOfMonth && isEndOfMonth(d))
143  return Calendar::endOfMonth(d1);
144 
145  return adjust(d1, c);
146  }
147  }

156  Date::serial_type Calendar::businessDaysBetween(const Date& from,
157  const Date& to,
158  bool includeFirst,
159  bool includeLast) const {
160  Date::serial_type wd = 0;
161  if (from != to) {
162  if (from < to) {
163  // the last one is treated separately to avoid
164  // incrementing Date::maxDate()
165  for (Date d = from; d < to; ++d) {
166  if (isBusinessDay(d))
167  ++wd;
168  }
169  if (isBusinessDay(to))
170  ++wd;
171  } else if (from > to) {
172  for (Date d = to; d < from; ++d) {
173  if (isBusinessDay(d))
174  ++wd;
175  }
176  if (isBusinessDay(from))
177  ++wd;
178  }
179 
180  if (isBusinessDay(from) && !includeFirst)
181  --wd;
182  if (isBusinessDay(to) && !includeLast)
183  --wd;
184 
185  if (from > to)
186  wd = -wd;
187  } else if (includeFirst && includeLast && isBusinessDay(from)) {
188  wd = 1;
189  }
190 
191  return wd;
192  }

281  const Date& from, const Date& to, bool includeWeekEnds) const {
282 
283  QL_REQUIRE(to>from, "'from' date ("
284  << from << ") must be earlier than 'to' date ("
285  << to << ")");
286  std::vector<Date> result;
287  for (Date d = from; d <= to; ++d) {
288  if (isHoliday(d) && (includeWeekEnds || !isWeekend(d.weekday())))
289  result.push_back(d);
290  }
