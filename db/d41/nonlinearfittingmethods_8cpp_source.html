55  DiscountFactor ExponentialSplinesFitting::discountFunction(const Array& x,
56  Time t) const {
57  DiscountFactor d = 0.0;
58  Size N = size();
59  Real kappa = x[N-1];
60  Real coeff = 0;
61 
62  if (!constrainAtZero_) {
63  for (Size i=0; i<N-1; ++i) {
64  d += x[i]* std::exp(-kappa * (i+1) * t);
65  }
66  } else {
67  // notation:
68  // d(t) = coeff* exp(-kappa*1*t) + x[0]* exp(-kappa*2*t) +
69  // x[1]* exp(-kappa*3*t) + ..+ x[7]* exp(-kappa*9*t)
70  for (Size i=0; i<N-1; i++) {
71  d += x[i]* std::exp(-kappa * (i+2) * t);
72  coeff += x[i];
73  }
74  coeff = 1.0- coeff;
75  d += coeff * std::exp(-kappa * t);
76  }
77  return d;
78  }

106  DiscountFactor NelsonSiegelFitting::discountFunction(const Array& x,
107  Time t) const {
108  Real kappa = x[size()-1];
109  Real zeroRate = x[0] + (x[1] + x[2])*
110  (1.0 - std::exp(-kappa*t))/
111  ((kappa+QL_EPSILON)*(t+QL_EPSILON)) -
112  (x[2])*std::exp(-kappa*t);
113  DiscountFactor d = std::exp(-zeroRate * t) ;
114  return d;
115  }

141  DiscountFactor SvenssonFitting::discountFunction(const Array& x,
142  Time t) const {
143  Real kappa = x[size()-2];
144  Real kappa_1 = x[size()-1];
145 
146  Real zeroRate = x[0] + (x[1] + x[2])*
147  (1.0 - std::exp(-kappa*t))/
148  ((kappa+QL_EPSILON)*(t+QL_EPSILON)) -
149  (x[2])*std::exp(-kappa*t) +
150  x[3]* (((1.0 - std::exp(-kappa_1*t))/((kappa_1+QL_EPSILON)*(t+QL_EPSILON)))- std::exp(-kappa_1*t));
151  DiscountFactor d = std::exp(-zeroRate * t) ;
152  return d;
153  }

156  CubicBSplinesFitting::CubicBSplinesFitting(
157  const std::vector<Time>& knots,
158  bool constrainAtZero,
159  const Array& weights,
160  const ext::shared_ptr<OptimizationMethod>& optimizationMethod,
161  const Array& l2,
162  const Real minCutoffTime,
163  const Real maxCutoffTime)
164  : FittedBondDiscountCurve::FittingMethod(
165  constrainAtZero, weights, optimizationMethod, l2, minCutoffTime, maxCutoffTime),
166  splines_(3, knots.size() - 5, knots) {
167 
168  QL_REQUIRE(knots.size() >= 8,
169  "At least 8 knots are required" );
170  Size basisFunctions = knots.size() - 4;
171 
172  if (constrainAtZero) {
173  size_ = basisFunctions-1;
174 
175  // Note: A small but nonzero N_th basis function at t=0 may
176  // lead to an ill conditioned problem
177  N_ = 1;
178 
179  QL_REQUIRE(std::abs(splines_(N_, 0.0)) > QL_EPSILON,
180  "N_th cubic B-spline must be nonzero at t=0");
181  } else {
182  size_ = basisFunctions;
183  N_ = 0;
184  }
185  }

230  DiscountFactor CubicBSplinesFitting::discountFunction(const Array& x,
231  Time t) const {
232  DiscountFactor d = 0.0;
233 
234  if (!constrainAtZero_) {
235  for (Size i=0; i<size_; ++i) {
236  d += x[i] * splines_(i,t);
237  }
238  } else {
239  const Real T = 0.0;
240  Real sum = 0.0;
241  for (Size i=0; i<size_; ++i) {
242  if (i < N_) {
243  d += x[i] * splines_(i,t);
244  sum += x[i] * splines_(i,T);
245  } else {
246  d += x[i] * splines_(i+1,t);
247  sum += x[i] * splines_(i+1,T);
248  }
249  }
250  Real coeff = 1.0 - sum;
251  coeff /= splines_(N_,T);
252  d += coeff * splines_(N_,t);
253  }
254 
255  return d;
256  }
