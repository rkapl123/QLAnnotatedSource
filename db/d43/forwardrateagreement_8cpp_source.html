78  // In theory, no need to implement this for a FRA (could directly
79  // supply a forwardValue). For the sake of keeping a consistent
80  // framework, we adhere to the concept of the forward contract as
81  // defined in the base class, with an underlying having a
82  // spotPrice (in this case, a loan or deposit with an NPV). Thus,
83  // spotValue() is defined here.
84  Real ForwardRateAgreement::spotValue() const {
85  calculate();
86  return notionalAmount_ *
87  forwardRate().compoundFactor(valueDate_, maturityDate_) *
88  discountCurve_->discount(maturityDate_);
89  }

108  void ForwardRateAgreement::calculateForwardRate() const {
109  if (useIndexedCoupon_)
110  forwardRate_ =
111  InterestRate(index_->fixing(fixingDate()), index_->dayCounter(), Simple, Once);
112  else
113  // par coupon approximation
114  forwardRate_ =
115  InterestRate((index_->forwardingTermStructure()->discount(valueDate_) /
116  index_->forwardingTermStructure()->discount(maturityDate_) -
117  1.0) /
118  index_->dayCounter().yearFraction(valueDate_, maturityDate_),
119  index_->dayCounter(), Simple, Once);
120  }
