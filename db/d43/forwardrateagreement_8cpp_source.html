30 ForwardRateAgreement::ForwardRateAgreement(const Date& valueDate,
31 const Date& maturityDate,
32 Position::Type type,
33 Rate strikeForwardRate,
34 Real notionalAmount,

36 Handle<YieldTermStructure> discountCurve,

42 discountCurve_(std::move(discountCurve)) {
43
44 maturityDate_ = calendar_.adjust(maturityDate_, businessDayConvention_);
45
46 registerWith(Settings::instance().evaluationDate());
47 registerWith(discountCurve_);
48

50
51 strikeForwardRate_ = InterestRate(strikeForwardRate,
52 index->dayCounter(),
53 Simple, Once);

55 strikeForwardRate_.compoundFactor(valueDate_,
56 maturityDate_);

59 strike));
60 // incomeDiscountCurve_ is irrelevant to an FRA
61 incomeDiscountCurve_ = discountCurve_;
62 // income is irrelevant to FRA - set it to zero
63 underlyingIncome_ = 0.0;
64
65 registerWith(index_);
66 }

97 // In theory, no need to implement this for a FRA (could directly
98 // supply a forwardValue). For the sake of keeping a consistent
99 // framework, we adhere to the concept of the forward contract as
100 // defined in the base class, with an underlying having a
101 // spotPrice (in this case, a loan or deposit with an NPV). Thus,
102 // spotValue() is defined here.
103 Real ForwardRateAgreement::spotValue() const {
104 calculate();
105
106 Handle<YieldTermStructure> discount =
107 discountCurve_.empty() ? index_->forwardingTermStructure() : discountCurve_;
108
109 return notionalAmount_ *
110 forwardRate().compoundFactor(valueDate_, maturityDate_) *
111 discount->discount(maturityDate_);
112 }

144 void ForwardRateAgreement::calculateForwardRate() const {
145 if (useIndexedCoupon_)
146 forwardRate_ =
147 InterestRate(index_->fixing(fixingDate()), index_->dayCounter(), Simple, Once);
148 else
149 // par coupon approximation
150 forwardRate_ =
151 InterestRate((index_->forwardingTermStructure()->discount(valueDate_) /
152 index_->forwardingTermStructure()->discount(maturityDate_) -
153 1.0) /
154 index_->dayCounter().yearFraction(valueDate_, maturityDate_),
155 index_->dayCounter(), Simple, Once);
156 }

158 void ForwardRateAgreement::calculateAmount() const {
159 calculateForwardRate();
160 Integer sign = fraType_ == Position::Long? 1 : -1;
161
162 Rate F = forwardRate_.rate();
163 Rate K = strikeForwardRate_.rate();
164 Time T = forwardRate_.dayCounter().yearFraction(valueDate_, maturityDate_);
165
166 amount_ = notionalAmount_ * sign * (F - K) * T / (1.0 + F * T);
167 }

180 InterestRate ForwardRateAgreement::impliedYield(Real underlyingSpotValue,
181 Real forwardValue,
182 Date settlementDate,
183 Compounding comp,
184 const DayCounter& dayCounter) {
185
186 Time t = dayCounter.yearFraction(settlementDate, maturityDate_);
187 Real compoundingFactor =
188 forwardValue / (underlyingSpotValue - spotIncome(incomeDiscountCurve_));
189 return InterestRate::impliedRate(compoundingFactor, dayCounter, comp, Annual, t);
190 }
