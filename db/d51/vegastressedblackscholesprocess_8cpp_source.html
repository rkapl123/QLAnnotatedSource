101 Real VegaStressedBlackScholesProcess::diffusion(Time t, Real x) const {
102 if (lowerTimeBorderForStressTest_ <= t && t <= upperTimeBorderForStressTest_
103 && lowerAssetBorderForStressTest_ <= x && x <= upperAssetBorderForStressTest_) {
104 return GeneralizedBlackScholesProcess::diffusion(t, x)+stressLevel_;
105 } else {
106 return GeneralizedBlackScholesProcess::diffusion(t, x);
107 }
