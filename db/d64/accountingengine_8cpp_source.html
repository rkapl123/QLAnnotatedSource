29  AccountingEngine::AccountingEngine(
30  const boost::shared_ptr<MarketModelEvolver>& evolver,
31  const Clone<MarketModelMultiProduct>& product,
32  Real initialNumeraireValue)
33  : evolver_(evolver), product_(product),
34  initialNumeraireValue_(initialNumeraireValue),
35  numberProducts_(product->numberOfProducts()),
36  numerairesHeld_(product->numberOfProducts()),
37  numberCashFlowsThisStep_(product->numberOfProducts()),
38  cashFlowsGenerated_(product->numberOfProducts()) {
39  for (Size i=0; i<numberProducts_; ++i)

47  for (Size j=0; j<cashFlowTimes.size(); ++j)

49  rateTimes));

53  Real AccountingEngine::singlePathValues(std::vector<Real>& values) {
54  std::fill(numerairesHeld_.begin(), numerairesHeld_.end(), 0.0);
55  Real weight = evolver_->startNewPath();
56  product_->reset();
57  Real principalInNumerairePortfolio = 1.0;
58 
59  bool done = false;
60  do {
61  Size thisStep = evolver_->currentStep();
62  weight *= evolver_->advanceStep();
63  done = product_->nextTimeStep(evolver_->currentState(),
64  numberCashFlowsThisStep_,
65  cashFlowsGenerated_);
66  Size numeraire =
67  evolver_->numeraires()[thisStep];
68 
69  // for each product...
70  for (Size i=0; i<numberProducts_; ++i) {
71  // ...and each cash flow...
72  const std::vector<MarketModelMultiProduct::CashFlow>& cashflows =
73  cashFlowsGenerated_[i];
74  for (Size j=0; j<numberCashFlowsThisStep_[i]; ++j) {
75  // ...convert the cash flow to numeraires.
76  // This is done by calculating the number of
77  // numeraire bonds corresponding to such cash flow...
78  const MarketModelDiscounter& discounter =
79  discounters_[cashflows[j].timeIndex];
80 
81  Real bonds = cashflows[j].amount *
82  discounter.numeraireBonds(evolver_->currentState(),
83  numeraire);
84 
85  // ...and adding the newly bought bonds to the number
86  // of numeraires held.
87  numerairesHeld_[i] += bonds/principalInNumerairePortfolio;
88  }
89  }
90 
91  if (!done) {
92 
93  // The numeraire might change between steps. This implies
94  // that we might have to convert the numeraire bonds for
95  // this step into a corresponding amount of numeraire
96  // bonds for the next step. This can be done by changing
97  // the principal of the numeraire and updating the number
98  // of bonds in the numeraire portfolio accordingly.
99 
100  Size nextNumeraire = evolver_->numeraires()[thisStep+1];
101 
102  principalInNumerairePortfolio *=
103  evolver_->currentState().discountRatio(numeraire,
104  nextNumeraire);
105  }
106 
107  } while (!done);
108 
109  for (Size i=0; i<numerairesHeld_.size(); ++i)
110  values[i] = numerairesHeld_[i] * initialNumeraireValue_;
111 
112  return weight;
113  }
