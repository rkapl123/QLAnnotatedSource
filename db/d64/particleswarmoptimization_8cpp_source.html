31  ParticleSwarmOptimization::ParticleSwarmOptimization(Size M,
32  boost::shared_ptr<Topology> topology,
33  boost::shared_ptr<Inertia> inertia,
34  Real c1, Real c2,
35  unsigned long seed)
36  : M_(M), rng_(seed),
37  topology_(topology),
38  inertia_(inertia) {
39  Real phi = c1 + c2;
40  QL_ENSURE(phi*phi - 4 * phi, "Invalid phi");
41  c0_ = 2.0 / std::abs(2.0 - phi - sqrt(phi*phi - 4 * phi));
42  c1_ = c0_*c1;
43  c2_ = c0_*c2;
44  }

54  void ParticleSwarmOptimization::startState(Problem &P, const EndCriteria &endCriteria) {
55  QL_REQUIRE(topology_, "Invalid topology");
56  QL_REQUIRE(inertia_, "Invalid inertia");
57  N_ = P.currentValue().size();
58  topology_->setSize(M_);
59  inertia_->setSize(M_, N_, c0_, endCriteria);
60  X_.reserve(M_);
61  V_.reserve(M_);
62  pBX_.reserve(M_);
63  pBF_ = Array(M_);
64  gBX_.reserve(M_);
65  gBF_ = Array(M_);
66  uX_ = P.constraint().upperBound(P.currentValue());
67  lX_ = P.constraint().lowerBound(P.currentValue());
68  Array bounds = uX_ - lX_;
69 
70  //Random initialization is done by Sobol sequence
71  SobolRsg sobol(N_ * 2);
72 
73  //Prepare containers
74  for (Size i = 0; i < M_; i++) {
75  const SobolRsg::sample_type::value_type &sample = sobol.nextSequence().value;
76  X_.push_back(Array(N_, 0.0));
77  Array& x = X_.back();
78  V_.push_back(Array(N_, 0.0));
79  Array& v = V_.back();
80  gBX_.push_back(Array(N_, 0.0));
81  for (Size j = 0; j < N_; j++) {
82  //Assign X=lb+(ub-lb)*random
83  x[j] = lX_[j] + bounds[j] * sample[2 * j];
84  //Assign V=(ub-lb)*2*random-(ub-lb) -> between (lb-ub) and (ub-lb)
85  v[j] = bounds[j] * (2.0*sample[2 * j + 1] - 1.0);
86  }
87  //Evaluate X and assign as personal best
88  pBX_.push_back(X_.back());
89  pBF_[i] = P.value(X_.back());
90  }
91 
92  //init topology & inertia
93  topology_->init(this);
94  inertia_->init(this);
95  }

97  EndCriteria::Type ParticleSwarmOptimization::minimize(Problem &P, const EndCriteria &endCriteria) {
98  QL_REQUIRE(!P.constraint().empty(), "PSO is a constrained optimizer");
99 
100  EndCriteria::Type ecType = EndCriteria::None;
101  P.reset();
102  Size iteration = 0;
103  Size iterationStat = 0;
104  Size maxIteration = endCriteria.maxIterations();
105  Size maxIStationary = endCriteria.maxStationaryStateIterations();
106  Real bestValue = QL_MAX_REAL;
107  Size bestPosition = 0;
108 
109  startState(P, endCriteria);
110  //Set best value & position
111  for (Size i = 0; i < M_; i++) {
112  if (pBF_[i] < bestValue) {
113  bestValue = pBF_[i];
114  bestPosition = i;
115  }
116  }
117 
118  //Run optimization
119  do {
120  iteration++;
121  iterationStat++;
122  //Check if stopping criteria is met
123  if (iteration > maxIteration || iterationStat > maxIStationary)
124  break;
125 
126  //According to the topology, determine best global position
127  topology_->findSocialBest();
128 
129  //Call inertia to change internal state
130  inertia_->setValues();
131 
132  //Loop over particles
133  for (Size i = 0; i < M_; i++) {
134  Array& x = X_[i];
135  Array& pB = pBX_[i];
136  const Array& gB = gBX_[i];
137  Array& v = V_[i];
138 
139  //Loop over dimensions
140  for (Size j = 0; j < N_; j++) {
141  //Update velocity
142  v[j] += c1_*rng_.nextReal()*(pB[j] - x[j]) + c2_*rng_.nextReal()*(gB[j] - x[j]);
143  //Update position
144  x[j] += v[j];
145  //Enforce bounds on positions
146  if (x[j] < lX_[j]) {
147  x[j] = lX_[j];
148  v[j] = 0.0;
149  }
150  else if (x[j] > uX_[j]) {
151  x[j] = uX_[j];
152  v[j] = 0.0;
153  }
154  }
155  //Evaluate x
156  Real f = P.value(x);
157  if (f < pBF_[i]) {
158  //Update personal best
159  pBF_[i] = f;
160  pB = x;
161  //Check stationary condition
162  if (f < bestValue) {
163  bestValue = f;
164  bestPosition = i;
165  iterationStat = 0;
166  }
167  }
168  }
169  } while (true);
170  if (iteration > maxIteration)
171  ecType = EndCriteria::MaxIterations;
172  else
173  ecType = EndCriteria::StationaryPoint;
174 
175  //Set result to best point
176  P.setCurrentValue(pBX_[bestPosition]);
177  P.setFunctionValue(bestValue);
178  return ecType;
179  }

181  void AdaptiveInertia::setValues() {
182  Real currBest = (*pBF_)[0];
183  for (Size i = 1; i < M_; i++) {
184  if (currBest >(*pBF_)[i]) currBest = (*pBF_)[i];
185  }
186  if (started_) { //First iteration leaves inertia unchanged
187  if (currBest < best_) {
188  best_ = currBest;
189  adaptiveCounter--;
190  }
191  else {
192  adaptiveCounter++;
193  }
194  if (adaptiveCounter > sh_) {
195  c0_ = std::max(minInertia_, std::min(maxInertia_, c0_*0.5));
196  }
197  else if (adaptiveCounter < sl_) {
198  c0_ = std::max(minInertia_, std::min(maxInertia_, c0_*2.0));
199  }
200  }
201  else {
202  best_ = currBest;
203  started_ = true;
204  }
205  for (Size i = 0; i < M_; i++) {
206  (*V_)[i] *= c0_;
207  }

210  void KNeighbors::findSocialBest() {
211  for (Size i = 0; i < M_; i++) {
212  Real bestF = (*pBF_)[i];
213  Size bestX = 0;
214  //Search K_ neightbors upwards
215  Size upper = std::min(i + K_, M_);
216  //Search K_ neighbors downwards
217  Size lower = std::max(i, K_ + 1) - K_ - 1;
218  for (Size j = lower; j < upper; j++) {
219  if ((*pBF_)[j] < bestF) {
220  bestF = (*pBF_)[j];
221  bestX = j;
222  }
223  }
224  if (i + K_ >= M_) { //loop around if i+K >= M_
225  for (Size j = 0; j < i + K_ - M_; j++) {
226  if ((*pBF_)[j] < bestF) {
227  bestF = (*pBF_)[j];
228  bestX = j;
229  }
230  }
231  }
232  else if (i < K_) {//loop around from above
233  for (Size j = M_ - (K_ - i) - 1; j < M_; j++) {
234  if ((*pBF_)[j] < bestF) {
235  bestF = (*pBF_)[j];
236  bestX = j;
237  }
238  }
239  }
240  (*gBX_)[i] = (*pBX_)[bestX];
241  (*gBF_)[i] = bestF;
242  }
243  }

288  void ClubsTopology::findSocialBest() {
289  //Update iteration
290  iteration_++;
291  bool reset = false;
292  if (iteration_ == resetIteration_) {
293  iteration_ = 0;
294  reset = true;
295  }
296 
297  //Find best by current club
298  for (Size i = 0; i < totalClubs_; i++) {
299  Real bestByClub = QL_MAX_REAL;
300  Real worstByClub = -QL_MAX_REAL;
301  Size bestP = 0;
302  Size worstP = 0;
303  const std::vector<bool> &particlesSet = particles4clubs_[i];
304  for (Size j = 0; j < M_; j++) {
305  if (particlesSet[j]) {
306  if (bestByClub >(*pBF_)[j]) {
307  bestByClub = (*pBF_)[j];
308  bestP = j;
309  }
310  else if (worstByClub < (*pBF_)[j]) {
311  worstByClub = (*pBF_)[j];
312  worstP = j;
313  }
314  }
315  }
316  bestByClub_[i] = bestP;
317  worstByClub_[i] = worstP;
318  }
319 
320  //Update clubs && global best
321  for (Size i = 0; i < M_; i++) {
322  std::vector<bool> &clubSet = clubs4particles_[i];
323  bool best = true;
324  bool worst = true;
325  Size currentClubs = 0;
326  for (Size j = 0; j < totalClubs_; j++) {
327  if (clubSet[j]) {
328  //If still thought of the best, check if best in club j
329  if (best && i != bestByClub_[j]) best = false;
330  //If still thought of the worst, check if worst in club j
331  if (worst && i != worstByClub_[j]) worst = false;
332  //Update currentClubs
333  currentClubs++;
334  }
335  }
336  //Update clubs
337  if (best) {
338  //Leave random club
339  leaveRandomClub(i, currentClubs);
340  }
341  else if (worst) {
342  //Join random club
343  joinRandomClub(i, currentClubs);
344  }
345  else if (reset && currentClubs != defaultClubs_) {
346  //If membership != defaultClubs_, then leave or join accordingly
347  if (currentClubs < defaultClubs_) {
348  //Join random club
349  joinRandomClub(i, currentClubs);
350  }
351  else {
352  //Leave random club
353  leaveRandomClub(i, currentClubs);
354  }
355  }
356 
357  //Update global best
358  Real bestNeighborF = QL_MAX_REAL;
359  Size bestNeighborX = 0;
360  for (Size j = 0; j < totalClubs_; j++) {
361  if (clubSet[j] && bestNeighborF >(*pBF_)[bestByClub_[j]]) {
362  bestNeighborF = (*pBF_)[bestByClub_[j]];
363  bestNeighborX = j;
364  }
365  }
366  (*gBX_)[i] = (*pBX_)[bestNeighborX];
367  (*gBF_)[i] = bestNeighborF;
368  }
369  }

371  void ClubsTopology::leaveRandomClub(Size particle, Size currentClubs) {
372  Size randIndex = distribution_(generator_,

388  void ClubsTopology::joinRandomClub(Size particle, Size currentClubs) {
389  Size randIndex = totalClubs_ == currentClubs ? 1 :
390  distribution_(generator_, uniform_integer::param_type(1, totalClubs_ - currentClubs));
391  Size index = 1;
392  std::vector<bool> &clubSet = clubs4particles_[particle];
393  for (Size j = 0; j < totalClubs_; j++) {
394  if (!clubSet[j]) {
395  if (index == randIndex) {
396  clubSet[j] = true;
397  particles4clubs_[j][particle] = true;
398  break;
399  }
400  index++;
401  }
