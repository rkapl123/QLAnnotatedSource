27  ParticleSwarmOptimization::ParticleSwarmOptimization(Size M,

50  void ParticleSwarmOptimization::startState(Problem &P, const EndCriteria &endCriteria) {
51  QL_REQUIRE(topology_, "Invalid topology");
52  QL_REQUIRE(inertia_, "Invalid inertia");
53  N_ = P.currentValue().size();
54  topology_->setSize(M_);
55  inertia_->setSize(M_, N_, c0_, endCriteria);
56  X_.reserve(M_);
57  V_.reserve(M_);
58  pBX_.reserve(M_);
59  pBF_ = Array(M_);
60  gBX_.reserve(M_);
61  gBF_ = Array(M_);
62  uX_ = P.constraint().upperBound(P.currentValue());
63  lX_ = P.constraint().lowerBound(P.currentValue());
64  Array bounds = uX_ - lX_;
65 
66  //Random initialization is done by Sobol sequence
67  SobolRsg sobol(N_ * 2);
68 
69  //Prepare containers
70  for (Size i = 0; i < M_; i++) {
71  const SobolRsg::sample_type::value_type &sample = sobol.nextSequence().value;
72  X_.push_back(Array(N_, 0.0));
73  Array& x = X_.back();
74  V_.push_back(Array(N_, 0.0));
75  Array& v = V_.back();
76  gBX_.push_back(Array(N_, 0.0));
77  for (Size j = 0; j < N_; j++) {
78  //Assign X=lb+(ub-lb)*random
79  x[j] = lX_[j] + bounds[j] * sample[2 * j];
80  //Assign V=(ub-lb)*2*random-(ub-lb) -> between (lb-ub) and (ub-lb)
81  v[j] = bounds[j] * (2.0*sample[2 * j + 1] - 1.0);
82  }
83  //Evaluate X and assign as personal best
84  pBX_.push_back(X_.back());
85  pBF_[i] = P.value(X_.back());
86  }
87 
88  //init topology & inertia
89  topology_->init(this);
90  inertia_->init(this);
91  }

93  EndCriteria::Type ParticleSwarmOptimization::minimize(Problem &P, const EndCriteria &endCriteria) {
94  QL_REQUIRE(!P.constraint().empty(), "PSO is a constrained optimizer");
95 
96  EndCriteria::Type ecType = EndCriteria::None;
97  P.reset();
98  Size iteration = 0;
99  Size iterationStat = 0;
100  Size maxIteration = endCriteria.maxIterations();
101  Size maxIStationary = endCriteria.maxStationaryStateIterations();
102  Real bestValue = QL_MAX_REAL;
103  Size bestPosition = 0;
104 
105  startState(P, endCriteria);
106  //Set best value & position
107  for (Size i = 0; i < M_; i++) {
108  if (pBF_[i] < bestValue) {
109  bestValue = pBF_[i];
110  bestPosition = i;
111  }
112  }
113 
114  //Run optimization
115  do {
116  iteration++;
117  iterationStat++;
118  //Check if stopping criteria is met
119  if (iteration > maxIteration || iterationStat > maxIStationary)
120  break;
121 
122  //According to the topology, determine best global position
123  topology_->findSocialBest();
124 
125  //Call inertia to change internal state
126  inertia_->setValues();
127 
128  //Loop over particles
129  for (Size i = 0; i < M_; i++) {
130  Array& x = X_[i];
131  Array& pB = pBX_[i];
132  const Array& gB = gBX_[i];
133  Array& v = V_[i];
134 
135  //Loop over dimensions
136  for (Size j = 0; j < N_; j++) {
137  //Update velocity
138  v[j] += c1_*rng_.nextReal()*(pB[j] - x[j]) + c2_*rng_.nextReal()*(gB[j] - x[j]);
139  //Update position
140  x[j] += v[j];
141  //Enforce bounds on positions
142  if (x[j] < lX_[j]) {
143  x[j] = lX_[j];
144  v[j] = 0.0;
145  }
146  else if (x[j] > uX_[j]) {
147  x[j] = uX_[j];
148  v[j] = 0.0;
149  }
150  }
151  //Evaluate x
152  Real f = P.value(x);
153  if (f < pBF_[i]) {
154  //Update personal best
155  pBF_[i] = f;
156  pB = x;
157  //Check stationary condition
158  if (f < bestValue) {
159  bestValue = f;
160  bestPosition = i;
161  iterationStat = 0;
162  }
163  }
164  }
165  } while (true);
166  if (iteration > maxIteration)
167  ecType = EndCriteria::MaxIterations;
168  else
169  ecType = EndCriteria::StationaryPoint;
170 
171  //Set result to best point
172  P.setCurrentValue(pBX_[bestPosition]);
173  P.setFunctionValue(bestValue);
174  return ecType;
175  }

177  void AdaptiveInertia::setValues() {
178  Real currBest = (*pBF_)[0];
179  for (Size i = 1; i < M_; i++) {
180  if (currBest >(*pBF_)[i]) currBest = (*pBF_)[i];
181  }
182  if (started_) { //First iteration leaves inertia unchanged
183  if (currBest < best_) {
184  best_ = currBest;
185  adaptiveCounter--;
186  }
187  else {
188  adaptiveCounter++;
189  }
190  if (adaptiveCounter > sh_) {
191  c0_ = std::max(minInertia_, std::min(maxInertia_, c0_*0.5));
192  }
193  else if (adaptiveCounter < sl_) {
194  c0_ = std::max(minInertia_, std::min(maxInertia_, c0_*2.0));
195  }
196  }
197  else {
198  best_ = currBest;
199  started_ = true;
200  }
201  for (Size i = 0; i < M_; i++) {
202  (*V_)[i] *= c0_;
203  }
204  }

206  void KNeighbors::findSocialBest() {
207  for (Size i = 0; i < M_; i++) {
208  Real bestF = (*pBF_)[i];
209  Size bestX = 0;
210  //Search K_ neightbors upwards
211  Size upper = std::min(i + K_, M_);
212  //Search K_ neighbors downwards
213  Size lower = std::max(i, K_ + 1) - K_ - 1;
214  for (Size j = lower; j < upper; j++) {
215  if ((*pBF_)[j] < bestF) {
216  bestF = (*pBF_)[j];
217  bestX = j;
218  }
219  }
220  if (i + K_ >= M_) { //loop around if i+K >= M_
221  for (Size j = 0; j < i + K_ - M_; j++) {
222  if ((*pBF_)[j] < bestF) {
223  bestF = (*pBF_)[j];
224  bestX = j;
225  }
226  }
227  }
228  else if (i < K_) {//loop around from above
229  for (Size j = M_ - (K_ - i) - 1; j < M_; j++) {
230  if ((*pBF_)[j] < bestF) {
231  bestF = (*pBF_)[j];
232  bestX = j;
233  }
234  }
235  }
236  (*gBX_)[i] = (*pBX_)[bestX];
237  (*gBF_)[i] = bestF;
238  }
239  }

284  void ClubsTopology::findSocialBest() {
285  //Update iteration
286  iteration_++;
287  bool reset = false;
288  if (iteration_ == resetIteration_) {
289  iteration_ = 0;
290  reset = true;
291  }
292 
293  //Find best by current club
294  for (Size i = 0; i < totalClubs_; i++) {
295  Real bestByClub = QL_MAX_REAL;
296  Real worstByClub = -QL_MAX_REAL;
297  Size bestP = 0;
298  Size worstP = 0;
299  const std::vector<bool> &particlesSet = particles4clubs_[i];
300  for (Size j = 0; j < M_; j++) {
301  if (particlesSet[j]) {
302  if (bestByClub >(*pBF_)[j]) {
303  bestByClub = (*pBF_)[j];
304  bestP = j;
305  }
306  else if (worstByClub < (*pBF_)[j]) {
307  worstByClub = (*pBF_)[j];
308  worstP = j;
309  }
310  }
311  }
312  bestByClub_[i] = bestP;
313  worstByClub_[i] = worstP;
314  }
315 
316  //Update clubs && global best
317  for (Size i = 0; i < M_; i++) {
318  std::vector<bool> &clubSet = clubs4particles_[i];
319  bool best = true;
320  bool worst = true;
321  Size currentClubs = 0;
322  for (Size j = 0; j < totalClubs_; j++) {
323  if (clubSet[j]) {
324  //If still thought of the best, check if best in club j
325  if (best && i != bestByClub_[j]) best = false;
326  //If still thought of the worst, check if worst in club j
327  if (worst && i != worstByClub_[j]) worst = false;
328  //Update currentClubs
329  currentClubs++;
330  }
331  }
332  //Update clubs
333  if (best) {
334  //Leave random club
335  leaveRandomClub(i, currentClubs);
336  }
337  else if (worst) {
338  //Join random club
339  joinRandomClub(i, currentClubs);
340  }
341  else if (reset && currentClubs != defaultClubs_) {
342  //If membership != defaultClubs_, then leave or join accordingly
343  if (currentClubs < defaultClubs_) {
344  //Join random club
345  joinRandomClub(i, currentClubs);
346  }
347  else {
348  //Leave random club
349  leaveRandomClub(i, currentClubs);
350  }
351  }
352 
353  //Update global best
354  Real bestNeighborF = QL_MAX_REAL;
355  Size bestNeighborX = 0;
356  for (Size j = 0; j < totalClubs_; j++) {
357  if (clubSet[j] && bestNeighborF >(*pBF_)[bestByClub_[j]]) {
358  bestNeighborF = (*pBF_)[bestByClub_[j]];
359  bestNeighborX = j;
360  }
361  }
362  (*gBX_)[i] = (*pBX_)[bestNeighborX];
363  (*gBF_)[i] = bestNeighborF;
364  }
365  }
