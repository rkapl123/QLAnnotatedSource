78  /* Criticism:
79  This model is not as generic as it could be. In principle a default loss
80  model dependent on a single factor correlation parameter is the only
81  restriction on the base loss model(s) type. This class however is tied to a
82  LatentModel single factor. But there is no need for the
83  underlying model to be of a latent type. This link is due to the copula
84  initialization traits which have to be present for non trivial copula
85  policies initialization (e.g. Student-T base correl models)
86 
87  Maybe a possibility is to pass copiable instances of the model and relinking
88  to the correlation in two internal copies.
89  */
90  template <class BaseModel_T, class Corr2DInt_T>
91  class BaseCorrelationLossModel : public DefaultLossModel,
92  public virtual Observer {
93  private:
94  typedef typename BaseModel_T::copulaType::initTraits initTraits;
95  public:
96  BaseCorrelationLossModel(
97  const Handle<BaseCorrelationTermStructure<Corr2DInt_T> >& correlTS,
98  const std::vector<Real>& recoveries,
99  const initTraits& traits = initTraits()
100  )
101  : localCorrelationAttach_(ext::make_shared<SimpleQuote>(
102  0.)),
103  localCorrelationDetach_(ext::make_shared<SimpleQuote>(
104  0.)),
105  recoveries_(recoveries),
106  correlTS_(correlTS),
107  copulaTraits_(traits)
108  {
109  registerWith(correlTS);
110  registerWith(Settings::instance().evaluationDate());
111  }
112  private:
113  // react to base correl surface notifications (quotes or reference date)
114  void update() {
115  setupModels();
116  // tell basket to notify instruments, etc, we are invalid
117  if(!basket_.empty()) basket_->notifyObservers();
118  }

120  /* Update model caches after basket assignement. */
121  void resetModel() {
122  remainingNotional_ = basket_->remainingNotional();
123  attachRatio_ = basket_->remainingAttachmentAmount()
124  /remainingNotional_;
125  detachRatio_ = basket_->remainingDetachmentAmount()
126  /remainingNotional_;
127 
128  basketAttach_ = ext::make_shared<Basket>(basket_->refDate(),basket_->remainingNames(),
129  basket_->remainingNotionals(),
130  basket_->pool(),
131  0.0,
132  attachRatio_,
133  basket_->claim()
134  );
135  basketDetach_ = ext::make_shared<Basket>(basket_->refDate(),basket_->remainingNames(),
136  basket_->remainingNotionals(),
137  basket_->pool(),
138  0.0,
139  detachRatio_,
140  basket_->claim()
141  );
142  setupModels();
143  }

145  the model is intended for pricing rather than ptfolio risk management.

148  protected:

153  private:

155  mutable Real remainingNotional_;

174  // Remember ETL returns the EL on the live part of the basket.
175  template<class LM, class I>
176  Real BaseCorrelationLossModel<LM, I>::expectedTrancheLoss(
177  const Date& d) const
178  {
179  Real correlK1 = correlTS_->correlation(d, attachRatio_);
180  Real correlK2 = correlTS_->correlation(d, detachRatio_);
181 
182  /* reset correl and call base models which have the different baskets
183  associated.*/
184  localCorrelationAttach_->setValue(correlK1);
185  Real expLossK1 =
186  basketAttach_->expectedTrancheLoss(d);
187  localCorrelationDetach_->setValue(correlK2);
188  Real expLossK2 =
189  basketDetach_->expectedTrancheLoss(d);
190  return expLossK2 - expLossK1;
191  }
