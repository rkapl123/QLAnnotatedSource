40  bool IMM::isIMMdate(const Date& date, bool mainCycle) {
41  if (date.weekday()!=Wednesday)

44  Day d = date.dayOfMonth();

134  Date IMM::date(const std::string& immCode,
135  const Date& refDate) {
136  QL_REQUIRE(isIMMcode(immCode, false),
137  immCode << " is not a valid IMM code");
138 
139  Date referenceDate = (refDate != Date() ?
140  refDate :
141  Date(Settings::instance().evaluationDate()));
142 
143  std::string code = to_upper_copy(immCode);
144  std::string ms = code.substr(0,1);
145  QuantLib::Month m;
146  if (ms=="F") m = January;
147  else if (ms=="G") m = February;
148  else if (ms=="H") m = March;
149  else if (ms=="J") m = April;
150  else if (ms=="K") m = May;
151  else if (ms=="M") m = June;
152  else if (ms=="N") m = July;
153  else if (ms=="Q") m = August;
154  else if (ms=="U") m = September;
155  else if (ms=="V") m = October;
156  else if (ms=="X") m = November;
157  else if (ms=="Z") m = December;
158  else QL_FAIL("invalid IMM month letter");
159 
160 // Year y = boost::lexical_cast<Year>(); // lexical_cast causes compilation errors with x64
161 
162  Year y= io::to_integer(code.substr(1,1));
163  /* year<1900 are not valid QuantLib years: to avoid a run-time
164  exception few lines below we need to add 10 years right away */
165  if (y==0 && referenceDate.year()<=1909) y+=10;
166  Year referenceYear = (referenceDate.year() % 10);
167  y += referenceDate.year() - referenceYear;
168  Date result = IMM::nextDate(Date(1, m, y), false);
169  if (result<referenceDate)
170  return IMM::nextDate(Date(1, m, y+10), false);
171 
172  return result;
173  }

175  Date IMM::nextDate(const Date& date, bool mainCycle) {
176  Date refDate = (date == Date() ?
177  Date(Settings::instance().evaluationDate()) :
178  date);
179  Year y = refDate.year();
180  QuantLib::Month m = refDate.month();
181 
182  Size offset = mainCycle ? 3 : 1;
183  Size skipMonths = offset-(m%offset);
184  if (skipMonths != offset || refDate.dayOfMonth() > 21) {
185  skipMonths += Size(m);
186  if (skipMonths<=12) {
187  m = QuantLib::Month(skipMonths);
188  } else {
189  m = QuantLib::Month(skipMonths-12);
190  y += 1;
191  }
192  }
193 
194  Date result = Date::nthWeekday(3, Wednesday, m, y);
195  if (result<=refDate)
196  result = nextDate(Date(22, m, y), mainCycle);
197  return result;
198  }
