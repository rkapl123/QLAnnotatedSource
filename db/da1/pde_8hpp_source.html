32  class PdeSecondOrderParabolic {
33  public:
34  virtual ~PdeSecondOrderParabolic() {}
35  virtual Real diffusion(Time t, Real x) const = 0;
36  virtual Real drift(Time t, Real x) const = 0;
37  virtual Real discount(Time t, Real x) const = 0;
38  virtual void generateOperator(Time t,
39  const TransformedGrid &tg,
40  TridiagonalOperator &L) const {
41  for (Size i=1; i < tg.size() - 1; i++) {
42  Real sigma = diffusion(t, tg.grid(i));
43  Real nu = drift(t, tg.grid(i));
44  Real r = discount(t, tg.grid(i));
45  Real sigma2 = sigma * sigma;
46 
47  Real pd = -(sigma2/tg.dxm(i)-nu)/ tg.dx(i);
48  Real pu = -(sigma2/tg.dxp(i)+nu)/ tg.dx(i);
49  Real pm = sigma2/(tg.dxm(i) * tg.dxp(i))+r;
50  L.setMidRow(i, pd,pm,pu);
51  }
52  }
53  };
