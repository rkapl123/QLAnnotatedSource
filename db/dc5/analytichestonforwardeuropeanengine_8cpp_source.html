26  class P12Integrand {
27  private:
28  ext::shared_ptr<AnalyticHestonEngine>& engine_;
29  Real logK_, phiRightLimit_;
30  Time tenor_;
31  std::complex<Real> i_, adj_;
32  public:
33  P12Integrand(ext::shared_ptr<AnalyticHestonEngine>& engine,
34  Real logK,
35  Time tenor,
36  bool P1, // true for P1, false for P2
37  Real phiRightLimit = 100) : engine_(engine), logK_(logK),
38  phiRightLimit_(phiRightLimit), tenor_(tenor), i_(std::complex<Real>(0.0, 1.0)) {
39 
40  // Only difference between P1 and P2 integral is the additional term in the chF evaluation
41  if (P1) {
42  adj_ = std::complex<Real>(0.0, -1.0);
43  } else {
44  adj_ = std::complex<Real>(0.0, 0.0);
45  }
46  }
47 
48  // QL Gaussian Quadrature - map phi from [-1 to 1] to {0, phiRightLimit]
49  double operator()(double phi) const {
50  double phiDash = (0.5+1e-8+0.5*phi) * phiRightLimit_; // Map phi to full range
51  return 0.5*phiRightLimit_*std::real((std::exp(-phiDash*logK_*i_) / (phiDash*i_)) * engine_->chF(phiDash+adj_, tenor_));
52  }
53  };

56  class P12HatIntegrand {
57  private:
58  Time tenor_, resetTime_;
59  Handle<Quote>& s0_;
60  bool P1_;
61  Real logK_, phiRightLimit_, nuRightLimit_;
62  const AnalyticHestonForwardEuropeanEngine* const parent_;
63  GaussLegendreIntegration innerIntegrator_;
64  public:
65  P12HatIntegrand(Time tenor,
66  Time resetTime,
67  Handle<Quote>& s0,
68  Real logK,
69  bool P1, // true for P1, false for P2
70  const AnalyticHestonForwardEuropeanEngine* const parent,
71  Real phiRightLimit,
72  Real nuRightLimit) : tenor_(tenor), resetTime_(resetTime),
73  s0_(s0), P1_(P1), logK_(logK), phiRightLimit_(phiRightLimit),
74  nuRightLimit_(nuRightLimit), parent_(parent), innerIntegrator_(128) {}
75  double operator()(double nu) const {
76 
77  // Rescale nu to [-1, 1]
78  Real nuDash = nuRightLimit_ * (0.5 * nu + 0.5 + 1e-8);
79 
80  // Calculate the chF from var(t) to expiry
81  ext::shared_ptr<AnalyticHestonEngine> engine = parent_->forwardChF(s0_, nuDash);
82  P12Integrand pIntegrand(engine, logK_, tenor_, P1_, phiRightLimit_);
83  Real p1Integral = innerIntegrator_(pIntegrand);
84 
85  // Calculate the value of the propagator to nu
86  Real propagator = parent_->propagator(resetTime_, nuDash);
87 
88  // Take the product, and scale integral's value back up to [0, right_lim]
89  return propagator * (0.5 + p1Integral/M_PI);
90  }
91  };

94  AnalyticHestonForwardEuropeanEngine::AnalyticHestonForwardEuropeanEngine(
95  ext::shared_ptr<HestonProcess> process, Size integrationOrder)
96  : process_(std::move(process)), integrationOrder_(integrationOrder), outerIntegrator_(128) {
97 
98  v0_ = process_->v0();
99  rho_ = process_->rho();
100  kappa_ = process_->kappa();
101  theta_ = process_->theta();
102  sigma_ = process_->sigma();
103  s0_ = process_->s0();
104 
105  QL_REQUIRE(sigma_ > 0.1,
106  "Very low values (<~10%) for Heston Vol-of-Vol cause numerical issues" \
107  "in this implementation of the propagator function, try using" \
108  "MCForwardEuropeanHestonEngine Monte-Carlo engine instead");
109 
110  riskFreeRate_ = process_->riskFreeRate();
111  dividendYield_ = process_->dividendYield();
112 
113  // Some of the required constant intermediate variables can be calculated now
114  kappaHat_ = kappa_ - rho_ * sigma_;
115  thetaHat_ = kappa_ * theta_ / kappaHat_;
116  R_ = 4 * kappaHat_ * thetaHat_ / (sigma_ * sigma_);
117  }

120  void AnalyticHestonForwardEuropeanEngine::calculate() const {
121  // This is a european option pricer
122  QL_REQUIRE(this->arguments_.exercise->type() == Exercise::European,
123  "not an European option");
124 
125  // We only price plain vanillas
126  ext::shared_ptr<PlainVanillaPayoff> payoff =
127  ext::dynamic_pointer_cast<PlainVanillaPayoff>(this->arguments_.payoff);
128  QL_REQUIRE(payoff, "non plain vanilla payoff given");
129 
130  Time resetTime = this->process_->time(this->arguments_.resetDate);
131  Time expiryTime = this->process_->time(this->arguments_.exercise->lastDate());
132  Time tenor = expiryTime - resetTime;
133  Real moneyness = this->arguments_.moneyness;
134 
135  // K needs to be scaled to forward AT RESET TIME, not spot...
136  Real expiryDcf = riskFreeRate_->discount(expiryTime);
137  Real resetDcf = riskFreeRate_->discount(resetTime);
138  Real expiryDividendDiscount = dividendYield_->discount(expiryTime);
139  Real resetDividendDiscount = dividendYield_->discount(resetTime);
140  Real expiryRatio = expiryDcf / expiryDividendDiscount;
141  Real resetRatio = resetDcf / resetDividendDiscount;
142 
143  QL_REQUIRE(resetTime >= 0.0, "Reset Date cannot be in the past");
144  QL_REQUIRE(expiryTime >= 0.0, "Expiry Date cannot be in the past");
145 
146  // Use some heuristics to decide upon phiRightLimit and nuRightLimit
147  Real phiRightLimit = 100.0;
148  Real nuRightLimit = std::max(2.0, 10.0 * (1+std::max(0.0, rho_)) * sigma_ * std::sqrt(resetTime * std::max(v0_, theta_)));
149 
150  // do the 2D integral calculation. For very short times, we just fall back on the standard
151  // calculation, both for accuracy and because tStar==0 causes some numerical issues...
152  std::pair<Real, Real> P1HatP2Hat;
153  if (resetTime <= 1e-3) {
154  Handle<Quote> tempQuote(ext::shared_ptr<Quote>(new SimpleQuote(s0_->value())));
155  P1HatP2Hat = calculateP1P2(tenor, tempQuote, moneyness * s0_->value(), expiryRatio, phiRightLimit);
156  } else {
157  P1HatP2Hat = calculateP1P2Hat(tenor, resetTime, moneyness, expiryRatio/resetRatio, phiRightLimit, nuRightLimit);
158  }
159 
160  // Apply the payoff functions
161  Real value = 0.0;
162  Real F = s0_->value() / expiryRatio;
163  switch (payoff->optionType()){
164  case Option::Call:
165  value = expiryDcf * (F*P1HatP2Hat.first - moneyness*s0_->value()*P1HatP2Hat.second/resetRatio);
166  break;
167  case Option::Put:
168  value = expiryDcf * (moneyness*s0_->value()*(1-P1HatP2Hat.second)/resetRatio - F*(1-P1HatP2Hat.first));
169  break;
170  default:
171  QL_FAIL("unknown option type");
172  }
173 
174  results_.value = value;
175 
176  results_.additionalResults["dcf"] = expiryDcf;
177  results_.additionalResults["qf"] = expiryDividendDiscount;
178  results_.additionalResults["expiryRatio"] = expiryRatio;
179  results_.additionalResults["resetRatio"] = resetRatio;
180  results_.additionalResults["moneyness"] = moneyness;
181  results_.additionalResults["s0"] = s0_->value();
182  results_.additionalResults["fwd"] = F;
183  results_.additionalResults["resetTime"] = resetTime;
184  results_.additionalResults["expiryTime"] = expiryTime;
185  results_.additionalResults["P1Hat"] = P1HatP2Hat.first;
186  results_.additionalResults["P2Hat"] = P1HatP2Hat.second;
187  results_.additionalResults["phiRightLimit"] = phiRightLimit;
188  results_.additionalResults["nuRightLimit"] = nuRightLimit;
189  }

192  std::pair<Real, Real> AnalyticHestonForwardEuropeanEngine::calculateP1P2Hat(Time tenor,
193  Time resetTime,
194  Real moneyness,
195  Real ratio,
196  Real phiRightLimit,
197  Real nuRightLimit) const {
198 
199  Handle<Quote> unitQuote(ext::shared_ptr<Quote>(new SimpleQuote(1.0)));
200 
201  // Re-expressing moneyness in terms of the forward here (strike fixes to spot, but in
202  // our pricing calculation we need to compare it to the future at expiry)
203  Real logMoneyness = std::log(moneyness*ratio);
204 
205  P12HatIntegrand p1HatIntegrand(tenor, resetTime, unitQuote, logMoneyness, true, this, phiRightLimit, nuRightLimit);
206  P12HatIntegrand p2HatIntegrand(tenor, resetTime, unitQuote, logMoneyness, false, this, phiRightLimit, nuRightLimit);
207 
208  Real p1HatIntegral = 0.5 * nuRightLimit * outerIntegrator_(p1HatIntegrand);
209  Real p2HatIntegral = 0.5 * nuRightLimit * outerIntegrator_(p2HatIntegrand);
210 
211  std::pair<Real, Real> P1HatP2Hat(p1HatIntegral, p2HatIntegral);
212 
213  return P1HatP2Hat;
214  }

217  Real AnalyticHestonForwardEuropeanEngine::propagator(Time resetTime,
218  Real varReset) const {
219  Real B, Lambda, term1, term2, term3;
220 
221  B = 4 * kappaHat_ / (sigma_ * sigma_ * (1 - std::exp(-kappaHat_ * resetTime)));
222  Lambda = B * std::exp(-kappaHat_ * resetTime) * v0_;
223 
224  // Now construct equation (18) from the paper term-by-term
225  term1 = std::exp(-0.5*(B * varReset + Lambda)) * B / 2;
226  term2 = std::pow(B * varReset / Lambda, 0.5*(R_/2 - 1));
227  term3 = modifiedBesselFunction_i(R_/2 - 1,std::sqrt(Lambda * B * varReset));
228 
229  return term1 * term2 * term3;
230  }

252  std::pair<Real, Real> AnalyticHestonForwardEuropeanEngine::calculateP1P2(Time tenor,
253  Handle<Quote>& St,
254  Real K,
255  Real ratio,
256  Real phiRightLimit) const {
257 
258  ext::shared_ptr<AnalyticHestonEngine> engine = forwardChF(St, v0_);
259  Real logK = std::log(K*ratio/St->value());
260 
261  // Integrate the CF and the complex integrand over positive phi
262  GaussLegendreIntegration integrator = GaussLegendreIntegration(128);
263  P12Integrand p1Integrand(engine, logK, tenor, true, phiRightLimit);
264  P12Integrand p2Integrand(engine, logK, tenor, false, phiRightLimit);
265 
266  Real p1Integral = integrator(p1Integrand);
267  Real p2Integral = integrator(p2Integrand);
268 
269  std::pair<Real, Real> P1P2(0.5 + p1Integral/M_PI, 0.5 + p2Integral/M_PI);
270 
271  return P1P2;
272  }
