99  inline Real PoissonDistribution::operator()(BigNatural k) const {
100  if (mu_==0.0) {
101  if (k==0) return 1.0;
102  else return 0.0;
103  }
104  Real logFactorial = Factorial::ln(k);
105  return std::exp(k*std::log(mu_) - logFactorial - mu_);
106  }

114  inline Real InverseCumulativePoisson::operator()(Real x) const {
115  QL_REQUIRE(x >= 0.0 && x <= 1.0,
116  "Inverse cumulative Poisson distribution is "
117  "only defined on the interval [0,1]");
118 
119  if (x == 1.0)
120  return QL_MAX_REAL;
121 
122  Real sum = 0.0;
123  BigNatural index = 0;
124  while (x > sum) {
125  sum += calcSummand(index);
126  index++;
127  }
128 
129  return Real(index-1);
130  }

132  inline Real InverseCumulativePoisson::calcSummand(BigNatural index) const {
133  return std::exp(-lambda_) * std::pow(lambda_, Integer(index)) /
134  Factorial::get(index);
135  }
