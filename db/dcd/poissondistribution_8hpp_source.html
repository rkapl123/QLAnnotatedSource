107  inline Real PoissonDistribution::operator()(BigNatural k) const {
108  if (mu_==0.0) {
109  if (k==0) return 1.0;
110  else return 0.0;
111  }
112  Real logFactorial = Factorial::ln(k);
113  return std::exp(k*std::log(mu_) - logFactorial - mu_);
114  }

122  inline Real InverseCumulativePoisson::operator()(Real x) const {
123  QL_REQUIRE(x >= 0.0 && x <= 1.0,
124  "Inverse cumulative Poisson distribution is "
125  "only defined on the interval [0,1]");
126 
127  if (x == 1.0)
128  return QL_MAX_REAL;
129 
130  Real sum = 0.0;
131  BigNatural index = 0;
132  while (x > sum) {
133  sum += calcSummand(index);
134  index++;
135  }
136 
137  return Real(index-1);
138  }
139 
140  inline Real InverseCumulativePoisson::calcSummand(BigNatural index) const {
141  return std::exp(-lambda_) * std::pow(lambda_, Integer(index)) /
142  Factorial::get(index);
143  }
