157  struct CachedSwapKey {
158  const boost::shared_ptr<SwapIndex> index;
159  const Date fixing;
160  const Period tenor;
161  bool operator==(const CachedSwapKey &o) const {
162  return index->name() == o.index->name() && fixing == o.fixing &&
163  tenor == o.tenor;
164  }
165  };
166 
167  struct CachedSwapKeyHasher
168  : std::unary_function<CachedSwapKey, std::size_t> {
169  std::size_t operator()(CachedSwapKey const &x) const {
170  std::size_t seed = 0;
171  boost::hash_combine(seed, x.index->name());
172  boost::hash_combine(seed, x.fixing.serialNumber());
173  boost::hash_combine(seed, x.tenor.length());
174  boost::hash_combine(seed, x.tenor.units());
175  return seed;
176  }
177  };

211  // retrieve underlying swap from cache if possible, otherwise
212  // create it and store it in the cache
213  boost::shared_ptr<VanillaSwap>
214  underlyingSwap(const boost::shared_ptr<SwapIndex> &index,
215  const Date &expiry, const Period &tenor) const {
216 
217  CachedSwapKey k = {index, expiry, tenor};
218  CacheType::iterator i = swapCache_.find(k);
219  if (i == swapCache_.end()) {
220  boost::shared_ptr<VanillaSwap> underlying =
221  index->clone(tenor)->underlyingSwap(expiry);
222  swapCache_.insert(std::make_pair(k, underlying));
223  return underlying;
224  }
225  return i->second;
226  }
