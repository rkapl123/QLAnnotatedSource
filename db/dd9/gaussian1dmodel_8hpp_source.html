147 // It is of great importance for performance reasons to cache underlying
148 // swaps generated from indexes. In addition the indexes may only be given
149 // as templates for the conventions with the tenor replaced by the actual
150 // one later on.
151
152 struct CachedSwapKey {
153 const ext::shared_ptr<SwapIndex> index;
154 const Date fixing;
155 const Period tenor;
156 bool operator==(const CachedSwapKey &o) const {
157 return index->name() == o.index->name() && fixing == o.fixing &&
158 tenor == o.tenor;
159 }
160 };
161
162 struct CachedSwapKeyHasher {
163 std::size_t operator()(CachedSwapKey const &x) const {
164 std::size_t seed = 0;
165 boost::hash_combine(seed, x.index->name());
166 boost::hash_combine(seed, x.fixing.serialNumber());
167 boost::hash_combine(seed, x.tenor.length());
168 boost::hash_combine(seed, x.tenor.units());
169 return seed;
170 }
171 };

198 // retrieve underlying swap from cache if possible, otherwise
199 // create it and store it in the cache
200 ext::shared_ptr<VanillaSwap>
201 underlyingSwap(const ext::shared_ptr<SwapIndex> &index,
202 const Date &expiry, const Period &tenor) const {
203
204 CachedSwapKey k = {index, expiry, tenor};
205 auto i = swapCache_.find(k);
206 if (i == swapCache_.end()) {
207 ext::shared_ptr<VanillaSwap> underlying =
208 index->clone(tenor)->underlyingSwap(expiry);
209 swapCache_.insert(std::make_pair(k, underlying));
210 return underlying;
211 }
212 return i->second;
213 }
