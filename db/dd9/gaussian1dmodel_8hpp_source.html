161  struct CachedSwapKey {
162  const ext::shared_ptr<SwapIndex> index;
163  const Date fixing;
164  const Period tenor;
165  bool operator==(const CachedSwapKey &o) const {
166  return index->name() == o.index->name() && fixing == o.fixing &&
167  tenor == o.tenor;
168  }
169  };
170 
171  struct CachedSwapKeyHasher {
172  std::size_t operator()(CachedSwapKey const &x) const {
173  std::size_t seed = 0;
174  boost::hash_combine(seed, x.index->name());
175  boost::hash_combine(seed, x.fixing.serialNumber());
176  boost::hash_combine(seed, x.tenor.length());
177  boost::hash_combine(seed, x.tenor.units());
178  return seed;
179  }
180  };

212  // retrieve underlying swap from cache if possible, otherwise
213  // create it and store it in the cache
214  ext::shared_ptr<VanillaSwap>
215  underlyingSwap(const ext::shared_ptr<SwapIndex> &index,
216  const Date &expiry, const Period &tenor) const {
217 
218  CachedSwapKey k = {index, expiry, tenor};
219  CacheType::iterator i = swapCache_.find(k);
220  if (i == swapCache_.end()) {
221  ext::shared_ptr<VanillaSwap> underlying =
222  index->clone(tenor)->underlyingSwap(expiry);
223  swapCache_.insert(std::make_pair(k, underlying));
224  return underlying;
225  }
226  return i->second;
227  }
