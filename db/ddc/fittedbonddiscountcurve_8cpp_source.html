96  void FittedBondDiscountCurve::performCalculations() const {
97 
98  QL_REQUIRE(!bondHelpers_.empty(), "no bondHelpers given");
99 
100  maxDate_ = Date::minDate();
101  Date refDate = referenceDate();
102 
103  // double check bond quotes still valid and/or instruments not expired
104  for (Size i=0; i<bondHelpers_.size(); ++i) {
105  ext::shared_ptr<Bond> bond = bondHelpers_[i]->bond();
106  QL_REQUIRE(bondHelpers_[i]->quote()->isValid(),
107  io::ordinal(i+1) << " bond (maturity: " <<
108  bond->maturityDate() << ") has an invalid price quote");
109  Date bondSettlement = bond->settlementDate();
110  QL_REQUIRE(bondSettlement>=refDate,
111  io::ordinal(i+1) << " bond settlemente date (" <<
112  bondSettlement << ") before curve reference date (" <<
113  refDate << ")");
114  QL_REQUIRE(BondFunctions::isTradable(*bond, bondSettlement),
115  io::ordinal(i+1) << " bond non tradable at " <<
116  bondSettlement << " settlement date (maturity"
117  " being " << bond->maturityDate() << ")");
118  maxDate_ = std::max(maxDate_, bondHelpers_[i]->pillarDate());
119  bondHelpers_[i]->setTermStructure(
120  const_cast<FittedBondDiscountCurve*>(this));
121  }
122  fittingMethod_->init();
123  fittingMethod_->calculate();
124  }

138  void FittedBondDiscountCurve::FittingMethod::init() {
139  // yield conventions
140  DayCounter yieldDC = curve_->dayCounter();
141  Compounding yieldComp = Compounded;
142  Frequency yieldFreq = Annual;
143 
144  Size n = curve_->bondHelpers_.size();
145  costFunction_ = ext::make_shared<FittingCost>(this);
146 
147  for (Size i=0; i<curve_->bondHelpers_.size(); ++i) {
148  curve_->bondHelpers_[i]->setTermStructure(curve_);
149  }
150 
151  if (calculateWeights_) {
152  if (weights_.empty())
153  weights_ = Array(n);
154 
155  Real squaredSum = 0.0;
156  for (Size i=0; i<curve_->bondHelpers_.size(); ++i) {
157  ext::shared_ptr<Bond> bond = curve_->bondHelpers_[i]->bond();
158 
159  Real cleanPrice = curve_->bondHelpers_[i]->quote()->value();
160 
161  Date bondSettlement = bond->settlementDate();
162  Rate ytm = BondFunctions::yield(*bond, cleanPrice,
163  yieldDC, yieldComp, yieldFreq,
164  bondSettlement);
165 
166  Time dur = BondFunctions::duration(*bond, ytm,
167  yieldDC, yieldComp, yieldFreq,
168  Duration::Modified,
169  bondSettlement);
170  weights_[i] = 1.0/dur;
171  squaredSum += weights_[i]*weights_[i];
172  }
173  weights_ /= std::sqrt(squaredSum);
174  }
175 
176  QL_REQUIRE(weights_.size() == n,
177  "Given weights do not cover all boostrapping helpers");
178 
179  if (!l2_.empty()) {
180  QL_REQUIRE(l2_.size() == size(),
181  "Given penalty factors do not cover all parameters");
182  }
183  }

185  void FittedBondDiscountCurve::FittingMethod::calculate() {
186 
187  FittingCost& costFunction = *costFunction_;
188  Constraint constraint = NoConstraint();
189 
190  // start with the guess solution, if it exists
191  Array x(size(), 0.0);
192  if (!curve_->guessSolution_.empty()) {
193  x = curve_->guessSolution_;
194  }
195 
196  if (curve_->maxEvaluations_ == 0)
197  {
198  // Don't calculate, simply use the given parameters to provide a fitted curve.
199  // This turns the fittedbonddiscountcurve into an evaluator of the parametric
200  // curve, for example allowing to use the parameters for a credit spread curve
201  // calculated with bonds in one currency to be coupled to a discount curve in
202  // another currency.
203 
204  QL_REQUIRE(!curve_->guessSolution_.empty(), "no guess provided");
205 
206  solution_ = curve_->guessSolution_;
207 
208  numberOfIterations_ = 0;
209  costValue_ = costFunction.value(solution_);
210 
211  return;
212  }
213 
214  //workaround for backwards compatibility
215  ext::shared_ptr<OptimizationMethod> optimization = optimizationMethod_;
216  if(!optimization){
217  optimization = ext::make_shared<Simplex>(curve_->simplexLambda_);
218  }
219  Problem problem(costFunction, constraint, x);
220 
221  Real rootEpsilon = curve_->accuracy_;
222  Real functionEpsilon = curve_->accuracy_;
223  Real gradientNormEpsilon = curve_->accuracy_;
224 
225  EndCriteria endCriteria(curve_->maxEvaluations_,
226  curve_->maxStationaryStateIterations_,
227  rootEpsilon,
228  functionEpsilon,
229  gradientNormEpsilon);
230 
231  optimization->minimize(problem,endCriteria);
232  solution_ = problem.currentValue();
233 
234  numberOfIterations_ = problem.functionEvaluation();
235  costValue_ = problem.functionValue();
236 
237  // save the results as the guess solution, in case of recalculation
238  curve_->guessSolution_ = solution_;
239  }

257  Disposable<Array>
258  FittedBondDiscountCurve::FittingMethod::FittingCost::values(
259  const Array &x) const {
260  Size n = fittingMethod_->curve_->bondHelpers_.size();
261  Size N = fittingMethod_->l2_.size();
262 
263  // set solution so that fittingMethod_->curve_ represents the current trial
264  // the final solution will be set in FittingMethod::calculate() later on
265  fittingMethod_->solution_ = x;
266 
267  Array values(n + N);
268  for (Size i=0; i<n; ++i) {
269  ext::shared_ptr<BondHelper> helper = fittingMethod_->curve_->bondHelpers_[i];
270  Real error = helper->impliedQuote() - helper->quote()->value();
271  Real weightedError = fittingMethod_->weights_[i] * error;
272  values[i] = weightedError * weightedError;
273  }
274 
275  if (N != 0) {
276  for (Size i = 0; i < N; ++i) {
277  Real error = x[i] - fittingMethod_->curve_->guessSolution_[i];
278  values[i + n] = fittingMethod_->l2_[i] * error * error;
279  }
280  }
281  return values;
282  }
