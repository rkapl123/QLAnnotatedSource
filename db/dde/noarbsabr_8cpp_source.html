45 NoArbSabrModel::NoArbSabrModel(const Real expiryTime, const Real forward,
46  const Real alpha, const Real beta, const Real nu,
47  const Real rho)
48  : expiryTime_(expiryTime), externalForward_(forward), alpha_(alpha),
49  beta_(beta), nu_(nu), rho_(rho), forward_(forward),
50  numericalForward_(forward) {
51 
52  QL_REQUIRE(expiryTime > 0.0 && expiryTime <= detail::NoArbSabrModel::expiryTime_max,
53  "expiryTime (" << expiryTime << ") out of bounds");
54  QL_REQUIRE(forward > 0.0, "forward (" << forward << ") must be positive");
55  QL_REQUIRE(beta >= detail::NoArbSabrModel::beta_min && beta <= detail::NoArbSabrModel::beta_max,
56  "beta (" << beta << ") out of bounds");
57  Real sigmaI = alpha * std::pow(forward, beta - 1.0);
58  QL_REQUIRE(sigmaI >= detail::NoArbSabrModel::sigmaI_min &&
59  sigmaI <= detail::NoArbSabrModel::sigmaI_max,
60  "sigmaI = alpha*forward^(beta-1.0) ("
61  << sigmaI << ") out of bounds, alpha=" << alpha
62  << " beta=" << beta << " forward=" << forward);
63  QL_REQUIRE(nu >= detail::NoArbSabrModel::nu_min && nu <= detail::NoArbSabrModel::nu_max,
64  "nu (" << nu << ") out of bounds");
65  QL_REQUIRE(rho >= detail::NoArbSabrModel::rho_min && rho <= detail::NoArbSabrModel::rho_max,
66  "rho (" << rho << ") out of bounds");
67 
68  // determine a region sufficient for integration in the normal case
69 
70  fmin_ = fmax_ = forward_;
71  for (Real tmp = p(fmax_);
72  tmp > std::max(detail::NoArbSabrModel::i_accuracy / std::max(1.0, fmax_ - fmin_),
73  detail::NoArbSabrModel::density_threshold);
74  tmp = p(fmax_)) {
75  fmax_ *= 2.0;
76  }
77  for (Real tmp = p(fmin_);
78  tmp > std::max(detail::NoArbSabrModel::i_accuracy / std::max(1.0, fmax_ - fmin_),
79  detail::NoArbSabrModel::density_threshold);
80  tmp = p(fmin_)) {
81  fmin_ *= 0.5;
82  }
83  fmin_ = std::max(detail::NoArbSabrModel::strike_min, fmin_);
84 
85  QL_REQUIRE(fmax_ > fmin_, "could not find a reasonable integration domain");
86 
87  integrator_ =
88  boost::shared_ptr<GaussLobattoIntegral>(new GaussLobattoIntegral(
89  detail::NoArbSabrModel::i_max_iterations, detail::NoArbSabrModel::i_accuracy));
90 
91  detail::D0Interpolator d0(forward_, expiryTime_, alpha_, beta_, nu_, rho_);
92  absProb_ = d0();
93 
94  try {
95  Brent b;
96  Real start = std::sqrt(externalForward_ - detail::NoArbSabrModel::strike_min);
97  Real tmp =
98  b.solve(boost::lambda::bind(&NoArbSabrModel::forwardError, this,
99  boost::lambda::_1),
100  detail::NoArbSabrModel::forward_accuracy, start,
101  std::min(detail::NoArbSabrModel::forward_search_step, start / 2.0));
102  forward_ = tmp * tmp + detail::NoArbSabrModel::strike_min;
103  } catch (Error&) {
104  // fall back to unadjusted forward
105  forward_ = externalForward_;
106  }
107 
108  Real d = forwardError(std::sqrt(forward_ - detail::NoArbSabrModel::strike_min));
109  numericalForward_ = d + externalForward_;
110 }

112 Real NoArbSabrModel::optionPrice(const Real strike) const {
113  if (p(std::max(forward_, strike)) < detail::NoArbSabrModel::density_threshold)
114  return 0.0;
115  return (1.0 - absProb_) *
116  ((*integrator_)(integrand(this, strike),
117  strike, std::max(fmax_, 2.0 * strike)) /
118  numericalIntegralOverP_);
119 }

121 Real NoArbSabrModel::digitalOptionPrice(const Real strike) const {
122  if (strike < QL_MIN_POSITIVE_REAL)
123  return 1.0;

132 Real NoArbSabrModel::forwardError(const Real forward) const {
133  forward_ = forward * forward + detail::NoArbSabrModel::strike_min;
134  numericalIntegralOverP_ = (*integrator_)(
135  std::bind1st(std::mem_fun(&NoArbSabrModel::p), this),
136  fmin_, fmax_);
137  return optionPrice(0.0) - externalForward_;
138 }

140 Real NoArbSabrModel::p(const Real f) const {
141 
142  if (f < detail::NoArbSabrModel::density_lower_bound ||
143  forward_ < detail::NoArbSabrModel::density_lower_bound)
144  return 0.0;
145 
146  Real fOmB = std::pow(f, 1.0 - beta_);
147  Real FOmB = std::pow(forward_, 1.0 - beta_);
148 
149  Real zf = fOmB / (alpha_ * (1.0 - beta_));
150  Real zF = FOmB / (alpha_ * (1.0 - beta_));
151  Real z = zF - zf;
152 
153  // Real JzF = std::sqrt(1.0 - 2.0 * rho_ * nu_ * zF + nu_ * nu_ * zF * zF);
154  Real Jmzf = std::sqrt(1.0 + 2.0 * rho_ * nu_ * zf + nu_ * nu_ * zf * zf);
155  Real Jz = std::sqrt(1.0 - 2.0 * rho_ * nu_ * z + nu_ * nu_ * z * z);
156 
157  Real xz = std::log((Jz - rho_ + nu_ * z) / (1.0 - rho_)) / nu_;
158  Real Bp_B = beta_ / FOmB;
159  // Real Bpp_B = beta_ * (2.0 * beta_ - 1.0) / (FOmB * FOmB);
160  Real kappa1 = 0.125 * nu_ * nu_ * (2.0 - 3.0 * rho_ * rho_) -
161  0.25 * rho_ * nu_ * alpha_ * Bp_B;
162  // Real kappa2 = alpha_ * alpha_ * (0.25 * Bpp_B - 0.375 * Bp_B * Bp_B);
163  Real gamma = 1.0 / (2.0 * (1.0 - beta_));
164  Real sqrtOmR = std::sqrt(1.0 - rho_ * rho_);
165  Real h = 0.5 * beta_ * rho_ / ((1.0 - beta_) * Jmzf * Jmzf) *
166  (nu_ * zf * std::log(zf * Jz / zF) +
167  (1 + rho_ * nu_ * zf) / sqrtOmR *
168  (std::atan((nu_ * z - rho_) / sqrtOmR) +
169  std::atan(rho_ / sqrtOmR)));
170 
171  Real res =
172  std::pow(Jz, -1.5) / (alpha_ * std::pow(f, beta_) * expiryTime_) *
173  std::pow(zf, 1.0 - gamma) * std::pow(zF, gamma) *
174  std::exp(-(xz * xz) / (2.0 * expiryTime_) +
175  (h + kappa1 * expiryTime_)) *
176  modifiedBesselFunction_i_exponentiallyWeighted(gamma,
177  zF * zf / expiryTime_);
178  return res;
179 }

215 Real D0Interpolator::operator()() const {
216 
217  // we do not need to check the indices here, because this is already
218  // done in the NoArbSabr constructor
219 
220  Size tauInd = std::upper_bound(tauG_.begin(), tauG_.end(), expiryTime_) -
221  tauG_.begin();
222  if (tauInd == tauG_.size())
223  --tauInd; // tau at upper bound
224  Real expiryTimeTmp = expiryTime_;
225  if (tauInd == 0) {
226  ++tauInd;
227  expiryTimeTmp = tauG_.front();
228  }
229  Real tauL = (expiryTimeTmp - tauG_[tauInd - 1]) /
230  (tauG_[tauInd] - tauG_[tauInd - 1]);
231 
232  int sigmaIInd =
233  sigmaIG_.size() -
234  (std::upper_bound(sigmaIG_.rbegin(), sigmaIG_.rend(), sigmaI_) -
235  sigmaIG_.rbegin());
236  if (sigmaIInd == 0)
237  ++sigmaIInd; // sigmaI at upper bound
238  Real sigmaIL = (sigmaI_ - sigmaIG_[sigmaIInd - 1]) /
239  (sigmaIG_[sigmaIInd] - sigmaIG_[sigmaIInd - 1]);
240 
241  int rhoInd =
242  rhoG_.size() -
243  (std::upper_bound(rhoG_.rbegin(), rhoG_.rend(), rho_) - rhoG_.rbegin());
244  if (rhoInd == 0) {
245  rhoInd++;
246  }
247  if (rhoInd == static_cast<int>(rhoG_.size())) {
248  rhoInd--;
249  }
250  Real rhoL =
251  (rho_ - rhoG_[rhoInd - 1]) / (rhoG_[rhoInd] - rhoG_[rhoInd - 1]);
252 
253  // for nu = 0 we know phi = 0.5*z_F^2
254  Size nuInd = std::upper_bound(nuG_.begin(), nuG_.end(), nu_) - nuG_.begin();
255  if (nuInd == nuG_.size())
256  --nuInd; // nu at upper bound
257  Real tmpNuG = nuInd > 0 ? nuG_[nuInd - 1] : 0.0;
258  Real nuL = (nu_ - tmpNuG) / (nuG_[nuInd] - tmpNuG);
259 
260  // for beta = 1 we know phi = 0.0
261  Size betaInd =
262  std::upper_bound(betaG_.begin(), betaG_.end(), beta_) - betaG_.begin();
263  Real tmpBetaG;
264  if (betaInd == betaG_.size())
265  tmpBetaG = 1.0;
266  else
267  tmpBetaG = betaG_[betaInd];
268  Real betaL =
269  (beta_ - betaG_[betaInd - 1]) / (tmpBetaG - betaG_[betaInd - 1]);
270 
271  Real phiRes = 0.0;
272  for (int iTau = -1; iTau <= 0; ++iTau) {
273  for (int iSigma = -1; iSigma <= 0; ++iSigma) {
274  for (int iRho = -1; iRho <= 0; ++iRho) {
275  for (int iNu = -1; iNu <= 0; ++iNu) {
276  for (int iBeta = -1; iBeta <= 0; ++iBeta) {
277  Real phiTmp;
278  if (iNu == -1 && nuInd == 0) {
279  phiTmp =
280  0.5 /
281  (sigmaI_ * sigmaI_ * (1.0 - beta_) *
282  (1.0 - beta_)); // this is 0.5*z_F^2, see above
283  } else {
284  if (iBeta == 0 && betaInd == betaG_.size()) {
285  phiTmp =
286  phi(detail::NoArbSabrModel::tiny_prob);
287  } else {
288  int ind = (tauInd + iTau +
289  (sigmaIInd + iSigma +
290  (rhoInd + iRho +
291  (nuInd + iNu + ((betaInd + iBeta) *
292  nuG_.size())) *
293  rhoG_.size()) *
294  sigmaIG_.size()) *
295  tauG_.size());
296  QL_REQUIRE(ind >= 0 && ind < 1209600,
297  "absorption matrix index ("
298  << ind << ") invalid");
299  phiTmp = phi((Real)sabrabsprob[ind] /
300  detail::NoArbSabrModel::nsim);
301  }
302  }
303  phiRes += phiTmp * (iTau == -1 ? (1.0 - tauL) : tauL) *
304  (iSigma == -1 ? (1.0 - sigmaIL) : sigmaIL) *
305  (iRho == -1 ? (1.0 - rhoL) : rhoL) *
306  (iNu == -1 ? (1.0 - nuL) : nuL) *
307  (iBeta == -1 ? (1.0 - betaL) : betaL);
308  }
309  }
310  }
311  }
312  }
313  return d0(phiRes);
314 }
