37  Uniform1dMesher(Real start, Real end, Size size)
38  : Fdm1dMesher(size) {
39  QL_REQUIRE(end > start, "end must be large than start");
40 
41  const Real dx = (end-start)/(size-1);
42 
43  for (Size i=0; i < size-1; ++i) {
44  locations_[i] = start + i*dx;
45  dplus_[i] = dminus_[i+1] = dx;
46  }
47 
48  locations_.back() = end;
49  dplus_.back() = dminus_.front() = Null<Real>();
50  }
51  };
52 }
