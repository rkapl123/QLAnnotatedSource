230  inline std::vector<Time> DiscretizedOption::mandatoryTimes() const {
231  std::vector<Time> times = underlying_->mandatoryTimes();
232  // discard negative times...
233  std::vector<Time>::const_iterator i =
234  std::find_if(exerciseTimes_.begin(),exerciseTimes_.end(),
235  greater_or_equal_to<Time>(0.0));
236  // and add the positive ones
237  times.insert(times.end(), i, exerciseTimes_.end());
238  return times;
