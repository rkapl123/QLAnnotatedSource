29  /* Generally inflation indices are available with a lag of 1month
30  and then observed with a lag of 2-3 months depending whether
31  they use an interpolated fixing or not. Here, we make the
32  swap use the interpolation of the index to avoid incompatibilities.
33  */
34  ZeroCouponInflationSwap::ZeroCouponInflationSwap(
35  Type type,
36  Real nominal,
37  const Date& startDate, // start date of contract (only)
38  const Date& maturity, // this is pre-adjustment!
39  Calendar fixCalendar,
40  BusinessDayConvention fixConvention,
41  DayCounter dayCounter,
42  Rate fixedRate,
43  const ext::shared_ptr<ZeroInflationIndex>& infIndex,
44  const Period& observationLag,
45  bool adjustInfObsDates,
46  Calendar infCalendar,
47  BusinessDayConvention infConvention)
48  : Swap(2), type_(type), nominal_(nominal), startDate_(startDate), maturityDate_(maturity),
49  fixCalendar_(std::move(fixCalendar)), fixConvention_(fixConvention), fixedRate_(fixedRate),
50  infIndex_(infIndex), observationLag_(observationLag), adjustInfObsDates_(adjustInfObsDates),
51  infCalendar_(std::move(infCalendar)), infConvention_(infConvention),
52  dayCounter_(std::move(dayCounter)) {
53  // first check compatibility of index and swap definitions
54  if (infIndex_->interpolated()) {
55  Period pShift(infIndex_->frequency());
56  QL_REQUIRE(observationLag_ - pShift > infIndex_->availabilityLag(),
57  "inconsistency between swap observation of index " << observationLag_ <<
58  " index availability " << infIndex_->availabilityLag() <<
59  " interpolated index period " << pShift <<
60  " and index availability " << infIndex_->availabilityLag() <<
61  " need (obsLag-index period) > availLag");
62  } else {
63  QL_REQUIRE(infIndex_->availabilityLag() < observationLag_,
64  "index tries to observe inflation fixings that do not yet exist: "
65  << " availability lag " << infIndex_->availabilityLag()
66  << " versus obs lag = " << observationLag_);
67  }
68 
69  if (infCalendar_==Calendar()) infCalendar_ = fixCalendar_;
70  if (infConvention_==BusinessDayConvention()) infConvention_ = fixConvention_;
71 
72  if (adjustInfObsDates_) {
73  baseDate_ = infCalendar_.adjust(startDate - observationLag_, infConvention_);
74  obsDate_ = infCalendar_.adjust(maturity - observationLag_, infConvention_);
75  } else {
76  baseDate_ = startDate - observationLag_;
77  obsDate_ = maturity - observationLag_;
78  }
79 
80  Date infPayDate = infCalendar_.adjust(maturity, infConvention_);
81  Date fixedPayDate = fixCalendar_.adjust(maturity, fixConvention_);
82 
83  // At this point the index may not be able to forecast
84  // i.e. do not want to force the existence of an inflation
85  // term structure before allowing users to create instruments.
86  Real T = inflationYearFraction(infIndex_->frequency(), infIndex_->interpolated(),
87  dayCounter_, baseDate_, obsDate_);
88  // N.B. the -1.0 is because swaps only exchange growth, not notionals as well
89  Real fixedAmount = nominal * ( std::pow(1.0 + fixedRate, T) - 1.0 );
90 
91  legs_[0].push_back(ext::shared_ptr<CashFlow>(
92  new SimpleCashFlow(fixedAmount, fixedPayDate)));
93  bool growthOnly = true;
94  legs_[1].push_back(ext::shared_ptr<CashFlow>(
95  new IndexedCashFlow(nominal,infIndex,baseDate_,obsDate_,infPayDate,growthOnly)));
96 
97  for (Size j=0; j<2; ++j) {
98  for (auto& i : legs_[j])
99  registerWith(i);
100  }
101 
102  switch (type_) {
103  case Payer:
104  payer_[0] = +1.0;
105  payer_[1] = -1.0;
106  break;
107  case Receiver:
108  payer_[0] = -1.0;
109  payer_[1] = +1.0;
110  break;
111  default:
112  QL_FAIL("Unknown zero-inflation-swap type");
113  }
114  }

132  Real ZeroCouponInflationSwap::fairRate() const {
133  // What does this mean before or after trade date?
134  // Always means that NPV is zero for _this_ instrument
135  // if it was created with _this_ rate
136  // _knowing_ the time from base to obs (etc).
137 
138  ext::shared_ptr<IndexedCashFlow> icf =
139  ext::dynamic_pointer_cast<IndexedCashFlow>(legs_[1].at(0));
140  QL_REQUIRE(icf,"failed to downcast to IndexedCashFlow in ::fairRate()");
141 
142  // +1 because the IndexedCashFlow has growthOnly=true
143  Real growth = icf->amount() / icf->notional() + 1.0;
144  Real T = inflationYearFraction(infIndex_->frequency(),
145  infIndex_->interpolated(),
146  dayCounter_, baseDate_, obsDate_);
147 
148  return std::pow(growth,1.0/T) - 1.0;
