28  /* Generally inflation indices are available with a lag of 1month
29  and then observed with a lag of 2-3 months depending whether
30  they use an interpolated fixing or not. Here, we make the
31  swap use the interpolation of the index to avoid incompatibilities.
32  */
33  ZeroCouponInflationSwap::ZeroCouponInflationSwap(
34  Type type,
35  Real nominal,
36  const Date& startDate, // start date of contract (only)
37  const Date& maturity, // this is pre-adjustment!
38  const Calendar& fixCalendar,
39  BusinessDayConvention fixConvention,
40  const DayCounter& dayCounter,
41  Rate fixedRate,
42  const boost::shared_ptr<ZeroInflationIndex> &infIndex,
43  const Period& observationLag,
44  bool adjustInfObsDates,
45  Calendar infCalendar,
46  BusinessDayConvention infConvention)
47  : Swap(2), type_(type), nominal_(nominal),
48  startDate_(startDate), maturityDate_(maturity),
49  fixCalendar_(fixCalendar), fixConvention_(fixConvention),
50  fixedRate_(fixedRate), infIndex_(infIndex),
51  observationLag_(observationLag), adjustInfObsDates_(adjustInfObsDates),
52  infCalendar_(infCalendar), infConvention_(infConvention),
53  dayCounter_(dayCounter) {
54  // first check compatibility of index and swap definitions
55  if (infIndex_->interpolated()) {
56  Period pShift(infIndex_->frequency());
57  QL_REQUIRE(observationLag_ - pShift > infIndex_->availabilityLag(),
58  "inconsistency between swap observation of index " << observationLag_ <<
59  " index availability " << infIndex_->availabilityLag() <<
60  " interpolated index period " << pShift <<
61  " and index availability " << infIndex_->availabilityLag() <<
62  " need (obsLag-index period) > availLag");
63  } else {
64  QL_REQUIRE(infIndex_->availabilityLag() < observationLag_,
65  "index tries to observe inflation fixings that do not yet exist: "
66  << " availability lag " << infIndex_->availabilityLag()
67  << " versus obs lag = " << observationLag_);
68  }
69 
70  if (infCalendar_==Calendar()) infCalendar_ = fixCalendar_;
71  if (infConvention_==BusinessDayConvention()) infConvention_ = fixConvention_;
72 
73  if (adjustInfObsDates_) {
74  baseDate_ = infCalendar_.adjust(startDate - observationLag_, infConvention_);
75  obsDate_ = infCalendar_.adjust(maturity - observationLag_, infConvention_);
76  } else {
77  baseDate_ = startDate - observationLag_;
78  obsDate_ = maturity - observationLag_;
79  }
80 
81  Date infPayDate = infCalendar_.adjust(maturity, infConvention_);
82  Date fixedPayDate = fixCalendar_.adjust(maturity, fixConvention_);
83 
84  // At this point the index may not be able to forecast
85  // i.e. do not want to force the existence of an inflation
86  // term structure before allowing users to create instruments.
87  Real T = inflationYearFraction(infIndex_->frequency(), infIndex_->interpolated(),
88  dayCounter_, baseDate_, obsDate_);
89  // N.B. the -1.0 is because swaps only exchange growth, not notionals as well
90  Real fixedAmount = nominal * ( std::pow(1.0 + fixedRate, T) - 1.0 );
91 
92  legs_[0].push_back(boost::shared_ptr<CashFlow>(
93  new SimpleCashFlow(fixedAmount, fixedPayDate)));
94  bool growthOnly = true;
95  legs_[1].push_back(boost::shared_ptr<CashFlow>(
96  new IndexedCashFlow(nominal,infIndex,baseDate_,obsDate_,infPayDate,growthOnly)));
97 
98  for (Size j=0; j<2; ++j) {
99  for (Leg::iterator i = legs_[j].begin(); i!= legs_[j].end(); ++i)
100  registerWith(*i);
101  }
102 
103  switch (type_) {
104  case Payer:
105  payer_[0] = +1.0;
106  payer_[1] = -1.0;
107  break;
108  case Receiver:
109  payer_[0] = -1.0;
110  payer_[1] = +1.0;
111  break;
112  default:
113  QL_FAIL("Unknown zero-inflation-swap type");
114  }
115 
116  }

135  Real ZeroCouponInflationSwap::fairRate() const {
136  // What does this mean before or after trade date?
137  // Always means that NPV is zero for _this_ instrument
138  // if it was created with _this_ rate
139  // _knowing_ the time from base to obs (etc).
140 
141  boost::shared_ptr<IndexedCashFlow> icf =
142  boost::dynamic_pointer_cast<IndexedCashFlow>(legs_[1].at(0));
143  QL_REQUIRE(icf,"failed to downcast to IndexedCashFlow in ::fairRate()");
144 
145  // +1 because the IndexedCashFlow has growthOnly=true
146  Real growth = icf->amount() / icf->notional() + 1.0;
147  Real T = inflationYearFraction(infIndex_->frequency(),
148  infIndex_->interpolated(),
149  dayCounter_, baseDate_, obsDate_);
150 
151  return std::pow(growth,1.0/T) - 1.0;
