38  class DecoratedHedge : public CallSpecifiedMultiProduct {
39  public:
40  explicit DecoratedHedge(const CallSpecifiedMultiProduct& product)
41  : CallSpecifiedMultiProduct(product) {
42  savedStates_.reserve(product.evolution().numberOfSteps());
43 
44  Size N = product.numberOfProducts();
45  numberCashFlowsThisStep_.resize(N);
46  cashFlowsGenerated_.resize(N);
47  for (Size i=0; i<N; ++i)
48  cashFlowsGenerated_[i].resize(
49  product.maxNumberOfCashFlowsPerProductPerStep());
50 
51  clear();
52  }

54  void reset() {
55  CallSpecifiedMultiProduct::reset();
56  disableCallability();
57  for (Size i=0; i<lastSavedStep_; ++i)
58  CallSpecifiedMultiProduct::nextTimeStep(
59  *savedStates_[i],
60  numberCashFlowsThisStep_,
61  cashFlowsGenerated_);
62  enableCallability();
63  }

111  UpperBoundEngine::UpperBoundEngine(
112  const ext::shared_ptr<MarketModelEvolver>& evolver,
113  const std::vector<ext::shared_ptr<MarketModelEvolver> >&
114  innerEvolvers,
115  const MarketModelMultiProduct& underlying,
116  const MarketModelExerciseValue& rebate,
117  const MarketModelMultiProduct& hedge,
118  const MarketModelExerciseValue& hedgeRebate,
119  const ExerciseStrategy<CurveState>& hedgeStrategy,
120  Real initialNumeraireValue)
121  : evolver_(evolver), innerEvolvers_(innerEvolvers),
122  composite_(MultiProductComposite()),
123  initialNumeraireValue_(initialNumeraireValue) {
124 
125  composite_.add(underlying);
126  composite_.add(ExerciseAdapter(rebate));
127  composite_.add(hedge);
128  composite_.add(ExerciseAdapter(hedgeRebate));
129  composite_.add(DecoratedHedge(CallSpecifiedMultiProduct(
130  hedge,hedgeStrategy,ExerciseAdapter(hedgeRebate))));
131  composite_.finalize();
132 
133  underlyingOffset_ = 0;
134  underlyingSize_ = underlying.numberOfProducts();
135  rebateOffset_ = underlyingSize_;
136  rebateSize_ = 1;
137  hedgeOffset_ = underlyingSize_+rebateSize_;
138  hedgeSize_ = hedge.numberOfProducts();
139  hedgeRebateOffset_ = underlyingSize_+rebateSize_+hedgeSize_;
140  hedgeRebateSize_ = 1;
141 
142 
143  numberOfProducts_ = composite_.numberOfProducts();
144 
145  const std::vector<Time>& evolutionTimes =
146  composite_.evolution().evolutionTimes();
147  numberOfSteps_ = evolutionTimes.size();
148 
149  isExerciseTime_.resize(evolutionTimes.size());
150  isExerciseTime_ = isInSubset(evolutionTimes,
151  hedgeStrategy.exerciseTimes());
152 
153  numberCashFlowsThisStep_.resize(numberOfProducts_);
154  cashFlowsGenerated_.resize(numberOfProducts_);
155  for (Size i=0; i<numberOfProducts_; ++i)
156  cashFlowsGenerated_[i].resize(
157  composite_.maxNumberOfCashFlowsPerProductPerStep());
158 
159  const std::vector<Time>& cashFlowTimes =
160  composite_.possibleCashFlowTimes();
161  const std::vector<Rate>& rateTimes =
162  composite_.evolution().rateTimes();
163  Size n =cashFlowTimes.size();
164  discounters_.reserve(n);
165  for (Size j=0; j<n; ++j)
166  discounters_.push_back(MarketModelDiscounter(cashFlowTimes[j],
167  rateTimes));
168  }

181  std::pair<Real,Real> UpperBoundEngine::singlePathValue(Size innerPaths) {
182 
183  DecoratedHedge& callable =
184  dynamic_cast<DecoratedHedge&>(composite_.item(4));
185  const ExerciseStrategy<CurveState>& strategy = callable.strategy();
186 
187 
188  Real maximumValue = QL_MIN_REAL;
189  Real numerairesHeld = 0.0;
190  Real weight = evolver_->startNewPath();
191  callable.clear();
192  composite_.reset();
193  callable.disableCallability();
194  Real principalInNumerairePortfolio = 1.0;
195  Size exercise = 0;
196 
197  for (Size k=0; k<numberOfSteps_; ++k) {
198  weight *= evolver_->advanceStep();
199 
200  composite_.nextTimeStep(evolver_->currentState(),
201  numberCashFlowsThisStep_,
202  cashFlowsGenerated_);
203 
204  // First, we accumulate cash flows from both the
205  // underlying...
206  Real underlyingCashFlows =
207  collectCashFlows(k,
208  principalInNumerairePortfolio,
209  underlyingOffset_,
210  underlyingOffset_+underlyingSize_);
211 
212  // ...and the hedge
213  Real hedgeCashFlows =
214  collectCashFlows(k,
215  principalInNumerairePortfolio,
216  hedgeOffset_,
217  hedgeOffset_+hedgeSize_);
218 
219  // we do the same for the rebates. Warning: this relies on
220  // the fact that on each exercise date an ExerciseAdapter
221  // generates a cash-flow equal to the exercise value
222  Real rebateCashFlow =
223  collectCashFlows(k,
224  principalInNumerairePortfolio,
225  rebateOffset_,
226  rebateOffset_+rebateSize_);
227 
228  Real hedgeRebateCashFlow =
229  collectCashFlows(k,
230  principalInNumerairePortfolio,
231  hedgeRebateOffset_,
232  hedgeRebateOffset_+hedgeRebateSize_);
233 
234 
235  numerairesHeld += underlyingCashFlows - hedgeCashFlows;
236 
237  // Second, we do the upper-bound thing
238  if (isExerciseTime_[k]) {
239 
240  Real unexercisedHedgeValue = 0.0;
241 
242  if (k != numberOfSteps_-1) {
243 
244  // Here, we setup the relevant inner evolver and
245  // the decorated callable hedge such that their
246  // reset() method brings them to the current point
247  // rather than the beginning of the path.
248 
249  ext::shared_ptr<MarketModelEvolver> currentEvolver =
250  innerEvolvers_[exercise++];
251  currentEvolver->setInitialState(evolver_->currentState());
252 
253  callable.stopRecording();
254  callable.enableCallability();
255  callable.save();
256 
257  // This allows us to write:
258  AccountingEngine engine(currentEvolver, callable,
259  1.0); // this causes the result
260  // to be in numeraire units
261  SequenceStatisticsInc innerStats(callable.numberOfProducts());
262  engine.multiplePathValues(innerStats, innerPaths);
263 
264  const std::vector<Real>& values = innerStats.mean();
265  unexercisedHedgeValue =
266  std::accumulate(values.begin(), values.end(), Real(0.0))
267  / principalInNumerairePortfolio;
268 
269  callable.disableCallability();
270  callable.startRecording();
271 
272  }
273 
274  // Now, we can calculate the total value of our hedged
275  // portfolio...
276  Real portfolioValue = numerairesHeld;
277  if (strategy.exercise(evolver_->currentState())) {
278  // get the rebates...
279  portfolioValue +=
280  rebateCashFlow - hedgeRebateCashFlow;
281  // ...and reinvest to rehedge
282  numerairesHeld +=
283  unexercisedHedgeValue - hedgeRebateCashFlow;
284  } else {
285  portfolioValue +=
286  rebateCashFlow - unexercisedHedgeValue;
287  }
288 
289  // ...and use it to update the maximum value
290  maximumValue = std::max(maximumValue, portfolioValue);
291  }
292 
293 
294  // Lastly, we do the homework for next step (if any)
295  if (k<numberOfSteps_-1) {
296 
297  // The numeraire might change between steps. This implies
298  // that we might have to convert the numeraire bonds for
299  // this step into a corresponding amount of numeraire
300  // bonds for the next step. This can be done by changing
301  // the principal of the numeraire and updating the number
302  // of bonds in the numeraire portfolio accordingly.
303 
304  Size numeraire = evolver_->numeraires()[k];
305  Size nextNumeraire = evolver_->numeraires()[k+1];
306 
307  principalInNumerairePortfolio *=
308  evolver_->currentState().discountRatio(numeraire,
309  nextNumeraire);
310  }
311 
312  }
313 
314  // finally, we update the maximum with the total accumulated
315  // cash flows (in case we never exercised)
316  maximumValue = std::max(maximumValue, numerairesHeld);
317 
318 
319  // all done; we just convert the result back to cash
320  maximumValue *= initialNumeraireValue_;
321 
322  return std::make_pair(maximumValue, weight);
323  }

326  Real UpperBoundEngine::collectCashFlows(Size currentStep,
327  Real principalInNumerairePortfolio,
328  Size beginProduct,
329  Size endProduct) const {
330  Size numeraire = evolver_->numeraires()[currentStep];
331 
332  Real numeraireUnits = 0.0;
333  // For each product in range...
334  for (Size i=beginProduct; i<endProduct; ++i) {
335  // ...and for each cash flow...
336  const std::vector<MarketModelMultiProduct::CashFlow>& cashflows =
337  cashFlowsGenerated_[i];
338  for (Size j=0; j<numberCashFlowsThisStep_[i]; ++j) {
339  // ...convert the cash flow to numeraires. This is
340  // done by calculating the number of numeraire bonds
341  // corresponding to such cash flow...
342  const MarketModelDiscounter& discounter =
343  discounters_[cashflows[j].timeIndex];
344  // ...and adding the newly bought bonds to the total
345  numeraireUnits += cashflows[j].amount *
346  discounter.numeraireBonds(evolver_->currentState(),
347  numeraire);
348  }
349  }
350  return numeraireUnits/principalInNumerairePortfolio;
351  }
