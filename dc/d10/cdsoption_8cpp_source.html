70  CdsOption::CdsOption(const boost::shared_ptr<CreditDefaultSwap>& swap,

72  bool knocksOut)
73  : Option(boost::shared_ptr<Payoff>(new NullPayoff), exercise),
74  swap_(swap), knocksOut_(knocksOut) {

123  Volatility CdsOption::impliedVolatility(
124  Real targetValue,
125  const Handle<YieldTermStructure>& termStructure,
126  const Handle<DefaultProbabilityTermStructure>& probability,
127  Real recoveryRate,
128  Real accuracy,
129  Size maxEvaluations,
130  Volatility minVol,
