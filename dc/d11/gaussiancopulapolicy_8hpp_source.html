47  /* check factors in LM are normalized. */

49  Real factorsNorm = std::inner_product(factorWeight.begin(), factorWeight.end(),
50  factorWeight.begin(), 0.);
51  QL_REQUIRE(factorsNorm < 1.,
52  "Non normal random factor combination.");
