26  KahaleSmileSection::KahaleSmileSection(
27  const boost::shared_ptr<SmileSection> source, const Real atm,
28  const bool interpolate, const bool exponentialExtrapolation,
29  const bool deleteArbitragePoints,
30  const std::vector<Real> &moneynessGrid, const Real gap,
31  const int forcedLeftIndex, const int forcedRightIndex)
32  : SmileSection(*source), source_(source), moneynessGrid_(moneynessGrid),
33  gap_(gap), interpolate_(interpolate),
34  exponentialExtrapolation_(exponentialExtrapolation),
35  forcedLeftIndex_(forcedLeftIndex), forcedRightIndex_(forcedRightIndex) {
36 
37  // only shifted lognormal smile sections are supported
38 
39  QL_REQUIRE(source->volatilityType() == ShiftedLognormal,
40  "KahaleSmileSection only supports shifted lognormal source "
41  "sections");
42 
43  ssutils_ = boost::shared_ptr<SmileSectionUtils>(new SmileSectionUtils(
44  *source, moneynessGrid, atm, deleteArbitragePoints));
45 
46  moneynessGrid_ = ssutils_->moneyGrid();
47  k_ = ssutils_->strikeGrid();
48  c_ = ssutils_->callPrices();
49  f_ = ssutils_->atmLevel();
50 
51  // for shifted smile sections we shift the forward and the strikes
52  // and do as if we were in a lognormal setting
53 
54  for(Size i=0;i<k_.size();++i) {
55  k_[i] += shift();
56  }
57 
58  f_ += shift();
59 
60  compute();
61  }

63  void KahaleSmileSection::compute() {
64 
65  std::pair<Size, Size> afIdx = ssutils_->arbitragefreeIndices();
66  leftIndex_ = afIdx.first;
67  rightIndex_ = afIdx.second;
68 
69  cFunctions_ = std::vector<boost::shared_ptr<cFunction> >(
70  rightIndex_ - leftIndex_ + 2);
71 
72  // extrapolation in the leftmost interval
73 
74  Brent brent;
75  bool success;
76  Real secl = 0.0;
77 
78  do {
79  success = true;
80  try {
81  Real k1 = k_[leftIndex_];
82  Real c1 = c_[leftIndex_];
83  Real c0 = c_[0];
84  secl = (c_[leftIndex_] - c_[0]) / (k_[leftIndex_] - k_[0]);
85  Real sec = (c_[leftIndex_ + 1] - c_[leftIndex_]) /
86  (k_[leftIndex_ + 1] - k_[leftIndex_]);
87  Real c1p;
88  if (interpolate_)
89  c1p = (secl + sec) / 2;
90  else {
91  c1p = -source_->digitalOptionPrice(
92  k1 - shift() + gap_ / 2.0, Option::Call, 1.0, gap_);
93  QL_REQUIRE(secl < c1p && c1p <= 0.0, "dummy");
94  // can not extrapolate so throw exception which is caught
95  // below
96  }
97  sHelper1 sh1(k1, c0, c1, c1p);
98  Real s = brent.solve(sh1, QL_KAHALE_ACC, 0.20, 0.00,
99  QL_KAHALE_SMAX); // numerical parameters
100  // hardcoded here
101  sh1(s);
102  boost::shared_ptr<cFunction> cFct1(
103  new cFunction(sh1.f_, s, 0.0, sh1.b_));
104  cFunctions_[0] = cFct1;
105  // sanity check - in rare cases we can get digitials
106  // which are not monotonic or greater than 1.0
107  // due to numerical effects. Move to the next index in
108  // these cases.
109  Real dig = digitalOptionPrice((k1 - shift()) / 2.0, Option::Call,
110  1.0, gap_);
111  QL_REQUIRE(dig >= -c1p && dig <= 1.0, "dummy");
112  if(static_cast<int>(leftIndex_) < forcedLeftIndex_) {
113  leftIndex_++;
114  success = false;
115  }
116  }
117  catch (...) {
118  leftIndex_++;
119  success = false;
120  }
121  } while (!success && leftIndex_ < rightIndex_);
122 
123  QL_REQUIRE(
124  leftIndex_ < rightIndex_,
125  "can not extrapolate to left, right index of af region reached ("
126  << rightIndex_ << ")");
127 
128  // interpolation
129 
130  Real cp0 = 0.0, cp1 = 0.0;
131 
132  if (interpolate_) {
133 
134  for (Size i = leftIndex_; i < rightIndex_; i++) {
135  Real k0 = k_[i];
136  Real k1 = k_[i + 1];
137  Real c0 = c_[i];
138  Real c1 = c_[i + 1];
139  Real sec = (c_[i + 1] - c_[i]) / (k_[i + 1] - k_[i]);
140  if (i == leftIndex_)
141  cp0 = leftIndex_ > 0 ? (secl + sec) / 2.0 : sec;
142  Real secr;
143  if (i == rightIndex_ - 1)
144  secr = 0.0;
145  else
146  secr = (c_[i + 2] - c_[i + 1]) / (k_[i + 2] - k_[i + 1]);
147  cp1 = (sec + secr) / 2.0;
148  aHelper ah(k0, k1, c0, c1, cp0, cp1);
149  Real a;
150  bool valid = false;
151  try {
152  a = brent.solve(
153  ah, QL_KAHALE_ACC, 0.5 * (cp1 + (1.0 + cp0)),
154  cp1 + QL_KAHALE_EPS, 1.0 + cp0 - QL_KAHALE_EPS);
155  // numerical parameters hardcoded here
156  valid = true;
157  }
158  catch (...) {
159  // delete the right point of the interval where we try to
160  // interpolate
161  moneynessGrid_.erase(moneynessGrid_.begin() + (i + 1));
162  k_.erase(k_.begin() + (i + 1));
163  c_.erase(c_.begin() + (i + 1));
164  cFunctions_.erase(cFunctions_.begin() + (i + 1));
165  rightIndex_--;
166  i--;
167  }
168  if (valid) {
169  ah(a);
170  boost::shared_ptr<cFunction> cFct(
171  new cFunction(ah.f_, ah.s_, a, ah.b_));
172  cFunctions_[leftIndex_ > 0 ? i - leftIndex_ + 1 : 0] = cFct;
173  cp0 = cp1;
174  }
175  }
176  }
177 
178  // extrapolation of right wing
179 
180  do {
181  success = true;
182  try {
183  Real k0 = k_[rightIndex_];
184  Real c0 = c_[rightIndex_];
185  Real cp0;
186  if (interpolate_)
187  cp0 = 0.5 * (c_[rightIndex_] - c_[rightIndex_ - 1]) /
188  (k_[rightIndex_] - k_[rightIndex_ - 1]);
189  else {
190  cp0 = -source_->digitalOptionPrice(
191  k0 - shift() - gap_ / 2.0, Option::Call, 1.0, gap_);
192  }
193  boost::shared_ptr<cFunction> cFct;
194  if (exponentialExtrapolation_) {
195  QL_REQUIRE(-cp0 / c0 > 0.0, "dummy"); // this is caught
196  // below
197  cFct = boost::shared_ptr<cFunction>(
198  new cFunction(-cp0 / c0, std::log(c0) - cp0 / c0 * k0));
199  } else {
200  sHelper sh(k0, c0, cp0);
201  Real s;
202  s = brent.solve(sh, QL_KAHALE_ACC, 0.20, 0.0,
203  QL_KAHALE_SMAX); // numerical parameters
204  // hardcoded here
205  sh(s);
206  cFct = boost::shared_ptr<cFunction>(
207  new cFunction(sh.f_, s, 0.0, 0.0));
208  }
209  cFunctions_[rightIndex_ - leftIndex_ + 1] = cFct;
210  }
211  catch (...) {
212  rightIndex_--;
213  success = false;
214  }
215  if(static_cast<int>(rightIndex_) > forcedRightIndex_) {
216  rightIndex_--;
217  success = false;
218  }
219  } while (!success && rightIndex_ > leftIndex_);
220 
221  QL_REQUIRE(
222  leftIndex_ < rightIndex_,
223  "can not extrapolate to right, left index of af region reached ("
224  << leftIndex_ << ")");
225  }

227  Real KahaleSmileSection::optionPrice(Rate strike, Option::Type type,
228  Real discount) const {
229  // option prices are directly available, so implement this function
230  // rather than use smileSection
231  // standard implementation
232  Real shifted_strike = std::max(strike + shift(), QL_KAHALE_EPS);
233  int i = index(shifted_strike);
234  if (interpolate_ ||
235  (i == 0 || i == (int)(rightIndex_ - leftIndex_ + 1)))
236  return discount *
237  (type == Option::Call
238  ? cFunctions_[i]->operator()(shifted_strike)
239  : cFunctions_[i]->operator()(shifted_strike) + shifted_strike - f_);
240  else
241  return source_->optionPrice(strike, type, discount);
242  }

244  Real KahaleSmileSection::volatilityImpl(Rate strike) const {
245  Real shifted_strike = std::max(strike + shift(), QL_KAHALE_EPS);
246  int i = index(shifted_strike);
247  if (!interpolate_ &&
248  !(i == 0 || i == (int)(rightIndex_ - leftIndex_ + 1)))
249  return source_->volatility(strike);
250  Real c = cFunctions_[i]->operator()(shifted_strike);
251  Real vol = 0.0;
252  try {
253  Option::Type type = shifted_strike >= f_ ? Option::Call : Option::Put;
254  vol = blackFormulaImpliedStdDev(
255  type, shifted_strike, f_,
256  type == Option::Put ? strike - f_ + c : c) /
257  sqrt(exerciseTime());
258  }
259  catch (...) {
260  }
261  return vol;
262  }

264  Size KahaleSmileSection::index(Rate strike) const {
265  int i =
266  static_cast<int>(std::upper_bound(k_.begin(), k_.end(), strike) -
267  k_.begin()) -
268  static_cast<int>(leftIndex_);
269  return std::max(
270  std::min(i, static_cast<int>(rightIndex_ - leftIndex_ + 1)), 0);
271  }
