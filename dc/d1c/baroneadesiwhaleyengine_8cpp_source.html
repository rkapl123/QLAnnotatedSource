37  // critical commodity price
38  Real BaroneAdesiWhaleyApproximationEngine::criticalPrice(
39  const boost::shared_ptr<StrikedTypePayoff>& payoff,
40  DiscountFactor riskFreeDiscount,
41  DiscountFactor dividendDiscount,
42  Real variance, Real tolerance) {
43 
44  // Calculation of seed value, Si
45  Real n= 2.0*std::log(dividendDiscount/riskFreeDiscount)/(variance);
46  Real m=-2.0*std::log(riskFreeDiscount)/(variance);
47  Real bT = std::log(dividendDiscount/riskFreeDiscount);
48 
49  Real qu, Su, h, Si;
50  switch (payoff->optionType()) {
51  case Option::Call:
52  qu = (-(n-1.0) + std::sqrt(((n-1.0)*(n-1.0)) + 4.0*m))/2.0;
53  Su = payoff->strike() / (1.0 - 1.0/qu);
54  h = -(bT + 2.0*std::sqrt(variance)) * payoff->strike() /
55  (Su - payoff->strike());
56  Si = payoff->strike() + (Su - payoff->strike()) *
57  (1.0 - std::exp(h));
58  break;
59  case Option::Put:
60  qu = (-(n-1.0) - std::sqrt(((n-1.0)*(n-1.0)) + 4.0*m))/2.0;
61  Su = payoff->strike() / (1.0 - 1.0/qu);
62  h = (bT - 2.0*std::sqrt(variance)) * payoff->strike() /
63  (payoff->strike() - Su);
64  Si = Su + (payoff->strike() - Su) * std::exp(h);
65  break;
66  default:
67  QL_FAIL("unknown option type");
68  }
69 
70 
71  // Newton Raphson algorithm for finding critical price Si
72  Real Q, LHS, RHS, bi;
73  Real forwardSi = Si * dividendDiscount / riskFreeDiscount;
74  Real d1 = (std::log(forwardSi/payoff->strike()) + 0.5*variance) /
75  std::sqrt(variance);
76  CumulativeNormalDistribution cumNormalDist;
77  Real K = (!close(riskFreeDiscount, 1.0, 1000))
78  ? -2.0*std::log(riskFreeDiscount)
79  / (variance*(1.0-riskFreeDiscount))
80  : 2.0/variance;
81  Real temp = blackFormula(payoff->optionType(), payoff->strike(),
82  forwardSi, std::sqrt(variance))*riskFreeDiscount;
83  switch (payoff->optionType()) {
84  case Option::Call:
85  Q = (-(n-1.0) + std::sqrt(((n-1.0)*(n-1.0)) + 4 * K)) / 2;
86  LHS = Si - payoff->strike();
87  RHS = temp + (1 - dividendDiscount * cumNormalDist(d1)) * Si / Q;
88  bi = dividendDiscount * cumNormalDist(d1) * (1 - 1/Q) +
89  (1 - dividendDiscount *
90  cumNormalDist.derivative(d1) / std::sqrt(variance)) / Q;
91  while (std::fabs(LHS - RHS)/payoff->strike() > tolerance) {
92  Si = (payoff->strike() + RHS - bi * Si) / (1 - bi);
93  forwardSi = Si * dividendDiscount / riskFreeDiscount;
94  d1 = (std::log(forwardSi/payoff->strike())+0.5*variance)
95  /std::sqrt(variance);
96  LHS = Si - payoff->strike();
97  Real temp2 = blackFormula(payoff->optionType(), payoff->strike(),
98  forwardSi, std::sqrt(variance))*riskFreeDiscount;
99  RHS = temp2 + (1 - dividendDiscount * cumNormalDist(d1)) * Si / Q;
100  bi = dividendDiscount * cumNormalDist(d1) * (1 - 1 / Q)
101  + (1 - dividendDiscount *
102  cumNormalDist.derivative(d1) / std::sqrt(variance))
103  / Q;
104  }
105  break;
106  case Option::Put:
107  Q = (-(n-1.0) - std::sqrt(((n-1.0)*(n-1.0)) + 4 * K)) / 2;
108  LHS = payoff->strike() - Si;
109  RHS = temp - (1 - dividendDiscount * cumNormalDist(-d1)) * Si / Q;
110  bi = -dividendDiscount * cumNormalDist(-d1) * (1 - 1/Q)
111  - (1 + dividendDiscount * cumNormalDist.derivative(-d1)
112  / std::sqrt(variance)) / Q;
113  while (std::fabs(LHS - RHS)/payoff->strike() > tolerance) {
114  Si = (payoff->strike() - RHS + bi * Si) / (1 + bi);
115  forwardSi = Si * dividendDiscount / riskFreeDiscount;
116  d1 = (std::log(forwardSi/payoff->strike())+0.5*variance)
117  /std::sqrt(variance);
118  LHS = payoff->strike() - Si;
119  Real temp2 = blackFormula(payoff->optionType(), payoff->strike(),
120  forwardSi, std::sqrt(variance))*riskFreeDiscount;
121  RHS = temp2 - (1 - dividendDiscount * cumNormalDist(-d1)) * Si / Q;
122  bi = -dividendDiscount * cumNormalDist(-d1) * (1 - 1 / Q)
123  - (1 + dividendDiscount * cumNormalDist.derivative(-d1)
124  / std::sqrt(variance)) / Q;
125  }
126  break;
127  default:
128  QL_FAIL("unknown option type");
129  }
130 
131  return Si;
132  }

134  void BaroneAdesiWhaleyApproximationEngine::calculate() const {
135 
136  QL_REQUIRE(arguments_.exercise->type() == Exercise::American,
137  "not an American Option");
138 
139  boost::shared_ptr<AmericanExercise> ex =
140  boost::dynamic_pointer_cast<AmericanExercise>(arguments_.exercise);
141  QL_REQUIRE(ex, "non-American exercise given");
142  QL_REQUIRE(!ex->payoffAtExpiry(),
143  "payoff at expiry not handled");
144 
145  boost::shared_ptr<StrikedTypePayoff> payoff =
146  boost::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
147  QL_REQUIRE(payoff, "non-striked payoff given");
148 
149  Real variance = process_->blackVolatility()->blackVariance(
150  ex->lastDate(), payoff->strike());
151  DiscountFactor dividendDiscount = process_->dividendYield()->discount(
152  ex->lastDate());
153  DiscountFactor riskFreeDiscount = process_->riskFreeRate()->discount(
154  ex->lastDate());
155  Real spot = process_->stateVariable()->value();
156  QL_REQUIRE(spot > 0.0, "negative or null underlying given");
157  Real forwardPrice = spot * dividendDiscount / riskFreeDiscount;
158  BlackCalculator black(payoff, forwardPrice, std::sqrt(variance),
159  riskFreeDiscount);
160 
161  if (dividendDiscount>=1.0 && payoff->optionType()==Option::Call) {
162  // early exercise never optimal
163  results_.value = black.value();
164  results_.delta = black.delta(spot);
165  results_.deltaForward = black.deltaForward();
166  results_.elasticity = black.elasticity(spot);
167  results_.gamma = black.gamma(spot);
168 
169  DayCounter rfdc = process_->riskFreeRate()->dayCounter();
170  DayCounter divdc = process_->dividendYield()->dayCounter();
171  DayCounter voldc = process_->blackVolatility()->dayCounter();
172  Time t =
173  rfdc.yearFraction(process_->riskFreeRate()->referenceDate(),
174  arguments_.exercise->lastDate());
175  results_.rho = black.rho(t);
176 
177  t = divdc.yearFraction(process_->dividendYield()->referenceDate(),
178  arguments_.exercise->lastDate());
179  results_.dividendRho = black.dividendRho(t);
180 
181  t = voldc.yearFraction(process_->blackVolatility()->referenceDate(),
182  arguments_.exercise->lastDate());
183  results_.vega = black.vega(t);
184  results_.theta = black.theta(spot, t);
185  results_.thetaPerDay = black.thetaPerDay(spot, t);
186 
187  results_.strikeSensitivity = black.strikeSensitivity();
188  results_.itmCashProbability = black.itmCashProbability();
189  } else {
190  // early exercise can be optimal
191  CumulativeNormalDistribution cumNormalDist;
192  Real tolerance = 1e-6;
193  Real Sk = criticalPrice(payoff, riskFreeDiscount,
194  dividendDiscount, variance, tolerance);
195  Real forwardSk = Sk * dividendDiscount / riskFreeDiscount;
196  Real d1 = (std::log(forwardSk/payoff->strike()) + 0.5*variance)
197  /std::sqrt(variance);
198  Real n = 2.0*std::log(dividendDiscount/riskFreeDiscount)/variance;
199  Real K = (!close(riskFreeDiscount, 1.0, 1000))
200  ? -2.0*std::log(riskFreeDiscount)
201  / (variance*(1.0-riskFreeDiscount))
202  : 2.0/variance;
203  Real Q, a;
204  switch (payoff->optionType()) {
205  case Option::Call:
206  Q = (-(n-1.0) + std::sqrt(((n-1.0)*(n-1.0))+4.0*K))/2.0;
207  a = (Sk/Q) * (1.0 - dividendDiscount * cumNormalDist(d1));
208  if (spot<Sk) {
209  results_.value = black.value() +
210  a * std::pow((spot/Sk), Q);
211  } else {
212  results_.value = spot - payoff->strike();
213  }
214  break;
215  case Option::Put:
216  Q = (-(n-1.0) - std::sqrt(((n-1.0)*(n-1.0))+4.0*K))/2.0;
217  a = -(Sk/Q) *
218  (1.0 - dividendDiscount * cumNormalDist(-d1));
219  if (spot>Sk) {
220  results_.value = black.value() +
221  a * std::pow((spot/Sk), Q);
222  } else {
223  results_.value = payoff->strike() - spot;
224  }
225  break;
226  default:
227  QL_FAIL("unknown option type");
228  }
229  } // end of "early exercise can be optimal"
230  }
