38  DiscretizedSwaption::DiscretizedSwaption(const Swaption::arguments& args,
39  const Date& referenceDate,
40  const DayCounter& dayCounter)
41  : DiscretizedOption(ext::shared_ptr<DiscretizedAsset>(),
42  args.exercise->type(),
43  std::vector<Time>()),
44  arguments_(args) {
45 
46  exerciseTimes_.resize(arguments_.exercise->dates().size());
47  for (Size i=0; i<exerciseTimes_.size(); ++i)
48  exerciseTimes_[i] =
49  dayCounter.yearFraction(referenceDate,
50  arguments_.exercise->date(i));
51 
52  // Date adjustments can get time vectors out of synch.
53  // Here, we try and collapse similar dates which could cause
54  // a mispricing.
55  for (Size i=0; i<arguments_.exercise->dates().size(); i++) {
56  Date exerciseDate = arguments_.exercise->date(i);
57  for (Size j=0; j<arguments_.fixedPayDates.size(); j++) {
58  if (withinNextWeek(exerciseDate,
59  arguments_.fixedPayDates[j])
60  // coupons in the future are dealt with below
61  && arguments_.fixedResetDates[j] < referenceDate)
62  arguments_.fixedPayDates[j] = exerciseDate;
63  }
64  for (Size j=0; j<arguments_.fixedResetDates.size(); j++) {
65  if (withinPreviousWeek(exerciseDate,
66  arguments_.fixedResetDates[j]))
67  arguments_.fixedResetDates[j] = exerciseDate;
68  }
69  for (Size j=0; j<arguments_.floatingResetDates.size(); j++) {
70  if (withinPreviousWeek(exerciseDate,
71  arguments_.floatingResetDates[j]))
72  arguments_.floatingResetDates[j] = exerciseDate;
73  }
74  }
75 
76  Time lastFixedPayment =
77  dayCounter.yearFraction(referenceDate,
78  arguments_.fixedPayDates.back());
79  Time lastFloatingPayment =
80  dayCounter.yearFraction(referenceDate,
81  arguments_.floatingPayDates.back());
82  lastPayment_ = std::max(lastFixedPayment,lastFloatingPayment);
83 
84  underlying_ = ext::shared_ptr<DiscretizedAsset>(
85  new DiscretizedSwap(arguments_,
86  referenceDate,
87  dayCounter));
88  }
