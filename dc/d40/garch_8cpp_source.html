70  Real Garch11CostFunction::value(const Array& x) const {
71  Real retval(0.0);
72  Real sigma2 = 0;
73  Real u2 = 0;
74  BOOST_FOREACH (Volatility r2, r2_) {
75  sigma2 = x[0] + x[1] * u2 + x[2] * sigma2;
76  u2 = r2;
77  retval += std::log(sigma2) + u2 / sigma2;
78  }
79  return retval / (2.0*r2_.size());

95  void Garch11CostFunction::gradient(Array& grad, const Array& x) const {
96  std::fill (grad.begin(), grad.end(), 0.0);
97  Real sigma2 = 0;
98  Real u2 = 0;
99  Real sigma2prev = sigma2;
100  Real u2prev = u2;
101  Real norm = 2.0 * r2_.size();
102  BOOST_FOREACH (Volatility r2, r2_) {
103  sigma2 = x[0] + x[1] * u2 + x[2] * sigma2;
104  u2 = r2;
105  Real w = (sigma2 - u2) / (sigma2*sigma2);
106  grad[0] += w;
107  grad[1] += u2prev * w;
108  grad[2] += sigma2prev * w;
109  u2prev = u2;
110  sigma2prev = sigma2;
111  }
112  std::transform(grad.begin(), grad.end(), grad.begin(),
113  divide_by<Real>(norm));
114  }

164  void FitAcfProblem::targetAndValue(const Array& x, Array& target,
165  Array& fct2fit) {
166  Real A4 = acf_[0] + A2_*A2_;
167  Real gamma = x[0];
168  Real beta = x[1];
169  target[0] = A2_*A2_/A4;
170  fct2fit[0] =
171  (1 - 3*gamma*gamma - 2*beta*beta + 4*beta*gamma)
172  / (3*(1 - gamma*gamma));
173  target[1] = acf_[1] / A4;
174  fct2fit[1] = gamma * (1 - fct2fit[0]) - beta;
175  for (std::size_t i = 2; i < idx_.size(); ++i) {
176  target[i] = acf_[idx_[i]] / A4;
177  fct2fit[i] = std::pow(gamma, (int)idx_[i]-1)* fct2fit[1];
178  }
179  }

181  void FitAcfProblem::targetValueAndGradient(const Array& x,
182  Matrix& grad_fct2fit,
183  Array& target,
184  Array& fct2fit) {
185  Real A4 = acf_[0] + A2_*A2_;
186  Real gamma = x[0];
187  Real beta = x[1];
188  target[0] = A2_*A2_/A4;
189  Real w1 = (1 - 3*gamma*gamma - 2*beta*beta + 4*beta*gamma);
190  Real w2 = (1 - gamma*gamma);
191  fct2fit[0] = w1 / (3*w2);
192  grad_fct2fit[0][0] = (2.0/3.0) * ((2*beta-3*gamma)*w2 + 2*w1*gamma) / (w2*w2);
193  grad_fct2fit[0][1] = (4.0/3.0) * (gamma - beta) / w2;
194  target[1] = acf_[1] / A4;
195  fct2fit[1] = gamma * (1 - fct2fit[0]) - beta;
196  grad_fct2fit[1][0] = (1 - fct2fit[0]) - gamma * grad_fct2fit[0][0];
197  grad_fct2fit[1][1] = -gamma * grad_fct2fit[0][1] - 1;
198  for (std::size_t i = 2; i < idx_.size(); ++i) {
199  target[i] = acf_[idx_[i]] / A4;
200  w1 = std::pow(gamma, (int)idx_[i]-1);
201  fct2fit[i] = w1 * fct2fit[1];
202  grad_fct2fit[i][0] = (idx_[i]-1) * (w1/gamma)*fct2fit[1] + w1*grad_fct2fit[1][0];
203  grad_fct2fit[i][1] = w1 * grad_fct2fit[1][1];
204  }
205  }

228  // Initial guess based on fitting ACF - initial guess for
229  // fitting acf is a moment matching estimates for mean(r2),
230  // acf(0), and acf(1).
231  Real initialGuess1(const Array &acf, Real mean_r2,
232  Real &alpha, Real &beta, Real &omega) {
233  Real A21 = acf[1];
234  Real A4 = acf[0] + mean_r2*mean_r2;
235 
236  Real A = mean_r2*mean_r2/A4; // 1/sigma^2
237  Real B = A21 / A4; // rho(1)
238 
239  Real gammaLower = A <= 1./3. - tol_level ? std::sqrt((1 - 3*A)/(3 - 3*A)) + tol_level : tol_level;
240  Garch11Constraint constraints(gammaLower, 1.0 - tol_level);
241 
242  Real gamma = gammaLower + (1 - gammaLower) * 0.5;
243  beta = std::min(gamma, std::max(gamma * (1 - A) - B, 0.0));
244  alpha = gamma - beta;
245  omega = mean_r2 * (1 - gamma);
246 
247  if (std::fabs(A-0.5) < QL_EPSILON) {
248  gamma = std::max(gammaLower, -(1+4*B*B)/(4*B));
249  beta = std::min(gamma, std::max(gamma * (1 - A) - B, 0.0));
250  alpha = gamma - beta;
251  omega = mean_r2 * (1 - gamma);
252  } else {
253  if (A > 1.0 - QL_EPSILON) {
254  gamma = std::max(gammaLower, -(1+B*B)/(2*B));
255  beta = std::min(gamma, std::max(gamma * (1 - A) - B, 0.0));
256  alpha = gamma - beta;
257  omega = mean_r2 * (1 - gamma);
258  } else {
259  Real D = (3*A-1)*(2*B*B+(1-A)*(2*A-1));
260  if (D >= 0) {
261  Real d = std::sqrt(D);
262  Real b = (B - d)/(2*A-1);
263  Real g = 0;
264  if (b >= tol_level && b <= 1.0 - tol_level) {
265  g = (b + B) / (1 - A);
266  }
267  if (g < gammaLower) {
268  b = (B + d)/(2*A-1);
269  if (b >= tol_level && b <= 1.0 - tol_level) {
270  g = (b + B) / (1 - A);
271  }
272  }
273  if (g >= gammaLower) {
274  gamma = g;
275  beta = std::min(gamma, std::max(gamma * (1 - A) - B, 0.0));
276  alpha = gamma - beta;
277  omega = mean_r2 * (1 - gamma);
278  }
279  }
280  }
281  }

284  std::size_t nCov = acf.size() - 1;
285  for (std::size_t i = 0; i <= nCov; ++i) {
286  if (i < 2 || (acf[i] > 0 && acf[i-1] > 0 && acf[i-1] > acf[i])) {
287  idx.push_back(i);
288  }
289  }

291  Array x(2);
292  x[0] = gamma;
293  x[1] = beta;

296  FitAcfConstraint c(gammaLower, 1.0 - tol_level);
297  NonLinearLeastSquare nnls(c);
298  nnls.setInitialValue(x);
299  FitAcfProblem pr(mean_r2, acf, idx);
300  x = nnls.perform(pr);
301  Array guess(3);
302  guess[0] = mean_r2 * (1 - x[0]);
303  guess[1] = x[0] - x[1];
304  guess[2] = x[1];
305  if (constraints.test(guess)) {
306  omega = guess[0];
307  alpha = guess[1];
308  beta = guess[2];
309  }
310  } catch (const std::exception &) {
311  // failed -- returning initial values
312  }
313  return gammaLower;
314  }

316  // Initial guess based on fitting ACF - initial guess for
317  // fitting acf is an estimate of gamma = alpfa+beta based on
318  // the property: acf(i+1) = gamma*acf(i) for i > 1.
319  Real initialGuess2 (const Array &acf, Real mean_r2,
320  Real &alpha, Real &beta, Real &omega) {
321  Real A21 = acf[1];
322  Real A4 = acf[0] + mean_r2*mean_r2;
323  Real A = mean_r2*mean_r2/A4; // 1/sigma^2
324  Real B = A21 / A4; // rho(1)
325  Real gammaLower = A <= 1./3. - tol_level ? std::sqrt((1 - 3*A)/(3 - 3*A)) + tol_level : tol_level;
326  Garch11Constraint constraints(gammaLower, 1.0 - tol_level);
327 
328  // ACF
329  Real gamma = 0;
330  std::size_t nn = 0;
331  std::vector<std::size_t> idx;
332  std::size_t nCov = acf.size() - 1;
333  for (std::size_t i = 0; i <= nCov; ++i) {
334  if (i < 2) idx.push_back(i);
335  if (i > 1 && acf[i] > 0 && acf[i-1] > 0 && acf[i-1] > acf[i]) {
336  gamma += acf[i]/acf[i-1];
337  nn++;
338  idx.push_back(i);
339  }
340  }
341  if (nn > 0)
342  gamma /= nn;
343  if (gamma < gammaLower) gamma = gammaLower;
344  beta = std::min(gamma, std::max(gamma * (1 - A) - B, 0.0));
345  omega = mean_r2 * (1 - gamma);
346 
347  Array x(2);
348  x[0] = gamma;
349  x[1] = beta;
350 
351  try {
352  FitAcfConstraint c(gammaLower, 1 - tol_level);
353  NonLinearLeastSquare nnls(c);
354  nnls.setInitialValue(x);
355  FitAcfProblem pr(mean_r2, acf, idx);
356  x = nnls.perform(pr);
357  Array guess(3);
358  guess[0] = mean_r2 * (1 - x[0]);
359  guess[1] = x[0] - x[1];
360  guess[2] = x[1];
361  if (constraints.test(guess)) {
362  omega = guess[0];
363  alpha = guess[1];
364  beta = guess[2];
365  }
366  } catch (const std::exception &) {
367  // failed -- returning initial values
368  }
369  return gammaLower;
370  }

374  Garch11::time_series
375  Garch11::calculate(const time_series& quoteSeries,
376  Real alpha, Real beta, Real omega) {
377  time_series retval;
378  const_iterator cur = quoteSeries.cbegin();
379  Real u = cur->second;
380  Real sigma2 = u*u;
381  while (++cur != quoteSeries.end()) {
382  sigma2 = omega + alpha * u * u + beta * sigma2;
383  retval[cur->first] = std::sqrt(sigma2);
384  u = cur->second;
385  }
386  sigma2 = omega + alpha * u * u + beta * sigma2;
387  --cur;
388  const_iterator prev = cur;
389  retval[cur->first + (cur->first - (--prev)->first) ] = std::sqrt(sigma2);
390  return retval;
391  }

403  ext::shared_ptr<Problem> Garch11::calibrate_r2(
404  Mode mode, const std::vector<Volatility> &r2, Real mean_r2,
405  OptimizationMethod &method, const EndCriteria &endCriteria,
406  Real &alpha, Real &beta, Real &omega) {
407  Real dataSize = Real(r2.size());
408  alpha = 0.0;
409  beta = 0.0;
410  omega = 0.0;
411  QL_REQUIRE (dataSize >= 4,
412  "Data series is too short to fit GARCH model");
413  QL_REQUIRE (mean_r2 > 0, "Data series is constant");
414  omega = mean_r2 * dataSize / (dataSize - 1);
415 
416  // ACF
417  Size maxLag = (Size)std::sqrt(dataSize);
418  Array acf(maxLag+1);
419  std::vector<Volatility> tmp(r2.size());
420  std::transform (r2.begin(), r2.end(), tmp.begin(),
421  subtract<Real>(mean_r2));
422  autocovariances (tmp.begin(), tmp.end(), acf.begin(), maxLag);
423  QL_REQUIRE (acf[0] > 0, "Data series is constant");
424 
425  Garch11CostFunction cost (r2);
426 
427  // two initial guesses based on fitting ACF
428  Real gammaLower = 0.0;
429  Array opt1(3);
430  Real fCost1 = QL_MAX_REAL;
431  if (mode != GammaGuess) {
432  gammaLower = initialGuess1(acf, mean_r2, opt1[1], opt1[2], opt1[0]);
433  fCost1 = cost.value(opt1);
434  }
435 
436  Array opt2(3);
437  Real fCost2 = QL_MAX_REAL;
438  if (mode != MomentMatchingGuess) {
439  gammaLower = initialGuess2(acf, mean_r2, opt2[1], opt2[2], opt2[0]);
440  fCost2 = cost.value(opt2);
441  }
442 
443  Garch11Constraint constraints(gammaLower, 1.0 - tol_level);
444 
445  ext::shared_ptr<Problem> ret;
446  if (mode != DoubleOptimization) {
447  try {
448  ret = calibrate_r2(r2, method, constraints, endCriteria,
449  fCost1 <= fCost2 ? opt1 : opt2,
450  alpha, beta, omega);
451  } catch (const std::exception &) {
452  if (fCost1 <= fCost2) {
453  alpha = opt1[1];
454  beta = opt1[2];
455  omega = opt1[0];
456  } else {
457  alpha = opt2[1];
458  beta = opt2[2];
459  omega = opt2[0];
460  }
461  }
462  } else {
463  ext::shared_ptr<Problem> ret1, ret2;
464  try {
465  ret1 = calibrate_r2(r2, method, constraints, endCriteria,
466  opt1, alpha, beta, omega);
467  opt1[1] = alpha;
468  opt1[2] = beta;
469  opt1[0] = omega;
470  if (constraints.test(opt1))
471  fCost1 = std::min(fCost1, cost.value(opt1));
472  } catch (const std::exception &) {
473  fCost1 = QL_MAX_REAL;
474  }
475 
476  try {
477  ret2 = calibrate_r2(r2, method, constraints, endCriteria,
478  opt2, alpha, beta, omega);
479  opt2[1] = alpha;
480  opt2[2] = beta;
481  opt2[0] = omega;
482  if (constraints.test(opt2))
483  fCost2 = std::min(fCost2, cost.value(opt2));
484  } catch (const std::exception &) {
485  fCost2 = QL_MAX_REAL;
486  }
487 
488  if (fCost1 <= fCost2) {
489  alpha = opt1[1];
490  beta = opt1[2];
491  omega = opt1[0];
492  ret = ret1;
493  } else {
494  alpha = opt2[1];
495  beta = opt2[2];
496  omega = opt2[0];
497  ret = ret2;
498  }
499  }
500  return ret;
501  }
