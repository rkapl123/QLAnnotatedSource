55  void FdExtOUJumpVanillaEngine::calculate() const {
56  // 1. Mesher
57  const Time maturity
58  = rTS_->dayCounter().yearFraction(rTS_->referenceDate(),
59  arguments_.exercise->lastDate());
60  const boost::shared_ptr<StochasticProcess1D> ouProcess(
61  process_->getExtendedOrnsteinUhlenbeckProcess());
62  const boost::shared_ptr<Fdm1dMesher> xMesher(
63  new FdmSimpleProcess1dMesher(xGrid_, ouProcess,maturity));
64 
65  const boost::shared_ptr<Fdm1dMesher> yMesher(
66  new ExponentialJump1dMesher(yGrid_,
67  process_->beta(),
68  process_->jumpIntensity(),
69  process_->eta()));
70 
71  const boost::shared_ptr<FdmMesher> mesher(
72  new FdmMesherComposite(xMesher, yMesher));
73 
74  // 2. Calculator
75  const boost::shared_ptr<FdmInnerValueCalculator> calculator(
76  new FdmExtOUJumpModelInnerValue(arguments_.payoff, mesher));
77 
78  // 3. Step conditions
79  const boost::shared_ptr<FdmStepConditionComposite> conditions =
80  FdmStepConditionComposite::vanillaComposite(
81  DividendSchedule(), arguments_.exercise,
82  mesher, calculator,
83  rTS_->referenceDate(), rTS_->dayCounter());
84 
85  // 4. Boundary conditions
86  const FdmBoundaryConditionSet boundaries;
87 
88  // 5. set-up solver
89  FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
90  calculator, maturity, tGrid_, 0 };
91 
92  const boost::shared_ptr<FdmExtOUJumpSolver> solver(
93  new FdmExtOUJumpSolver(Handle<ExtOUWithJumpsProcess>(process_),
94  rTS_, solverDesc, schemeDesc_));
95 
96  const Real x = process_->initialValues()[0];
97  const Real y = process_->initialValues()[1];
98  results_.value = solver->valueAt(x, y);
99  }
