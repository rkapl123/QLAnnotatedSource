57 Time Coupon::accruedPeriod(const Date& d) const {
58 if (d <= accrualStartDate_ || d > paymentDate_) {
59 return 0.0;
60 } else if (tradingExCoupon(d)) {
61 return -dayCounter().yearFraction(d, std::max(d, accrualEndDate_),
62 refPeriodStart_, refPeriodEnd_);
63 } else {
64 return dayCounter().yearFraction(accrualStartDate_,
65 std::min(d, accrualEndDate_),
66 refPeriodStart_,
67 refPeriodEnd_);
68 }
69 }
