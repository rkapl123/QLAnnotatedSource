75  void FdOrnsteinUhlenbeckVanillaEngine::calculate() const {
76 
77  // 1. Mesher
78  const ext::shared_ptr<StrikedTypePayoff> payoff =
79  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
80 
81  const DayCounter dc = rTS_->dayCounter();
82  const Date referenceDate = rTS_->referenceDate();
83 
84  const Time maturity = dc.yearFraction(
85  referenceDate, arguments_.exercise->lastDate());
86 
87  const ext::shared_ptr<Fdm1dMesher> equityMesher(
88  new FdmSimpleProcess1dMesher(
89  xGrid_, process_, maturity, 1, epsilon_));
90 
91  const ext::shared_ptr<FdmMesher> mesher (
92  new FdmMesherComposite(equityMesher));
93 
94  // 2. Calculator
95  const ext::shared_ptr<FdmInnerValueCalculator> calculator(
96  new FdmOUInnerValue(payoff, mesher, 0));
97 
98  // 3. Step conditions
99  const ext::shared_ptr<FdmStepConditionComposite> conditions =
100  FdmStepConditionComposite::vanillaComposite(
101  arguments_.cashFlow, arguments_.exercise,
102  mesher, calculator,
103  referenceDate, dc);
104 
105  // 4. Boundary conditions
106  const FdmBoundaryConditionSet boundaries;
107 
108  // 5. Solver
109  FdmSolverDesc solverDesc = { mesher, boundaries, conditions, calculator,
110  maturity, tGrid_, dampingSteps_ };
111 
112  const ext::shared_ptr<FdmOrnsteinUhlenbeckOp> op(
113  new FdmOrnsteinUhlenbeckOp(mesher, process_, rTS_, 0));
114 
115  const ext::shared_ptr<Fdm1DimSolver> solver(
116  new Fdm1DimSolver(solverDesc, schemeDesc_, op));
117 
118  const Real spot = process_->x0();
119 
120  results_.value = solver->interpolateAt(spot);
121  results_.delta = solver->derivativeX(spot);
122  results_.gamma = solver->derivativeXX(spot);
123  results_.theta = solver->thetaAt(spot);
124  }
125 }
