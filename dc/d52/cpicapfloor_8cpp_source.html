36  CPICapFloor::CPICapFloor(Option::Type type,
37  Real nominal,
38  const Date& startDate, // start date of contract (only)
39  Real baseCPI,
40  const Date& maturity, // this is pre-adjustment!
41  const Calendar& fixCalendar,
42  BusinessDayConvention fixConvention,
43  const Calendar& payCalendar,
44  BusinessDayConvention payConvention,
45  Rate strike,
46  const Handle<ZeroInflationIndex> &infIndex,
47  const Period& observationLag,
48  CPI::InterpolationType observationInterpolation)
49  : type_(type), nominal_(nominal), startDate_(startDate), baseCPI_(baseCPI),
50  maturity_(maturity), fixCalendar_(fixCalendar), fixConvention_(fixConvention),
51  payCalendar_(payCalendar), payConvention_(payConvention),
52  strike_(strike), infIndex_(infIndex), observationLag_(observationLag),
53  observationInterpolation_(observationInterpolation)
54  {
55  QL_REQUIRE(fixCalendar_ != Calendar(),"CPICapFloor: fixing calendar may not be null.");
56  QL_REQUIRE(payCalendar_ != Calendar(),"CPICapFloor: payment calendar may not be null.");
57 
58  if (observationInterpolation_ == CPI::Flat ||
59  (observationInterpolation_ == CPI::AsIndex && !infIndex_->interpolated())
60  ) {
61  QL_REQUIRE(observationLag_ >= infIndex_->availabilityLag(),
62  "CPIcapfloor's observationLag must be at least availabilityLag of inflation index: "
63  <<"when the observation is effectively flat"
64  << observationLag_ << " vs " << infIndex_->availabilityLag());
65  }
66  if (observationInterpolation_ == CPI::Linear ||
67  (observationInterpolation_ == CPI::AsIndex && infIndex_->interpolated())
68  ) {
69  QL_REQUIRE(observationLag_ > infIndex_->availabilityLag(),
70  "CPIcapfloor's observationLag must be greater than availabilityLag of inflation index: "
71  <<"when the observation is effectively linear"
72  << observationLag_ << " vs " << infIndex_->availabilityLag());
73  }
74  }
