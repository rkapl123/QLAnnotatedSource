66 CPICapFloor::CPICapFloor(Option::Type type,
67 Real nominal,
68 const Date& startDate, // start date of contract (only)
69 Real baseCPI,
70 const Date& maturity, // this is pre-adjustment!
71 Calendar fixCalendar,
72 BusinessDayConvention fixConvention,
73 Calendar payCalendar,
74 BusinessDayConvention payConvention,
75 Rate strike,
76 const Handle<ZeroInflationIndex>& infIndex,
77 const Period& observationLag,
78 CPI::InterpolationType observationInterpolation)
79 : type_(type), nominal_(nominal), startDate_(startDate), baseCPI_(baseCPI), maturity_(maturity),
80 fixCalendar_(std::move(fixCalendar)), fixConvention_(fixConvention),
81 payCalendar_(std::move(payCalendar)), payConvention_(payConvention), strike_(strike),
82 index_(*infIndex), observationLag_(observationLag),
83 observationInterpolation_(observationInterpolation), infIndex_(infIndex) {
84 QL_REQUIRE(index_, "no inflation index passed");
85 QL_REQUIRE(fixCalendar_ != Calendar(), "no fixing calendar passed");
86 QL_REQUIRE(payCalendar_ != Calendar(), "no payment calendar passed");
87
88 if (!detail::CPI::isInterpolated(index_, observationInterpolation_)) {
89 QL_REQUIRE(observationLag_ >= index_->availabilityLag(),
90 "CPIcapfloor's observationLag must be at least availabilityLag of inflation index: "
91 <<"when the observation is effectively flat"
92 << observationLag_ << " vs " << index_->availabilityLag());
93 } else {
94 QL_REQUIRE(observationLag_ > index_->availabilityLag(),
95 "CPIcapfloor's observationLag must be greater than availabilityLag of inflation index: "
96 <<"when the observation is effectively linear"
97 << observationLag_ << " vs " << index_->availabilityLag());
98 }
99 }
