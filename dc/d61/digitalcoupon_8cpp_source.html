40  underlying->accrualStartDate(),

52  isCallCashOrNothing_(false), isPutCashOrNothing_(false),
53  callLeftEps_(replication->gap()/2.), callRightEps_(replication->gap()/2.),
54  putLeftEps_(replication->gap()/2.), putRightEps_(replication->gap()/2.),
55  hasPutStrike_(false), hasCallStrike_(false),
56  replicationType_(replication->replicationType()) {
57 
58  QL_REQUIRE(replication->gap()>0.0, "Non positive epsilon not allowed");
59 
60  if (putStrike == Null<Rate>()) {
61  QL_REQUIRE(putDigitalPayoff == Null<Rate>(),
62  "Put Cash rate non allowed if put strike is null");
63  }
64  if (callStrike == Null<Rate>()) {
65  QL_REQUIRE(callDigitalPayoff == Null<Rate>(),
66  "Call Cash rate non allowed if call strike is null");
67  }
68  if (callStrike != Null<Rate>()) {
69  QL_REQUIRE(callStrike >= 0., "negative call strike not allowed");
70  hasCallStrike_ = true;
71  callStrike_ = callStrike;
72  QL_REQUIRE(callStrike_>=replication->gap()/2., "call strike < eps/2");
73  switch (callPosition) {
74  case Position::Long :
75  callCsi_ = 1.0;
76  break;
77  case Position::Short :
78  callCsi_ = -1.0;
79  break;
80  default:
81  QL_FAIL("unsupported position type");
82  }
83  if (callDigitalPayoff != Null<Rate>()){
84  callDigitalPayoff_ = callDigitalPayoff;
85  isCallCashOrNothing_ = true;
86  }
87  }
88  if (putStrike != Null<Rate>()){
89  QL_REQUIRE(putStrike >= 0., "negative put strike not allowed");
90  hasPutStrike_ = true;
91  putStrike_ = putStrike;
92  switch (putPosition) {
93  case Position::Long :
94  putCsi_ = 1.0;
95  break;
96  case Position::Short :
97  putCsi_ = -1.0;
98  break;
99  default:
100  QL_FAIL("unsupported position type");
101  }
102  if (putDigitalPayoff != Null<Rate>()){
103  putDigitalPayoff_ = putDigitalPayoff;
104  isPutCashOrNothing_ = true;
105  }
106  }
107 
108  switch (replicationType_) {
109  case Replication::Central :
110  // do nothing
111  break;
112  case Replication::Sub :
113  if (hasCallStrike_) {
114  switch (callPosition) {
115  case Position::Long :
116  callLeftEps_ = 0.;
117  callRightEps_ = replication->gap();
118  break;
119  case Position::Short :
120  callLeftEps_ = replication->gap();
121  callRightEps_ = 0.;
122  break;
123  default:
124  QL_FAIL("unsupported position type");
125  }
126  }
127  if (hasPutStrike_) {
128  switch (putPosition) {
129  case Position::Long :
130  putLeftEps_ = replication->gap();
131  putRightEps_ = 0.;
132  break;
133  case Position::Short :
134  putLeftEps_ = 0.;
135  putRightEps_ = replication->gap();
136  break;
137  default:
138  QL_FAIL("unsupported position type");
139  }
140  }
141  break;
142  case Replication::Super :
143  if (hasCallStrike_) {
144  switch (callPosition) {
145  case Position::Long :
146  callLeftEps_ = replication->gap();
147  callRightEps_ = 0.;
148  break;
149  case Position::Short :
150  callLeftEps_ = 0.;
151  callRightEps_ = replication->gap();
152  break;
153  default:
154  QL_FAIL("unsupported position type");
155  }
156  }
157  if (hasPutStrike_) {
158  switch (putPosition) {
159  case Position::Long :
160  putLeftEps_ = 0.;
161  putRightEps_ = replication->gap();
162  break;
163  case Position::Short :
164  putLeftEps_ = replication->gap();
165  putRightEps_ = 0.;
166  break;
167  default:
168  QL_FAIL("unsupported position type");
169  }
170  }
171  break;
172  default:
173  QL_FAIL("unsupported replication type");
174  }
175 
176  registerWith(underlying);

180  Rate DigitalCoupon::callOptionRate() const {
181 
182  Rate callOptionRate = Rate(0.);
183  if(hasCallStrike_) {
184  // Step function
185  callOptionRate = isCallCashOrNothing_ ? callDigitalPayoff_ : callStrike_;
186  CappedFlooredCoupon next(underlying_, callStrike_ + callRightEps_);
187  CappedFlooredCoupon previous(underlying_, callStrike_ - callLeftEps_);
188  callOptionRate *= (next.rate() - previous.rate())
189  / (callLeftEps_ + callRightEps_);
190  if (!isCallCashOrNothing_) {
191  // Call
192  CappedFlooredCoupon atStrike(underlying_, callStrike_);
193  Rate call = underlying_->rate() - atStrike.rate();
194  // Sum up
195  callOptionRate += call;
196  }
197  }
198  return callOptionRate;
199  }

201  Rate DigitalCoupon::putOptionRate() const {
202 
203  Rate putOptionRate = Rate(0.);
204  if(hasPutStrike_) {
205  // Step function
206  putOptionRate = isPutCashOrNothing_ ? putDigitalPayoff_ : putStrike_;
207  CappedFlooredCoupon next(underlying_, Null<Rate>(), putStrike_ + putRightEps_);
208  CappedFlooredCoupon previous(underlying_, Null<Rate>(), putStrike_ - putLeftEps_);
209  putOptionRate *= (next.rate() - previous.rate())
210  / (putLeftEps_ + putRightEps_);
211  if (!isPutCashOrNothing_) {
212  // Put
213  CappedFlooredCoupon atStrike(underlying_, Null<Rate>(), putStrike_);
214  Rate put = - underlying_->rate() + atStrike.rate();
215  // Sum up
216  putOptionRate -= put;
217  }
218  }
219  return putOptionRate;
220  }

222  Rate DigitalCoupon::rate() const {
223 
224  QL_REQUIRE(underlying_->pricer(), "pricer not set");
225 
226  Date fixingDate = underlying_->fixingDate();
227  Date today = Settings::instance().evaluationDate();
228  bool enforceTodaysHistoricFixings =
229  Settings::instance().enforcesTodaysHistoricFixings();
230  Rate underlyingRate = underlying_->rate();
231  if (fixingDate < today ||
232  ((fixingDate == today) && enforceTodaysHistoricFixings)) {
233  // must have been fixed
234  return underlyingRate + callCsi_ * callPayoff() + putCsi_ * putPayoff();
235  }
236  if (fixingDate == today) {
237  // might have been fixed
238  Rate pastFixing =
239  IndexManager::instance().getHistory((underlying_->index())->name())[fixingDate];
240  if (pastFixing != Null<Real>()) {
241  return underlyingRate + callCsi_ * callPayoff() + putCsi_ * putPayoff();
242  } else
243  return underlyingRate + callCsi_ * callOptionRate() + putCsi_ * putOptionRate();
244  }
245  return underlyingRate + callCsi_ * callOptionRate() + putCsi_ * putOptionRate();
246  }

294  Rate DigitalCoupon::callPayoff() const {
295  // to use only if index has fixed
296  Rate payoff(0.);
297  if(hasCallStrike_) {
298  Rate underlyingRate = underlying_->rate();
299  if ( (underlyingRate - callStrike_) > 1.e-16 ) {
300  payoff = isCallCashOrNothing_ ? callDigitalPayoff_ : underlyingRate;
301  } else {
302  if (isCallATMIncluded_) {
303  if ( std::abs(callStrike_ - underlyingRate) <= 1.e-16 )
304  payoff = isCallCashOrNothing_ ? callDigitalPayoff_ : underlyingRate;
305  }
306  }
307  }
308  return payoff;
309  }
