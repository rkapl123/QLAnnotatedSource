96  template <class ModelType> inline
97  Real FdmAffineModelSwapInnerValue<ModelType>::innerValue(
98  const FdmLinearOpIterator& iter, Time t) {
99 
100  const Date& iterExerciseDate = exerciseDates_.find(t)->second;
101 
102  const Array disRate(getState(disModel_, t, iter));
103  const Array fwdRate(getState(fwdModel_, t, iter));
104 
105  if (disTs_.empty() || iterExerciseDate != disTs_->referenceDate()) {
106 
107  const Handle<YieldTermStructure> discount
108  = disModel_->termStructure();
109 
110  disTs_.linkTo(ext::shared_ptr<YieldTermStructure>(
111  new FdmAffineModelTermStructure(disRate,
112  discount->calendar(), discount->dayCounter(),
113  iterExerciseDate, discount->referenceDate(),
114  disModel_)));
115 
116  const Handle<YieldTermStructure> fwd = fwdModel_->termStructure();
117 
118  fwdTs_.linkTo(ext::shared_ptr<YieldTermStructure>(
119  new FdmAffineModelTermStructure(fwdRate,
120  fwd->calendar(), fwd->dayCounter(),
121  iterExerciseDate, fwd->referenceDate(),
122  fwdModel_)));
123 
124  }
125  else {
126  ext::dynamic_pointer_cast<FdmAffineModelTermStructure>(
127  disTs_.currentLink())->setVariable(disRate);
128  ext::dynamic_pointer_cast<FdmAffineModelTermStructure>(
129  fwdTs_.currentLink())->setVariable(fwdRate);
130  }
131 
132  Real npv = 0.0;
133  for (Size j = 0; j < 2; j++) {
134  for (Leg::const_iterator i = swap_->leg(j).begin();
135  i != swap_->leg(j).end(); ++i) {
136  npv += ext::dynamic_pointer_cast<Coupon>(*i)
137  ->accrualStartDate() >= iterExerciseDate
138  ? (*i)->amount() * disTs_->discount((*i)->date())
139  : 0.0;
140  }
141  if (j == 0)
142  npv *= -1.0;
143  }
144  if (swap_->type() == VanillaSwap::Receiver)
145  npv *= -1.0;
146 
147  return std::max(0.0, npv);
148  }
