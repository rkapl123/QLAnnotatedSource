54  ImpliedSwaptionVolHelper::ImpliedSwaptionVolHelper(
55  const Swaption& swaption,
56  const Handle<YieldTermStructure>& discountCurve,
57  Real targetValue,
58  Real displacement,
59  VolatilityType type)
60  : discountCurve_(discountCurve), targetValue_(targetValue) {
61 
62  // set an implausible value, so that calculation is forced
63  // at first ImpliedSwaptionVolHelper::operator()(Volatility x) call
64  vol_ = boost::shared_ptr<SimpleQuote>(new SimpleQuote(-1.0));
65  Handle<Quote> h(vol_);
66 
67  switch (type) {
68  case ShiftedLognormal:
69  engine_ = boost::make_shared<BlackSwaptionEngine>(
70  discountCurve_, h, Actual365Fixed(), displacement);
71  break;
72  case Normal:
73  engine_ = boost::make_shared<BachelierSwaptionEngine>(
74  discountCurve_, h, Actual365Fixed());
75  break;
76  default:
77  QL_FAIL("unknown VolatilityType (" << type << ")");
78  break;
79  }
80  swaption.setupArguments(engine_->getArguments());
81  results_ = dynamic_cast<const Instrument::results *>(
82  engine_->getResults());
83  }

85  Real ImpliedSwaptionVolHelper::operator()(Volatility x) const {
86  if (x!=vol_->value()) {
87  vol_->setValue(x);
88  engine_->calculate();
89  }
90  return results_->value-targetValue_;
91  }

93  Real ImpliedSwaptionVolHelper::derivative(Volatility x) const {
94  if (x!=vol_->value()) {
95  vol_->setValue(x);
96  engine_->calculate();
97  }
98  std::map<std::string,boost::any>::const_iterator vega_ =
99  results_->additionalResults.find("vega");
100  QL_REQUIRE(vega_ != results_->additionalResults.end(),
101  "vega not provided");
102  return boost::any_cast<Real>(vega_->second);
103  }
104  }
