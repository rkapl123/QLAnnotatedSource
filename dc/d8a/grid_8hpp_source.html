37  inline Disposable<Array> CenteredGrid(Real center, Real dx,
38  Size steps) {
39  Array result(steps+1);
40  for (Size i=0; i<steps+1; i++)
41  result[i] = center + (i - steps/2.0)*dx;
42  return result;
43  }
44 
45  inline Disposable<Array> BoundedGrid(Real xMin, Real xMax,
46  Size steps) {
47  Array result(steps+1);
48  Real x=xMin, dx=(xMax-xMin)/steps;
49  for (Size i=0; i<steps+1; i++, x+=dx)
50  result[i] = x;
51  return result;
52  }

54  inline Disposable<Array> BoundedLogGrid(Real xMin, Real xMax,
55  Size steps) {
56  Array result(steps+1);
57  Real gridLogSpacing = (std::log(xMax) - std::log(xMin)) /
58  (steps);
59  Real edx = std::exp(gridLogSpacing);
60  result[0] = xMin;
61  for (Size j=1; j < steps+1; j++) {
62  result[j] = result[j-1]*edx;
63  }
64  return result;
65  }
66 }
