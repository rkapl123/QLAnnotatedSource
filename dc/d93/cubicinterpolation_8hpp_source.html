155  template <class I1, class I2>
156  CubicInterpolation(const I1& xBegin,
157  const I1& xEnd,

159  CubicInterpolation::DerivativeApprox da,

161  CubicInterpolation::BoundaryCondition leftCond,
162  Real leftConditionValue,
163  CubicInterpolation::BoundaryCondition rightCond,
164  Real rightConditionValue) {
165  impl_ = boost::shared_ptr<Interpolation::Impl>(new

167  da,
168  monotonic,
169  leftCond,
170  leftConditionValue,
171  rightCond,
172  rightConditionValue));
173  impl_->update();
174  coeffs_ =
175  boost::dynamic_pointer_cast<detail::CoefficientHolder>(impl_);
176  }

178  return coeffs_->primitiveConst_;
179  }
180  const std::vector<Real>& aCoefficients() const { return coeffs_->a_; }

182  const std::vector<Real>& cCoefficients() const { return coeffs_->c_; }
183  const std::vector<bool>& monotonicityAdjustments() const {
184  return coeffs_->monotonicityAdjustments_;
185  }
186  private:
187  boost::shared_ptr<detail::CoefficientHolder> coeffs_;
188  };

360  template <class I1, class I2>
361  class CubicInterpolationImpl : public CoefficientHolder,
362  public Interpolation::templateImpl<I1,I2> {
363  public:
364  CubicInterpolationImpl(const I1& xBegin,
365  const I1& xEnd,
366  const I2& yBegin,
367  CubicInterpolation::DerivativeApprox da,
368  bool monotonic,
369  CubicInterpolation::BoundaryCondition leftCondition,
370  Real leftConditionValue,
371  CubicInterpolation::BoundaryCondition rightCondition,
372  Real rightConditionValue)
373  : CoefficientHolder(xEnd-xBegin),
374  Interpolation::templateImpl<I1,I2>(xBegin, xEnd, yBegin,
375  Cubic::requiredPoints),
376  da_(da),
377  monotonic_(monotonic),
378  leftType_(leftCondition), rightType_(rightCondition),
379  leftValue_(leftConditionValue),
380  rightValue_(rightConditionValue),
381  tmp_(n_), dx_(n_-1), S_(n_-1), L_(n_) {
382  if (leftType_ == CubicInterpolation::Lagrange
383  || rightType_ == CubicInterpolation::Lagrange) {
384  QL_REQUIRE((xEnd-xBegin) >= 4,
385  "Lagrange boundary condition requires at least "
386  "4 points (" << (xEnd-xBegin) << " are given)");
387  }
388  }

390  void update() {
391 
392  for (Size i=0; i<n_-1; ++i) {
393  dx_[i] = this->xBegin_[i+1] - this->xBegin_[i];
394  S_[i] = (this->yBegin_[i+1] - this->yBegin_[i])/dx_[i];
395  }
396 
397  // first derivative approximation
398  if (da_==CubicInterpolation::Spline) {
399  for (Size i=1; i<n_-1; ++i) {
400  L_.setMidRow(i, dx_[i], 2.0*(dx_[i]+dx_[i-1]), dx_[i-1]);
401  tmp_[i] = 3.0*(dx_[i]*S_[i-1] + dx_[i-1]*S_[i]);
402  }
403 
404  // left boundary condition
405  switch (leftType_) {
406  case CubicInterpolation::NotAKnot:
407  // ignoring end condition value
408  L_.setFirstRow(dx_[1]*(dx_[1]+dx_[0]),
409  (dx_[0]+dx_[1])*(dx_[0]+dx_[1]));
410  tmp_[0] = S_[0]*dx_[1]*(2.0*dx_[1]+3.0*dx_[0]) +
411  S_[1]*dx_[0]*dx_[0];
412  break;
413  case CubicInterpolation::FirstDerivative:
414  L_.setFirstRow(1.0, 0.0);
415  tmp_[0] = leftValue_;
416  break;
417  case CubicInterpolation::SecondDerivative:
418  L_.setFirstRow(2.0, 1.0);
419  tmp_[0] = 3.0*S_[0] - leftValue_*dx_[0]/2.0;
420  break;
421  case CubicInterpolation::Periodic:
422  QL_FAIL("this end condition is not implemented yet");
423  case CubicInterpolation::Lagrange:
424  L_.setFirstRow(1.0, 0.0);
425  tmp_[0] = cubicInterpolatingPolynomialDerivative(
426  this->xBegin_[0],this->xBegin_[1],
427  this->xBegin_[2],this->xBegin_[3],
428  this->yBegin_[0],this->yBegin_[1],
429  this->yBegin_[2],this->yBegin_[3],
430  this->xBegin_[0]);
431  break;
432  default:
433  QL_FAIL("unknown end condition");
434  }
435 
436  // right boundary condition
437  switch (rightType_) {
438  case CubicInterpolation::NotAKnot:
439  // ignoring end condition value
440  L_.setLastRow(-(dx_[n_-2]+dx_[n_-3])*(dx_[n_-2]+dx_[n_-3]),
441  -dx_[n_-3]*(dx_[n_-3]+dx_[n_-2]));
442  tmp_[n_-1] = -S_[n_-3]*dx_[n_-2]*dx_[n_-2] -
443  S_[n_-2]*dx_[n_-3]*(3.0*dx_[n_-2]+2.0*dx_[n_-3]);
444  break;
445  case CubicInterpolation::FirstDerivative:
446  L_.setLastRow(0.0, 1.0);
447  tmp_[n_-1] = rightValue_;
448  break;
449  case CubicInterpolation::SecondDerivative:
450  L_.setLastRow(1.0, 2.0);
451  tmp_[n_-1] = 3.0*S_[n_-2] + rightValue_*dx_[n_-2]/2.0;
452  break;
453  case CubicInterpolation::Periodic:
454  QL_FAIL("this end condition is not implemented yet");
455  case CubicInterpolation::Lagrange:
456  L_.setLastRow(0.0,1.0);
457  tmp_[n_-1] = cubicInterpolatingPolynomialDerivative(
458  this->xBegin_[n_-4],this->xBegin_[n_-3],
459  this->xBegin_[n_-2],this->xBegin_[n_-1],
460  this->yBegin_[n_-4],this->yBegin_[n_-3],
461  this->yBegin_[n_-2],this->yBegin_[n_-1],
462  this->xBegin_[n_-1]);
463  break;
464  default:
465  QL_FAIL("unknown end condition");
466  }
467 
468  // solve the system
469  L_.solveFor(tmp_, tmp_);
470  } else if (da_==CubicInterpolation::SplineOM1) {
471  Matrix T_(n_-2, n_, 0.0);
472  for (Size i=0; i<n_-2; ++i) {
473  T_[i][i]=dx_[i]/6.0;
474  T_[i][i+1]=(dx_[i+1]+dx_[i])/3.0;
475  T_[i][i+2]=dx_[i+1]/6.0;
476  }
477  Matrix S_(n_-2, n_, 0.0);
478  for (Size i=0; i<n_-2; ++i) {
479  S_[i][i]=1.0/dx_[i];
480  S_[i][i+1]=-(1.0/dx_[i+1]+1.0/dx_[i]);
481  S_[i][i+2]=1.0/dx_[i+1];
482  }
483  Matrix Up_(n_, 2, 0.0);
484  Up_[0][0]=1;
485  Up_[n_-1][1]=1;
486  Matrix Us_(n_, n_-2, 0.0);
487  for (Size i=0; i<n_-2; ++i)
488  Us_[i+1][i]=1;
489  Matrix Z_ = Us_*inverse(T_*Us_);
490  Matrix I_(n_, n_, 0.0);
491  for (Size i=0; i<n_; ++i)
492  I_[i][i]=1;
493  Matrix V_ = (I_-Z_*T_)*Up_;
494  Matrix W_ = Z_*S_;
495  Matrix Q_(n_, n_, 0.0);
496  Q_[0][0]=1.0/(n_-1)*dx_[0]*dx_[0]*dx_[0];
497  Q_[0][1]=7.0/8*1.0/(n_-1)*dx_[0]*dx_[0]*dx_[0];
498  for (Size i=1; i<n_-1; ++i) {
499  Q_[i][i-1]=7.0/8*1.0/(n_-1)*dx_[i-1]*dx_[i-1]*dx_[i-1];
500  Q_[i][i]=1.0/(n_-1)*dx_[i]*dx_[i]*dx_[i]+1.0/(n_-1)*dx_[i-1]*dx_[i-1]*dx_[i-1];
501  Q_[i][i+1]=7.0/8*1.0/(n_-1)*dx_[i]*dx_[i]*dx_[i];
502  }
503  Q_[n_-1][n_-2]=7.0/8*1.0/(n_-1)*dx_[n_-2]*dx_[n_-2]*dx_[n_-2];
504  Q_[n_-1][n_-1]=1.0/(n_-1)*dx_[n_-2]*dx_[n_-2]*dx_[n_-2];
505  Matrix J_ = (I_-V_*inverse(transpose(V_)*Q_*V_)*transpose(V_)*Q_)*W_;
506  Array Y_(n_);
507  for (Size i=0; i<n_; ++i)
508  Y_[i]=this->yBegin_[i];
509  Array D_ = J_*Y_;
510  for (Size i=0; i<n_-1; ++i)
511  tmp_[i]=(Y_[i+1]-Y_[i])/dx_[i]-(2.0*D_[i]+D_[i+1])*dx_[i]/6.0;
512  tmp_[n_-1]=tmp_[n_-2]+D_[n_-2]*dx_[n_-2]+(D_[n_-1]-D_[n_-2])*dx_[n_-2]/2.0;
513 
514  } else if (da_==CubicInterpolation::SplineOM2) {
515  Matrix T_(n_-2, n_, 0.0);
516  for (Size i=0; i<n_-2; ++i) {
517  T_[i][i]=dx_[i]/6.0;
518  T_[i][i+1]=(dx_[i]+dx_[i+1])/3.0;
519  T_[i][i+2]=dx_[i+1]/6.0;
520  }
521  Matrix S_(n_-2, n_, 0.0);
522  for (Size i=0; i<n_-2; ++i) {
523  S_[i][i]=1.0/dx_[i];
524  S_[i][i+1]=-(1.0/dx_[i+1]+1.0/dx_[i]);
525  S_[i][i+2]=1.0/dx_[i+1];
526  }
527  Matrix Up_(n_, 2, 0.0);
528  Up_[0][0]=1;
529  Up_[n_-1][1]=1;
530  Matrix Us_(n_, n_-2, 0.0);
531  for (Size i=0; i<n_-2; ++i)
532  Us_[i+1][i]=1;
533  Matrix Z_ = Us_*inverse(T_*Us_);
534  Matrix I_(n_, n_, 0.0);
535  for (Size i=0; i<n_; ++i)
536  I_[i][i]=1;
537  Matrix V_ = (I_-Z_*T_)*Up_;
538  Matrix W_ = Z_*S_;
539  Matrix Q_(n_, n_, 0.0);
540  Q_[0][0]=1.0/(n_-1)*dx_[0];
541  Q_[0][1]=1.0/2*1.0/(n_-1)*dx_[0];
542  for (Size i=1; i<n_-1; ++i) {
543  Q_[i][i-1]=1.0/2*1.0/(n_-1)*dx_[i-1];
544  Q_[i][i]=1.0/(n_-1)*dx_[i]+1.0/(n_-1)*dx_[i-1];
545  Q_[i][i+1]=1.0/2*1.0/(n_-1)*dx_[i];
546  }
547  Q_[n_-1][n_-2]=1.0/2*1.0/(n_-1)*dx_[n_-2];
548  Q_[n_-1][n_-1]=1.0/(n_-1)*dx_[n_-2];
549  Matrix J_ = (I_-V_*inverse(transpose(V_)*Q_*V_)*transpose(V_)*Q_)*W_;
550  Array Y_(n_);
551  for (Size i=0; i<n_; ++i)
552  Y_[i]=this->yBegin_[i];
553  Array D_ = J_*Y_;
554  for (Size i=0; i<n_-1; ++i)
555  tmp_[i]=(Y_[i+1]-Y_[i])/dx_[i]-(2.0*D_[i]+D_[i+1])*dx_[i]/6.0;
556  tmp_[n_-1]=tmp_[n_-2]+D_[n_-2]*dx_[n_-2]+(D_[n_-1]-D_[n_-2])*dx_[n_-2]/2.0;
557  } else { // local schemes
558  if (n_==2)
559  tmp_[0] = tmp_[1] = S_[0];
560  else {
561  switch (da_) {
562  case CubicInterpolation::FourthOrder:
563  QL_FAIL("FourthOrder not implemented yet");
564  break;
565  case CubicInterpolation::Parabolic:
566  // intermediate points
567  for (Size i=1; i<n_-1; ++i)
568  tmp_[i] = (dx_[i-1]*S_[i]+dx_[i]*S_[i-1])/(dx_[i]+dx_[i-1]);
569  // end points
570  tmp_[0] = ((2.0*dx_[ 0]+dx_[ 1])*S_[ 0] - dx_[ 0]*S_[ 1]) / (dx_[ 0]+dx_[ 1]);
571  tmp_[n_-1] = ((2.0*dx_[n_-2]+dx_[n_-3])*S_[n_-2] - dx_[n_-2]*S_[n_-3]) / (dx_[n_-2]+dx_[n_-3]);
572  break;
573  case CubicInterpolation::FritschButland:
574  // intermediate points
575  for (Size i=1; i<n_-1; ++i) {
576  Real Smin = std::min(S_[i-1], S_[i]);
577  Real Smax = std::max(S_[i-1], S_[i]);
578  if(Smax+2.0*Smin == 0){
579  if (Smin*Smax < 0)
580  tmp_[i] = QL_MIN_REAL;
581  else if (Smin*Smax == 0)
582  tmp_[i] = 0;
583  else
584  tmp_[i] = QL_MAX_REAL;
585  }
586  else
587  tmp_[i] = 3.0*Smin*Smax/(Smax+2.0*Smin);
588  }
589  // end points
590  tmp_[0] = ((2.0*dx_[ 0]+dx_[ 1])*S_[ 0] - dx_[ 0]*S_[ 1]) / (dx_[ 0]+dx_[ 1]);
591  tmp_[n_-1] = ((2.0*dx_[n_-2]+dx_[n_-3])*S_[n_-2] - dx_[n_-2]*S_[n_-3]) / (dx_[n_-2]+dx_[n_-3]);
592  break;
593  case CubicInterpolation::Akima:
594  tmp_[0] = (std::abs(S_[1]-S_[0])*2*S_[0]*S_[1]+std::abs(2*S_[0]*S_[1]-4*S_[0]*S_[0]*S_[1])*S_[0])/(std::abs(S_[1]-S_[0])+std::abs(2*S_[0]*S_[1]-4*S_[0]*S_[0]*S_[1]));
595  tmp_[1] = (std::abs(S_[2]-S_[1])*S_[0]+std::abs(S_[0]-2*S_[0]*S_[1])*S_[1])/(std::abs(S_[2]-S_[1])+std::abs(S_[0]-2*S_[0]*S_[1]));
596  for (Size i=2; i<n_-2; ++i) {
597  if ((S_[i-2]==S_[i-1]) && (S_[i]!=S_[i+1]))
598  tmp_[i] = S_[i-1];
599  else if ((S_[i-2]!=S_[i-1]) && (S_[i]==S_[i+1]))
600  tmp_[i] = S_[i];
601  else if (S_[i]==S_[i-1])
602  tmp_[i] = S_[i];
603  else if ((S_[i-2]==S_[i-1]) && (S_[i-1]!=S_[i]) && (S_[i]==S_[i+1]))
604  tmp_[i] = (S_[i-1]+S_[i])/2.0;
605  else
606  tmp_[i] = (std::abs(S_[i+1]-S_[i])*S_[i-1]+std::abs(S_[i-1]-S_[i-2])*S_[i])/(std::abs(S_[i+1]-S_[i])+std::abs(S_[i-1]-S_[i-2]));
607  }
608  tmp_[n_-2] = (std::abs(2*S_[n_-2]*S_[n_-3]-S_[n_-2])*S_[n_-3]+std::abs(S_[n_-3]-S_[n_-4])*S_[n_-2])/(std::abs(2*S_[n_-2]*S_[n_-3]-S_[n_-2])+std::abs(S_[n_-3]-S_[n_-4]));
609  tmp_[n_-1] = (std::abs(4*S_[n_-2]*S_[n_-2]*S_[n_-3]-2*S_[n_-2]*S_[n_-3])*S_[n_-2]+std::abs(S_[n_-2]-S_[n_-3])*2*S_[n_-2]*S_[n_-3])/(std::abs(4*S_[n_-2]*S_[n_-2]*S_[n_-3]-2*S_[n_-2]*S_[n_-3])+std::abs(S_[n_-2]-S_[n_-3]));
610  break;
611  case CubicInterpolation::Kruger:
612  // intermediate points
613  for (Size i=1; i<n_-1; ++i) {
614  if (S_[i-1]*S_[i]<0.0)
615  // slope changes sign at point
616  tmp_[i] = 0.0;
617  else
618  // slope will be between the slopes of the adjacent
619  // straight lines and should approach zero if the
620  // slope of either line approaches zero
621  tmp_[i] = 2.0/(1.0/S_[i-1]+1.0/S_[i]);
622  }
623  // end points
624  tmp_[0] = (3.0*S_[0]-tmp_[1])/2.0;
625  tmp_[n_-1] = (3.0*S_[n_-2]-tmp_[n_-2])/2.0;
626  break;
627  case CubicInterpolation::Harmonic:
628  // intermediate points
629  for (Size i=1; i<n_-1; ++i) {
630  Real w1 = 2*dx_[i]+dx_[i-1];
631  Real w2 = dx_[i]+2*dx_[i-1];
632  if (S_[i-1]*S_[i]<=0.0)
633  // slope changes sign at point
634  tmp_[i] = 0.0;
635  else
636  // weighted harmonic mean of S_[i] and S_[i-1] if they
637  // have the same sign; otherwise 0
638  tmp_[i] = (w1+w2)/(w1/S_[i-1]+w2/S_[i]);
639  }
640  // end points [0]
641  tmp_[0] = ((2 * dx_[0] + dx_[1])*S_[0] - dx_[0] * S_[1]) / (dx_[1] + dx_[0]);
642  if (tmp_[0]*S_[0]<0.0) {
643  tmp_[0] = 0;
644  }
645  else if (S_[0]*S_[1]<0) {
646  if (std::fabs(tmp_[0])>std::fabs(3*S_[0])) {
647  tmp_[0] = 3*S_[0];
648  }
649  }
650  // end points [n-1]
651  tmp_[n_-1] = ((2*dx_[n_-2]+dx_[n_-3])*S_[n_-2]-dx_[n_-2]*S_[n_-3])/(dx_[n_-3]+dx_[n_-2]);
652  if (tmp_[n_-1]*S_[n_-2]<0.0) {
653  tmp_[n_-1] = 0;
654  }
655  else if (S_[n_-2]*S_[n_-3]<0) {
656  if (std::fabs(tmp_[n_-1])>std::fabs(3*S_[n_-2])) {
657  tmp_[n_-1] = 3*S_[n_-2];
658  }
659  }
660  break;
661  default:
662  QL_FAIL("unknown scheme");
663  }
664  }
665  }
666 
667  std::fill(monotonicityAdjustments_.begin(),
668  monotonicityAdjustments_.end(), false);
669  // Hyman monotonicity constrained filter
670  if (monotonic_) {
671  Real correction;
672  Real pm, pu, pd, M;
673  for (Size i=0; i<n_; ++i) {
674  if (i==0) {
675  if (tmp_[i]*S_[0]>0.0) {
676  correction = tmp_[i]/std::fabs(tmp_[i]) *
677  std::min<Real>(std::fabs(tmp_[i]),
678  std::fabs(3.0*S_[0]));
679  } else {
680  correction = 0.0;
681  }
682  if (correction!=tmp_[i]) {
683  tmp_[i] = correction;
684  monotonicityAdjustments_[i] = true;
685  }
686  } else if (i==n_-1) {
687  if (tmp_[i]*S_[n_-2]>0.0) {
688  correction = tmp_[i]/std::fabs(tmp_[i]) *
689  std::min<Real>(std::fabs(tmp_[i]),
690  std::fabs(3.0*S_[n_-2]));
691  } else {
692  correction = 0.0;
693  }
694  if (correction!=tmp_[i]) {
695  tmp_[i] = correction;
696  monotonicityAdjustments_[i] = true;
697  }
698  } else {
699  pm=(S_[i-1]*dx_[i]+S_[i]*dx_[i-1])/
700  (dx_[i-1]+dx_[i]);
701  M = 3.0 * std::min(std::min(std::fabs(S_[i-1]),
702  std::fabs(S_[i])),
703  std::fabs(pm));
704  if (i>1) {
705  if ((S_[i-1]-S_[i-2])*(S_[i]-S_[i-1])>0.0) {
706  pd=(S_[i-1]*(2.0*dx_[i-1]+dx_[i-2])
707  -S_[i-2]*dx_[i-1])/
708  (dx_[i-2]+dx_[i-1]);
709  if (pm*pd>0.0 && pm*(S_[i-1]-S_[i-2])>0.0) {
710  M = std::max<Real>(M, 1.5*std::min(
711  std::fabs(pm),std::fabs(pd)));
712  }
713  }
714  }
715  if (i<n_-2) {
716  if ((S_[i]-S_[i-1])*(S_[i+1]-S_[i])>0.0) {
717  pu=(S_[i]*(2.0*dx_[i]+dx_[i+1])-S_[i+1]*dx_[i])/
718  (dx_[i]+dx_[i+1]);
719  if (pm*pu>0.0 && -pm*(S_[i]-S_[i-1])>0.0) {
720  M = std::max<Real>(M, 1.5*std::min(
721  std::fabs(pm),std::fabs(pu)));
722  }
723  }
724  }
725  if (tmp_[i]*pm>0.0) {
726  correction = tmp_[i]/std::fabs(tmp_[i]) *
727  std::min(std::fabs(tmp_[i]), M);
728  } else {
729  correction = 0.0;
730  }
731  if (correction!=tmp_[i]) {
732  tmp_[i] = correction;
733  monotonicityAdjustments_[i] = true;
734  }
735  }
736  }
737  }
738 
739 
740  // cubic coefficients
741  for (Size i=0; i<n_-1; ++i) {
742  a_[i] = tmp_[i];
743  b_[i] = (3.0*S_[i] - tmp_[i+1] - 2.0*tmp_[i])/dx_[i];
744  c_[i] = (tmp_[i+1] + tmp_[i] - 2.0*S_[i])/(dx_[i]*dx_[i]);
745  }
746 
747  primitiveConst_[0] = 0.0;
748  for (Size i=1; i<n_-1; ++i) {
749  primitiveConst_[i] = primitiveConst_[i-1]
750  + dx_[i-1] *
751  (this->yBegin_[i-1] + dx_[i-1] *
752  (a_[i-1]/2.0 + dx_[i-1] *
753  (b_[i-1]/3.0 + dx_[i-1] * c_[i-1]/4.0)));
754  }
755  }
756  Real value(Real x) const {
757  Size j = this->locate(x);
758  Real dx_ = x-this->xBegin_[j];
759  return this->yBegin_[j] + dx_*(a_[j] + dx_*(b_[j] + dx_*c_[j]));
760  }
761  Real primitive(Real x) const {
762  Size j = this->locate(x);
763  Real dx_ = x-this->xBegin_[j];
764  return primitiveConst_[j]
765  + dx_*(this->yBegin_[j] + dx_*(a_[j]/2.0
766  + dx_*(b_[j]/3.0 + dx_*c_[j]/4.0)));
767  }
768  Real derivative(Real x) const {
769  Size j = this->locate(x);
770  Real dx_ = x-this->xBegin_[j];
771  return a_[j] + (2.0*b_[j] + 3.0*c_[j]*dx_)*dx_;
772  }
773  Real secondDerivative(Real x) const {
774  Size j = this->locate(x);
775  Real dx_ = x-this->xBegin_[j];
776  return 2.0*b_[j] + 6.0*c_[j]*dx_;
777  }

795  ((a-b)*(a-c)*(a-d)*(b-c)*(b-d)*(c-d));
