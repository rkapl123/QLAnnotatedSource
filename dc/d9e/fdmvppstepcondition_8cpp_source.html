87  void FdmVPPStepCondition::applyTo(Array& a, Time t) const {
88  boost::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
89 
90  const Size nStates = layout->dim()[stateDirection_];
91  const FdmLinearOpIterator endIter = layout->end();
92 
93  for (FdmLinearOpIterator iter=layout->begin();iter != endIter; ++iter) {
94  a[iter.index()] += evolve(iter, t);
95  }
96 
97  for (FdmLinearOpIterator iter=layout->begin();iter != endIter; ++iter) {
98  if (!iter.coordinates()[stateDirection_]) {
99 
100  Array x(nStates);
101  for (Size i=0; i < nStates; ++i) {
102  x[i] = a[layout->neighbourhood(iter, stateDirection_, i)];
103  }
104 
105  const Real gasPrice = gasPrice_->innerValue(iter, t);
106  x = changeState(gasPrice, x, t);
107  for (Size i=0; i < nStates; ++i) {
108  a[layout->neighbourhood(iter, stateDirection_, i)] = x[i];
109  }
110  }
111  }
112  }

114  Real FdmVPPStepCondition::evolve(
115  const FdmLinearOpIterator& iter, Time t) const {
116 
117  const Size state = iter.coordinates()[stateDirection_];
118 
119  if (stateEvolveFcts_[state].empty()) {
120  return 0.0;
121  }
122  else {
123  const Real sparkSpread = sparkSpreadPrice_->innerValue(iter, t);
124  return stateEvolveFcts_[state](sparkSpread);
125  }
126  }
