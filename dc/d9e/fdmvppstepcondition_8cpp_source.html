33  FdmVPPStepCondition::FdmVPPStepCondition(
34  const FdmVPPStepConditionParams& params,
35  Size nStates,
36  const FdmVPPStepConditionMesher& mesh,
37  ext::shared_ptr<FdmInnerValueCalculator> gasPrice,
38  ext::shared_ptr<FdmInnerValueCalculator> sparkSpreadPrice)
39  : heatRate_(params.heatRate), pMin_(params.pMin), pMax_(params.pMax), tMinUp_(params.tMinUp),
40  tMinDown_(params.tMinDown), startUpFuel_(params.startUpFuel),
41  startUpFixCost_(params.startUpFixCost), fuelCostAddon_(params.fuelCostAddon),
42  stateDirection_(mesh.stateDirection), nStates_(nStates), mesher_(mesh.mesher),
43  gasPrice_(std::move(gasPrice)), sparkSpreadPrice_(std::move(sparkSpreadPrice)),
44  stateEvolveFcts_(nStates_) {
45 
46  QL_REQUIRE(nStates_ == mesher_->layout()->dim()[stateDirection_],
47  "mesher does not fit to vpp arguments");
48 
49  for (Size i=0; i < nStates_; ++i) {
50  const Size j = i % (2*tMinUp_ + tMinDown_);
51 
52  if (j < tMinUp_) {
53  stateEvolveFcts_[i] = [&](Real x){ return evolveAtPMin(x); };
54  }
55  else if (j < 2*tMinUp_){
56  stateEvolveFcts_[i] = [&](Real x) { return evolveAtPMax(x); };
57  }
58  }
59  }

67  void FdmVPPStepCondition::applyTo(Array& a, Time t) const {
68  ext::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
69 
70  const Size nStates = layout->dim()[stateDirection_];
71  const FdmLinearOpIterator endIter = layout->end();
72 
73  for (FdmLinearOpIterator iter=layout->begin();iter != endIter; ++iter) {
74  a[iter.index()] += evolve(iter, t);
75  }
76 
77  for (FdmLinearOpIterator iter=layout->begin();iter != endIter; ++iter) {
78  if (iter.coordinates()[stateDirection_] == 0U) {
79 
80  Array x(nStates);
81  for (Size i=0; i < nStates; ++i) {
82  x[i] = a[layout->neighbourhood(iter, stateDirection_, i)];
83  }
84 
85  const Real gasPrice = gasPrice_->innerValue(iter, t);
86  x = changeState(gasPrice, x, t);
87  for (Size i=0; i < nStates; ++i) {
88  a[layout->neighbourhood(iter, stateDirection_, i)] = x[i];
89  }
90  }
91  }
92  }

94  Real FdmVPPStepCondition::evolve(
95  const FdmLinearOpIterator& iter, Time t) const {
96 
97  const Size state = iter.coordinates()[stateDirection_];
98 
99  if (!(stateEvolveFcts_[state])) {
100  return 0.0;
101  }
102  else {
103  const Real sparkSpread = sparkSpreadPrice_->innerValue(iter, t);
104  return stateEvolveFcts_[state](sparkSpread);
105  }
106  }
