29 OptionletStripper::OptionletStripper(
30  const ext::shared_ptr< CapFloorTermVolSurface > &termVolSurface,
31  const ext::shared_ptr< IborIndex > &iborIndex,
32  const Handle< YieldTermStructure > &discount, const VolatilityType type,
33  const Real displacement)
34  : termVolSurface_(termVolSurface), iborIndex_(iborIndex),
35  discount_(discount), nStrikes_(termVolSurface->strikes().size()),
36  volatilityType_(type), displacement_(displacement) {
37 
38  if (volatilityType_ == Normal) {
39  QL_REQUIRE(displacement_ == 0.0,
40  "non-null displacement is not allowed with Normal model");
41  }
42 
43  registerWith(termVolSurface);
44  registerWith(iborIndex_);
45  registerWith(discount_);
46  registerWith(Settings::instance().evaluationDate());
47 
48  Period indexTenor = iborIndex_->tenor();
49  Period maxCapFloorTenor = termVolSurface->optionTenors().back();
50 
51  // optionlet tenors and capFloor lengths
52  optionletTenors_.push_back(indexTenor);
53  capFloorLengths_.push_back(optionletTenors_.back()+indexTenor);
54  QL_REQUIRE(maxCapFloorTenor>=capFloorLengths_.back(),
55  "too short (" << maxCapFloorTenor <<
56  ") capfloor term vol termVolSurface");
57  Period nextCapFloorLength = capFloorLengths_.back()+indexTenor;
58  while (nextCapFloorLength<=maxCapFloorTenor) {
59  optionletTenors_.push_back(capFloorLengths_.back());
60  capFloorLengths_.push_back(nextCapFloorLength);
61  nextCapFloorLength += indexTenor;
62  }
63  nOptionletTenors_ = optionletTenors_.size();
64 
65  optionletVolatilities_ =
66  vector<vector<Volatility> >(nOptionletTenors_,
67  vector<Volatility>(nStrikes_));
68  optionletStrikes_ = vector<vector<Rate> >(nOptionletTenors_,
69  termVolSurface->strikes());
70  optionletDates_ = vector<Date>(nOptionletTenors_);
71  optionletTimes_ = vector<Time>(nOptionletTenors_);
72  atmOptionletRate_ = vector<Rate>(nOptionletTenors_);
73  optionletPaymentDates_ = vector<Date>(nOptionletTenors_);
74  optionletAccrualPeriods_ = vector<Time>(nOptionletTenors_);
75  }
