53 void InflationCoupon::setPricer(const ext::shared_ptr<InflationCouponPricer>& pricer) {
54 QL_REQUIRE(checkPricerImpl(pricer),"pricer given is wrong type");
55 if (pricer_ != nullptr)
56 unregisterWith(pricer_);
57 pricer_ = pricer;
58 if (pricer_ != nullptr)
59 registerWith(pricer_);
60 update();
61 }

64 Rate InflationCoupon::rate() const {
65 QL_REQUIRE(pricer_, "pricer not set");
66 // we know it is the correct type because checkPricerImpl checks on setting
67 // in general pricer_ will be a derived class, as will *this on calling
68 pricer_->initialize(*this);
69 return pricer_->swapletRate();
70 }

73 Real InflationCoupon::accruedAmount(const Date& d) const {
74 if (d <= accrualStartDate_ || d > paymentDate_) {
75 return 0.0;
76 } else {
77 return nominal() * rate() * accruedPeriod(d);
78 }
79 }

82 Date InflationCoupon::fixingDate() const {
83
84 // fixing calendar is usually the null calendar for inflation indices
85 return index_->fixingCalendar().advance(refPeriodEnd_-observationLag_,
86 -static_cast<Integer>(fixingDays_), Days, ModifiedPreceding);
87 }
