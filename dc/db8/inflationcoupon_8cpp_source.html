61  Rate InflationCoupon::rate() const {

70  Real InflationCoupon::accruedAmount(const Date& d) const {
71  if (d <= accrualStartDate_ || d > paymentDate_) {
72  return 0.0;
73  } else {
74  return nominal() * rate() *
75  dayCounter().yearFraction(accrualStartDate_,
76  std::min(d, accrualEndDate_),
77  refPeriodStart_,
78  refPeriodEnd_);
79  }
80  }
