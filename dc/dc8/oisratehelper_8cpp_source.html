58  void OISRateHelper::initializeDates() {
59 
60  // dummy OvernightIndex with curve/swap arguments
61  // review here
62  ext::shared_ptr<IborIndex> clonedIborIndex =
63  overnightIndex_->clone(termStructureHandle_);
64  ext::shared_ptr<OvernightIndex> clonedOvernightIndex =
65  ext::dynamic_pointer_cast<OvernightIndex>(clonedIborIndex);
66 
67  // input discount curve Handle might be empty now but it could
68  // be assigned a curve later; use a RelinkableHandle here
69  swap_ = MakeOIS(tenor_, clonedOvernightIndex, 0.0, forwardStart_)
70  .withDiscountingTermStructure(discountRelinkableHandle_)
71  .withSettlementDays(settlementDays_)
72  .withTelescopicValueDates(telescopicValueDates_)
73  .withPaymentLag(paymentLag_)
74  .withPaymentAdjustment(paymentConvention_)
75  .withPaymentFrequency(paymentFrequency_)
76  .withPaymentCalendar(paymentCalendar_)
77  .withOvernightLegSpread(overnightSpread_)
78  .withAveragingMethod(averagingMethod_);
79 
80  earliestDate_ = swap_->startDate();
81  maturityDate_ = swap_->maturityDate();
82 
83  Date lastPaymentDate = std::max(swap_->overnightLeg().back()->date(),
84  swap_->fixedLeg().back()->date());
85  latestRelevantDate_ = std::max(maturityDate_, lastPaymentDate);
86 
87  switch (pillarChoice_) {
88  case Pillar::MaturityDate:
89  pillarDate_ = maturityDate_;
90  break;
91  case Pillar::LastRelevantDate:
92  pillarDate_ = latestRelevantDate_;
93  break;
94  case Pillar::CustomDate:
95  // pillarDate_ already assigned at construction time
96  QL_REQUIRE(pillarDate_ >= earliestDate_,
97  "pillar date (" << pillarDate_ << ") must be later "
98  "than or equal to the instrument's earliest date (" <<
99  earliestDate_ << ")");
100  QL_REQUIRE(pillarDate_ <= latestRelevantDate_,
101  "pillar date (" << pillarDate_ << ") must be before "
102  "or equal to the instrument's latest relevant date (" <<
103  latestRelevantDate_ << ")");
104  break;
105  default:
106  QL_FAIL("unknown Pillar::Choice(" << Integer(pillarChoice_) << ")");
107  }
108 
109  latestDate_ = std::max(swap_->maturityDate(), lastPaymentDate);
110  }

112  void OISRateHelper::setTermStructure(YieldTermStructure* t) {
113  // do not set the relinkable handle as an observer -
114  // force recalculation when needed
115  bool observer = false;
116 
117  ext::shared_ptr<YieldTermStructure> temp(t, null_deleter());
118  termStructureHandle_.linkTo(temp, observer);
119 
120  if (discountHandle_.empty())
121  discountRelinkableHandle_.linkTo(temp, observer);
122  else
123  discountRelinkableHandle_.linkTo(*discountHandle_, observer);
124 
125  RelativeDateRateHelper::setTermStructure(t);
126  }
