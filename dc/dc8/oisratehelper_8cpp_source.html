59 void OISRateHelper::initializeDates() {
60
61 // dummy OvernightIndex with curve/swap arguments
62 // review here
63 ext::shared_ptr<IborIndex> clonedIborIndex =
64 overnightIndex_->clone(termStructureHandle_);
65 ext::shared_ptr<OvernightIndex> clonedOvernightIndex =
66 ext::dynamic_pointer_cast<OvernightIndex>(clonedIborIndex);
67 // input discount curve Handle might be empty now but it could
68 // be assigned a curve later; use a RelinkableHandle here
69 MakeOIS tmp = MakeOIS(tenor_, clonedOvernightIndex, 0.0, forwardStart_)
70 .withDiscountingTermStructure(discountRelinkableHandle_)
71 .withSettlementDays(settlementDays_)
72 .withTelescopicValueDates(telescopicValueDates_)
73 .withPaymentLag(paymentLag_)
74 .withPaymentAdjustment(paymentConvention_)
75 .withPaymentFrequency(paymentFrequency_)
76 .withPaymentCalendar(paymentCalendar_)
77 .withOvernightLegSpread(overnightSpread_)
78 .withAveragingMethod(averagingMethod_);
79 if (endOfMonth_) {
80 swap_ = tmp.withEndOfMonth(*endOfMonth_);
81 } else {
82 swap_ = tmp;
83 }
84
85 earliestDate_ = swap_->startDate();
86 maturityDate_ = swap_->maturityDate();
87
88 Date lastPaymentDate = std::max(swap_->overnightLeg().back()->date(),
89 swap_->fixedLeg().back()->date());
90 latestRelevantDate_ = std::max(maturityDate_, lastPaymentDate);
91
92 switch (pillarChoice_) {
93 case Pillar::MaturityDate:
94 pillarDate_ = maturityDate_;
95 break;
96 case Pillar::LastRelevantDate:
97 pillarDate_ = latestRelevantDate_;
98 break;
99 case Pillar::CustomDate:
100 // pillarDate_ already assigned at construction time
101 QL_REQUIRE(pillarDate_ >= earliestDate_,
102 "pillar date (" << pillarDate_ << ") must be later "
103 "than or equal to the instrument's earliest date (" <<
104 earliestDate_ << ")");
105 QL_REQUIRE(pillarDate_ <= latestRelevantDate_,
106 "pillar date (" << pillarDate_ << ") must be before "
107 "or equal to the instrument's latest relevant date (" <<
108 latestRelevantDate_ << ")");
109 break;
110 default:
111 QL_FAIL("unknown Pillar::Choice(" << Integer(pillarChoice_) << ")");
112 }
113
114 latestDate_ = std::max(swap_->maturityDate(), lastPaymentDate);
115 }

117 void OISRateHelper::setTermStructure(YieldTermStructure* t) {
118 // do not set the relinkable handle as an observer -
119 // force recalculation when needed
120 bool observer = false;
121
122 ext::shared_ptr<YieldTermStructure> temp(t, null_deleter());
123 termStructureHandle_.linkTo(temp, observer);
124
125 if (discountHandle_.empty())
126 discountRelinkableHandle_.linkTo(temp, observer);
127 else
128 discountRelinkableHandle_.linkTo(*discountHandle_, observer);
129
130 RelativeDateRateHelper::setTermStructure(t);
131 }

148 DatedOISRateHelper::DatedOISRateHelper(const Date& startDate,
149 const Date& endDate,
150 const Handle<Quote>& fixedRate,
151 const ext::shared_ptr<OvernightIndex>& overnightIndex,
152 Handle<YieldTermStructure> discount,
153 bool telescopicValueDates,
154 RateAveraging::Type averagingMethod)
155 : RateHelper(fixedRate), discountHandle_(std::move(discount)),
156 telescopicValueDates_(telescopicValueDates),
157 averagingMethod_(averagingMethod) {
158
159 registerWith(overnightIndex);
160 registerWith(discountHandle_);
161
162 // dummy OvernightIndex with curve/swap arguments
163 // review here
164 ext::shared_ptr<IborIndex> clonedIborIndex =
165 overnightIndex->clone(termStructureHandle_);
166 ext::shared_ptr<OvernightIndex> clonedOvernightIndex =
167 ext::dynamic_pointer_cast<OvernightIndex>(clonedIborIndex);
168
169 // input discount curve Handle might be empty now but it could
170 // be assigned a curve later; use a RelinkableHandle here
171 swap_ = MakeOIS(Period(), clonedOvernightIndex, 0.0)
172 .withDiscountingTermStructure(discountRelinkableHandle_)
173 .withEffectiveDate(startDate)
174 .withTerminationDate(endDate)
175 .withTelescopicValueDates(telescopicValueDates_)
176 .withAveragingMethod(averagingMethod_);
177
178 earliestDate_ = swap_->startDate();
179 Date lastPaymentDate = std::max(swap_->overnightLeg().back()->date(),
180 swap_->fixedLeg().back()->date());
181 latestDate_ = std::max(swap_->maturityDate(), lastPaymentDate);
182 }
