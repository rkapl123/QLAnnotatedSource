33  void AnalyticCapFloorEngine::calculate() const {
34  QL_REQUIRE(!model_.empty(), "null model");
35 
36  Date referenceDate;
37  DayCounter dayCounter;
38 
39  ext::shared_ptr<TermStructureConsistentModel> tsmodel =
40  ext::dynamic_pointer_cast<TermStructureConsistentModel>(*model_);
41  if (tsmodel != nullptr) {
42  referenceDate = tsmodel->termStructure()->referenceDate();
43  dayCounter = tsmodel->termStructure()->dayCounter();
44  } else {
45  referenceDate = termStructure_->referenceDate();
46  dayCounter = termStructure_->dayCounter();
47  }
48 
49  Real value = 0.0;
50  CapFloor::Type type = arguments_.type;
51  Size nPeriods = arguments_.endDates.size();
52 
53  bool includeRefDatePayments =
54  Settings::instance().includeReferenceDateEvents();
55  if (referenceDate == Settings::instance().evaluationDate()) {
56  boost::optional<bool> includeTodaysPayments =
57  Settings::instance().includeTodaysCashFlows();
58  if (includeTodaysPayments) // NOLINT(readability-implicit-bool-conversion)
59  includeRefDatePayments = *includeTodaysPayments;
60  }
61 
62  for (Size i=0; i<nPeriods; i++) {
63 
64  Time fixingTime =
65  dayCounter.yearFraction(referenceDate,
66  arguments_.fixingDates[i]);
67  Time paymentTime =
68  dayCounter.yearFraction(referenceDate,
69  arguments_.endDates[i]);
70 
71  bool not_expired =
72  includeRefDatePayments ? paymentTime >= 0.0 : paymentTime > 0.0;
73 
74  if (not_expired) {
75  Time tenor = arguments_.accrualTimes[i];
76  Rate fixing = arguments_.forwards[i];
77  if (fixingTime <= 0.0) {
78  if (type == CapFloor::Cap || type == CapFloor::Collar) {
79  DiscountFactor discount = model_->discount(paymentTime);
80  Rate strike = arguments_.capRates[i];
81  value += discount * arguments_.nominals[i] * tenor
82  * arguments_.gearings[i]
83  * std::max(0.0, fixing - strike);
84  }
85  if (type == CapFloor::Floor || type == CapFloor::Collar) {
86  DiscountFactor discount = model_->discount(paymentTime);
87  Rate strike = arguments_.floorRates[i];
88  Real mult = (type == CapFloor::Floor) ? 1.0 : -1.0;
89  value += discount * arguments_.nominals[i] * tenor
90  * mult * arguments_.gearings[i]
91  * std::max(0.0, strike - fixing);
92  }
93  } else {
94  Time maturity =
95  dayCounter.yearFraction(referenceDate,
96  arguments_.startDates[i]);
97  if (type == CapFloor::Cap || type == CapFloor::Collar) {
98  Real temp = 1.0+arguments_.capRates[i]*tenor;
99  value += arguments_.nominals[i] *
100  arguments_.gearings[i] * temp *
101  model_->discountBondOption(Option::Put, 1.0/temp,
102  maturity, paymentTime);
103  }
104  if (type == CapFloor::Floor || type == CapFloor::Collar) {
105  Real temp = 1.0+arguments_.floorRates[i]*tenor;
106  Real mult = (type == CapFloor::Floor) ? 1.0 : -1.0;
107  value += arguments_.nominals[i] *
108  arguments_.gearings[i] * temp * mult *
109  model_->discountBondOption(Option::Call, 1.0/temp,
110  maturity, paymentTime);
111  }
112  }
113  }
114  }
115 
116  results_.value = value;
117  }
