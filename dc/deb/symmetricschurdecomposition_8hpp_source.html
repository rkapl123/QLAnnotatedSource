67 inline void SymmetricSchurDecomposition::jacobiRotate_(
68 Matrix &m, Real rot, Real dil, Size j1,
69 Size k1, Size j2, Size k2) const {
70 Real x1, x2;
71 x1 = m[j1][k1];
72 x2 = m[j2][k2];
73 m[j1][k1] = x1 - dil*(x2 + x1*rot);
74 m[j2][k2] = x2 + dil*(x1 - x2*rot);
75 }
