74 void OvernightIndexedSwap::initialize(const Schedule& schedule) {
75 if (fixedDC_==DayCounter())
76 fixedDC_ = overnightIndex_->dayCounter();
77 legs_[0] = FixedRateLeg(schedule)
78 .withNotionals(nominals_)
79 .withCouponRates(fixedRate_, fixedDC_)
80 .withPaymentLag(paymentLag_)
81 .withPaymentAdjustment(paymentAdjustment_)
82 .withPaymentCalendar(paymentCalendar_);
83
84 legs_[1] = OvernightLeg(schedule, overnightIndex_)
85 .withNotionals(nominals_)
86 .withSpreads(spread_)
87 .withTelescopicValueDates(telescopicValueDates_)
88 .withPaymentLag(paymentLag_)
89 .withPaymentAdjustment(paymentAdjustment_)
90 .withPaymentCalendar(paymentCalendar_)
91 .withAveragingMethod(averagingMethod_);
92
93 for (Size j=0; j<2; ++j) {
94 for (auto& i : legs_[j])
95 registerWith(i);
96 }
97
98 switch (type_) {
99 case Payer:
100 payer_[0] = -1.0;
101 payer_[1] = +1.0;
102 break;
103 case Receiver:
104 payer_[0] = +1.0;
105 payer_[1] = -1.0;
106 break;
107 default:
108 QL_FAIL("Unknown overnight-swap type");
109 }
110 }
