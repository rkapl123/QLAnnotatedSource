46  void TreeCapFloorEngine::calculate() const {
47 
48  QL_REQUIRE(!model_.empty(), "no model specified");
49 
50  Date referenceDate;
51  DayCounter dayCounter;
52 
53  boost::shared_ptr<TermStructureConsistentModel> tsmodel =
54  boost::dynamic_pointer_cast<TermStructureConsistentModel>(*model_);
55  if (tsmodel) {
56  referenceDate = tsmodel->termStructure()->referenceDate();
57  dayCounter = tsmodel->termStructure()->dayCounter();
58  } else {
59  referenceDate = termStructure_->referenceDate();
60  dayCounter = termStructure_->dayCounter();
61  }
62 
63  DiscretizedCapFloor capfloor(arguments_, referenceDate, dayCounter);
64  boost::shared_ptr<Lattice> lattice;
65 
66  if (lattice_) {
67  lattice = lattice_;
68  } else {
69  std::vector<Time> times = capfloor.mandatoryTimes();
70  TimeGrid timeGrid(times.begin(), times.end(), timeSteps_);
71  lattice = model_->tree(timeGrid);
72  }
73 
74  Time firstTime = dayCounter.yearFraction(referenceDate,
75  arguments_.startDates.front());
76  Time lastTime = dayCounter.yearFraction(referenceDate,
77  arguments_.endDates.back());
78  capfloor.initialize(lattice, lastTime);
79  capfloor.rollback(firstTime);
80 
81  results_.value = capfloor.presentValue();
82  }

84 }
