123 ZabrModel::fdPrice(const std::vector<Real> &strikes) const {
124 
125  // TODO check strikes to be increasing
126  // TODO put these parameters somewhere
127  const Real start =
128  std::min(0.00001, strikes.front() * 0.5); // lowest strike for grid
129  const Real end =
130  std::max(0.10, strikes.back() * 1.5); // highest strike for grid
131  const Size size = 500; // grid points
132  const Real density = 0.1; // density for concentrating mesher
133  const Size steps =
134  (Size)std::ceil(expiryTime_ * 24); // number of steps in dimension t
135  const Size dampingSteps = 5; // thereof damping steps
136 
137  // Layout
138  std::vector<Size> dim(1, size);
139  const ext::shared_ptr<FdmLinearOpLayout> layout(
140  new FdmLinearOpLayout(dim));
141 
142  // Mesher
143  const ext::shared_ptr<Fdm1dMesher> m1(new Concentrating1dMesher(
144  start, end, size, std::pair<Real, Real>(forward_, density), true));
145  // const ext::shared_ptr<Fdm1dMesher> m1(new
146  // Uniform1dMesher(start,end,size));
147  // const ext::shared_ptr<Fdm1dMesher> m1a(new
148  // Uniform1dMesher(start,0.03,101));
149  // const ext::shared_ptr<Fdm1dMesher> m1b(new

151  // const ext::shared_ptr<Fdm1dMesher> m1(new Glued1dMesher(*m1a,*m1b));
152  const std::vector<ext::shared_ptr<Fdm1dMesher> > meshers(1, m1);
153  const ext::shared_ptr<FdmMesher> mesher(
154  new FdmMesherComposite(layout, meshers));
155 
156  // Boundary conditions
157  FdmBoundaryConditionSet boundaries;
158 
159  // initial values
160  Array rhs(mesher->layout()->size());
161  for (FdmLinearOpIterator iter = layout->begin(); iter != layout->end();
162  ++iter) {
163  Real k = mesher->location(iter, 0);
164  rhs[iter.index()] = std::max(forward_ - k, 0.0);
165  }
166 
167  // local vols (TODO how can we avoid these Array / vector copying?)
168  Array k = mesher->locations(0);
169  std::vector<Real> kv(k.size());
170  std::copy(k.begin(), k.end(), kv.begin());
171  std::vector<Real> locVolv = localVolatility(kv);
172  Array locVol(locVolv.size());
173  std::copy(locVolv.begin(), locVolv.end(), locVol.begin());
174 
175  // solver
176  ext::shared_ptr<FdmDupire1dOp> map(new FdmDupire1dOp(mesher, locVol));
177  FdmBackwardSolver solver(map, boundaries,
178  ext::shared_ptr<FdmStepConditionComposite>(),
179  FdmSchemeDesc::Douglas());
180  solver.rollback(rhs, expiryTime_, 0.0, steps, dampingSteps);
181 
182  // interpolate solution
183  ext::shared_ptr<Interpolation> solution(new CubicInterpolation(
184  k.begin(), k.end(), rhs.begin(), CubicInterpolation::Spline, true,
185  CubicInterpolation::SecondDerivative, 0.0,
186  CubicInterpolation::SecondDerivative, 0.0));
187  // ext::shared_ptr<Interpolation> solution(new
188  // LinearInterpolation(k.begin(),k.end(),rhs.begin()));
189  solution->disableExtrapolation();
190  std::vector<Real> result(strikes.size());
191  std::transform(strikes.begin(), strikes.end(), result.begin(), *solution);
192  return result;

195 Real ZabrModel::fullFdPrice(const Real strike) const {
196 
197  // TODO what are good values here, still experimenting with them
198  Real eps = 0.01;
199  Real scaleFactor = 1.5;
200  Real normInvEps = InverseCumulativeNormal()(1.0 - eps);
201  Real alphaI = alpha_ * std::pow(forward_, beta_ - 1.0);
202  // nu is already standardized within this class ...
203  Real v0 = alpha_ * std::exp(-scaleFactor * normInvEps *
204  std::sqrt(expiryTime_) * nu_);
205  Real v1 = alpha_ *
206  std::exp(scaleFactor * normInvEps * std::sqrt(expiryTime_) * nu_);
207  Real f0 = forward_ * std::exp(-scaleFactor * normInvEps *
208  std::sqrt(expiryTime_) * alphaI);
209  Real f1 = forward_ * std::exp(scaleFactor * normInvEps *
210  std::sqrt(expiryTime_) * alphaI);
211  v1 = std::min(v1, 2.0);
212  f0 = std::min(strike / 2.0, f0);
213  f1 = std::max(strike * 1.5, std::min(f1, std::max(2.0, strike * 1.5)));
214 
215  const Size sizef = 100;
216  const Size sizev = 100;
217  const Size steps = Size(24 * expiryTime_ + 1);
218  const Size dampingSteps = 5;
219  const Real densityf = 0.1;
220  const Real densityv = 0.1;
221 
222  QL_REQUIRE(strike >= f0 && strike <= f1,
223  "strike (" << strike << ") must be inside pde grid [" << f0
224  << ";" << f1 << "]");
225 
226  // Layout
227  std::vector<Size> dim;
228  dim.push_back(sizef);
229  dim.push_back(sizev);
230  const ext::shared_ptr<FdmLinearOpLayout> layout(
231  new FdmLinearOpLayout(dim));
232 
233  // Mesher
234  // two concentrating mesher around f and k to get the mesher for f
235  const Real x0 = std::min(forward_, strike);
236  const Real x1 = std::max(forward_, strike);
237  const Size sizefa = std::max<Size>(
238  4, (Size)std::ceil(((x0 + x1) / 2.0 - f0) / (f1 - f0) * (Real)sizef));
239  const Size sizefb = sizef - sizefa + 1; // common point, so we can spend
240  // one more here
241  const ext::shared_ptr<Fdm1dMesher> mfa(
242  new Concentrating1dMesher(f0, (x0 + x1) / 2.0, sizefa,
243  std::pair<Real, Real>(x0, densityf), true));
244  const ext::shared_ptr<Fdm1dMesher> mfb(
245  new Concentrating1dMesher((x0 + x1) / 2.0, f1, sizefb,
246  std::pair<Real, Real>(x1, densityf), true));
247  const ext::shared_ptr<Fdm1dMesher> mf(new Glued1dMesher(*mfa, *mfb));
248 
249  // concentraing mesher around f to get the forward mesher
250  // const ext::shared_ptr<Fdm1dMesher> mf(new Concentrating1dMesher(
251  // f0, f1, sizef, std::pair<Real, Real>(forward_, densityf), true));
252 
253  // Volatility mesher
254  const ext::shared_ptr<Fdm1dMesher> mv(new Concentrating1dMesher(
255  v0, v1, sizev, std::pair<Real, Real>(alpha_, densityv), true));
256 
257  // uniform meshers
258  // const ext::shared_ptr<Fdm1dMesher> mf(new
259  // Uniform1dMesher(f0,f1,sizef));
260  // const ext::shared_ptr<Fdm1dMesher> mv(new
261  // Uniform1dMesher(v0,v1,sizev));
262 
263  std::vector<ext::shared_ptr<Fdm1dMesher> > meshers;
264  meshers.push_back(mf);
265  meshers.push_back(mv);
266  const ext::shared_ptr<FdmMesher> mesher(
267  new FdmMesherComposite(layout, meshers));
268 
269  // initial values
270  Array rhs(mesher->layout()->size());
271  std::vector<Real> f_;
272  std::vector<Real> v_;
273  for (FdmLinearOpIterator iter = layout->begin(); iter != layout->end();
274  ++iter) {
275  Real f = mesher->location(iter, 0);
276  // Real v = mesher->location(iter, 0);
277  rhs[iter.index()] = std::max(f - strike, 0.0);
278  if (iter.coordinates()[1] == 0U)
279  f_.push_back(mesher->location(iter, 0));
280  if (iter.coordinates()[0] == 0U)
281  v_.push_back(mesher->location(iter, 1));
282  }
283 
284  // Boundary conditions
285  FdmBoundaryConditionSet boundaries;
286 
287  ext::shared_ptr<FdmZabrOp> map(
288  new FdmZabrOp(mesher, beta_, nu_, rho_, gamma_));
289  FdmBackwardSolver solver(map, boundaries,
290  ext::shared_ptr<FdmStepConditionComposite>(),
291  FdmSchemeDesc::/*CraigSneyd()*/ Hundsdorfer());
292 
293  solver.rollback(rhs, expiryTime_, 0.0, steps, dampingSteps);
294 
295  // interpolate solution (this is not necessary when using concentrating
296  // meshers with required point)
297  Matrix result(f_.size(), v_.size());
298  for (Size j = 0; j < v_.size(); ++j)
299  std::copy(rhs.begin() + j * f_.size(),
300  rhs.begin() + (j + 1) * f_.size(), result.row_begin(j));
301  ext::shared_ptr<BicubicSpline> interpolation =
302  ext::make_shared<BicubicSpline>(
303  f_.begin(), f_.end(), v_.begin(), v_.end(), result);
304  interpolation->disableExtrapolation();
305  return (*interpolation)(forward_, alpha_);
306 }

312 Disposable<std::vector<Real> >
313 ZabrModel::x(const std::vector<Real> &strikes) const {
314 
315  QL_REQUIRE(strikes[0] > 0.0 || beta_ < 1.0,
316  "strikes must be positive (" << strikes[0] << ") if beta = 1");
317  for (auto i = strikes.begin() + 1; i != strikes.end(); ++i)
318  QL_REQUIRE(*i > *(i - 1), "strikes must be strictly ascending ("
319  << *(i - 1) << "," << *i << ")");
320 
321  AdaptiveRungeKutta<Real> rk(1.0E-8, 1.0E-5,
322  0.0); // TODO move the parameters here as
323  // parameters with default values to
324  // the constructor
325  std::vector<Real> y(strikes.size()), result(strikes.size());
326  std::transform(strikes.rbegin(), strikes.rend(), y.begin(),
327  [&](Real _k) { return this->y(_k); });
328 
329  if (close(gamma_, 1.0)) {
330  for (Size m = 0; m < y.size(); m++) {
331  Real J = std::sqrt(1.0 + nu_ * nu_ * y[m] * y[m] -
332  2.0 * rho_ * nu_ * y[m]);
333  result[y.size() - 1 - m] =
334  std::log((J + nu_ * y[m] - rho_) / (1.0 - rho_)) / nu_;
335  }
336  } else {
337  Size ynz = std::upper_bound(y.begin(), y.end(), 0.0) - y.begin();
338  if (ynz > 0)
339  if (close(y[ynz - 1], 0.0))
340  ynz--;
341  if (ynz == y.size())
342  ynz--;
343 
344  for (int dir = 1; dir >= -1; dir -= 2) {
345  Real y0 = 0.0, u0 = 0.0;
346  for (int m = ynz + (dir == -1 ? -1 : 0);
347  dir == -1 ? m >= 0 : m < (int)y.size(); m += dir) {
348  Real u = rk([&](Real _y, Real _u){ return F(_y, _u); },
349  u0, y0, y[m]);
350  result[y.size() - 1 - m] = u * pow(alpha_, 1.0 - gamma_);
351  u0 = u;
352  y0 = y[m];
353  }
354  }
355  }
356 
357  return result;
358 }

360 Real ZabrModel::y(const Real strike) const {
361 
362  if (close(beta_, 1.0)) {
363  return std::log(forward_ / strike) * std::pow(alpha_, gamma_ - 2.0);
364  } else {
365  return (strike < 0.0
366  ? std::pow(forward_, 1.0 - beta_) +
367  std::pow(-strike, 1.0 - beta_)
368  : std::pow(forward_, 1.0 - beta_) -
369  std::pow(strike, 1.0 - beta_)) *
370  std::pow(alpha_, gamma_ - 2.0) / (1.0 - beta_);
371  }
372 }

374 Real ZabrModel::F(const Real y, const Real u) const {
375  Real A = 1.0 + (gamma_ - 2.0) * (gamma_ - 2.0) * nu_ * nu_ * y * y +
376  2.0 * rho_ * (gamma_ - 2.0) * nu_ * y;
377  Real B = 2.0 * rho_ * (1.0 - gamma_) * nu_ +
378  2.0 * (1.0 - gamma_) * (gamma_ - 2.0) * nu_ * nu_ * y;
379  Real C = (1.0 - gamma_) * (1.0 - gamma_) * nu_ * nu_;
380  return (-B * u + std::sqrt(B * B * u * u - 4.0 * A * (C * u * u - 1.0))) /
381  (2.0 * A);
382 }
