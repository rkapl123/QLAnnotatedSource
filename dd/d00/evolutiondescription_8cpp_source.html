113  void checkCompatibility(const EvolutionDescription& evolution,
114  const std::vector<Size>& numeraires)
115  {
116  const std::vector<Time>& evolutionTimes = evolution.evolutionTimes();
117  Size n = evolutionTimes.size();
118  QL_REQUIRE(numeraires.size() == n,
119  "Size mismatch between numeraires (" << numeraires.size()
120  << ") and evolution times (" << n << ")");
121 
122  const std::vector<Time>& rateTimes = evolution.rateTimes();
123  for (Size i=0; i<n-1; i++)
124  QL_REQUIRE(rateTimes[numeraires[i]] >= evolutionTimes[i],
125  io::ordinal(i+1) << " step, evolution time " <<
126  evolutionTimes[i] << ": the numeraire (" << numeraires[i] <<
127  "), corresponding to rate time " <<
128  rateTimes[numeraires[i]] << ", is expired");
129  }

138  bool isInMoneyMarketPlusMeasure(const EvolutionDescription& evolution,
139  const std::vector<Size>& numeraires,
140  Size offset) {
141  bool res = true;
142  const std::vector<Time>& rateTimes = evolution.rateTimes();
143  Size maxNumeraire=rateTimes.size()-1;
144  QL_REQUIRE(offset<=maxNumeraire,
145  "offset (" << offset <<
146  ") is greater than the max allowed value for numeraire ("
147  << maxNumeraire << ")");
148  const std::vector<Time>& evolutionTimes = evolution.evolutionTimes();
149  for (Size i=0, j=0; i<evolutionTimes.size(); ++i) {
150  while (rateTimes[j] < evolutionTimes[i])
151  j++;
152  res = (numeraires[i] == std::min(j+offset, maxNumeraire)) && res;
153  }
154  return res;
155  }

168  std::vector<Size> moneyMarketPlusMeasure(const EvolutionDescription& ev,
169  Size offset) {
170  const std::vector<Time>& rateTimes = ev.rateTimes();
171  Size maxNumeraire = rateTimes.size()-1;
172  QL_REQUIRE(offset<=maxNumeraire,
173  "offset (" << offset <<
174  ") is greater than the max allowed value for numeraire ("
175  << maxNumeraire << ")");
176 
177  const std::vector<Time>& evolutionTimes = ev.evolutionTimes();
178  Size n = evolutionTimes.size();
179  std::vector<Size> numeraires(n);
180  for (Size i=0, j=0; i<n; ++i) {
181  while (rateTimes[j] < evolutionTimes[i])
182  j++;
183  numeraires[i] = std::min(j+offset, maxNumeraire);
184  }
185  return numeraires;
186  }
