36  /* A note on the number of buckets: As it is now the code goes splitting
37  losses into buckets from loses equal to zero to losses up to the value of
38  the underlying basket. This is in view of a stochastic loss given default
39  but in a constant LGD situation this is a waste and it is more efficient to
40  go up to the attainable losses.
41  \todo Extend to the multifactor case for a generic LM
42  */
43  template<class copulaPolicy>
44  class HomogeneousPoolLossModel : public DefaultLossModel {
45  private:
46  void resetModel();
47  public:
48  HomogeneousPoolLossModel(
49  const ext::shared_ptr<ConstantLossLatentmodel<copulaPolicy> >&
50  copula,
51  Size nBuckets,
52  Real max = 5.,
53  Real min = -5.,
54  Real nSteps = 50)
55  : copula_(copula),
56  nBuckets_(nBuckets),
57  max_(max), min_(min), nSteps_(nSteps), delta_((max - min)/nSteps)
58  {
59  QL_REQUIRE(copula->numFactors() == 1,
60  "Inhomogeneous model not implemented for multifactor");
61  }
62  protected:
63  Distribution lossDistrib(const Date& d) const;
64  public:
65  Real expectedTrancheLoss(const Date& d) const {
66  return lossDistrib(d).cumulativeExcessProbability(attachAmount_,
67  detachAmount_);
68  // This one if the distribution is over the whole loss structure:
69  // but it becomes very expensive
70  /*
71  return lossDistrib(d).trancheExpectedValue(attach_ * notional_,
72  detach_ * notional_);
73  */
74  }
75  Real percentile(const Date& d, Real percentile) const {
76  Real portfLoss = lossDistrib(d).confidenceLevel(percentile);
77  return std::min(std::max(portfLoss - attachAmount_, 0.),
78  detachAmount_ - attachAmount_);
79  }
80  Real expectedShortfall(const Date& d, Probability percentile) const {
81  Distribution dist = lossDistrib(d);
82  dist.tranche(attachAmount_, detachAmount_);
83  return dist.expectedShortfall(percentile);
84  }
85  protected:
86  const ext::shared_ptr<ConstantLossLatentmodel<copulaPolicy> > copula_;
87  Size nBuckets_;
88  mutable Real attach_, detach_, notional_, attachAmount_, detachAmount_;
89  mutable std::vector<Real> notionals_;
90  private:
91  // integration:
92  // \todo move integration to latent model types when moving to a
93  // multifactor version
94  const Real max_;// redundant?
95  const Real min_;
96  const Real nSteps_;
97  const Real delta_;
98  };

106  template<class CP>
107  void HomogeneousPoolLossModel<CP>::resetModel()
108  {
109  // need to be capped now since the limit amounts might be over the
110  // remaining notional (think amortizing)
111  attach_ = std::min(basket_->remainingAttachmentAmount() /
112  basket_->remainingNotional(), 1.);
113  detach_ = std::min(basket_->remainingDetachmentAmount() /
114  basket_->remainingNotional(), 1.);
115  notional_ = basket_->remainingNotional();
116  notionals_ = basket_->remainingNotionals();
117  attachAmount_ = basket_->remainingAttachmentAmount();
118  detachAmount_ = basket_->remainingDetachmentAmount();
119 
120  copula_->resetBasket(basket_.currentLink());
121  }

123  template<class CP>
124  Distribution HomogeneousPoolLossModel<CP>::lossDistrib(
125  const Date& d) const
126  {
127  LossDistHomogeneous bucktLDistBuff(nBuckets_, detachAmount_);
128 
129  std::vector<Real> lgd;// switch to a mutable cache member
130  std::vector<Real> recoveries = copula_->recoveries();
131  std::transform(recoveries.begin(), recoveries.end(),
132  std::back_inserter(lgd),
133  subtract_from<Real>(1.0));
134  std::transform(lgd.begin(), lgd.end(), notionals_.begin(),
135  lgd.begin(), std::multiplies<Real>());
136  std::vector<Real> prob = basket_->remainingProbabilities(d);
137  for(Size iName=0; iName<prob.size(); iName++)
138  prob[iName] = copula_->inverseCumulativeY(prob[iName], iName);
139 
140  // integrate locally (1 factor).
141  // use explicitly a 1D latent model object?
142  Distribution dist(nBuckets_, 0.0,
143  detachAmount_);
144  //notional_);
145  std::vector<Real> mkft(1, min_ + delta_ /2.);
146  for (Size i = 0; i < nSteps_; i++) {
147  std::vector<Real> conditionalProbs;
148  for(Size iName=0; iName<notionals_.size(); iName++)
149  conditionalProbs.push_back(
150  copula_->conditionalDefaultProbabilityInvP(prob[iName], iName,
151  mkft));
152  Distribution bld = bucktLDistBuff(lgd, conditionalProbs);
153  Real densitydm = delta_ * copula_->density(mkft);
154  // also, instead of calling the static method it could be wrapped
155  // through an inlined call in the latent model
156  for (Size j = 0; j < nBuckets_; j++)
157  dist.addDensity(j, bld.density(j) * densitydm);
158  mkft[0] += delta_;
159  }
160  return dist;
161  }
