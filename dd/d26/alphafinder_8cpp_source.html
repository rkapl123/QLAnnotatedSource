94  Real Minimize(Real low,
95  Real high,
96  Real tolerance,
97  T& theObject,
98  Real (T::*Value)(Real),
99  bool (T::*Condition)(Real),
100  bool& failed) {
101 
102  Real leftValue = (theObject.*Value)(low);
103  Real rightValue = (theObject.*Value)(high);
104  Real W = 0.5*(3.0-std::sqrt(5.0));
105  Real x=W*low+(1-W)*high;
106  Real midValue = (theObject.*Value)(x);
107 
108  failed = true;
109 
110  while(high - low > tolerance) {
111 
112  if (x - low > high -x) // left interval is bigger
113  {
114  Real tentativeNewMid = W*low+(1-W)*x;
115  Real tentativeNewMidValue = (theObject.*Value)(tentativeNewMid);
116  bool conditioner = (theObject.*Condition)(tentativeNewMidValue);
117  if (!conditioner) {
118  if ((theObject.*Condition)(x))
119  return x;
120  else
121  if (leftValue < rightValue)
122  return low;
123  else
124  return high;
125  }
126 
127  if (tentativeNewMidValue < midValue) // go left
128  {
129  high =x;
130  rightValue = midValue;
131  x = tentativeNewMid;
132  midValue = tentativeNewMidValue;
133  }
134  else // go right
135  {
136  low = tentativeNewMid;
137  leftValue = tentativeNewMidValue;
138  }
139  }
140  else
141  {
142  Real tentativeNewMid = W*x+(1-W)*high;
143  Real tentativeNewMidValue = (theObject.*Value)(tentativeNewMid);
144  bool conditioner = (theObject.*Condition)(tentativeNewMidValue);
145  if (!conditioner) {
146  if ((theObject.*Condition)(x))
147  return x;
148  else
149  if (leftValue < rightValue)
150  return low;
151  else
152  return high;
153  }
154 
155  if (tentativeNewMidValue < midValue) // go right
156  {
157  low =x;
158  leftValue = midValue;
159  x = tentativeNewMid;
160  midValue = tentativeNewMidValue;
161  }
162  else // go left
163  {
164  high = tentativeNewMid;
165  rightValue = tentativeNewMidValue;
166  }
167  }
168 
169 
170 
171 
172  }
173  failed = false;
174  return x;

231  bool AlphaFinder::finalPart(Real alphaFound,
232  Integer stepindex,
233  const std::vector<Volatility>& ratetwohomogeneousvols,
234  Real quadraticPart,
235  Real linearPart,
236  Real constantPart,
237  Real& alpha,
238  Real& a,
239  Real& b,
240  std::vector<Volatility>& ratetwovols) {
241  alpha = alphaFound;
242  quadratic q2(quadraticPart, linearPart, constantPart-targetVariance_ );
243  parametricform_->setAlpha(alpha);
244  Real y; // dummy
245  q2.roots(a,y);
246 
247  Real varSoFar=0.0;
248  for (Integer i =0; i < stepindex+1; ++i) {
249  ratetwovols[i] = ratetwohomogeneousvols[i] *
250  (*parametricform_)(i) * a;
251  varSoFar += ratetwovols[i]* ratetwovols[i];
252  }
253 
254  Real VarToFind = totalVar_-varSoFar;
255  if (VarToFind < 0)
256  return false;
257  Real requiredSd = std::sqrt(VarToFind);
258  b = requiredSd / (ratetwohomogeneousvols[stepindex+1] *
259  (*parametricform_)(stepindex));
260  ratetwovols[stepindex+1] = requiredSd;
261  return true;
262  }

296  bool AlphaFinder::solve(Real alpha0,
297  Integer stepindex, // caplet index
298  const std::vector<Volatility>& rateonevols,
299  const std::vector<Volatility>& ratetwohomogeneousvols,
300  const std::vector<Real>& correlations,
301  Real w0,
302  Real w1,
303  Real targetVariance,
304  Real tolerance,
305  Real alphaMax,
306  Real alphaMin,
307  Integer steps,
308  Real& alpha,
309  Real& a,
310  Real& b,
311  std::vector<Volatility>& ratetwovols) {
312  stepindex_=stepindex;
313  rateonevols_=rateonevols;
314  ratetwohomogeneousvols_=ratetwohomogeneousvols;
315  correlations_=correlations;
316  w0_=w0;
317  w1_=w1;
318  totalVar_=0;
319  for (Size i=0; i <=static_cast<Size>(stepindex)+1; ++i)
320  totalVar_+=ratetwohomogeneousvols[i]*ratetwohomogeneousvols[i];
321  targetVariance_ = targetVariance;
322 
323  // constant part will not depend on alpha
324 
325  constantPart_ =0.0;
326  for (Integer i=0; i < stepindex+1; ++i)
327  constantPart_+=rateonevols[i]*rateonevols[i];
328 
329  constantPart_ *= w0*w0;
330 
331  // compute linear part with initial alpha
332  Real valueAtTP = valueAtTurningPoint(alpha0);
333 
334  if (valueAtTP <= targetVariance) {
335  finalPart(alpha0,
336  stepindex,
337  ratetwohomogeneousvols,
338  quadraticPart_,
339  linearPart_,
340  constantPart_,
341  alpha,
342  a,
343  b,
344  ratetwovols);
345  return true;
346  }
347 
348  // we now have to solve
349  Real bottomValue = valueAtTurningPoint(alphaMin);
350  Real bottomAlpha = alphaMin;
351  Real topValue = valueAtTurningPoint(alphaMax);
352  Real topAlpha = alphaMax;
353  Real bilimit = alpha0;
354 
355  if (bottomValue > targetVariance && topValue > targetVariance) {
356  // see if if ok at some intermediate point by stepping through
357  Integer i=1;
358  while ( i < steps && topValue> targetVariance) {
359  topAlpha = alpha0 + (alphaMax-alpha0)*(i+0.0)/(steps+0.0);
360  topValue=valueAtTurningPoint(topAlpha);
361  ++i;
362  }
363 
364  if (topValue <= targetVariance)
365  bilimit = alpha0 + (topAlpha-alpha0)*(i-2.0)/(steps+0.0);
366  }
367 
368  if (bottomValue > targetVariance && topValue > targetVariance) {
369  // see if if ok at some intermediate point by stepping through
370  Integer i=1;
371  while ( i < steps && topValue> targetVariance) {
372  bottomAlpha = alpha0 + (alphaMin-alpha0)*(i+0.0)/(steps+0.0);
373  bottomValue=valueAtTurningPoint(bottomAlpha);
374  ++i;
375  }
376 
377  if (bottomValue <= targetVariance)
378  bilimit = alpha0 +(bottomAlpha-alpha0)*(i-2.0)/(steps+0.0);
379  }
380 
381  if (bottomValue > targetVariance && topValue > targetVariance)
382  return false;
383 
384  if (bottomValue <= targetVariance) {
385  // then find root of increasing function
386  // (or as if increasing function)
387  alpha = Bisection<AlphaFinder>(
388  targetVariance,
389  bottomAlpha,
390  bilimit,
391  tolerance,
392  *this,
393  &AlphaFinder::valueAtTurningPoint);
394  } else {
395  // find root of decreasing function (or as if decreasing function)
396  alpha=Bisection<AlphaFinder>(
397  -targetVariance,
398  bilimit,
399  topAlpha,
400  tolerance,
401  *this,
402  &AlphaFinder::minusValueAtTurningPoint);
403  }
404  finalPart(alpha,
405  stepindex,
406  ratetwohomogeneousvols,
407  quadraticPart_,
408  linearPart_,
409  constantPart_,
410  alpha,
411  a,
412  b,
413  ratetwovols);
414  return true;
415  }

445  targetVariance_=targetVariance;

452 
453  constantPart_ *= w0*w0;
454 
455  Real alpha1 = alphaMin;
456  Real alpha2 = alphaMax;
457 
458  // compute linear part with initial alpha
459  bool alpha0OK = testIfSolutionExists(alpha0);
460  bool alphaMaxOK = testIfSolutionExists(alphaMax);
461  bool alphaMinOK = testIfSolutionExists(alphaMin);
462

464 
465  if (foundOKPoint) {
466  if (!alphaMinOK) {
467  // lower alpha is bad

469  // must die somewhere in between

471  alphaMin,

474  *this,

476  } else {

478  alpha1 = FindLowestOK<AlphaFinder>(
479  alpha0,
480  alphaMax,
481  tolerance,
482  *this,
483  &AlphaFinder::testIfSolutionExists);
484  }
485  }
486 
487 
488  if (!alphaMaxOK) {
489  // higher alpha is bad
490  alpha2 = FindHighestOK<AlphaFinder>(
491  alpha1,
492  alphaMax,
493  tolerance,
494  *this,
495  &AlphaFinder::testIfSolutionExists);
496  } else
497  alpha2= alphaMax;
498  }
499  else {
500  // ok let's see if we can find a value of alpha that works
501  bool foundUpOK = false;
502  bool foundDownOK = false;
503  Real alphaUp = alpha0, alphaDown = alpha0;
504  Real stepSize = (alphaMax-alpha0)/steps;
505 
506  for (Size j=0;
507  j<static_cast<Size>(steps) && !foundUpOK && !foundDownOK;
508  ++j) {
509  alphaUp = alpha0+j*stepSize;
510  foundUpOK=testIfSolutionExists(alphaUp);
511  alphaDown = alpha0-j*stepSize;
512  foundDownOK=testIfSolutionExists(alphaDown);
513  }
514  foundOKPoint = foundUpOK || foundDownOK;
515  if (!foundOKPoint)
516  return false;
517 
518  if (foundUpOK) {
519  alpha1 = alphaUp;
520  alpha2 = FindHighestOK<AlphaFinder>(
521  alpha1,
522  alphaMax,
523  tolerance,
524  *this,
525  &AlphaFinder::testIfSolutionExists);
526  } else {
527  alpha2 = alphaDown;
528  alpha1 = FindLowestOK<AlphaFinder>(
529  alphaMin,
530  alpha2,
531  tolerance,
532  *this,
533  &AlphaFinder::testIfSolutionExists);
534  }
535  }
536 
537  // we have now found alpha1, alpha2 such that solution exists
538  // at endpoints. we now want to minimize within that interval
539  bool failed;
540  alpha = Minimize<AlphaFinder>(
541  alpha1,
542  alpha2,
543  tolerance,
544  *this,
545  &AlphaFinder::homogeneityfailure,
546  &AlphaFinder::testIfSolutionExists,

553  computeLinearPart(alpha),
554  constantPart_,

557  b,
558  ratetwovols);
