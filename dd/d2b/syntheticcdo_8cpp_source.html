35 SyntheticCDO::SyntheticCDO(const ext::shared_ptr<Basket>& basket,
36 Protection::Side side,
37 const Schedule& schedule,
38 Rate upfrontRate,
39 Rate runningRate,
40 const DayCounter& dayCounter,
41 BusinessDayConvention paymentConvention,
42 boost::optional<Real> notional)
43 : basket_(basket), side_(side), upfrontRate_(upfrontRate), runningRate_(runningRate),
44 leverageFactor_(notional ? notional.get() / basket->trancheNotional() : Real(1.)), // NOLINT(readability-implicit-bool-conversion)
45 dayCounter_(dayCounter), paymentConvention_(paymentConvention) {
46 QL_REQUIRE(!basket->names().empty(), "basket is empty");
47 // Basket inception must lie before contract protection start.
48 QL_REQUIRE(basket->refDate() <= schedule.startDate(),
49 //using the start date of the schedule might be wrong, think of the
50 // CDS rule
51 "Basket did not exist before contract start.");
52
53 // Notice the notional is that of the basket at basket inception, some
54 // names might have defaulted in between
55 normalizedLeg_ = FixedRateLeg(schedule)
56 .withNotionals(basket_->trancheNotional() * leverageFactor_)
57 .withCouponRates(runningRate, dayCounter)
58 .withPaymentAdjustment(paymentConvention);
59
60 // Date today = Settings::instance().evaluationDate();
61
62 // register with probabilities if the corresponding issuer is, baskets
63 // are not registered with the DTS
64 for (Size i = 0; i < basket->names().size(); i++) {
65 /* This turns out to be a problem: depends on today but I am not
66 modifying the registrations, if we go back in time in the
67 calculations this would left me unregistered to some. Not impossible
68 to de-register and register when updating but i am dropping it.
69
70 if(!basket->pool()->get(basket->names()[i]).
71 defaultedBetween(schedule.dates()[0], today,
72 basket->pool()->defaultKeys()[i]))
73 */
74 // registers with the associated curve (issuer and event type)
75 // \todo make it possible to access them by name instead of index
76 registerWith(basket->pool()->get(basket->names()[i]).
77 defaultProbability(basket->pool()->defaultKeys()[i]));
78 /* \todo Issuers should be observables/obsrvr and they would in turn
79 regiter with the DTS; only we might get updates from curves we do
80 not use.
81 */
82 }
83 registerWith(basket_);
84 }

108 Rate SyntheticCDO::fairPremium () const {
109 calculate();
110 QL_REQUIRE(premiumValue_ != 0, "Attempted divide by zero while calculating syntheticCDO premium.");
111 return runningRate_
112 * (protectionValue_ - upfrontPremiumValue_) / premiumValue_;
113 }

228 Real SyntheticCDO::implicitCorrelation(const std::vector<Real>& recoveries,
229 const Handle<YieldTermStructure>& discountCurve,
230 Real targetNPV,
231 Real accuracy) const
232 {
233 ext::shared_ptr<SimpleQuote> correl(new SimpleQuote(0.0));
234
235 ext::shared_ptr<GaussianLHPLossModel> lhp(new
236 GaussianLHPLossModel(Handle<Quote>(correl), recoveries));
237
238 // lock
239 basket_->setLossModel(lhp);
240
241 MidPointCDOEngine engineIC(discountCurve);
242 setupArguments(engineIC.getArguments());
243 const auto* results = dynamic_cast<const SyntheticCDO::results*>(engineIC.getResults());
244
245 // aviod recal of the basket on engine updates through the quote
246 basket_->recalculate();
247 basket_->freeze();
248
249 ObjectiveFunction f(targetNPV, *correl, engineIC, results);
250 Rate guess = 0.001;
251 // Rate step = guess*0.1;
252
253 // wrap/catch to be able to unfreeze the basket:
254 Real solution = Brent().solve(f, accuracy, guess, QL_EPSILON, 1.-QL_EPSILON);
255 basket_->unfreeze();
256 return solution;
257 }
