80 void setupTimes(const std::vector<Date>& dates,
81 Date referenceDate,
82 const DayCounter& dayCounter) {
83 times_.resize(dates.size());
84 times_[0] = dayCounter.yearFraction(referenceDate, dates[0]);
85 for (Size i = 1; i < dates.size(); i++) {
86 QL_REQUIRE(dates[i] > dates[i-1],
87 "dates not sorted: " << dates[i] << " passed after " << dates[i-1]);
88
89 times_[i] = dayCounter.yearFraction(referenceDate, dates[i]);
90 QL_REQUIRE(!close(this->times_[i], this->times_[i-1]),
91 "two passed dates (" << dates[i-1] << " and " << dates[i]
92 << ") correspond to the same time "
93 << "under this curve's day count convention (" << dayCounter << ")");
94 }
95 }

97 void setupInterpolation() {
98 interpolation_ = interpolator_.interpolate(times_.begin(),
99 times_.end(),
100 data_.begin());
101 }

108 // Usually, the maximum date is the one corresponding to the
109 // last node. However, it might happen that a bit of
110 // extrapolation is used by construction; for instance, when a
111 // curve is bootstrapped and the last relevant date for an
112 // instrument is after the corresponding pillar.
113 // We provide here a slot to store this information, so that
114 // it's available to all derived classes (we should have
115 // probably done the same with the dates_ vector, but moving
116 // it here might not be entirely backwards-compatible).
