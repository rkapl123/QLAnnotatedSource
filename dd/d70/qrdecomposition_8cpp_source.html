30  Disposable<std::vector<Size> > qrDecomposition(const Matrix& M,
31  Matrix& q, Matrix& r,
32  bool pivot) {
33  Matrix mT = transpose(M);
34  const Size m = M.rows();
35  const Size n = M.columns();
36 
37  std::unique_ptr<int[]> lipvt(new int[n]);
38  std::unique_ptr<Real[]> rdiag(new Real[n]);
39  std::unique_ptr<Real[]> wa(new Real[n]);
40 
41  MINPACK::qrfac(m, n, mT.begin(), 0, (pivot)?1:0,
42  lipvt.get(), n, rdiag.get(), rdiag.get(), wa.get());
43  if (r.columns() != n || r.rows() !=n)
44  r = Matrix(n, n);
45 
46  for (Size i=0; i < n; ++i) {
47  std::fill(r.row_begin(i), r.row_begin(i)+i, 0.0);
48  r[i][i] = rdiag[i];
49  if (i < m) {
50  std::copy(mT.column_begin(i)+i+1, mT.column_end(i),
51  r.row_begin(i)+i+1);
52  }
53  else {
54  std::fill(r.row_begin(i)+i+1, r.row_end(i), 0.0);
55  }
56  }
57 
58  if (q.rows() != m || q.columns() != n)
59  q = Matrix(m, n);
60 
61  if (m > n) {
62  std::fill(q.begin(), q.end(), 0.0);
63 
64  Integer u = std::min(n,m);
65  for (Integer i=0; i < u; ++i)
66  q[i][i] = 1.0;
67 
68  Array v(m);
69  for (Integer i=u-1; i >=0; --i) {
70  if (std::fabs(mT[i][i]) > QL_EPSILON) {
71  const Real tau = 1.0/mT[i][i];
72 
73  std::fill(v.begin(), v.begin()+i, 0.0);
74  std::copy(mT.row_begin(i)+i, mT.row_end(i), v.begin()+i);
75 
76  Array w(n, 0.0);
77  for (Size l=0; l < n; ++l)
78  w[l] += std::inner_product(
79  v.begin()+i, v.end(), q.column_begin(l)+i, 0.0);
80 
81  for (Size k=i; k < m; ++k) {
82  const Real a = tau*v[k];
83  for (Size l=0; l < n; ++l)
84  q[k][l] -= a*w[l];
85  }
86  }
87  }
88  }
89  else {
90  Array w(m);
91  for (Size k=0; k < m; ++k) {
92  std::fill(w.begin(), w.end(), 0.0);
93  w[k] = 1.0;
94 
95  for (Size j=0; j < std::min(n, m); ++j) {
96  const Real t3 = mT[j][j];
97  if (t3 != 0.0) {
98  const Real t
99  = std::inner_product(mT.row_begin(j)+j, mT.row_end(j),
100  w.begin()+j, 0.0)/t3;
101  for (Size i=j; i<m; ++i) {
102  w[i]-=mT[j][i]*t;
103  }
104  }
105  q[k][j] = w[j];
106  }
107  std::fill(q.row_begin(k) + std::min(n, m), q.row_end(k), 0.0);
108  }
109  }
110 
111  std::vector<Size> ipvt(n);
112 
113  if (pivot) {
114  std::copy(lipvt.get(), lipvt.get()+n, ipvt.begin());
115  }
116  else {
117  for (Size i=0; i < n; ++i)
118  ipvt[i] = i;
119  }
120 
121  return ipvt;
122  }
