113  template<class T>
114  Disposable<std::vector<T> > AdaptiveRungeKutta<T>::operator()(
115  const OdeFct& ode,
116  const std::vector<T>& y1,
117  const Real x1,
118  const Real x2) {
119  Size n = y1.size();
120  std::vector<T> y(y1);
121  std::vector<Real> yScale(n);
122  Real x = x1;
123  Real h = h1_* (x1<=x2 ? 1 : -1);
124  Real hnext,hdid;
125 
126  for (Size nstp=1; nstp<=ADAPTIVERK_MAXSTP; nstp++) {
127  std::vector<T> dydx=ode(x,y);
128  for (Size i=0;i<n;i++)
129  yScale[i] = std::abs(y[i])+std::abs(dydx[i]*h)+ADAPTIVERK_TINY;
130  if ((x+h-x2)*(x+h-x1) > 0.0)
131  h=x2-x;
132  rkqs(y,dydx,x,h,eps_,yScale,hdid,hnext,ode);
133 
134  if ((x-x2)*(x2-x1) >= 0.0)
135  return y;
136 
137  if (std::fabs(hnext) <= hmin_)
138  QL_FAIL("Step size (" << hnext << ") too small ("
139  << hmin_ << " min) in AdaptiveRungeKutta");
140  h=hnext;
141  }
142  QL_FAIL("Too many steps (" << ADAPTIVERK_MAXSTP
143  << ") in AdaptiveRungeKutta");
144  }

172  template<class T>
173  void AdaptiveRungeKutta<T>::rkqs(std::vector<T>& y,
174  const std::vector<T>& dydx,
175  Real& x,
176  const Real htry,
177  const Real eps,
178  const std::vector<Real>& yScale,
179  Real& hdid,
180  Real& hnext,
181  const OdeFct& derivs) {
182  Size n=y.size();
183  Real errmax,xnew;
184  std::vector<T> yerr(n),ytemp(n);
185 
186  Real h=htry;
187 
188  for(;;) {
189  rkck(y,dydx,x,h,ytemp,yerr,derivs);
190  errmax=0.0;
191  for (Size i=0;i<n;i++)
192  errmax=std::max(errmax,std::abs(yerr[i]/yScale[i]));
193  errmax/=eps;
194  if (errmax>1.0) {
195  Real htemp1 = ADAPTIVERK_SAFETY*h*std::pow(errmax,ADAPTIVERK_PSHRINK);
196  Real htemp2 = h / 10;
197  // These would be std::min and std::max, of course,
198  // but VC++14 had problems inlining them and caused
199  // the wrong results to be calculated. The problem
200  // seems to be fixed in update 3, but let's keep this
201  // implementation for compatibility.
202  Real max_positive = htemp1 > htemp2 ? htemp1 : htemp2;
203  Real max_negative = htemp1 < htemp2 ? htemp1 : htemp2;
204  h = ((h >= 0.0) ? max_positive : max_negative);
205  xnew=x+h;
206  if (xnew==x)
207  QL_FAIL("Stepsize underflow (" << h << " at x = " << x
208  << ") in AdaptiveRungeKutta::rkqs");
209  continue;
210  } else {
211  if (errmax>ADAPTIVERK_ERRCON)
212  hnext=ADAPTIVERK_SAFETY*h*std::pow(errmax,ADAPTIVERK_PGROW);
213  else
214  hnext=5.0*h;
215  x+=(hdid=h);
216  for (Size i=0;i<n;i++)
217  y[i]=ytemp[i];
218  break;
219  }
220  }
221  }

223  template <class T>
224  void AdaptiveRungeKutta<T>::rkck(const std::vector<T>& y,
225  const std::vector<T>& dydx,
226  Real x,
227  const Real h,
228  std::vector<T>& yout,
229  std::vector<T> &yerr,
230  const OdeFct& derivs) {
231 
232  Size n=y.size();
233  std::vector<T> ak2(n),ak3(n),ak4(n),ak5(n),ak6(n),ytemp(n);
234 
235  // first step
236  for (Size i=0;i<n;i++)
237  ytemp[i]=y[i]+b21*h*dydx[i];
238 
239  // second step
240  ak2=derivs(x+a2*h,ytemp);
241  for (Size i=0;i<n;i++)
242  ytemp[i]=y[i]+h*(b31*dydx[i]+b32*ak2[i]);
243 
244  // third step
245  ak3=derivs(x+a3*h,ytemp);
246  for (Size i=0;i<n;i++)
247  ytemp[i]=y[i]+h*(b41*dydx[i]+b42*ak2[i]+b43*ak3[i]);
248 
249  // fourth step
250  ak4=derivs(x+a4*h,ytemp);
251  for (Size i=0;i<n;i++)
252  ytemp[i]=y[i]+h*(b51*dydx[i]+b52*ak2[i]+b53*ak3[i]+b54*ak4[i]);
253 
254  // fifth step
255  ak5=derivs(x+a5*h,ytemp);
256  for (Size i=0;i<n;i++)
257  ytemp[i]=y[i]+h*(b61*dydx[i]+b62*ak2[i]+b63*ak3[i]+b64*ak4[i]+b65*ak5[i]);
258 
259  // sixth step
260  ak6=derivs(x+a6*h,ytemp);
261  for (Size i=0;i<n;i++) {
262  yout[i]=y[i]+h*(c1*dydx[i]+c3*ak3[i]+c4*ak4[i]+c6*ak6[i]);
263  yerr[i]=h*(dc1*dydx[i]+dc3*ak3[i]+dc4*ak4[i]+dc5*ak5[i]+dc6*ak6[i]);
264  }
265  }
