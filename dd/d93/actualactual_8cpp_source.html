30  template <class T>
31  Integer findCouponsPerYear(const T& impl,
32  Date refStart, Date refEnd) {
33  // This will only work for day counts longer than 15 days.
34  auto months = (Integer)std::lround(12 * Real(impl.dayCount(refStart, refEnd)) / 365.0);
35  return (Integer)std::lround(12.0 / Real(months));
36  }

49  std::vector<Date> getListOfPeriodDatesIncludingQuasiPayments(
50  const Schedule& schedule) {
51  // Process the schedule into an array of dates.
52  Date issueDate = schedule.date(0);
53  std::vector<Date> newDates = schedule.dates();
54 
55  if (!schedule.hasIsRegular() || !schedule.isRegular(1))
56  {
57  Date firstCoupon = schedule.date(1);
58 
59  Date notionalFirstCoupon =
60  schedule.calendar().advance(firstCoupon,
61  -schedule.tenor(),
62  schedule.businessDayConvention(),
63  schedule.endOfMonth());
64 
65  newDates[0] = notionalFirstCoupon;
66 
67  //long first coupon
68  if (notionalFirstCoupon > issueDate) {
69  Date priorNotionalCoupon =
70  schedule.calendar().advance(notionalFirstCoupon,
71  -schedule.tenor(),
72  schedule.businessDayConvention(),
73  schedule.endOfMonth());
74  newDates.insert(newDates.begin(),
75  priorNotionalCoupon); //insert as the first element?
76  }
77  }
78 
79  if (!schedule.hasIsRegular() || !schedule.isRegular(schedule.size() - 1))
80  {
81  Date notionalLastCoupon =
82  schedule.calendar().advance(schedule.date(schedule.size() - 2),
83  schedule.tenor(),
84  schedule.businessDayConvention(),
85  schedule.endOfMonth());
86 
87  newDates[schedule.size() - 1] = notionalLastCoupon;
88 
89  if (notionalLastCoupon < schedule.endDate())
90  {
91  Date nextNotionalCoupon =
92  schedule.calendar().advance(notionalLastCoupon,
93  schedule.tenor(),
94  schedule.businessDayConvention(),
95  schedule.endOfMonth());
96  newDates.push_back(nextNotionalCoupon);
97  }
98  }
99 
100  return newDates;
101  }

103  template <class T>
104  Time yearFractionWithReferenceDates(const T& impl,
105  const Date& d1, const Date& d2,
106  const Date& d3, const Date& d4) {
107  QL_REQUIRE(d1 <= d2,
108  "This function is only correct if d1 <= d2\n"
109  "d1: " << d1 << " d2: " << d2);
110 
111  Real referenceDayCount = Real(impl.dayCount(d3, d4));
112  //guess how many coupon periods per year:
113  Integer couponsPerYear;
114  if (referenceDayCount < 16) {
115  couponsPerYear = 1;
116  referenceDayCount = impl.dayCount(d1, d1 + 1 * Years);
117  }
118  else {
119  couponsPerYear = findCouponsPerYear(impl, d3, d4);
120  }
121  return Real(impl.dayCount(d1, d2)) / (referenceDayCount*couponsPerYear);
122  }

149  Time ActualActual::ISMA_Impl::yearFraction(const Date& d1,
150  const Date& d2,
151  const Date& d3,
152  const Date& d4) const {
153  if (d1 == d2) {
154  return 0.0;
155  } else if (d2 < d1) {
156  return -yearFraction(d2, d1, d3, d4);
157  }
158 
159  std::vector<Date> couponDates =
160  getListOfPeriodDatesIncludingQuasiPayments(schedule_);
161 
162  Real yearFractionSum = 0.0;
163  for (Size i = 0; i < couponDates.size() - 1; i++) {
164  Date startReferencePeriod = couponDates[i];
165  Date endReferencePeriod = couponDates[i + 1];
166  if (d1 < endReferencePeriod && d2 > startReferencePeriod) {
167  yearFractionSum +=
168  yearFractionWithReferenceDates(*this,
169  std::max(d1, startReferencePeriod),
170  std::min(d2, endReferencePeriod),
171  startReferencePeriod,
172  endReferencePeriod);
173  }
174  }
175  return yearFractionSum;
176  }

179  Time ActualActual::Old_ISMA_Impl::yearFraction(const Date& d1,
180  const Date& d2,
181  const Date& d3,
182  const Date& d4) const {
183  if (d1 == d2)
184  return 0.0;
185 
186  if (d1 > d2)
187  return -yearFraction(d2,d1,d3,d4);
188 
189  // when the reference period is not specified, try taking
190  // it equal to (d1,d2)
191  Date refPeriodStart = (d3 != Date() ? d3 : d1);
192  Date refPeriodEnd = (d4 != Date() ? d4 : d2);
193 
194  QL_REQUIRE(refPeriodEnd > refPeriodStart && refPeriodEnd > d1,
195  "invalid reference period: "
196  << "date 1: " << d1
197  << ", date 2: " << d2
198  << ", reference period start: " << refPeriodStart
199  << ", reference period end: " << refPeriodEnd);
200 
201  // estimate roughly the length in months of a period
202  auto months = (Integer)std::lround(12 * Real(refPeriodEnd - refPeriodStart) / 365);
203 
204  // for short periods...
205  if (months == 0) {
206  // ...take the reference period as 1 year from d1
207  refPeriodStart = d1;
208  refPeriodEnd = d1 + 1*Years;
209  months = 12;
210  }
211 
212  Time period = Real(months)/12.0;
213 
214  if (d2 <= refPeriodEnd) {
215  // here refPeriodEnd is a future (notional?) payment date
216  if (d1 >= refPeriodStart) {
217  // here refPeriodStart is the last (maybe notional)
218  // payment date.
219  // refPeriodStart <= d1 <= d2 <= refPeriodEnd
220  // [maybe the equality should be enforced, since
221  // refPeriodStart < d1 <= d2 < refPeriodEnd
222  // could give wrong results] ???
223  return period*Real(daysBetween(d1,d2)) /
224  daysBetween(refPeriodStart,refPeriodEnd);
225  } else {
226  // here refPeriodStart is the next (maybe notional)
227  // payment date and refPeriodEnd is the second next
228  // (maybe notional) payment date.
229  // d1 < refPeriodStart < refPeriodEnd
230  // AND d2 <= refPeriodEnd
231  // this case is long first coupon
232 
233  // the last notional payment date
234  Date previousRef = refPeriodStart - months*Months;
235 
236  if (d2 > refPeriodStart)
237  return yearFraction(d1, refPeriodStart, previousRef,
238  refPeriodStart) +
239  yearFraction(refPeriodStart, d2, refPeriodStart,
240  refPeriodEnd);
241  else
242  return yearFraction(d1,d2,previousRef,refPeriodStart);
243  }
244  } else {
245  // here refPeriodEnd is the last (notional?) payment date
246  // d1 < refPeriodEnd < d2 AND refPeriodStart < refPeriodEnd
247  QL_REQUIRE(refPeriodStart<=d1,
248  "invalid dates: "
249  "d1 < refPeriodStart < refPeriodEnd < d2");
250  // now it is: refPeriodStart <= d1 < refPeriodEnd < d2
251 
252  // the part from d1 to refPeriodEnd
253  Time sum = yearFraction(d1, refPeriodEnd,
254  refPeriodStart, refPeriodEnd);
255 
256  // the part from refPeriodEnd to d2
257  // count how many regular periods are in [refPeriodEnd, d2],
258  // then add the remaining time
259  Integer i=0;
260  Date newRefStart, newRefEnd;
261  for (;;) {
262  newRefStart = refPeriodEnd + (months*i)*Months;
263  newRefEnd = refPeriodEnd + (months*(i+1))*Months;
264  if (d2 < newRefEnd) {
265  break;
266  } else {
267  sum += period;
268  i++;
269  }
270  }
271  sum += yearFraction(newRefStart,d2,newRefStart,newRefEnd);
272  return sum;
273  }
274  }

277  Time ActualActual::ISDA_Impl::yearFraction(const Date& d1,
278  const Date& d2,
279  const Date&,
280  const Date&) const {
281  if (d1 == d2)
282  return 0.0;
283 
284  if (d1 > d2)
285  return -yearFraction(d2,d1,Date(),Date());
286 
287  Integer y1 = d1.year(), y2 = d2.year();
288  Real dib1 = (Date::isLeap(y1) ? 366.0 : 365.0),
289  dib2 = (Date::isLeap(y2) ? 366.0 : 365.0);
290 
291  Time sum = y2 - y1 - 1;
292  // FLOATING_POINT_EXCEPTION
293  sum += daysBetween(d1, Date(1,January,y1+1))/dib1;
294  sum += daysBetween(Date(1,January,y2),d2)/dib2;
295  return sum;
296  }

299  Time ActualActual::AFB_Impl::yearFraction(const Date& d1,
300  const Date& d2,
301  const Date&,
302  const Date&) const {
303  if (d1 == d2)
304  return 0.0;
305 
306  if (d1 > d2)
307  return -yearFraction(d2,d1,Date(),Date());
308 
309  Date newD2=d2, temp=d2;
310  Time sum = 0.0;
311  while (temp > d1) {
312  temp = newD2 - 1*Years;
313  if (temp.dayOfMonth()==28 && temp.month()==2
314  && Date::isLeap(temp.year())) {
315  temp += 1;
316  }
317  if (temp>=d1) {
318  sum += 1.0;
319  newD2 = temp;
320  }
321  }
322 
323  Real den = 365.0;
324 
325  if (Date::isLeap(newD2.year())) {
326  temp = Date(29, February, newD2.year());
327  if (newD2>temp && d1<=temp)
328  den += 1.0;
329  } else if (Date::isLeap(d1.year())) {
330  temp = Date(29, February, d1.year());
331  if (newD2>temp && d1<=temp)
332  den += 1.0;
333  }
334 
335  return sum+daysBetween(d1, newD2)/den;
336  }
