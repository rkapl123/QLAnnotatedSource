44  Time ActualActual::ISMA_Impl::yearFraction(const Date& d1,
45  const Date& d2,
46  const Date& d3,
47  const Date& d4) const {
48  if (d1 == d2)
49  return 0.0;
50 
51  if (d1 > d2)
52  return -yearFraction(d2,d1,d3,d4);
53 
54  // when the reference period is not specified, try taking
55  // it equal to (d1,d2)
56  Date refPeriodStart = (d3 != Date() ? d3 : d1);
57  Date refPeriodEnd = (d4 != Date() ? d4 : d2);
58 
59  QL_REQUIRE(refPeriodEnd > refPeriodStart && refPeriodEnd > d1,
60  "invalid reference period: "
61  << "date 1: " << d1
62  << ", date 2: " << d2
63  << ", reference period start: " << refPeriodStart
64  << ", reference period end: " << refPeriodEnd);
65 
66  // estimate roughly the length in months of a period
67  Integer months =
68  Integer(0.5+12*Real(refPeriodEnd-refPeriodStart)/365);
69 
70  // for short periods...
71  if (months == 0) {
72  // ...take the reference period as 1 year from d1
73  refPeriodStart = d1;
74  refPeriodEnd = d1 + 1*Years;
75  months = 12;
76  }
77 
78  Time period = Real(months)/12.0;
79 
80  if (d2 <= refPeriodEnd) {
81  // here refPeriodEnd is a future (notional?) payment date
82  if (d1 >= refPeriodStart) {
83  // here refPeriodStart is the last (maybe notional)
84  // payment date.
85  // refPeriodStart <= d1 <= d2 <= refPeriodEnd
86  // [maybe the equality should be enforced, since
87  // refPeriodStart < d1 <= d2 < refPeriodEnd
88  // could give wrong results] ???
89  return period*Real(daysBetween(d1,d2)) /
90  daysBetween(refPeriodStart,refPeriodEnd);
91  } else {
92  // here refPeriodStart is the next (maybe notional)
93  // payment date and refPeriodEnd is the second next
94  // (maybe notional) payment date.
95  // d1 < refPeriodStart < refPeriodEnd
96  // AND d2 <= refPeriodEnd
97  // this case is long first coupon
98 
99  // the last notional payment date
100  Date previousRef;
101  if (schedule_.empty()) {
102  previousRef = refPeriodStart - months*Months;
103  } else {
104  previousRef = schedule_.calendar().advance(refPeriodStart,
105  -schedule_.tenor(),
106  schedule_.businessDayConvention(),
107  schedule_.endOfMonth());
108  }
109 
110  if (d2 > refPeriodStart)
111  return yearFraction(d1, refPeriodStart, previousRef,
112  refPeriodStart) +
113  yearFraction(refPeriodStart, d2, refPeriodStart,
114  refPeriodEnd);
115  else
116  return yearFraction(d1,d2,previousRef,refPeriodStart);
117  }
118  } else {
119  // here refPeriodEnd is the last (notional?) payment date
120  // d1 < refPeriodEnd < d2 AND refPeriodStart < refPeriodEnd
121  QL_REQUIRE(refPeriodStart<=d1,
122  "invalid dates: "
123  "d1 < refPeriodStart < refPeriodEnd < d2");
124  // now it is: refPeriodStart <= d1 < refPeriodEnd < d2
125 
126  // the part from d1 to refPeriodEnd
127  Time sum = yearFraction(d1, refPeriodEnd,
128  refPeriodStart, refPeriodEnd);
129 
130  // the part from refPeriodEnd to d2
131  // count how many regular periods are in [refPeriodEnd, d2],
132  // then add the remaining time
133  Integer i=0;
134  Date newRefStart, newRefEnd;
135  for (;;) {
136  newRefStart = refPeriodEnd + (months*i)*Months;
137  newRefEnd = refPeriodEnd + (months*(i+1))*Months;
138  if (d2 < newRefEnd) {
139  break;
140  } else {
141  sum += period;
142  i++;
143  }
144  }
145  sum += yearFraction(newRefStart,d2,newRefStart,newRefEnd);
146  return sum;
147  }
148  }

150  Time ActualActual::ISDA_Impl::yearFraction(const Date& d1,
151  const Date& d2,
152  const Date&,
153  const Date&) const {
154  if (d1 == d2)
155  return 0.0;
156 
157  if (d1 > d2)
158  return -yearFraction(d2,d1,Date(),Date());
159 
160  Integer y1 = d1.year(), y2 = d2.year();
161  Real dib1 = (Date::isLeap(y1) ? 366.0 : 365.0),
162  dib2 = (Date::isLeap(y2) ? 366.0 : 365.0);
163 
164  Time sum = y2 - y1 - 1;
165  // FLOATING_POINT_EXCEPTION
166  sum += daysBetween(d1, Date(1,January,y1+1))/dib1;
167  sum += daysBetween(Date(1,January,y2),d2)/dib2;
168  return sum;
169  }

171  Time ActualActual::AFB_Impl::yearFraction(const Date& d1,
172  const Date& d2,
173  const Date&,
174  const Date&) const {
175  if (d1 == d2)
176  return 0.0;
177 
178  if (d1 > d2)
179  return -yearFraction(d2,d1,Date(),Date());
180 
181  Date newD2=d2, temp=d2;
182  Time sum = 0.0;
183  while (temp > d1) {
184  temp = newD2 - 1*Years;
185  if (temp.dayOfMonth()==28 && temp.month()==2
186  && Date::isLeap(temp.year())) {
187  temp += 1;
188  }
189  if (temp>=d1) {
190  sum += 1.0;
191  newD2 = temp;
192  }
193  }
194 
195  Real den = 365.0;
196 
197  if (Date::isLeap(newD2.year())) {
198  temp = Date(29, February, newD2.year());
199  if (newD2>temp && d1<=temp)
200  den += 1.0;
201  } else if (Date::isLeap(d1.year())) {
202  temp = Date(29, February, d1.year());
203  if (newD2>temp && d1<=temp)
204  den += 1.0;
205  }
206 
207  return sum+daysBetween(d1, newD2)/den;
208  }
