69  Real FloatingRateCoupon::accruedAmount(const Date& d) const {
70  if (d <= accrualStartDate_ || d > paymentDate_) {
71  // out of coupon range
72  return 0.0;
73  } else if (tradingExCoupon(d)) {
74  return -nominal() * rate() *
75  dayCounter().yearFraction(d, std::max(d, accrualEndDate_),
76  refPeriodStart_, refPeriodEnd_);
77  } else {
78  // usual case
79  return nominal() * rate() *
80  dayCounter().yearFraction(accrualStartDate_,
81  std::min(d, accrualEndDate_),
82  refPeriodStart_,
83  refPeriodEnd_);
84  }
85  }

87  Date FloatingRateCoupon::fixingDate() const {
88  // if isInArrears_ fix at the end of period
89  Date refDate = isInArrears_ ? accrualEndDate_ : accrualStartDate_;
