64  template <class T>
65  void BinomialConvertibleEngine<T>::calculate() const {
66 
67  DayCounter rfdc = process_->riskFreeRate()->dayCounter();
68  DayCounter divdc = process_->dividendYield()->dayCounter();
69  DayCounter voldc = process_->blackVolatility()->dayCounter();
70  Calendar volcal = process_->blackVolatility()->calendar();
71 
72  Real s0 = process_->x0();
73  QL_REQUIRE(s0 > 0.0, "negative or null underlying");
74  Volatility v = process_->blackVolatility()->blackVol(
75  arguments_.exercise->lastDate(), s0);
76  Date maturityDate = arguments_.exercise->lastDate();
77  Rate riskFreeRate = process_->riskFreeRate()->zeroRate(
78  maturityDate, rfdc, Continuous, NoFrequency);
79  Rate q = process_->dividendYield()->zeroRate(
80  maturityDate, divdc, Continuous, NoFrequency);
81  Date referenceDate = process_->riskFreeRate()->referenceDate();
82 
83  // subtract dividends
84  Size i;
85  for (i=0; i<arguments_.dividends.size(); i++) {
86  if (arguments_.dividends[i]->date() >= referenceDate)
87  s0 -= arguments_.dividends[i]->amount() *
88  process_->riskFreeRate()->discount(
89  arguments_.dividends[i]->date());
90  }
91  QL_REQUIRE(s0 > 0.0,
92  "negative value after subtracting dividends");
93 
94  // binomial trees with constant coefficient
95  Handle<Quote> underlying(ext::shared_ptr<Quote>(new SimpleQuote(s0)));
96  Handle<YieldTermStructure> flatRiskFree(
97  ext::shared_ptr<YieldTermStructure>(
98  new FlatForward(referenceDate, riskFreeRate, rfdc)));
99  Handle<YieldTermStructure> flatDividends(
100  ext::shared_ptr<YieldTermStructure>(
101  new FlatForward(referenceDate, q, divdc)));
102  Handle<BlackVolTermStructure> flatVol(
103  ext::shared_ptr<BlackVolTermStructure>(
104  new BlackConstantVol(referenceDate, volcal, v, voldc)));
105 
106  ext::shared_ptr<PlainVanillaPayoff> payoff =
107  ext::dynamic_pointer_cast<PlainVanillaPayoff>(arguments_.payoff);
108  QL_REQUIRE(payoff, "non-plain payoff given");
109 
110  Time maturity = rfdc.yearFraction(arguments_.settlementDate,
111  maturityDate);
112 
113  ext::shared_ptr<GeneralizedBlackScholesProcess> bs(
114  new GeneralizedBlackScholesProcess(underlying, flatDividends,
115  flatRiskFree, flatVol));
116  ext::shared_ptr<T> tree(new T(bs, maturity, timeSteps_,
117  payoff->strike()));
118 
119  Real creditSpread = arguments_.creditSpread->value();
120 
121  ext::shared_ptr<Lattice> lattice(
122  new TsiveriotisFernandesLattice<T>(tree,riskFreeRate,maturity,
123  timeSteps_,creditSpread,v,q));
124 
125  DiscretizedConvertible convertible(arguments_, bs,
126  TimeGrid(maturity, timeSteps_));
127 
128  convertible.initialize(lattice, maturity);
129  convertible.rollback(0.0);
130  results_.value = convertible.presentValue();
131  QL_ENSURE(results_.value < std::numeric_limits<Real>::max(),
132  "floating-point overflow on tree grid");
133  }
134 
135 }
