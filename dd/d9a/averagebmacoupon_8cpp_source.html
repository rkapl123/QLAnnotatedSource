37  Rate swapletRate() const {
38  const std::vector<Date>& fixingDates = coupon_->fixingDates();
39  const boost::shared_ptr<InterestRateIndex>& index =
40  coupon_->index();
41 
42  Natural cutoffDays = 0; // to be verified
43  Date startDate = coupon_->accrualStartDate() - cutoffDays,
44  endDate = coupon_->accrualEndDate() - cutoffDays,
45  d1 = startDate,
46  d2 = startDate;
47 
48  QL_REQUIRE (fixingDates.size() > 0, "fixing date list empty");
49  QL_REQUIRE (index->valueDate(fixingDates.front()) <= startDate,
50  "first fixing date valid after period start");
51  QL_REQUIRE (index->valueDate(fixingDates.back()) >= endDate,
52  "last fixing date valid before period end");
53 
54  Rate avgBMA = 0.0;
55  Integer days = 0;
56  for (Size i=0; i<fixingDates.size() - 1; ++i) {
57  Date valueDate = index->valueDate(fixingDates[i]);
58  Date nextValueDate = index->valueDate(fixingDates[i+1]);
59 
60  if (fixingDates[i] >= endDate || valueDate >= endDate)
61  break;
62  if (fixingDates[i+1] < startDate
63  || nextValueDate <= startDate)
64  continue;
65 
66  d2 = std::min(nextValueDate, endDate);
67 
68  avgBMA += index->fixing(fixingDates[i]) * (d2 - d1);
69 
70  days += d2 - d1;
71  d1 = d2;
72  }
73  avgBMA /= (endDate - startDate);
74 
75  QL_ENSURE(days == endDate - startDate,
76  "averaging days " << days << " differ from "
77  "interest days " << (endDate - startDate));
78 
79  return coupon_->gearing()*avgBMA + coupon_->spread();
80  }

105  Real nominal,
106  const Date& startDate,
107  const Date& endDate,
108  const boost::shared_ptr<BMAIndex>& index,
109  Real gearing, Spread spread,
110  const Date& refPeriodStart,
111  const Date& refPeriodEnd,
112  const DayCounter& dayCounter)
113  : FloatingRateCoupon(paymentDate, nominal, startDate, endDate,
114  index->fixingDays(), index, gearing, spread,
115  refPeriodStart, refPeriodEnd, dayCounter, false)
116  {
117  Calendar cal = index->fixingCalendar();
118  Integer fixingDays = Integer(index->fixingDays());
119  fixingDays += bmaCutoffDays;
120  Date fixingStart = cal.advance(startDate, -fixingDays*Days, Preceding);
121  fixingSchedule_ = index->fixingSchedule(fixingStart, endDate);
122 
123  setPricer(boost::shared_ptr<FloatingRateCouponPricer>(
124  new AverageBMACouponPricer));

211  AverageBMALeg::operator Leg() const {
212 
213  QL_REQUIRE(!notionals_.empty(), "no notional given");
214 
215  Leg cashflows;
216 
217  // the following is not always correct
218  Calendar calendar = schedule_.calendar();
219 
220  Date refStart, start, refEnd, end;
221  Date paymentDate;
222 
223  Size n = schedule_.size()-1;
224  for (Size i=0; i<n; ++i) {
225  refStart = start = schedule_.date(i);
226  refEnd = end = schedule_.date(i+1);
227  paymentDate = calendar.adjust(end, paymentAdjustment_);
228  if (i == 0 && schedule_.hasIsRegular() && !schedule_.isRegular(i+1)
229  && schedule_.hasTenor())
230  refStart = calendar.adjust(end - schedule_.tenor(),
231  paymentAdjustment_);
232  if (i == n-1 && schedule_.hasIsRegular() && !schedule_.isRegular(i+1)
233  && schedule_.hasTenor())
234  refEnd = calendar.adjust(start + schedule_.tenor(),
235  paymentAdjustment_);
236 
237  cashflows.push_back(boost::shared_ptr<CashFlow>(new
238  AverageBMACoupon(paymentDate,
239  detail::get(notionals_, i, notionals_.back()),
240  start, end,
241  index_,
242  detail::get(gearings_, i, 1.0),
243  detail::get(spreads_, i, 0.0),
244  refStart, refEnd,
245  paymentDayCounter_)));
246  }
247 
248  return cashflows;
249  }
