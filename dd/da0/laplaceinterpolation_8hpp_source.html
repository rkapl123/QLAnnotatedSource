39 template <class M> void laplaceInterpolation(M &A, Real relTol = 1E-6) {
40 
41  struct f_A {
42  const SparseMatrix &g;
43  explicit f_A(const SparseMatrix &g) : g(g) {}
44  Disposable<Array> operator()(const Array &x) const {
45  return prod(g, x);
46  }
47  };
48 
49  Size m = A.rows();
50  Size n = A.columns();
51 
52  QL_REQUIRE(n > 1 && m > 1, "matrix (" << m << "," << n
53  << ") must at least be 2x2");
54 
55  SparseMatrix g(m * n, m * n, 5 * m * n);
56  Array rhs(m * n, 0.0), guess(m * n, 0.0);
57  Real guessTmp = 0.0;
58  Size i1, i2, i3, i4, j1, j2, j3, j4;
59  bool inner;
60 
61  for (Size l = 0, i = 0; i < m; ++i) {
62  for (Size j = 0; j < n; ++j) {
63 
64  inner = false;
65 
66  // top
67  if (i == 0) {
68  if (j == 0) {
69  i1 = 0;
70  j1 = 1;
71  i2 = 1;
72  j2 = 0;
73  } else {
74  if (j == n - 1) {
75  i1 = 0;
76  j1 = n - 2;
77  i2 = 1;
78  j2 = n - 1;
79  } else {
80  i1 = i2 = 0;
81  j1 = j - 1;
82  j2 = j + 1;
83  }
84  }
85  }
86 
87  // bottom
88  if (i == m - 1) {
89  if (j == 0) {
90  i1 = m - 1;
91  j1 = 1;
92  i2 = m - 2;
93  j2 = 0;
94  } else {
95  if (j == n - 1) {
96  i1 = m - 1;
97  j1 = n - 2;
98  i2 = m - 2;
99  j2 = n - 1;
100  } else {
101  i1 = i2 = m - 1;
102  j1 = j - 1;
103  j2 = j + 1;
104  }
105  }
106  }
107 
108  // left / right
109  if (i > 0 && i < m - 1) {
110  if (j == 0 || j == n - 1) {
111  j1 = j2 = j;
112  i1 = i - 1;
113  i2 = i + 1;
114  } else {
115  inner = true;
116  i1 = i - 1;
117  i2 = i - 1;
118  i3 = i + 1;
119  i4 = i + 1;
120  j1 = j - 1;
121  j2 = j + 1;
122  j3 = j - 1;
123  j4 = j + 1;
124  }
125  }
126 
127  g(l, i * n + j) = 1.0;
128  if (A[i][j] == Null<Real>()) {
129  if (inner) {
130  g(l, i1 * n + j1) = -0.25;
131  g(l, i2 * n + j2) = -0.25;
132  g(l, i3 * n + j3) = -0.25;
133  g(l, i4 * n + j4) = -0.25;
134  } else {
135  g(l, i1 * n + j1) = -0.5;
136  g(l, i2 * n + j2) = -0.5;
137  }
138  rhs[l] = 0.0;
139  guess[l] = guessTmp;
140  } else {
141  rhs[l] = A[i][j];
142  guess[l] = guessTmp = A[i][j];
143  }
144  ++l;
145  }
146  }
147 
148  // solve the equation (preconditioner is identiy)
149  Array s = BiCGstab(f_A(g), 10 * m * n, relTol)
150  .solve(rhs, guess)
151  .x;
152 
153  // replace missing values by solution
154  for (Size i = 0; i < m; ++i) {
155  for (Size j = 0; j < n; ++j) {
156  if (A[i][j] == Null<Real>()) {
157  A[i][j] = s[i * n + j];
158  }
159  }
160  }
161 };
