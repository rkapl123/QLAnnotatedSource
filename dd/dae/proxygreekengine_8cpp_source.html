81  void ProxyGreekEngine::multiplePathValues(
82  SequenceStatisticsInc& stats,
83  std::vector<std::vector<SequenceStatisticsInc> >& modifiedStats,
84  Size numberOfPaths) {
85  Size N = product_->numberOfProducts();
86 
87  std::vector<Real> values(N);
88  std::vector<std::vector<std::vector<Real> > > modifiedValues;
89  modifiedValues.resize(constrainedEvolvers_.size());
90  for (Size i=0; i<modifiedValues.size(); ++i) {
91  modifiedValues[i].resize(constrainedEvolvers_[i].size());
92  for (Size j=0; j<modifiedValues[i].size(); ++j)
93  modifiedValues[i][j].resize(N);
94  }
95 
96  std::vector<Real> results(N);
97 
98  for (Size i=0; i<numberOfPaths; ++i) {
99  singlePathValues(values, modifiedValues);
100  stats.add(values);
101 
102  for (Size j=0; j<diffWeights_.size(); ++j) {
103  for (Size k=0; k<diffWeights_[j].size(); ++k) {
104  const std::vector<Real>& weights = diffWeights_[j][k];
105  for (Size l=0; l<N; ++l) {
106  results[l] = weights[0]*values[l];
107  for (Size n=1; n<weights.size(); ++n)
108  results[l] += weights[n]*modifiedValues[j][n-1][l];
109  }
110  modifiedStats[j][k].add(results);
111  }
112  }
113  }

116  void ProxyGreekEngine::singleEvolverValues(MarketModelEvolver& evolver,
117  std::vector<Real>& values,
118  bool storeRates) {
119 
120  std::fill(numerairesHeld_.begin(), numerairesHeld_.end(), 0.0);
121  Real weight = evolver.startNewPath();
122  product_->reset();
123  Real principalInNumerairePortfolio = 1.0;
124 
125  if (storeRates)
126  constraintsActive_ =false;
127  // std::fill(constraintsActive_.begin(),
128  // constraintsActive_.end(),
129  // false);
130  // }
131 
132  bool done = false;
133  do {
134  Size thisStep = evolver.currentStep();
135  weight *= evolver.advanceStep();
136  done = product_->nextTimeStep(evolver.currentState(),
137  numberCashFlowsThisStep_,
138  cashFlowsGenerated_);
139  if (storeRates) {
140  constraints_[thisStep] = evolver.currentState().swapRate(
141  startIndexOfConstraint_[thisStep],
142  endIndexOfConstraint_[thisStep]);
143  constraintsActive_[thisStep] = true;
144  }
145 
146  Size numeraire =
147  evolver.numeraires()[thisStep];
148 
149  // for each product...
150  for (Size i=0; i<numberProducts_; ++i) {
151  // ...and each cash flow...
152  const std::vector<MarketModelMultiProduct::CashFlow>& cashflows =
153  cashFlowsGenerated_[i];
154  for (Size j=0; j<numberCashFlowsThisStep_[i]; ++j) {
155  // ...convert the cash flow to numeraires.
156  // This is done by calculating the number of
157  // numeraire bonds corresponding to such cash flow...
158  const MarketModelDiscounter& discounter =
159  discounters_[cashflows[j].timeIndex];
160 
161  Real bonds = cashflows[j].amount *
162  discounter.numeraireBonds(evolver.currentState(),
163  numeraire);
164 
165  // ...and adding the newly bought bonds to the number
166  // of numeraires held.
167  numerairesHeld_[i] +=
168  weight*bonds/principalInNumerairePortfolio;
169  }
170  }
171 
172  if (!done) {
173 
174  // The numeraire might change between steps. This implies
175  // that we might have to convert the numeraire bonds for
176  // this step into a corresponding amount of numeraire
177  // bonds for the next step. This can be done by changing
178  // the principal of the numeraire and updating the number
179  // of bonds in the numeraire portfolio accordingly.
180 
181  Size nextNumeraire = evolver.numeraires()[thisStep+1];
182 
183  principalInNumerairePortfolio *=
184  evolver.currentState().discountRatio(numeraire,
185  nextNumeraire);
186  }
187 
188  } while (!done);
189 
190  for (Size i=0; i<numerairesHeld_.size(); ++i)
191  values[i] = numerairesHeld_[i] * initialNumeraireValue_;
192 
193  }
194 
195 }
