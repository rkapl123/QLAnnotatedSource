55  Real GaussLobattoIntegral::calculateAbsTolerance(
56  const ext::function<Real (Real)>& f,
57  Real a, Real b) const {
58 
59 
60  Real relTol = std::max(relAccuracy_, QL_EPSILON);
61 
62  const Real m = (a+b)/2;
63  const Real h = (b-a)/2;
64  const Real y1 = f(a);
65  const Real y3 = f(m-alpha_*h);
66  const Real y5 = f(m-beta_*h);
67  const Real y7 = f(m);
68  const Real y9 = f(m+beta_*h);
69  const Real y11= f(m+alpha_*h);
70  const Real y13= f(b);
71 
72  const Real f1 = f(m-x1_*h);
73  const Real f2 = f(m+x1_*h);
74  const Real f3 = f(m-x2_*h);
75  const Real f4 = f(m+x2_*h);
76  const Real f5 = f(m-x3_*h);
77  const Real f6 = f(m+x3_*h);
78 
79  Real acc=h*(0.0158271919734801831*(y1+y13)
80  +0.0942738402188500455*(f1+f2)
81  +0.1550719873365853963*(y3+y11)
82  +0.1888215739601824544*(f3+f4)
83  +0.1997734052268585268*(y5+y9)
84  +0.2249264653333395270*(f5+f6)
85  +0.2426110719014077338*y7);
86 
87  increaseNumberOfEvaluations(13);
88  if (acc == 0.0 && ( f1 != 0.0 || f2 != 0.0 || f3 != 0.0
89  || f4 != 0.0 || f5 != 0.0 || f6 != 0.0)) {
90  QL_FAIL("can not calculate absolute accuracy "
91  "from relative accuracy");
92  }
93 
94  Real r = 1.0;
95  if (useConvergenceEstimate_) {
96  const Real integral2 = (h/6)*(y1+y13+5*(y5+y9));
97  const Real integral1 = (h/1470)*(77*(y1+y13)+432*(y3+y11)+
98  625*(y5+y9)+672*y7);
99 
100  if (std::fabs(integral2-acc) != 0.0)
101  r = std::fabs(integral1-acc)/std::fabs(integral2-acc);
102  if (r == 0.0 || r > 1.0)
103  r = 1.0;
104  }
105 
106  if (relAccuracy_ != Null<Real>())
107  return std::min(absoluteAccuracy(), acc*relTol)/(r*QL_EPSILON);
108  else {
109  return absoluteAccuracy()/(r*QL_EPSILON);
110  }
111  }

113  Real GaussLobattoIntegral::adaptivGaussLobattoStep(
114  const ext::function<Real (Real)>& f,
115  Real a, Real b, Real fa, Real fb,
116  Real acc) const {
117  QL_REQUIRE(numberOfEvaluations() < maxEvaluations(),
118  "max number of iterations reached");
119 
120  const Real h=(b-a)/2;
121  const Real m=(a+b)/2;
122 
123  const Real mll=m-alpha_*h;
124  const Real ml =m-beta_*h;
125  const Real mr =m+beta_*h;
126  const Real mrr=m+alpha_*h;
127 
128  const Real fmll= f(mll);
129  const Real fml = f(ml);
130  const Real fm = f(m);
131  const Real fmr = f(mr);
132  const Real fmrr= f(mrr);
133  increaseNumberOfEvaluations(5);
134 
135  const Real integral2=(h/6)*(fa+fb+5*(fml+fmr));
136  const Real integral1=(h/1470)*(77*(fa+fb)
137  +432*(fmll+fmrr)+625*(fml+fmr)+672*fm);
138 
139  // avoid 80 bit logic on x86 cpu
140  volatile Real dist = acc + (integral1-integral2);
141  if(Real(dist)==acc || mll<=a || b<=mrr) {
142  QL_REQUIRE(m>a && b>m,"Interval contains no more machine number");
143  return integral1;
144  }
145  else {
146  return adaptivGaussLobattoStep(f,a,mll,fa,fmll,acc)
147  + adaptivGaussLobattoStep(f,mll,ml,fmll,fml,acc)
148  + adaptivGaussLobattoStep(f,ml,m,fml,fm,acc)
149  + adaptivGaussLobattoStep(f,m,mr,fm,fmr,acc)
150  + adaptivGaussLobattoStep(f,mr,mrr,fmr,fmrr,acc)
151  + adaptivGaussLobattoStep(f,mrr,b,fmrr,fb,acc);
152  }
153  }
