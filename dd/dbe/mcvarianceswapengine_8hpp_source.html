47  class MCVarianceSwapEngine : public VarianceSwap::engine,
48  public McSimulation<SingleVariate,RNG,S> {
49  public:
50  typedef
51  typename McSimulation<SingleVariate,RNG,S>::path_generator_type
52  path_generator_type;

54  typename McSimulation<SingleVariate,RNG,S>::path_pricer_type

58  // constructor
59  MCVarianceSwapEngine(
60  const ext::shared_ptr<GeneralizedBlackScholesProcess>& process,
61  Size timeSteps,
62  Size timeStepsPerYear,
63  bool brownianBridge,
64  bool antitheticVariate,
65  Size requiredSamples,
66  Real requiredTolerance,
67  Size maxSamples,
68  BigNatural seed);
69  // calculate variance via Monte Carlo
70  void calculate() const {
71  McSimulation<SingleVariate,RNG,S>::calculate(requiredTolerance_,
72  requiredSamples_,
73  maxSamples_);
74  results_.variance =
75  this->mcModel_->sampleAccumulator().mean();
76 
77  DiscountFactor riskFreeDiscount =
78  process_->riskFreeRate()->discount(arguments_.maturityDate);
79  Real multiplier;
80  switch (arguments_.position) {
81  case Position::Long:
82  multiplier = 1.0;
83  break;
84  case Position::Short:
85  multiplier = -1.0;
86  break;
87  default:
88  QL_FAIL("Unknown position");
89  }
90  multiplier *= riskFreeDiscount * arguments_.notional;
91 
92  results_.value =
93  multiplier * (results_.variance - arguments_.strike);
94 
95  if (RNG::allowsErrorEstimate) {
96  Real varianceError =
97  this->mcModel_->sampleAccumulator().errorEstimate();
98  results_.errorEstimate = multiplier * varianceError;
99  }
100  }
101  protected:
102  // McSimulation implementation
103  ext::shared_ptr<path_pricer_type> pathPricer() const;
104  TimeGrid timeGrid() const;
105 
106  ext::shared_ptr<path_generator_type> pathGenerator() const {
107 
108  Size dimensions = process_->factors();
109 
110  TimeGrid grid = timeGrid();
111  typename RNG::rsg_type gen =
112  RNG::make_sequence_generator(dimensions*(grid.size()-1),seed_);
113 
114  return ext::shared_ptr<path_generator_type>(
115  new path_generator_type(process_, grid, gen,
116  brownianBridge_));
117  }
118  // data members
119  ext::shared_ptr<GeneralizedBlackScholesProcess> process_;
120  Size timeSteps_, timeStepsPerYear_;
121  Size requiredSamples_, maxSamples_;
122  Real requiredTolerance_;
123  bool brownianBridge_;
124  BigNatural seed_;
125  };

199  template <class RNG, class S>
200  inline TimeGrid MCVarianceSwapEngine<RNG,S>::timeGrid() const {
201 
202  Time t = this->process_->time(this->arguments_.maturityDate);
203 
204  if (timeSteps_ != Null<Size>()) {
205  return TimeGrid(t, this->timeSteps_);
206  } else if (timeStepsPerYear_ != Null<Size>()) {
207  Size steps = static_cast<Size>(timeStepsPerYear_*t);
208  return TimeGrid(t, std::max<Size>(steps, 1));
209  } else {
210  QL_FAIL("time steps not specified");
211  }
212  }

338  inline Real VariancePathPricer::operator()(const Path& path) const {
339  QL_REQUIRE(path.length() > 0, "the path cannot be empty");
340  Time t0 = path.timeGrid().front();
341  Time t = path.timeGrid().back();
342  Time dt = path.timeGrid().dt(0);
343  SegmentIntegral integrator(static_cast<Size>(t/dt));
344  detail::Integrand f(path, process_);
345  return integrator(f,t0,t)/t;
346  }
