44  Real GBSMRNDCalculator::cdf(Real k, Time t) const {
45  const Handle<BlackVolTermStructure> volTS
46  = process_->blackVolatility();
47 
48  const Real dk = 1e-3*k;
49  const Real dvol_dk
50  = (volTS->blackVol(t, k+dk) - volTS->blackVol(t, k-dk)) / (2*dk);
51 
52  const DiscountFactor dR
53  = process_->riskFreeRate()->discount(t, true);
54  const DiscountFactor dD
55  = process_->dividendYield()->discount(t, true);
56 
57  const Real forward = process_->x0() * dD / dR;
58  const Real stdDev = std::sqrt(
59  process_->blackVolatility()->blackVariance(t, k, true));
60 
61  if (forward <= k) {
62  const BlackCalculator calc(Option::Call, k, forward, stdDev, dR);
63 
64  return 1.0 + ( calc.strikeSensitivity()
65  + calc.vega(t) * dvol_dk) /dR;
66  }
67  else {
68  const BlackCalculator calc(Option::Put, k, forward, stdDev, dR);
69 
70  return ( calc.strikeSensitivity()
71  + calc.vega(t) * dvol_dk) /dR;
72  }
73  }

75  Real GBSMRNDCalculator::invcdf(Real q, Time t) const {
76  const Real fwd = process_->x0()
77  / process_->riskFreeRate()->discount(t, true)
78  * process_->dividendYield()->discount(t, true);
79 
80  const Volatility atmVariance = std::sqrt(
81  process_->blackVolatility()->blackVariance(t, fwd, true));
82 
83  const Real atmX = InverseCumulativeNormal()(q);
84 
85  const Real guess = fwd*std::exp(atmVariance*atmX);
86 
87  Real lower = guess;
88  while (guess/lower < 65535.0 && cdf(lower, t) > q)
89  lower*=0.5;
90 
91  Real upper = guess;
92  while (upper/guess < 65535.0 && cdf(upper, t) < q) upper*=2;
93 
94  QL_REQUIRE(guess/lower < 65535.0 && upper/guess < 65535.0,
95  "Could not find an start interval with ("
96  << lower << ", " << upper << ") -> ("
97  << cdf(lower, t) << ", " << cdf(upper, t) << ")");
98 
99  return Brent().solve(
100  [&](Real _k){ return cdf(_k, t) - q; },
101  1e-10, 0.5*(lower+upper), lower, upper);
102  }
