47  void SwaptionVolCube2::performCalculations() const{
48 
49  SwaptionVolatilityCube::performCalculations();
51  for (Size i=0; i<nStrikes_; i++)
52  for (Size j=0; j<nOptionTenors_; j++)
53  for (Size k=0; k<nSwapTenors_; k++) {
54  volSpreadsMatrix_[i][j][k] =
55  volSpreads_[j*nSwapTenors_+k][i]->value();
56  }
58  for (Size i=0; i<nStrikes_; i++) {
59  volSpreadsInterpolator_[i] = BilinearInterpolation(
60  swapLengths_.begin(), swapLengths_.end(),
61  optionTimes_.begin(), optionTimes_.end(),
62  volSpreadsMatrix_[i]);
63  volSpreadsInterpolator_[i].enableExtrapolation();
64  }

72  Date optionDate = optionDateFromTime(optionTime);

84  boost::shared_ptr<SmileSection>
85  SwaptionVolCube2::smileSectionImpl(const Date& optionDate,
86  const Period& swapTenor) const {
87  calculate();
88  Rate atmForward = atmStrike(optionDate, swapTenor);
89  Volatility atmVol = atmVol_->volatility(optionDate,
90  swapTenor,
91  atmForward);
92  Time optionTime = timeFromReference(optionDate);
93  Real exerciseTimeSqrt = std::sqrt(optionTime);
94  std::vector<Real> strikes, stdDevs;
95  strikes.reserve(nStrikes_);
96  stdDevs.reserve(nStrikes_);
97  Time length = swapLength(swapTenor);
98  for (Size i=0; i<nStrikes_; ++i) {
99  strikes.push_back(atmForward + strikeSpreads_[i]);
100  stdDevs.push_back(exerciseTimeSqrt*(
101  atmVol + volSpreadsInterpolator_[i](length, optionTime)));
102  }
103  Real shift = atmVol_->shift(optionTime,length);
104  return boost::shared_ptr<SmileSection>(new
105  InterpolatedSmileSection<Linear>(optionTime,
106  strikes,
107  stdDevs,
108  atmForward,
109  Linear(),
110  Actual365Fixed(),
111  volatilityType(),
112  shift));
113  }
