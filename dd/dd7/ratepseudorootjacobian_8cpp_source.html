79  void RatePseudoRootJacobianNumerical::getBumps(const std::vector<Rate>& oldRates,
80  const std::vector<Real>& , // not used in the numerical implementation
81  const std::vector<Rate>& newRates,
82  const std::vector<Real>& gaussians,
83  Matrix& B)
84  {
85  Size numberRates = taus_.size();
86 
87  QL_REQUIRE(B.rows()==numberBumps_,
88  "B.rows()<> numberBumps_");
89 
90  QL_REQUIRE(B.columns()==taus_.size(),
91  "B.columns()<> number of rates");
92 
93  for (Size i =0; i < numberBumps_; ++i)
94  {
95  const Matrix& pseudo = pseudoBumped_[i];
96  driftsComputers_[i].compute(oldRates,
97  drifts_);
98 
99  for (Size j =0; j < aliveIndex_; ++j)
100  B[i][j]=0.0;
101 
102  for (Size j=aliveIndex_; j < numberRates; ++j)
103  {
104  bumpedRates_[j] = std::log(oldRates[j]+displacements_[j]);
105 
106  for (Size k=0; k < factors_; ++k)
107  bumpedRates_[j] += -0.5*pseudo[j][k]*pseudo[j][k];
108 
109  bumpedRates_[j] +=drifts_[j];
110 
111  for (Size k=0; k < factors_; ++k)
112  bumpedRates_[j] += pseudo[j][k]*gaussians[k];
113 
114  bumpedRates_[j] = std::exp(bumpedRates_[j]);
115  bumpedRates_[j] -= displacements_[j];
116  Real tmp = bumpedRates_[j] - newRates[j];
117 
118  B[i][j] = tmp;
119  }
120  }

167  void RatePseudoRootJacobian::getBumps(const std::vector<Rate>& oldRates,
168  const std::vector<Real>& discountRatios,
169  const std::vector<Rate>& newRates,
170  const std::vector<Real>& gaussians,
171  Matrix& B)
172  {
173  Size numberRates = taus_.size();
174 
175  QL_REQUIRE(B.rows() == numberBumps_, "we need B.rows() which is " << B.rows() << " to equal numberBumps_ which is " << numberBumps_);
176  QL_REQUIRE(B.columns() == numberRates, "we need B.columns() which is " << B.columns() << " to equal numberRates which is " << numberRates);
177 
178 
179  for (Size j=aliveIndex_; j < numberRates; ++j)
180  ratios_[j] = (oldRates[j] + displacements_[j])*discountRatios[j+1];
181 
182  for (Size f=0; f < factors_; ++f)
183  {
184  e_[aliveIndex_][f] = 0;
185 
186  for (Size j= aliveIndex_+1; j < numberRates; ++j)
187  e_[j][f] = e_[j-1][f] + ratios_[j-1]*pseudoRoot_[j-1][f];
188  }
189 
190 
191  for (Size f=0; f < factors_; ++f)
192  for (Size j=aliveIndex_; j < numberRates; ++j)
193  {
194  for (Size k= aliveIndex_; k < j ; ++k)
195  allDerivatives_[j][k][f] = newRates[j]*ratios_[k]*taus_[k]*pseudoRoot_[j][f];
196 
197  // GG don't seem to have the 2, this term is miniscule in any case
198  Real tmp = //2*
199  2*ratios_[j]*taus_[j]*pseudoRoot_[j][f];
200  tmp -= pseudoRoot_[j][f];
201  tmp += e_[j][f]*taus_[j];
202  tmp += gaussians[f];
203  tmp *= (newRates[j]+displacements_[j]);
204 
205 
206  allDerivatives_[j][j][f] =tmp;
207 
208  for (Size k= j+1; k < numberRates ; ++k)
209  allDerivatives_[j][k][f]=0.0;
210 
211 
212  }
213 
214 
215  for (Size i =0; i < numberBumps_; ++i)
216  {
217  Size j=0;
218 
219  for (; j < aliveIndex_; ++j)
220  {
221  B[i][j]=0.0;
222  }
223  for (; j < numberRates; ++j)
224  {
225  Real sum =0.0;
226 
227  for (Size k=aliveIndex_; k < numberRates; ++k)
228  for (Size f=0; f < factors_; ++f)
