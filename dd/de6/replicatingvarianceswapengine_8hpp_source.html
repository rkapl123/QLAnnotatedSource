94  inline void ReplicatingVarianceSwapEngine::computeOptionWeights(
95  const std::vector<Real>& availStrikes,
96  const Option::Type type,
97  weights_type& optionWeights) const {
98  if (availStrikes.empty())
99  return;
100 
101  std::vector<Real> strikes = availStrikes;
102 
103  // add end-strike for piecewise approximation
104  switch (type) {
105  case Option::Call:
106  std::sort(strikes.begin(), strikes.end());
107  strikes.push_back(strikes.back() + dk_);
108  break;
109  case Option::Put:
110  std::sort(strikes.begin(), strikes.end(), std::greater<Real>());
111  strikes.push_back(std::max(strikes.back() - dk_, 0.0));
112  break;
113  default:
114  QL_FAIL("invalid option type");
115  }
116 
117  // remove duplicate strikes
118  std::vector<Real>::iterator last =
119  std::unique(strikes.begin(), strikes.end());
120  strikes.erase(last, strikes.end());
121 
122  // compute weights
123  Real f = strikes.front();
124  Real slope, prevSlope = 0.0;
125 
126 
127 
128 
129  for (std::vector<Real>::const_iterator k=strikes.begin();
130  // added end-strike discarded
131  k<strikes.end()-1;
132  ++k) {
133  slope = std::fabs((computeLogPayoff(*(k+1), f) -
134  computeLogPayoff(*k, f))/
135  (*(k+1) - *k));
136  ext::shared_ptr<StrikedTypePayoff> payoff(
137  new PlainVanillaPayoff(type, *k));
138  if ( k == strikes.begin() )
139  optionWeights.push_back(std::make_pair(payoff,slope));
140  else
141  optionWeights.push_back(
142  std::make_pair(payoff, slope - prevSlope));
143  prevSlope = slope;
144  }
145  }

148  inline Real ReplicatingVarianceSwapEngine::computeLogPayoff(
149  const Real strike,
150  const Real callPutStrikeBoundary) const {
151  Real f = callPutStrikeBoundary;
152  return (2.0/residualTime()) * (((strike - f)/f) - std::log(strike/f));
153  }

156  inline
157  Real ReplicatingVarianceSwapEngine::computeReplicatingPortfolio(
158  const weights_type& optionWeights) const {
159 
160  ext::shared_ptr<Exercise> exercise(
161  new EuropeanExercise(arguments_.maturityDate));
162  ext::shared_ptr<PricingEngine> optionEngine(
163  new AnalyticEuropeanEngine(process_));
164  Real optionsValue = 0.0;
165 
166  for (weights_type::const_iterator i = optionWeights.begin();
167  i < optionWeights.end(); ++i) {
168  ext::shared_ptr<StrikedTypePayoff> payoff = i->first;
169  EuropeanOption option(payoff, exercise);
170  option.setPricingEngine(optionEngine);
171  Real weight = i->second;
172  optionsValue += option.NPV() * weight;
173  }
174 
175  Real f = optionWeights.front().first->strike();
176  return 2.0 * riskFreeRate() -
177  2.0/residualTime() *
178  (((underlying()/riskFreeDiscount() - f)/f) +
179  std::log(f/underlying())) +
180  optionsValue/riskFreeDiscount();
181  }

184  // calculate variance via replicating portfolio
185  inline void ReplicatingVarianceSwapEngine::calculate() const {
186  weights_type optionWeigths;
187  computeOptionWeights(callStrikes_, Option::Call, optionWeigths);
188  computeOptionWeights(putStrikes_, Option::Put, optionWeigths);
189 
190  results_.variance = computeReplicatingPortfolio(optionWeigths);
191 
192  DiscountFactor riskFreeDiscount =
193  process_->riskFreeRate()->discount(arguments_.maturityDate);
194  Real multiplier;
195  switch (arguments_.position) {
196  case Position::Long:
197  multiplier = 1.0;
198  break;
199  case Position::Short:
200  multiplier = -1.0;
201  break;
202  default:
203  QL_FAIL("Unknown position");
204  }
205  results_.value = multiplier * riskFreeDiscount * arguments_.notional *
206  (results_.variance - arguments_.strike);
207 
208  results_.additionalResults["optionWeights"] = optionWeigths;
209  }
