53  void FdG2SwaptionEngine::calculate() const {
54 
55  // 1. Term structure
56  const Handle<YieldTermStructure> ts = model_->termStructure();
57 
58  // 2. Mesher
59  const DayCounter dc = ts->dayCounter();
60  const Date referenceDate = ts->referenceDate();
61  const Time maturity = dc.yearFraction(referenceDate,
62  arguments_.exercise->lastDate());
63 
64  const ext::shared_ptr<OrnsteinUhlenbeckProcess> process1(
65  new OrnsteinUhlenbeckProcess(model_->a(), model_->sigma()));
66 
67  const ext::shared_ptr<OrnsteinUhlenbeckProcess> process2(
68  new OrnsteinUhlenbeckProcess(model_->b(), model_->eta()));
69 
70  const ext::shared_ptr<Fdm1dMesher> xMesher(
71  new FdmSimpleProcess1dMesher(xGrid_,process1,maturity,1,invEps_));
72 
73  const ext::shared_ptr<Fdm1dMesher> yMesher(
74  new FdmSimpleProcess1dMesher(yGrid_,process2,maturity,1,invEps_));
75 
76  const ext::shared_ptr<FdmMesher> mesher(
77  new FdmMesherComposite(xMesher, yMesher));
78 
79  // 3. Inner Value Calculator
80  const std::vector<Date>& exerciseDates = arguments_.exercise->dates();
81  std::map<Time, Date> t2d;
82 
83  for (Size i=0; i < exerciseDates.size(); ++i) {
84  const Time t = dc.yearFraction(referenceDate, exerciseDates[i]);
85  QL_REQUIRE(t >= 0, "exercise dates must not contain past date");
86 
87  t2d[t] = exerciseDates[i];
88  }
89 
90  const Handle<YieldTermStructure> disTs = model_->termStructure();
91  const Handle<YieldTermStructure> fwdTs
92  = arguments_.swap->iborIndex()->forwardingTermStructure();
93 
94  QL_REQUIRE(fwdTs->dayCounter() == disTs->dayCounter(),
95  "day counter of forward and discount curve must match");
96  QL_REQUIRE(fwdTs->referenceDate() == disTs->referenceDate(),
97  "reference date of forward and discount curve must match");
98 
99  const ext::shared_ptr<G2> fwdModel(
100  new G2(fwdTs, model_->a(), model_->sigma(),
101  model_->b(), model_->eta(), model_->rho()));
102 
103  const ext::shared_ptr<FdmInnerValueCalculator> calculator(
104  new FdmAffineModelSwapInnerValue<G2>(
105  model_.currentLink(), fwdModel,
106  arguments_.swap, t2d, mesher, 0));
107 
108  // 4. Step conditions
109  const ext::shared_ptr<FdmStepConditionComposite> conditions =
110  FdmStepConditionComposite::vanillaComposite(
111  DividendSchedule(), arguments_.exercise,
112  mesher, calculator, referenceDate, dc);
113 
114  // 5. Boundary conditions
115  const FdmBoundaryConditionSet boundaries;
116 
117  // 6. Solver
118  FdmSolverDesc solverDesc = { mesher, boundaries, conditions,
119  calculator, maturity,
120  tGrid_, dampingSteps_ };
121 
122  const boost::scoped_ptr<FdmG2Solver> solver(
123  new FdmG2Solver(model_, solverDesc, schemeDesc_));
124 
125  results_.value = solver->valueAt(0.0, 0.0);
126  }
127 }
