103 void CdsHelper::initializeDates() {
104
105 protectionStart_ = evaluationDate_ + settlementDays_;
106
107 Date startDate = startDate_ == Date() ? protectionStart_ : startDate_;
108 // Only adjust start date if rule is not CDS or CDS2015. Unsure about OldCDS.
109 if (rule_ != DateGeneration::CDS && rule_ != DateGeneration::CDS2015) {
110 startDate = calendar_.adjust(startDate, paymentConvention_);
111 }
112
113 Date endDate;
114 if (rule_ == DateGeneration::CDS2015 || rule_ == DateGeneration::CDS || rule_ == DateGeneration::OldCDS) {
115 Date refDate = startDate_ == Date() ? evaluationDate_ : startDate_;
116 endDate = cdsMaturity(refDate, tenor_, rule_);
117 } else {
118 // Keep the old logic here
119 Date refDate = startDate_ == Date() ? protectionStart_ : startDate_ + settlementDays_;
120 endDate = refDate + tenor_;
121 }
122
123 schedule_ =
124 MakeSchedule().from(startDate)
125 .to(endDate)
126 .withFrequency(frequency_)
127 .withCalendar(calendar_)
128 .withConvention(paymentConvention_)
129 .withTerminationDateConvention(Unadjusted)
130 .withRule(rule_);
131 earliestDate_ = schedule_.dates().front();
132 latestDate_ = calendar_.adjust(schedule_.dates().back(),
133 paymentConvention_);
134 if (model_ == CreditDefaultSwap::ISDA)
135 ++latestDate_;
136 }

187 void SpreadCdsHelper::resetEngine() {
188 swap_ = ext::shared_ptr<CreditDefaultSwap>(new CreditDefaultSwap(
189 Protection::Buyer, 100.0, 0.01, schedule_, paymentConvention_,
190 dayCounter_, settlesAccrual_, paysAtDefaultTime_, protectionStart_,
191 ext::shared_ptr<Claim>(), lastPeriodDC_, rebatesAccrual_, evaluationDate_));
192
193 switch (model_) {
194 case CreditDefaultSwap::ISDA:
195 swap_->setPricingEngine(ext::make_shared<IsdaCdsEngine>(
196 probability_, recoveryRate_, discountCurve_, false,
197 IsdaCdsEngine::Taylor, IsdaCdsEngine::HalfDayBias,
198 IsdaCdsEngine::Piecewise));
199 break;
200 case CreditDefaultSwap::Midpoint:
201 swap_->setPricingEngine(ext::make_shared<MidPointCdsEngine>(
202 probability_, recoveryRate_, discountCurve_));
203 break;
204 default:
205 QL_FAIL("unknown CDS pricing model: " << model_);
206 }
207 }
