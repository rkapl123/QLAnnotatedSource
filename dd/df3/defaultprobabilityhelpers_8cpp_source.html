97  void CdsHelper::initializeDates() {
98  protectionStart_ = evaluationDate_ + settlementDays_;
99  Date startDate, endDate;
100  if(startDate_ == Date()) {
101  startDate = calendar_.adjust(protectionStart_,
102  paymentConvention_);
103  if (rule_ == DateGeneration::CDS || rule_ == DateGeneration::CDS2015) { // for standard CDS ..
104  // .. the start date is not adjusted
105  startDate = protectionStart_;
106  }
107  // .. and (in any case) the end date rolls by 3 month as
108  // soon as the trade date falls on an IMM date,
109  // or the March or September IMM date in case of the CDS2015 rule.
110  endDate = protectionStart_ + tenor_;
111 
112  } else {
113  if(!schedule_.empty()) return; //no need to update schedule
114  startDate = calendar_.adjust(startDate_, paymentConvention_);
115  endDate = startDate_ + settlementDays_ + tenor_;
116  }
117  schedule_ =
118  MakeSchedule().from(startDate)
119  .to(endDate)
120  .withFrequency(frequency_)
121  .withCalendar(calendar_)
122  .withConvention(paymentConvention_)
123  .withTerminationDateConvention(Unadjusted)
124  .withRule(rule_);
125  earliestDate_ = schedule_.dates().front();
126  latestDate_ = calendar_.adjust(schedule_.dates().back(),
127  paymentConvention_);
128  if (model_ == CreditDefaultSwap::ISDA)
129  ++latestDate_;
130  }

181  void SpreadCdsHelper::resetEngine() {
182  swap_ = ext::shared_ptr<CreditDefaultSwap>(new CreditDefaultSwap(
183  Protection::Buyer, 100.0, 0.01, schedule_, paymentConvention_,
184  dayCounter_, settlesAccrual_, paysAtDefaultTime_, protectionStart_,
185  ext::shared_ptr<Claim>(), lastPeriodDC_, rebatesAccrual_));
186 
187  switch (model_) {
188  case CreditDefaultSwap::ISDA:
189  swap_->setPricingEngine(ext::make_shared<IsdaCdsEngine>(
190  probability_, recoveryRate_, discountCurve_, false,
191  IsdaCdsEngine::Taylor, IsdaCdsEngine::HalfDayBias,
192  IsdaCdsEngine::Piecewise));
193  break;
194  case CreditDefaultSwap::Midpoint:
195  swap_->setPricingEngine(ext::make_shared<MidPointCdsEngine>(
196  probability_, recoveryRate_, discountCurve_));
197  break;
198  default:
199  QL_FAIL("unknown CDS pricing model: " << model_);
200  }
201  }
