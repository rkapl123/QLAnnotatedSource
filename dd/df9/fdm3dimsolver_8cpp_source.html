33  Fdm3DimSolver::Fdm3DimSolver(
34  const FdmSolverDesc& solverDesc,
35  const FdmSchemeDesc& schemeDesc,
36  const boost::shared_ptr<FdmLinearOpComposite>& op)
37  : solverDesc_(solverDesc),
38  schemeDesc_(schemeDesc),
39  op_(op),
40  thetaCondition_(boost::make_shared<FdmSnapshotCondition>(
41  0.99*std::min(1.0/365.0,
42  solverDesc.condition->stoppingTimes().empty()
43  ? solverDesc.maturity :
44  solverDesc.condition->stoppingTimes().front()))),
45  conditions_(FdmStepConditionComposite::joinConditions(thetaCondition_,
46  solverDesc.condition)),
47  initialValues_(solverDesc.mesher->layout()->size()),
48  resultValues_ (solverDesc.mesher->layout()->dim()[2],
49  Matrix(solverDesc.mesher->layout()->dim()[1],
50  solverDesc.mesher->layout()->dim()[0])),
51  interpolation_(solverDesc.mesher->layout()->dim()[2]) {
52 
53  const boost::shared_ptr<FdmMesher> mesher = solverDesc.mesher;
54  const boost::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
55 
56  x_.reserve(layout->dim()[0]);
57  y_.reserve(layout->dim()[1]);
58  z_.reserve(layout->dim()[2]);
59 
60  const FdmLinearOpIterator endIter = layout->end();
61  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
62  ++iter) {
63  initialValues_[iter.index()]
64  = solverDesc.calculator->avgInnerValue(iter,
65  solverDesc.maturity);
66 
67 
68  if (!iter.coordinates()[1] && !iter.coordinates()[2]) {
69  x_.push_back(mesher->location(iter, 0));
70  }
71  if (!iter.coordinates()[0] && !iter.coordinates()[2]) {
72  y_.push_back(mesher->location(iter, 1));
73  }
74  if (!iter.coordinates()[0] && !iter.coordinates()[1]) {
75  z_.push_back(mesher->location(iter, 2));
76  }
77  }
78  }

80  void Fdm3DimSolver::performCalculations() const {
81  Array rhs(initialValues_.size());
82  std::copy(initialValues_.begin(), initialValues_.end(), rhs.begin());
83 
84  FdmBackwardSolver(op_, solverDesc_.bcSet, conditions_, schemeDesc_)
85  .rollback(rhs, solverDesc_.maturity, 0.0,
86  solverDesc_.timeSteps, solverDesc_.dampingSteps);
87 
88  for (Size i=0; i < z_.size(); ++i) {
89  std::copy(rhs.begin()+i *y_.size()*x_.size(),
90  rhs.begin()+(i+1)*y_.size()*x_.size(),
91  resultValues_[i].begin());
92 
93  interpolation_[i] = boost::make_shared<BicubicSpline>(x_.begin(), x_.end(),
94  y_.begin(), y_.end(),
95  resultValues_[i]);
96  }
97  }

99  Real Fdm3DimSolver::interpolateAt(Real x, Real y, Rate z) const {

102  Array zArray(z_.size());

110  Real Fdm3DimSolver::thetaAt(Real x, Real y, Rate z) const {
111  QL_REQUIRE(conditions_->stoppingTimes().front() > 0.0,
112  "stopping time at zero-> can't calculate theta");
113  calculate();
114 
115  const Array& rhs = thetaCondition_->getValues();
116  std::vector<Matrix> thetaValues(z_.size(), Matrix(y_.size(),x_.size()));
117  for (Size i=0; i < z_.size(); ++i) {
118  std::copy(rhs.begin()+i *y_.size()*x_.size(),
119  rhs.begin()+(i+1)*y_.size()*x_.size(),
120  thetaValues[i].begin());
121  }
122 
123  Array zArray(z_.size());
124  for (Size i=0; i < z_.size(); ++i) {
125  zArray[i] = BicubicSpline(x_.begin(),x_.end(),
126  y_.begin(),y_.end(), thetaValues[i])(x,y);
127  }
128 
129  return (MonotonicCubicNaturalSpline(z_.begin(), z_.end(),
130  zArray.begin())(z)
131  - interpolateAt(x, y, z)) / thetaCondition_->getTime();
132  }
