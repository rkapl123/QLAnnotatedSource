32  Fdm3DimSolver::Fdm3DimSolver(
33  const FdmSolverDesc& solverDesc,
34  const FdmSchemeDesc& schemeDesc,
35  const ext::shared_ptr<FdmLinearOpComposite>& op)
36  : solverDesc_(solverDesc),
37  schemeDesc_(schemeDesc),
38  op_(op),
39  thetaCondition_(ext::make_shared<FdmSnapshotCondition>(
40  0.99*std::min(1.0/365.0,
41  solverDesc.condition->stoppingTimes().empty()
42  ? solverDesc.maturity :
43  solverDesc.condition->stoppingTimes().front()))),
44  conditions_(FdmStepConditionComposite::joinConditions(thetaCondition_,
45  solverDesc.condition)),
46  initialValues_(solverDesc.mesher->layout()->size()),
47  resultValues_ (solverDesc.mesher->layout()->dim()[2],
48  Matrix(solverDesc.mesher->layout()->dim()[1],
49  solverDesc.mesher->layout()->dim()[0])),
50  interpolation_(solverDesc.mesher->layout()->dim()[2]) {
51 
52  const ext::shared_ptr<FdmMesher> mesher = solverDesc.mesher;
53  const ext::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
54 
55  x_.reserve(layout->dim()[0]);
56  y_.reserve(layout->dim()[1]);
57  z_.reserve(layout->dim()[2]);
58 
59  const FdmLinearOpIterator endIter = layout->end();
60  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
61  ++iter) {
62  initialValues_[iter.index()]
63  = solverDesc.calculator->avgInnerValue(iter,
64  solverDesc.maturity);
65 
66 
67  if ((iter.coordinates()[1] == 0U) && (iter.coordinates()[2] == 0U)) {
68  x_.push_back(mesher->location(iter, 0));
69  }
70  if ((iter.coordinates()[0] == 0U) && (iter.coordinates()[2] == 0U)) {
71  y_.push_back(mesher->location(iter, 1));
72  }
73  if ((iter.coordinates()[0] == 0U) && (iter.coordinates()[1] == 0U)) {
74  z_.push_back(mesher->location(iter, 2));
75  }
76  }
77  }

79  void Fdm3DimSolver::performCalculations() const {
80  Array rhs(initialValues_.size());
81  std::copy(initialValues_.begin(), initialValues_.end(), rhs.begin());
82 
83  FdmBackwardSolver(op_, solverDesc_.bcSet, conditions_, schemeDesc_)
84  .rollback(rhs, solverDesc_.maturity, 0.0,
85  solverDesc_.timeSteps, solverDesc_.dampingSteps);
86 
87  for (Size i=0; i < z_.size(); ++i) {
88  std::copy(rhs.begin()+i *y_.size()*x_.size(),
89  rhs.begin()+(i+1)*y_.size()*x_.size(),
90  resultValues_[i].begin());
91 
92  interpolation_[i] = ext::make_shared<BicubicSpline>(x_.begin(), x_.end(),
93  y_.begin(), y_.end(),
94  resultValues_[i]);
95  }
96  }

109  Real Fdm3DimSolver::thetaAt(Real x, Real y, Rate z) const {
110  if (conditions_->stoppingTimes().front() == 0.0)
111  return Null<Real>();
112 
113  calculate();
114 
115  const Array& rhs = thetaCondition_->getValues();
116  std::vector<Matrix> thetaValues(z_.size(), Matrix(y_.size(),x_.size()));
117  for (Size i=0; i < z_.size(); ++i) {
118  std::copy(rhs.begin()+i *y_.size()*x_.size(),
119  rhs.begin()+(i+1)*y_.size()*x_.size(),
120  thetaValues[i].begin());
121  }
122 
123  Array zArray(z_.size());
124  for (Size i=0; i < z_.size(); ++i) {
125  zArray[i] = BicubicSpline(x_.begin(),x_.end(),
126  y_.begin(),y_.end(), thetaValues[i])(x,y);
127  }
128 
129  return (MonotonicCubicNaturalSpline(z_.begin(), z_.end(),
130  zArray.begin())(z)
131  - interpolateAt(x, y, z)) / thetaCondition_->getTime();
132  }
