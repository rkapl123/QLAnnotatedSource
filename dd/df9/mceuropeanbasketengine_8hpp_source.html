174  template <class RNG, class S>
175  inline TimeGrid MCEuropeanBasketEngine<RNG,S>::timeGrid() const {
176 
177  Time residualTime = processes_->time(
178  this->arguments_.exercise->lastDate());
179  if (timeSteps_ != Null<Size>()) {
180  return TimeGrid(residualTime, timeSteps_);
181  } else if (timeStepsPerYear_ != Null<Size>()) {
182  Size steps = static_cast<Size>(timeStepsPerYear_*residualTime);
183  return TimeGrid(residualTime, std::max<Size>(steps, 1));
184  } else {
185  QL_FAIL("time steps not specified");
186  }
187  }

189  template <class RNG, class S>
190  inline
191  boost::shared_ptr<typename MCEuropeanBasketEngine<RNG,S>::path_pricer_type>
192  MCEuropeanBasketEngine<RNG,S>::pathPricer() const {
193 
194  boost::shared_ptr<BasketPayoff> payoff =
195  boost::dynamic_pointer_cast<BasketPayoff>(arguments_.payoff);
196  QL_REQUIRE(payoff, "non-basket payoff given");
197 
198  boost::shared_ptr<GeneralizedBlackScholesProcess> process =
199  boost::dynamic_pointer_cast<GeneralizedBlackScholesProcess>(
200  processes_->process(0));
201  QL_REQUIRE(process, "Black-Scholes process required");
202 
203  return boost::shared_ptr<
204  typename MCEuropeanBasketEngine<RNG,S>::path_pricer_type>(
205  new EuropeanMultiPathPricer(payoff,
206  process->riskFreeRate()->discount(
207  arguments_.exercise->lastDate())));
208  }
