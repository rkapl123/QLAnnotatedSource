39  CumulativeBehrensFisher::CumulativeBehrensFisher(const std::vector<Integer>& degreesFreedom,
40  const std::vector<Real>& factors)
41  : degreesFreedom_(degreesFreedom), factors_(factors), polyConvolved_(std::vector<Real>(1, 1.))
42 
43  {
44  QL_REQUIRE(degreesFreedom.size() == factors.size(),
45  "Incompatible sizes in convolution.");
46  for (int i : degreesFreedom) {
47  QL_REQUIRE(i % 2 != 0, "Even degree of freedom not allowed");
48  QL_REQUIRE(i >= 0, "Negative degree of freedom not allowed");
49  }
50  for(Size i=0; i<degreesFreedom_.size(); i++)
51  polynCharFnc_.push_back(polynCharactT((degreesFreedom[i]-1)/2));
52  // adjust the polynomial coefficients by the factors in the linear
53  // combination:
54  for(Size i=0; i<degreesFreedom_.size(); i++) {
55  Real multiplier = 1.;
56  for(Size k=1; k<polynCharFnc_[i].size(); k++) {
57  multiplier *= std::abs(factors_[i]);
58  polynCharFnc_[i][k] *= multiplier;
59  }
60  }
61  //convolution, here it is a product of polynomials and exponentials
62  for (auto& i : polynCharFnc_)
63  polyConvolved_ = convolveVectorPolynomials(polyConvolved_, i);
64  // trim possible zeros that might have arised:
65  auto it = polyConvolved_.rbegin();
66  while (it != polyConvolved_.rend()) {
67  if (*it == 0.) {
68  polyConvolved_.pop_back();
69  it = polyConvolved_.rbegin();
70  }else{
71  break;
72  }
73  }
74  // cache 'a' value (the exponent)
75  for(Size i=0; i<degreesFreedom_.size(); i++)
76  a_ += std::sqrt(static_cast<Real>(degreesFreedom_[i]))
77  * std::abs(factors_[i]);
78  a2_ = a_ * a_;
79  }

81  Disposable<std::vector<Real> >
82  CumulativeBehrensFisher::polynCharactT(Natural n) const {
83  Natural nu = 2 * n +1;
84  std::vector<Real> low(1,1.), high(1,1.);
85  high.push_back(std::sqrt(static_cast<Real>(nu)));
86  if(n==0) return low;
87  if(n==1) return high;
88 
89  for(Size k=1; k<n; k++) {
90  std::vector<Real> recursionFactor(1,0.); // 0 coef
91  recursionFactor.push_back(0.); // 1 coef
92  recursionFactor.push_back(nu/((2.*k+1.)*(2.*k-1.))); // 2 coef
93  std::vector<Real> lowUp =
94  convolveVectorPolynomials(recursionFactor, low);
95  //add them up:
96  for(Size i=0; i<high.size(); i++)
97  lowUp[i] += high[i];
98  low = high;
99  high = lowUp;
100  }
101  return high;
102  }

104  Disposable<std::vector<Real> >
105  CumulativeBehrensFisher::convolveVectorPolynomials(
106  const std::vector<Real>& v1,
107  const std::vector<Real>& v2) const {
108  #if defined(QL_EXTRA_SAFETY_CHECKS)
109  QL_REQUIRE(!v1.empty() && !v2.empty(),
110  "Incorrect vectors in polynomial.");
111  #endif
112 
113  const std::vector<Real>& shorter = v1.size() < v2.size() ? v1 : v2;
114  const std::vector<Real>& longer = (v1 == shorter) ? v2 : v1;
115 
116  Size newDegree = v1.size()+v2.size()-2;
117  std::vector<Real> resultB(newDegree+1, 0.);
118  for(Size polyOrdr=0; polyOrdr<resultB.size(); polyOrdr++) {
119  for(Size i=std::max<Integer>(0, polyOrdr-longer.size()+1);
120  i<=std::min(polyOrdr, shorter.size()-1); i++)
121  resultB[polyOrdr] += shorter[i]*longer[polyOrdr-i];
122  }
123  return resultB;
124  }

126  Probability CumulativeBehrensFisher::operator()(const Real x) const {
127  // 1st & 0th terms with the table integration
128  Real integral = polyConvolved_[0] * std::atan(x/a_);
129  Real squared = a2_ + x*x;
130  Real rootsqr = std::sqrt(squared);
131  Real atan2xa = std::atan2(-x,a_);
132  if(polyConvolved_.size()>1)
133  integral += polyConvolved_[1] * x/squared;
134 
135  for(Size exponent = 2; exponent <polyConvolved_.size(); exponent++) {
136  integral -= polyConvolved_[exponent] *
137  Factorial::get(exponent-1) * std::sin((exponent)*atan2xa)
138  /std::pow(rootsqr, static_cast<Real>(exponent));
139  }
140  return .5 + integral / M_PI;
141  }

143  Probability
144  CumulativeBehrensFisher::density(const Real x) const {
145  Real squared = a2_ + x*x;
146  Real integral = polyConvolved_[0] * a_ / squared;
147  Real rootsqr = std::sqrt(squared);
148  Real atan2xa = std::atan2(-x,a_);
149  for(Size exponent=1; exponent <polyConvolved_.size(); exponent++) {
150  integral += polyConvolved_[exponent] *
151  Factorial::get(exponent) * std::cos((exponent+1)*atan2xa)
152  /std::pow(rootsqr, static_cast<Real>(exponent+1) );
153  }
154  return integral / M_PI;
155  }

167  Real InverseCumulativeBehrensFisher::operator()(const Probability q) const {
168  using namespace ext::placeholders;
169 
170  Probability effectiveq;
171  Real sign;
172  // since the distrib is symmetric solve only on the right side:
173  if(q==0.5) {
174  return 0.;
175  }else if(q < 0.5) {
176  sign = -1.;
177  effectiveq = 1.-q;
178  }else{
179  sign = 1.;
180  effectiveq = q;
181  }
182  Real xMin =
183  InverseCumulativeNormal::standard_value(effectiveq) * normSqr_;
184  // inversion will fail at the Brent's bounds-check if this is not enough
185  // (q is very close to 1.), in a bad combination fails around 1.-1.e-7
186  Real xMax = 1.e6;
187  return sign *
188  Brent().solve(ext::bind(subtract<Real>(effectiveq),
189  ext::bind(
190  &CumulativeBehrensFisher::operator(),
191  distrib_, _1)), accuracy_, (xMin+xMax)/2., xMin, xMax);
192  }
