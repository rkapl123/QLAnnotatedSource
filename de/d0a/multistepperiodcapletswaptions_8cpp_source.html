77  bool MultiStepPeriodCapletSwaptions::nextTimeStep(
78  const CurveState& currentState,
79  std::vector<Size>& numberCashFlowsThisStep,
80  std::vector<std::vector<MarketModelMultiProduct::CashFlow> >&
81  genCashFlows)
82  {
83 
84  for (Size i=0; i < numberCashFlowsThisStep.size(); ++i)
85  numberCashFlowsThisStep[i]=0UL;
86 
87  if (currentIndex_ >=offset_ && (currentIndex_ - offset_) % period_ ==0)
88  {
89  // caplet first
90  Real df = currentState.discountRatio(currentIndex_ + period_, currentIndex_);
91  Time tau = rateTimes_[currentIndex_+period_]- rateTimes_[currentIndex_];
92  Real forward = (1.0/df-1.0)/tau;
93  Real value = (*forwardPayOffs_[productIndex_])(forward);
94  value *= tau*currentState.discountRatio(currentIndex_+period_,currentIndex_);
95 
96  if (value >0)
97  {
98  numberCashFlowsThisStep[productIndex_]=1UL;
99  genCashFlows[productIndex_][0].amount = value;
100  genCashFlows[productIndex_][0].timeIndex= productIndex_;
101  }
102 
103  // now swaption
104 
105  unsigned long numberPeriods = numberBigFRAs_ - productIndex_;
106  Real B = 0.0;
107  double P0 = 1.0; // i.e currentState.discountRatio(currentIndex_,currentIndex_);
108  Real Pn = currentState.discountRatio(currentIndex_ + numberPeriods*period_, currentIndex_);
109  for (unsigned long i=0; i < numberPeriods; ++i)
110  {
111  Time tau = rateTimes_[currentIndex_+(i+1)*period_]- rateTimes_[currentIndex_+i*period_];
112  B+= tau*currentState.discountRatio(currentIndex_+(i+1)*period_,currentIndex_);
113  }
114 
115 
116  Real swapRate = (P0-Pn)/B;
117 
118  Real swaptionValue= (*swapPayOffs_[productIndex_])(swapRate);
119  swaptionValue *=B;
120 
121  if (swaptionValue >0)
122  {
123  numberCashFlowsThisStep[productIndex_+numberBigFRAs_]=1UL;
124  genCashFlows[productIndex_+numberBigFRAs_][0].amount = swaptionValue;
125  genCashFlows[productIndex_+numberBigFRAs_][0].timeIndex=productIndex_+numberBigFRAs_;
126  }
127 
128  ++productIndex_;
129 
130  }
131 
132  ++currentIndex_;
133 
134  bool terminate =(productIndex_ >= numberBigFRAs_);
135 
136 
137  return terminate;
138  }
