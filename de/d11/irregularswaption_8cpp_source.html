49  IrregularImpliedVolHelper::IrregularImpliedVolHelper(
50  const IrregularSwaption& swaption,
51  const Handle<YieldTermStructure>& discountCurve,
52  Real targetValue)
53  : discountCurve_(discountCurve), targetValue_(targetValue) {
54 
55  vol_ = boost::shared_ptr<SimpleQuote>(new SimpleQuote(-1.0));
56  Handle<Quote> h(vol_);
57  engine_ = boost::shared_ptr<PricingEngine>(new
58  BlackSwaptionEngine(discountCurve_, h));
59  swaption.setupArguments(engine_->getArguments());
60 
61  results_ =
62  dynamic_cast<const Instrument::results*>(engine_->getResults());
63  }

65  Real IrregularImpliedVolHelper::operator()(Volatility x) const {
66  if (x!=vol_->value()) {
67  vol_->setValue(x);
68  engine_->calculate();
69  }
70  return results_->value-targetValue_;
71  }

73  Real IrregularImpliedVolHelper::derivative(Volatility x) const {
74  if (x!=vol_->value()) {
75  vol_->setValue(x);
76  engine_->calculate();
77  }
78  std::map<std::string,boost::any>::const_iterator vega_ =
79  results_->additionalResults.find("vega");
80  QL_REQUIRE(vega_ != results_->additionalResults.end(),
81  "vega not provided");
82  return boost::any_cast<Real>(vega_->second);
83  }
84  }
