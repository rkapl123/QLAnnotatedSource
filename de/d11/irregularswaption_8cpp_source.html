35 class IrregularImpliedVolHelper {
36 public:
37 IrregularImpliedVolHelper(const IrregularSwaption&,
38 Handle<YieldTermStructure> discountCurve,
39 Real targetValue);
40 Real operator()(Volatility x) const;
41 Real derivative(Volatility x) const;
42 private:
43 ext::shared_ptr<PricingEngine> engine_;
44 Handle<YieldTermStructure> discountCurve_;
45 Real targetValue_;
46 ext::shared_ptr<SimpleQuote> vol_;
47 const Instrument::results* results_;
48 };

50 IrregularImpliedVolHelper::IrregularImpliedVolHelper(
51 const IrregularSwaption& swaption,
52 Handle<YieldTermStructure> discountCurve,
53 Real targetValue)
54 : discountCurve_(std::move(discountCurve)), targetValue_(targetValue),
55 vol_(ext::make_shared<SimpleQuote>(-1.0)) {
56
57 Handle<Quote> h(vol_);
58 engine_ = ext::shared_ptr<PricingEngine>(new
59 BlackSwaptionEngine(discountCurve_, h));
60 swaption.setupArguments(engine_->getArguments());
61
62 results_ =
63 dynamic_cast<const Instrument::results*>(engine_->getResults());
64 }
65
66 Real IrregularImpliedVolHelper::operator()(Volatility x) const {
67 if (x!=vol_->value()) {
68 vol_->setValue(x);
69 engine_->calculate();
70 }
71 return results_->value-targetValue_;
72 }
73
74 Real IrregularImpliedVolHelper::derivative(Volatility x) const {
75 if (x!=vol_->value()) {
76 vol_->setValue(x);
77 engine_->calculate();
78 }
79 auto vega_ = results_->additionalResults.find("vega");
80 QL_REQUIRE(vega_ != results_->additionalResults.end(),
81 "vega not provided");
82 return boost::any_cast<Real>(vega_->second);
83 }
84 }

110 void IrregularSwaption::setupArguments(PricingEngine::arguments* args) const {
111
112 swap_->setupArguments(args);
113
114 auto* arguments = dynamic_cast<IrregularSwaption::arguments*>(args);

129 Volatility IrregularSwaption::impliedVolatility(
130 Real targetValue,
131 const Handle<YieldTermStructure>& discountCurve,
132 Volatility guess,
133 Real accuracy,
134 Natural maxEvaluations,
135 Volatility minVol,
136 Volatility maxVol) const {
137 calculate();
138 QL_REQUIRE(!isExpired(), "instrument expired");
139
140 IrregularImpliedVolHelper f(*this, discountCurve, targetValue);
141 //Brent solver;
142 NewtonSafe solver;
143 solver.setMaxEvaluations(maxEvaluations);
144 return solver.solve(f, accuracy, guess, minVol, maxVol);
145 }
