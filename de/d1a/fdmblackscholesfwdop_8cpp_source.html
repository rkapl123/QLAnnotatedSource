51  void FdmBlackScholesFwdOp::setTime(Time t1, Time t2) {
52  const Rate r = rTS_->forwardRate(t1, t2, Continuous).rate();
53  const Rate q = qTS_->forwardRate(t1, t2, Continuous).rate();
54 
55  if (localVol_) {
56  const boost::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
57  const FdmLinearOpIterator endIter = layout->end();
58 
59  Array v(layout->size());
60  for (FdmLinearOpIterator iter = layout->begin();
61  iter != endIter; ++iter) {
62  const Size i = iter.index();
63 
64  if (illegalLocalVolOverwrite_ < 0.0) {
65  v[i] = square<Real>()(
66  localVol_->localVol(0.5*(t1+t2), x_[i], true));
67  }
68  else {
69  try {
70  v[i] = square<Real>()(
71  localVol_->localVol(0.5*(t1+t2), x_[i], true));
72  } catch (Error&) {
73  v[i] = square<Real>()(illegalLocalVolOverwrite_);
74  }
75  }
76  }
77  mapT_.axpyb(Array(1, 1.0), dxMap_.multR(- r + q + 0.5*v),
78  dxxMap_.multR(0.5*v), Array(1, 0.0));
79  }
80  else {
81  const Real v
82  = volTS_->blackForwardVariance(t1, t2, strike_)/(t2-t1);
83  mapT_.axpyb(Array(1, - r + q + 0.5*v), dxMap_,
84  dxxMap_.mult(0.5*Array(mesher_->layout()->size(), v)),
85  Array(1, 0.0));
86  }
87  }
