67 Time time2,
68 bool extrapolate) const {
69 QL_REQUIRE(time1<time2, "wrong times");
70 Real var1 = atmVariance(time1, extrapolate);
71 Real var2 = atmVariance(time2, extrapolate);
72 QL_ENSURE(var1<var2, "non-increasing variances");
