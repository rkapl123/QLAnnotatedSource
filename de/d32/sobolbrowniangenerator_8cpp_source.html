27  void fillByFactor(std::vector<std::vector<Size> >& M,
28  Size factors, Size steps) {

30  for (Size i=0; i<factors; ++i)

32  M[i][j] = counter++;

35  void fillByStep(std::vector<std::vector<Size> >& M,
36  Size factors, Size steps) {
37  Size counter = 0;
38  for (Size j=0; j<steps; ++j)
39  for (Size i=0; i<factors; ++i)
40  M[i][j] = counter++;
41  }

43  // variate 2 is used for the second factor's full path
44  void fillByDiagonal(std::vector<std::vector<Size> >& M,
45  Size factors, Size steps) {
46  // starting position of the current diagonal
47  Size i0 = 0, j0 = 0;
48  // current position
49  Size i=0, j=0;
50  Size counter = 0;
51  while (counter < factors*steps) {
52  M[i][j] = counter++;
53  if (i == 0 || j == steps-1) {
54  // we completed a diagonal and have to start a new one
55  if (i0 < factors-1) {
56  // we start the path of the next factor
57  i0 = i0+1;
58  j0 = 0;
59  } else {
60  // we move along the path of the last factor
61  i0 = factors-1;
62  j0 = j0+1;
63  }
64  i = i0;
65  j = j0;
66  } else {
67  // we move along the diagonal
68  i = i-1;
69  j = j+1;
70  }
71  }
72  }

110  SobolBrownianGenerator::SobolBrownianGenerator(
111  Size factors,
112  Size steps,
113  Ordering ordering,
114  unsigned long seed,
115  SobolRsg::DirectionIntegers integers)
116  : factors_(factors), steps_(steps), ordering_(ordering),
117  generator_(SobolRsg(factors*steps, seed, integers),
118  InverseCumulativeNormal()),
119  bridge_(steps), lastStep_(0),
120  orderedIndices_(factors, std::vector<Size>(steps)),
121  bridgedVariates_(factors, std::vector<Real>(steps)) {
122 
123  switch (ordering_) {
124  case Factors:
125  fillByFactor(orderedIndices_, factors_, steps_);
126  break;
127  case Steps:
128  fillByStep(orderedIndices_, factors_, steps_);
129  break;
130  case Diagonal:
131  fillByDiagonal(orderedIndices_, factors_, steps_);
132  break;
133  default:
134  QL_FAIL("unknown ordering");
135  }
136  }

139  Real SobolBrownianGenerator::nextPath() {
140  typedef InverseCumulativeRsg<SobolRsg,
141  InverseCumulativeNormal>::sample_type
142  sample_type;
143 
144  const sample_type& sample = generator_.nextSequence();
145  // Brownian-bridge the variates according to the ordered indices
146  for (Size i=0; i<factors_; ++i) {
147  bridge_.transform(boost::make_permutation_iterator(
148  sample.value.begin(),
149  orderedIndices_[i].begin()),
150  boost::make_permutation_iterator(
151  sample.value.begin(),
152  orderedIndices_[i].end()),
153  bridgedVariates_[i].begin());
154  }
155  lastStep_ = 0;
156  return sample.weight;
157  }

165  std::vector<std::vector<Real> > SobolBrownianGenerator::transform(
166  const std::vector<std::vector<Real> >& variates) {
167 
168  QL_REQUIRE( (variates.size() == factors_*steps_),
169  "inconsistent variate vector");
170 
171  const Size dim = factors_*steps_;
172  const Size nPaths = variates.front().size();
173 
174  std::vector<std::vector<Real> >
175  retVal(factors_, std::vector<Real>(nPaths*steps_));
176 
177  for (Size j=0; j < nPaths; ++j) {
178  std::vector<Real> sample(steps_*factors_);
179  for (Size k=0; k < dim; ++k) {
180  sample[k] = variates[k][j];
181  }
182  for (Size i=0; i<factors_; ++i) {
183  bridge_.transform(boost::make_permutation_iterator(
184  sample.begin(),
185  orderedIndices_[i].begin()),
186  boost::make_permutation_iterator(
187  sample.begin(),
188  orderedIndices_[i].end()),
189  retVal[i].begin()+j*steps_);
190  }
191  }
192 
193  return retVal;
194  }

196  Real SobolBrownianGenerator::nextStep(std::vector<Real>& output) {
197  #if defined(QL_EXTRA_SAFETY_CHECKS)
198  QL_REQUIRE(output.size() == factors_, "size mismatch");
199  QL_REQUIRE(lastStep_<steps_, "sequence exhausted");
200  #endif
201  for (Size i=0; i<factors_; ++i)
202  output[i] = bridgedVariates_[i][lastStep_];
203  ++lastStep_;
204  return 1.0;
205  }
