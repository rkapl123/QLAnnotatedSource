100  MakeCms::operator boost::shared_ptr<Swap>() const {
101 
102  Date startDate;
103  if (effectiveDate_ != Date())
104  startDate = effectiveDate_;
105  else {
106  Natural fixingDays = iborIndex_->fixingDays();
107  Date refDate = Settings::instance().evaluationDate();
108  // if the evaluation date is not a business day
109  // then move to the next business day
110  refDate = floatCalendar_.adjust(refDate);
111  Date spotDate = floatCalendar_.advance(refDate,
112  fixingDays*Days);
113  startDate = spotDate+forwardStart_;
114  }
115 
116  Date terminationDate = startDate+swapTenor_;
117 
118  Schedule cmsSchedule(startDate, terminationDate,
119  cmsTenor_, cmsCalendar_,
120  cmsConvention_,
121  cmsTerminationDateConvention_,
122  cmsRule_, cmsEndOfMonth_,
123  cmsFirstDate_, cmsNextToLastDate_);
124 
125  Schedule floatSchedule(startDate, terminationDate,
126  floatTenor_, floatCalendar_,
127  floatConvention_,
128  floatTerminationDateConvention_,
129  floatRule_ , floatEndOfMonth_,
130  floatFirstDate_, floatNextToLastDate_);
131 
132  Leg cmsLeg = CmsLeg(cmsSchedule, swapIndex_)
133  .withNotionals(nominal_)
134  .withPaymentDayCounter(cmsDayCount_)
135  .withPaymentAdjustment(cmsConvention_)
136  .withFixingDays(swapIndex_->fixingDays())
137  .withGearings(cmsGearing_)
138  .withSpreads(cmsSpread_)
139  .withCaps(cmsCap_)
140  .withFloors(cmsFloor_);
141  if (couponPricer_)
142  setCouponPricer(cmsLeg, couponPricer_);
143 
144  Rate usedSpread = iborSpread_;
145  if (useAtmSpread_) {
146  QL_REQUIRE(!iborIndex_->forwardingTermStructure().empty(),
147  "null term structure set to this instance of " <<
148  iborIndex_->name());
149  QL_REQUIRE(!swapIndex_->forwardingTermStructure().empty(),
150  "null term structure set to this instance of " <<
151  swapIndex_->name());
152  QL_REQUIRE(couponPricer_,
153  "no CmsCouponPricer set (yet)");
154  Leg floatLeg = IborLeg(floatSchedule, iborIndex_)
155  .withNotionals(nominal_)
156  .withPaymentDayCounter(floatDayCount_)
157  .withPaymentAdjustment(floatConvention_)
158  .withFixingDays(iborIndex_->fixingDays());
159 
160  Swap temp(cmsLeg, floatLeg);
161  temp.setPricingEngine(engine_);
162 
163  Real npv = temp.legNPV(0)+temp.legNPV(1);
164 
165  usedSpread = -npv/temp.legBPS(1)*1e-4;
166  } else {
167  QL_REQUIRE(usedSpread != Null<Spread>(),
168  "null spread set");
169  }
170 
171  Leg floatLeg = IborLeg(floatSchedule, iborIndex_)
172  .withNotionals(nominal_)
173  .withPaymentDayCounter(floatDayCount_)
174  .withPaymentAdjustment(floatConvention_)
175  .withFixingDays(iborIndex_->fixingDays())
176  .withSpreads(usedSpread);
177 
178  boost::shared_ptr<Swap> swap;
179  if (payCms_)
180  swap = boost::make_shared<Swap>(cmsLeg, floatLeg);
181  else
182  swap = boost::make_shared<Swap>(floatLeg, cmsLeg);
183  swap->setPricingEngine(engine_);
184  return swap;
185  }
