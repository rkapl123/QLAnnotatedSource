94  void updateAlphaVec(){
95  // Function calculates the alpha vector with given
96  // fixed pillars+values
97 
98  // Write Matrix M
99  Real tmp=0.0;
100 
101  for(Size rowIt=0; rowIt<xSize_;++rowIt){
102 
103  yVec_[rowIt]=this->yBegin_[rowIt];
104  tmp=1.0/gammaFunc(this->xBegin_[rowIt]);
105 
106  for(Size colIt=0; colIt<xSize_;++colIt){
107  M_[rowIt][colIt]=kernelAbs(this->xBegin_[rowIt],
108  this->xBegin_[colIt])*tmp;
109  }
110  }
111 
112  // Solve y=M*\alpha for \alpha
113  alphaVec_ = qrSolve(M_, yVec_);
114 
115  // check if inversion worked up to a reasonable precision.
116  // I've chosen not to check determinant(M_)!=0 before solving
117 
118  Array diffVec=Abs(M_*alphaVec_ - yVec_);
119 
120  for (Size i=0; i<diffVec.size(); ++i) {
121  QL_REQUIRE(diffVec[i] < invPrec_,
122  "Inversion failed in 1d kernel interpolation");
123  }
124  }
