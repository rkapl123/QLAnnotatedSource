32 //! K-interpolated YoY optionlet volatility
33 /*! The stripper provides curves in the T direction along each K.
34 We don't know whether this is interpolating or fitting in the
35 T direction. Our K direction interpolations are not model
36 fitting.
37
38 An alternative design would be a
39 FittedYoYOptionletVolatilitySurface taking a model, e.g. SABR
40 in the interest rate world. This could use the same stripping
41 in the T direction along each K.
42
43 \bug Tests currently fail.
44 */

178 template<class Interpolator1D>
179 Volatility KInterpolatedYoYOptionletVolatilitySurface<Interpolator1D>::
180 volatilityImpl(Time length, Rate strike) const {
181
182 auto years = (Natural)floor(length);
183 auto days = (Natural)floor((length - years) * 365.0);
184 Date d = referenceDate() + Period(years, Years) + Period(days, Days);
185
186 return this->volatilityImpl(d, strike);
187 }

189 template<class Interpolator1D>
190 void KInterpolatedYoYOptionletVolatilitySurface<Interpolator1D>::
191 updateSlice(const Date &d) const {
192
193 if (!lastDateisSet_ || d != lastDate_ ) {
194 slice_ = yoyOptionletStripper_->slice(d);
195
196 tempKinterpolation_ =
197 factory1D_.interpolate( slice_.first.begin(),
198 slice_.first.end(),
199 slice_.second.begin() );
200 lastDateisSet_ = true;
201 lastDate_ = d;
202 }
203 }
