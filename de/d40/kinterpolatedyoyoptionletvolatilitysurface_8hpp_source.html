165  template<class Interpolator1D>
166  Volatility KInterpolatedYoYOptionletVolatilitySurface<Interpolator1D>::
167  volatilityImpl(Time length, Rate strike) const {
168 
169  Natural years = (Natural)floor(length);
170  Natural days = (Natural)floor((length - years) * 365.0);
171  Date d = referenceDate() + Period(years, Years) + Period(days, Days);
172 
173  return this->volatilityImpl(d, strike);
174  }

176  template<class Interpolator1D>
177  void KInterpolatedYoYOptionletVolatilitySurface<Interpolator1D>::
178  updateSlice(const Date &d) const {
179 
180  if (!lastDateisSet_ || d != lastDate_ ) {
181  slice_ = yoyOptionletStripper_->slice(d);
182 
183  tempKinterpolation_ =
184  factory1D_.interpolate( slice_.first.begin(),
185  slice_.first.end(),
186  slice_.second.begin() );
187  lastDateisSet_ = true;
188  lastDate_ = d;
189  }
190  }
