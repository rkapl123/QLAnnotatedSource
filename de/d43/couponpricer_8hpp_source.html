198  inline Real BlackIborCouponPricer::swapletPrice() const {
199  // past or future fixing is managed in InterestRateIndex::fixing()
200 
201  Real swapletPrice = adjustedFixing() * accrualPeriod_ * discount_;
202  return gearing_ * swapletPrice + spreadLegValue_;
203  }
204 
205  inline Rate BlackIborCouponPricer::swapletRate() const {
206  return gearing_ * adjustedFixing() + spread_;
207  }

209  inline Real BlackIborCouponPricer::capletPrice(Rate effectiveCap) const {
