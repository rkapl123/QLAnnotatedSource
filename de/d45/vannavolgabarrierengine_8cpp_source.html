36  VannaVolgaBarrierEngine::VannaVolgaBarrierEngine(const Handle<DeltaVolQuote>& atmVol,
37  const Handle<DeltaVolQuote>& vol25Put,
38  const Handle<DeltaVolQuote>& vol25Call,
39  const Handle<Quote>& spotFX,
40  const Handle<YieldTermStructure>& domesticTS,
41  const Handle<YieldTermStructure>& foreignTS,
42  const bool adaptVanDelta,
43  const Real bsPriceWithSmile)
44  : atmVol_(atmVol), vol25Put_(vol25Put), vol25Call_(vol25Call), T_(atmVol_->maturity()),
45  spotFX_(spotFX), domesticTS_(domesticTS), foreignTS_(foreignTS),
46  adaptVanDelta_(adaptVanDelta), bsPriceWithSmile_(bsPriceWithSmile) {
47  QL_REQUIRE(vol25Put_->delta() == -0.25, "25 delta put is required by vanna volga method");
48  QL_REQUIRE(vol25Call_->delta() == 0.25, "25 delta call is required by vanna volga method");
49 
50  QL_REQUIRE(vol25Put_->maturity() == vol25Call_->maturity() &&
51  vol25Put_->maturity() == atmVol_->maturity(),
52  "Maturity of 3 vols are not the same");
53 
54  QL_REQUIRE(!domesticTS_.empty(), "domestic yield curve is not defined");
55  QL_REQUIRE(!foreignTS_.empty(), "foreign yield curve is not defined");
56 
57  registerWith(atmVol_);
58  registerWith(vol25Put_);
59  registerWith(vol25Call_);
60  registerWith(spotFX_);
61  registerWith(domesticTS_);
62  registerWith(foreignTS_);
63  }

65  void VannaVolgaBarrierEngine::calculate() const {
66 
67  QL_REQUIRE(arguments_.barrierType == Barrier::UpIn || arguments_.barrierType == Barrier::UpOut ||
68  arguments_.barrierType == Barrier::DownIn || arguments_.barrierType == Barrier::DownOut,
69  "Invalid barrier type");
70 
71  const Real sigmaShift_vega = 0.0001;
72  const Real sigmaShift_volga = 0.0001;
73  const Real spotShift_delta = 0.0001 * spotFX_->value();
74  const Real sigmaShift_vanna = 0.0001;
75 
76  Handle<Quote> x0Quote(
77  ext::make_shared<SimpleQuote>(spotFX_->value())); //used for shift
78  Handle<Quote> atmVolQuote(
79  ext::make_shared<SimpleQuote>(atmVol_->value())); //used for shift
80 
81  ext::shared_ptr<BlackVolTermStructure> blackVolTS =
82  ext::make_shared<BlackConstantVol>(
83  Settings::instance().evaluationDate(),
84  NullCalendar(), atmVolQuote, Actual365Fixed());
85  ext::shared_ptr<BlackScholesMertonProcess> stochProcess =
86  ext::make_shared<BlackScholesMertonProcess>(
87  x0Quote,
88  foreignTS_,
89  domesticTS_,
90  Handle<BlackVolTermStructure>(blackVolTS));
91 
92  ext::shared_ptr<PricingEngine> engineBS =
93  ext::make_shared<AnalyticBarrierEngine>(stochProcess);
94 
95  BlackDeltaCalculator blackDeltaCalculatorAtm(
96  Option::Call, atmVol_->deltaType(), x0Quote->value(),
97  domesticTS_->discount(T_), foreignTS_->discount(T_),
98  atmVol_->value() * sqrt(T_));
99  Real atmStrike = blackDeltaCalculatorAtm.atmStrike(atmVol_->atmType());
100 
101  Real call25Vol = vol25Call_->value();
102  Real put25Vol = vol25Put_->value();
103 
104  BlackDeltaCalculator blackDeltaCalculatorPut25(Option::Put, vol25Put_->deltaType(), x0Quote->value(),
105  domesticTS_->discount(T_), foreignTS_->discount(T_),
106  put25Vol * sqrt(T_));
107  Real put25Strike = blackDeltaCalculatorPut25.strikeFromDelta(-0.25);
108  BlackDeltaCalculator blackDeltaCalculatorCall25(Option::Call, vol25Call_->deltaType(), x0Quote->value(),
109  domesticTS_->discount(T_), foreignTS_->discount(T_),
110  call25Vol * sqrt(T_));
111  Real call25Strike = blackDeltaCalculatorCall25.strikeFromDelta(0.25);
112 
113 
114  //here use vanna volga interpolated smile to price vanilla
115  std::vector<Real> strikes;
116  std::vector<Real> vols;
117  strikes.push_back(put25Strike);
118  vols.push_back(put25Vol);
119  strikes.push_back(atmStrike);
120  vols.push_back(atmVol_->value());
121  strikes.push_back(call25Strike);
122  vols.push_back(call25Vol);
123  VannaVolga vannaVolga(x0Quote->value(), domesticTS_->discount(T_), foreignTS_->discount(T_), T_);
124  Interpolation interpolation = vannaVolga.interpolate(strikes.begin(), strikes.end(), vols.begin());
125  interpolation.enableExtrapolation();
126  const ext::shared_ptr<StrikedTypePayoff> payoff =
127  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
128  Real strikeVol = interpolation(payoff->strike());
129 
130  //vanilla option price
131  Real vanillaOption = blackFormula(payoff->optionType(), payoff->strike(),
132  x0Quote->value()* foreignTS_->discount(T_)/ domesticTS_->discount(T_),
133  strikeVol * sqrt(T_),
134  domesticTS_->discount(T_));
135 
136  //spot > barrier up&out 0
137  if(x0Quote->value() >= arguments_.barrier && arguments_.barrierType == Barrier::UpOut){
138  results_.value = 0.0;
139  results_.additionalResults["VanillaPrice"] = adaptVanDelta_? bsPriceWithSmile_ : vanillaOption;
140  results_.additionalResults["BarrierInPrice"] = adaptVanDelta_? bsPriceWithSmile_ : vanillaOption;
141  results_.additionalResults["BarrierOutPrice"] = 0.0;
142  }
143  //spot > barrier up&in vanilla
144  else if(x0Quote->value() >= arguments_.barrier && arguments_.barrierType == Barrier::UpIn){
145  results_.value = adaptVanDelta_? bsPriceWithSmile_ : vanillaOption;
146  results_.additionalResults["VanillaPrice"] = adaptVanDelta_? bsPriceWithSmile_ : vanillaOption;
147  results_.additionalResults["BarrierInPrice"] = adaptVanDelta_? bsPriceWithSmile_ : vanillaOption;
148  results_.additionalResults["BarrierOutPrice"] = 0.0;
149  }
150  //spot < barrier down&out 0
151  else if(x0Quote->value() <= arguments_.barrier && arguments_.barrierType == Barrier::DownOut){
152  results_.value = 0.0;
153  results_.additionalResults["VanillaPrice"] = adaptVanDelta_? bsPriceWithSmile_ : vanillaOption;
154  results_.additionalResults["BarrierInPrice"] = adaptVanDelta_? bsPriceWithSmile_ : vanillaOption;
155  results_.additionalResults["BarrierOutPrice"] = 0.0;
156  }
157  //spot < barrier down&in vanilla
158  else if(x0Quote->value() <= arguments_.barrier && arguments_.barrierType == Barrier::DownIn){
159  results_.value = adaptVanDelta_? bsPriceWithSmile_ : vanillaOption;
160  results_.additionalResults["VanillaPrice"] = adaptVanDelta_? bsPriceWithSmile_ : vanillaOption;
161  results_.additionalResults["BarrierInPrice"] = adaptVanDelta_? bsPriceWithSmile_ : vanillaOption;
162  results_.additionalResults["BarrierOutPrice"] = 0.0;
163  }
164  else{
165 
166  //set up BS barrier option pricing
167  //only calculate out barrier option price
168  // in barrier price = vanilla - out barrier
169  Barrier::Type barrierType;
170  if(arguments_.barrierType == Barrier::UpOut)
171  barrierType = arguments_.barrierType;
172  else if(arguments_.barrierType == Barrier::UpIn)
173  barrierType = Barrier::UpOut;
174  else if(arguments_.barrierType == Barrier::DownOut)
175  barrierType = arguments_.barrierType;
176  else
177  barrierType = Barrier::DownOut;
178 
179  BarrierOption barrierOption(barrierType,
180  arguments_.barrier,
181  arguments_.rebate,
182  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff),
183  arguments_.exercise);
184 
185  barrierOption.setPricingEngine(engineBS);
186 
187  //BS price with atm vol
188  Real priceBS = barrierOption.NPV();
189 
190  Real priceAtmCallBS = blackFormula(Option::Call,atmStrike,
191  x0Quote->value()* foreignTS_->discount(T_)/ domesticTS_->discount(T_),
192  atmVol_->value() * sqrt(T_),
193  domesticTS_->discount(T_));
194  Real price25CallBS = blackFormula(Option::Call,call25Strike,
195  x0Quote->value()* foreignTS_->discount(T_)/ domesticTS_->discount(T_),
196  atmVol_->value() * sqrt(T_),
197  domesticTS_->discount(T_));
198  Real price25PutBS = blackFormula(Option::Put,put25Strike,
199  x0Quote->value()* foreignTS_->discount(T_)/ domesticTS_->discount(T_),
200  atmVol_->value() * sqrt(T_),
201  domesticTS_->discount(T_));
202 
203  //market price
204  Real priceAtmCallMkt = blackFormula(Option::Call,atmStrike,
205  x0Quote->value()* foreignTS_->discount(T_)/ domesticTS_->discount(T_),
206  atmVol_->value() * sqrt(T_),
207  domesticTS_->discount(T_));
208 
209  Real price25CallMkt = blackFormula(Option::Call,call25Strike,
210  x0Quote->value()* foreignTS_->discount(T_)/ domesticTS_->discount(T_),
211  call25Vol * sqrt(T_),
212  domesticTS_->discount(T_));
213  Real price25PutMkt = blackFormula(Option::Put,put25Strike,
214  x0Quote->value()* foreignTS_->discount(T_)/ domesticTS_->discount(T_),
215  put25Vol * sqrt(T_),
216  domesticTS_->discount(T_));
217 
218 
219  //Analytical Black Scholes formula for vanilla option
220  NormalDistribution norm;
221  Real d1atm = (std::log(x0Quote->value()* foreignTS_->discount(T_)/ domesticTS_->discount(T_)/atmStrike)
222  + 0.5*std::pow(atmVolQuote->value(),2.0) * T_)/(atmVolQuote->value() * sqrt(T_));
223  Real vegaAtm_Analytical = x0Quote->value() * norm(d1atm) * sqrt(T_) * foreignTS_->discount(T_);
224  Real vannaAtm_Analytical = vegaAtm_Analytical/x0Quote->value() *(1.0 - d1atm/(atmVolQuote->value()*sqrt(T_)));
225  Real volgaAtm_Analytical = vegaAtm_Analytical * d1atm * (d1atm - atmVolQuote->value() * sqrt(T_))/atmVolQuote->value();
226 
227  Real d125call = (std::log(x0Quote->value()* foreignTS_->discount(T_)/ domesticTS_->discount(T_)/call25Strike)
228  + 0.5*std::pow(atmVolQuote->value(),2.0) * T_)/(atmVolQuote->value() * sqrt(T_));
229  Real vega25Call_Analytical = x0Quote->value() * norm(d125call) * sqrt(T_) * foreignTS_->discount(T_);
230  Real vanna25Call_Analytical = vega25Call_Analytical/x0Quote->value() *(1.0 - d125call/(atmVolQuote->value()*sqrt(T_)));
231  Real volga25Call_Analytical = vega25Call_Analytical * d125call * (d125call - atmVolQuote->value() * sqrt(T_))/atmVolQuote->value();
232 
233  Real d125Put = (std::log(x0Quote->value()* foreignTS_->discount(T_)/ domesticTS_->discount(T_)/put25Strike)
234  + 0.5*std::pow(atmVolQuote->value(),2.0) * T_)/(atmVolQuote->value() * sqrt(T_));
235  Real vega25Put_Analytical = x0Quote->value() * norm(d125Put) * sqrt(T_) * foreignTS_->discount(T_);
236  Real vanna25Put_Analytical = vega25Put_Analytical/x0Quote->value() *(1.0 - d125Put/(atmVolQuote->value()*sqrt(T_)));
237  Real volga25Put_Analytical = vega25Put_Analytical * d125Put * (d125Put - atmVolQuote->value() * sqrt(T_))/atmVolQuote->value();
238 
239 
240  //BS vega
241  ext::static_pointer_cast<SimpleQuote> (atmVolQuote.currentLink())->setValue(atmVolQuote->value() + sigmaShift_vega);
242  barrierOption.recalculate();
243  Real vegaBarBS = (barrierOption.NPV() - priceBS)/sigmaShift_vega;
244 
245  ext::static_pointer_cast<SimpleQuote> (atmVolQuote.currentLink())->setValue(atmVolQuote->value() - sigmaShift_vega);//setback
246 
247  //BS volga
248 
249  //vegaBar2
250  //base NPV
251  ext::static_pointer_cast<SimpleQuote> (atmVolQuote.currentLink())->setValue(atmVolQuote->value() + sigmaShift_volga);
252  barrierOption.recalculate();
253  Real priceBS2 = barrierOption.NPV();
254 
255  //shifted npv
256  ext::static_pointer_cast<SimpleQuote> (atmVolQuote.currentLink())->setValue(atmVolQuote->value() + sigmaShift_vega);
257  barrierOption.recalculate();
258  Real vegaBarBS2 = (barrierOption.NPV() - priceBS2)/sigmaShift_vega;
259  Real volgaBarBS = (vegaBarBS2 - vegaBarBS)/sigmaShift_volga;
260 
261  ext::static_pointer_cast<SimpleQuote> (atmVolQuote.currentLink())->setValue(atmVolQuote->value()
262  - sigmaShift_volga
263  - sigmaShift_vega);//setback
264 
265  //BS Delta
266  //base delta
267  ext::static_pointer_cast<SimpleQuote> (x0Quote.currentLink())->setValue(x0Quote->value() + spotShift_delta);//shift forth
268  barrierOption.recalculate();
269  Real priceBS_delta1 = barrierOption.NPV();
270 
271  ext::static_pointer_cast<SimpleQuote> (x0Quote.currentLink())->setValue(x0Quote->value() - 2 * spotShift_delta);//shift back
272  barrierOption.recalculate();
273  Real priceBS_delta2 = barrierOption.NPV();
274 
275  ext::static_pointer_cast<SimpleQuote> (x0Quote.currentLink())->setValue(x0Quote->value() + spotShift_delta);//set back
276  Real deltaBar1 = (priceBS_delta1 - priceBS_delta2)/(2.0*spotShift_delta);
277 
278  //shifted delta
279  ext::static_pointer_cast<SimpleQuote> (atmVolQuote.currentLink())->setValue(atmVolQuote->value() + sigmaShift_vanna);//shift sigma
280  ext::static_pointer_cast<SimpleQuote> (x0Quote.currentLink())->setValue(x0Quote->value() + spotShift_delta);//shift forth
281  barrierOption.recalculate();
282  priceBS_delta1 = barrierOption.NPV();
283 
284  ext::static_pointer_cast<SimpleQuote> (x0Quote.currentLink())->setValue(x0Quote->value() - 2 * spotShift_delta);//shift back
285  barrierOption.recalculate();
286  priceBS_delta2 = barrierOption.NPV();
287 
288  ext::static_pointer_cast<SimpleQuote> (x0Quote.currentLink())->setValue(x0Quote->value() + spotShift_delta);//set back
289  Real deltaBar2 = (priceBS_delta1 - priceBS_delta2)/(2.0*spotShift_delta);
290 
291  Real vannaBarBS = (deltaBar2 - deltaBar1)/sigmaShift_vanna;
292 
293  ext::static_pointer_cast<SimpleQuote> (atmVolQuote.currentLink())->setValue(atmVolQuote->value() - sigmaShift_vanna);//set back
294 
295  //Matrix
296  Matrix A(3,3,0.0);
297 
298  //analytical
299  A[0][0] = vegaAtm_Analytical;
300  A[0][1] = vega25Call_Analytical;
301  A[0][2] = vega25Put_Analytical;
302  A[1][0] = vannaAtm_Analytical;
303  A[1][1] = vanna25Call_Analytical;
304  A[1][2] = vanna25Put_Analytical;
305  A[2][0] = volgaAtm_Analytical;
306  A[2][1] = volga25Call_Analytical;
307  A[2][2] = volga25Put_Analytical;
308 
309  Array b(3,0.0);
310  b[0] = vegaBarBS;
311  b[1] = vannaBarBS;
312  b[2] = volgaBarBS;
313 
314  Array q = inverse(A) * b;
315 
316  //touch probability
317  CumulativeNormalDistribution cnd;
318  Real mu = domesticTS_->zeroRate(T_, Continuous) - foreignTS_->zeroRate(T_, Continuous) - pow(atmVol_->value(), 2.0)/2.0;
319  Real h2 = (log(arguments_.barrier/x0Quote->value()) + mu*T_)/(atmVol_->value()*sqrt(T_));
320  Real h2Prime = (log(x0Quote->value()/arguments_.barrier) + mu*T_)/(atmVol_->value()*sqrt(T_));
321  Real probTouch = 0.0;
322  if(arguments_.barrierType == Barrier::UpIn || arguments_.barrierType == Barrier::UpOut)
323  probTouch = cnd(h2Prime) + pow(arguments_.barrier/x0Quote->value(), 2.0*mu/pow(atmVol_->value(), 2.0))*cnd(-h2);
324  else
325  probTouch = cnd(-h2Prime) + pow(arguments_.barrier/x0Quote->value(), 2.0*mu/pow(atmVol_->value(), 2.0))*cnd(h2);
326  Real p_survival = 1.0 - probTouch;
327 
328  Real lambda = p_survival ;
329  Real adjust = q[0]*(priceAtmCallMkt - priceAtmCallBS)
330  + q[1]*(price25CallMkt - price25CallBS)
331  + q[2]*(price25PutMkt - price25PutBS);
332  Real outPrice = priceBS + lambda*adjust;//
333  Real inPrice;
334 
335  //adapt Vanilla delta
336  if (adaptVanDelta_) {
337  outPrice += lambda*(bsPriceWithSmile_ - vanillaOption);
338  //capfloored by (0, vanilla)
339  outPrice = std::max(0.0, std::min(bsPriceWithSmile_, outPrice));
340  inPrice = bsPriceWithSmile_ - outPrice;
341  }
342  else{
343  //capfloored by (0, vanilla)
344  outPrice = std::max(0.0, std::min(vanillaOption, outPrice));
345  inPrice = vanillaOption - outPrice;
346  }
347 
348  if(arguments_.barrierType == Barrier::DownOut || arguments_.barrierType == Barrier::UpOut)
349  results_.value = outPrice;
350  else
351  results_.value = inPrice;
352  results_.additionalResults["VanillaPrice"] = vanillaOption;
353  results_.additionalResults["BarrierInPrice"] = inPrice;
354  results_.additionalResults["BarrierOutPrice"] = outPrice;
355  results_.additionalResults["lambda"] = lambda;
356  }
357  }
