164  template <class GenericEngine, template <class> class MC,
165  class RNG, class S>
166  inline
167  TimeGrid MCLongstaffSchwartzPathEngine<GenericEngine,MC,RNG,S>::timeGrid()
168  const {
169  const std::vector<Date> & fixings = this->arguments_.fixingDates;
170  const Size numberOfFixings = fixings.size();
171 
172  std::vector<Time> fixingTimes(numberOfFixings);
173  for (Size i = 0; i < numberOfFixings; ++i) {
174  fixingTimes[i] =
175  this->process_->time(fixings[i]);
176  }
177 
178  const Size numberOfTimeSteps = timeSteps_ != Null<Size>() ? timeSteps_ : timeStepsPerYear_ * fixingTimes.back();
179 
180  return TimeGrid(fixingTimes.begin(), fixingTimes.end(), numberOfTimeSteps);
181  }
