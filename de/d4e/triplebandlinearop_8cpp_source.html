30  TripleBandLinearOp::TripleBandLinearOp(
31  Size direction,
32  const ext::shared_ptr<FdmMesher>& mesher)
33  : direction_(direction),
34  i0_ (new Size[mesher->layout()->size()]),
35  i2_ (new Size[mesher->layout()->size()]),
36  reverseIndex_ (new Size[mesher->layout()->size()]),
37  lower_ (new Real[mesher->layout()->size()]),
38  diag_ (new Real[mesher->layout()->size()]),
39  upper_ (new Real[mesher->layout()->size()]),
40  mesher_(mesher) {
41 
42  const ext::shared_ptr<FdmLinearOpLayout> layout = mesher->layout();
43  const FdmLinearOpIterator endIter = layout->end();
44 
45  std::vector<Size> newDim(layout->dim());
46  std::iter_swap(newDim.begin(), newDim.begin()+direction_);
47  std::vector<Size> newSpacing = FdmLinearOpLayout(newDim).spacing();
48  std::iter_swap(newSpacing.begin(), newSpacing.begin()+direction_);
49 
50  for (FdmLinearOpIterator iter = layout->begin(); iter!=endIter; ++iter) {
51  const Size i = iter.index();
52 
53  i0_[i] = layout->neighbourhood(iter, direction, -1);
54  i2_[i] = layout->neighbourhood(iter, direction, 1);
55 
56  const std::vector<Size>& coordinates = iter.coordinates();
57  const Size newIndex =
58  std::inner_product(coordinates.begin(), coordinates.end(),
59  newSpacing.begin(), Size(0));
60  reverseIndex_[newIndex] = i;
61  }
62  }

106  void TripleBandLinearOp::axpyb(const Array& a,
107  const TripleBandLinearOp& x,
108  const TripleBandLinearOp& y,
109  const Array& b) {
110  const Size size = mesher_->layout()->size();
111 
112  Real *diag(diag_.get());
113  Real *lower(lower_.get());
114  Real *upper(upper_.get());
115 
116  const Real *y_diag (y.diag_.get());
117  const Real *y_lower(y.lower_.get());
118  const Real *y_upper(y.upper_.get());
119 
120  if (a.empty()) {
121  if (b.empty()) {
122  //#pragma omp parallel for
123  for (Size i=0; i < size; ++i) {
124  diag[i] = y_diag[i];
125  lower[i] = y_lower[i];
126  upper[i] = y_upper[i];
127  }
128  }
129  else {
130  Array::const_iterator bptr(b.begin());
131  const Size binc = (b.size() > 1) ? 1 : 0;
132  //#pragma omp parallel for
133  for (Size i=0; i < size; ++i) {
134  diag[i] = y_diag[i] + bptr[i*binc];
135  lower[i] = y_lower[i];
136  upper[i] = y_upper[i];
137  }
138  }
139  }
140  else if (b.empty()) {
141  Array::const_iterator aptr(a.begin());
142  const Size ainc = (a.size() > 1) ? 1 : 0;
143 
144  const Real *x_diag (x.diag_.get());
145  const Real *x_lower(x.lower_.get());
146  const Real *x_upper(x.upper_.get());
147 
148  //#pragma omp parallel for
149  for (Size i=0; i < size; ++i) {
150  const Real s = aptr[i*ainc];
151  diag[i] = y_diag[i] + s*x_diag[i];
152  lower[i] = y_lower[i] + s*x_lower[i];
153  upper[i] = y_upper[i] + s*x_upper[i];
154  }
155  }
156  else {
157  Array::const_iterator bptr(b.begin());
158  const Size binc = (b.size() > 1) ? 1 : 0;
159 
160  Array::const_iterator aptr(a.begin());
161  const Size ainc = (a.size() > 1) ? 1 : 0;
162 
163  const Real *x_diag (x.diag_.get());
164  const Real *x_lower(x.lower_.get());
165  const Real *x_upper(x.upper_.get());
166 
167  //#pragma omp parallel for
168  for (Size i=0; i < size; ++i) {
169  const Real s = aptr[i*ainc];
170  diag[i] = y_diag[i] + s*x_diag[i] + bptr[i*binc];
171  lower[i] = y_lower[i] + s*x_lower[i];
172  upper[i] = y_upper[i] + s*x_upper[i];
173  }
174  }
175  }

209  Disposable<TripleBandLinearOp> TripleBandLinearOp::multR(const Array& u) const {
210  const ext::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
211  const Size size = layout->size();
212  QL_REQUIRE(u.size() == size, "inconsistent size of rhs");
213  TripleBandLinearOp retVal(direction_, mesher_);
214 
215  #pragma omp parallel for
216  for (long i=0; i < (long)size; ++i) {
217  const Real sm1 = i > 0? u[i-1] : 1.0;
218  const Real s0 = u[i];
219  const Real sp1 = i < (long)size-1? u[i+1] : 1.0;
220  retVal.lower_[i]= lower_[i]*sm1;
221  retVal.diag_[i] = diag_[i]*s0;
222  retVal.upper_[i]= upper_[i]*sp1;
223  }
224 
225  return retVal;

243  Disposable<Array> TripleBandLinearOp::apply(const Array& r) const {
244  const ext::shared_ptr<FdmLinearOpLayout> index = mesher_->layout();
245 
246  QL_REQUIRE(r.size() == index->size(), "inconsistent length of r");
247 
248  const Real* lptr = lower_.get();
249  const Real* dptr = diag_.get();
250  const Real* uptr = upper_.get();
251  const Size* i0ptr = i0_.get();
252  const Size* i2ptr = i2_.get();
253 
254  array_type retVal(r.size());
255  //#pragma omp parallel for
256  for (Size i=0; i < index->size(); ++i) {
257  retVal[i] = r[i0ptr[i]]*lptr[i]+r[i]*dptr[i]+r[i2ptr[i]]*uptr[i];
258  }
259 
260  return retVal;
261  }

280  Disposable<Array>
281  TripleBandLinearOp::solve_splitting(const Array& r, Real a, Real b) const {
282  const ext::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
283  QL_REQUIRE(r.size() == layout->size(), "inconsistent size of rhs");
284 
285 #ifdef QL_EXTRA_SAFETY_CHECKS
286  for (FdmLinearOpIterator iter = layout->begin();
287  iter!=layout->end(); ++iter) {
288  const std::vector<Size>& coordinates = iter.coordinates();
289  QL_REQUIRE( coordinates[direction_] != 0
290  || lower_[iter.index()] == 0,"removing non zero entry!");
291  QL_REQUIRE( coordinates[direction_] != layout->dim()[direction_]-1
292  || upper_[iter.index()] == 0,"removing non zero entry!");
293  }
294 #endif
295 
296  Array retVal(r.size()), tmp(r.size());
297 
298  const Real* lptr = lower_.get();
299  const Real* dptr = diag_.get();
300  const Real* uptr = upper_.get();
301 
302  // Thomson algorithm to solve a tridiagonal system.
303  // Example code taken from Tridiagonalopertor and
304  // changed to fit for the triple band operator.
305  Size rim1 = reverseIndex_[0];
306  Real bet=1.0/(a*dptr[rim1]+b);
307  QL_REQUIRE(bet != 0.0, "division by zero");
308  retVal[reverseIndex_[0]] = r[rim1]*bet;
309 
310  for (Size j=1; j<=layout->size()-1; j++){
311  const Size ri = reverseIndex_[j];
312  tmp[j] = a*uptr[rim1]*bet;
313 
314  bet=b+a*(dptr[ri]-tmp[j]*lptr[ri]);
315  QL_ENSURE(bet != 0.0, "division by zero");
316  bet=1.0/bet;
317 
318  retVal[ri] = (r[ri]-a*lptr[ri]*retVal[rim1])*bet;
319  rim1 = ri;
320  }
321  // cannot be j>=0 with Size j
322  for (Size j=layout->size()-2; j>0; --j)
323  retVal[reverseIndex_[j]] -= tmp[j+1]*retVal[reverseIndex_[j+1]];
324  retVal[reverseIndex_[0]] -= tmp[1]*retVal[reverseIndex_[1]];
325 
326  return retVal;
327  }
