66  MakeVanillaSwap::operator ext::shared_ptr<VanillaSwap>() const {
67 
68  Date startDate;
69  if (effectiveDate_ != Date())
70  startDate = effectiveDate_;
71  else {
72  Date refDate = Settings::instance().evaluationDate();
73  // if the evaluation date is not a business day
74  // then move to the next business day
75  refDate = floatCalendar_.adjust(refDate);
76  // use index valueDate interface wherever possible to estimate spot date.
77  // Unless we pass an explicit settlementDays_ which does not match the index-defined number of fixing days.
78  Date spotDate;
79  if (settlementDays_ == Null<Natural>())
80  spotDate = iborIndex_->valueDate(refDate);
81  else
82  spotDate = floatCalendar_.advance(refDate, settlementDays_ * Days);
83  startDate = spotDate+forwardStart_;
84  if (forwardStart_.length()<0)
85  startDate = floatCalendar_.adjust(startDate,
86  Preceding);
87  else if (forwardStart_.length()>0)
88  startDate = floatCalendar_.adjust(startDate,
89  Following);
90  // no explicit date adjustment needed for forwardStart_.length()==0 (already handled by spotDate arithmetic above)
91  }
92 
93  Date endDate = terminationDate_;
94  if (endDate == Date()) {
95  if (floatEndOfMonth_)
96  endDate = floatCalendar_.advance(startDate,
97  swapTenor_,
98  ModifiedFollowing,
99  floatEndOfMonth_);
100  else
101  endDate = startDate + swapTenor_;
102  }
103 
104  const Currency& curr = iborIndex_->currency();
105  Period fixedTenor;
106  if (fixedTenor_ != Period())
107  fixedTenor = fixedTenor_;
108  else {
109  if ((curr == EURCurrency()) ||
110  (curr == USDCurrency()) ||
111  (curr == CHFCurrency()) ||
112  (curr == SEKCurrency()) ||
113  (curr == GBPCurrency() && swapTenor_ <= 1 * Years))
114  fixedTenor = Period(1, Years);
115  else if ((curr == GBPCurrency() && swapTenor_ > 1 * Years) ||
116  (curr == JPYCurrency()) ||
117  (curr == AUDCurrency() && swapTenor_ >= 4 * Years))
118  fixedTenor = Period(6, Months);
119  else if ((curr == HKDCurrency() ||
120  (curr == AUDCurrency() && swapTenor_ < 4 * Years)))
121  fixedTenor = Period(3, Months);
122  else
123  QL_FAIL("unknown fixed leg default tenor for " << curr);
124  }
125 
126  Schedule fixedSchedule(startDate, endDate,
127  fixedTenor, fixedCalendar_,
128  fixedConvention_,
129  fixedTerminationDateConvention_,
130  fixedRule_, fixedEndOfMonth_,
131  fixedFirstDate_, fixedNextToLastDate_);
132 
133  Schedule floatSchedule(startDate, endDate,
134  floatTenor_, floatCalendar_,
135  floatConvention_,
136  floatTerminationDateConvention_,
137  floatRule_, floatEndOfMonth_,
138  floatFirstDate_, floatNextToLastDate_);
139 
140  DayCounter fixedDayCount;
141  if (fixedDayCount_ != DayCounter())
142  fixedDayCount = fixedDayCount_;
143  else {
144  if (curr == USDCurrency())
145  fixedDayCount = Actual360();
146  else if (curr == EURCurrency() || curr == CHFCurrency() ||
147  curr == SEKCurrency())
148  fixedDayCount = Thirty360(Thirty360::BondBasis);
149  else if (curr == GBPCurrency() || curr == JPYCurrency() ||
150  curr == AUDCurrency() || curr == HKDCurrency() ||
151  curr == THBCurrency())
152  fixedDayCount = Actual365Fixed();
153  else
154  QL_FAIL("unknown fixed leg day counter for " << curr);
155  }
156 
157  Rate usedFixedRate = fixedRate_;
158  if (fixedRate_ == Null<Rate>()) {
159  VanillaSwap temp(type_, nominal_,
160  fixedSchedule,
161  0.0, // fixed rate
162  fixedDayCount,
163  floatSchedule, iborIndex_,
164  floatSpread_, floatDayCount_);
165  if (engine_ == 0) {
166  Handle<YieldTermStructure> disc =
167  iborIndex_->forwardingTermStructure();
168  QL_REQUIRE(!disc.empty(),
169  "null term structure set to this instance of " <<
170  iborIndex_->name());
171  bool includeSettlementDateFlows = false;
172  ext::shared_ptr<PricingEngine> engine(new
173  DiscountingSwapEngine(disc, includeSettlementDateFlows));
174  temp.setPricingEngine(engine);
175  } else
176  temp.setPricingEngine(engine_);
177 
178  usedFixedRate = temp.fairRate();
179  }
180 
181  ext::shared_ptr<VanillaSwap> swap(new
182  VanillaSwap(type_, nominal_,
183  fixedSchedule,
184  usedFixedRate, fixedDayCount,
185  floatSchedule,
186  iborIndex_, floatSpread_, floatDayCount_));
187 
188  if (engine_ == 0) {
189  Handle<YieldTermStructure> disc =
190  iborIndex_->forwardingTermStructure();
191  bool includeSettlementDateFlows = false;
192  ext::shared_ptr<PricingEngine> engine(new
193  DiscountingSwapEngine(disc, includeSettlementDateFlows));
194  swap->setPricingEngine(engine);
195  } else
196  swap->setPricingEngine(engine_);
197 
198  return swap;
199  }
