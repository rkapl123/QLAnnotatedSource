66  MakeVanillaSwap::operator shared_ptr<VanillaSwap>() const {
67 
68  Date startDate;
69  if (effectiveDate_ != Date())
70  startDate = effectiveDate_;
71  else {
72  Date refDate = Settings::instance().evaluationDate();
73  // if the evaluation date is not a business day
74  // then move to the next business day
75  refDate = floatCalendar_.adjust(refDate);
76  Date spotDate = floatCalendar_.advance(refDate,
77  settlementDays_*Days);
78  startDate = spotDate+forwardStart_;
79  if (forwardStart_.length()<0)
80  startDate = floatCalendar_.adjust(startDate,
81  Preceding);
82  else
83  startDate = floatCalendar_.adjust(startDate,
84  Following);
85  }
86 
87  Date endDate = terminationDate_;
88  if (endDate == Date()) {
89  if (floatEndOfMonth_)
90  endDate = floatCalendar_.advance(startDate,
91  swapTenor_,
92  ModifiedFollowing,
93  floatEndOfMonth_);
94  else
95  endDate = startDate + swapTenor_;
96  }
97 
98  const Currency& curr = iborIndex_->currency();
99  Period fixedTenor;
100  if (fixedTenor_ != Period())
101  fixedTenor = fixedTenor_;
102  else {
103  if ((curr == EURCurrency()) ||
104  (curr == USDCurrency()) ||
105  (curr == CHFCurrency()) ||
106  (curr == SEKCurrency()) ||
107  (curr == GBPCurrency() && swapTenor_ <= 1 * Years))
108  fixedTenor = Period(1, Years);
109  else if ((curr == GBPCurrency() && swapTenor_ > 1 * Years) ||
110  (curr == JPYCurrency()) ||
111  (curr == AUDCurrency() && swapTenor_ >= 4 * Years))
112  fixedTenor = Period(6, Months);
113  else if ((curr == HKDCurrency() ||
114  (curr == AUDCurrency() && swapTenor_ < 4 * Years)))
115  fixedTenor = Period(3, Months);
116  else
117  QL_FAIL("unknown fixed leg default tenor for " << curr);
118  }
119 
120  Schedule fixedSchedule(startDate, endDate,
121  fixedTenor, fixedCalendar_,
122  fixedConvention_,
123  fixedTerminationDateConvention_,
124  fixedRule_, fixedEndOfMonth_,
125  fixedFirstDate_, fixedNextToLastDate_);
126 
127  Schedule floatSchedule(startDate, endDate,
128  floatTenor_, floatCalendar_,
129  floatConvention_,
130  floatTerminationDateConvention_,
131  floatRule_, floatEndOfMonth_,
132  floatFirstDate_, floatNextToLastDate_);
133 
134  DayCounter fixedDayCount;
135  if (fixedDayCount_ != DayCounter())
136  fixedDayCount = fixedDayCount_;
137  else {
138  if (curr == USDCurrency())
139  fixedDayCount = Actual360();
140  else if (curr == EURCurrency() || curr == CHFCurrency() ||
141  curr == SEKCurrency())
142  fixedDayCount = Thirty360(Thirty360::BondBasis);
143  else if (curr == GBPCurrency() || curr == JPYCurrency() ||
144  curr == AUDCurrency() || curr == HKDCurrency())
145  fixedDayCount = Actual365Fixed();
146  else
147  QL_FAIL("unknown fixed leg day counter for " << curr);
148  }
149 
150  Rate usedFixedRate = fixedRate_;
151  if (fixedRate_ == Null<Rate>()) {
152  VanillaSwap temp(type_, nominal_,
153  fixedSchedule,
154  0.0, // fixed rate
155  fixedDayCount,
156  floatSchedule, iborIndex_,
157  floatSpread_, floatDayCount_);
158  if (engine_ == 0) {
159  Handle<YieldTermStructure> disc =
160  iborIndex_->forwardingTermStructure();
161  QL_REQUIRE(!disc.empty(),
162  "null term structure set to this instance of " <<
163  iborIndex_->name());
164  bool includeSettlementDateFlows = false;
165  shared_ptr<PricingEngine> engine(new
166  DiscountingSwapEngine(disc, includeSettlementDateFlows));
167  temp.setPricingEngine(engine);
168  } else
169  temp.setPricingEngine(engine_);
170 
171  usedFixedRate = temp.fairRate();
172  }
173 
174  shared_ptr<VanillaSwap> swap(new
175  VanillaSwap(type_, nominal_,
176  fixedSchedule,
177  usedFixedRate, fixedDayCount,
178  floatSchedule,
179  iborIndex_, floatSpread_, floatDayCount_));
180 
181  if (engine_ == 0) {
182  Handle<YieldTermStructure> disc =
183  iborIndex_->forwardingTermStructure();
184  bool includeSettlementDateFlows = false;
185  shared_ptr<PricingEngine> engine(new
186  DiscountingSwapEngine(disc, includeSettlementDateFlows));
187  swap->setPricingEngine(engine);
188  } else
189  swap->setPricingEngine(engine_);
190 
191  return swap;
192  }
