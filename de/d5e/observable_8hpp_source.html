317 class Proxy {
318 public:
319 explicit Proxy(Observer* const observer)
320 : active_ (true),
321 observer_(observer) {
322 }
323
324 void update() const {
325 std::lock_guard<std::recursive_mutex> lock(mutex_);
326 if (active_) {
327 // c++17 is required if used with std::shared_ptr<T>
328 const ext::weak_ptr<Observer> o
329 = observer_->weak_from_this();
330
331 //check for empty weak reference
332 //https://stackoverflow.com/questions/45507041/how-to-check-if-weak-ptr-is-empty-non-assigned
333 const ext::weak_ptr<Observer> empty;
334 if (o.owner_before(empty) || empty.owner_before(o)) {
335 const ext::shared_ptr<Observer> obs(o.lock());
336 if (obs)
337 obs->update();
338 }
339 else {
340 observer_->update();
341 }
342 }
343 }
344
345 void deactivate() {
346 std::lock_guard<std::recursive_mutex> lock(mutex_);
347 active_ = false;
348 }
349
350 private:
351 bool active_;
352 mutable std::recursive_mutex mutex_;
353 Observer* const observer_;
354 };

453
454 inline void ObservableSettings::enableUpdates() {
455 std::lock_guard<std::mutex> lock(mutex_);
456
457 // if there are outstanding deferred updates, do the notification
458 updatesType_ = UpdatesEnabled;
459
460 if (deferredObservers_.size()) {
461 bool successful = true;
462 std::string errMsg;
463
464 for (auto i=deferredObservers_.begin();
465 i!=deferredObservers_.end(); ++i) {
466 try {
467 const ext::shared_ptr<Observer::Proxy> proxy = i->lock();
468 if (proxy)
469 proxy->update();
470 } catch (std::exception& e) {
471 successful = false;
472 errMsg = e.what();
473 } catch (...) {
474 successful = false;
475 }
476 }
477
478 deferredObservers_.clear();
479
480 QL_ENSURE(successful,
481 "could not notify one or more observers: " << errMsg);
482 }
