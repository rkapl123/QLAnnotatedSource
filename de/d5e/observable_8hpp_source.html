334 class Proxy {
335 public:
336 explicit Proxy(Observer* const observer)
337 : active_ (true),
338 observer_(observer) {
339 }
340
341 void update() const {
342 std::lock_guard<std::recursive_mutex> lock(mutex_);
343 if (active_) {
344 // c++17 is required if used with std::shared_ptr<T>
345 const ext::weak_ptr<Observer> o
346 = observer_->weak_from_this();
347
348 //check for empty weak reference
349 //https://stackoverflow.com/questions/45507041/how-to-check-if-weak-ptr-is-empty-non-assigned
350 const ext::weak_ptr<Observer> empty;
351 if (o.owner_before(empty) || empty.owner_before(o)) {
352 const ext::shared_ptr<Observer> obs(o.lock());
353 if (obs)
354 obs->update();
355 }
356 else {
357 observer_->update();
358 }
359 }
360 }
361
362 void deactivate() {
363 std::lock_guard<std::recursive_mutex> lock(mutex_);
364 active_ = false;
365 }

367 private:
368 bool active_;
369 mutable std::recursive_mutex mutex_;
370 Observer* const observer_;
371 };

456 inline void ObservableSettings::enableUpdates() {
457 std::lock_guard<std::mutex> lock(mutex_);
458
459 // if there are outstanding deferred updates, do the notification
460 updatesType_ = UpdatesEnabled;
461
462 if (deferredObservers_.size()) {
463 bool successful = true;
464 std::string errMsg;
465
466 for (auto i=deferredObservers_.begin();
467 i!=deferredObservers_.end(); ++i) {
468 try {
469 const ext::shared_ptr<Observer::Proxy> proxy = i->lock();
470 if (proxy)
471 proxy->update();
472 } catch (std::exception& e) {
473 successful = false;
474 errMsg = e.what();
475 } catch (...) {
476 successful = false;
477 }
478 }
479
480 deferredObservers_.clear();
481
482 QL_ENSURE(successful,
483 "could not notify one or more observers: " << errMsg);
484 }
