308  void update() const {
309  boost::lock_guard<boost::recursive_mutex> lock(mutex_);
310  if (active_) {
311  // c++17 is required if used with std::shared_ptr<T>
312  const ext::weak_ptr<Observer> o
313  = observer_->weak_from_this();
314 
315  //check for empty weak reference
316  //https://stackoverflow.com/questions/45507041/how-to-check-if-weak-ptr-is-empty-non-assigned
317  const ext::weak_ptr<Observer> empty;
318  if (o.owner_before(empty) || empty.owner_before(o)) {
319  const ext::shared_ptr<Observer> obs(o.lock());
320  if (obs)
321  obs->update();
322  }
323  else {
324  observer_->update();
325  }
326  }
327  }

330  boost::lock_guard<boost::recursive_mutex> lock(mutex_);
331  active_ = false;
332  }

426  inline void ObservableSettings::enableUpdates() {
427  boost::lock_guard<boost::mutex> lock(mutex_);
428 
429  // if there are outstanding deferred updates, do the notification
430  updatesType_ = UpdatesEnabled;
431 
432  if (deferredObservers_.size()) {
433  bool successful = true;
434  std::string errMsg;
435 
436  for (iterator i=deferredObservers_.begin();
437  i!=deferredObservers_.end(); ++i) {
438  try {
439  const ext::shared_ptr<Observer::Proxy> proxy = i->lock();
440  if (proxy)
441  proxy->update();
442  } catch (std::exception& e) {
443  successful = false;
444  errMsg = e.what();
445  } catch (...) {
446  successful = false;
447  }
448  }
449 
450  deferredObservers_.clear();
451 
452  QL_ENSURE(successful,
453  "could not notify one or more observers: " << errMsg);
454  }
455  }
