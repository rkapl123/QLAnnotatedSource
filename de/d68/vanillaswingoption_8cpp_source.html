31  const Size secPerDay = 24u*3600u;

33  std::pair<std::vector<Date>, std::vector<Size> >
34  createDateTimes(const Date& from, const Date& to, Size stepSize) {
35 
36  std::vector<Size> secs;
37  std::vector<Date> dates;
38 
39  Date iterDate = from;
40  Size iterStepSize = 0u;
41 
42  while (iterDate <= to) {
43  dates.push_back(iterDate);
44  secs.push_back(iterStepSize);
45 
46  iterStepSize+=stepSize;
47  if (iterStepSize >= secPerDay) {
48  iterDate+=1L;
49  iterStepSize%=secPerDay;
50  }
51  }
52 
53  return std::pair<std::vector<Date>,std::vector<Size> >(dates, secs);
54  }
55  }

57  SwingExercise::SwingExercise(const std::vector<Date>& dates,

62  QL_REQUIRE(dates_.size() == seconds_.size(),
63  "dates and seconds must have the same size");
64  for (Size i=0; i < dates_.size(); ++i) {
65  QL_REQUIRE(seconds_[i] < secPerDay,

85  std::vector<Time> SwingExercise::exerciseTimes(const DayCounter& dc,
86  const Date& refDate) const {
87  std::vector<Time> exerciseTimes;
88  exerciseTimes.reserve(dates().size());
89  for (Size i=0; i<dates().size(); ++i) {
90  Time t = dc.yearFraction(refDate, dates()[i]);
91 
92  const Time dt
93  = dc.yearFraction(refDate, dates()[i]+Period(1u, Days)) - t;
94 
95  t += dt*seconds()[i]/(24*3600.);
96 
97  QL_REQUIRE(t >= 0, "exercise dates must not contain past date");
98  exerciseTimes.push_back(t);
99  }
100 
101  return exerciseTimes;
102  }
