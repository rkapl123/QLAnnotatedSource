104  public:
105  XABRInterpolationImpl(const I1& xBegin,
106  const I1& xEnd,
107  const I2& yBegin,
108  Time t,
109  const Real& forward,
110  const std::vector<Real>& params,
111  const std::vector<bool>& paramIsFixed,
112  bool vegaWeighted,
113  ext::shared_ptr<EndCriteria> endCriteria,
114  ext::shared_ptr<OptimizationMethod> optMethod,
115  const Real errorAccept,
116  const bool useMaxError,
117  const Size maxGuesses,
118  const std::vector<Real>& addParams = std::vector<Real>())
119  : Interpolation::templateImpl<I1, I2>(xBegin, xEnd, yBegin, 1),
120  XABRCoeffHolder<Model>(t, forward, params, paramIsFixed, addParams),
121  endCriteria_(std::move(endCriteria)), optMethod_(std::move(optMethod)),
122  errorAccept_(errorAccept), useMaxError_(useMaxError), maxGuesses_(maxGuesses),
123  vegaWeighted_(vegaWeighted) {
124  // if no optimization method or endCriteria is provided, we provide one
125  if (!optMethod_)
126  optMethod_ = ext::shared_ptr<OptimizationMethod>(
127  new LevenbergMarquardt(1e-8, 1e-8, 1e-8));
128  // optMethod_ = ext::shared_ptr<OptimizationMethod>(new
129  // Simplex(0.01));
130  if (!endCriteria_) {
131  endCriteria_ = ext::make_shared<EndCriteria>(
132  60000, 100, 1e-8, 1e-8, 1e-8);
133  }
134  this->weights_ =
135  std::vector<Real>(xEnd - xBegin, 1.0 / (xEnd - xBegin));
136  }

138  void update() override {
139 
140  this->updateModelInstance();
141 
142  // we should also check that y contains positive values only
143 
144  // we must update weights if it is vegaWeighted
145  if (vegaWeighted_) {
146  I1 x = this->xBegin_;
147  I2 y = this->yBegin_;
148  // std::vector<Real>::iterator w = weights_.begin();
149  this->weights_.clear();
150  Real weightsSum = 0.0;
151  for (; x != this->xEnd_; ++x, ++y) {
152  Real stdDev = std::sqrt((*y) * (*y) * this->t_);
153  this->weights_.push_back(Model().weight(*x, this->forward_, stdDev,
154  this->addParams_));
155  weightsSum += this->weights_.back();
156  }
157  // weight normalization
158  auto w = this->weights_.begin();
159  for (; w != this->weights_.end(); ++w)
160  *w /= weightsSum;
161  }
162 
163  // there is nothing to optimize
164  if (std::accumulate(this->paramIsFixed_.begin(),
165  this->paramIsFixed_.end(), true,
166  std::logical_and<bool>())) {
167  this->error_ = interpolationError();
168  this->maxError_ = interpolationMaxError();
169  this->XABREndCriteria_ = EndCriteria::None;
170  return;
171  } else {
172  XABRError costFunction(this);
173 
174  Array guess(Model().dimension());
175  for (Size i = 0; i < guess.size(); ++i)
176  guess[i] = this->params_[i];
177 
178  Size iterations = 0;
179  Size freeParameters = 0;
180  Real bestError = QL_MAX_REAL;
181  Array bestParameters;
182  for (Size i = 0; i < Model().dimension(); ++i)
183  if (!this->paramIsFixed_[i])
184  ++freeParameters;
185  HaltonRsg halton(freeParameters, 42);
186  EndCriteria::Type tmpEndCriteria;
187  Real tmpInterpolationError;
188 
189  do {
190 
191  if (iterations > 0) {
192  HaltonRsg::sample_type s = halton.nextSequence();
193  Model().guess(guess, this->paramIsFixed_, this->forward_,
194  this->t_, s.value, this->addParams_);
195  for (Size i = 0; i < this->paramIsFixed_.size(); ++i)
196  if (this->paramIsFixed_[i])
197  guess[i] = this->params_[i];
198  }
199 
200  Array inversedTransformatedGuess(Model().inverse(
201  guess, this->paramIsFixed_, this->params_, this->forward_));
202 
203  ProjectedCostFunction constrainedXABRError(
204  costFunction, inversedTransformatedGuess,
205  this->paramIsFixed_);
206 
207  Array projectedGuess(
208  constrainedXABRError.project(inversedTransformatedGuess));
209 
210  NoConstraint constraint;
211  Problem problem(constrainedXABRError, constraint,
212  projectedGuess);
213  tmpEndCriteria = optMethod_->minimize(problem, *endCriteria_);
214  Array projectedResult(problem.currentValue());
215  Array transfResult(
216  constrainedXABRError.include(projectedResult));
217 
218  Array result = Model().direct(transfResult, this->paramIsFixed_,
219  this->params_, this->forward_);
220  tmpInterpolationError = useMaxError_ ? interpolationMaxError()
221  : interpolationError();
222 
223  if (tmpInterpolationError < bestError) {
224  bestError = tmpInterpolationError;
225  bestParameters = result;
226  this->XABREndCriteria_ = tmpEndCriteria;
227  }
228 
229  } while (++iterations < maxGuesses_ &&
230  tmpInterpolationError > errorAccept_);
231 
232  for (Size i = 0; i < bestParameters.size(); ++i)
233  this->params_[i] = bestParameters[i];
234 
235  this->error_ = interpolationError();
236  this->maxError_ = interpolationMaxError();
237  }
238  }

246  // calculate total squared weighted difference (L2 norm)
247  Real interpolationSquaredError() const {
248  Real error, totalError = 0.0;
249  I1 x = this->xBegin_;
250  I2 y = this->yBegin_;
251  auto w = this->weights_.begin();
252  for (; x != this->xEnd_; ++x, ++y, ++w) {
253  error = (value(*x) - *y);
254  totalError += error * error * (*w);
255  }
256  return totalError;
257  }
258 
259  // calculate weighted differences
260  Disposable<Array> interpolationErrors() const {
261  Array results(this->xEnd_ - this->xBegin_);
262  I1 x = this->xBegin_;
263  Array::iterator r = results.begin();
264  I2 y = this->yBegin_;
265  auto w = this->weights_.begin();
266  for (; x != this->xEnd_; ++x, ++r, ++w, ++y) {
267  *r = (value(*x) - *y) * std::sqrt(*w);
268  }
269  return results;
270  }
271 
272  Real interpolationError() const {
273  Size n = this->xEnd_ - this->xBegin_;
274  Real squaredError = interpolationSquaredError();
275  return std::sqrt(n * squaredError / (n==1 ? 1 : (n - 1)));
276  }
277 
278  Real interpolationMaxError() const {
279  Real error, maxError = QL_MIN_REAL;
280  I1 i = this->xBegin_;
281  I2 j = this->yBegin_;
282  for (; i != this->xEnd_; ++i, ++j) {
283  error = std::fabs(value(*i) - *j);
284  maxError = std::max(maxError, error);
285  }
286  return maxError;
287  }
