169  T* instance = instance_.load(boost::memory_order_consume);

171  if (!instance) {
172  boost::mutex::scoped_lock guard(mutex_);
173  instance = instance_.load(boost::memory_order_consume);
174  if (!instance) {
175  instance = new T();
176  instance_.store(instance, boost::memory_order_release);
177  }
