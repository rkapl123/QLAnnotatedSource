27  SwapRateTrigger::SwapRateTrigger(const std::vector<Time>& rateTimes,
28  std::vector<Rate> swapTriggers,
29  const std::vector<Time>& exerciseTimes)
30  : rateTimes_(rateTimes), swapTriggers_(std::move(swapTriggers)), exerciseTimes_(exerciseTimes),
31  rateIndex_(exerciseTimes.size()) {
32 
33  checkIncreasingTimes(rateTimes);
34  QL_REQUIRE(rateTimes.size()>1,
35  "Rate times must contain at least two values");
36 
37  checkIncreasingTimes(exerciseTimes);
38 
39  QL_REQUIRE(swapTriggers_.size()==exerciseTimes_.size(),
40  "swapTriggers/exerciseTimes mismatch");
41  Size j = 0;
42  for (Size i=0; i<exerciseTimes.size(); ++i) {
43  while (j < rateTimes.size() && rateTimes[j] < exerciseTimes[i])
44  ++j;
45  rateIndex_[i] = j;
46  }
47  }
