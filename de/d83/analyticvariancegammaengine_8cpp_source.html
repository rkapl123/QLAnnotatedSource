32  class Integrand : std::unary_function<Real,Real> {
33  public:
34  Integrand(const boost::shared_ptr<StrikedTypePayoff>& payoff,
35  Real s0, Real t, Real riskFreeDiscount, Real dividendDiscount,
36  Real sigma, Real nu, Real theta)
37  : payoff_(payoff), s0_(s0), t_(t), riskFreeDiscount_(riskFreeDiscount),
38  dividendDiscount_(dividendDiscount),
39  sigma_(sigma), nu_(nu), theta_(theta) {
40  omega_ = std::log(1.0 - theta_ * nu_ - (sigma_ * sigma_ * nu_) / 2.0) / nu_;
41  // We can precompute the denominator of the gamma pdf (does not depend on x)
42  // shape = t_/nu_, scale = nu_
43  GammaFunction gf;
44  gammaDenom_ = std::exp(gf.logValue(t_ / nu_)) * std::pow(nu_, t_ / nu_);
45  }
46 
47  Real operator()(Real x) const {
48  // Compute adjusted black scholes price
49  Real s0_adj = s0_ * std::exp(theta_ * x + omega_ * t_ + (sigma_ * sigma_ * x) / 2.0);
50  Real vol_adj = sigma_ * std::sqrt(x / t_);
51  vol_adj *= std::sqrt(t_);
52 
53  BlackScholesCalculator bs(payoff_, s0_adj, dividendDiscount_, vol_adj, riskFreeDiscount_);
54  Real bsprice = bs.value();
55 
56  // Multiply by gamma distribution
57  Real gamp = (std::pow(x, t_ / nu_ - 1.0) * std::exp(-x / nu_)) / gammaDenom_;
58  Real result = bsprice * gamp;
59  return result;
60  }
61 
62  private:
63  boost::shared_ptr<StrikedTypePayoff> payoff_;
64  Real s0_;
65  Real t_;
66  Real riskFreeDiscount_;
67  Real dividendDiscount_;
68  Rate sigma_;
69  Real nu_;
70  Real theta_;
71  Real omega_;
72  Real gammaDenom_;
73  };

85  void VarianceGammaEngine::calculate() const {
86 
87  QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
88  "not an European Option");
89 
90  boost::shared_ptr<StrikedTypePayoff> payoff =
91  boost::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
92  QL_REQUIRE(payoff, "non-striked payoff given");
93 
94  DiscountFactor dividendDiscount =
95  process_->dividendYield()->discount(
96  arguments_.exercise->lastDate());
97  DiscountFactor riskFreeDiscount =
98  process_->riskFreeRate()->discount(arguments_.exercise->lastDate());
99 
100  DayCounter rfdc = process_->riskFreeRate()->dayCounter();
101  Time t = rfdc.yearFraction(process_->riskFreeRate()->referenceDate(),
102  arguments_.exercise->lastDate());
103 
104  Integrand f(payoff,
105  process_->x0(),
106  t, riskFreeDiscount, dividendDiscount,
107  process_->sigma(), process_->nu(), process_->theta());
108 
109  Real infinity = 15.0 * std::sqrt(process_->nu() * t);
110  Real target = absErr_*1e-4;
111  Real val = f(infinity);
112  while (std::abs(val)>target){
113  infinity*=1.5;
114  val = f(infinity);
115  }
116  // the integration is split due to occasional singularities at 0
117  Real split = 0.1;
118  GaussKronrodNonAdaptive integrator1(absErr_, 1000, 0);
119  Real pvA = integrator1(f, 0, split);
120  GaussLobattoIntegral integrator2(2000, absErr_);
121  Real pvB = integrator2(f, split, infinity);
122  results_.value = pvA + pvB;
123  }
