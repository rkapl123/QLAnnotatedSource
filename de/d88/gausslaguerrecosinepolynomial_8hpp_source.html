33  class GaussLaguerreTrigonometricBase
34  : public MomentBasedGaussianPolynomial<mp_real> {
35  public:
36  explicit GaussLaguerreTrigonometricBase(Real u)
37  : u_(u) { }

39  protected:
40  virtual mp_real m0() const = 0;
41  virtual mp_real m1() const = 0;

43  mp_real moment_(Size n) const { //NOLINT(bugprone-virtual-near-miss)
44  if (m_.size() <= n)
45  m_.resize(n+1, std::numeric_limits<mp_real>::quiet_NaN());
46 
47  if (boost::math::isnan(m_[n])) {
48  if (n == 0)
49  m_[0] = m0();
50  else if (n == 1)
51  m_[1] = m1();
52  else
53  m_[n] = (2*n*moment_(n-1)
54  - n*(n-1)*moment_(n-2))/(1+u_*u_);
55  }
56 
57  return m_[n];
58  }
59  mp_real fact(Size n) const {
60  if (f_.size() <= n)
61  f_.resize(n+1, std::numeric_limits<mp_real>::quiet_NaN());
62 
63  if (boost::math::isnan(f_[n])) {
64  if (n == 0)
65  f_[0] = 1.0;
66  else
67  f_[n] = n*fact(n-1);
68  }
69  return f_[n];
70 
71  }
72  const Real u_;
73 
74  private:
75  mutable std::vector<mp_real> m_, f_;
76  };
