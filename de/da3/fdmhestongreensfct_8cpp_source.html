44  Disposable<Array> FdmHestonGreensFct::get(Time t, Algorithm algorithm)
45  const {
46  const Rate r = process_->riskFreeRate()->forwardRate(0, t, Continuous);
47  const Rate q = process_->dividendYield()->forwardRate(0,t, Continuous);
48 
49  const Real s0 = process_->s0()->value();
50  const Real v0 = process_->v0();
51  const Real x0 = std::log(s0) + (r-q-0.5*v0*l0_*l0_)*t;
52 
53  const Real rho = process_->rho();
54  const Real theta = process_->theta();
55  const Real kappa = process_->kappa();
56  const Real sigma = process_->sigma();
57 
58  const boost::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
59  const FdmLinearOpIterator endIter = layout->end();
60 
61  Array p(mesher_->layout()->size());
62  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
63  ++iter) {
64  const Real x = mesher_->location(iter, 0);
65  const Real v = (trafoType_ != FdmSquareRootFwdOp::Log)
66  ? mesher_->location(iter, 1)
67  : std::exp(mesher_->location(iter, 1));
68 
69  Real retVal;
70  switch (algorithm) {
71  case ZeroCorrelation:
72  {
73  const Real sd_x = l0_*std::sqrt(v0*t);
74  const Real p_x = M_1_SQRTPI*M_SQRT1_2/sd_x
75  * std::exp(-0.5*square<Real>()((x - x0)/sd_x));
76  const Real p_v = SquareRootProcessRNDCalculator(
77  v0, kappa, theta, sigma).pdf(v, t);
78 
79  retVal = p_v*p_x;
80  }
81  break;
82  case SemiAnalytical:
83  retVal = process_->pdf(x, v, t, 1e-4);
84  break;
85  case Gaussian:
86  {
87  const Real sd_x = l0_*std::sqrt(v0*t);
88  const Real sd_v = sigma*std::sqrt(v0*t);
89  const Real z0 = v0 + kappa*(theta - v0)*t;
90  retVal = 1.0/(M_TWOPI*sd_x*sd_v*std::sqrt(1-rho*rho))
91  *std::exp(-( square<Real>()((x-x0)/sd_x)
92  + square<Real>()((v-z0)/sd_v)
93  - 2*rho*(x-x0)*(v-z0)/(sd_x*sd_v))
94  /(2*(1-rho*rho)) );
95  }
96  break;
97  default:
98  QL_FAIL("unknown algorithm");
99  }
100 
101  switch (trafoType_) {
102  case FdmSquareRootFwdOp::Log:
103  retVal*=v;
104  break;
105  case FdmSquareRootFwdOp::Plain:
106  break;
107  case FdmSquareRootFwdOp::Power:
108  retVal*=std::pow(v, 1.0 - 2*kappa*theta/(sigma*sigma));
109  break;
110  default:
111  QL_FAIL("unknown transformation type");
112  }
113 
114  p[iter.index()] = retVal;
115  }
116 
117  return p;
118  }
