41 Array FdmHestonGreensFct::get(Time t, Algorithm algorithm) const {
42
43 const Rate r = process_->riskFreeRate()->forwardRate(0, t, Continuous);
44 const Rate q = process_->dividendYield()->forwardRate(0,t, Continuous);
45
46 const Real s0 = process_->s0()->value();
47 const Real v0 = process_->v0();
48 const Real x0 = std::log(s0) + (r-q-0.5*v0*l0_*l0_)*t;
49
50 const Real rho = process_->rho();
51 const Real theta = process_->theta();
52 const Real kappa = process_->kappa();
53 const Real sigma = process_->sigma();
54
55 const ext::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
56 const FdmLinearOpIterator endIter = layout->end();
57
58 Array p(mesher_->layout()->size());
59 for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
60 ++iter) {
61 const Real x = mesher_->location(iter, 0);
62 const Real v = (trafoType_ != FdmSquareRootFwdOp::Log)
63 ? mesher_->location(iter, 1)
64 : std::exp(mesher_->location(iter, 1));
65
66 Real retVal;
67 switch (algorithm) {
68 case ZeroCorrelation:
69 {
70 const Real sd_x = l0_*std::sqrt(v0*t);
71 const Real p_x = M_1_SQRTPI*M_SQRT1_2/sd_x
72 * std::exp(-0.5*squared((x - x0)/sd_x));
73 const Real p_v = SquareRootProcessRNDCalculator(
74 v0, kappa, theta, sigma).pdf(v, t);
75
76 retVal = p_v*p_x;
77 }
78 break;
79 case SemiAnalytical:
80 retVal = process_->pdf(x, v, t, 1e-4);
81 break;
82 case Gaussian:
83 {
84 const Real sd_x = l0_*std::sqrt(v0*t);
85 const Real sd_v = sigma*std::sqrt(v0*t);
86 const Real z0 = v0 + kappa*(theta - v0)*t;
87 retVal = 1.0/(M_TWOPI*sd_x*sd_v*std::sqrt(1-rho*rho))
88 *std::exp(-( squared((x-x0)/sd_x)
89 + squared((v-z0)/sd_v)
90 - 2*rho*(x-x0)*(v-z0)/(sd_x*sd_v))
91 /(2*(1-rho*rho)) );
92 }
93 break;
94 default:
95 QL_FAIL("unknown algorithm");
96 }
97
98 switch (trafoType_) {
99 case FdmSquareRootFwdOp::Log:
100 retVal*=v;
101 break;
102 case FdmSquareRootFwdOp::Plain:
103 break;
104 case FdmSquareRootFwdOp::Power:
105 retVal*=std::pow(v, 1.0 - 2*kappa*theta/(sigma*sigma));
106 break;
107 default:
108 QL_FAIL("unknown transformation type");
109 }
110
111 p[iter.index()] = retVal;
112 }
113
114 return p;
115 }
116}
