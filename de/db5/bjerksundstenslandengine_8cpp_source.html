33  Real phi(Real S, Real gamma, Real H, Real I,
34  Real rT, Real bT, Real variance) {
35 
36  Real lambda = (-rT + gamma * bT + 0.5 * gamma * (gamma - 1.0)
37  * variance);
38  Real d = -(std::log(S / H) + (bT + (gamma - 0.5) * variance) )
39  / std::sqrt(variance);
40  Real kappa = 2.0 * bT / variance + (2.0 * gamma - 1.0);
41  return std::exp(lambda) * (cumNormalDist(d)
42  - std::pow((I / S), kappa) *
43  cumNormalDist(d - 2.0 * std::log(I/S) / std::sqrt(variance)));
44  }

47  Real americanCallApproximation(Real S, Real X,
48  Real rfD, Real dD, Real variance) {
49 
50  Real bT = std::log(dD/rfD);
51  Real rT = std::log(1.0/rfD);
52 
53  Real beta = (0.5 - bT/variance) +
54  std::sqrt(std::pow((bT/variance - 0.5), Real(2.0))
55  + 2.0 * rT/variance);
56  Real BInfinity = beta / (beta - 1.0) * X;
57  // Real B0 = std::max(X, std::log(rfD) / std::log(dD) * X);
58  Real B0 = std::max(X, rT / (rT - bT) * X);
59  Real ht = -(bT + 2.0*std::sqrt(variance)) * B0 / (BInfinity - B0);
60 
61  // investigate what happen to I for dD->0.0
62  Real I = B0 + (BInfinity - B0) * (1 - std::exp(ht));
63  QL_REQUIRE(I >= X,
64  "Bjerksund-Stensland approximation not applicable "
65  "to this set of parameters");
66  if (S >= I) {
67  return S - X;
68  } else {
69  // investigate what happen to alpha for dD->0.0
70  return (I - X) * std::pow(S/I, beta)
71  *(1 - phi(S, beta, I, I, rT, bT, variance))
72  + S * phi(S, 1.0, I, I, rT, bT, variance)
73  - S * phi(S, 1.0, X, I, rT, bT, variance)
74  - X * phi(S, 0.0, I, I, rT, bT, variance)
75  + X * phi(S, 0.0, X, I, rT, bT, variance);
76  }
77  }

86  void BjerksundStenslandApproximationEngine::calculate() const {
87 
88  QL_REQUIRE(arguments_.exercise->type() == Exercise::American,
89  "not an American Option");
90 
91  ext::shared_ptr<AmericanExercise> ex =
92  ext::dynamic_pointer_cast<AmericanExercise>(arguments_.exercise);
93  QL_REQUIRE(ex, "non-American exercise given");
94  QL_REQUIRE(!ex->payoffAtExpiry(),
95  "payoff at expiry not handled");
96 
97  ext::shared_ptr<PlainVanillaPayoff> payoff =
98  ext::dynamic_pointer_cast<PlainVanillaPayoff>(arguments_.payoff);
99  QL_REQUIRE(payoff, "non-plain payoff given");
100 
101  Real variance =
102  process_->blackVolatility()->blackVariance(ex->lastDate(),
103  payoff->strike());
104  DiscountFactor dividendDiscount =
105  process_->dividendYield()->discount(ex->lastDate());
106  DiscountFactor riskFreeDiscount =
107  process_->riskFreeRate()->discount(ex->lastDate());
108  Real spot = process_->stateVariable()->value();
109  QL_REQUIRE(spot > 0.0, "negative or null underlying given");
110  Real strike = payoff->strike();
111 
112  if (payoff->optionType()==Option::Put) {
113  // use put-call simmetry
114  std::swap(spot, strike);
115  std::swap(riskFreeDiscount, dividendDiscount);
116  payoff = ext::make_shared<PlainVanillaPayoff>(
117  Option::Call, strike);
118  }
119 
120  if (dividendDiscount>=1.0) {
121  // early exercise is never optimal - use Black formula
122  Real forwardPrice = spot * dividendDiscount / riskFreeDiscount;
123  BlackCalculator black(payoff, forwardPrice, std::sqrt(variance),
124  riskFreeDiscount);
125 
126  results_.value = black.value();
127  results_.delta = black.delta(spot);
128  results_.deltaForward = black.deltaForward();
129  results_.elasticity = black.elasticity(spot);
130  results_.gamma = black.gamma(spot);
131 
132  DayCounter rfdc = process_->riskFreeRate()->dayCounter();
133  DayCounter divdc = process_->dividendYield()->dayCounter();
134  DayCounter voldc = process_->blackVolatility()->dayCounter();
135  Time t =
136  rfdc.yearFraction(process_->riskFreeRate()->referenceDate(),
137  arguments_.exercise->lastDate());
138  results_.rho = black.rho(t);
139 
140  t = divdc.yearFraction(process_->dividendYield()->referenceDate(),
141  arguments_.exercise->lastDate());
142  results_.dividendRho = black.dividendRho(t);
143 
144  t = voldc.yearFraction(process_->blackVolatility()->referenceDate(),
145  arguments_.exercise->lastDate());
146  results_.vega = black.vega(t);
147  results_.theta = black.theta(spot, t);
148  results_.thetaPerDay = black.thetaPerDay(spot, t);
149 
150  results_.strikeSensitivity = black.strikeSensitivity();
151  results_.itmCashProbability = black.itmCashProbability();
152  } else {
153  // early exercise can be optimal - use approximation
154  results_.value = americanCallApproximation(spot,
155  strike,
156  riskFreeDiscount,
157  dividendDiscount,
158  variance);
159  }
160  }
