47  Period PeriodParser::parse(const std::string& str) {
48  QL_REQUIRE(str.length()>1, "period string length must be at least 2");
49 
50  std::vector<std::string > subStrings;
51  std::string reducedString = str;
52 
53  Size iPos, reducedStringDim = 100000, max_iter = 0;
54  while (reducedStringDim>0) {
55  iPos = reducedString.find_first_of("DdWwMmYy");
56  Size subStringDim = iPos+1;
57  reducedStringDim = reducedString.length()-subStringDim;
58  subStrings.push_back(reducedString.substr(0, subStringDim));
59  reducedString = reducedString.substr(iPos+1, reducedStringDim);
60  ++max_iter;
61  QL_REQUIRE(max_iter<str.length(), "unknown '" << str << "' unit");
62  }
63 
64  Period result = parseOnePeriod(subStrings[0]);
65  for (Size i=1; i<subStrings.size(); ++i)
66  result += parseOnePeriod(subStrings[i]);
67  return result;
68  }

70  Period PeriodParser::parseOnePeriod(const std::string& str) {
71  QL_REQUIRE(str.length()>1, "single period require a string of at "
72  "least 2 characters");
73 
74  Size iPos = str.find_first_of("DdWwMmYy");
75  QL_REQUIRE(iPos==str.length()-1, "unknown '" <<
76  str.substr(str.length()-1, str.length()) << "' unit");
77  TimeUnit units = Days;
78  char abbr = static_cast<char>(std::toupper(str[iPos]));
79  if (abbr == 'D') units = Days;
80  else if (abbr == 'W') units = Weeks;
81  else if (abbr == 'M') units = Months;
82  else if (abbr == 'Y') units = Years;
83 
84  Size nPos = str.find_first_of("-+0123456789");
85  QL_REQUIRE(nPos<iPos, "no numbers of " << units << " provided");
86  Integer n;
87  try {
88  n = std::stoi(str.substr(nPos,iPos));
89  } catch (std::exception& e) {
90  QL_FAIL("unable to parse the number of units of " << units <<
91  " in '" << str << "'. Error:" << e.what());
92  }
93 
94  return {n, units};
95  }

97  Date DateParser::parseFormatted(const std::string& str,
98  const std::string& fmt) {
99  #ifndef QL_PATCH_SOLARIS
100  using namespace boost::gregorian;
101 
102  date boostDate;
103  std::istringstream is(str);
104  is.imbue(std::locale(std::locale(), new date_input_facet(fmt)));
105  is >> boostDate;
106  date_duration noDays = boostDate - date(1901, 1, 1);
107  return Date(1, January, 1901) + noDays.days();
