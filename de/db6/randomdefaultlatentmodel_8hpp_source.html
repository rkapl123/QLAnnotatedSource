228  template<template <class, class> class D, class C, class URNG>
229  Probability RandomLM<D, C, URNG>::probAtLeastNEvents(Size n,
230  const Date& d) const
231  {
232  calculate();
233  Date today = Settings::instance().evaluationDate();
234 
235  QL_REQUIRE(d>today, "Date for statistic must be in the future.");
236  // casted to natural to avoid warning, we have just checked the sign
237  Natural val = d.serialNumber() - today.serialNumber();
238 
239  if(n==0) return 1.;

241  Real counts = 0.;
242  for(Size iSim=0; iSim < nSims_; iSim++) {
243  Size simCount = 0;

245  getSim(iSim);
246  for(Size iEvt=0; iEvt < events.size(); iEvt++)

255  template<template <class, class> class D, class C, class URNG>
256  Disposable<std::vector<Probability> >
257  RandomLM<D, C, URNG>::probsBeingNthEvent(Size n,
258  const Date& d) const
259  {
260  calculate();
261  Size basketSize = basket_->size();
262 
263  QL_REQUIRE(n>0 && n<=basketSize, "Impossible number of defaults.");
264  Date today = Settings::instance().evaluationDate();
265 
266  QL_REQUIRE(d>today, "Date for statistic must be in the future.");
267  // casted to natural to avoid warning, we have just checked the sign
268  Natural val = d.serialNumber() - today.serialNumber();
269 
270  std::vector<Probability> hitsByDate(basketSize, 0.);
271  for(Size iSim=0; iSim < nSims_; iSim++) {
272  const std::vector<simEvent<D<C, URNG> > >& events = getSim(iSim);
273  std::map<unsigned short, unsigned short> namesDefaulting;
274  for(Size iEvt=0; iEvt < events.size(); iEvt++) {
275  // if event is within time horizon...
276  if(val > events[iEvt].dayFromRef)
277  //...count it. notice insertion sorts by date.
278  namesDefaulting.insert(std::make_pair<unsigned short,
279  unsigned short>(events[iEvt].dayFromRef,
280  events[iEvt].nameIdx));
281  }
282  if(namesDefaulting.size() >= n) {
283  std::map<unsigned short, unsigned short>::const_iterator
284  itdefs = namesDefaulting.begin();
285  // locate nth default in time:
286  std::advance(itdefs, n-1);
287  // update statistic:
288  hitsByDate[itdefs->second]++;
289  }
290  }
291  std::transform(hitsByDate.begin(), hitsByDate.end(),
292  hitsByDate.begin(),
293  divide_by<Real>(Real(nSims_)));
294  return hitsByDate;
295  // \todo Provide confidence interval
296  }

299  template<template <class, class> class D, class C, class URNG>
300  Real RandomLM<D, C, URNG>::defaultCorrelation(const Date& d,
301  Size iName, Size jName) const
302  {
303  // a control variate with the probabilities is possible
304  calculate();
305  Date today = Settings::instance().evaluationDate();
306 
307  QL_REQUIRE(d>today, "Date for statistic must be in the future.");
308  // casted to natural to avoid warning, we have just checked the sign
309  Natural val = d.serialNumber() - today.serialNumber();
310 
311  Real expectedDefiDefj = 0.;// E[1_i 1_j]
312  // the rest of magnitudes have known values (probabilities) but that
313  // would distort the simulation results.
314  Real expectedDefi = 0.;
315  Real expectedDefj = 0.;
316  for(Size iSim=0; iSim < nSims_; iSim++) {
317  const std::vector<simEvent<D<C, URNG> > >& events = getSim(iSim);
318  Real imatch = 0., jmatch = 0.;
319  for(Size iEvt=0; iEvt < events.size(); iEvt++) {
320  if((val > events[iEvt].dayFromRef) &&
321  (events[iEvt].nameIdx == iName)) imatch = 1.;
322  if((val > events[iEvt].dayFromRef) &&
323  (events[iEvt].nameIdx == jName)) jmatch = 1.;
324  }
325  expectedDefiDefj += imatch * jmatch;
326  expectedDefi += imatch;
327  expectedDefj += jmatch;
328  }
329  expectedDefiDefj = expectedDefiDefj / (nSims_-1);// unbiased
330  expectedDefi = expectedDefi / nSims_;
331  expectedDefj = expectedDefj / nSims_;
332 
333  return (expectedDefiDefj - expectedDefi*expectedDefj) /
334  std::sqrt((expectedDefi*expectedDefj*(1.-expectedDefi)
335  *(1.-expectedDefj)));
336  // \todo Provide confidence interval
337  }

347  template<template <class, class> class D, class C, class URNG>
348  std::pair<Real, Real> RandomLM<D, C, URNG>::expectedTrancheLossInterval(
349  const Date& d, Probability confidencePerc) const
350  {
351  calculate();
352  Date today = Settings::instance().evaluationDate();
353  Date::serial_type val = d.serialNumber() - today.serialNumber();
354 
355  Real attachAmount = basket_->attachmentAmount();
356  Real detachAmount = basket_->detachmentAmount();
357 
358  // Real trancheLoss= 0.;
359  GeneralStatistics lossStats;
360  for(Size iSim=0; iSim < nSims_; iSim++) {
361  const std::vector<simEvent<D<C, URNG> > >& events = getSim(iSim);
362 
363  Real portfSimLoss=0.;
364  for(Size iEvt=0; iEvt < events.size(); iEvt++) {
365  // if event is within time horizon...
366  if(val > static_cast<Date::serial_type>(
367  events[iEvt].dayFromRef)) {
368  Size iName = events[iEvt].nameIdx;
369  // ...and is contained in the basket.
370  portfSimLoss +=
371  basket_->exposure(basket_->names()[iName],
372  Date(events[iEvt].dayFromRef +
373  today.serialNumber())) *
374  (1.-getEventRecovery(events[iEvt]));
375  }
376  }
377  lossStats.add(// d ates? current losses? realized defaults, not yet
378  std::min(std::max(portfSimLoss - attachAmount, 0.),
379  detachAmount - attachAmount) );
380  }
381  return std::make_pair(lossStats.mean(), lossStats.errorEstimate() *
382  InverseCumulativeNormal::standard_value(0.5*(1.+confidencePerc)));
383  }

446  template<template <class, class> class D, class C, class URNG>
447  Real RandomLM<D, C, URNG>::expectedShortfall(const Date& d,
448  Real percent) const {
449 
450  const Date today = Settings::instance().evaluationDate();
451  QL_REQUIRE(d >= today,
452  "Requested percentile date must lie after computation date.");
453  calculate();
454 
455  Real attachAmount = basket_->attachmentAmount();
456  Real detachAmount = basket_->detachmentAmount();
457 
458  Date::serial_type val = d.serialNumber() - today.serialNumber();
459  if(val <= 0) return 0.;// plus basket realized losses
460 
461  //GenericRiskStatistics<GeneralStatistics> statsX;
462  std::vector<Real> losses;
463  for(Size iSim=0; iSim < nSims_; iSim++) {
464  const std::vector<simEvent<D<C, URNG> > >& events = getSim(iSim);
465  Real portfSimLoss=0.;
466  for(Size iEvt=0; iEvt < events.size(); iEvt++) {
467  if(val > static_cast<Date::serial_type>(
468  events[iEvt].dayFromRef)) {
469  Size iName = events[iEvt].nameIdx;
470  // ...and is contained in the basket.
471  //if(basket_->pool()->has(copula_->pool()->names()[iName]))
472  portfSimLoss +=
473  basket_->exposure(basket_->names()[iName],
474  Date(events[iEvt].dayFromRef +
475  today.serialNumber())) *
476  (1.-getEventRecovery(events[iEvt]));
477  }
478  }
479  portfSimLoss = std::min(std::max(portfSimLoss - attachAmount, 0.),
480  detachAmount - attachAmount);
481  losses.push_back(portfSimLoss);
482  }
483 
484  std::sort(losses.begin(), losses.end());
485  Real posit = std::ceil(percent * nSims_);
486  posit = posit >= 0. ? posit : 0.;
487  Size position = static_cast<Size>(posit);
488  Real perctlInf = losses[position];//q_{\alpha}
489 
490  // the prob of values strictly larger than the quantile value.
491  Probability probOverQ =
492  static_cast<Real>(std::distance(losses.begin() + position,
493  losses.end())) / static_cast<Real>(nSims_);
494 
495  return ( perctlInf * (1.-percent-probOverQ) +//<-correction term
496  std::accumulate(losses.begin() + position, losses.end(),
497  Real(0.))/nSims_
498  )/(1.-percent);
499 
500  /* Alternative ESF definition; find the first loss larger than the
501  one of the percentile. Notice the choice here, the expected shortfall
502  is understood in the sense that we are looking for the average given
503  than losses are above a certain value rather than above a certain
504  probability:
505  (Unlikely to be the algorithm of choice)*/
506  /*
507  std::vector<Real>::iterator itPastPerc =
508  std::find_if(losses.begin() + position, losses.end(),
509  greater_or_equal_to<Real>(perctlInf));
510  // notice if the sample is flat at the end this might be zero
511  Size pointsOverVal = nSims_ - std::distance(itPastPerc, losses.end());
512  return pointsOverVal == 0 ? 0. :
513  std::accumulate(itPastPerc, losses.end(), 0.) / pointsOverVal;
514  */
515 
516  /* For the definition of ESF see for instance: 'Quantitative Risk
517  Management' by A.J. McNeil, R.Frey and P.Embrechts, princeton series in
518  finance, 2005; equations on page 39 sect 2.12:
519  $q_{\alpha}(F) = inf{x \in R : F(x) \le \alpha}$
520  and equation 2.25 on p. 45:
521  $ESF_{\alpha} = \frac{1}{1-\alpha} [E(L; L \ge q_{\alpha} ) +
522  q_{\alpha} (1-\alpha-P(L \ge q_{\alpha})) ]$
523  The second term accounts for non continuous distributions.
524  */
525  }

535  /* See Appendix-A of "Evaluating value-at-risk methodologies: Accuracy
536  versus computational time.", M. Pritsker, Wharton FIC, November 1996

538  the true value being within the interval; which is different to the error

540  */
541  template<template <class, class> class D, class C, class URNG>
542  ext::tuple<Real, Real, Real> // disposable?
543  RandomLM<D, C, URNG>::percentileAndInterval(const Date& d,
544  Real percentile) const {
545 
546  QL_REQUIRE(percentile >= 0. && percentile <= 1.,
547  "Incorrect percentile");
548  calculate();
549 
550  Real attachAmount = basket_->attachmentAmount();
551  Real detachAmount = basket_->detachmentAmount();
552 
553  std::vector<Real> rankLosses;
554  Date today = Settings::instance().evaluationDate();
555  Date::serial_type val = d.serialNumber() - today.serialNumber();
556  for(Size iSim=0; iSim < nSims_; iSim++) {
557  const std::vector<simEvent<D<C, URNG> > >& events = getSim(iSim);
558  Real portfSimLoss=0.;
559  for(Size iEvt=0; iEvt < events.size(); iEvt++) {
560  if(val > static_cast<Date::serial_type>(
561  events[iEvt].dayFromRef)) {
562  Size iName = events[iEvt].nameIdx;
563  // if(basket_->pool()->has(copula_->pool()->names()[iName]))
564  portfSimLoss +=
565  basket_->exposure(basket_->names()[iName],
566  Date(events[iEvt].dayFromRef +
567  today.serialNumber())) *
568  (1.-getEventRecovery(events[iEvt]));
569  }
570  }
571  portfSimLoss = std::min(std::max(portfSimLoss - attachAmount, 0.),
572  detachAmount - attachAmount);
573  // update dataset for rank stat:
574  rankLosses.push_back(portfSimLoss);
575  }
576 
577  std::sort(rankLosses.begin(), rankLosses.end());
578  Size quantilePosition = static_cast<Size>(floor(nSims_*percentile));
579  Real quantileValue = rankLosses[quantilePosition];
580 
581  // compute confidence interval:
582  const Probability confInterval = 0.95;// as an argument?
583  Real lowerPercentile, upperPercentile;
584  Size r = quantilePosition - 1;
585  Size s = quantilePosition + 1;
586  bool rLocked = false,
587  sLocked = false;
588  // Size rfinal = 0,
589  // sfinal = 0;
590  for(Size delta=1; delta < quantilePosition; delta++) {
591  Real cached =
592  incompleteBetaFunction(Real(s), Real(nSims_+1-s),
593  percentile, 1.e-8, 500);
594  Real pMinus =
595  /* There was a fix in the repository on the gammadistribution. It
596  might impact these, it might be neccesary to multiply these values
597  by '-1'*/
598  incompleteBetaFunction(Real(r+1), Real(nSims_-r),
599  percentile, 1.e-8, 500)
600  - cached;
601  Real pPlus =
602  incompleteBetaFunction(Real(r), Real(nSims_-r+1),
603  percentile, 1.e-8, 500)
604  - cached;
605  if((pMinus > confInterval) && !rLocked ) {
606  // rfinal = r + 1;
607  rLocked = true;
608  }
609  if((pPlus >= confInterval) && !sLocked) {
610  // sfinal = s;
611  sLocked = true;
612  }
613  if(rLocked && sLocked) break;
614  r--;
615  s++;
616  s = std::min(nSims_-1, s);
617  }
618  lowerPercentile = rankLosses[r];
619  upperPercentile = rankLosses[s];
620 
621  return {quantileValue, lowerPercentile, upperPercentile};
622  }

658

660  const std::vector<simEvent<D<C, URNG> > >& events = getSim(iSim);
661  Real portfSimLoss=0.;
662  //std::vector<Real> splitBuffer(numLiveNames_, 0.);
663  std::vector<simEvent<D<C, URNG> > > splitEventsBuffer;
664 
665  for(Size iEvt=0; iEvt < events.size(); iEvt++) {
666  if(val > static_cast<Date::serial_type>(
667  events[iEvt].dayFromRef)) {
668  Size iName = events[iEvt].nameIdx;
669  // if(basket_->pool()->has(copula_->pool()->names()[iName])) {
670  portfSimLoss +=

675  //and will sort later if buffer applies:
676  splitEventsBuffer.push_back(events[iEvt]);
677  }
678  }
679  portfSimLoss = std::min(std::max(portfSimLoss - attachAmount, 0.),
680  detachAmount - attachAmount);
681 
682  /* second pass; split is conditional to total losses within target
683  losses/percentile: */
684  Real ptflCumulLoss = 0.;
685  if(portfSimLoss > loss) {
686  std::sort(splitEventsBuffer.begin(), splitEventsBuffer.end());
687  //NOW THIS:
688  split.assign(numLiveNames, 0.);
689  /* if the name triggered a loss in the portf limits assign
690  this loss to that name.. */
691  for(Size i=0; i<splitEventsBuffer.size(); i++) {
692  Size iName = splitEventsBuffer[i].nameIdx;
693  Real lossName =
694  // allows amortizing (others should be like this)
695  // basket_->remainingNotionals(Date(simsBuffer_[i].dayFromRef +
696  // today.serialNumber()))[iName] *
697  basket_->exposure(basket_->names()[iName],
698  Date(splitEventsBuffer[i].dayFromRef +
699  today.serialNumber())) *
700  (1.-getEventRecovery(splitEventsBuffer[i]));

703  std::min(std::max(ptflCumulLoss - attachAmount, 0.),
704  detachAmount - attachAmount);
705  ptflCumulLoss += lossName;
706  Real tranchedLossAfter =
707  std::min(std::max(ptflCumulLoss - attachAmount, 0.),
708  detachAmount - attachAmount);
709  // assign new losses:
710  split[iName] += tranchedLossAfter - tranchedLossBefore;
711  }
712  for(Size iName=0; iName<numLiveNames; iName++) {
713  splitStats[iName].add(split[iName] /
714  std::min(std::max(ptflCumulLoss - attachAmount, 0.),
715  detachAmount - attachAmount) );
716  }
717  }
718  }

721  std::vector<Real> means, rangeUp, rangeDown;
722  Real confidFactor = InverseCumulativeNormal()(0.5+confInterval/2.);
723  for(Size iName=0; iName<numLiveNames; iName++) {
724  means.push_back(splitStats[iName].mean());
725  Real error = confidFactor * splitStats[iName].errorEstimate() ;
726  rangeDown.push_back(means.back() - error);
727  rangeUp.push_back(means.back() + error);
728  }
729 
730  std::vector<std::vector<Real> > results;
731  results.push_back(means);
732  results.push_back(rangeDown);
733  results.push_back(rangeUp);
734
