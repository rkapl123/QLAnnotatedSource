142  template <class RNG, class S>
143  inline TimeGrid MCDiscreteAveragingAsianEngine<RNG,S>::timeGrid() const {
144 
145  Date referenceDate = process_->riskFreeRate()->referenceDate();
146  DayCounter voldc = process_->blackVolatility()->dayCounter();
147  std::vector<Time> fixingTimes;
148  Size i;
149  for (i=0; i<arguments_.fixingDates.size(); i++) {
150  if (arguments_.fixingDates[i]>=referenceDate) {
151  Time t = voldc.yearFraction(referenceDate,
152  arguments_.fixingDates[i]);
153  fixingTimes.push_back(t);
154  }
155  }
156 
157  if (fixingTimes.empty() ||
158  (fixingTimes.size() == 1 && fixingTimes.front() == 0.0))
159  throw detail::PastFixingsOnly();
160 
161  return TimeGrid(fixingTimes.begin(), fixingTimes.end());
162  }
