52  Real ExtendedCoxIngersollRoss::A(Time t, Time s) const {
53  Real pt = termStructure()->discount(t);
54  Real ps = termStructure()->discount(s);
55  Real value = CoxIngersollRoss::A(t,s)*std::exp(B(t,s)*phi_(t))*
56  (ps*CoxIngersollRoss::A(0.0,t)*std::exp(-B(0.0,t)*x0()))/
57  (pt*CoxIngersollRoss::A(0.0,s)*std::exp(-B(0.0,s)*x0()));
58  return value;
59  }

61  Real ExtendedCoxIngersollRoss::discountBondOption(Option::Type type,
62  Real strike,
63  Time t, Time s) const {
64 
65  QL_REQUIRE(strike>0.0, "strike must be positive");
66 
67  DiscountFactor discountT = termStructure()->discount(t);
68  DiscountFactor discountS = termStructure()->discount(s);
69  if (t < QL_EPSILON) {
70  switch(type) {
71  case Option::Call:
72  return std::max<Real>(discountS - strike, 0.0);
73  case Option::Put:
74  return std::max<Real>(strike - discountS, 0.0);
75  default: QL_FAIL("unsupported option type");
76  }
77  }
78 
79  Real sigma2 = sigma()*sigma();
80  Real h = std::sqrt(k()*k() + 2.0*sigma2);
81  Real r0 = termStructure()->forwardRate(0.0, 0.0,
82  Continuous, NoFrequency);
83  Real b = B(t,s);
84 
85  Real rho = 2.0*h/(sigma2*(std::exp(h*t) - 1.0));
86  Real psi = (k() + h)/sigma2;
87 
88  Real df = 4.0*k()*theta()/sigma2;
89  Real ncps = 2.0*rho*rho*(r0-phi_(0.0))*std::exp(h*t)/(rho+psi+b);
90  Real ncpt = 2.0*rho*rho*(r0-phi_(0.0))*std::exp(h*t)/(rho+psi);
91 
92  NonCentralCumulativeChiSquareDistribution chis(df, ncps);
93  NonCentralCumulativeChiSquareDistribution chit(df, ncpt);
94 
95  Real z = std::log(CoxIngersollRoss::A(t,s)/strike)/b;
96  Real call = discountS*chis(2.0*z*(rho+psi+b)) -
97  strike*discountT*chit(2.0*z*(rho+psi));
98  if (type == Option::Call)
99  return call;
100  else
101  return call - discountS + strike*discountT;
102  }
