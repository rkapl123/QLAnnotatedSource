34  FdmOrnsteinUhlenbeckOp::FdmOrnsteinUhlenbeckOp(
35  const ext::shared_ptr<FdmMesher>& mesher,
36  ext::shared_ptr<OrnsteinUhlenbeckProcess> process,
37  ext::shared_ptr<YieldTermStructure> rTS,
38  Size direction)
39  : mesher_(mesher), process_(std::move(process)), rTS_(std::move(rTS)), direction_(direction),
40  m_(direction, mesher), mapX_(direction, mesher) {
41 
42  const ext::shared_ptr<FdmLinearOpLayout> layout=mesher_->layout();
43 
44  Array drift(layout->size());
45  const Array x(mesher_->locations(direction));
46 
47  for (FdmLinearOpIterator iter=layout->begin(), endIter=layout->end();
48  iter!=endIter; ++iter) {
49  const Size i = iter.index();
50  drift[i] = process_->drift(0.0, x[i]);
51  }
52 
53  m_.axpyb(drift, FirstDerivativeOp(direction, mesher),
54  SecondDerivativeOp(direction, mesher)
55  .mult(0.5*square<Real>()(process_->volatility())
56  *Array(mesher->layout()->size(), 1.0)), Array());
57  }
