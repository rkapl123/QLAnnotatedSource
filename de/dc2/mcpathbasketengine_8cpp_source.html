31  Real EuropeanPathMultiPathPricer::operator()(const MultiPath& multiPath)
32  const {
33 
34  Size n = multiPath.pathSize();
35  QL_REQUIRE(n > 0, "the path cannot be empty");
36 
37  Size numberOfAssets = multiPath.assetNumber();
38  QL_REQUIRE(numberOfAssets > 0, "there must be some paths");
39 
40  const Size numberOfTimes = timePositions_.size();
41 
42  Matrix path(numberOfAssets, numberOfTimes, Null<Real>());
43 
44  for (Size i = 0; i < numberOfTimes; ++i) {
45  const Size pos = timePositions_[i];
46  for (Size j = 0; j < numberOfAssets; ++j)
47  path[j][i] = multiPath[j][pos];
48  }
49 
50  Array values(numberOfTimes, 0.0);
51 
52  // ignored
53  Array exercises;
54  std::vector<Array> states;
55 
56  payoff_->value(path, forwardTermStructures_, values, exercises, states);
57 
58  // in this engine we ignore early exercise
59 
60  Real discountedPayoff = DotProduct(values, discounts_);
61 
62  return discountedPayoff;
63  }
