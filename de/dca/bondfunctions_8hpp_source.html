164  Frequency frequency,
165  Date settlementDate = Date(),
166  Real accuracy = 1.0e-10,
167  Rate guess = 0.05) {
168  if (settlementDate == Date())
169  settlementDate = bond.settlementDate();
170 
171  QL_REQUIRE(BondFunctions::isTradable(bond, settlementDate),
172  "non tradable at " << settlementDate <<
173  " (maturity being " << bond.maturityDate() << ")");
174 
175  Real dirtyPrice = cleanPrice + bond.accruedAmount(settlementDate);
176  dirtyPrice /= 100.0 / bond.notional(settlementDate);
177 
178  return CashFlows::yield<Solver>(solver, bond.cashflows(),
179  dirtyPrice, dayCounter, compounding,
180  frequency, false, settlementDate,
181  settlementDate, accuracy, guess);
