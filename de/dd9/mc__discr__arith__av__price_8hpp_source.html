121  MCDiscreteArithmeticAPEngine<RNG,S>::pathPricer() const {
122 
123  ext::shared_ptr<PlainVanillaPayoff> payoff =
124  ext::dynamic_pointer_cast<PlainVanillaPayoff>(
125  this->arguments_.payoff);

135  new ArithmeticAPOPathPricer(
136  payoff->optionType(),
137  payoff->strike(),
138  this->process_->riskFreeRate()->discount(

140  this->arguments_.runningAccumulator,
