26  void historicalRatesAnalysis(
27  SequenceStatistics& statistics,
28  std::vector<Date>& skippedDates,
29  std::vector<std::string>& skippedDatesErrorMessage,
30  const Date& startDate,
31  const Date& endDate,
32  const Period& step,
33  const std::vector<boost::shared_ptr<InterestRateIndex> >& indexes) {
34 
35  skippedDates.clear();
36  skippedDatesErrorMessage.clear();
37 
38  Size nRates = indexes.size();
39  statistics.reset(nRates);
40 
41  std::vector<Rate> sample(nRates);
42  std::vector<Rate> prevSample(nRates);
43  std::vector<Rate> sampleDiff(nRates);
44 
45  Calendar cal = indexes[0]->fixingCalendar();
46  // start with a valid business date
47  Date currentDate = cal.advance(startDate, 1*Days, Following);
48  bool isFirst = true;
49  // Loop over the historical dataset
50  for (; currentDate<=endDate;
51  currentDate = cal.advance(currentDate, step, Following)) {
52 
53  try {
54  for (Size i=0; i<nRates; ++i) {
55  Rate fixing = indexes[i]->fixing(currentDate, false);
56  sample[i] = fixing;
57  }
58  } catch (std::exception& e) {
59  skippedDates.push_back(currentDate);
60  skippedDatesErrorMessage.push_back(e.what());
61  continue;
62  }
63 
64  // From 2nd step onwards, calculate forward rate
65  // relative differences
66  if (!isFirst){
67  for (Size i=0; i<nRates; ++i)
68  sampleDiff[i] = sample[i]/prevSample[i] -1.0;
69  // add observation
70  statistics.add(sampleDiff.begin(), sampleDiff.end());
71  }
72  else
73  isFirst = false;
74 
75  // Store last calculated forward rates
76  std::swap(prevSample, sample);
77 
78  }
79  }
