33  FdmHestonVarianceMesher::FdmHestonVarianceMesher(
34  Size size,
35  const boost::shared_ptr<HestonProcess> & process,
36  Time maturity, Size tAvgSteps, Real epsilon)
37  : Fdm1dMesher(size) {
38 
39  std::vector<Real> vGrid(size, 0.0), pGrid(size, 0.0);
40  const Real df = 4*process->theta()*process->kappa()/
41  square<Real>()(process->sigma());
42  try {
43  std::multiset<std::pair<Real, Real> > grid;
44 
45  for (Size l=1; l<=tAvgSteps; ++l) {
46  const Real t = (maturity*l)/tAvgSteps;
47  const Real ncp = 4*process->kappa()*std::exp(-process->kappa()*t)
48  /(square<Real>()(process->sigma())
49  *(1-std::exp(-process->kappa()*t)))*process->v0();
50  const Real k = square<Real>()(process->sigma())
51  *(1-std::exp(-process->kappa()*t))/(4*process->kappa());
52 
53  const Real qMin = 0.0; // v_min = 0.0;
54  const Real qMax = std::max(process->v0(),
55  k*InverseNonCentralCumulativeChiSquareDistribution(
56  df, ncp, 100, 1e-8)(1-epsilon));
57 
58  const Real minVStep=(qMax-qMin)/(50*size);
59  Real ps,p = 0.0;
60 
61  Real vTmp = qMin;
62  grid.insert(std::pair<Real, Real>(qMin, epsilon));
63 
64  for (Size i=1; i < size; ++i) {
65  ps = (1 - epsilon - p)/(size-i);
66  p += ps;
67  const Real tmp = k*InverseNonCentralCumulativeChiSquareDistribution(
68  df, ncp, 100, 1e-8)(p);
69 
70  const Real vx = std::max(vTmp+minVStep, tmp);
71  p = NonCentralCumulativeChiSquareDistribution(df, ncp)(vx/k);
72  vTmp=vx;
73  grid.insert(std::pair<Real, Real>(vx, p));
74  }
75  }
76  QL_REQUIRE(grid.size() == size*tAvgSteps,
77  "something wrong with the grid size");
78 
79  std::vector<std::pair<Real, Real> > tp(grid.begin(), grid.end());
80 
81  for (Size i=0; i < size; ++i) {
82  const Size b = (i*tp.size())/size;
83  const Size e = ((i+1)*tp.size())/size;
84  for (Size j=b; j < e; ++j) {
85  vGrid[i]+=tp[j].first/(e-b);
86  pGrid[i]+=tp[j].second/(e-b);
87  }
88  }
89  }
90  catch (const Error&) {
91  // use default mesh
92  const Real vol = process->sigma()*
93  std::sqrt(process->theta()/(2*process->kappa()));
94 
95  const Real mean = process->theta();
96  const Real upperBound = std::max(process->v0()+4*vol, mean+4*vol);
97  const Real lowerBound
98  = std::max(0.0, std::min(process->v0()-4*vol, mean-4*vol));
99 
100  for (Size i=0; i < size; ++i) {
101  pGrid[i] = i/(size-1.0);
102  vGrid[i] = lowerBound + i*(upperBound-lowerBound)/(size-1.0);
103  }
104  }
105 
106  Real skewHint = ((process->kappa() != 0.0)
107  ? std::max(1.0, process->sigma()/process->kappa()) : 1.0);
108 
109  std::sort(pGrid.begin(), pGrid.end());
110  volaEstimate_ = GaussLobattoIntegral(100000, 1e-4)(
111  boost::function1<Real, Real>(
112  compose(std::ptr_fun<Real, Real>(std::sqrt),
113  LinearInterpolation(pGrid.begin(), pGrid.end(),
114  vGrid.begin()))),
115  pGrid.front(), pGrid.back())*std::pow(skewHint, 1.5);
116 
117  const Real v0 = process->v0();
118  for (Size i=1; i<vGrid.size(); ++i) {
119  if (vGrid[i-1] <= v0 && vGrid[i] >= v0) {
120  if (std::fabs(vGrid[i-1] - v0) < std::fabs(vGrid[i] - v0))
121  vGrid[i-1] = v0;
122  else
123  vGrid[i] = v0;
124  }
125  }
126 
127  std::copy(vGrid.begin(), vGrid.end(), locations_.begin());
128 
129  for (Size i=0; i < size-1; ++i) {
130  dminus_[i+1] = dplus_[i] = vGrid[i+1] - vGrid[i];
131  }
132  dplus_.back() = dminus_.front() = Null<Real>();
133  }
134 }
