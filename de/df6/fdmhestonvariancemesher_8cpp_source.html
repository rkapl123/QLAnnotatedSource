58  FdmHestonVarianceMesher::FdmHestonVarianceMesher(
59  Size size,
60  const ext::shared_ptr<HestonProcess> & process,
61  Time maturity, Size tAvgSteps, Real epsilon,
62  Real mixingFactor)
63  : Fdm1dMesher(size) {
64 
65  std::vector<Real> vGrid(size, 0.0), pGrid(size, 0.0);
66  const Real mixedSigma = process->sigma()*mixingFactor;
67  const Real df = 4*process->theta()*process->kappa()/
68  square<Real>()(mixedSigma);
69  try {
70  std::multiset<std::pair<Real, Real> > grid;
71 
72  for (Size l=1; l<=tAvgSteps; ++l) {
73  const Real t = (maturity*l)/tAvgSteps;
74  const Real ncp = 4*process->kappa()*std::exp(-process->kappa()*t)
75  /(square<Real>()(mixedSigma)
76  *(1-std::exp(-process->kappa()*t)))*process->v0();
77  const Real k = square<Real>()(mixedSigma)
78  *(1-std::exp(-process->kappa()*t))/(4*process->kappa());
79 
80  const Real qMin = 0.0; // v_min = 0.0;
81  const Real qMax = std::max(process->v0(),
82  k*InverseNonCentralCumulativeChiSquareDistribution(
83  df, ncp, 100, 1e-8)(1-epsilon));
84 
85  const Real minVStep=(qMax-qMin)/(50*size);
86  Real ps,p = 0.0;
87 
88  Real vTmp = qMin;
89  grid.insert(std::pair<Real, Real>(qMin, epsilon));
90 
91  for (Size i=1; i < size; ++i) {
92  ps = (1 - epsilon - p)/(size-i);
93  p += ps;
94  const Real tmp = k*InverseNonCentralCumulativeChiSquareDistribution(
95  df, ncp, 100, 1e-8)(p);
96 
97  const Real vx = std::max(vTmp+minVStep, tmp);
98  p = NonCentralCumulativeChiSquareDistribution(df, ncp)(vx/k);
99  vTmp=vx;
100  grid.insert(std::pair<Real, Real>(vx, p));
101  }
102  }
103  QL_REQUIRE(grid.size() == size*tAvgSteps,
104  "something wrong with the grid size");
105 
106  const std::vector<std::pair<Real, Real> > tp(grid.begin(), grid.end());
107 
108  for (Size i=0; i < size; ++i) {
109  const Size b = (i*tp.size())/size;
110  const Size e = ((i+1)*tp.size())/size;
111  for (Size j=b; j < e; ++j) {
112  vGrid[i]+=tp[j].first/(e-b);
113  pGrid[i]+=tp[j].second/(e-b);
114  }
115  }
116  }
117  catch (const Error&) {
118  // use default mesh
119  const Real vol = mixedSigma*
120  std::sqrt(process->theta()/(2*process->kappa()));
121 
122  const Real mean = process->theta();
123  const Real upperBound = std::max(process->v0()+4*vol, mean+4*vol);
124  const Real lowerBound
125  = std::max(0.0, std::min(process->v0()-4*vol, mean-4*vol));
126 
127  for (Size i=0; i < size; ++i) {
128  pGrid[i] = i/(size-1.0);
129  vGrid[i] = lowerBound + i*(upperBound-lowerBound)/(size-1.0);
130  }
131  }
132 
133  Real skewHint = ((process->kappa() != 0.0)
134  ? std::max(1.0, mixedSigma/process->kappa()) : 1.0);
135 
136  std::sort(pGrid.begin(), pGrid.end());
137  volaEstimate_ = GaussLobattoIntegral(100000, 1e-4)(
138  interpolated_volatility(pGrid, vGrid),
139  pGrid.front(), pGrid.back())*std::pow(skewHint, 1.5);
140 
141  const Real v0 = process->v0();
142  for (Size i=1; i<vGrid.size(); ++i) {
143  if (vGrid[i-1] <= v0 && vGrid[i] >= v0) {
144  if (std::fabs(vGrid[i-1] - v0) < std::fabs(vGrid[i] - v0))
145  vGrid[i-1] = v0;
146  else
147  vGrid[i] = v0;
148  }
149  }
150 
151  std::copy(vGrid.begin(), vGrid.end(), locations_.begin());
152 
153  for (Size i=0; i < size-1; ++i) {
154  dminus_[i+1] = dplus_[i] = vGrid[i+1] - vGrid[i];
155  }
156  dplus_.back() = dminus_.front() = Null<Real>();
157  }

160  FdmHestonLocalVolatilityVarianceMesher::FdmHestonLocalVolatilityVarianceMesher(
161  Size size,
162  const ext::shared_ptr<HestonProcess>& process,
163  const ext::shared_ptr<LocalVolTermStructure>& leverageFct,
164  Time maturity, Size tAvgSteps, Real epsilon,
165  Real mixingFactor)
166  : Fdm1dMesher(size) {
167 
168  const FdmHestonVarianceMesher mesher(
169  size, process, maturity, tAvgSteps, epsilon, mixingFactor);
170 
171  for (Size i=0; i < size; ++i) {
172  dplus_[i] = mesher.dplus(i);
173  dminus_[i] = mesher.dminus(i);
174  locations_[i] = mesher.location(i);
175  }
176 
177  volaEstimate_ = mesher.volaEstimate();
178 
179  if (leverageFct != 0) {
180  typedef boost::accumulators::accumulator_set<
181  Real, boost::accumulators::stats<
182  boost::accumulators::tag::mean> >
183  accumulator_set;
184 
185  accumulator_set acc;
186 
187  const Real s0 = process->s0()->value();
188 
189  acc(leverageFct->localVol(0, s0, true));
190 
191  const Handle<YieldTermStructure> rTS = process->riskFreeRate();
192  const Handle<YieldTermStructure> qTS = process->dividendYield();
193 
194  for (Size l=1; l <= tAvgSteps; ++l) {
195  const Real t = (maturity*l)/tAvgSteps;
196  const Real vol = volaEstimate_ * boost::accumulators::mean(acc);
197 
198  const Real fwd = s0*qTS->discount(t)/rTS->discount(t);
199 
200  const Size sAvgSteps = 50;
201 
202  std::vector<Real> u(sAvgSteps), sig(sAvgSteps);
203 
204  for (Size i=0; i < sAvgSteps; ++i) {
205  u[i] = epsilon + ((1-2*epsilon)/(sAvgSteps-1))*i;
206  const Real x = InverseCumulativeNormal()(u[i]);
207 
208  const Real gf = x*vol*std::sqrt(t);
209  const Real f = fwd*std::exp(gf);
210 
211  sig[i] = square<Real>()(leverageFct->localVol(t, f, true));
212  }
213 
214  const Real leverageAvg =
215  GaussLobattoIntegral(10000, 1e-4)(
216  interpolated_volatility(u, sig), u.front(), u.back())
217  / (1-2*epsilon);
218 
219  acc(leverageAvg);
220  }
221  volaEstimate_ *= boost::accumulators::mean(acc);
222  }
223  }
