64  void AnalyticBinaryBarrierEngine::calculate() const {
65 
66  ext::shared_ptr<AmericanExercise> ex =
67  ext::dynamic_pointer_cast<AmericanExercise>(arguments_.exercise);
68  QL_REQUIRE(ex, "non-American exercise given");
69  QL_REQUIRE(ex->payoffAtExpiry(), "payoff must be at expiry");
70  QL_REQUIRE(ex->dates()[0] <=
71  process_->blackVolatility()->referenceDate(),
72  "American option with window exercise not handled yet");
73 
74  ext::shared_ptr<StrikedTypePayoff> payoff =
75  ext::dynamic_pointer_cast<StrikedTypePayoff>(arguments_.payoff);
76  QL_REQUIRE(payoff, "non-striked payoff given");
77 
78  Real spot = process_->stateVariable()->value();
79  QL_REQUIRE(spot > 0.0, "negative or null underlying given");
80 
81  Real variance =
82  process_->blackVolatility()->blackVariance(ex->lastDate(),
83  payoff->strike());
84  Real barrier = arguments_.barrier;
85  QL_REQUIRE(barrier>0.0,
86  "positive barrier value required");
87  Barrier::Type barrierType = arguments_.barrierType;
88 
89  // KO degenerate cases
90  if ( (barrierType == Barrier::DownOut && spot <= barrier) ||
91  (barrierType == Barrier::UpOut && spot >= barrier))
92  {
93  // knocked out, no value
94  results_.value = 0;
95  results_.delta = 0;
96  results_.gamma = 0;
97  results_.vega = 0;
98  results_.theta = 0;
99  results_.rho = 0;
100  results_.dividendRho = 0;
101  return;
102  }
103 
104  // KI degenerate cases
105  if ((barrierType == Barrier::DownIn && spot <= barrier) ||
106  (barrierType == Barrier::UpIn && spot >= barrier)) {
107  // knocked in - is a digital european
108  ext::shared_ptr<Exercise> exercise(new EuropeanExercise(
109  arguments_.exercise->lastDate()));
110 
111  ext::shared_ptr<PricingEngine> engine(
112  new AnalyticEuropeanEngine(process_));
113 
114  VanillaOption opt(payoff, exercise);
115  opt.setPricingEngine(engine);
116  results_.value = opt.NPV();
117  results_.delta = opt.delta();
118  results_.gamma = opt.gamma();
119  results_.vega = opt.vega();
120  results_.theta = opt.theta();
121  results_.rho = opt.rho();
122  results_.dividendRho = opt.dividendRho();
123  return;
124  }
125 
126  Rate riskFreeDiscount =
127  process_->riskFreeRate()->discount(ex->lastDate());
128 
129  AnalyticBinaryBarrierEngine_helper helper(process_,
130  payoff, ex, arguments_);
131  results_.value = helper.payoffAtExpiry(spot, variance, riskFreeDiscount);
132  }

134  Real AnalyticBinaryBarrierEngine_helper::payoffAtExpiry(
135  Real spot, Real variance, Real discount)
136  {
137  Rate dividendDiscount =
138  process_->dividendYield()->discount(exercise_->lastDate());
139 
140  QL_REQUIRE(spot>0.0,
141  "positive spot value required");
142 
143  QL_REQUIRE(discount>0.0,
144  "positive discount required");
145 
146  QL_REQUIRE(dividendDiscount>0.0,
147  "positive dividend discount required");
148 
149  QL_REQUIRE(variance>=0.0,
150  "negative variance not allowed");
151 
152  Option::Type type = payoff_->optionType();
153  Real strike = payoff_->strike();
154  Real barrier = arguments_.barrier;
155  QL_REQUIRE(barrier>0.0,
156  "positive barrier value required");
157  Barrier::Type barrierType = arguments_.barrierType;
158 
159  Real stdDev = std::sqrt(variance);
160  Real mu = std::log(dividendDiscount/discount)/variance - 0.5;
161  Real K = 0;
162 
163  // binary cash-or-nothing payoff?
164  ext::shared_ptr<CashOrNothingPayoff> coo =
165  ext::dynamic_pointer_cast<CashOrNothingPayoff>(payoff_);
166  if (coo != 0) {
167  K = coo->cashPayoff();
168  }
169 
170  // binary asset-or-nothing payoff?
171  ext::shared_ptr<AssetOrNothingPayoff> aoo =
172  ext::dynamic_pointer_cast<AssetOrNothingPayoff>(payoff_);
173  if (aoo != 0) {
174  mu += 1.0;
175  K = spot * dividendDiscount / discount; // forward
176  }
177 
178  Real log_S_X = std::log(spot/strike);
179  Real log_S_H = std::log(spot/barrier);
180  Real log_H_S = std::log(barrier/spot);
181  Real log_H2_SX = std::log(barrier*barrier/(spot*strike));
182  Real H_S_2mu = std::pow(barrier/spot, 2*mu);
183 
184  Real eta = (barrierType == Barrier::DownIn ||
185  barrierType == Barrier::DownOut ? 1.0 : -1.0);
186  Real phi = (type == Option::Call ? 1.0 : -1.0);
187 
188  Real x1, x2, y1, y2;
189  Real cum_x1, cum_x2, cum_y1, cum_y2;
190  if (variance>=QL_EPSILON) {
191 
192  // we calculate using mu*stddev instead of (mu+1)*stddev
193  // because cash-or-nothing don't need it. asset-or-nothing
194  // mu is really mu+1
195  x1 = phi*(log_S_X/stdDev + mu*stdDev);
196  x2 = phi*(log_S_H/stdDev + mu*stdDev);
197  y1 = eta*(log_H2_SX/stdDev + mu*stdDev);
198  y2 = eta*(log_H_S/stdDev + mu*stdDev);
199 
200  CumulativeNormalDistribution f;
201  cum_x1 = f(x1);
202  cum_x2 = f(x2);
203  cum_y1 = f(y1);
204  cum_y2 = f(y2);
205  } else {
206  if (log_S_X>0)
207  cum_x1= 1.0;
208  else
209  cum_x1= 0.0;
210  if (log_S_H>0)
211  cum_x2= 1.0;
212  else
213  cum_x2= 0.0;
214  if (log_H2_SX>0)
215  cum_y1= 1.0;
216  else
217  cum_y1= 0.0;
218  if (log_H_S>0)
219  cum_y2= 1.0;
220  else
221  cum_y2= 0.0;
222  }
223 
224  Real alpha = 0;
225 
226  switch (barrierType) {
227  case Barrier::DownIn:
228  if (type == Option::Call) {
229  // down-in and call
230  if (strike >= barrier) {
231  // B3 (eta=1, phi=1)
232  alpha = H_S_2mu * cum_y1;
233  } else {
234  // B1-B2+B4 (eta=1, phi=1)
235  alpha = cum_x1 - cum_x2 + H_S_2mu * cum_y2;
236  }
237  }
238  else {
239  // down-in and put
240  if (strike >= barrier) {
241  // B2-B3+B4 (eta=1, phi=-1)
242  alpha = cum_x2 + H_S_2mu*(-cum_y1 + cum_y2);
243  } else {
244  // B1 (eta=1, phi=-1)
245  alpha = cum_x1;
246  }
247  }
248  break;
249 
250  case Barrier::UpIn:
251  if (type == Option::Call) {
252  // up-in and call
253  if (strike >= barrier) {
254  // B1 (eta=-1, phi=1)
255  alpha = cum_x1;
256  } else {
257  // B2-B3+B4 (eta=-1, phi=1)
258  alpha = cum_x2 + H_S_2mu * (-cum_y1 + cum_y2);
259  }
260  }
261  else {
262  // up-in and put
263  if (strike >= barrier) {
264  // B1-B2+B4 (eta=-1, phi=-1)
265  alpha = cum_x1 - cum_x2 + H_S_2mu * cum_y2;
266  } else {
267  // B3 (eta=-1, phi=-1)
268  alpha = H_S_2mu * cum_y1;
269  }
270  }
271  break;
272 
273  case Barrier::DownOut:
274  if (type == Option::Call) {
275  // down-out and call
276  if (strike >= barrier) {
277  // B1-B3 (eta=1, phi=1)
278  alpha = cum_x1 - H_S_2mu * cum_y1;
279  } else {
280  // B2-B4 (eta=1, phi=1)
281  alpha = cum_x2 - H_S_2mu * cum_y2;
282  }
283  }
284  else {
285  // down-out and put
286  if (strike >= barrier) {
287  // B1-B2+B3-B4 (eta=1, phi=-1)
288  alpha = cum_x1 - cum_x2 + H_S_2mu * (cum_y1-cum_y2);
289  } else {
290  // always 0
291  alpha = 0;
292  }
293  }
294  break;
295  case Barrier::UpOut:
296  if (type == Option::Call) {
297  // up-out and call
298  if (strike >= barrier) {
299  // always 0
300  alpha = 0;
301  } else {
302  // B1-B2+B3-B4 (eta=-1, phi=1)
303  alpha = cum_x1 - cum_x2 + H_S_2mu * (cum_y1-cum_y2);
304  }
305  }
306  else {
307  // up-out and put
308  if (strike >= barrier) {
309  // B2-B4 (eta=-1, phi=-1)
310  alpha = cum_x2 - H_S_2mu * cum_y2;
311  } else {
312  // B1-B3 (eta=-1, phi=-1)
313  alpha = cum_x1 - H_S_2mu * cum_y1;
314  }
315  }
316  break;
317  default:
318  QL_FAIL("invalid barrier type");
319  }
320 
321  return discount * K * alpha;
322  }
