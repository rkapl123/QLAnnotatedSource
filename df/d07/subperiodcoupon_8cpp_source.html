30  SubPeriodsCoupon::SubPeriodsCoupon(const Date& paymentDate,
31  Real nominal,
32  const Date& startDate,
33  const Date& endDate,
34  Natural fixingDays,
35  const ext::shared_ptr<IborIndex>& index,
36  Real gearing,
37  Rate couponSpread,
38  Rate rateSpread,
39  const Date& refPeriodStart,
40  const Date& refPeriodEnd,
41  const DayCounter& dayCounter,
42  const Date& exCouponDate)
43  : FloatingRateCoupon(paymentDate, nominal, startDate, endDate,
44  fixingDays, index, gearing, couponSpread,
45  refPeriodStart, refPeriodEnd, dayCounter,
46  false, exCouponDate),
47  rateSpread_(rateSpread) {
48  Schedule sch = MakeSchedule()
49  .from(startDate)
50  .to(endDate)
51  .withTenor(index->tenor())
52  .withCalendar(index->fixingCalendar())
53  .withConvention(index->businessDayConvention())
54  .backwards()
55  .endOfMonth(index->endOfMonth());
56  valueDates_ = sch.dates();
57 
58  // fixing dates
59  n_ = valueDates_.size() - 1;
60  if (index->fixingDays() == 0) {
61  fixingDates_ = std::vector<Date>(valueDates_.begin(), valueDates_.end() - 1);
62  } else {
63  fixingDates_.resize(n_);
64  for (Size i = 0; i < n_; ++i)
65  fixingDates_[i] = index->fixingDate(valueDates_[i]);
66  }
67 
68  // accrual of sub-periods
69  dt_.resize(n_);
70  const DayCounter& dc = index->dayCounter();
71  for (Size i = 0; i < n_; ++i)
72  dt_[i] = dc.yearFraction(valueDates_[i], valueDates_[i + 1]);
73  }

83  void SubPeriodsPricer::initialize(const FloatingRateCoupon& coupon) {
84  coupon_ = dynamic_cast<const SubPeriodsCoupon*>(&coupon);
85  QL_REQUIRE(coupon_, "sub-periods coupon required");
86 
87  ext::shared_ptr<IborIndex> index = ext::dynamic_pointer_cast<IborIndex>(coupon_->index());
88  if (!index) {
89  // coupon was right, index is not
90  QL_FAIL("IborIndex required");
91  }
92 
93  QL_REQUIRE(coupon_->accrualPeriod() != 0.0, "null accrual period");
94 
95  const std::vector<Date>& fixingDates = coupon_->fixingDates();
96  Size n = fixingDates.size();
97  subPeriodFixings_.resize(n);
98 
99  for (Size i = 0; i < n; i++) {
100  subPeriodFixings_[i] = index->fixing(fixingDates[i]) + coupon_->rateSpread();
101  }
102  }

124  Real AveragingRatePricer::swapletRate() const {
125  // past or future fixing is managed in InterestRateIndex::fixing()
126 
127  Size nCount = subPeriodFixings_.size();
128  const std::vector<Time>& subPeriodFractions = coupon_->dt();
129  Real aggregateFactor = 0.0;
130  for (Size i = 0; i < nCount; i++) {
131  aggregateFactor += subPeriodFixings_[i] * subPeriodFractions[i];
132  }
133 
134  Real rate = aggregateFactor / coupon_->accrualPeriod();
135  return coupon_->gearing() * rate + coupon_->spread();
136  }

138  Real CompoundingRatePricer::swapletRate() const {
139  // past or future fixing is managed in InterestRateIndex::fixing()
140 
141  Real compoundFactor = 1.0;
142  const std::vector<Time>& subPeriodFractions = coupon_->dt();
143  Size nCount = subPeriodFixings_.size();
144  for (Size i = 0; i < nCount; i++) {
145  compoundFactor *= (1.0 + subPeriodFixings_[i] * subPeriodFractions[i]);
146  }
147 
148  Real rate = (compoundFactor - 1.0) / coupon_->accrualPeriod();
149  return coupon_->gearing() * rate + coupon_->spread();
150  }

244  SubPeriodsLeg::operator Leg() const {
245  Leg cashflows;
246  Calendar calendar = schedule_.calendar();
247  Date refStart, start, refEnd, end, exCouponDate;
248  Date paymentDate;
249 
250  Size n = schedule_.size() - 1;
251  QL_REQUIRE(!notionals_.empty(), "no notional given");
252  QL_REQUIRE(notionals_.size() <= n,
253  "too many nominals (" << notionals_.size() << "), only " << n << " required");
254  QL_REQUIRE(gearings_.size() <= n,
255  "too many gearings (" << gearings_.size() << "), only " << n << " required");
256  QL_REQUIRE(couponSpreads_.size() <= n,
257  "too many coupon spreads (" << couponSpreads_.size() << "), only " << n << " required");
258  QL_REQUIRE(rateSpreads_.size() <= n,
259  "too many rate spreads (" << rateSpreads_.size() << "), only " << n << " required");
260  QL_REQUIRE(fixingDays_.size() <= n,
261  "too many fixing days (" << fixingDays_.size() << "), only " << n << " required");
262 
263  for (Size i = 0; i < n; ++i) {
264  refStart = start = schedule_.date(i);
265  refEnd = end = schedule_.date(i + 1);
266  paymentDate = paymentCalendar_.advance(end, paymentLag_, Days, paymentAdjustment_);
267 
268  if (i == 0 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1))
269  refStart = calendar.adjust(end - schedule_.tenor(), paymentAdjustment_);
270  if (i == n - 1 && schedule_.hasIsRegular() && !schedule_.isRegular(i + 1))
271  refEnd = calendar.adjust(start + schedule_.tenor(), paymentAdjustment_);
272  if (exCouponPeriod_ != Period()) {
273  if (exCouponCalendar_.empty()) {
274  exCouponDate = calendar.advance(paymentDate, -exCouponPeriod_,
275  exCouponAdjustment_, exCouponEndOfMonth_);
276  } else {
277  exCouponDate = exCouponCalendar_.advance(
278  paymentDate, -exCouponPeriod_, exCouponAdjustment_, exCouponEndOfMonth_);
279  }
280  }
281  cashflows.push_back(ext::shared_ptr<CashFlow>(new SubPeriodsCoupon(
282  paymentDate, detail::get(notionals_, i, notionals_.back()), start, end,
283  detail::get(fixingDays_, i, index_->fixingDays()), index_,
284  detail::get(gearings_, i, 1.0), detail::get(couponSpreads_, i, 0.0),
285  detail::get(rateSpreads_, i, 0.0), refStart, refEnd, paymentDayCounter_,
286  exCouponDate)));
287  }
288 
289  switch (averagingMethod_) {
290  case RateAveraging::Simple:
291  setCouponPricer(cashflows,
292  ext::shared_ptr<FloatingRateCouponPricer>(new AveragingRatePricer));
293  break;
294  case RateAveraging::Compound:
295  setCouponPricer(cashflows, ext::shared_ptr<FloatingRateCouponPricer>(
296  new CompoundingRatePricer));
297  break;
298  default:
299  QL_FAIL("unknown compounding convention (" << Integer(averagingMethod_) << ")");
300  }
301  return cashflows;
302  }
303 }
