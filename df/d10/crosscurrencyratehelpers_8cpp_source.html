29  ext::shared_ptr<Swap>
30  CrossCurrencyBasisSwapRateHelper::buildCrossCurrencyLeg(const Date& evaluationDate,
31  const Period& tenor,
32  Natural fixingDays,
33  const Calendar& calendar,
34  BusinessDayConvention convention,
35  bool endOfMonth,
36  const ext::shared_ptr<IborIndex>& idx,
37  VanillaSwap::Type type,
38  Real notional,
39  Spread basis) {
40  bool isPayer = (type == VanillaSwap::Payer);
41  Date referenceDate = calendar.adjust(evaluationDate);
42  Date earliestDate = calendar.advance(referenceDate, fixingDays * Days, convention);
43  Date maturity = earliestDate + tenor;
44 
45  Schedule schedule = MakeSchedule()
46  .from(earliestDate)
47  .to(maturity)
48  .withTenor(idx->tenor())
49  .withCalendar(calendar)
50  .withConvention(convention)
51  .endOfMonth(endOfMonth)
52  .backwards();
53 
54  Leg leg = IborLeg(schedule, idx).withNotionals(notional).withSpreads(basis);
55  Date lastPaymentDate = leg.back()->date();
56  leg.push_back(ext::make_shared<SimpleCashFlow>(notional, lastPaymentDate));
57 
58  return ext::make_shared<Swap>(std::vector<Leg>(1, leg), std::vector<bool>(1, isPayer));
59  }

86  void CrossCurrencyBasisSwapRateHelper::initializeDates() {
87  baseCcyLeg_ = CrossCurrencyBasisSwapRateHelper::buildCrossCurrencyLeg(
88  evaluationDate_, tenor_, fixingDays_, calendar_, convention_, endOfMonth_, baseCcyIdx_,
89  VanillaSwap::Receiver);
90  quoteCcyLeg_ = CrossCurrencyBasisSwapRateHelper::buildCrossCurrencyLeg(
91  evaluationDate_, tenor_, fixingDays_, calendar_, convention_, endOfMonth_, quoteCcyIdx_,
92  VanillaSwap::Payer);
93 
94  earliestDate_ = std::min(baseCcyLeg_->startDate(), quoteCcyLeg_->startDate());
95  latestDate_ = std::max(baseCcyLeg_->maturityDate(), quoteCcyLeg_->maturityDate());
96 
97  ext::shared_ptr<PricingEngine> otherCcyEngine(
98  new DiscountingSwapEngine(termStructureHandle_));
99  ext::shared_ptr<PricingEngine> collateralCcyEngine(
100  new DiscountingSwapEngine(collateralHandle_));
101 
102  if (isFxBaseCurrencyCollateralCurrency_) {
103  baseCcyLeg_->setPricingEngine(collateralCcyEngine);
104  quoteCcyLeg_->setPricingEngine(otherCcyEngine);
105  } else {
106  baseCcyLeg_->setPricingEngine(otherCcyEngine);
107  quoteCcyLeg_->setPricingEngine(collateralCcyEngine);
108  }

111  Real CrossCurrencyBasisSwapRateHelper::impliedQuote() const {
112  QL_REQUIRE(termStructure_ != nullptr, "term structure not set");
113  QL_REQUIRE(!collateralHandle_.empty(), "collateral term structure not set");
114 
115  baseCcyLeg_->recalculate();
116  Real npvBaseCcy = baseCcyLeg_->NPV();
117 
118  quoteCcyLeg_->recalculate();
119  Real npvQuoteCcy = quoteCcyLeg_->NPV();
120 
121  const Spread basisPoint = 1.0e-4;
122  Real bps = isBasisOnFxBaseCurrencyLeg_ ? baseCcyLeg_->legBPS(0) : quoteCcyLeg_->legBPS(0);
123 
124  return -(npvQuoteCcy + npvBaseCcy) / bps * basisPoint;
125  }
