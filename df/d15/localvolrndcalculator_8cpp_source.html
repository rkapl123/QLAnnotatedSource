111  Real LocalVolRNDCalculator::pdf(Real x, Time t) const {
112  calculate();
113 
114  QL_REQUIRE(t > 0, "positive time expected");
115  QL_REQUIRE(t <= timeGrid_->back(),
116  "given time exceeds local vol time grid");
117 
118  const Time tMin = std::min(timeGrid_->at(1), 1.0/365);
119 
120  if (t <= tMin) {
121  const Volatility vol = localVol_->localVol(0.0, spot_->value());
122  const Volatility stdDev = vol * std::sqrt(t);
123  const Real xm = - 0.5 * stdDev * stdDev +
124  std::log(spot_->value() * qTS_->discount(t)/rTS_->discount(t));
125 
126  return GaussianDistribution(xm, stdDev)(x);
127  }
128  else if (t <= timeGrid_->at(1)) {
129  const Volatility vol = localVol_->localVol(0.0, spot_->value());
130  const Volatility stdDev = vol * std::sqrt(tMin);
131  const Real xm = - 0.5 * stdDev * stdDev +
132  std::log(spot_->value() * qTS_->discount(tMin)/rTS_->discount(tMin));
133 
134  const GaussianDistribution gaussianPDF(xm, stdDev);
135 
136  const Time deltaT = timeGrid_->at(1) - tMin;
137  return gaussianPDF(x)*(timeGrid_->at(1) - t)/deltaT
138  + probabilityInterpolation(0, x)*(t - tMin)/deltaT;
139  }
140  else {
141  const TimeGrid::const_iterator lb
142  = std::lower_bound(timeGrid_->begin(), timeGrid_->end(), t);
143  const TimeGrid::const_iterator llb = lb-1;
144 
145  const Size idx = std::distance(timeGrid_->begin(), lb)-1;
146 
147  const Time deltaT = *lb - *llb;
148  return probabilityInterpolation(idx-1, x)*(*lb - t)/deltaT
149  + probabilityInterpolation(idx, x)*(t - *llb)/deltaT;
150  }
151  }

153  Real LocalVolRNDCalculator::cdf(Real x, Time t) const {
154  calculate();
155 
156  // get the left side of the integral
157  const Time tc = timeGrid_->closestTime(t);
158  const Size idx = (tc > t) ? timeGrid_->index(tc)-1
159  : std::min(xm_.size()-1, timeGrid_->index(tc));
160 
161  Real xl = xm_[idx]->locations().front();
162  Real xr = xm_[idx]->locations().back();
163 
164  if (x < xl)
165  return 0.0;
166  else if (x > xr)
167  return 1.0;
168 
169  // left or right hand integral
170  if (x > 0.5*(xr+xl)) {
171  while (pdf(xr, t) > 0.01*localVolProbEps_) xr*=1.1;
172  return 1.0-GaussLobattoIntegral(maxIter_, 0.1*localVolProbEps_)(
173  boost::bind(&LocalVolRNDCalculator::pdf, this, _1, t), x, xr);
174  }
175  else {
176  while (pdf(xl, t) > 0.01*localVolProbEps_) xl*=0.9;
177 
178  return GaussLobattoIntegral(maxIter_, 0.1*localVolProbEps_)(
179  boost::bind(&LocalVolRNDCalculator::pdf, this, _1, t), xl, x);
180  }
181  }

183  Real LocalVolRNDCalculator::invcdf(Real p, Time t) const {
184  calculate();
185 
186  const Time closeGridTime(timeGrid_->closestTime(t));
187  if (closeGridTime == 0.0) {
188  return RiskNeutralDensityCalculator::InvCDFHelper(
189  this, std::log(spot_->value()), 0.1*localVolProbEps_, maxIter_)
190  .inverseCDF(p, t);
191  }
192  else {
193  Array xp(xGrid_);
194  const Size idx = timeGrid_->index(closeGridTime)-1;
195 
196  const Array x(xm_[idx]->locations().begin(),
197  xm_[idx]->locations().end());
198  std::transform(x.begin(), x.end(), pm_->row_begin(idx), xp.begin(),
199  std::multiplies<Real>());
200 
201  const Real xm = DiscreteSimpsonIntegral()(x, xp);
202  return RiskNeutralDensityCalculator::InvCDFHelper(
203  this, xm, 0.1*localVolProbEps_, maxIter_).inverseCDF(p, t);
204  }

227  void LocalVolRNDCalculator::performCalculations() const {
228  rescaleTimeSteps_.clear();
229 
230  const Time sT = timeGrid_->at(1);
231  Time t = std::min(sT, (gaussianStepSize_ > 0.0) ? gaussianStepSize_
232  : 0.5*sT);
233  const Volatility vol = localVol_->localVol(0.0, spot_->value());
234 
235  const Volatility stdDev = vol * std::sqrt(t);
236  Real xm = - 0.5 * stdDev * stdDev +
237  std::log(spot_->value() * qTS_->discount(t)/rTS_->discount(t));
238 
239  const Volatility stdDevOfFirstStep = vol * std::sqrt(sT);
240  const Real normInvEps = InverseCumulativeNormal()(1 - localVolProbEps_);
241  Real sLowerBound = xm - normInvEps * stdDevOfFirstStep;
242  Real sUpperBound = xm + normInvEps * stdDevOfFirstStep;
243 
244  boost::shared_ptr<Fdm1dMesher> mesher(
245  new Concentrating1dMesher(sLowerBound, sUpperBound, xGrid_,
246  std::make_pair(xm, x0Density_), true));
247 
248  Array p(mesher->size());
249  Array x(mesher->locations().begin(), mesher->locations().end());
250 
251  const GaussianDistribution gaussianPDF(xm, vol * std::sqrt(t));
252 
253  for (Size idx=0; idx < p.size(); ++idx) {
254  p[idx] = gaussianPDF(x[idx]);
255  }
256  p = rescalePDF(x, p);
257 
258  QL_REQUIRE(x.size() > 10, "x grid is too small. "
259  "Minimum size is greater than 10");
260 
261  const Size b = std::max(Size(1), Size(x.size()*0.04));
262 
263  boost::shared_ptr<DouglasScheme> evolver(
264  new DouglasScheme(0.5,
265  boost::make_shared<FdmLocalVolFwdOp>(
266  boost::make_shared<FdmMesherComposite>(mesher),
267  spot_, rTS_, qTS_, localVol_)));
268 
269  pFct_.resize(tGrid_);
270 
271  for (Size i=1; i <= tGrid_; ++i) {
272  const Time dt = timeGrid_->at(i) - t;
273 
274  // leaking probability mass?
275  const Real maxLeftValue =
276  std::max(std::fabs(*std::min_element(p.begin(), p.begin()+b)),
277  std::fabs(*std::max_element(p.begin(), p.begin()+b)));
278  const Real maxRightValue =
279  std::max(std::fabs(*std::min_element(p.end()-b, p.end())),
280  std::fabs(*std::max_element(p.end()-b, p.end())));
281 
282  if (std::max(maxLeftValue, maxRightValue) > localVolProbEps_) {
283  rescaleTimeSteps_.push_back(i);
284 
285  const Real oldLowerBound = sLowerBound;
286  const Real oldUpperBound = sUpperBound;
287 
288  xm = DiscreteSimpsonIntegral()(x, x*p);
289  Array vols(x.size());
290  for (Size j=0; j < vols.size(); ++j) {
291  vols[j] = localVol_->localVol(t + dt, std::exp(x[j]));
292  }
293 
294  const Real vm = DiscreteSimpsonIntegral()(x, vols)
295  /(x.back() - x.front());
296 
297  const Size scalingSteps
298  = std::max(3, Integer(0.01*timeGrid_->size()));
299 
300  const Real scalingFactor
301  = std::sqrt(timeGrid_->at(
302  std::min(timeGrid_->size()-1, i+scalingSteps)))
303  * vm;
304 
305  if (maxLeftValue > localVolProbEps_)
306  sLowerBound -= scalingFactor*xm;
307  if (maxRightValue > localVolProbEps_)
308  sUpperBound += scalingFactor*xm;
309 
310  mesher = boost::shared_ptr<Fdm1dMesher>(
311  new Concentrating1dMesher(sLowerBound, sUpperBound, xGrid_,
312  std::make_pair(xm, 0.1), false));
313 
314  const CubicNaturalSpline pSpline(x.begin(), x.end(), p.begin());
315  const Array xn(mesher->locations().begin(),
316  mesher->locations().end());
317  Array pn(xn.size(), 0.0);
318 
319  for (Size j=0; j < xn.size(); ++j) {
320  if (xn[j] >= oldLowerBound && xn[j] <= oldUpperBound)
321  pn[j] = pSpline(xn[j]);
322  }
323 
324  x = xn;
325  p = rescalePDF(xn, pn);
326 
327  evolver = boost::make_shared<DouglasScheme>(0.5,
328  boost::make_shared<FdmLocalVolFwdOp>(
329  boost::make_shared<FdmMesherComposite>(mesher),
330  spot_, rTS_, qTS_, localVol_));
331  }
332  evolver->setStep(dt);
333  t+=dt;
334 
335  if (dt > QL_EPSILON) {
336  evolver->step(p, t);
337  p = rescalePDF(x, p);
338  }
339 
340  xm_[i-1] = mesher;
341  std::copy(p.begin(), p.end(), pm_->row_begin(i-1));
342  pFct_[i-1] = boost::make_shared<CubicNaturalSpline>(
343  xm_[i-1]->locations().begin(),
344  xm_[i-1]->locations().end(),
345  pm_->row_begin(i-1));
346  }
347  }
