135  void MarkovFunctional::initialize() {
136 
137  QL_MFMESSAGE(modelOutputs_, "initializing");
138  modelOutputs_.dirty_ = true;
139 
140  modelOutputs_.settings_ = modelSettings_;
141 
142  GaussHermiteIntegration gaussHermite(
143  modelSettings_.gaussHermitePoints_);
144  normalIntegralX_ = gaussHermite.x();
145  normalIntegralW_ = gaussHermite.weights();
146  for (Size i = 0; i < normalIntegralX_.size(); i++) {
147  normalIntegralW_[i] *=
148  std::exp(-normalIntegralX_[i] * normalIntegralX_[i]) * M_1_SQRTPI;
149  normalIntegralX_[i] *= M_SQRT2;
150  }
151 
152  volsteptimesArray_ = Array(volstepdates_.size());
153 
154  updateTimes1();
155 
156  if (capletCalibrated_) {
157  for (auto capletExpirie : capletExpiries_) {
158  makeCapletCalibrationPoint(capletExpirie);
159  }
160  } else {
161  std::vector<Date>::const_iterator i;
162  std::vector<Period>::const_iterator j;
163  for (i = swaptionExpiries_.begin(), j = swaptionTenors_.begin();
164  i != swaptionExpiries_.end(); ++i, ++j) {
165  makeSwaptionCalibrationPoint(*i, *j);
166  }
167  }
168 
169  bool done;
170  numeraireDate_ = Date::minDate();
171  do {
172  Date numeraireKnown = numeraireDate_;
173  done = true;
174  for (auto i = calibrationPoints_.rbegin(); i != calibrationPoints_.rend() && done;
175  ++i) {
176  if (i->second.paymentDates_.back() > numeraireDate_) {
177  numeraireDate_ = i->second.paymentDates_.back();
178  numeraireKnown = i->second.paymentDates_.back();
179  if (i != calibrationPoints_.rbegin()) {
180  done = false;
181  }
182  }
183  // Inlining this into the loop condition causes
184  // a bogus compilation error wih g++ 4.0.1 on Mac OS X
185  std::vector<Date>::const_reverse_iterator rend =
186  i->second.paymentDates_.rend();
187  for (std::vector<Date>::const_reverse_iterator j =
188  i->second.paymentDates_.rbegin();
189  j != rend && done; ++j) {
190  if (*j < numeraireKnown) {
191  if (capletCalibrated_) {
192  makeCapletCalibrationPoint(*j);
193  done = false;
194  break;
195  } else {
196  UpRounding rounder(0);
197  makeSwaptionCalibrationPoint(
198  *j,
199  Period(
200  static_cast<Integer>(rounder(
201  (swapIndexBase_->dayCounter()
202  .yearFraction(*j, numeraireKnown) -
203  0.5 / 365) *
204  12.0)),
205  Months));
206  done = false;
207  break;
208  }
209  }
210  }
211  if (done) {
212  numeraireKnown = i->first;
213  }
214  }
215  } while (!done);
216 
217  updateTimes2();
218 
219  sigma_ =
220  PiecewiseConstantParameter(volsteptimes_, PositiveConstraint());
221  for (Size i = 0; i < sigma_.size(); i++) {
222  sigma_.setParam(i, volatilities_[i]);
223  }
224 
225  stateProcess_ = ext::make_shared<MfStateProcess>(
226  reversion_(0.0), volsteptimesArray_, sigma_.params());
227 
228  y_ = yGrid(modelSettings_.yStdDevs_, modelSettings_.yGridPoints_);
229 
230  discreteNumeraire_ = ext::make_shared<Matrix>(
231  times_.size(), 2 * modelSettings_.yGridPoints_ + 1, 1.0);
232  for (Size i = 0; i < times_.size(); i++) {
233  ext::shared_ptr<Interpolation> numInt(new CubicInterpolation(
234  y_.begin(), y_.end(), discreteNumeraire_->row_begin(i),
235  CubicInterpolation::Spline, true, CubicInterpolation::Lagrange,
236  0.0, CubicInterpolation::Lagrange, 0.0));
237  numInt->enableExtrapolation();
238  numeraire_.push_back(numInt);
239  }
240 
241  registerWith(termStructure());
242  if (!swaptionVol_.empty())
243  registerWith(swaptionVol_);
244  if (!capletVol_.empty())
245  registerWith(capletVol_);
246  }

248  void MarkovFunctional::makeSwaptionCalibrationPoint(const Date &expiry,
249  const Period &tenor) {
250 
251  QL_REQUIRE(calibrationPoints_.count(expiry) == 0,
252  "swaption expiry ("
253  << expiry
254  << ") occurs more than once in calibration set");
255 
256  CalibrationPoint p;
257  p.isCaplet_ = false;
258  p.tenor_ = tenor;
259 
260  ext::shared_ptr<VanillaSwap> underlying = underlyingSwap(swapIndexBase_, expiry, tenor);
261 
262  Schedule sched = underlying->fixedSchedule();
263  const Calendar& cal = sched.calendar();
264  BusinessDayConvention bdc = underlying->paymentConvention();
265 
266  for (unsigned int k = 1; k < sched.size(); k++) {
267  p.yearFractions_.push_back(
268  swapIndexBase_->dayCounter().yearFraction(
269  k == 1 ? expiry : sched.date(k - 1), sched.date(k)));
270  p.paymentDates_.push_back(cal.adjust(sched.date(k), bdc));
271  }
272  calibrationPoints_[expiry] = p;
273  }

275  void MarkovFunctional::makeCapletCalibrationPoint(const Date &expiry) {
276 
277  QL_REQUIRE(
278  calibrationPoints_.count(expiry) == 0,
279  "caplet expiry (" << expiry
280  << ") occurs more than once in calibration set");
281 
282  CalibrationPoint p;
283  p.isCaplet_ = true;
284  // p.expiry_ = expiry;
285  p.tenor_ = iborIndex_->tenor();
286  Date valueDate = iborIndex_->valueDate(expiry);
287  Date endDate = iborIndex_->fixingCalendar().advance(
288  valueDate, iborIndex_->tenor(), iborIndex_->businessDayConvention(),
289  iborIndex_->endOfMonth());
290  // FIXME Here we should use a calculation date calendar ?
291  p.paymentDates_.push_back(endDate);
292  p.yearFractions_.push_back(
293  iborIndex_->dayCounter().yearFraction(expiry, endDate));
294  // adjust the first period to start on expiry
295  calibrationPoints_[expiry] = p;
296  }

298  void MarkovFunctional::updateSmiles() const {
299 
300  QL_MFMESSAGE(modelOutputs_, "updating smiles");
301  modelOutputs_.dirty_ = true;
302 
303  arbitrageIndices_.clear();
304 
305  Size pointIndex = 0;
306 
307  for (auto i = calibrationPoints_.rbegin(); i != calibrationPoints_.rend(); ++i) {
308 
309  ext::shared_ptr<SmileSection> smileSection;
310  if (i->second.isCaplet_) {
311  i->second.annuity_ =
312  i->second.yearFractions_[0] *
313  termStructure()->discount(i->second.paymentDates_[0], true);
314  i->second.atm_ = (termStructure()->discount(i->first, true) -
315  termStructure()->discount(
316  i->second.paymentDates_[0], true)) /
317  i->second.annuity_;
318  smileSection = capletVol_->smileSection(i->first, true);
319  } else {
320  Real annuity = 0.0;
321  for (unsigned int k = 0; k < i->second.paymentDates_.size();
322  k++) {
323  annuity += i->second.yearFractions_[k] *
324  termStructure()->discount(
325  i->second.paymentDates_[k], true);
326  }
327  i->second.annuity_ = annuity;
328  i->second.atm_ = (termStructure()->discount(i->first, true) -
329  termStructure()->discount(
330  i->second.paymentDates_.back(), true)) /
331  annuity;
332  smileSection = swaptionVol_->smileSection(
333  i->first, i->second.tenor_, true);
334  }
335 
336  i->second.rawSmileSection_ = ext::shared_ptr<SmileSection>(
337  new AtmSmileSection(smileSection, i->second.atm_));
338 
339  int forcedLeftIndex = -1;
340  int forcedRightIndex = QL_MAX_INTEGER;
341  if(forcedArbitrageIndices_.size() > pointIndex) {
342  forcedLeftIndex = forcedArbitrageIndices_[pointIndex].first;
343  forcedRightIndex = forcedArbitrageIndices_[pointIndex].second;
344  }
345 
346  if ((modelSettings_.adjustments_ & ModelSettings::KahaleSmile) != 0) {
347 
348  i->second.smileSection_ = ext::make_shared<KahaleSmileSection>(
349 
350  i->second.rawSmileSection_, i->second.atm_,
351  (modelSettings_.adjustments_ &
352  ModelSettings::KahaleInterpolation) != 0,
353  (modelSettings_.adjustments_ &
354  ModelSettings::SmileExponentialExtrapolation) != 0,
355  (modelSettings_.adjustments_ &
356  ModelSettings::SmileDeleteArbitragePoints) != 0,
357  modelSettings_.smileMoneynessCheckpoints_,
358  modelSettings_.digitalGap_,
359  forcedLeftIndex, forcedRightIndex);
360 
361  arbitrageIndices_.push_back(
362  ext::dynamic_pointer_cast<KahaleSmileSection>(
363  i->second.smileSection_)->coreIndices());
364 
365  } else {
366 
367  if ((modelSettings_.adjustments_ & ModelSettings::SabrSmile) != 0) {
368 
369  SmileSectionUtils ssutils(
370  *i->second.rawSmileSection_,
371  modelSettings_.smileMoneynessCheckpoints_);
372  std::vector<Real> k = ssutils.strikeGrid();
373  k.erase(k.begin()); // the first strike is zero which we do
374  // not want in the sabr calibration
375  QL_REQUIRE(i->second.rawSmileSection_->volatilityType() ==
376  ShiftedLognormal,
377  "MarkovFunctional: SABR calibration to normal "
378  "input volatilities is not supported");
379  QL_REQUIRE(
380  k.size() >= 4,
381  "for sabr calibration at least 4 points are needed (is "
382  << k.size() << ")");
383  std::vector<Real> v;
384  v.reserve(k.size());
385  for (double j : k) {
386  v.push_back(i->second.rawSmileSection_->volatility(j));
387  }
388 
389  // TODO should we fix beta to avoid numerical instabilities
390  // during calibration ?
391  ext::shared_ptr<SabrInterpolatedSmileSection> sabrSection(
392  new SabrInterpolatedSmileSection(
393  i->first, i->second.atm_, k, false,
394  i->second.rawSmileSection_->volatility(
395  i->second.atm_),
396  v, 0.03, 0.80, 0.50, 0.00, false, false, false,
397  false, true, ext::shared_ptr<EndCriteria>(),
398  ext::shared_ptr<OptimizationMethod>(),
399  Actual365Fixed(),
400  i->second.rawSmileSection_->shift()));
401 
402  // we make the sabr section arbitrage free by superimposing
403  // a kahalesection
404 
405  i->second.smileSection_ = ext::make_shared<
406  KahaleSmileSection>(
407  sabrSection, i->second.atm_, false,
408  (modelSettings_.adjustments_ &
409  ModelSettings::SmileExponentialExtrapolation) != 0,
410  (modelSettings_.adjustments_ &
411  ModelSettings::SmileDeleteArbitragePoints) != 0,
412  modelSettings_.smileMoneynessCheckpoints_,
413  modelSettings_.digitalGap_,
414  forcedLeftIndex, forcedRightIndex);
415 
416  arbitrageIndices_.push_back(
417  ext::dynamic_pointer_cast<KahaleSmileSection>(
418  i->second.smileSection_)->coreIndices());
419 
420  } else if ((modelSettings_.adjustments_ & ModelSettings::CustomSmile) != 0) {
421 
422  // Custom smile section is af by assumption
423  i->second.smileSection_ =
424  modelSettings_.customSmileFactory_->smileSection(
425  i->second.rawSmileSection_, i->second.atm_);
426  arbitrageIndices_.emplace_back(Null<Size>(), Null<Size>());
427  } else { // no smile pretreatment
428 
429  i->second.smileSection_ = i->second.rawSmileSection_;
430  }
431  }
432 
433  // custom smile will take care of this itself
434  if ((modelSettings_.adjustments_ & ModelSettings::CustomSmile) == 0) {
435  i->second.minRateDigital_ =
436  i->second.smileSection_->digitalOptionPrice(
437  modelSettings_.lowerRateBound_ -
438  i->second.smileSection_->shift(),
439  Option::Call, i->second.annuity_,
440  modelSettings_.digitalGap_);
441  i->second.maxRateDigital_ =
442  i->second.smileSection_->digitalOptionPrice(
443  modelSettings_.upperRateBound_ -
444  i->second.smileSection_->shift(),
445  Option::Call, i->second.annuity_,
446  modelSettings_.digitalGap_);
447  }
448 
449  ++pointIndex;
450  }
451  }

453  void MarkovFunctional::updateNumeraireTabulation() const {
454 
455  QL_MFMESSAGE(modelOutputs_, "updating numeraire tabulation");
456  modelOutputs_.dirty_ = true;
457 
458  modelOutputs_.adjustmentFactors_.clear();
459  modelOutputs_.digitalsAdjustmentFactors_.clear();
460 
461  int idx = times_.size() - 2;
462 
463  for (auto i = calibrationPoints_.rbegin(); i != calibrationPoints_.rend(); ++i, --idx) {
464 
465  ext::shared_ptr<CustomSmileSection> mfSec;
466  if ((modelSettings_.adjustments_ & ModelSettings::CustomSmile) != 0) {
467  mfSec = ext::dynamic_pointer_cast<CustomSmileSection>(
468  i->second.smileSection_);
469  QL_REQUIRE(mfSec,
470  "no CustomSmileSection given, this is unexpected...");
471  }
472 
473  Array discreteDeflatedAnnuities(y_.size(), 0.0);
474  Array deflatedFinalPayments;
475 
476  Real numeraire0 = termStructure()->discount(numeraireTime_, true);
477  Real normalization =
478  termStructure()->discount(times_[idx], true) / numeraire0;
479 
480  for (unsigned int k = 0; k < i->second.paymentDates_.size(); k++) {
481  deflatedFinalPayments =
482  deflatedZerobondArray(termStructure()->timeFromReference(
483  i->second.paymentDates_[k]),
484  times_[idx], y_);
485  discreteDeflatedAnnuities +=
486  deflatedFinalPayments * i->second.yearFractions_[k];
487  }
488 
489  CubicInterpolation deflatedAnnuities(
490  y_.begin(), y_.end(), discreteDeflatedAnnuities.begin(),
491  CubicInterpolation::Spline, true, CubicInterpolation::Lagrange,
492  0.0, CubicInterpolation::Lagrange, 0.0);
493  deflatedAnnuities.enableExtrapolation();
494 
495  Real digitalsCorrectionFactor = 1.0;
496  modelOutputs_.digitalsAdjustmentFactors_.insert(
497  modelOutputs_.digitalsAdjustmentFactors_.begin(),
498  digitalsCorrectionFactor);
499 
500  Real digital = 0.0, swapRate, swapRate0;
501 
502  for (int c = 0;
503  c == 0 ||
504  (c == 1 && ((modelSettings_.adjustments_ & ModelSettings::AdjustDigitals) != 0));
505  c++) {
506 
507  if (c == 1) {
508  digitalsCorrectionFactor = i->second.annuity_ / digital;
509  modelOutputs_.digitalsAdjustmentFactors_.front() =
510  digitalsCorrectionFactor;
511  }
512 
513  digital = 0.0;
514  swapRate0 =
515  modelSettings_.upperRateBound_ / 2.0; // initial guess
516  for (int j = y_.size() - 1; j >= 0; j--) {
517 
518  Real integral = 0.0;
519 
520  if (j == (int)(y_.size() - 1)) {
521  if ((modelSettings_.adjustments_ &
522  ModelSettings::NoPayoffExtrapolation) == 0) {
523  if ((modelSettings_.adjustments_ &
524  ModelSettings::ExtrapolatePayoffFlat) != 0) {
525  integral = gaussianShiftedPolynomialIntegral(
526  0.0, 0.0, 0.0, 0.0,
527  discreteDeflatedAnnuities[j - 1], y_[j - 1],
528  y_[j], 100.0);
529  } else {
530  Real ca =
531  deflatedAnnuities.aCoefficients()[j - 1];
532  Real cb =
533  deflatedAnnuities.bCoefficients()[j - 1];
534  Real cc =
535  deflatedAnnuities.cCoefficients()[j - 1];
536  integral = gaussianShiftedPolynomialIntegral(
537  0.0, cc, cb, ca,
538  discreteDeflatedAnnuities[j - 1], y_[j - 1],
539  y_[j], 100.0);
540  }
541  }
542  } else {
543  Real ca = deflatedAnnuities.aCoefficients()[j];
544  Real cb = deflatedAnnuities.bCoefficients()[j];
545  Real cc = deflatedAnnuities.cCoefficients()[j];
546  integral = gaussianShiftedPolynomialIntegral(
547  0.0, cc, cb, ca, discreteDeflatedAnnuities[j],
548  y_[j], y_[j], y_[j + 1]);
549  }
550 
551  if (integral < 0) {
552  QL_MFMESSAGE(modelOutputs_,
553  "WARNING: integral for digitalPrice is "
554  "negative for j="
555  << j << " (" << integral
556  << ") --- reset it to zero.");
557  integral = 0.0;
558  }
559 
560  digital += integral * numeraire0 * digitalsCorrectionFactor;
561 
562  bool check = true;
563  if ((modelSettings_.adjustments_ & ModelSettings::CustomSmile) != 0) {
564  swapRate = mfSec->inverseDigitalCall(
565  digital, i->second.annuity_);
566  } else if (digital >= i->second.minRateDigital_) {
567  swapRate = modelSettings_.lowerRateBound_ -
568  i->second.rawSmileSection_->shift();
569  check = false;
570  } else if (digital <= i->second.maxRateDigital_) {
571  swapRate = modelSettings_.upperRateBound_;
572  check = false;
573  } else {
574  swapRate = marketSwapRate(
575  i->first, i->second, digital, swapRate0,
576  i->second.rawSmileSection_->shift());
577  }
578  if (check && j < (int)y_.size() - 1 &&
579  swapRate > swapRate0) {
580  QL_MFMESSAGE(
581  modelOutputs_,
582  "WARNING: swap rate is decreasing in y for "
583  "t=" << times_[idx]
584  << ", j=" << j << " (y, swap rate) is ("
585  << y_[j] << "," << swapRate << ") but for j="
586  << j + 1 << " it is (" << y_[j + 1] << ","
587  << swapRate0 << ") --- reset rate to "
588  << swapRate0 << " in node j=" << j);
589  swapRate = swapRate0;
590  }
591  swapRate0 = swapRate;
592  Real numeraire =
593  1.0 / std::max(swapRate * discreteDeflatedAnnuities[j] +
594  deflatedFinalPayments[j], 1E-6);
595  (*discreteNumeraire_)[idx][j] = numeraire * normalization;
596  }
597  }
598 
599  if ((modelSettings_.adjustments_ & ModelSettings::AdjustYts) != 0) {
600  numeraire_[idx]->update();
601  Real modelDeflatedZerobond = deflatedZerobond(times_[idx], 0.0);

611  modelDeflatedZerobond / marketDeflatedZerobond);

716  Disposable<Array> MarkovFunctional::numeraireArray(const Time t, const Array& y) const {
717 
718  calculate();
719  Array res(y.size(), termStructure()->discount(numeraireTime_, true));
720  if (t < QL_EPSILON)
721  return res;
722 
723  Real inverseNormalization =
724  termStructure()->discount(numeraireTime_, true) /
725  termStructure()->discount(t, true);
726 
727  Time tz = std::min(t, times_.back());
728  Size i = std::min<Size>(
729  std::upper_bound(times_.begin(), times_.end() - 1, t) -
730  times_.begin(),
731  times_.size() - 1);
732 
733  Real ta = times_[i - 1];
734  Real tb = times_[i];
735  Real dt = tb - ta;
736 
737  for (Size j = 0; j < y.size(); j++) {
738  Real yv = y[j];
739  if (yv < y_.front())
740  yv = y_.front();
741  // FIXME flat extrapolation should be incoperated into interpolation
742  // object, see above
743  if (yv > y_.back())
744  yv = y_.back();
745  Real na = (*numeraire_[i - 1])(yv);
746  Real nb = (*numeraire_[i])(yv);
747  res[j] =
748  inverseNormalization / ((tz - ta) / nb + (tb - tz) / na) * dt;
749  // linear in reciprocal of normalized numeraire
750  }
751 
752  return res;
753  }

767

769

772  Real stdDev_0_T = stateProcess_->stdDeviation(0.0, 0.0, T);

782  for (Size i = 0; i < modelSettings_.gaussHermitePoints_; i++) {
