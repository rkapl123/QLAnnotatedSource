69  Real SabrSmileSection::varianceImpl(Rate strike) const {
70  strike = std::max(0.00001 - shift(),strike);
71  Volatility vol = unsafeShiftedSabrVolatility(
72  strike, forward_, exerciseTime(), alpha_, beta_, nu_, rho_, shift_);
