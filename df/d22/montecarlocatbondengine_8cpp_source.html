37  QL_REQUIRE(!discountCurve_.empty(),
38  "discounting term structure handle is empty");

40  results_.valuationDate = (*discountCurve_)->referenceDate();
41 
42  bool includeRefDateFlows = includeSettlementDateFlows_ ? // NOLINT(readability-implicit-bool-conversion)
43  *includeSettlementDateFlows_ :
44  Settings::instance().includeReferenceDateEvents();
45 
46  Real lossProbability;
47  Real exhaustionProbability;
48  Real expectedLoss;
49 
50  results_.value = npv(includeRefDateFlows,
51  results_.valuationDate,
52  results_.valuationDate,
53  lossProbability,
54  exhaustionProbability,
55  expectedLoss);
56 
57  results_.lossProbability = lossProbability;
58  results_.exhaustionProbability = exhaustionProbability;
59  results_.expectedLoss = expectedLoss;
60 
61  // a bond's cashflow on settlement date is never taken into
62  // account, so we might have to play it safe and recalculate
63  if (!includeRefDateFlows
64  && results_.valuationDate == arguments_.settlementDate) {
65  // same parameters as above, we can avoid another call
66  results_.settlementValue = results_.value;
67  } else {
68  // no such luck
69  results_.settlementValue =
70  npv(includeRefDateFlows, arguments_.settlementDate, arguments_.settlementDate, lossProbability, exhaustionProbability, expectedLoss);
71  }

74  Real MonteCarloCatBondEngine::npv(bool includeSettlementDateFlows, Date settlementDate, Date npvDate, Real& lossProbability, Real &exhaustionProbability, Real& expectedLoss) const
75  {
76  const size_t MAX_PATHS = 10000; //TODO
77  lossProbability = 0.0;
78  exhaustionProbability = 0.0;
79  expectedLoss = 0.0;
80  if (arguments_.cashflows.empty())
81  return 0.0;
82 
83  if (settlementDate == Date())
84  settlementDate = Settings::instance().evaluationDate();
85 
86  if (npvDate == Date())
87  npvDate = settlementDate;
88 
89  Real totalNPV = 0.0;
90  Date effectiveDate = std::max(arguments_.startDate, settlementDate);
91  Date maturityDate = (*arguments_.cashflows.rbegin())->date();
92  ext::shared_ptr<CatSimulation> catSimulation = catRisk_->newSimulation(effectiveDate, maturityDate);
93  std::vector<std::pair<Date, Real> > eventsPath;
94  NotionalPath notionalPath;
95  Real riskFreeNPV = pathNpv(includeSettlementDateFlows, settlementDate, notionalPath);
96  size_t pathCount=0;
97  while(catSimulation->nextPath(eventsPath) && pathCount<MAX_PATHS)
98  {
99  arguments_.notionalRisk->updatePath(eventsPath, notionalPath);
100  if(notionalPath.loss()>0) { //optimization, most paths will not include any loss
101  totalNPV += pathNpv(includeSettlementDateFlows, settlementDate, notionalPath);
102  lossProbability+=1;
103  if (notionalPath.loss()==1)
104  exhaustionProbability+=1;
105  expectedLoss+=notionalPath.loss();
106  } else {
107  totalNPV += riskFreeNPV;
108  }
109  pathCount++;
110  }
111  lossProbability/=pathCount;
112  exhaustionProbability/=pathCount;
113  expectedLoss/=pathCount;
114  return totalNPV/(pathCount*discountCurve_->discount(npvDate));
115  }

117  Real MonteCarloCatBondEngine::pathNpv(bool includeSettlementDateFlows,
118  Date settlementDate,
119  const NotionalPath& notionalPath) const {
120  Real totalNPV = 0.0;
121  for (auto& cashflow : arguments_.cashflows) {
122  if (!cashflow->hasOccurred(settlementDate, includeSettlementDateFlows)) {
123  Real amount = cashFlowRiskyValue(cashflow, notionalPath);
124  totalNPV += amount * discountCurve_->discount(cashflow->date());
125  }
126  }
127  return totalNPV;
128  }
