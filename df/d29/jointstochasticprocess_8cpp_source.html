31  JointStochasticProcess::JointStochasticProcess(
32  const std::vector<ext::shared_ptr<StochasticProcess> > & l,
33  Size factors)
34  : l_ (l),
35  size_ (0),
36  factors_(factors),
37  modelFactors_(0) {
38 
39  for (const_iterator iter=l_.begin(); iter != l_.end(); ++iter) {
40  registerWith(*iter);
41  }
42 
43  vsize_.reserve (l_.size()+1);
44  vfactors_.reserve(l_.size()+1);
45 
46  for (const_iterator iter = l_.begin(); iter != l_.end(); ++iter) {
47  vsize_.push_back(size_);
48  size_ += (*iter)->size();
49 
50  vfactors_.push_back(modelFactors_);
51  modelFactors_ += (*iter)->factors();
52  }

54  vsize_.push_back(size_);

57  if (factors_ == Null<Size>()) {
58  factors_ = modelFactors_;
59  } else {
60  QL_REQUIRE(factors_ <= size_, "too many factors given");
61  }

135  Disposable<Matrix> JointStochasticProcess::covariance(Time t0,
136  const Array& x0,
137  Time dt) const {
138 
139  // get the model intrinsic covariance matrix
140  Matrix retVal(size(), size(), 0.0);
141 
142  for (Size j=0; j < l_.size(); ++j) {
143 
144  const Size vs = vsize_[j];
145  const Matrix& pCov = l_[j]->covariance(t0, slice(x0,j), dt);
146 
147  for (Size i=0; i < pCov.rows(); ++i) {
148  std::copy(pCov.row_begin(i), pCov.row_end(i),
149  retVal.row_begin(vs+i) + vs);
150  }
151  }
152 
153  // add the cross model covariance matrix
154  const Array& volatility = Sqrt(retVal.diagonal());
155  Matrix crossModelCovar = this->crossModelCorrelation(t0, x0);
156 
157  for (Size i=0; i < size(); ++i) {
158  for (Size j=0; j < size(); ++j) {
159  crossModelCovar[i][j] *= volatility[i]*volatility[j];
160  }
161  }
162 
163  retVal += crossModelCovar;
164 
165  return retVal;
166  }

190  Disposable<Array> JointStochasticProcess::evolve(
191  Time t0, const Array& x0, Time dt, const Array& dw) const {
192  Array dv(modelFactors_);
193 
194  if ( correlationIsStateDependent()
195  || correlationCache_.count(CachingKey(t0, dt)) == 0) {
196  Matrix cov = covariance(t0, x0, dt);
197 
198  const Array& sqrtDiag = Sqrt(cov.diagonal());
199  for (Size i=0; i < cov.rows(); ++i) {
200  for (Size j=i; j < cov.columns(); ++j) {
201  const Real div = sqrtDiag[i]*sqrtDiag[j];
202 
203  cov[i][j] = cov[j][i] = ( div > 0) ? cov[i][j]/div : 0.0;
204  }
205  }
206 
207  Matrix diff(size(), modelFactors_, 0.0);
208 
209  for (Size j = 0; j < l_.size(); ++j) {
210  const Size vs = vsize_ [j];
211  const Size vf = vfactors_[j];
212 
213  Matrix stdDev = l_[j]->stdDeviation(t0, slice(x0,j), dt);
214 
215  for (Size i=0; i < stdDev.rows(); ++i) {
216  const Volatility vol = std::sqrt(
217  std::inner_product(stdDev.row_begin(i),
218  stdDev.row_end(i),
219  stdDev.row_begin(i), 0.0));
220  if (vol > 0.0) {
221  std::transform(stdDev.row_begin(i), stdDev.row_end(i),
222  stdDev.row_begin(i),
223  divide_by<Real>(vol));
224  }
225  else {
226  // keep the svd happy
227  std::fill(stdDev.row_begin(i), stdDev.row_end(i),
228  100*i*QL_EPSILON);
229  }
230  }
231 
232  SVD svd(stdDev);
233  const Array& s = svd.singularValues();
234  Matrix w(s.size(), s.size(), 0.0);
235  for (Size i=0; i < s.size(); ++i) {
236  if (std::fabs(s[i]) > std::sqrt(QL_EPSILON)) {
237  w[i][i] = 1.0/s[i];
238  }
239  }
240 
241  const Matrix inv = svd.U() * w * transpose(svd.V());
242 
243  for (Size i=0; i < stdDev.rows(); ++i) {
244  std::copy(inv.row_begin(i), inv.row_end(i),
245  diff.row_begin(i+vs)+vf);
246  }
247  }
248 
249  Matrix rs = rankReducedSqrt(cov, factors_, 1.0,
250  SalvagingAlgorithm::Spectral);
251 
252  if (rs.columns() < factors_) {
253  // less eigenvalues than expected factors.
254  // fill the rest with zero's.
255  Matrix tmp = Matrix(cov.rows(), factors_, 0.0);
256  for (Size i=0; i < cov.rows(); ++i) {
257  std::copy(rs.row_begin(i), rs.row_end(i),
258  tmp.row_begin(i));
259  }
260  rs = tmp;
261  }
262 
263  const Matrix m = transpose(diff) * rs;
264 
265  if (!correlationIsStateDependent()) {
266  correlationCache_[CachingKey(t0,dt)] = m;
267  }
268  dv = m*dw;
269  }
270  else {
271  if (!correlationIsStateDependent()) {
272  dv = correlationCache_[CachingKey(t0,dt)] * dw;
273  }
274  }
275 
276  this->preEvolve(t0, x0, dt, dv);
277 
278 
279  Array retVal(size());
280  for (const_iterator iter = l_.begin(); iter != l_.end(); ++iter) {
281  const Size i = iter - l_.begin();
282 
283  Array dz((*iter)->factors());
284  std::copy(dv.begin()+vfactors_[i],
285  dv.begin()+vfactors_[i] + (*iter)->factors(),
286  dz.begin());
287  Array x((*iter)->size());
288  std::copy(x0.begin()+vsize_[i],
289  x0.begin()+vsize_[i] + (*iter)->size(),
290  x.begin());
291  const Array r = (*iter)->evolve(t0, x, dt, dz);
292  std::copy(r.begin(), r.end(), retVal.begin()+vsize_[i]);
293  }
294 
295  return this->postEvolve(t0, x0, dt, dv, retVal);
296  }
