31  GaussianQuadrature::GaussianQuadrature(
32  Size n,
33  const GaussianOrthogonalPolynomial& orthPoly)
34  : x_(n), w_(n) {
35 
36  // set-up matrix to compute the roots and the weights
37  Array e(n-1);
38 
39  Size i;
40  for (i=1; i < n; ++i) {
41  x_[i] = orthPoly.alpha(i);
42  e[i-1] = std::sqrt(orthPoly.beta(i));
43  }
44  x_[0] = orthPoly.alpha(0);
45 
46  TqrEigenDecomposition tqr(
47  x_, e,
48  TqrEigenDecomposition::OnlyFirstRowEigenVector,
49  TqrEigenDecomposition::Overrelaxation);
50 
51  x_ = tqr.eigenvalues();
52  const Matrix& ev = tqr.eigenvectors();
53 
54  Real mu_0 = orthPoly.mu_0();
55  for (i=0; i<n; ++i) {
56  w_[i] = mu_0*ev[0][i]*ev[0][i] / orthPoly.w(x_[i]);
57  }
58  }

61  void TabulatedGaussLegendre::order(Size order) {
62  switch(order) {
63  case(6):
64  order_=order; x_=x6; w_=w6; n_=n6;
65  break;
66  case(7):
67  order_=order; x_=x7; w_=w7; n_=n7;
68  break;
69  case(12):
70  order_=order; x_=x12; w_=w12; n_=n12;
71  break;
72  case(20):
73  order_=order; x_=x20; w_=w20; n_=n20;
74  break;
75  default:
76  QL_FAIL("order " << order << " not supported");
77  }
78  }
