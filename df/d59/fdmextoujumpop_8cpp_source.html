66  FdmExtOUJumpOp::FdmExtOUJumpOp(
67  const ext::shared_ptr<FdmMesher>& mesher,
68  const ext::shared_ptr<ExtOUWithJumpsProcess>& process,
69  const ext::shared_ptr<YieldTermStructure>& rTS,
70  const FdmBoundaryConditionSet& bcSet,
71  Size integroIntegrationOrder)
72  : mesher_ (mesher),
73  process_(process),
74  rTS_ (rTS),
75  bcSet_ (bcSet),
76  gaussLaguerreIntegration_(integroIntegrationOrder),
77  x_ (mesher->locations(0)),
78  ouOp_ (new FdmExtendedOrnsteinUhlenbeckOp(
79  mesher,
80  process->getExtendedOrnsteinUhlenbeckProcess(), rTS, bcSet)),
81  dyMap_ (FirstDerivativeOp(1, mesher)
82  .mult(-process->beta()*mesher->locations(1)))
83  {
84 #if !defined(QL_NO_UBLAS_SUPPORT)
85  const Real eta = process_->eta();
86  const Real lambda = process_->jumpIntensity();
87 
88  const Array yInt = gaussLaguerreIntegration_.x();
89  const Array weights= gaussLaguerreIntegration_.weights();
90 
91  integroPart_ = SparseMatrix(mesher_->layout()->size(),
92  mesher_->layout()->size());
93 
94  const ext::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
95  const FdmLinearOpIterator endIter = layout->end();
96 
97  Array yLoc(mesher_->layout()->dim()[1]);
98  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
99  ++iter) {
100  yLoc[iter.coordinates()[1]] = mesher_->location(iter, 1);
101  }
102 
103  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
104  ++iter) {
105 
106  const Size diag = iter.index();
107  integroPart_(diag, diag) -= lambda;
108 
109  const Real y = mesher_->location(iter, 1);
110  const Integer yIndex = iter.coordinates()[1];
111 
112  for (Size i=0; i < yInt.size(); ++i) {
113  const Real weight = std::exp(-yInt[i])*weights[i];
114 
115  const Real ys = y + yInt[i]/eta;
116  const Integer l = (ys > yLoc.back()) ? yLoc.size()-2
117  : std::upper_bound(yLoc.begin(),
118  yLoc.end()-1, ys) - yLoc.begin()-1;
119 
120  const Real s = (ys-yLoc[l])/(yLoc[l+1]-yLoc[l]);
121  integroPart_(diag, layout->neighbourhood(iter, 1, l-yIndex))
122  += weight*lambda*(1-s);
123  integroPart_(diag, layout->neighbourhood(iter, 1, l+1-yIndex))
124  += weight*lambda*s;
125  }
126  }
127 #endif
128  }

188  Real FdmExtOUJumpOp::IntegroIntegrand::operator()(Real u) const {
189  const Real y = y_ + u/eta_;
190  Real valueOfDerivative = (*interpl_)(y, true);
191 
192  for (FdmBoundaryConditionSet::const_iterator iter=bcSet_.begin();
193  iter < bcSet_.end(); ++iter) {
194  const ext::shared_ptr<FdmDirichletBoundary> dirichletBC =
195  ext::dynamic_pointer_cast<FdmDirichletBoundary>(*iter);
196 
197  if (dirichletBC != 0) {
198  valueOfDerivative=
199  dirichletBC->applyAfterApplying(y, valueOfDerivative);
200  }
201  }
202 
203  return std::exp(-u)*valueOfDerivative;
204  }

206  Disposable<Array> FdmExtOUJumpOp::integro(const Array& r) const {
207  Array integral(r.size());
208  const ext::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
209  const Size extraDims=layout->size()/(layout->dim()[0]*layout->dim()[1]);
210 
211  std::vector<Array> y(extraDims, Array(layout->dim()[1]));
212  std::vector<Matrix> f(extraDims,
213  Matrix(layout->dim()[1], layout->dim()[0]));
214 
215  const FdmLinearOpIterator endIter = layout->end();
216  for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
217  ++iter) {
218  const Size i = iter.coordinates()[0];
219  const Size j = iter.coordinates()[1];
220  const Size k = iter.index() / (layout->dim()[0]*layout->dim()[1]);
221 
222  y[k][j] = mesher_->location(iter, 1);
223  f[k][j][i] = r[iter.index()];
224  }
225  std::vector<std::vector<ext::shared_ptr<LinearInterpolation> > >
226  interpl(extraDims, std::vector<
227  ext::shared_ptr<LinearInterpolation> >(f[0].columns()));
228 
229  for (Size k=0; k < extraDims; ++k) {
230  for (Size i=0; i < f[k].columns(); ++i) {
231  interpl[k][i] = ext::shared_ptr<LinearInterpolation>(
232  new LinearInterpolation(y[k].begin(), y[k].end(),
233  f[k].column_begin(i)));
234  }
235  }
236 
237  const Real eta = process_->eta();
238 
239  for (FdmLinearOpIterator iter=layout->begin(); iter!=endIter; ++iter) {
240  const Size i = iter.coordinates()[0];
241  const Size j = iter.coordinates()[1];
242  const Size k = iter.index() / (layout->dim()[0]*layout->dim()[1]);
243 
244  integral[iter.index()] = gaussLaguerreIntegration_(
245  IntegroIntegrand(interpl[k][i], bcSet_, y[k][j], eta));
246  }
247 
248  return process_->jumpIntensity()*(integral-r);
249  }
