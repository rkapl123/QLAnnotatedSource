48 FdmExtOUJumpOp::FdmExtOUJumpOp(
49 const ext::shared_ptr<FdmMesher>& mesher,
50 const ext::shared_ptr<ExtOUWithJumpsProcess>& process,
51 const ext::shared_ptr<YieldTermStructure>& rTS,
52 const FdmBoundaryConditionSet& bcSet,
53 Size integroIntegrationOrder)
54 : mesher_ (mesher),
55 process_(process),
56 rTS_ (rTS),
57 bcSet_ (bcSet),
58 gaussLaguerreIntegration_(integroIntegrationOrder),
59 x_ (mesher->locations(0)),
60 ouOp_ (new FdmExtendedOrnsteinUhlenbeckOp(
61 mesher,
62 process->getExtendedOrnsteinUhlenbeckProcess(), rTS, bcSet)),
63 dyMap_ (FirstDerivativeOp(1, mesher)
64 .mult(-process->beta()*mesher->locations(1)))
65 {
66 const Real eta = process_->eta();
67 const Real lambda = process_->jumpIntensity();
68
69 const Array yInt = gaussLaguerreIntegration_.x();
70 const Array weights= gaussLaguerreIntegration_.weights();
71
72 integroPart_ = SparseMatrix(mesher_->layout()->size(),
73 mesher_->layout()->size());
74
75 const ext::shared_ptr<FdmLinearOpLayout> layout = mesher_->layout();
76 const FdmLinearOpIterator endIter = layout->end();
77
78 Array yLoc(mesher_->layout()->dim()[1]);
79 for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
80 ++iter) {
81 yLoc[iter.coordinates()[1]] = mesher_->location(iter, 1);
82 }
83
84 for (FdmLinearOpIterator iter = layout->begin(); iter != endIter;
85 ++iter) {
86
87 const Size diag = iter.index();
88 integroPart_(diag, diag) -= lambda;
89
90 const Real y = mesher_->location(iter, 1);
91 const Integer yIndex = iter.coordinates()[1];
92
93 for (Size i=0; i < yInt.size(); ++i) {
94 const Real weight = std::exp(-yInt[i])*weights[i];
95
96 const Real ys = y + yInt[i]/eta;
97 const Integer l = (ys > yLoc.back()) ? yLoc.size()-2
98 : std::upper_bound(yLoc.begin(),
99 yLoc.end()-1, ys) - yLoc.begin()-1;
100
101 const Real s = (ys-yLoc[l])/(yLoc[l+1]-yLoc[l]);
102 integroPart_(diag, layout->neighbourhood(iter, 1, l-yIndex))
103 += weight*lambda*(1-s);
104 integroPart_(diag, layout->neighbourhood(iter, 1, l+1-yIndex))
105 += weight*lambda*s;
106 }
107 }
108 }
