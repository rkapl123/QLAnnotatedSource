41  HaganIrregularSwaptionEngine::Basket::Basket(
42  ext::shared_ptr<IrregularSwap> swap,
43  Handle<YieldTermStructure> termStructure,
44  Handle<SwaptionVolatilityStructure> volatilityStructure)
45  : swap_(std::move(swap)), termStructure_(std::move(termStructure)),
46  volatilityStructure_(std::move(volatilityStructure)), targetNPV_(0.0), lambda_(0.0) {
47 
48  engine_ = ext::shared_ptr<PricingEngine>(new DiscountingSwapEngine(termStructure_));
49 
50  // store swap npv
51  swap_->setPricingEngine(engine_);
52  targetNPV_ = swap_->NPV();
53 
54  // build standard swaps
55 
56  const Leg& fixedLeg = swap_->fixedLeg();
57  const Leg& floatLeg = swap_->floatingLeg();
58 
59  Leg fixedCFS, floatCFS;
60 
61  for (Size i = 0; i < fixedLeg.size(); ++i) {
62  // retrieve fixed rate coupon from fixed leg
63  ext::shared_ptr<FixedRateCoupon> coupon =
64  ext::dynamic_pointer_cast<FixedRateCoupon>(fixedLeg[i]);
65  QL_REQUIRE(coupon, "dynamic cast of fixed leg coupon failed.");
66 
67  expiries_.push_back(coupon->date());
68 
69  ext::shared_ptr<FixedRateCoupon> newCpn = ext::make_shared<FixedRateCoupon>(
70  coupon->date(), 1.0, coupon->rate(), coupon->dayCounter(),
71  coupon->accrualStartDate(), coupon->accrualEndDate(),
72  coupon->referencePeriodStart(), coupon->referencePeriodEnd());
73 
74  fixedCFS.push_back(newCpn);
75 
76  annuities_.push_back(10000 * CashFlows::bps(fixedCFS, **termStructure_, true));
77 
78  floatCFS.clear();
79 
80  for (const auto& j : floatLeg) {
81  // retrieve ibor coupon from floating leg
82  ext::shared_ptr<IborCoupon> coupon = ext::dynamic_pointer_cast<IborCoupon>(j);
83  QL_REQUIRE(coupon, "dynamic cast of float leg coupon failed.");
84 
85  if (coupon->date() <= expiries_[i]) {
86  ext::shared_ptr<IborCoupon> newCpn = ext::shared_ptr<IborCoupon>(new IborCoupon(
87  coupon->date(), 1.0, coupon->accrualStartDate(), coupon->accrualEndDate(),
88  coupon->fixingDays(), coupon->iborIndex(), 1.0, coupon->spread(),
89  coupon->referencePeriodStart(), coupon->referencePeriodEnd(),
90  coupon->dayCounter(), coupon->isInArrears()));
91 
92 
93  if (!newCpn->isInArrears())
94  newCpn->setPricer(
95  ext::shared_ptr<FloatingRateCouponPricer>(new BlackIborCouponPricer()));
96 
97  floatCFS.push_back(newCpn);
98  }
99  }
100 
101  Real floatLegNPV = CashFlows::npv(floatCFS, **termStructure_, true);
102 
103  fairRates_.push_back(floatLegNPV / annuities_[i]);
104  }
105  }

108  //computes a replication of the swap in terms of a basket of vanilla swaps
109  //by solving a linear system of equation
110  Disposable<Array> HaganIrregularSwaptionEngine::Basket::compute(Rate lambda) const {
111 
112  //update members
113  lambda_ = lambda;
114 
115  Size n = swap_->fixedLeg().size();
116 
117  //build linear system of equations
118  Matrix arr(n,n,0.0);
119  Array rhs(n);
120 
121 
122  //fill the matrix describing the linear system of equations by looping over rows
123  for(Size r = 0; r < n; ++r)
124  {
125 
126  ext::shared_ptr<FixedRateCoupon> cpn_r = ext::dynamic_pointer_cast<FixedRateCoupon>(swap_->fixedLeg()[r]);
127  QL_REQUIRE(cpn_r,"Cast to fixed rate coupon failed.");
128 
129  //looping over columns
130  for(Size c = r; c < n; ++c){
131 
132  //set homogenous part of lse
133  arr[r][c] = ( fairRates_[c] + lambda_ ) * cpn_r->accrualPeriod();
134  }
135 
136  // add nominal repayment for i-th swap
137  arr[r][r] += 1;
138  }
139 
140 
141  for(Size r = 0; r < n; ++r)
142  {
143  ext::shared_ptr<FixedRateCoupon> cpn_r = ext::dynamic_pointer_cast<FixedRateCoupon>(swap_->fixedLeg()[r]);
144 
145  // set inhomogenity of lse
146  Real N_r = cpn_r->nominal();
147 
148  if(r < n - 1){
149 
150  ext::shared_ptr<FixedRateCoupon> cpn_rp1 = ext::dynamic_pointer_cast<FixedRateCoupon>(swap_->fixedLeg()[r+1]);
151 
152  Real N_rp1 = cpn_rp1->nominal();
153 
154  rhs[r] = N_r * (cpn_r->rate()) * cpn_r->accrualPeriod() + (N_r - N_rp1);
155 
156  } else {
157 
158  rhs[r] = N_r * (cpn_r->rate()) * cpn_r->accrualPeriod() + N_r;
159 
160  }
161 
162  }
163 
164 
165  SVD svd(arr);
166 
167  Disposable<Array> weights = svd.solveFor(rhs);
168 
169  return weights;
170 
171  }

175  Real HaganIrregularSwaptionEngine::Basket::operator()(Rate lambda) const {
176 
177  Disposable<Array> weights = compute(lambda);
178 
179  Real defect = -targetNPV_;
180 
181  for(Size i=0; i< weights.size();++i) defect -= swap_->type()*lambda*weights[i]*annuities_[i];
182 
183  return defect;
184  }

189  //creates a standard swap by deducing its conventions from market data objects
190  ext::shared_ptr<VanillaSwap> HaganIrregularSwaptionEngine::Basket::component(Size i) const {
191 
192  ext::shared_ptr<IborCoupon> iborCpn = ext::dynamic_pointer_cast<IborCoupon>(swap_->floatingLeg()[0]);
193  QL_REQUIRE(iborCpn,"dynamic cast of float leg coupon failed. Can't find index.");
194  ext::shared_ptr<IborIndex> iborIndex = iborCpn->iborIndex();
195 
196 
197  Period dummySwapLength = Period(1,Years);
198 
199  ext::shared_ptr<VanillaSwap> memberSwap_ = MakeVanillaSwap(dummySwapLength,iborIndex)
200  .withType(VanillaSwap::Type(swap_->type()))
201  .withEffectiveDate(swap_->startDate())
202  .withTerminationDate(expiries_[i])
203  .withRule(DateGeneration::Backward)
204  .withDiscountingTermStructure(termStructure_);
205 
206  Real stdAnnuity = 10000*CashFlows::bps(memberSwap_->fixedLeg(),**termStructure_,true);
207 
208  //compute annuity transformed rate
209  Rate transformedRate = (fairRates_[i]+lambda_)*annuities_[i]/stdAnnuity;
210 
211  memberSwap_ = MakeVanillaSwap(dummySwapLength,iborIndex,transformedRate)
212  .withType(VanillaSwap::Type(swap_->type()))
213  .withEffectiveDate(swap_->startDate())
214  .withTerminationDate(expiries_[i])
215  .withRule(DateGeneration::Backward)
216  .withDiscountingTermStructure(termStructure_);
217 
218 
219  return memberSwap_;
220 
221  }

239  void HaganIrregularSwaptionEngine::calculate() const {
240 
241  //check exercise type
242  ext::shared_ptr<Exercise> exercise_ = this->arguments_.exercise;
243  QL_REQUIRE(exercise_->type() == QuantLib::Exercise::European,"swaption must be european");
244 
245  //extract the underlying irregular swap
246  ext::shared_ptr<IrregularSwap> swap_ = this->arguments_.swap;
247 
248 
249  //Reshuffle spread from float to fixed (, i.e. remove spread from float side by finding the adjusted fixed coupon
250  //such that the NPV of the swap stays constant).
251  Leg fixedLeg = swap_->fixedLeg();
252  Real fxdLgBPS = CashFlows::bps(fixedLeg,**termStructure_,true);
253 
254  Leg floatLeg = swap_->floatingLeg();
255  Real fltLgNPV = CashFlows::npv(floatLeg,**termStructure_,true);
256  Real fltLgBPS = CashFlows::bps(floatLeg,**termStructure_,true);
257 
258 
259  Leg floatCFS,fixedCFS;
260 
261  floatCFS.clear();
262 
263  for (auto& j : floatLeg) {
264  //retrieve ibor coupon from floating leg
265  ext::shared_ptr<IborCoupon> coupon = ext::dynamic_pointer_cast<IborCoupon>(j);
266  QL_REQUIRE(coupon,"dynamic cast of float leg coupon failed.");
267 
268  ext::shared_ptr<IborCoupon> newCpn = ext::shared_ptr<IborCoupon> (
269  new IborCoupon(coupon->date(),
270  coupon->nominal(),
271  coupon->accrualStartDate(),
272  coupon->accrualEndDate(),
273  coupon->fixingDays(),
274  coupon->iborIndex(),
275  coupon->gearing(),
276  0.0,
277  coupon->referencePeriodStart(),
278  coupon->referencePeriodEnd(),
279  coupon->dayCounter(),
280  coupon->isInArrears()));
281 
282 
283  if (!newCpn->isInArrears())
284  newCpn->setPricer(
285  ext::shared_ptr<FloatingRateCouponPricer>(
286  new BlackIborCouponPricer()));
287 
288  floatCFS.push_back(newCpn);
289  }
290 
291 
292  Real sprdLgNPV = fltLgNPV - CashFlows::npv(floatCFS,**termStructure_,true);
293  Rate avgSpread = sprdLgNPV/fltLgBPS/10000;
294 
295  Rate cpn_adjustment = avgSpread*fltLgBPS/fxdLgBPS;
296 
297  fixedCFS.clear();
298 
299  for (auto& i : fixedLeg) {
300  //retrieve fixed rate coupon from fixed leg
301  ext::shared_ptr<FixedRateCoupon> coupon = ext::dynamic_pointer_cast<FixedRateCoupon>(i);
302  QL_REQUIRE(coupon,"dynamic cast of fixed leg coupon failed.");
303 
304  ext::shared_ptr<FixedRateCoupon> newCpn = ext::make_shared<FixedRateCoupon> (
305  coupon->date(),
306  coupon->nominal(),
307  coupon->rate()-cpn_adjustment,
308  coupon->dayCounter(),
309  coupon->accrualStartDate(),
310  coupon->accrualEndDate(),
311  coupon->referencePeriodStart(),
312  coupon->referencePeriodEnd());
313 
314  fixedCFS.push_back(newCpn);
315  }
316 
317 
318  //this is the irregular swap with spread removed
319  swap_ = ext::make_shared<IrregularSwap>(arguments_.swap->type(),fixedCFS,floatCFS);
320 
321 
322 
323  //Sets up the basket by implementing the methodology described in
324  //P.S.Hagan "Callable Swaps and Bermudan 'Exercise into Swaptions'"
325  Basket basket(swap_,termStructure_,volatilityStructure_);
326 
327 
329  //find lambda //
331 
332  Bisection s1d;
333 
334  Rate minLambda = -0.5;
335  Rate maxLambda = 0.5;
336  s1d.setMaxEvaluations(10000);
337  s1d.setLowerBound(minLambda);
338  s1d.setUpperBound(maxLambda);
339  s1d.solve(basket,1.0e-8,0.01, minLambda, maxLambda);
340 
341 
343  // compute the price of the irreg swaption as the sum of the prices of the regular swaptions //
345 
346 
347  results_.value = HKPrice(basket,exercise_);
348  }

353  // Computes irregular swaption price according to P.J. Hunt, J.E. Kennedy: //
354  // "Implied interest rate pricing models", Finance Stochast. 2, 275293 (1998) //
356 
357  Real HaganIrregularSwaptionEngine::HKPrice(Basket& basket,ext::shared_ptr<Exercise>& exercise) const {
358 
359  // Black 76 Swaption Engine: assumes that the swaptions exercise date equals the swap start date
360  ext::shared_ptr<PricingEngine> blackSwaptionEngine =
361  ext::shared_ptr<PricingEngine>(new BlackSwaptionEngine(termStructure_,volatilityStructure_));
362 
363  //retrieve weights of underlying swaps
364  Disposable<Array> weights = basket.weights();
365 
366  Real npv = 0.0;
367 
368  for(Size i=0; i<weights.size(); ++i)
369  {
370  ext::shared_ptr<VanillaSwap> pvSwap_ = basket.component(i);
371  Swaption swaption = Swaption(pvSwap_,exercise);
372  swaption.setPricingEngine(blackSwaptionEngine);
373  npv += weights[i]*swaption.NPV();
374  }
375 
376  return npv;
377 
378  }
