32  EventSetSimulation::EventSetSimulation(
33  ext::shared_ptr<std::vector<std::pair<Date, Real> > > events,
34  Date eventsStart,
35  Date eventsEnd,
36  Date start,
37  Date end)
38  : CatSimulation(start, end), events_(std::move(events)), eventsStart_(eventsStart),
39  eventsEnd_(eventsEnd), i_(0) {
40  years_ = end_.year()-start_.year();
41  if(eventsStart_.month()<start_.month()
42  || (eventsStart_.month()==start_.month()
43  && eventsStart_.dayOfMonth()<=start_.dayOfMonth())) {
44  periodStart_ = Date(start_.dayOfMonth(), start_.month(), eventsStart_.year());
45  } else {
46  periodStart_ = Date(start_.dayOfMonth(), start_.month(), eventsStart_.year()+1);
47  }
48  periodEnd_ = Date(end_.dayOfMonth(), end_.month(), periodStart_.year()+years_);
49  while(i_<events_->size() && (*events_)[i_].first<periodStart_) ++i_; //i points to the first element after the start of the relevant period.
50  }

52  bool EventSetSimulation::nextPath(std::vector< std::pair< Date, Real > >& path) {
53  path.resize(0);
54  if(periodEnd_>eventsEnd_) //Ran out of event data
55  return false;
56 
57  while(i_<events_->size() && (*events_)[i_].first<periodStart_) {
58  ++i_; //skip the elements between the previous period and this period
59  }
60  while(i_<events_->size() && (*events_)[i_].first<=periodEnd_){
61  std::pair<Date, Real> e(events_->at(i_).first+(start_.year() - periodStart_.year())*Years, events_->at(i_).second);
62  path.push_back(e);
63  ++i_; //i points to the first element after the start of the relevant period.
64  }
65  if(start_+years_*Years<end_) {
66  periodStart_+=(years_+1)*Years;
67  periodEnd_+=(years_+1)*Years;
68  } else {
69  periodStart_+=years_*Years;
70  periodEnd_+=years_*Years;
71  }
72  return true;
73  }

103  bool BetaRiskSimulation::nextPath(std::vector<std::pair<Date, Real> > &path)
104  {
105  path.resize(0);
106  Real eventFraction = exponential_();
107  while(eventFraction<=yearFraction_)
108  {
109  Integer days = round(eventFraction*dayCount_/yearFraction_);
110  Date eventDate = start_ + days*Days;
111  if(eventDate<=end_)
112  {
113  path.push_back(std::pair<Date, Real> (eventDate, generateBeta()));
114  }
115  else break;
116  eventFraction = exponential_();
117  }
118  return true;
119  }

121  BetaRisk::BetaRisk(Real maxLoss,
122  Real years,
123  Real mean,
124  Real stdDev)
125  : maxLoss_(maxLoss), lambda_(1.0/years) {
126  QL_REQUIRE(mean<maxLoss, "Mean "<<mean<<"of the loss distribution must be less than the maximum loss "<<maxLoss);
127  Real normalizedMean = mean/maxLoss;
128  Real normalizedVar = stdDev*stdDev/(maxLoss*maxLoss);
129  QL_REQUIRE(normalizedVar<normalizedMean*(1.0-normalizedMean), "Standard deviation of "<<stdDev<<" is impossible to achieve in gamma distribution with mean "<<mean);
130  Real nu = normalizedMean*(1.0-normalizedMean)/normalizedVar - 1.0;
131  alpha_=normalizedMean*nu;
132  beta_=(1.0-normalizedMean)*nu;
133  }
