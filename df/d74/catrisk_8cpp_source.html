31  EventSetSimulation::EventSetSimulation(
32  const ext::shared_ptr<std::vector<std::pair<Date, Real> > >& events,
33  Date eventsStart,
34  Date eventsEnd,
35  Date start,
36  Date end)
37  : CatSimulation(start, end), events_(events), eventsStart_(eventsStart), eventsEnd_(eventsEnd),
38  i_(0) {
39  years_ = end_.year()-start_.year();
40  if(eventsStart_.month()<start_.month()
41  || (eventsStart_.month()==start_.month()
42  && eventsStart_.dayOfMonth()<=start_.dayOfMonth())) {
43  periodStart_ = Date(start_.dayOfMonth(), start_.month(), eventsStart_.year());
44  } else {
45  periodStart_ = Date(start_.dayOfMonth(), start_.month(), eventsStart_.year()+1);
46  }
47  periodEnd_ = Date(end_.dayOfMonth(), end_.month(), periodStart_.year()+years_);
48  while(i_<events_->size() && (*events_)[i_].first<periodStart_) ++i_; //i points to the first element after the start of the relevant period.
49  }

51  bool EventSetSimulation::nextPath(std::vector< std::pair< Date, Real > >& path) {
52  path.resize(0);
53  if(periodEnd_>eventsEnd_) //Ran out of event data
54  return false;
55 
56  while(i_<events_->size() && (*events_)[i_].first<periodStart_) {
57  ++i_; //skip the elements between the previous period and this period
58  }
59  while(i_<events_->size() && (*events_)[i_].first<=periodEnd_){
60  std::pair<Date, Real> e(events_->at(i_).first+(start_.year() - periodStart_.year())*Years, events_->at(i_).second);
61  path.push_back(e);
62  ++i_; //i points to the first element after the start of the relevant period.
63  }
64  if(start_+years_*Years<end_) {
65  periodStart_+=(years_+1)*Years;
66  periodEnd_+=(years_+1)*Years;
67  } else {
68  periodStart_+=years_*Years;
69  periodEnd_+=years_*Years;
70  }
71  return true;
72  }

102  bool BetaRiskSimulation::nextPath(std::vector<std::pair<Date, Real> > &path)
103  {
104  path.resize(0);
105  Real eventFraction = exponential_();
106  while(eventFraction<=yearFraction_)
107  {
108  Integer days = round(eventFraction*dayCount_/yearFraction_);
109  Date eventDate = start_ + days*Days;
110  if(eventDate<=end_)
111  {
112  path.push_back(std::pair<Date, Real> (eventDate, generateBeta()));
113  }
114  else break;
115  eventFraction = exponential_();
116  }
117  return true;
118  }

120  BetaRisk::BetaRisk(Real maxLoss,
121  Real years,
122  Real mean,
123  Real stdDev)
124  : maxLoss_(maxLoss), lambda_(1.0/years) {
125  QL_REQUIRE(mean<maxLoss, "Mean "<<mean<<"of the loss distribution must be less than the maximum loss "<<maxLoss);
126  Real normalizedMean = mean/maxLoss;
127  Real normalizedVar = stdDev*stdDev/(maxLoss*maxLoss);
128  QL_REQUIRE(normalizedVar<normalizedMean*(1.0-normalizedMean), "Standard deviation of "<<stdDev<<" is impossible to achieve in gamma distribution with mean "<<mean);
129  Real nu = normalizedMean*(1.0-normalizedMean)/normalizedVar - 1.0;
130  alpha_=normalizedMean*nu;
131  beta_=(1.0-normalizedMean)*nu;
132  }
