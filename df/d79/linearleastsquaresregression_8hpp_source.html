69  template <class xContainer>
70  class LinearFcts<xContainer, false> {
71  public:
72  typedef typename xContainer::value_type ArgumentType;
73  LinearFcts (const xContainer &x, Real intercept) {
74  if (intercept != 0.0)
75  v.push_back(constant<ArgumentType, Real>(intercept));
76  Size m = x.begin()->size();
77  for (Size i = 0; i < m; ++i)
78  v.push_back(LinearFct<ArgumentType>(i));
