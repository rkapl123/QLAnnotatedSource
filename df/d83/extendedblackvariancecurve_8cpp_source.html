25  ExtendedBlackVarianceCurve::ExtendedBlackVarianceCurve(
26  const Date& referenceDate,
27  const std::vector<Date>& dates,
28  const std::vector<Handle<Quote> >& volatilities,

41  variances_ = std::vector<Real>(dates.size()+1);
42  times_ = std::vector<Time>(dates.size()+1);

44  times_[0] = 0.0;
45  for (Size j=1; j<=dates.size(); ++j) {
46  times_[j] = timeFromReference(dates[j-1]);
47  QL_REQUIRE(times_[j]>times_[j-1],
48  "dates must be sorted unique!");
49  }

51  setVariances();
52  setInterpolation<Linear>();
53 
54  for (Size j=0; j<volatilities_.size(); ++j)
55  registerWith(volatilities_[j]);

58  void ExtendedBlackVarianceCurve::setVariances() {
59  variances_[0] = 0.0;
60  for (Size j=1; j<=volatilities_.size(); j++) {
61  Volatility sigma = volatilities_[j-1]->value();
62  variances_[j] = times_[j] * sigma * sigma;
63  QL_REQUIRE(variances_[j]>=variances_[j-1]
64  || !forceMonotoneVariance_,
65  "variance must be non-decreasing");
66  }
67  }

75  Real ExtendedBlackVarianceCurve::blackVarianceImpl(Time t, Real) const {
76  if (t<=times_.back()) {
