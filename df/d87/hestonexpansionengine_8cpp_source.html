61  void HestonExpansionEngine::calculate() const
62  {
63  // this is a european option pricer
64  QL_REQUIRE(arguments_.exercise->type() == Exercise::European,
65  "not an European option");
66 
67  // plain vanilla
68  boost::shared_ptr<PlainVanillaPayoff> payoff =
69  boost::dynamic_pointer_cast<PlainVanillaPayoff>(arguments_.payoff);
70  QL_REQUIRE(payoff, "non plain vanilla payoff given");
71 
72  const boost::shared_ptr<HestonProcess>& process = model_->process();
73 
74  const Real riskFreeDiscount = process->riskFreeRate()->discount(
75  arguments_.exercise->lastDate());
76  const Real dividendDiscount = process->dividendYield()->discount(
77  arguments_.exercise->lastDate());
78 
79  const Real spotPrice = process->s0()->value();
80  QL_REQUIRE(spotPrice > 0.0, "negative or null underlying given");
81 
82  const Real strikePrice = payoff->strike();
83  const Real term = process->time(arguments_.exercise->lastDate());
84 
85  //possible optimization:
86  // if term=lastTerm & model=lastModel & formula=lastApprox, reuse approx.
87  const Real forward = spotPrice*dividendDiscount/riskFreeDiscount;
88  Real vol;
89  switch(formula_) {
90  case LPP2: {
91  LPP2HestonExpansion expansion(model_->kappa(), model_->theta(),
92  model_->sigma(), model_->v0(),
93  model_->rho(), term);
94  vol = expansion.impliedVolatility(strikePrice, forward);
95  break;
96  }
97  case LPP3: {
98  LPP3HestonExpansion expansion(model_->kappa(), model_->theta(),
99  model_->sigma(), model_->v0(),
100  model_->rho(), term);
101  vol = expansion.impliedVolatility(strikePrice, forward);
102  break;
103  }
104  case Forde: {
105  FordeHestonExpansion expansion(model_->kappa(), model_->theta(),
106  model_->sigma(), model_->v0(),
107  model_->rho(), term);
108  vol = expansion.impliedVolatility(strikePrice, forward);
109  break;
110  }
111  default:
112  QL_FAIL("unknown expansion formula");
113  }
114  const Real price = blackFormula(payoff, forward, vol*sqrt(term),
115  riskFreeDiscount, 0);
116  results_.value = price;
117  }

211  FordeHestonExpansion::FordeHestonExpansion(Real kappa, Real theta,
212  Real sigma, Real v0,
213  Real rho, Real term) {
214  Real v0Sqrt = sqrt(v0);
215  Real rhoBarSquare = 1 - rho * rho;
216  Real sigma00 = v0Sqrt;
217  Real sigma01 = v0Sqrt * (rho * sigma / (4 * v0)); //term in x
218  Real sigma02 = v0Sqrt * ((1 - 5 * rho * rho / 2) / 24 * sigma * sigma/ (v0 * v0)); //term in x*x
219  Real a00 = -sigma * sigma / 12 * (1 - rho * rho / 4) + v0 * rho * sigma / 4 + kappa / 2 * (theta - v0);
220  Real a01 = rho * sigma / (24 * v0) * (sigma * sigma * rhoBarSquare - 2 * kappa * (theta + v0) + v0 * rho * sigma); //term in x
221  Real a02 = (176 * sigma * sigma - 480 * kappa * theta - 712 * rho * rho * sigma * sigma + 521 * rho * rho * rho * rho * sigma * sigma + 40 * sigma * rho * rho * rho * v0 + 1040 * kappa * theta * rho * rho - 80 * v0 * kappa * rho * rho) * sigma * sigma / (v0 * v0 * 7680);
222  coeffs[0] = sigma00*sigma00+a00*term;
223  coeffs[1] = sigma00*sigma01*2+a01*term;
224  coeffs[2] = sigma00*sigma02*2+sigma01*sigma01+a02*term;
225  coeffs[3] = sigma01*sigma02*2;
226  coeffs[4] = sigma02*sigma02;
227  }
