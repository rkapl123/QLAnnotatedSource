27  CreditRiskPlus::CreditRiskPlus(
28  const std::vector<Real> &exposure,
29  const std::vector<Real> &defaultProbability,
30  const std::vector<Size> &sector,

34  relativeDefaultVariance_(relativeDefaultVariance),

36 
37  m_ = exposure_.size();

39  QL_REQUIRE(m_ > 0, "no exposures given");
40  QL_REQUIRE(m_ == pd_.size(), "number of exposures ("
41  << m_

60  exposureSum_ = 0.0;
61  el_ = 0.0;
62  el2_ = 0.0;
63  for (Size i = 0; i < m_; ++i) {
64  QL_REQUIRE(exposure_[i] >= 0.0, "exposure #"
65  << i << " is negative ("

67  QL_REQUIRE(pd_[i] > 0.0, "pd #" << i << " is negative (" << pd_[i]
68  << ")");

70  << ") is out of range 0..."
71  << (n_ - 1));
72  exposureSum_ += exposure_[i];
73  el_ += pd_[i] * exposure_[i];
74  el2_ += pd_[i] * exposure_[i]*exposure_[i];
75  }

82  Real CreditRiskPlus::lossQuantile(const Real p) {
83 
84  Size i = 0;
85  Real sum = loss_[0];
86  while(i < upperIndex_-1 && sum < p) {
87  ++i;
88  sum += loss_[i];
89  }
90 
91  if(loss_[0] >= p)
92  return 0.0;
93 
94  Real p1 = sum - loss_[i];
95  Real p2 = sum >= p ? sum : 1.0;
96  Real l1 = (i - 1) * unit_;
97  Real l2 = i * unit_;
98 
99  return l1 + (p - p1) / (p2 - p1) * (l2 - l1);
100  }

102  void CreditRiskPlus::compute() {
103 
104  std::vector<Real> sectorPdSum_, sectorSpecTerms_;
105 
106  sectorPdSum_ = std::vector<Real>(n_, 0.0);
107  sectorExposure_ = std::vector<Real>(n_, 0.0);
108  sectorEl_ = std::vector<Real>(n_, 0.0);
109  sectorSpecTerms_ = std::vector<Real>(n_, 0.0);
110  sectorUl_ = std::vector<Real>(n_, 0.0);
111  marginalLoss_ = std::vector<Real>(m_, 0.0);
112 
113  std::vector<Real> pdAdj(m_, 0.0);
114 
115  // compute exposure bands
116 
117  unsigned long maxNu_ = 0;
118  upperIndex_ = 0;
119 
120  // map of nuC_ to expected loss
121  std::map<unsigned long, Real, std::less<unsigned long> > epsNuC_;
122 
123  std::map<unsigned long, Real, std::less<unsigned long> >::iterator iter;
124 
125  for (Size k = 0; k < m_; ++k) {
126  unsigned long exUnit = (unsigned long)(std::floor(0.5 + exposure_[k] / unit_)); // round
127  if (exposure_[k] > 0 && exUnit == 0)
128  exUnit = 1; // but avoid zero exposure
129  if (exUnit > maxNu_)
130  maxNu_ = exUnit;
131  pdAdj[k] = exposure_[k] > 0.0
132  ? exposure_[k] * pd_[k] / (exUnit * unit_)
133  : 0.0; // adjusted pd
134  Real el = exUnit * pdAdj[k];
135  if (exUnit > 0) {
136  iter = epsNuC_.find(exUnit);
137  if (iter == epsNuC_.end()) {
138  epsNuC_.insert(std::pair<unsigned long, Real>(exUnit, el));
139  } else {
140  (*iter).second += el;
141  }
142  upperIndex_ += exUnit;
143  }
144  }
145 
146  // compute per sector figures
147 
148  Real pdSum_ = 0;
149  for (Size k = 0; k < m_; ++k) {
150  pdSum_ += pdAdj[k];
151  sectorPdSum_[sector_[k]] += pd_[k];
152  sectorExposure_[sector_[k]] += exposure_[k];
153  sectorEl_[sector_[k]] += exposure_[k] * pd_[k];
154  }
155 
156  for (Size i = 0; i < n_; ++i) {
157 
158  // precompute sector specific terms (formula 15 in [1])
159 
160  sectorSpecTerms_[i] += relativeDefaultVariance_[i] * sectorEl_[i];
161  for (Size j = 0; j < n_; ++j) {
162  if (j != i) {
163  sectorSpecTerms_[i] +=
164  correlation_[i][j] *
165  std::sqrt(relativeDefaultVariance_[i] *
166  relativeDefaultVariance_[j]) *
167  sectorEl_[j];
168  }
169  }
170  }
171 
172  // compute synthetic standard deviation (formula 12 in [1])
173 
174  ul_ = 0.0;
175  for (Size i = 0; i < n_; ++i) {
176  sectorUl_[i] =
177  relativeDefaultVariance_[i] * sectorEl_[i] * sectorEl_[i];
178  ul_ += sectorUl_[i];
179  for (Size j = 0; j < n_; ++j) {
180  if (j != i) {
181  ul_ += correlation_[i][j] *
182  std::sqrt(relativeDefaultVariance_[i] *
183  relativeDefaultVariance_[j]) *
184  sectorEl_[i] * sectorEl_[j];
185  }
186  }
187  }
188 
189  Real matchUl_ = ul_; // formula 13 in [1], rhs
190  for (Size k = 0; k < m_; ++k) {
191  Real tmp = pd_[k] * exposure_[k] * exposure_[k];
192  sectorUl_[sector_[k]] += tmp;
193  ul_ += tmp;
194  }
195  ul_ = std::sqrt(ul_);
196  for (Size i = 0; i < n_; ++i)
197  sectorUl_[i] = std::sqrt(sectorUl_[i]);
198 
199  // compute risk contributions (formula 15 in [1])
200 
201  for (Size k = 0; k < m_; ++k) {
202  marginalLoss_[k] = pd_[k] * exposure_[k] / ul_ *
203  (sectorSpecTerms_[sector_[k]] + exposure_[k]);
204  }
205 
206  // compute sigmaC_ and deduced figures
207 
208  Real sigmaC_ = pdSum_ * sqrt(matchUl_ / (el_ * el_));
209  Real alphaC_ = pdSum_ * pdSum_ / (sigmaC_ * sigmaC_);
210  Real betaC_ = sigmaC_ * sigmaC_ / pdSum_;
211  Real pC_ = betaC_ / (1.0 + betaC_);
212 
213  // compute loss distribution
214 
215  loss_.clear();
216  loss_.push_back(std::pow(1.0 - pC_, alphaC_)); // A(0)
217 
218  Real res;
219  for (unsigned long n = 0; n < upperIndex_ - 1; ++n) { // compute A(n+1)
220  // recursively
221  res = 0.0;
222  for (unsigned long j = 0;
223  j <= std::min<unsigned long>(maxNu_ - 1, n); ++j) {
224  iter = epsNuC_.find(j + 1);
225  if (iter != epsNuC_.end()) {
226  res += (*iter).second * loss_[n - j] * alphaC_;
227  if (j <= n - 1)
228  res += (*iter).second / ((Real)(j + 1)) *
229  ((Real)(n - j)) * loss_[n - j];
230  }
231  }
232  loss_.push_back(res * pC_ / (pdSum_ * ((Real)(n + 1))));
233  }
234  }
235 }
