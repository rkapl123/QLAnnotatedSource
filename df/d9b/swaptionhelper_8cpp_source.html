119  Real SwaptionHelper::blackPrice(Volatility sigma) const {
120  calculate();
121  Handle<Quote> vol(ext::shared_ptr<Quote>(new SimpleQuote(sigma)));
122  ext::shared_ptr<PricingEngine> engine;
123  switch(volatilityType_) {
124  case ShiftedLognormal:
125  engine = ext::make_shared<BlackSwaptionEngine>(
126  termStructure_, vol, Actual365Fixed(), shift_);
127  break;
128  case Normal:
129  engine = ext::make_shared<BachelierSwaptionEngine>(
130  termStructure_, vol, Actual365Fixed());
131  break;
132  default:
133  QL_FAIL("can not construct engine: " << volatilityType_);
134  break;
135  }
136  swaption_->setPricingEngine(engine);
137  Real value = swaption_->NPV();
138  swaption_->setPricingEngine(engine_);
139  return value;
140  }

142  void SwaptionHelper::performCalculations() const {
143 
144  Calendar calendar = index_->fixingCalendar();
145  Natural fixingDays = index_->fixingDays();
146 
147  Date exerciseDate = exerciseDate_;
148  if (exerciseDate == Null<Date>())
149  exerciseDate = calendar.advance(termStructure_->referenceDate(),
150  maturity_,
151  index_->businessDayConvention());
152 
153  Date startDate = calendar.advance(exerciseDate,
154  fixingDays, Days,
155  index_->businessDayConvention());
156 
157  Date endDate = endDate_;
158  if (endDate == Null<Date>())
159  endDate = calendar.advance(startDate, length_,
160  index_->businessDayConvention());
161 
162  Schedule fixedSchedule(startDate, endDate, fixedLegTenor_, calendar,
163  index_->businessDayConvention(),
164  index_->businessDayConvention(),
165  DateGeneration::Forward, false);
166  Schedule floatSchedule(startDate, endDate, index_->tenor(), calendar,
167  index_->businessDayConvention(),
168  index_->businessDayConvention(),
169  DateGeneration::Forward, false);
170 
171  ext::shared_ptr<PricingEngine> swapEngine(
172  new DiscountingSwapEngine(termStructure_, false));
173 
174  VanillaSwap::Type type = VanillaSwap::Receiver;
175 
176  VanillaSwap temp(VanillaSwap::Receiver, nominal_,
177  fixedSchedule, 0.0, fixedLegDayCounter_,
178  floatSchedule, index_, 0.0, floatingLegDayCounter_);
179  temp.setPricingEngine(swapEngine);
180  Real forward = temp.fairRate();
181  if(strike_ == Null<Real>()) {
182  exerciseRate_ = forward;
183  }
184  else {
185  exerciseRate_ = strike_;
186  type = strike_ <= forward ? VanillaSwap::Receiver : VanillaSwap::Payer;
187  // ensure that calibration instrument is out of the money
188  }
189  swap_ = ext::make_shared<VanillaSwap>(
190  type, nominal_,
191  fixedSchedule, exerciseRate_, fixedLegDayCounter_,
192  floatSchedule, index_, 0.0, floatingLegDayCounter_);
193  swap_->setPricingEngine(swapEngine);
194 
195  ext::shared_ptr<Exercise> exercise(new EuropeanExercise(exerciseDate));
196 
197  swaption_ = ext::make_shared<Swaption>(swap_, exercise);
198 
199  BlackCalibrationHelper::performCalculations();
200 
201  }
