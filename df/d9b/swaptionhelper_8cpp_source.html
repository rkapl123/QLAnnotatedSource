119 Real SwaptionHelper::blackPrice(Volatility sigma) const {
120 calculate();
121 Handle<Quote> vol(ext::shared_ptr<Quote>(new SimpleQuote(sigma)));
122 ext::shared_ptr<PricingEngine> engine;
123 switch(volatilityType_) {
124 case ShiftedLognormal:
125 engine = ext::make_shared<BlackSwaptionEngine>(
126 termStructure_, vol, Actual365Fixed(), shift_);
127 break;
128 case Normal:
129 engine = ext::make_shared<BachelierSwaptionEngine>(
130 termStructure_, vol, Actual365Fixed());
131 break;
132 default:
133 QL_FAIL("can not construct engine: " << volatilityType_);
134 break;
135 }
136 swaption_->setPricingEngine(engine);
137 Real value = swaption_->NPV();
138 swaption_->setPricingEngine(engine_);
139 return value;
140 }

142 void SwaptionHelper::performCalculations() const {
143
144 Calendar calendar = index_->fixingCalendar();
145
146 Date exerciseDate = exerciseDate_;
147 if (exerciseDate == Null<Date>())
148 exerciseDate = calendar.advance(termStructure_->referenceDate(),
149 maturity_,
150 index_->businessDayConvention());
151
152 Date startDate = index_->valueDate(index_->fixingCalendar().adjust(exerciseDate));
153
154 Date endDate = endDate_;
155 if (endDate == Null<Date>())
156 endDate = calendar.advance(startDate, length_,
157 index_->businessDayConvention());
158
159 Schedule fixedSchedule(startDate, endDate, fixedLegTenor_, calendar,
160 index_->businessDayConvention(),
161 index_->businessDayConvention(),
162 DateGeneration::Forward, false);
163 Schedule floatSchedule(startDate, endDate, index_->tenor(), calendar,
164 index_->businessDayConvention(),
165 index_->businessDayConvention(),
166 DateGeneration::Forward, false);
167
168 ext::shared_ptr<PricingEngine> swapEngine(
169 new DiscountingSwapEngine(termStructure_, false));
170
171 Swap::Type type = Swap::Receiver;
172
173 VanillaSwap temp(Swap::Receiver, nominal_,
174 fixedSchedule, 0.0, fixedLegDayCounter_,
175 floatSchedule, index_, 0.0, floatingLegDayCounter_);
176 temp.setPricingEngine(swapEngine);
177 Real forward = temp.fairRate();
178 if(strike_ == Null<Real>()) {
179 exerciseRate_ = forward;
180 }
181 else {
182 exerciseRate_ = strike_;
183 type = strike_ <= forward ? Swap::Receiver : Swap::Payer;
184 // ensure that calibration instrument is out of the money
185 }
186 swap_ = ext::make_shared<VanillaSwap>(
187 type, nominal_,
188 fixedSchedule, exerciseRate_, fixedLegDayCounter_,
189 floatSchedule, index_, 0.0, floatingLegDayCounter_);
190 swap_->setPricingEngine(swapEngine);
191
192 ext::shared_ptr<Exercise> exercise(new EuropeanExercise(exerciseDate));
193
194 swaption_ = ext::make_shared<Swaption>(swap_, exercise);
195
196 BlackCalibrationHelper::performCalculations();
197
198 }
