38  void CoterminalSwapCurveState::setOnCoterminalSwapRates(
39  const std::vector<Rate>& rates,
40  Size firstValidIndex) {
41  QL_REQUIRE(rates.size()==numberOfRates_,
42  "rates mismatch: " <<
43  numberOfRates_ << " required, " <<
44  rates.size() << " provided");
45  QL_REQUIRE(firstValidIndex<numberOfRates_,
46  "first valid index must be less than " <<
47  numberOfRates_ << ": " <<
48  firstValidIndex << " not allowed");
49  // first copy input...
50  first_ = firstValidIndex;
51  std::copy(rates.begin()+first_, rates.end(),
52  cotSwapRates_.begin()+first_);
53  // ...then calculate discount ratios and coterminal annuities:
54  // reference discount bond = P(n) (the last one)
55  // discRatios_[numberOfRates_] = P(n)/P(n) = 1.0 by construction/definition
56  cotAnnuities_[numberOfRates_-1] = rateTaus_[numberOfRates_-1];
57  // j < n
58  for (Size i=numberOfRates_-1; i>first_; --i) {
59  discRatios_[i] = 1.0 + cotSwapRates_[i] * cotAnnuities_[i];
60  cotAnnuities_[i-1] = cotAnnuities_[i] + rateTaus_[i-1] * discRatios_[i];
61  }
62  discRatios_[first_] = 1.0 + cotSwapRates_[first_] * cotAnnuities_[first_];
63 
64  // Insert here lazy evaluation of:
65  // - forward rates
66  // - constant maturity swap rates/annuities
67  }
