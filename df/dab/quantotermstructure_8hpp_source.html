113  inline Date QuantoTermStructure::maxDate() const {
114  Date maxDate = std::min(underlyingDividendTS_->maxDate(),
115  riskFreeTS_->maxDate());
116  maxDate = std::min(maxDate, foreignRiskFreeTS_->maxDate());
117  maxDate = std::min(maxDate, underlyingBlackVolTS_->maxDate());
118  maxDate = std::min(maxDate, exchRateBlackVolTS_->maxDate());
119  return maxDate;
120  }

122  inline Rate QuantoTermStructure::zeroYieldImpl(Time t) const {
123  // warning: here it is assumed that all TS have the same daycount.
124  // It should be QL_REQUIREd
125  return underlyingDividendTS_->zeroRate(t, Continuous, NoFrequency, true)
126  + riskFreeTS_->zeroRate(t, Continuous, NoFrequency, true)
127  - foreignRiskFreeTS_->zeroRate(t, Continuous, NoFrequency, true)
128  + underlyingExchRateCorrelation_
129  * underlyingBlackVolTS_->blackVol(t, strike_, true)
130  * exchRateBlackVolTS_->blackVol(t, exchRateATMlevel_, true);
131  }
