587  inline const Disposable<Matrix> outerProduct(Iterator1 v1begin,
588  Iterator1 v1end,
589  Iterator2 v2begin,
590  Iterator2 v2end) {
591 
592  Size size1 = std::distance(v1begin, v1end);

596  QL_REQUIRE(size2>0, "null second vector");

598  Matrix result(size1, size2);
599

601  std::transform(v2begin, v2end, result.row_begin(i),
602  std::bind1st(std::multiplies<Real>(), *v1begin));
