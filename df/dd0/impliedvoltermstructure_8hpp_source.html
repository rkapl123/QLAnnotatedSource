97  inline Real ImpliedVolTermStructure::blackVarianceImpl(Time t,
98  Real strike) const {
99  /* timeShift (and/or variance) variance at evaluation date
100  cannot be cached since the original curve could change
101  between invocations of this method */
102  Time timeShift =
103  dayCounter().yearFraction(originalTS_->referenceDate(),
104  referenceDate());
105  /* t is relative to the current reference date
106  and needs to be converted to the time relative
107  to the reference date of the original curve */
108  return originalTS_->blackForwardVariance(timeShift,
109  timeShift+t,
110  strike,
111  true);
112  }
