61  Disposable<Matrix> KlugeExtOUProcess::diffusion(Time t, const Array& x)
62  const{
63  Matrix retVal(size(), factors(), 0.0);
64 
65  Volatility vol = ouProcess_->diffusion(t, x.back());
66 
67  retVal[0][0] = klugeProcess_->diffusion(t, x)[0][0];
68  retVal[size()][0] = rho_*vol;
69  retVal[size()][factors()] = sqrtMRho_*vol;

71  return retVal;
