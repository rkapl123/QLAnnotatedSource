45 bool Italy::SettlementImpl::isBusinessDay(const Date& date) const {
46 Weekday w = date.weekday();
47 Day d = date.dayOfMonth(), dd = date.dayOfYear();
48 Month m = date.month();
49 Year y = date.year();
50 Day em = easterMonday(y);
51 if (isWeekend(w)
52 // New Year's Day
53 || (d == 1 && m == January)
54 // Epiphany
55 || (d == 6 && m == January)
56 // Easter Monday
57 || (dd == em)
58 // Liberation Day
59 || (d == 25 && m == April)
60 // Labour Day
61 || (d == 1 && m == May)
62 // Republic Day
63 || (d == 2 && m == June && y >= 2000)
64 // Assumption
65 || (d == 15 && m == August)
66 // All Saints' Day
67 || (d == 1 && m == November)
68 // Immaculate Conception
69 || (d == 8 && m == December)
70 // Christmas
71 || (d == 25 && m == December)
72 // St. Stephen
73 || (d == 26 && m == December)
74 // December 31st, 1999 only
75 || (d == 31 && m == December && y == 1999))
76 return false; // NOLINT(readability-simplify-boolean-expr)
77 return true;
78 }
