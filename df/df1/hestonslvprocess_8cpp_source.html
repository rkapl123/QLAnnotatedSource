51  Disposable<Array> HestonSLVProcess::drift(Time t, const Array& x) const {
52  Array tmp(2);
53 
54  const Volatility vol =
55  std::max(1e-8, std::sqrt(x[1])*leverageFct_->localVol(t, x[0], true));
56 
57  tmp[0] = riskFreeRate()->forwardRate(t, t, Continuous)
58  - dividendYield()->forwardRate(t, t, Continuous)
59  - 0.5*vol*vol;
60 
61  tmp[1] = kappa_*(theta_ - x[1]);
62 
63  return tmp;

66  Disposable<Matrix> HestonSLVProcess::diffusion(Time t, const Array& x)
67  const {
68 
69  const Real vol =
70  std::max(1e-8, std::sqrt(x[1])*leverageFct_->localVol(t, x[0], true));
71 
72  const Real sigma2 = mixedSigma_ * std::sqrt(x[1]);
73  const Real sqrhov = std::sqrt(1.0 - rho_*rho_);
74 
75  Matrix tmp(2,2);
76  tmp[0][0] = vol; tmp[0][1] = 0.0;
77  tmp[1][0] = rho_*sigma2; tmp[1][1] = sqrhov*sigma2;
78

82  Disposable<Array> HestonSLVProcess::evolve(
83  Time t0, const Array& x0, Time dt, const Array& dw) const {
84  Array retVal(2);
85 
86  const Real ex = std::exp(-kappa_*dt);
87 
88  const Real m = theta_+(x0[1]-theta_)*ex;
89  const Real s2 = x0[1]*mixedSigma_*mixedSigma_*ex/kappa_*(1-ex)
90  + theta_*mixedSigma_*mixedSigma_/(2*kappa_)*(1-ex)*(1-ex);
91  const Real psi = s2/(m*m);
92 
93  if (psi < 1.5) {
94  const Real b2 = 2/psi-1+std::sqrt(2/psi*(2/psi-1));
95  const Real b = std::sqrt(b2);
96  const Real a = m/(1+b2);
97 
98  retVal[1] = a*(b+dw[1])*(b+dw[1]);
99  }
100  else {
101  const Real p = (psi-1)/(psi+1);
102  const Real beta = (1-p)/m;
103  const Real u = CumulativeNormalDistribution()(dw[1]);
104 
105  retVal[1] = ((u <= p) ? 0.0 : std::log((1-p)/(1-u))/beta);
106  }
107 
108  const Real mu = riskFreeRate()->forwardRate(t0, t0+dt, Continuous)
109  - dividendYield()->forwardRate(t0, t0+dt, Continuous);
110 
111  const Real rho1 = std::sqrt(1-rho_*rho_);
112 
113  const Volatility l_0 = leverageFct_->localVol(t0, x0[0], true);
114  const Real v_0 = 0.5*(x0[1]+retVal[1])*l_0*l_0;
115 
116  retVal[0] = x0[0]*std::exp(mu*dt - 0.5*v_0*dt
117  + rho_/mixedSigma_*l_0 * (
118  retVal[1] - kappa_*theta_*dt
119  + 0.5*(x0[1]+retVal[1])*kappa_*dt - x0[1])
120  + rho1*std::sqrt(v_0*dt)*dw[0]);
121 
122  return retVal;
123  }
