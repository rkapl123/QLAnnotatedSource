191 inline Real CommodityIndex::forwardPrice(const Date& date) const {
192 try {
193 Real forwardPrice =
194 forwardCurve_->price(date, exchangeContracts_, nearbyOffset_);
195 return forwardPrice * forwardCurveUomConversionFactor_;
196 } catch (const std::exception& e) {
197 QL_FAIL("error fetching forward price for index " << name_
198 << ": " << e.what());
