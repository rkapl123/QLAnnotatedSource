29  template<class F>
30  Real BrentMinimize(Real low,
31  Real mid,
32  Real high,
33  Real tolerance,
34  Size maxIt,
35  const F& objectiveFunction) {
36  Real W = 0.5*(3.0-std::sqrt(5.0));
37  Real x = W*low+(1-W)*high;
38  if (mid > low && mid < high)
39  x = mid;
40 
41  Real midValue = objectiveFunction(x);
42 
43  Size iterations = 0;
44  while (high-low > tolerance && iterations < maxIt) {
45  if (x - low > high -x) { // left interval is bigger
46  Real tentativeNewMid = W*low+(1-W)*x;
47  Real tentativeNewMidValue =
48  objectiveFunction(tentativeNewMid);
49 
50  if (tentativeNewMidValue < midValue) { // go left
51  high =x;
52  x = tentativeNewMid;
53  midValue = tentativeNewMidValue;
54  } else { // go right
55  low = tentativeNewMid;
56  }
57  } else {
58  Real tentativeNewMid = W*x+(1-W)*high;
59  Real tentativeNewMidValue =
60  objectiveFunction(tentativeNewMid);
61 
62  if (tentativeNewMidValue < midValue) { // go right
63  low =x;
64  x = tentativeNewMid;
65  midValue = tentativeNewMidValue;
66  } else { // go left
67  high = tentativeNewMid;
68  }
69  }
70  ++iterations;
71  }
72  return x;
73  }
74  }

76  SphereCylinderOptimizer::SphereCylinderOptimizer(Real r,
77  Real s,
78  Real alpha,
79  Real z1,
80  Real z2,
81  Real z3,
82  Real zweight)
83  : r_(r), s_(s), alpha_(alpha), z1_(z1), z2_(z2), z3_(z3), zweight_(zweight)
84  {
85 
86  QL_REQUIRE(r>0,
87  "sphere must have positive radius");
88 
89  s = std::max(s,0.0);
90  QL_REQUIRE(alpha>0,
91  "cylinder centre must have positive coordinate");
92 
93  nonEmpty_ = std::fabs(alpha - s) <= r;
94 
95  Real cylinderInside = r * r - (s + alpha) * (s + alpha);
96 
97  if (cylinderInside > 0.0) {
98  topValue_ = alpha + s;
99  bottomValue_ = alpha - s;
100  }
101  else
102  {
103 
104  bottomValue_ = alpha-s;
105  Real tmp = r*r - (s *s+alpha*alpha);
106 
107  if ( tmp <=0)
108  { // max to left of maximimum
109  Real topValue2 = std::sqrt(s*s - tmp*tmp/(4*alpha*alpha));
110  topValue_ = alpha -std::sqrt(s*s - topValue2*topValue2);
111 
112  }
113  else
114  {
115  topValue_ = alpha+ tmp/(2.0*alpha);
116 
117  }
118 
119  }

127  void SphereCylinderOptimizer::findClosest(Size maxIterations,
128  Real tolerance,
129  Real& y1,
130  Real& y2,
131  Real& y3) const
132  {

134 
135  Real x1,x2,x3;
136  findByProjection(x1,x2,x3);
137 
138  y1 = BrentMinimize(
139  bottomValue_, x1, topValue_,tolerance, maxIterations,
140  ext::bind(

143  y3= std::sqrt(r_*r_ - y1*y1-y2*y2);

146  Real SphereCylinderOptimizer::objectiveFunction(Real x1) const
147  {
148  // Real x1 = alpha_ - std::sqrt(s_*s_-x2*x2);
149 
150  Real x2sq = s_*s_ - (x1-alpha_)*(x1-alpha_);
151  // a negative number will be minuscule and a result of rounding error
152  Real x2 = x2sq >= 0.0 ? std::sqrt(x2sq) : 0.0;
153  Real x3= std::sqrt(r_*r_ - x1*x1-x2*x2);
154 
155  Real err=0.0;
156  err+= (x1-z1_)*(x1-z1_);
157  err+= (x2-z2_)*(x2-z2_);
158  err+= (x3-z3_)*(x3-z3_)*zweight_;
159 
160  return err;
161  }

163  bool SphereCylinderOptimizer::findByProjection(Real& y1,
164  Real& y2,
165  Real& y3) const {
166  Real z1moved = z1_-alpha_;
167  Real distance = std::sqrt( z1moved*z1moved + z2_*z2_);
168  Real scale = s_/distance;
169  Real y1moved = z1moved*scale;
170  y1 = alpha_+ y1moved;
171  y2 = scale*z2_;
172  Real residual = r_*r_ - y1*y1 -y2*y2;
173  if (residual >=0.0) {
174  y3 = std::sqrt(residual);
175  return true;
176  }
177  // we are outside the sphere
178  if (!isIntersectionNonEmpty()) {
179  y3=0.0;
180  return false;
181  }
182 
183  // intersection is non-empty but projection point is outside sphere
184  // so take rightmost point
185  y3 = 0.0;
186  y1 = topValue_;
187  y2 = std::sqrt(r_*r_ -y1*y1);
188 
189  return true;
190  }
