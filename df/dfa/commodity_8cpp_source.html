45  const SecondaryCostAmounts& secondaryCostAmounts) {
46  std::string currencyCode;
47  Real totalAmount = 0;
48 
49  out << "secondary costs" << std::endl;
50  for (SecondaryCostAmounts::const_iterator i = secondaryCostAmounts.begin();
51  i != secondaryCostAmounts.end(); ++i) {
52  Real amount = i->second.value();
53  if (currencyCode.empty())
54  currencyCode = i->second.currency().code();

56  out << std::setw(28) << std::left << i->first
